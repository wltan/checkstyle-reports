25  package java.rmi;
26  
27  /**
28   * An <code>RMISecurityException</code> signals that a security exception
29   * has occurred during the execution of one of
30   * <code>java.rmi.RMISecurityManager</code>'s methods.
31   *
32   * @author  Roger Riggs
33   * @since   JDK1.1
34   * @deprecated Use {@link java.lang.SecurityException} instead.
35   * Application code should never directly reference this class, and
36   * <code>RMISecurityManager</code> no longer throws this subclass of
37   * <code>java.lang.SecurityException</code>.
38   */
39  @Deprecated
40  public class RMISecurityException extends java.lang.SecurityException {
41  
42      /* indicate compatibility with JDK 1.1.x version of class */
43       private static final long serialVersionUID = -8433406075740433514L;
44  
45      /**
46       * Construct an <code>RMISecurityException</code> with a detail message.
47       * @param name the detail message
48       * @since JDK1.1
49       * @deprecated no replacement
50       */
51      @Deprecated
52      public RMISecurityException(String name) {
53          super(name);
54      }
55  
56      /**
57       * Construct an <code>RMISecurityException</code> with a detail message.
58       * @param name the detail message
59       * @param arg ignored
60       * @since JDK1.1
61       * @deprecated no replacement
62       */
63      @Deprecated
64      public RMISecurityException(String name, String arg) {
65          this(name);
66      }
67  }

60  /**
61   * Defines the internal contract between the <tt>SessionFactory</tt> and other parts of
62   * Hibernate such as implementors of <tt>Type</tt>.
63   *
64   * @see org.hibernate.SessionFactory
65   * @see org.hibernate.internal.SessionFactoryImpl
66   * @author Gavin King
67   */
68  public interface SessionFactoryImplementor extends Mapping, SessionFactory {
69  	@Override
70  	public SessionBuilderImplementor withOptions();
71  
72  	/**
73  	 * Retrieve the {@link Type} resolver associated with this factory.
74  	 *
75  	 * @return The type resolver
76  	 */
77  	public TypeResolver getTypeResolver();
78  
79  	/**
80  	 * Get a copy of the Properties used to configure this session factory.
81  	 *
82  	 * @return The properties.
83  	 */
84  	public Properties getProperties();
85  
86  	/**
87  	 * Get the persister for the named entity
88  	 *
89  	 * @param entityName The name of the entity for which to retrieve the persister.
90  	 * @return The persister
91  	 * @throws MappingException Indicates persister could not be found with that name.
92  	 */
93  	public EntityPersister getEntityPersister(String entityName) throws MappingException;
94  
95  	/**
96  	 * Get all entity persisters as a Map, which entity name its the key and the persister is the value.
97  	 *
98  	 * @return The Map contains all entity persisters.
99  	 */
100 	public Map<String,EntityPersister> getEntityPersisters();
101 
102 	/**
103 	 * Get the persister object for a collection role.
104 	 *
105 	 * @param role The role (name) of the collection for which to retrieve the
106 	 * persister.
107 	 * @return The persister
108 	 * @throws MappingException Indicates persister could not be found with that role.
109 	 */
110 	public CollectionPersister getCollectionPersister(String role) throws MappingException;
111 
112 	/**
113 	 * Get all collection persisters as a Map, which collection role as the key and the persister is the value.
114 	 *
115 	 * @return The Map contains all collection persisters.
116 	 */
117 	public Map<String, CollectionPersister> getCollectionPersisters();
118 
119 	/**
120 	 * Get the JdbcServices.
121 	 * @return the JdbcServices
122 	 */
123 	public JdbcServices getJdbcServices();
124 
125 	/**
126 	 * Get the SQL dialect.
127 	 * <p/>
128 	 * Shorthand for {@code getJdbcServices().getDialect()}
129 	 *
130 	 * @return The dialect
131 	 */
132 	public Dialect getDialect();
133 
134 	/**
135 	 * Get the factory scoped interceptor for this factory.
136 	 *
137 	 * @return The factory scope interceptor, or null if none.
138 	 */
139 	public Interceptor getInterceptor();
140 
141 	public QueryPlanCache getQueryPlanCache();
142 
143 	/**
144 	 * Get the return types of a query
145 	 */
146 	public Type[] getReturnTypes(String queryString) throws HibernateException;
147 
148 	/**
149 	 * Get the return aliases of a query
150 	 */
151 	public String[] getReturnAliases(String queryString) throws HibernateException;
152 
153 	/**
154 	 * Get the connection provider
155 	 *
156 	 * @deprecated Access to connections via {@link org.hibernate.engine.jdbc.spi.JdbcConnectionAccess} should
157 	 * be preferred over access via {@link ConnectionProvider}, whenever possible.
158 	 * {@link org.hibernate.engine.jdbc.spi.JdbcConnectionAccess} is tied to the Hibernate Session to
159 	 * properly account for contextual information.  See {@link SessionImplementor#getJdbcConnectionAccess()}
160 	 */
161 	@Deprecated
162 	public ConnectionProvider getConnectionProvider();
163 	/**
164 	 * Get the names of all persistent classes that implement/extend the given interface/class
165 	 */
166 	public String[] getImplementors(String className) throws MappingException;
167 	/**
168 	 * Get a class name, using query language imports
169 	 */
170 	public String getImportedClassName(String name);
171 
172 	/**
173 	 * Get the default query cache
174 	 */
175 	public QueryCache getQueryCache();
176 	/**
177 	 * Get a particular named query cache, or the default cache
178 	 * @param regionName the name of the cache region, or null for the default query cache
179 	 * @return the existing cache, or a newly created cache if none by that region name
180 	 */
181 	public QueryCache getQueryCache(String regionName) throws HibernateException;
182 
183 	/**
184 	 * Get the cache of table update timestamps
185 	 */
186 	public UpdateTimestampsCache getUpdateTimestampsCache();
187 	/**
188 	 * Statistics SPI
189 	 */
190 	public StatisticsImplementor getStatisticsImplementor();
191 
192 	public NamedQueryDefinition getNamedQuery(String queryName);
193 	public NamedSQLQueryDefinition getNamedSQLQuery(String queryName);
194 	public ResultSetMappingDefinition getResultSetMapping(String name);
195 
196 	/**
197 	 * Get the identifier generator for the hierarchy
198 	 */
199 	public IdentifierGenerator getIdentifierGenerator(String rootEntityName);
200 
201 	/**
202 	 * Get a named second-level cache region
203 	 *
204 	 * @param regionName The name of the region to retrieve.
205 	 * @return The region
206 	 */
207 	public Region getSecondLevelCacheRegion(String regionName);
208 	
209 	/**
210 	 * Get a named naturalId cache region
211 	 *
212 	 * @param regionName The name of the region to retrieve.
213 	 * @return The region
214 	 */
215 	public Region getNaturalIdCacheRegion(String regionName);
216 
217 	/**
218 	 * Get a map of all the second level cache regions currently maintained in
219 	 * this session factory.  The map is structured with the region name as the
220 	 * key and the {@link Region} instances as the values.
221 	 *
222 	 * @return The map of regions
223 	 */
224 	public Map getAllSecondLevelCacheRegions();
225 
226 	/**
227 	 * Retrieves the SQLExceptionConverter in effect for this SessionFactory.
228 	 *
229 	 * @return The SQLExceptionConverter for this SessionFactory.
230 	 *
231 	 */
232 	public SQLExceptionConverter getSQLExceptionConverter();
233 	   // TODO: deprecate???
234 
235 	/**
236 	 * Retrieves the SqlExceptionHelper in effect for this SessionFactory.
237 	 *
238 	 * @return The SqlExceptionHelper for this SessionFactory.
239 	 *
240 	 */
241     public SqlExceptionHelper getSQLExceptionHelper();
242 
243 	public Settings getSettings();
244 
245 	/**
246 	 * Get a nontransactional "current" session for Hibernate EntityManager
247 	 */
248 	public Session openTemporarySession() throws HibernateException;
249 
250 	/**
251 	 * Retrieves a set of all the collection roles in which the given entity
252 	 * is a participant, as either an index or an element.
253 	 *
254 	 * @param entityName The entity name for which to get the collection roles.
255 	 * @return set of all the collection roles in which the given entityName participates.
256 	 */
257 	public Set<String> getCollectionRolesByEntityParticipant(String entityName);
258 
259 	public EntityNotFoundDelegate getEntityNotFoundDelegate();
260 
261 	public SQLFunctionRegistry getSqlFunctionRegistry();
262 
263 	/**
264 	 * Retrieve fetch profile by name.
265 	 *
266 	 * @param name The name of the profile to retrieve.
267 	 * @return The profile definition
268 	 */
269 	public FetchProfile getFetchProfile(String name);
270 
271 	public ServiceRegistryImplementor getServiceRegistry();
272 
273 	public void addObserver(SessionFactoryObserver observer);
274 
275 	public CustomEntityDirtinessStrategy getCustomEntityDirtinessStrategy();
276 
277 	public CurrentTenantIdentifierResolver getCurrentTenantIdentifierResolver();
278 }

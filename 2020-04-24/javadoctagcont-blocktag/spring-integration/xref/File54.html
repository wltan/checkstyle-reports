109 	 * @param uri The URI.
110 	 * @param webClient The WebClient to use.

112 	public WebFluxRequestExecutingMessageHandler(String uri, @Nullable WebClient webClient) {
113 		this(new LiteralExpression(uri), webClient);
114 		/*
115 		 *  We'd prefer to do this assertion first, but the compiler doesn't allow it. However,
116 		 *  it's safe because the literal expression simply wraps the String variable, even
117 		 *  when null.
118 		 */
119 		Assert.hasText(uri, "URI is required");

122 	/**
123 	 * Create a handler that will send requests to the provided URI using a provided WebClient.
124 	 * @param uriExpression A SpEL Expression that can be resolved against the message object and
125 	 * {@link org.springframework.beans.factory.BeanFactory}.
126 	 * @param webClient The WebClient to use.
127 	 */
128 	public WebFluxRequestExecutingMessageHandler(Expression uriExpression, @Nullable WebClient webClient) {
129 		super(uriExpression);
130 		this.webClientExplicitlySet = webClient != null;
131 		this.webClient =
132 				!this.webClientExplicitlySet
133 						? WebClient.builder().uriBuilderFactory(this.uriFactory).build()
134 						: webClient;
135 		this.setAsync(true);
136 	}

206 	@Override
207 	@Nullable
208 	protected Object exchange(Object uri, HttpMethod httpMethod, HttpEntity<?> httpRequest,
209 			Object expectedResponseType, Message<?> requestMessage, Map<String, ?> uriVariables) {
210 
211 		WebClient.RequestBodySpec requestSpec =
212 				createRequestBodySpec(uri, httpMethod, httpRequest, requestMessage, uriVariables);
213 
214 		Mono<ClientResponse> responseMono = exchangeForResponseMono(requestSpec);
215 
216 		if (isExpectReply()) {
217 			return createReplyFromResponse(expectedResponseType, responseMono);
218 		}
219 		else {
220 			responseMono.subscribe(v -> { }, ex -> sendErrorMessage(requestMessage, ex));
221 			return null;
222 		}
223 	}

225 	private Object createReplyFromResponse(Object expectedResponseType, Mono<ClientResponse> responseMono) {
226 		return responseMono
227 				.flatMap(response -> {
228 							ResponseEntity.BodyBuilder httpEntityBuilder =
229 									ResponseEntity.status(response.statusCode())
230 											.headers(response.headers().asHttpHeaders());
231 
232 							Mono<?> bodyMono;
233 
234 							if (expectedResponseType != null) {
235 								if (this.replyPayloadToFlux) {
236 									BodyExtractor<? extends Flux<?>, ReactiveHttpInputMessage> extractor;
237 									if (expectedResponseType instanceof ParameterizedTypeReference<?>) {
238 										extractor = BodyExtractors.toFlux(
239 												(ParameterizedTypeReference<?>) expectedResponseType);
240 									}
241 									else {
242 										extractor = BodyExtractors.toFlux((Class<?>) expectedResponseType);
243 									}
244 									Flux<?> flux = response.body(extractor);
245 									bodyMono = Mono.just(flux);
246 								}
247 								else {
248 									BodyExtractor<? extends Mono<?>, ReactiveHttpInputMessage> extractor;
249 									if (expectedResponseType instanceof ParameterizedTypeReference<?>) {
250 										extractor = BodyExtractors.toMono(
251 												(ParameterizedTypeReference<?>) expectedResponseType);
252 									}
253 									else {
254 										extractor = BodyExtractors.toMono((Class<?>) expectedResponseType);
255 									}
256 									bodyMono = response.body(extractor);
257 								}
258 							}
259 							else if (this.bodyExtractor != null) {
260 								Object body = response.body(this.bodyExtractor);
261 								if (body instanceof Mono) {
262 									bodyMono = (Mono<?>) body;
263 								}
264 								else {
265 									bodyMono = Mono.just(body);
266 								}
267 							}
268 							else {
269 								bodyMono = Mono.empty();
270 							}
271 
272 							return bodyMono
273 									.map(httpEntityBuilder::body)
274 									.defaultIfEmpty(httpEntityBuilder.build());
275 						}
276 				)
277 				.map(this::getReply);
278 	}

280 	private WebClient.RequestBodySpec createRequestBodySpec(Object uri, HttpMethod httpMethod,
281 			HttpEntity<?> httpRequest, Message<?> requestMessage, Map<String, ?> uriVariables) {
282 
283 		WebClient.RequestBodyUriSpec requestBodyUriSpec = this.webClient.method(httpMethod);
284 		WebClient.RequestBodySpec requestSpec;
285 
286 		if (uri instanceof URI) {
287 			requestSpec = requestBodyUriSpec.uri((URI) uri);
288 		}
289 		else {
290 			requestSpec = requestBodyUriSpec.uri((String) uri, uriVariables);
291 		}
292 
293 		requestSpec = requestSpec.headers(headers -> headers.putAll(httpRequest.getHeaders()));
294 		BodyInserter<?, ? super ClientHttpRequest> inserter = buildBodyInserterForRequest(requestMessage, httpRequest);
295 		if (inserter != null) {
296 			requestSpec.body(inserter);
297 		}
298 		return requestSpec;
299 	}

301 	private Mono<ClientResponse> exchangeForResponseMono(WebClient.RequestBodySpec requestSpec) {
302 		return requestSpec.exchange()
303 				.flatMap(response -> {
304 					HttpStatus httpStatus = response.statusCode();
305 					if (httpStatus.isError()) {
306 						return response.body(BodyExtractors.toDataBuffers())
307 								.reduce(DataBuffer::write)
308 								.map(dataBuffer -> {
309 									byte[] bytes = new byte[dataBuffer.readableByteCount()];
310 									dataBuffer.read(bytes);
311 									DataBufferUtils.release(dataBuffer);
312 									return bytes;
313 								})
314 								.defaultIfEmpty(new byte[0])
315 								.map(bodyBytes -> {
316 											throw new WebClientResponseException(
317 													"ClientResponse has erroneous status code: "
318 															+ httpStatus.value() + " "
319 															+ httpStatus.getReasonPhrase(),
320 													httpStatus.value(),
321 													httpStatus.getReasonPhrase(),
322 													response.headers().asHttpHeaders(),
323 													bodyBytes,
324 													response.headers().contentType()
325 															.map(MimeType::getCharset)
326 															.orElse(StandardCharsets.ISO_8859_1));
327 										}
328 								);
329 					}
330 					else {
331 						return Mono.just(response);
332 					}
333 				});
334 	}

336 	@Nullable
337 	private BodyInserter<?, ? super ClientHttpRequest> buildBodyInserterForRequest(Message<?> requestMessage,
338 			HttpEntity<?> httpRequest) {
339 
340 		Object requestBody = httpRequest.getBody();
341 		if (requestBody == null) {
342 			return null;
343 		}
344 
345 		BodyInserter<?, ? super ClientHttpRequest> inserter = null;
346 		if (requestBody instanceof Resource) {
347 			inserter = BodyInserters.fromResource((Resource) requestBody);
348 		}
349 		else if (requestBody instanceof Publisher<?>) {
350 			inserter = buildBodyInserterForPublisher(requestMessage, (Publisher<?>) requestBody);
351 		}
352 		else if (requestBody instanceof MultiValueMap<?, ?>) {
353 			inserter = buildBodyInserterForMultiValueMap((MultiValueMap<?, ?>) requestBody,
354 					httpRequest.getHeaders().getContentType());
355 		}
356 		else {
357 			inserter = BodyInserters.fromValue(requestBody);
358 		}
359 		return inserter;
360 	}
361 
362 	@SuppressWarnings("unchecked")
363 	private <T, P extends Publisher<T>> BodyInserter<P, ? super ClientHttpRequest> buildBodyInserterForPublisher(
364 			Message<?> requestMessage, P publisher) {
365 
366 		BodyInserter<P, ? super ClientHttpRequest> inserter;
367 		Object publisherElementType = evaluateTypeFromExpression(requestMessage,
368 				this.publisherElementTypeExpression, "publisherElementType");
369 		if (publisherElementType instanceof Class<?>) {
370 			inserter = BodyInserters.fromPublisher(publisher, (Class<T>) publisherElementType);
371 		}
372 		else if (publisherElementType instanceof ParameterizedTypeReference<?>) {
373 			inserter = BodyInserters.fromPublisher(publisher, (ParameterizedTypeReference<T>) publisherElementType);
374 		}
375 		else {
376 			inserter = BodyInserters.fromPublisher(publisher, (Class<T>) Object.class);
377 		}
378 		return inserter;
379 	}
380 
381 	@Nullable
382 	@SuppressWarnings("unchecked")
383 	private static BodyInserters.FormInserter<?> buildBodyInserterForMultiValueMap(
384 			MultiValueMap<?, ?> requestBody, MediaType contentType) {
385 
386 		if (MediaType.APPLICATION_FORM_URLENCODED.equals(contentType)) {
387 			return BodyInserters.fromFormData((MultiValueMap<String, String>) requestBody);
388 		}
389 		else if (MediaType.MULTIPART_FORM_DATA.equals(contentType)) {
390 			return BodyInserters.fromMultipartData((MultiValueMap<String, ?>) requestBody);
391 		}
392 		else {
393 			return null;
394 		}
395 	}

70  /**
71   * A {@link org.springframework.messaging.MessageHandler} implementation
72   * that writes the Message payload to a
73   * file. If the payload is a File object, it will copy the File to the specified
74   * destination directory. If the payload is a byte array, a String or an
75   * InputStream it will be written directly. Otherwise, the payload type is
76   * unsupported, and an Exception will be thrown.
77   * <p>
78   * To append a new-line after each write, set the
79   * {@link #setAppendNewLine(boolean) appendNewLine} flag to 'true'. It is 'false' by default.
80   * <p>
81   * If the 'deleteSourceFiles' flag is set to true, the original Files will be
82   * deleted. The default value for that flag is <em>false</em>. See the
83   * {@link #setDeleteSourceFiles(boolean)} method javadoc for more information.
84   * <p>
85   * Other transformers may be useful to precede this handler. For example, any
86   * Serializable object payload can be converted into a byte array by the
87   * {@link org.springframework.integration.transformer.PayloadSerializingTransformer}.
88   * Likewise, any Object can be converted to a String based on its
89   * <code>toString()</code> method by the
90   * {@link org.springframework.integration.transformer.ObjectToStringTransformer}.
91   * <p>
92   * {@link FileExistsMode#APPEND} adds content to an existing file; the file is closed after
93   * each write.
94   * {@link FileExistsMode#APPEND_NO_FLUSH} adds content to an existing file and the file
95   * is left open without flushing any data. Data will be flushed based on the
96   * {@link #setFlushInterval(long) flushInterval} or when a message is sent to the
97   * {@link #trigger(Message)} method, or a
98   * {@link #flushIfNeeded(MessageFlushPredicate, Message) flushIfNeeded}
99   * method is called.
100  *
101  * @author Mark Fisher
102  * @author Iwein Fuld
103  * @author Alex Peters
104  * @author Oleg Zhurakousky
105  * @author Artem Bilan
106  * @author Gunnar Hillert
107  * @author Gary Russell
108  * @author Tony Falabella
109  * @author Alen Turkovic
110  */
111 public class FileWritingMessageHandler extends AbstractReplyProducingMessageHandler
112 		implements Lifecycle, MessageTriggerAction {

222 	 * Will set the {@link FileExistsMode} that specifies what will happen in
223 	 * case the destination exists. For example {@link FileExistsMode#APPEND}
224 	 * instructs this handler to append data to the existing file rather then
225 	 * creating a new file for each {@link Message}.
226 	 * <p> If set to {@link FileExistsMode#APPEND}, the adapter will also
227 	 * create a real instance of the {@link LockRegistry} to ensure that there
228 	 * is no collisions when multiple threads are writing to the same file.
229 	 * <p> Otherwise the LockRegistry is set to {@link PassThruLockRegistry} which
230 	 * has no effect.
231 	 * <p> With {@link FileExistsMode#REPLACE_IF_MODIFIED}, if the file exists,
232 	 * it is only replaced if its last modified timestamp is different to the
233 	 * source; otherwise, the write is ignored. For {@link File} payloads,
234 	 * the actual timestamp of the {@link File} is compared; for other payloads,
235 	 * the {@link FileHeaders#SET_MODIFIED} is compared to the existing file.
236 	 * If the header is missing, or its value is not a {@link Number}, the file
237 	 * is always replaced. This mode will typically only make sense if
238 	 * {@link #setPreserveTimestamp(boolean) preserveTimestamp} is true.
239 	 * @param fileExistsMode Must not be null
240 	 * @see #setPreserveTimestamp(boolean)
241 	 */
242 	public void setFileExistsMode(FileExistsMode fileExistsMode) {
243 		Assert.notNull(fileExistsMode, "'fileExistsMode' must not be null.");
244 		this.fileExistsMode = fileExistsMode;
245 
246 		if (FileExistsMode.APPEND.equals(fileExistsMode)
247 				|| FileExistsMode.APPEND_NO_FLUSH.equals(this.fileExistsMode)) {
248 			this.lockRegistry = this.lockRegistry instanceof PassThruLockRegistry
249 					? new DefaultLockRegistry()
250 					: this.lockRegistry;
251 		}
252 	}

382 	/**
383 	 * Set the file permissions after uploading, e.g. 0600 for
384 	 * owner read/write. Only applies to file systems that support posix
385 	 * file permissions.
386 	 * @param chmod the permissions.
387 	 * @throws IllegalArgumentException if the value is higher than 0777.
388 	 * @since 5.0
389 	 */
390 	public void setChmod(int chmod) {
391 		Assert.isTrue(chmod >= 0 && chmod <= 0777, // NOSONAR permissions octal
392 				"'chmod' must be between 0 and 0777 (octal)");
393 		if (!FileUtils.IS_POSIX) {
394 			this.logger.error("'chmod' setting ignored - the file system does not support Posix attributes");
395 			return;
396 		}
397 		BitSet bits = BitSet.valueOf(new byte[] { (byte) chmod, (byte) (chmod >> 8) }); // NOSONAR
398 		this.permissions = bits.stream()
399 				.boxed()
400 				.map((b) -> POSIX_FILE_PERMISSIONS[b])
401 				.collect(Collectors.toSet());
402 	}

447 	@Override
448 	public void start() {
449 		if (this.flushTask == null && FileExistsMode.APPEND_NO_FLUSH.equals(this.fileExistsMode)) {
450 			TaskScheduler taskScheduler = getTaskScheduler();
451 			Assert.state(taskScheduler != null, "'taskScheduler' is required for FileExistsMode.APPEND_NO_FLUSH");
452 			this.flushTask = taskScheduler.scheduleAtFixedRate(new Flusher(), this.flushInterval / 3); // NOSONAR
453 		}
454 	}
455 
456 	@Override
457 	public void stop() {
458 		synchronized (this) {
459 			if (this.flushTask != null) {
460 				this.flushTask.cancel(true);
461 				this.flushTask = null;
462 			}
463 		}
464 		Flusher flusher = new Flusher();
465 		flusher.run();
466 		boolean needInterrupt = this.fileStates.size() > 0;
467 		int n = 0;
468 		while (n++ < 10 && this.fileStates.size() > 0) { // NOSONAR
469 			try {
470 				Thread.sleep(1);
471 			}
472 			catch (InterruptedException e) {
473 				// cancel the interrupt
474 			}
475 			flusher.run();
476 		}
477 		if (this.fileStates.size() > 0) {
478 			this.logger.error("Failed to flush after multiple attempts, while stopping: " + this.fileStates.keySet());
479 		}
480 		if (needInterrupt) {
481 			Thread.currentThread().interrupt();
482 		}
483 	}

490 	private void validateDestinationDirectory(File destinationDirectory, boolean autoCreateDirectory) {
491 		if (!destinationDirectory.exists() && autoCreateDirectory) {
492 			Assert.isTrue(destinationDirectory.mkdirs(),
493 					() -> "Destination directory [" + destinationDirectory + "] could not be created.");
494 		}
495 		Assert.isTrue(destinationDirectory.exists(),
496 				() -> "Destination directory [" + destinationDirectory + "] does not exist.");
497 		Assert.isTrue(destinationDirectory.isDirectory(),
498 				() -> "Destination path [" + destinationDirectory + "] does not point to a directory.");
499 		Assert.isTrue(Files.isWritable(destinationDirectory.toPath()),
500 				() -> "Destination directory [" + destinationDirectory + "] is not writable.");
501 	}

503 	@Override // NOSONAR
504 	protected Object handleRequestMessage(Message<?> requestMessage) {
505 		Object payload = requestMessage.getPayload();
506 		String generatedFileName = this.fileNameGenerator.generateFileName(requestMessage);
507 		File originalFileFromHeader = retrieveOriginalFileFromHeader(requestMessage);
508 
509 		File destinationDirectoryToUse = evaluateDestinationDirectoryExpression(requestMessage);
510 
511 		File tempFile = new File(destinationDirectoryToUse, generatedFileName + this.temporaryFileSuffix);
512 		File resultFile = new File(destinationDirectoryToUse, generatedFileName);
513 
514 		boolean exists = resultFile.exists();
515 		if (exists && FileExistsMode.FAIL.equals(this.fileExistsMode)) {
516 			throw new MessageHandlingException(requestMessage,
517 					"Failed to process message in the [" + this
518 							+ "]. The destination file already exists at '" + resultFile.getAbsolutePath() + "'.");
519 		}
520 
521 		Object timestamp = requestMessage.getHeaders().get(FileHeaders.SET_MODIFIED);
522 		if (payload instanceof File) {
523 			timestamp = ((File) payload).lastModified();
524 		}
525 		boolean ignore = (FileExistsMode.IGNORE.equals(this.fileExistsMode) // NOSONAR
526 				&& (exists || (StringUtils.hasText(this.temporaryFileSuffix) && tempFile.exists())))
527 				|| ((exists && FileExistsMode.REPLACE_IF_MODIFIED.equals(this.fileExistsMode))
528 				&& (timestamp instanceof Number
529 				&& ((Number) timestamp).longValue() == resultFile.lastModified()));
530 		if (!ignore) {
531 			try {
532 				if (!exists &&
533 						generatedFileName.replaceAll("/", Matcher.quoteReplacement(File.separator))
534 								.contains(File.separator)) {
535 					resultFile.getParentFile().mkdirs(); //NOSONAR - will fail on the writing below
536 				}
537 
538 				resultFile = writeMessageToFile(requestMessage, originalFileFromHeader, tempFile, resultFile,
539 						timestamp);
540 			}
541 			catch (Exception e) {
542 				throw IntegrationUtils.wrapInHandlingExceptionIfNecessary(requestMessage,
543 						() -> "failed to write Message payload to file in the [" + this + ']', e);
544 			}
545 		}
546 
547 		if (!this.expectReply) {
548 			return null;
549 		}
550 
551 		if (resultFile != null && originalFileFromHeader == null && payload instanceof File) {
552 			return getMessageBuilderFactory()
553 					.withPayload(resultFile)
554 					.setHeader(FileHeaders.ORIGINAL_FILE, payload);
555 		}
556 		return resultFile;
557 	}

559 	private File writeMessageToFile(Message<?> requestMessage, File originalFileFromHeader, File tempFile,
560 			File resultFile, Object timestamp) throws IOException {
561 
562 		File fileToReturn;
563 		Object payload = requestMessage.getPayload();
564 		if (payload instanceof File) {
565 			fileToReturn = handleFileMessage((File) payload, tempFile, resultFile, requestMessage);
566 		}
567 		else if (payload instanceof InputStream) {
568 			fileToReturn = handleInputStreamMessage((InputStream) payload, originalFileFromHeader, tempFile,
569 					resultFile, requestMessage);
570 		}
571 		else if (payload instanceof byte[]) {
572 			fileToReturn = handleByteArrayMessage((byte[]) payload, originalFileFromHeader, tempFile, resultFile,
573 					requestMessage);
574 		}
575 		else if (payload instanceof String) {
576 			fileToReturn = handleStringMessage((String) payload, originalFileFromHeader, tempFile, resultFile,
577 					requestMessage);
578 		}
579 		else {
580 			throw new IllegalArgumentException(
581 					"Unsupported Message payload type [" + payload.getClass().getName() + "]");
582 		}
583 
584 		if (this.preserveTimestamp) {
585 			if (timestamp instanceof Number) {
586 				if (!fileToReturn.setLastModified(((Number) timestamp).longValue())) {
587 					throw new IllegalStateException("Could not set last modified '" + timestamp
588 							+ "' timestamp on file: " + fileToReturn);
589 				}
590 			}
591 			else if (this.logger.isWarnEnabled()) {
592 				this.logger.warn("Could not set lastModified, header " + FileHeaders.SET_MODIFIED
593 						+ " must be a Number, not " + (timestamp == null ? "null" : timestamp.getClass()));
594 			}
595 		}
596 		return fileToReturn;
597 	}

600 	 * Retrieves the File instance from the {@link FileHeaders#ORIGINAL_FILE}
601 	 * header if available. If the value is not a File instance or a String
602 	 * representation of a file path, this will return <code>null</code>.
603 	 */

605 		Object value = message.getHeaders().get(FileHeaders.ORIGINAL_FILE);
606 		if (value instanceof File) {

608 		}

613 	}

628 	private File handleInputStreamMessage(InputStream sourceFileInputStream, File originalFile, File tempFile,
629 			File resultFile, Message<?> requestMessage) throws IOException {
630 
631 		boolean append =
632 				FileExistsMode.APPEND.equals(this.fileExistsMode)
633 						|| FileExistsMode.APPEND_NO_FLUSH.equals(this.fileExistsMode);
634 
635 		if (append) {
636 			final File fileToWriteTo = determineFileToWrite(resultFile, tempFile);
637 
638 			WhileLockedProcessor whileLockedProcessor = new WhileLockedProcessor(this.lockRegistry,
639 					fileToWriteTo.getAbsolutePath()) {
640 
641 				@Override
642 				protected void whileLocked() throws IOException {
643 					if (FileWritingMessageHandler.this.newFileCallback != null && !fileToWriteTo.exists()) {
644 						FileWritingMessageHandler.this.newFileCallback.accept(fileToWriteTo, requestMessage);
645 					}
646 
647 					appendStreamToFile(fileToWriteTo, sourceFileInputStream);
648 				}
649 
650 			};
651 			whileLockedProcessor.doWhileLocked();
652 			cleanUpAfterCopy(fileToWriteTo, resultFile, originalFile);
653 			return resultFile;
654 		}
655 		else {
656 
657 			try (InputStream inputStream = sourceFileInputStream;
658 					OutputStream outputStream =
659 							new BufferedOutputStream(new FileOutputStream(tempFile), this.bufferSize)) {
660 
661 				byte[] buffer = new byte[StreamUtils.BUFFER_SIZE];
662 				int bytesRead;
663 				while ((bytesRead = inputStream.read(buffer)) != -1) {
664 					outputStream.write(buffer, 0, bytesRead);
665 				}
666 				if (this.appendNewLine) {
667 					outputStream.write(System.lineSeparator().getBytes());
668 				}
669 				outputStream.flush();
670 			}
671 			cleanUpAfterCopy(tempFile, resultFile, originalFile);
672 			return resultFile;
673 		}
674 	}

676 	private void appendStreamToFile(File fileToWriteTo, InputStream sourceFileInputStream) throws IOException {
677 		FileState state = getFileState(fileToWriteTo, false);
678 		BufferedOutputStream bos = null;
679 		try (InputStream inputStream = sourceFileInputStream) {
680 			bos = state != null ? state.stream : createOutputStream(fileToWriteTo, true);
681 			byte[] buffer = new byte[StreamUtils.BUFFER_SIZE];
682 			int bytesRead = -1;
683 			while ((bytesRead = inputStream.read(buffer)) != -1) {
684 				bos.write(buffer, 0, bytesRead);
685 			}
686 			if (FileWritingMessageHandler.this.appendNewLine) {
687 				bos.write(System.lineSeparator().getBytes());
688 			}
689 		}
690 		finally {
691 			try {
692 				if (state == null || FileWritingMessageHandler.this.flushTask == null) {
693 					if (bos != null) {
694 						bos.close();
695 					}
696 					clearState(fileToWriteTo, state);
697 				}
698 				else {
699 					state.lastWrite = System.currentTimeMillis();
700 				}
701 			}
702 			catch (IOException ex) {
703 			}
704 		}
705 	}

814 	private File determineFileToWrite(File resultFile, File tempFile) {
815 
816 		final File fileToWriteTo;
817 
818 		switch (this.fileExistsMode) {
819 			case APPEND:
820 			case APPEND_NO_FLUSH:
821 				fileToWriteTo = resultFile;

823 			case FAIL:
824 			case IGNORE:

887 	private synchronized FileState getFileState(File fileToWriteTo, boolean isString)
888 			throws FileNotFoundException {
889 
890 		FileState state;
891 		boolean appendNoFlush = FileExistsMode.APPEND_NO_FLUSH.equals(this.fileExistsMode);
892 		if (appendNoFlush) {
893 			String absolutePath = fileToWriteTo.getAbsolutePath();
894 			state = this.fileStates.get(absolutePath);
895 			if (state != null // NOSONAR
896 					&& ((isString && state.stream != null) || (!isString && state.writer != null))) {
897 				state.close();
898 				state = null;
899 				this.fileStates.remove(absolutePath);
900 			}
901 			if (state == null) {
902 				if (isString) {
903 					state = new FileState(createWriter(fileToWriteTo, true),
904 							this.lockRegistry.obtain(fileToWriteTo.getAbsolutePath()));
905 				}
906 				else {
907 					state = new FileState(createOutputStream(fileToWriteTo, true),
908 							this.lockRegistry.obtain(fileToWriteTo.getAbsolutePath()));
909 				}
910 				this.fileStates.put(absolutePath, state);
911 			}
912 			state.lastWrite = Long.MAX_VALUE; // prevent flush while we write
913 		}
914 		else {
915 			state = null;
916 		}
917 		return state;
918 	}

987 	private Map<String, FileState> findFilesToFlush(MessageFlushPredicate flushPredicate, Message<?> filterMessage) {
988 		Map<String, FileState> toRemove = new HashMap<>();
989 		synchronized (this) {
990 			Iterator<Entry<String, FileState>> iterator = this.fileStates.entrySet().iterator();
991 			while (iterator.hasNext()) {
992 				Entry<String, FileState> entry = iterator.next();
993 				FileState state = entry.getValue();
994 				if (flushPredicate.shouldFlush(entry.getKey(), state.firstWrite, state.lastWrite, filterMessage)) {
995 					iterator.remove();
996 					toRemove.put(entry.getKey(), state);
997 				}
998 			}
999 		}
1000 		return toRemove;
1001 	}

1009 	private void doFlush(Map<String, FileState> toRemove) {
1010 		Map<String, FileState> toRestore = new HashMap<>();
1011 		boolean interrupted = false;
1012 		for (Entry<String, FileState> entry : toRemove.entrySet()) {
1013 			if (!interrupted && entry.getValue().close()) {
1014 				if (FileWritingMessageHandler.this.logger.isDebugEnabled()) {
1015 					FileWritingMessageHandler.this.logger.debug("Flushed: " + entry.getKey());
1016 				}
1017 			}
1018 			else { // interrupted (stop), re-add
1019 				interrupted = true;
1020 				toRestore.put(entry.getKey(), entry.getValue());
1021 			}
1022 		}
1023 		if (interrupted) {
1024 			if (FileWritingMessageHandler.this.logger.isDebugEnabled()) {
1025 				FileWritingMessageHandler.this.logger
1026 						.debug("Interrupted during flush; not flushed: " + toRestore.keySet());
1027 			}
1028 			synchronized (this) {
1029 				for (Entry<String, FileState> entry : toRestore.entrySet()) {
1030 					this.fileStates.putIfAbsent(entry.getKey(), entry.getValue());
1031 				}
1032 			}
1033 		}
1034 	}

1040 	private static final class FileState {
1041 
1042 		private final BufferedWriter writer;
1043 
1044 		private final BufferedOutputStream stream;
1045 
1046 		private final Lock lock;
1047 
1048 		private final long firstWrite = System.currentTimeMillis();
1049 
1050 		private volatile long lastWrite;
1051 
1052 		FileState(BufferedWriter writer, Lock lock) {
1053 			this.writer = writer;
1054 			this.stream = null;
1055 			this.lock = lock;
1056 		}
1057 
1058 		FileState(BufferedOutputStream stream, Lock lock) {
1059 			this.writer = null;
1060 			this.stream = stream;
1061 			this.lock = lock;
1062 		}
1063 
1064 		private boolean close() {
1065 			try {
1066 				this.lock.lockInterruptibly();
1067 				try {
1068 					if (this.writer != null) {
1069 						this.writer.close();
1070 					}
1071 					else {
1072 						this.stream.close();
1073 					}
1074 				}
1075 				catch (IOException e) {
1076 					// ignore
1077 				}
1078 				return true;
1079 			}
1080 			catch (InterruptedException e1) {
1081 				Thread.currentThread().interrupt();
1082 				return false;
1083 			}
1084 			finally {
1085 				this.lock.unlock();
1086 			}
1087 		}

1089 	}

1091 	private final class Flusher implements Runnable {
1092 
1093 		Flusher() {
1094 		}
1095 
1096 		@Override
1097 		public void run() {
1098 			Map<String, FileState> toRemove = new HashMap<>();
1099 			synchronized (FileWritingMessageHandler.this) {
1100 				long expired = FileWritingMessageHandler.this.flushTask == null ? Long.MAX_VALUE
1101 						: (System.currentTimeMillis() - FileWritingMessageHandler.this.flushInterval);
1102 				Iterator<Entry<String, FileState>> iterator =
1103 						FileWritingMessageHandler.this.fileStates.entrySet().iterator();
1104 				while (iterator.hasNext()) {
1105 					Entry<String, FileState> entry = iterator.next();
1106 					FileState state = entry.getValue();
1107 					if (state.lastWrite < expired ||
1108 							(!FileWritingMessageHandler.this.flushWhenIdle && state.firstWrite < expired)) {
1109 						toRemove.put(entry.getKey(), state);
1110 						iterator.remove();
1111 					}
1112 				}
1113 			}
1114 			doFlush(toRemove);
1115 		}
1116

1170

1172 		}
1173 
1174 		@Override
1175 		public boolean shouldFlush(String fileAbsolutePath, long firstWrite, long lastWrite,
1176 				Message<?> triggerMessage) {
1177 			Pattern pattern;
1178 			if (triggerMessage.getPayload() instanceof String) {
1179 				pattern = Pattern.compile((String) triggerMessage.getPayload());
1180 			}

1183 			}
1184 			else {

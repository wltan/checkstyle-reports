17  package org.springframework.integration.config;
18  
19  import java.lang.annotation.Documented;
20  import java.lang.annotation.ElementType;
21  import java.lang.annotation.Retention;
22  import java.lang.annotation.RetentionPolicy;
23  import java.lang.annotation.Target;
24  
25  import org.springframework.context.annotation.Import;
26  import org.springframework.core.annotation.AliasFor;
27  
28  /**
29   * Enables default configuring of management in Spring Integration components in an existing application.
30   *
31   * <p>The resulting {@link IntegrationManagementConfigurer}
32   * bean is defined under the name {@code integrationManagementConfigurer}.
33   *
34   * @author Gary Russell
35   *
36   * @since 4.2
37   */
38  @Target(ElementType.TYPE)
39  @Retention(RetentionPolicy.RUNTIME)
40  @Documented
41  @Import(IntegrationManagementConfiguration.class)
42  public @interface EnableIntegrationManagement {
43  
44  	/**
45  	 * A list of simple patterns for component names for which message counts will be
46  	 * enabled (defaults to '*'). Enables message
47  	 * counting (`sendCount`, `errorCount`, `receiveCount`) for those components that
48  	 * support counters (channels, message handlers, etc). This is the initial setting
49  	 * only, individual components can have counts enabled/disabled at runtime. May be
50  	 * overridden by an entry in {@link #statsEnabled() statsEnabled} which is additional
51  	 * functionality over simple counts. If a pattern starts with `!`, counts are disabled
52  	 * for matches. For components that match multiple patterns, the first pattern wins.
53  	 * Disabling counts at runtime also disables stats.
54  	 * Defaults to no components, unless JMX is enabled in which case, defaults to all
55  	 * components. Overrides {@link #defaultCountsEnabled()} for matching bean names.
56  	 * @return the patterns.
57  	 * @deprecated in favor of 'metersEnabled'.
58  	 */
59  	@Deprecated
60  	@AliasFor("metersEnabled")
61  	String[] countsEnabled() default "*";
62  
63  	/**
64  	 * A list of simple patterns for component names for which message counts will be
65  	 * enabled (defaults to '*'). Enables message
66  	 * counting (`sendCount`, `errorCount`, `receiveCount`) for those components that
67  	 * support counters (channels, message handlers, etc). This is the initial setting
68  	 * only, individual components can have counts enabled/disabled at runtime. May be
69  	 * overridden by an entry in {@link #statsEnabled() statsEnabled} which is additional
70  	 * functionality over simple counts. If a pattern starts with `!`, counts are disabled
71  	 * for matches. For components that match multiple patterns, the first pattern wins.
72  	 * Disabling counts at runtime also disables stats.
73  	 * Defaults to no components, unless JMX is enabled in which case, defaults to all
74  	 * components. Overrides {@link #defaultCountsEnabled()} for matching bean names.
75  	 * @return the patterns.
76  	 */
77  	@AliasFor("countsEnabled")
78  	String[] metersEnabled() default "*";
79  
80  	/**
81  	 * A list of simple patterns for component names for which message statistics will be
82  	 * enabled (response times, rates etc), as well as counts (a positive match here
83  	 * overrides {@link #countsEnabled() countsEnabled}, you can't have statistics without
84  	 * counts). (defaults to '*'). Enables
85  	 * statistics for those components that support statistics (channels - when sending,
86  	 * message handlers, etc). This is the initial setting only, individual components can
87  	 * have stats enabled/disabled at runtime. If a pattern starts with `!`, stats (and
88  	 * counts) are disabled for matches. Note: this means that '!foo' here will disable
89  	 * stats and counts for 'foo' even if counts are enabled for 'foo' in
90  	 * {@link #countsEnabled() countsEnabled}. For components
91  	 * that match multiple patterns, the first pattern wins. Enabling stats at runtime
92  	 * also enables counts.
93  	 * Defaults to no components, unless JMX is enabled in which case, defaults to all
94  	 * components.
95  	 * @return the patterns.
96  	 * @deprecated in favor of dimensional metrics via
97  	 * {@link org.springframework.integration.support.management.metrics.MeterFacade}.
98  	 * Built-in metrics will be removed in a future release.
99  	 */
100 	@Deprecated
101 	String[] statsEnabled() default "*";
102 
103 	/**
104 	 * The default setting for enabling counts when a bean name is not matched by
105 	 * {@link #countsEnabled() countsEnabled}.
106 	 * @return the value; false by default, or true when JMX is enabled.
107 	 */
108 	String defaultCountsEnabled() default "false";
109 
110 	/**
111 	 * The default setting for enabling statistics when a bean name is not matched by
112 	 * {@link #statsEnabled() statsEnabled}.
113 	 * @return the value; false by default, or true when JMX is enabled.
114 	 * @deprecated in favor of dimensional metrics via
115 	 * {@link org.springframework.integration.support.management.metrics.MeterFacade}.
116 	 * Built-in metrics will be removed in a future release.
117 	 */
118 	@Deprecated
119 	String defaultStatsEnabled() default "false";
120 
121 	/**
122 	 * Use to disable all logging in the main message flow in framework components. When 'false', such logging will be
123 	 * skipped, regardless of logging level. When 'true', the logging is controlled as normal by the logging
124 	 * subsystem log level configuration.
125 	 * <p>
126 	 * It has been found that in high-volume messaging environments, calls to methods such as
127 	 * {@code logger.isDebuggingEnabled()} can be quite expensive and account for an inordinate amount of CPU
128 	 * time.
129 	 * <p>
130 	 * Set this to false to disable logging by default in all framework components that implement
131 	 * {@link org.springframework.integration.support.management.IntegrationManagement}
132 	 * (channels, message handlers etc). This turns off logging such as
133 	 * "PreSend on channel", "Received message" etc.
134 	 * <p>
135 	 * After the context is initialized, individual components can have their setting changed by invoking
136 	 * {@link org.springframework.integration.support.management.IntegrationManagement#setLoggingEnabled(boolean)}.
137 	 * @return the value; true by default.
138 	 */
139 	String defaultLoggingEnabled() default "true";
140 
141 	/**
142 	 * The bean name of a {@code MetricsFactory}. The {@code DefaultMetricsFactory} is used
143 	 * if omitted.
144 	 * @return the bean name.
145 	 * @deprecated in favor of dimensional metrics via
146 	 * {@link org.springframework.integration.support.management.metrics.MeterFacade}.
147 	 * Built-in metrics will be removed in a future release.
148 	 */
149 	@Deprecated
150 	String metricsFactory() default "";
151 
152 }

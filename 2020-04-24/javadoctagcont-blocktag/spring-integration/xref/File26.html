17  package org.springframework.integration.support.management;
18  
19  import org.springframework.jmx.export.annotation.ManagedMetric;
20  import org.springframework.jmx.support.MetricType;
21  
22  /**
23   * Interface for all message channel monitors containing accessors for various useful
24   * metrics that are generic for all channel types.
25   *
26   * @author Dave Syer
27   * @author Gary Russell
28   * @deprecated in favor of dimensional metrics via
29   * {@link org.springframework.integration.support.management.metrics.MeterFacade}.
30   * Built-in metrics will be removed in a future release.
31   * @since 2.0
32   */
33  @Deprecated
34  public interface MessageChannelMetrics extends BaseChannelMetrics {
35  
36  	/**
37  	 * @return the number of successful sends
38  	 */
39  	@ManagedMetric(metricType = MetricType.COUNTER, displayName = "MessageChannel Send Count")
40  	int getSendCount();
41  
42  	/**
43  	 * @return the number of successful sends
44  	 * @since 3.0
45  	 */
46  	@ManagedMetric(metricType = MetricType.COUNTER, displayName = "MessageChannel Send Count")
47  	long getSendCountLong();
48  
49  
50  	@Override
51  	default long sendCount() {
52  		return getSendCountLong();
53  	}
54  
55  	/**
56  	 * @return the number of failed sends (either throwing an exception or rejected by the channel)
57  	 */
58  	@ManagedMetric(metricType = MetricType.COUNTER, displayName = "MessageChannel Send Error Count")
59  	int getSendErrorCount();
60  
61  	/**
62  	 * @return the number of failed sends (either throwing an exception or rejected by the channel)
63  	 * @since 3.0
64  	 */
65  	@ManagedMetric(metricType = MetricType.COUNTER, displayName = "MessageChannel Send Error Count")
66  	long getSendErrorCountLong();
67  
68  	@Override
69  	default long sendErrorCount() {
70  		return getSendErrorCountLong();
71  	}
72  
73  	/**
74  	 * @return the time in milliseconds since the last send
75  	 */
76  	@ManagedMetric(metricType = MetricType.GAUGE, displayName = "Channel Time Since Last Send in Milliseconds")
77  	double getTimeSinceLastSend();
78  
79  	/**
80  	 * @return the mean send rate (per second)
81  	 */
82  	@ManagedMetric(metricType = MetricType.GAUGE, displayName = "Channel Send Rate per Second")
83  	double getMeanSendRate();
84  
85  	/**
86  	 * @return the mean error rate (per second).  Errors comprise all failed sends.
87  	 */
88  	@ManagedMetric(metricType = MetricType.GAUGE, displayName = "Channel Error Rate per Second")
89  	double getMeanErrorRate();
90  
91  	/**
92  	 * @return the mean ratio of failed to successful sends in approximately the last minute
93  	 */
94  	@ManagedMetric(metricType = MetricType.GAUGE, displayName = "Mean Channel Error Ratio per Minute")
95  	double getMeanErrorRatio();
96  
97  	/**
98  	 * @return the mean send duration (milliseconds)
99  	 */
100 	@ManagedMetric(metricType = MetricType.GAUGE, displayName = "Channel Send Mean Duration in Milliseconds")
101 	double getMeanSendDuration();
102 
103 	/**
104 	 * @return the minimum send duration (milliseconds) since startup
105 	 */
106 	@ManagedMetric(metricType = MetricType.GAUGE, displayName = "Channel Send Min Duration in Milliseconds")
107 	double getMinSendDuration();
108 
109 	/**
110 	 * @return the maximum send duration (milliseconds) since startup
111 	 */
112 	@ManagedMetric(metricType = MetricType.GAUGE, displayName = "Channel Send Max Duration in Milliseconds")
113 	double getMaxSendDuration();
114 
115 	/**
116 	 * @return the standard deviation send duration (milliseconds)
117 	 */
118 	@ManagedMetric(metricType = MetricType.GAUGE, displayName = "Channel Send Standard Deviation Duration in Milliseconds")
119 	double getStandardDeviationSendDuration();
120 
121 	/**
122 	 * @return summary statistics about the send duration (milliseconds)
123 	 */
124 	Statistics getSendDuration();
125 
126 	/**
127 	 * @return summary statistics about the send rates (per second)
128 	 */
129 	Statistics getSendRate();
130 
131 	/**
132 	 * @return summary statistics about the error rates (per second)
133 	 */
134 	Statistics getErrorRate();
135 
136 }

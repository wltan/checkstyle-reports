37  /**
38   * Requires the requesting user to be {@link org.jsecurity.subject.Subject#isAuthenticated() authenticated} for the
39   * request to continue, and if they're not, forces the user to login via the HTTP Basic protocol-specific challenge.
40   * Upon successful login, they're allowed to continue on to the requested resource/url.
41   *
42   * <p>This implementation is a 'clean room' Java implementation of Basic HTTP Authentication specification per
43   * <a href="ftp://ftp.isi.edu/in-notes/rfc2617.txt">RFC 2617</a>.</p>
44   *
45   * <p>Basic authentication functions as follows:</p>
46   *
47   * <ol>
48   * <li>A request comes in for a resource that requires authentication.</li>
49   * <li>The server replies with a 401 response status, sets the <code>WWW-Authenticate</code> header, and the contents of a
50   * page informing the user that the incoming resource requires authentication.</li>
51   * <li>Upon receiving this <code>WWW-Authenticate</code> challenge from the server, the client then takes a
52   * username and a password and puts them in the following format:
53   * <p><code>username:password</code></p></li>
54   * <li>This token is then base 64 encoded.</li>
55   * <li>The client then sends another request for the same resource with the following header:<p/>
56   * <p><code>Authorization: Basic <em>Base64_encoded_username_and_password</em></code></p></li>
57   * </ol>
58   *
59   * <p>The {@link #onAccessDenied(javax.servlet.ServletRequest, javax.servlet.ServletResponse)} method will
60   * only be called if the subject making the request is not
61   * {@link org.jsecurity.subject.Subject#isAuthenticated() authenticated} </p>
62   *
63   * @author Allan Ditzel
64   * @author Les Hazlewood
65   * @see <a href="ftp://ftp.isi.edu/in-notes/rfc2617.txt">RFC 2617</a>
66   * @see <a href="http://en.wikipedia.org/wiki/Basic_access_authentication">Basic Access Authentication</a>
67   * @since 0.9
68   */
69  public class BasicHttpAuthenticationFilter extends AuthenticationFilter {

95       * Per RFC 2617, this name name is displayed to the end user when they are asked to authenticate.  Unless overridden
96       * by the {@link #setApplicationName(String) setApplicationName(String)} method, the default value is 'application'.
97       * <p/>
98       * Please see {@link #setApplicationName(String) setApplicationName(String)} for an example of how this functions.
99       *
100      * @return the name to use in the ServletResponse's 'WWW-Authenticate' header.
101      */

107      * Sets the name to use in the ServletResponse's <b><code>WWW-Authenticate</code></b> header.
108      * <p/>

110      * by this method, the default value is &quot;application&quot;
111      * <p/>
112      * For example, setting this property to the value <b><code>Awesome Webapp</code></b> will result in the

114      * <p/>

116      * <p/>
117      * Side note: As you can see from the header text, the HTTP Basic specification calls
118      * this the authentication 'realm', but we call this the 'applicationName' instead to avoid confusion with
119      * JSecurity's Realm constructs.
120      *
121      * @param applicationName the name to use in the ServletResponse's 'WWW-Authenticate' header.
122      */

124         this.applicationName = applicationName;

177     /**
178      * Processes unauthenticated requests. It handles the two-stage request/challenge authentication protocol.
179      *
180      * @param request  incoming ServletRequest
181      * @param response outgoing ServletResponse
182      * @return true if the request should be processed; false if the request should not continue to be processed
183      */
184     protected boolean onAccessDenied(ServletRequest request, ServletResponse response) {
185         boolean loggedIn = false; //false by default or we wouldn't be in this method
186         if (isLoginAttempt(request, response)) {
187             loggedIn = executeLogin(request, response);
188         }
189         if (!loggedIn) {
190             sendChallenge(request, response);
191         }
192         return loggedIn;
193     }

195     /**
196      * Determines whether the incoming request is an attempt to log in.
197      * <p/>
198      * The default implementation obtains the value of the request's
199      * {@link #AUTHORIZATION_HEADER AUTHORIZATION_HEADER}, and if it is not <code>null</code>, delegates
200      * to {@link #isLoginAttempt(String) isLoginAttempt(authzHeaderValue)}. If the header is <code>null</code>,
201      * <code>false</code> is returned.
202      *
203      * @param request  incoming ServletRequest
204      * @param response outgoing ServletResponse
205      * @return true if the incoming request is an attempt to log in based, false otherwise
206      */
207     protected boolean isLoginAttempt(ServletRequest request, ServletResponse response) {
208         String authzHeader = getAuthzHeader(request);
209         return authzHeader != null && isLoginAttempt(authzHeader);
210     }

212     /**
213      * Returns the {@link #AUTHORIZATION_HEADER AUTHORIZATION_HEADER} from the specified ServletRequest.
214      * <p/>
215      * This implementation merely casts the request to an <code>HttpServletRequest</code> and returns the header:
216      * <p/>
217      * <code>HttpServletRequest httpRequest = {@link WebUtils#toHttp(javax.servlet.ServletRequest) toHttp(reaquest)};<br/>
218      * return httpRequest.getHeader({@link #AUTHORIZATION_HEADER AUTHORIZATION_HEADER});</code>
219      *
220      * @param request the incoming <code>ServletRequest</code>
221      * @return the <code>Authorization</code> header's value.
222      */
223     protected String getAuthzHeader( ServletRequest request ) {
224         HttpServletRequest httpRequest = toHttp(request);
225         return httpRequest.getHeader(AUTHORIZATION_HEADER);
226     }

228     /**
229      * Default implementation that returns <code>true</code> if the specified <code>authzHeader</code>
230      * starts with the same (case-insensitive) characters specified by the
231      * {@link #getAuthzScheme() authzScheme}, <code>false</code> otherwise.
232      * <p/>
233      * That is:
234      * <p/>
235      * <code>String authzScheme = getAuthzScheme().toLowerCase();<br/>
236      * return authzHeader.toLowerCase().startsWith(authzScheme);</code>
237      *
238      * @param authzHeader the 'Authorization' header value (guaranteed to be non-null if the
239      *                    {@link #isLoginAttempt(javax.servlet.ServletRequest, javax.servlet.ServletResponse)} method is not overriden).
240      * @return <code>true</code> if the authzHeader value matches that configured as defined by
241      *         the {@link #getAuthzScheme() authzScheme}.
242      */
243     protected boolean isLoginAttempt(String authzHeader) {
244         String authzScheme = getAuthzScheme().toLowerCase();
245         return authzHeader.toLowerCase().startsWith(authzScheme);
246     }

248     /**
249      * Builds the challenge for authorization by setting a HTTP <code>401</code> (Unauthorized) status as well as the
250      * response's {@link #AUTHENTICATE_HEADER AUTHENTICATE_HEADER}.
251      * <p/>
252      * The header value constructed is equal to:
253      * <p/>
254      * <code>{@link #getAuthcScheme() getAuthcScheme()} + " realm=\"" + {@link #getApplicationName() getApplicationName()} + "\"";</code>
255      *
256      * @param request  incoming ServletRequest, ignored by this implementation
257      * @param response outgoing ServletResponse
258      * @return false - this sends the challenge to be sent back
259      */
260     protected boolean sendChallenge(ServletRequest request, ServletResponse response) {
261         if (log.isDebugEnabled()) {
262             log.debug("Authentication required: sending 401 Authentication challenge response.");
263         }
264         HttpServletResponse httpResponse = toHttp(response);
265         httpResponse.setStatus(HttpServletResponse.SC_UNAUTHORIZED);
266         String authcHeader = getAuthcScheme() + " realm=\"" + getApplicationName() + "\"";
267         httpResponse.setHeader(AUTHENTICATE_HEADER, authcHeader);
268         return false;
269     }

271     /**
272      * Executes a login attempt with the provided credentials in the http header and returns <code>true</code>
273      * if the login attempt is successful and <code>false</code> otherwise.
274      * <p/>
275      * This implementation:
276      * <ol><li>acquires the username and password based on the request's
277      * {@link #getAuthzHeader(javax.servlet.ServletRequest) authorization header} via the
278      * {@link #getPrincipalsAndCredentials(String, javax.servlet.ServletRequest) getPrincipalsAndCredentials} method</li>
279      * <li>The return value of that method is converted to an <code>AuthenticationToken</code> via the
280      * {@link #createToken(String, String, javax.servlet.ServletRequest) createToken} method</li>
281      * <li>Finally, the login attempt is executed using that token by calling
282      * {@link #executeLogin(org.jsecurity.authc.AuthenticationToken, javax.servlet.ServletRequest, javax.servlet.ServletResponse)}</li>
283      * </ol>
284      *
285      * @param request  incoming ServletRequest
286      * @param response outgoing ServletResponse
287      * @return true if the subject was successfully logged in, false otherwise
288      */
289     protected boolean executeLogin(ServletRequest request, ServletResponse response) {
290         if (log.isDebugEnabled()) {
291             log.debug("Attempting to authenticate Subject based on Http BASIC Authentication request...");
292         }
293 
294         String authorizationHeader = getAuthzHeader(request);
295         if (authorizationHeader == null || authorizationHeader.length() == 0 ) {
296             return false;
297         }
298 
299         if (log.isDebugEnabled()) {
300             log.debug("Attempting to execute login with headers [" + authorizationHeader + "]");
301         }
302 
303         String[] prinCred = getPrincipalsAndCredentials(authorizationHeader, request);
304         if ( prinCred == null || prinCred.length < 2 ) {
305             return false;
306         }
307 
308         String username = prinCred[0];
309         String password = prinCred[1];
310 
311         if (log.isDebugEnabled()) {
312             log.debug("Processing login request for username [" + username + "]");
313         }
314 
315         AuthenticationToken token = createToken(username, password, request );
316         if ( token != null ) {
317             return executeLogin(token, request, response );
318         }
319 
320         //always default to false.  If we've made it to this point in the code, that
321         //means the authentication attempt either never occured, or wasn't successful:
322         return false;
323     }

325     /**
326      * Executes a login attmept for the
327      * {@link #getSubject(javax.servlet.ServletRequest, javax.servlet.ServletResponse) currently executing}
328      * <code>Subject</code> using the specified authentication <code>token</code>.
329      * <p/>
330      * The login attempt constitutes calling {@link Subject#login currentSubject.login(token)}.  If the method
331      * call returns successfully, <code>true</code> is returned, <code>false</code> otherwise.
332      * 
333      * @param token the <code>AuthenticationToken</code> representing the submitted username and password.
334      * @param request the incoming ServletRequest
335      * @param response the outgoing ServletResponse
336      * @return <code>true</code> if the authentication attempt is successful, <code>false</code> otherwise.
337      */
338     protected boolean executeLogin( AuthenticationToken token, ServletRequest request, ServletResponse response ) {
339         Subject subject = getSubject(request, response);
340         if ( token != null && subject != null ) {
341             try {
342                 subject.login(token);
343                 if (log.isDebugEnabled()) {
344                     log.debug("Successfully logged in user [" + token.getPrincipal() + "]");
345                 }
346                 return true;
347             } catch (AuthenticationException ae) {
348                 if (log.isDebugEnabled()) {
349                     log.debug("Unable to log in user [" + token.getPrincipal()+ "]", ae);
350                 }
351             }
352         }
353 
354         //always default to false - authentication attempt never occurred or wasn't successful:
355         return false;
356     }

358     /**
359      * Returns the username obtained from the
360      * {@link #getAuthzHeader(javax.servlet.ServletRequest) authorizationHeader}.
361      * <p/>
362      * Once the <code>authzHeader is split per the RFC (based on the space character, " "), the resulting split tokens
363      * are translated into the username/password pair by the
364      * {@link #getPrincipalsAndCredentials(String, String) getPrincipalsAndCredentials(scheme,encoded)} method.
365      *
366      * @param authorizationHeader the authorization header obtained from the request.
367      * @param request the incoming ServletRequest
368      * @return the username (index 0)/password pair (index 1) submitted by the user for the given header value and request.
369      * @see #getAuthzHeader(javax.servlet.ServletRequest)
370      */
371     protected String[] getPrincipalsAndCredentials( String authorizationHeader, ServletRequest request ) {
372         if ( authorizationHeader == null ) {
373             return null;
374         }
375         String[] authTokens = authorizationHeader.split(" ");
376         if ( authTokens == null || authTokens.length < 2 ) {
377             return null;
378         }
379         return getPrincipalsAndCredentials(authTokens[0], authTokens[1]);
380     }
381 
382     /**
383      * Returns the username and password pair based on the specified <code>encoded</code> String obtained from
384      * the request's authorization header.
385      * <p/>
386      * Per RFC 2617, the default implementation first Base64 decodes the string and then splits the resulting decoded
387      * string into two based on the ":" character.  That is:
388      * <p/>
389      * <code>String decoded = Base64.decodeToString(encoded);<br/>
390      * return decoded.split(":");</code>
391      *
392      * @param scheme the {@link #getAuthcScheme() authcScheme} found in the request
393      * {@link #getAuthzHeader(javax.servlet.ServletRequest) authzHeader}.  It is ignored by this implementation,
394      * but available to overriding implementations should they find it useful.
395      * @param encoded the Base64-encoded username:password value found after the scheme in the header
396      * @return the username (index 0)/password (index 1) pair obtained from the encoded header data.
397      */
398     protected String[] getPrincipalsAndCredentials( String scheme, String encoded ) {
399         String decoded = Base64.decodeToString(encoded);
400         return decoded.split(":");
401     }

403     /**
404      * Creates an AuthenticationToken based on the username and password and incoming request to be submitted to
405      * the {@link Subject#login Subject.login} method for authentication.
406      * <p/>
407      * The default implementation acquires the request's associated
408      * {@link #getInetAddress(javax.servlet.ServletRequest) inetAddress} as well as a potential
409      * {@link #isRememberMeEnabled(javax.servlet.ServletRequest) rememberMe} status, and with the given
410      * <code>username</code> and <code>password</code>, returns a
411      * <code>new {@link org.jsecurity.authc.UsernamePasswordToken UsernamePasswordToken}</code>.  That is:
412      * <p/>
413      * <code>InetAddress addr = getInetAddress(request);<br/>
414      * boolean rememberMe = isRememberMeEnabled(request);<br/>
415      * return new UsernamePasswordToken(username, password, rememberMe, addr );</code>
416      * <p/>
417      * It should be noted that Basic HTTP Authentication does not support any concept of <code>rememberMe</code, but
418      * we still allow subclasses to enable this feature for any given request via the
419      * {@link #isRememberMeEnabled(javax.servlet.ServletRequest) isRememberMeEnabled} method if subclasses wish to
420      * override that method in custom environments.
421      *
422      * @param username the username obtained from the request's 'Authorization' header.
423      * @param password the password obtained from the request's 'Authorization' header.
424      * @param request the incoming ServletRequest.
425      * @return a constructed <code>AuthenticationToken</code> that will be used to execute a login attempt for the
426      * current <code>Subject</code>.
427      */
428     protected AuthenticationToken createToken( String username, String password, ServletRequest request ) {
429         InetAddress addr = getInetAddress(request);
430         boolean rememberMe = isRememberMeEnabled(request);
431         return new UsernamePasswordToken(username, password, rememberMe, addr );
432     }

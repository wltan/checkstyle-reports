28  /**
29   * The Try control gives us the ability write safe code without focusing on try-catch blocks in the presence of exceptions.
30   * <p>
31   * The following exceptions are considered to be fatal/non-recoverable:
32   * <ul>
33   * <li>{@linkplain InterruptedException}</li>
34   * <li>{@linkplain LinkageError}</li>
35   * <li>{@linkplain ThreadDeath}</li>
36   * <li>{@linkplain VirtualMachineError} (i.e. {@linkplain OutOfMemoryError} or {@linkplain StackOverflowError})</li>
37   * </ul>
38   * <p>
39   * <strong>Important note:</strong> Try may re-throw (undeclared) exceptions, e.g. on {@code get()}. From within a
40   * dynamic proxy {@link java.lang.reflect.InvocationHandler} this will lead to an
41   * {@link java.lang.reflect.UndeclaredThrowableException}. For more information, please read
42   * <a href="https://docs.oracle.com/javase/8/docs/technotes/guides/reflection/proxy.html">Dynamic Proxy Classes</a>.
43   *
44   * @param <T> Value type in the case of success.
45   * @author Daniel Dietrich
46   */
47  public interface Try<T> extends Value<T>, Serializable {

52       * Creates a Try of a CheckedFunction0.
53       *
54       * @param supplier A checked supplier
55       * @param <T>      Component type
56       * @return {@code Success(supplier.apply())} if no exception occurs, otherwise {@code Failure(throwable)} if an
57       * exception occurs calling {@code supplier.apply()}.
58       */

60          Objects.requireNonNull(supplier, "supplier is null");
61          try {
62              return new Success<>(supplier.apply());

64              return new Failure<>(t);
65          }

124     /**
125      * Reduces many {@code Try}s into a single {@code Try} by transforming an
126      * {@code Iterable<Try<? extends T>>} into a {@code Try<Seq<T>>}. If any of
127      * the {@code Try}s are {@link Try.Failure}, then this returns a {@link Try.Failure}.
128      *
129      * @param values An {@link Iterable} of {@code Try}s
130      * @param <T>    type of the Trys
131      * @return A {@code Try} of a {@link Seq} of results
132      * @throws NullPointerException if {@code values} is null
133      */
134     static <T> Try<Seq<T>> sequence(Iterable<? extends Try<? extends T>> values) {
135         Objects.requireNonNull(values, "values is null");
136         Vector<T> vector = Vector.empty();
137         for (Try<? extends T> value : values) {
138             if (value.isFailure()) {
139                 return Try.failure(value.getCause());
140             }
141             vector = vector.append(value.get());
142         }
143         return Try.success(vector);
144     }

24  package org.hibernate;
25  
26  /**
27   * Loads an entity by its natural identifier
28   * 
29   * @author Eric Dalquist
30   * @author Steve Ebersole
31   *
32   * @see org.hibernate.annotations.NaturalId
33   */
34  public interface NaturalIdLoadAccess {
35  	/**
36  	 * Specify the {@link LockOptions} to use when retrieving the entity.
37  	 *
38  	 * @param lockOptions The lock options to use.
39  	 *
40  	 * @return {@code this}, for method chaining
41  	 */
42  	public NaturalIdLoadAccess with(LockOptions lockOptions);
43  
44  	/**
45  	 * Add a NaturalId attribute value.
46  	 * 
47  	 * @param attributeName The entity attribute name that is marked as a NaturalId
48  	 * @param value The value of the attribute
49  	 *
50  	 * @return {@code this}, for method chaining
51  	 */
52  	public NaturalIdLoadAccess using(String attributeName, Object value);
53  
54  	/**
55  	 * For entities with mutable natural ids, should Hibernate perform "synchronization" prior to performing
56  	 * lookups?  The default is to perform "synchronization" (for correctness).
57  	 * <p/>
58  	 * "synchronization" here indicates updating the natural-id -> pk cross reference maintained as part of the
59  	 * session.  When enabled, prior to performing the lookup, Hibernate will check all entities of the given
60  	 * type associated with the session to see if its natural-id values have changed and, if so, update the
61  	 * cross reference.  There is a performance impact associated with this, so if application developers are
62  	 * certain the natural-ids in play have not changed, this setting can be disabled to circumvent that impact.
63  	 * However, disabling this setting when natural-ids values have changed can result in incorrect results!
64  	 *
65  	 * @param enabled Should synchronization be performed?  {@code true} indicates synchronization will be performed;
66  	 * {@code false} indicates it will be circumvented.
67  	 *
68  	 * @return {@code this}, for method chaining
69  	 */
70  	public NaturalIdLoadAccess setSynchronizationEnabled(boolean enabled);
71  
72  	/**
73  	 * Return the persistent instance with the natural id value(s) defined by the call(s) to {@link #using}.  This
74  	 * method might return a proxied instance that is initialized on-demand, when a non-identifier method is accessed.
75  	 *
76  	 * You should not use this method to determine if an instance exists; to check for existence, use {@link #load}
77  	 * instead.  Use this only to retrieve an instance that you assume exists, where non-existence would be an
78  	 * actual error.
79  	 *
80  	 * @return the persistent instance or proxy
81  	 */
82  	public Object getReference();
83  
84  	/**
85  	 * Return the persistent instance with the natural id value(s) defined by the call(s) to {@link #using}, or
86  	 * {@code null} if there is no such persistent instance.  If the instance is already associated with the session,
87  	 * return that instance, initializing it if needed.  This method never returns an uninitialized instance.
88  	 *
89  	 * @return The persistent instance or {@code null} 
90  	 */
91  	public Object load();
92  
93  }

25  package org.hibernate.transaction;
26  import java.util.Properties;
27  import javax.transaction.Transaction;
28  import javax.transaction.TransactionManager;
29  
30  import org.hibernate.HibernateException;
31  
32  /**
33   * Contract for locating the JTA {@link TransactionManager} on given platform.
34   * <p/>
35   * NOTE: this contract has expanded over time, and basically is a platform
36   * abstraction contract for JTA-related information.
37   *
38   * @author Gavin King
39   */
40  public interface TransactionManagerLookup {
41  
42  	/**
43  	 * Obtain the JTA {@link TransactionManager}.
44  	 *
45  	 * @param props The configuration properties.
46  	 * @return The JTA {@link TransactionManager}.
47  	 *
48  	 * @throws HibernateException Indicates problem locating {@link TransactionManager}.
49  	 */
50  	public TransactionManager getTransactionManager(Properties props) throws HibernateException;
51  
52  	/**
53  	 * Return the JNDI namespace of the JTA
54  	 * {@link javax.transaction.UserTransaction} for this platform or <tt>null</tt>;
55  	 * optional operation.
56  	 *
57  	 * @return The JNDI namespace where we can locate the
58  	 * {@link javax.transaction.UserTransaction} for this platform.
59  	 */
60  	public String getUserTransactionName();
61  
62  	/**
63  	 * Determine an identifier for the given transaction appropriate for use in caching/lookup usages.
64  	 * <p/>
65  	 * Generally speaking the transaction itself will be returned here.  This method was added specifically
66  	 * for use in WebSphere and other unfriendly JEE containers (although WebSphere is still the only known
67  	 * such brain-dead, sales-driven impl).
68  	 *
69  	 * @param transaction The transaction to be identified.
70  	 * @return An appropropriate identifier
71  	 */
72  	public Object getTransactionIdentifier(Transaction transaction);
73  }

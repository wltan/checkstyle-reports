36  /**
37   * <p>A Properties collection which preserves comments and whitespace
38   * present in the input stream from which it was loaded.</p>
39   * <p>The class defers the usual work of the <a href="http://java.sun.com/j2se/1.3/docs/api/java/util/Properties.html">java.util.Properties</a>
40   * class to there, but it also keeps track of the contents of the
41   * input stream from which it was loaded (if applicable), so that it can
42   * write out the properties in as close a form as possible to the input.</p>
43   * If no changes occur to property values, the output should be the same
44   * as the input, except for the leading date stamp, as normal for a
45   * properties file. Properties added are appended to the file. Properties
46   * whose values are changed are changed in place. Properties that are
47   * removed are excised. If the <code>removeComments</code> flag is set,
48   * then the comments immediately preceding the property are also removed.</p>
49   * <p>If a second set of properties is loaded into an existing set, the
50   * lines of the second set are added to the end. Note however, that if a
51   * property already stored is present in a stream subsequently loaded, then
52   * that property is removed before the new value is set. For example,
53   * consider the file</p>
54   * <pre> # the first line
55   * alpha=one
56   *
57   * # the second line
58   * beta=two</pre>
59   * <p>This file is loaded, and then the following is also loaded into the
60   * same <code>LayoutPreservingProperties</code> object</p>
61   * <pre> # association
62   * beta=band
63   *
64   * # and finally
65   * gamma=rays</pre>
66   * </p>The resulting collection sequence of logical lines depends on whether
67   * or not <code>removeComments</code> was set at the time the second stream
68   * is loaded. If it is set, then the resulting list of lines is</p>
69   * <pre> # the first line
70   * alpha=one
71   *
72   * # association
73   * beta=band
74   *
75   * # and finally
76   * gamma=rays</pre>
77   * <p>If the flag is not set, then the comment "the second line" is retained,
78   * although the key-value pair <code>beta=two</code> is removed.</p>
79   */
80  public class LayoutPreservingProperties extends Properties {
81      private String LS = StringUtils.LINE_SEP;

163     /**
164      * Store a new key-value pair, or add a new one. The normal
165      * functionality is taken care of by the superclass in the call to
166      * {@link #setProperty}; this method takes care of this classes
167      * extensions.
168      * @param key the key of the property to be stored
169      * @param value the value to be stored
170      */
171     private void innerSetProperty(String key, String value) {
172         value = escapeValue(value);
173 
174         if (keyedPairLines.containsKey(key)) {
175             Integer i = (Integer) keyedPairLines.get(key);
176             Pair p = (Pair) logicalLines.get(i.intValue());
177             p.setValue(value);
178         } else {
179             key = escapeName(key);
180             Pair p = new Pair(key, value);
181             p.setNew(true);
182             keyedPairLines.put(key, new Integer(logicalLines.size()));
183             logicalLines.add(p);
184         }
185     }

187     public void clear() {
188         super.clear();
189         keyedPairLines.clear();
190         logicalLines.clear();
191     }

193     public Object remove(Object key) {
194         Object obj = super.remove(key);
195         Integer i = (Integer) keyedPairLines.remove(key);
196         if (null != i) {
197             if (removeComments) {
198                 removeCommentsEndingAt(i.intValue());
199             }
200             logicalLines.set(i.intValue(), null);
201         }
202         return obj;
203     }

205     public Object clone() {
206         LayoutPreservingProperties dolly =
207             (LayoutPreservingProperties) super.clone();
208         dolly.keyedPairLines = (HashMap) this.keyedPairLines.clone();
209         dolly.logicalLines = (ArrayList) this.logicalLines.clone();
210         final int size = dolly.logicalLines.size();
211         for (int j = 0; j < size; j++) {
212             LogicalLine line = (LogicalLine) dolly.logicalLines.get(j);
213             if (line instanceof Pair) {
214                 Pair p = (Pair) line;
215                 dolly.logicalLines.set(j, p.clone());
216             }
217             // no reason to clone other lines are they are immutable
218         }
219         return dolly;
220     }

254     public void store(OutputStream out, String header) throws IOException {
255         OutputStreamWriter osw = new OutputStreamWriter(out, ResourceUtils.ISO_8859_1);
256 
257         int skipLines = 0;
258         int totalLines = logicalLines.size();
259 
260         if (header != null) {
261             osw.write("#" + header + LS);
262             if (totalLines > 0
263                 && logicalLines.get(0) instanceof Comment
264                 && header.equals(logicalLines.get(0).toString().substring(1))) {
265                 skipLines = 1;
266             }
267         }
268 
269         // we may be updatiung a file written by this class, replace
270         // the date comment instead of adding a new one and preserving
271         // the one written last time
272         if (totalLines > skipLines
273             && logicalLines.get(skipLines) instanceof Comment) {
274             try {
275                 DateUtils.parseDateFromHeader(logicalLines
276                                               .get(skipLines)
277                                               .toString().substring(1));
278                 skipLines++;
279             } catch (java.text.ParseException pe) {
280                 // not an existing date comment
281             }
282         }
283         osw.write("#" + DateUtils.getDateForHeader() + LS);
284 
285         boolean writtenSep = false;
286         for (Iterator i = logicalLines.subList(skipLines, totalLines).iterator();
287              i.hasNext(); ) {
288             LogicalLine line = (LogicalLine) i.next();
289             if (line instanceof Pair) {
290                 if (((Pair)line).isNew()) {
291                     if (!writtenSep) {
292                         osw.write(LS);
293                         writtenSep = true;
294                     }
295                 }
296                 osw.write(line.toString() + LS);
297             }
298             else if (line != null) {
299                 osw.write(line.toString() + LS);
300             }
301         }
302         osw.close();
303     }

305     /**
306      * Reads a properties file into an internally maintained
307      * collection of logical lines (possibly spanning physcial lines),
308      * which make up the comments, blank lines and properties of the
309      * file.
310      * @param is the stream from which to read the data
311      */
312     private String readLines(InputStream is) throws IOException {
313         InputStreamReader isr = new InputStreamReader(is, ResourceUtils.ISO_8859_1);
314         PushbackReader pbr = new PushbackReader(isr, 1);
315 
316         if (logicalLines.size() > 0) {
317             // we add a blank line for spacing
318             logicalLines.add(new Blank());
319         }
320 
321         String s = readFirstLine(pbr);
322         BufferedReader br = new BufferedReader(pbr);
323 
324         boolean continuation = false;
325         boolean comment = false;
326         StringBuffer fileBuffer = new StringBuffer();
327         StringBuffer logicalLineBuffer = new StringBuffer();
328         while (s != null) {
329             fileBuffer.append(s).append(LS);
330 
331             if (continuation) {
332                 // put in the line feed that was removed
333                 s = "\n" + s;
334             } else {
335                 // could be a comment, if first non-whitespace is a # or !
336                 comment = s.matches("^( |\t|\f)*(#|!).*");
337             }
338 
339             // continuation if not a comment and the line ends is an
340             // odd number of backslashes
341             if (!comment) {
342                 continuation = requiresContinuation(s);
343             }
344 
345             logicalLineBuffer.append(s);
346 
347             if (!continuation) {
348                 LogicalLine line = null;
349                 if (comment) {
350                     line = new Comment(logicalLineBuffer.toString());
351                 } else if (logicalLineBuffer.toString().trim().length() == 0) {
352                     line = new Blank();
353                 } else {
354                     line = new Pair(logicalLineBuffer.toString());
355                     String key = unescape(((Pair)line).getName());
356                     if (keyedPairLines.containsKey(key)) {
357                         // this key is already present, so we remove it and add
358                         // the new one
359                         remove(key);
360                     }
361                     keyedPairLines.put(key, new Integer(logicalLines.size()));
362                 }
363                 logicalLines.add(line);
364                 logicalLineBuffer.setLength(0);
365             }
366             s = br.readLine();
367         }
368         return fileBuffer.toString();
369     }

371     /**
372      * Reads the first line and determines the EOL-style of the file
373      * (relies on the style to be consistent, of course).
374      *
375      * <p>Sets LS as a side-effect.</p>
376      *
377      * @return the first line without any line separator, leaves the
378      * reader positioned after the first line separator
379      *
380      * @since Ant 1.8.2
381      */
382     private String readFirstLine(PushbackReader r) throws IOException {
383         StringBuffer sb = new StringBuffer(80);
384         int ch = r.read();
385         boolean hasCR = false;
386         // when reaching EOF before the first EOL, assume native line
387         // feeds
388         LS = StringUtils.LINE_SEP;
389 
390         while (ch >= 0) {
391             if (hasCR && ch != '\n') {
392                 // line feed is sole CR
393                 r.unread(ch);
394                 break;
395             }
396 
397             if (ch == '\r') {
398                 LS = "\r";
399                 hasCR = true;
400             } else if (ch == '\n') {
401                 LS = hasCR ? "\r\n" : "\n";
402                 break;
403             } else {
404                 sb.append((char) ch);
405             }
406             ch = r.read();
407         }
408         return sb.toString();
409     }

411     /**
412      * Returns <code>true</code> if the line represented by
413      * <code>s</code> is to be continued on the next line of the file,
414      * or <code>false</code> otherwise.
415      * @param s the contents of the line to examine
416      * @return <code>true</code> if the line is to be continued,
417      * <code>false</code> otherwise
418      */
419     private boolean requiresContinuation(String s) {
420         char[] ca = s.toCharArray();
421         int i = ca.length - 1;
422         while (i > 0 && ca[i] == '\\') {
423             i--;
424         }
425         // trailing backslashes
426         int tb = ca.length - i - 1;
427         return tb % 2 == 1;
428     }

430     /**
431      * Unescape the string according to the rules for a Properites
432      * file, as laid out in the docs for <a
433      * href="http://java.sun.com/j2se/1.3/docs/api/java/util/Properties.html">java.util.Properties</a>.
434      * @param s the string to unescape (coming from the source file)
435      * @return the unescaped string
436      */
437     private String unescape(String s) {
438         /*
439          * The following combinations are converted:
440          * \n  newline
441          * \r  carraige return
442          * \f  form feed
443          * \t  tab
444          * \\  backslash
445          * \u0000  unicode character
446          * Any other slash is ignored, so
447          * \b  becomes 'b'.
448          */
449 
450         char[] ch = new char[s.length() + 1];
451         s.getChars(0, s.length(), ch, 0);
452         ch[s.length()] = '\n';
453         StringBuffer buffy = new StringBuffer(s.length());
454         for (int i = 0; i < ch.length; i++) {
455             char c = ch[i];
456             if (c == '\n') {
457                 // we have hit out end-of-string marker
458                 break;
459             }
460             else if (c == '\\') {
461                 // possibly an escape sequence
462                 c = ch[++i];
463                 if (c == 'n')
464                     buffy.append('\n');
465                 else if (c == 'r')
466                     buffy.append('\r');
467                 else if (c == 'f')
468                     buffy.append('\f');
469                 else if (c == 't')
470                     buffy.append('\t');
471                 else if (c == 'u') {
472                     // handle unicode escapes
473                     c = unescapeUnicode(ch, i+1);
474                     i += 4;
475                     buffy.append(c);
476                 }
477                 else
478                     buffy.append(c);
479             }
480             else {
481                 buffy.append(c);
482             }
483         }
484         return buffy.toString();
485     }

492      */
493     private char unescapeUnicode(char[] ch, int i) {
494         String s = new String(ch, i, 4);
495         return (char) Integer.parseInt(s, 16);

522     /**
523      * Escape the string <code>s</code> according to the rules in the
524      * docs for <a
525      * href="http://java.sun.com/j2se/1.3/docs/api/java/util/Properties.html">java.util.Properties</a>.
526      * @param s the string to escape
527      * @param escapeAllSpaces if <code>true</code> the method escapes
528      * all the spaces, if <code>false</code>, it escapes only the
529      * leading whitespace
530      * @return the escaped string
531      */
532     private String escape(String s, boolean escapeAllSpaces) {
533         if (s == null) {
534             return null;
535         }
536 
537         char[] ch = new char[s.length()];
538         s.getChars(0, s.length(), ch, 0);
539         String forEscaping = "\t\f\r\n\\:=#!";
540         String escaped = "tfrn\\:=#!";
541         StringBuffer buffy = new StringBuffer(s.length());
542         boolean leadingSpace = true;
543         for (int i = 0; i < ch.length; i++) {
544             char c = ch[i];
545             if (c == ' ') {
546                 if (escapeAllSpaces || leadingSpace) {
547                     buffy.append("\\");
548                 }
549             } else {
550                 leadingSpace = false;
551             }
552             int p = forEscaping.indexOf(c);
553             if (p != -1) {
554                 buffy.append("\\").append(escaped.substring(p,p+1));
555             } else if (c < 0x0020 || c > 0x007e) {
556                 buffy.append(escapeUnicode(c));
557             } else {
558                 buffy.append(c);
559             }
560         }
561         return buffy.toString();
562     }

574     /**
575      * Remove the comments in the leading up the {@link logicalLines}
576      * list leading up to line <code>pos</code>.
577      * @param pos the line number to which the comments lead
578      */
579     private void removeCommentsEndingAt(int pos) {
580         /* We want to remove comments preceding this position. Step
581          * back counting blank lines (call this range B1) until we hit
582          * something non-blank. If what we hit is not a comment, then
583          * exit. If what we hit is a comment, then step back counting
584          * comment lines (call this range C1). Nullify lines in C1 and
585          * B1.
586          */
587 
588         int end = pos - 1;
589 
590         // step pos back until it hits something non-blank
591         for (pos = end; pos > 0; pos--) {
592             if (!(logicalLines.get(pos) instanceof Blank)) {
593                 break;
594             }
595         }
596 
597         // if the thing it hits is not a comment, then we have nothing
598         // to remove
599         if (!(logicalLines.get(pos) instanceof Comment)) {
600             return;
601         }
602 
603         // step back until we hit the start of the comment
604         for (; pos >= 0; pos--) {
605             if (!(logicalLines.get(pos) instanceof Comment)) {
606                 break;
607             }
608         }
609 
610         // now we want to delete from pos+1 to end
611         for (pos++ ;pos <= end; pos++) {
612             logicalLines.set(pos, null);
613         }
614     }

705         private void parsePair(String text) {
706             // need to find first non-escaped '=', ':', '\t' or ' '.
707             int pos = findFirstSeparator(text);
708             if (pos == -1) {
709                 // trim leading whitespace only
710                 name = text;
711                 value = null;
712             }
713             else {
714                 name = text.substring(0, pos);
715                 value = text.substring(pos+1, text.length());
716             }
717             // trim leading whitespace only
718             name = stripStart(name, " \t\f");
719         }
720 
721         private String stripStart(String s, String chars) {
722             if (s == null) {
723                 return null;
724             }
725 
726             int i = 0;
727             for (;i < s.length(); i++) {
728                 if (chars.indexOf(s.charAt(i)) == -1) {
729                     break;
730                 }
731             }
732             if (i == s.length()) {
733                 return "";
734             }
735             return s.substring(i);
736         }
737 
738         private int findFirstSeparator(String s) {
739             // Replace double backslashes with underscores so that they don't
740             // confuse us looking for '\t' or '\=', for example, but they also
741             // don't change the position of other characters
742             s = s.replaceAll("\\\\\\\\", "__");
743 
744             // Replace single backslashes followed by separators, so we don't
745             // pick them up
746             s = s.replaceAll("\\\\=", "__");
747             s = s.replaceAll("\\\\:", "__");
748             s = s.replaceAll("\\\\ ", "__");
749             s = s.replaceAll("\\\\t", "__");
750 
751             // Now only the unescaped separators are left
752             return indexOfAny(s, " :=\t");
753         }
754 
755         private int indexOfAny(String s, String chars) {
756             if (s == null || chars == null) {
757                 return -1;
758             }
759 
760             int p = s.length() + 1;
761             for (int i = 0; i < chars.length(); i++) {
762                 int x = s.indexOf(chars.charAt(i));
763                 if (x != -1 && x < p) {
764                     p = x;
765                 }
766             }
767             if (p == s.length() + 1) {
768                 return -1;
769             }
770             return p;
771         }
772     }

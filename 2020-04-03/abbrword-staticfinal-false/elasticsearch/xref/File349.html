68  public class LocalIndexShardGateway extends AbstractIndexShardComponent implements IndexShardGateway {
69  
70      private static final int RECOVERY_TRANSLOG_RENAME_RETRIES = 3;
71  
72      private final ThreadPool threadPool;
73      private final MappingUpdatedAction mappingUpdatedAction;
74      private final IndexService indexService;
75      private final IndexShard indexShard;
76  
77      private final TimeValue waitForMappingUpdatePostRecovery;
78  
79      private volatile ScheduledFuture flushScheduler;
80      private final TimeValue syncInterval;
81      private final CancellableThreads cancellableThreads = new CancellableThreads();
82  
83      @Inject
84      public LocalIndexShardGateway(ShardId shardId, @IndexSettings Settings indexSettings, ThreadPool threadPool, MappingUpdatedAction mappingUpdatedAction,
85                                    IndexService indexService, IndexShard indexShard) {
86          super(shardId, indexSettings);
87          this.threadPool = threadPool;
88          this.mappingUpdatedAction = mappingUpdatedAction;
89          this.indexService = indexService;
90          this.indexShard = indexShard;
91  
92          this.waitForMappingUpdatePostRecovery = componentSettings.getAsTime("wait_for_mapping_update_post_recovery", TimeValue.timeValueSeconds(30));
93          syncInterval = componentSettings.getAsTime("sync", TimeValue.timeValueSeconds(5));
94          if (syncInterval.millis() > 0) {
95              this.indexShard.translog().syncOnEachOperation(false);
96              flushScheduler = threadPool.schedule(syncInterval, ThreadPool.Names.SAME, new Sync());
97          } else if (syncInterval.millis() == 0) {
98              flushScheduler = null;
99              this.indexShard.translog().syncOnEachOperation(true);
100         } else {
101             flushScheduler = null;
102         }
103     }

110     @Override
111     public void recover(boolean indexShouldExists, RecoveryState recoveryState) throws IndexShardGatewayRecoveryException {
112         indexShard.prepareForIndexRecovery();
113         long version = -1;
114         long translogId = -1;
115         SegmentInfos si = null;
116         final Set<String> typesToUpdate = Sets.newHashSet();
117         indexShard.store().incRef();
118         try {
119             try {
120                 indexShard.store().failIfCorrupted();
121                 try {
122                     si = Lucene.readSegmentInfos(indexShard.store().directory());
123                 } catch (Throwable e) {
124                     String files = "_unknown_";
125                     try {
126                         files = Arrays.toString(indexShard.store().directory().listAll());
127                     } catch (Throwable e1) {
128                         files += " (failure=" + ExceptionsHelper.detailedMessage(e1) + ")";
129                     }
130                     if (indexShouldExists && indexShard.indexService().store().persistent()) {
131                         throw new IndexShardGatewayRecoveryException(shardId(), "shard allocated for local recovery (post api), should exist, but doesn't, current files: " + files, e);
132                     }
133                 }
134                 if (si != null) {
135                     if (indexShouldExists) {
136                         version = si.getVersion();
137                         /**
138                          * We generate the translog ID before each lucene commit to ensure that
139                          * we can read the current translog ID safely when we recover. The commits metadata
140                          * therefor contains always the current / active translog ID.
141                          */
142                         if (si.getUserData().containsKey(Translog.TRANSLOG_ID_KEY)) {
143                             translogId = Long.parseLong(si.getUserData().get(Translog.TRANSLOG_ID_KEY));
144                         } else {
145                             translogId = version;
146                         }
147                         logger.trace("using existing shard data, translog id [{}]", translogId);
148                     } else {
149                         // it exists on the directory, but shouldn't exist on the FS, its a leftover (possibly dangling)
150                         // its a "new index create" API, we have to do something, so better to clean it than use same data
151                         logger.trace("cleaning existing shard, shouldn't exists");
152                         IndexWriter writer = new IndexWriter(indexShard.store().directory(), new IndexWriterConfig(Lucene.VERSION, Lucene.STANDARD_ANALYZER).setOpenMode(IndexWriterConfig.OpenMode.CREATE));
153                         writer.close();
154                     }
155                 }
156             } catch (Throwable e) {
157                 throw new IndexShardGatewayRecoveryException(shardId(), "failed to fetch index version after copying it over", e);
158             }
159             recoveryState.getIndex().updateVersion(version);
160 
161             // since we recover from local, just fill the files and size
162             try {
163                 final RecoveryState.Index index = recoveryState.getIndex();
164                 if (si != null) {
165                     final Directory directory = indexShard.store().directory();
166                     for (String name : Lucene.files(si)) {
167                         long length = directory.fileLength(name);
168                         index.addFileDetail(name, length, true);
169                     }
170                 }
171             } catch (IOException e) {
172                 logger.debug("failed to list file details", e);
173             }
174 
175             File recoveringTranslogFile = null;
176             if (translogId == -1) {
177                 logger.trace("no translog id set (indexShouldExist [{}])", indexShouldExists);
178             } else {
179 
180                 // move an existing translog, if exists, to "recovering" state, and start reading from it
181                 FsTranslog translog = (FsTranslog) indexShard.translog();
182                 String translogName = "translog-" + translogId;
183                 String recoverTranslogName = translogName + ".recovering";
184 
185                 logger.trace("try recover from translog file {} locations: {}", translogName, Arrays.toString(translog.locations()));
186                 for (File translogLocation : translog.locations()) {
187                     File tmpRecoveringFile = new File(translogLocation, recoverTranslogName);
188                     if (!tmpRecoveringFile.exists()) {
189                         File tmpTranslogFile = new File(translogLocation, translogName);
190                         if (tmpTranslogFile.exists()) {
191                             logger.trace("Translog file found in {} - renaming", translogLocation);
192                             boolean success = false;
193                             for (int i = 0; i < RECOVERY_TRANSLOG_RENAME_RETRIES; i++) {
194                                 if (tmpTranslogFile.renameTo(tmpRecoveringFile)) {
195 
196                                     recoveringTranslogFile = tmpRecoveringFile;
197                                     logger.trace("Renamed translog from {} to {}", tmpTranslogFile.getName(), recoveringTranslogFile.getName());
198                                     success = true;
199                                     break;
200                                 }
201                             }
202                             if (success == false) {
203                                 try {
204                                     // this is a fallback logic that to ensure we can recover from the file.
205                                     // on windows a virus-scanner etc can hold on to the file and after retrying
206                                     // we just skip the recovery and the engine will reuse the file and truncate it.
207                                     // in 2.0 this is all not needed since translog files are write once.
208                                     Files.copy(tmpTranslogFile.toPath(), tmpRecoveringFile.toPath());
209                                     recoveringTranslogFile = tmpRecoveringFile;
210                                     logger.trace("Copied translog from {} to {}", tmpTranslogFile.getName(), recoveringTranslogFile.getName());
211                                 } catch (IOException ex) {
212                                     throw new ElasticsearchException("failed to copy recovery file", ex);
213                                 }
214                             }
215                         } else {
216                             logger.trace("Translog file NOT found in {} - continue", translogLocation);
217                         }
218                     } else {
219                         recoveringTranslogFile = tmpRecoveringFile;
220                         break;
221                     }
222                 }
223             }
224             // we must do this *after* we capture translog name so the engine creation will not make a new one.
225             // also we have to do this regardless of whether we have a translog, to follow the recovery stages.
226             indexShard.prepareForTranslogRecovery();
227 
228             if (recoveringTranslogFile == null || !recoveringTranslogFile.exists()) {
229                 // no translog to recovery from, start and bail
230                 // no translog files, bail
231                 recoveryState.getTranslog().totalOperations(0);
232                 recoveryState.getTranslog().totalOperationsOnStart(0);
233                 indexShard.finalizeRecovery();
234                 indexShard.postRecovery("post recovery from gateway, no translog");
235                 // no index, just start the shard and bail
236                 return;
237             }
238 
239             StreamInput in = null;
240             if (logger.isTraceEnabled()) {
241                 logger.trace("recovering translog file: {} length: {}", recoveringTranslogFile, recoveringTranslogFile.length());
242             }
243             try {
244                 TranslogStream stream = TranslogStreams.translogStreamFor(recoveringTranslogFile);
245                 try {
246                     in = stream.openInput(recoveringTranslogFile);
247                 } catch (TruncatedTranslogException e) {
248                     // file is empty or header has been half-written and should be ignored
249                     logger.trace("ignoring truncation exception, the translog is either empty or half-written", e);
250                 }
251                 while (true) {
252                     if (in == null) {
253                         break;
254                     }
255                     Translog.Operation operation;
256                     try {
257                         if (stream instanceof LegacyTranslogStream) {
258                             in.readInt(); // ignored opSize
259                         }
260                         operation = stream.read(in);
261                     } catch (TruncatedTranslogException|EOFException e) {
262                         // ignore, not properly written the last op
263                         logger.trace("ignoring translog EOF exception, the last operation was not properly written", e);
264                         break;
265                     } catch (IOException e) {
266                         // ignore, not properly written last op
267                         logger.trace("ignoring translog IO exception, the last operation was not properly written", e);
268                         break;
269                     }
270                     try {
271                         Engine.IndexingOperation potentialIndexOperation = indexShard.performRecoveryOperation(operation);
272                         if (potentialIndexOperation != null && potentialIndexOperation.parsedDoc().mappingsModified()) {
273                             if (!typesToUpdate.contains(potentialIndexOperation.docMapper().type())) {
274                                 typesToUpdate.add(potentialIndexOperation.docMapper().type());
275                             }
276                         }
277                         recoveryState.getTranslog().incrementRecoveredOperations();
278                     } catch (ElasticsearchException e) {
279                         if (e.status() == RestStatus.BAD_REQUEST) {
280                             // mainly for MapperParsingException and Failure to detect xcontent
281                             logger.info("ignoring recovery of a corrupt translog entry", e);
282                         } else {
283                             throw e;
284                         }
285                     }
286                 }
287             } catch (Throwable e) {
288                 // we failed to recovery, make sure to delete the translog file (and keep the recovering one)
289                 indexShard.translog().closeWithDelete();
290                 throw new IndexShardGatewayRecoveryException(shardId, "failed to recover shard", e);
291             } finally {
292                 IOUtils.closeWhileHandlingException(in);
293             }
294             // we flush to trim the translog, in case it was big.
295             try {
296                 FlushRequest flushRequest = new FlushRequest();
297                 flushRequest.force(false);
298                 flushRequest.waitIfOngoing(false);
299                 indexShard.flush(flushRequest);
300             } catch (FlushNotAllowedEngineException e) {
301                 // to be safe we catch the FNAEX , at this point no one can recover
302                 logger.debug("skipping flush at end of recovery (not allowed)", e);
303             }
304             indexShard.finalizeRecovery();
305             indexShard.postRecovery("post recovery from gateway");
306 
307             try {
308                 Files.deleteIfExists(recoveringTranslogFile.toPath());
309             } catch (Exception ex) {
310                 logger.debug("Failed to delete recovering translog file {}", ex, recoveringTranslogFile);
311             }
312         } finally {
313             indexShard.store().decRef();
314         }
315 
316         for (final String type : typesToUpdate) {
317             final CountDownLatch latch = new CountDownLatch(1);
318             mappingUpdatedAction.updateMappingOnMaster(indexService.index().name(), indexService.mapperService().documentMapper(type), indexService.indexUUID(), new MappingUpdatedAction.MappingUpdateListener() {
319                 @Override
320                 public void onMappingUpdate() {
321                     latch.countDown();
322                 }
323 
324                 @Override
325                 public void onFailure(Throwable t) {
326                     latch.countDown();
327                     logger.debug("failed to send mapping update post recovery to master for [{}]", t, type);
328                 }
329             });
330             cancellableThreads.execute(new CancellableThreads.Interruptable() {
331                 @Override
332                 public void run() throws InterruptedException {
333                     try {
334                         boolean waited = latch.await(waitForMappingUpdatePostRecovery.millis(), TimeUnit.MILLISECONDS);
335                         if (!waited) {
336                             logger.debug("waited for mapping update on master for [{}], yet timed out", type);
337                         }
338                     } catch (InterruptedException e) {
339                         logger.debug("interrupted while waiting for mapping update");
340                     }
341                 }
342             });
343 
344         }
345     }
346 
347     @Override
348     public String type() {
349         return "local";
350     }
351 
352 
353     @Override
354     public void close() {
355         FutureUtils.cancel(flushScheduler);
356         cancellableThreads.cancel("closed");
357     }
358 
359     class Sync implements Runnable {
360         @Override
361         public void run() {
362             // don't re-schedule  if its closed..., we are done
363             if (indexShard.state() == IndexShardState.CLOSED) {
364                 return;
365             }
366             if (indexShard.state() == IndexShardState.STARTED && indexShard.translog().syncNeeded()) {
367                 threadPool.executor(ThreadPool.Names.FLUSH).execute(new Runnable() {
368                     @Override
369                     public void run() {
370                         try {
371                             indexShard.translog().sync();
372                         } catch (Exception e) {
373                             if (indexShard.state() == IndexShardState.STARTED) {
374                                 logger.warn("failed to sync translog", e);
375                             }
376                         }
377                         if (indexShard.state() != IndexShardState.CLOSED) {
378                             flushScheduler = threadPool.schedule(syncInterval, ThreadPool.Names.SAME, Sync.this);
379                         }
380                     }
381                 });
382             } else {
383                 flushScheduler = threadPool.schedule(syncInterval, ThreadPool.Names.SAME, Sync.this);
384             }
385         }
386     }

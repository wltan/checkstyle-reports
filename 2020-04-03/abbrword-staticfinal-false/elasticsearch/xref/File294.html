50  public class PublishClusterStateAction extends AbstractComponent {
51  
52      public static final String ACTION_NAME = "internal:discovery/zen/publish";
53  
54      public static interface NewClusterStateListener {
55  
56          static interface NewStateProcessed {
57  
58              void onNewClusterStateProcessed();
59  
60              void onNewClusterStateFailed(Throwable t);
61          }
62  
63          void onNewClusterState(ClusterState clusterState, NewStateProcessed newStateProcessed);
64      }
65  
66      private final TransportService transportService;
67      private final DiscoveryNodesProvider nodesProvider;
68      private final NewClusterStateListener listener;
69      private final DiscoverySettings discoverySettings;
70      private final ClusterName clusterName;
71  
72      public PublishClusterStateAction(Settings settings, TransportService transportService, DiscoveryNodesProvider nodesProvider,
73                                       NewClusterStateListener listener, DiscoverySettings discoverySettings, ClusterName clusterName) {
74          super(settings);
75          this.transportService = transportService;
76          this.nodesProvider = nodesProvider;
77          this.listener = listener;
78          this.discoverySettings = discoverySettings;
79          this.clusterName = clusterName;
80          transportService.registerHandler(ACTION_NAME, new PublishClusterStateRequestHandler());
81      }
82  
83      public void close() {
84          transportService.removeHandler(ACTION_NAME);
85      }
86  
87      public void publish(ClusterState clusterState, final Discovery.AckListener ackListener) {
88          Set<DiscoveryNode> nodesToPublishTo = new HashSet<>(clusterState.nodes().size());
89          DiscoveryNode localNode = nodesProvider.nodes().localNode();
90          for (final DiscoveryNode node : clusterState.nodes()) {
91              if (node.equals(localNode)) {
92                  continue;
93              }
94              nodesToPublishTo.add(node);
95          }
96          publish(clusterState, nodesToPublishTo, new AckClusterStatePublishResponseHandler(nodesToPublishTo, ackListener));
97      }
98  
99      private void publish(final ClusterState clusterState, final Set<DiscoveryNode> nodesToPublishTo,
100                          final BlockingClusterStatePublishResponseHandler publishResponseHandler) {
101 
102         Map<Version, BytesReference> serializedStates = Maps.newHashMap();
103 
104         final AtomicBoolean timedOutWaitingForNodes = new AtomicBoolean(false);
105         final TimeValue publishTimeout = discoverySettings.getPublishTimeout();
106 
107         for (final DiscoveryNode node : nodesToPublishTo) {
108 
109             // try and serialize the cluster state once (or per version), so we don't serialize it
110             // per node when we send it over the wire, compress it while we are at it...
111             BytesReference bytes = serializedStates.get(node.version());
112             if (bytes == null) {
113                 try {
114                     BytesStreamOutput bStream = new BytesStreamOutput();
115                     StreamOutput stream = new HandlesStreamOutput(CompressorFactory.defaultCompressor().streamOutput(bStream));
116                     stream.setVersion(node.version());
117                     ClusterState.Builder.writeTo(clusterState, stream);
118                     stream.close();
119                     bytes = bStream.bytes();
120                     serializedStates.put(node.version(), bytes);
121                 } catch (Throwable e) {
122                     logger.warn("failed to serialize cluster_state before publishing it to node {}", e, node);
123                     publishResponseHandler.onFailure(node, e);
124                     continue;
125                 }
126             }
127             try {
128                 TransportRequestOptions options = TransportRequestOptions.options().withType(TransportRequestOptions.Type.STATE).withCompress(false);
129                 // no need to put a timeout on the options here, because we want the response to eventually be received
130                 // and not log an error if it arrives after the timeout
131                 transportService.sendRequest(node, ACTION_NAME,
132                         new BytesTransportRequest(bytes, node.version()),
133                         options, // no need to compress, we already compressed the bytes
134 
135                         new EmptyTransportResponseHandler(ThreadPool.Names.SAME) {
136 
137                             @Override
138                             public void handleResponse(TransportResponse.Empty response) {
139                                 if (timedOutWaitingForNodes.get()) {
140                                     logger.debug("node {} responded for cluster state [{}] (took longer than [{}])", node, clusterState.version(), publishTimeout);
141                                 }
142                                 publishResponseHandler.onResponse(node);
143                             }
144 
145                             @Override
146                             public void handleException(TransportException exp) {
147                                 logger.debug("failed to send cluster state to {}", exp, node);
148                                 publishResponseHandler.onFailure(node, exp);
149                             }
150                         });
151             } catch (Throwable t) {
152                 logger.debug("error sending cluster state to {}", t, node);
153                 publishResponseHandler.onFailure(node, t);
154             }
155         }
156 
157         if (publishTimeout.millis() > 0) {
158             // only wait if the publish timeout is configured...
159             try {
160                 timedOutWaitingForNodes.set(!publishResponseHandler.awaitAllNodes(publishTimeout));
161                 if (timedOutWaitingForNodes.get()) {
162                     DiscoveryNode[] pendingNodes = publishResponseHandler.pendingNodes();
163                     // everyone may have just responded
164                     if (pendingNodes.length > 0) {
165                         logger.warn("timed out waiting for all nodes to process published state [{}] (timeout [{}], pending nodes: {})", clusterState.version(), publishTimeout, pendingNodes);
166                     }
167                 }
168             } catch (InterruptedException e) {
169                 // ignore & restore interrupt
170                 Thread.currentThread().interrupt();
171             }
172         }
173     }
174 
175     private class PublishClusterStateRequestHandler extends BaseTransportRequestHandler<BytesTransportRequest> {
176 
177         @Override
178         public BytesTransportRequest newInstance() {
179             return new BytesTransportRequest();
180         }
181 
182         @Override
183         public void messageReceived(BytesTransportRequest request, final TransportChannel channel) throws Exception {
184             Compressor compressor = CompressorFactory.compressor(request.bytes());
185             StreamInput in;
186             if (compressor != null) {
187                 in = CachedStreamInput.cachedHandlesCompressed(compressor, request.bytes().streamInput());
188             } else {
189                 in = CachedStreamInput.cachedHandles(request.bytes().streamInput());
190             }
191             in.setVersion(request.version());
192             ClusterState clusterState = ClusterState.Builder.readFrom(in, nodesProvider.nodes().localNode(), clusterName);
193             clusterState.status(ClusterState.ClusterStateStatus.RECEIVED);
194             logger.debug("received cluster state version {}", clusterState.version());
195             try {
196                 listener.onNewClusterState(clusterState, new NewClusterStateListener.NewStateProcessed() {
197                     @Override
198                     public void onNewClusterStateProcessed() {
199                         try {
200                             channel.sendResponse(TransportResponse.Empty.INSTANCE);
201                         } catch (Throwable e) {
202                             logger.debug("failed to send response on cluster state processed", e);
203                         }
204                     }
205 
206                     @Override
207                     public void onNewClusterStateFailed(Throwable t) {
208                         try {
209                             channel.sendResponse(t);
210                         } catch (Throwable e) {
211                             logger.debug("failed to send response on cluster state processed", e);
212                         }
213                     }
214                 });
215             } catch (Exception e) {
216                 logger.warn("unexpected error while processing cluster state version [{}]", e, clusterState.version());
217                 try {
218                     channel.sendResponse(e);
219                 } catch (Throwable e1) {
220                     logger.debug("failed to send response on cluster state processed", e1);
221                 }
222             }
223         }
224 
225         @Override
226         public String executor() {
227             return ThreadPool.Names.SAME;
228         }
229     }

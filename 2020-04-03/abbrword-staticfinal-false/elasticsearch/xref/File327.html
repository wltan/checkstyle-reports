38  /**
39   * <p>
40   * A {@link PostingsFormat} useful for low doc-frequency fields such as primary
41   * keys. Bloom filters are maintained in a ".blm" file which offers "fast-fail"
42   * for reads in segments known to have no record of the key. A choice of
43   * delegate PostingsFormat is used to record all other Postings data.
44   * </p>
45   * <p>
46   * This is a special bloom filter version, based on {@link org.elasticsearch.common.util.BloomFilter} and inspired
47   * by Lucene {@link org.apache.lucene.codecs.bloom.BloomFilteringPostingsFormat}.
48   * </p>
49   * @deprecated only for reading old segments
50   */
51  @Deprecated
52  public class BloomFilterPostingsFormat extends PostingsFormat {
53  
54      public static final String BLOOM_CODEC_NAME = "XBloomFilter"; // the Lucene one is named BloomFilter
55      public static final int BLOOM_CODEC_VERSION = 1;
56      public static final int BLOOM_CODEC_VERSION_CHECKSUM = 2;
57      public static final int BLOOM_CODEC_VERSION_CURRENT = BLOOM_CODEC_VERSION_CHECKSUM;
58  
59      /**
60       * Extension of Bloom Filters file
61       */
62      static final String BLOOM_EXTENSION = "blm";
63  
64      private BloomFilter.Factory bloomFilterFactory = BloomFilter.Factory.DEFAULT;
65      private PostingsFormat delegatePostingsFormat;
66  
67      /**
68       * Creates Bloom filters for a selection of fields created in the index. This
69       * is recorded as a set of Bitsets held as a segment summary in an additional
70       * "blm" file. This PostingsFormat delegates to a choice of delegate
71       * PostingsFormat for encoding all other postings data.
72       *
73       * @param delegatePostingsFormat The PostingsFormat that records all the non-bloom filter data i.e.
74       *                               postings info.
75       * @param bloomFilterFactory     The {@link BloomFilter.Factory} responsible for sizing BloomFilters
76       *                               appropriately
77       */
78      public BloomFilterPostingsFormat(PostingsFormat delegatePostingsFormat,
79                                       BloomFilter.Factory bloomFilterFactory) {
80          super(BLOOM_CODEC_NAME);
81          this.delegatePostingsFormat = delegatePostingsFormat;
82          this.bloomFilterFactory = bloomFilterFactory;
83      }
84  
85      // Used only by core Lucene at read-time via Service Provider instantiation -
86      // do not use at Write-time in application code.
87      public BloomFilterPostingsFormat() {
88          super(BLOOM_CODEC_NAME);
89      }

92      public BloomFilteredFieldsConsumer fieldsConsumer(SegmentWriteState state) throws IOException {

94      }

96      @Override
97      public BloomFilteredFieldsProducer fieldsProducer(SegmentReadState state)
98              throws IOException {
99          return new BloomFilteredFieldsProducer(state);

102     public final class BloomFilteredFieldsProducer extends FieldsProducer {
103         private FieldsProducer delegateFieldsProducer;
104         HashMap<String, BloomFilter> bloomsByFieldName = new HashMap<>();
105         private final IndexInput data;
106 
107         // for internal use only
108         FieldsProducer getDelegate() {
109             return delegateFieldsProducer;
110         }
111 
112         public BloomFilteredFieldsProducer(SegmentReadState state)
113                 throws IOException {
114 
115             String bloomFileName = IndexFileNames.segmentFileName(
116                     state.segmentInfo.name, state.segmentSuffix, BLOOM_EXTENSION);
117             boolean success = false;
118             try {
119                 data = state.directory.openChecksumInput(bloomFileName, state.context);
120                 int version = CodecUtil.checkHeader(data, BLOOM_CODEC_NAME, BLOOM_CODEC_VERSION,
121                         BLOOM_CODEC_VERSION_CURRENT);
122                 // // Load the hash function used in the BloomFilter
123                 // hashFunction = HashFunction.forName(bloomIn.readString());
124                 // Load the delegate postings format
125                final String delegatePostings = data.readString();
126                 this.delegateFieldsProducer = PostingsFormat.forName(delegatePostings)
127                         .fieldsProducer(state);
128                success = true;
129             } finally {
130               if (!success) {
131                   IOUtils.closeWhileHandlingException(this);
132               }
133             }
134         }
135 
136         @Override
137         public Iterator<String> iterator() {
138             return delegateFieldsProducer.iterator();
139         }
140 
141         @Override
142         public void close() throws IOException {
143             IOUtils.close(data, delegateFieldsProducer);
144         }
145 
146         @Override
147         public Terms terms(String field) throws IOException {
148             BloomFilter filter = bloomsByFieldName.get(field);
149             if (filter == null) {
150                 return delegateFieldsProducer.terms(field);
151             } else {
152                 Terms result = delegateFieldsProducer.terms(field);
153                 if (result == null) {
154                     return null;
155                 }
156                 return new BloomFilteredTerms(result, filter);
157             }
158         }
159 
160         @Override
161         public int size() {
162             return delegateFieldsProducer.size();
163         }
164 
165         public long getUniqueTermCount() throws IOException {
166             return delegateFieldsProducer.getUniqueTermCount();
167         }
168 
169         @Override
170         public long ramBytesUsed() {
171             long size = delegateFieldsProducer.ramBytesUsed();
172             for (BloomFilter bloomFilter : bloomsByFieldName.values()) {
173                 size += bloomFilter.getSizeInBytes();
174             }
175             return size;
176         }
177 
178         @Override
179         public void checkIntegrity() throws IOException {
180             delegateFieldsProducer.checkIntegrity();
181         }
182     }

184     public static final class BloomFilteredTerms extends FilterAtomicReader.FilterTerms {
185         private BloomFilter filter;
186 
187         public BloomFilteredTerms(Terms terms, BloomFilter filter) {
188             super(terms);
189             this.filter = filter;
190         }
191 
192         public BloomFilter getFilter() {
193             return filter;
194         }
195 
196         @Override
197         public TermsEnum iterator(TermsEnum reuse) throws IOException {
198             TermsEnum result;
199             if ((reuse != null) && (reuse instanceof BloomFilteredTermsEnum)) {
200                 // recycle the existing BloomFilteredTermsEnum by asking the delegate
201                 // to recycle its contained TermsEnum
202                 BloomFilteredTermsEnum bfte = (BloomFilteredTermsEnum) reuse;
203                 if (bfte.filter == filter) {
204                     bfte.reset(this.in);
205                     return bfte;
206                 }
207                 reuse = bfte.reuse;
208             }
209             // We have been handed something we cannot reuse (either null, wrong
210             // class or wrong filter) so allocate a new object
211             result = new BloomFilteredTermsEnum(this.in, reuse, filter);
212             return result;
213         }
214     }

216     static final class BloomFilteredTermsEnum extends TermsEnum {
217 
218         private Terms delegateTerms;
219         private TermsEnum delegateTermsEnum;
220         private TermsEnum reuse;
221         private BloomFilter filter;
222 
223         public BloomFilteredTermsEnum(Terms other, TermsEnum reuse, BloomFilter filter) {
224             this.delegateTerms = other;
225             this.reuse = reuse;
226             this.filter = filter;
227         }
228 
229         void reset(Terms others) {
230             reuse = this.delegateTermsEnum;
231             this.delegateTermsEnum = null;
232             this.delegateTerms = others;
233         }
234 
235         private TermsEnum getDelegate() throws IOException {
236             if (delegateTermsEnum == null) {
237                 /* pull the iterator only if we really need it -
238                  * this can be a relatively heavy operation depending on the 
239                  * delegate postings format and they underlying directory
240                  * (clone IndexInput) */
241                 delegateTermsEnum = delegateTerms.iterator(reuse);
242             }
243             return delegateTermsEnum;
244         }
245 
246         @Override
247         public final BytesRef next() throws IOException {
248             return getDelegate().next();
249         }
250 
251         @Override
252         public final Comparator<BytesRef> getComparator() {
253             return delegateTerms.getComparator();
254         }
255 
256         @Override
257         public final boolean seekExact(BytesRef text)
258                 throws IOException {
259             // The magical fail-fast speed up that is the entire point of all of
260             // this code - save a disk seek if there is a match on an in-memory
261             // structure
262             // that may occasionally give a false positive but guaranteed no false
263             // negatives
264             if (!filter.mightContain(text)) {
265                 return false;
266             }
267             return getDelegate().seekExact(text);
268         }
269 
270         @Override
271         public final SeekStatus seekCeil(BytesRef text)
272                 throws IOException {
273             return getDelegate().seekCeil(text);
274         }
275 
276         @Override
277         public final void seekExact(long ord) throws IOException {
278             getDelegate().seekExact(ord);
279         }
280 
281         @Override
282         public final BytesRef term() throws IOException {
283             return getDelegate().term();
284         }
285 
286         @Override
287         public final long ord() throws IOException {
288             return getDelegate().ord();
289         }
290 
291         @Override
292         public final int docFreq() throws IOException {
293             return getDelegate().docFreq();
294         }
295 
296         @Override
297         public final long totalTermFreq() throws IOException {
298             return getDelegate().totalTermFreq();
299         }
300 
301 
302         @Override
303         public DocsAndPositionsEnum docsAndPositions(Bits liveDocs,
304                                                      DocsAndPositionsEnum reuse, int flags) throws IOException {
305             return getDelegate().docsAndPositions(liveDocs, reuse, flags);
306         }
307 
308         @Override
309         public DocsEnum docs(Bits liveDocs, DocsEnum reuse, int flags)
310                 throws IOException {
311             return getDelegate().docs(liveDocs, reuse, flags);
312         }

317     // TODO: would be great to move this out to test code, but the interaction between es090 and bloom is complex
318     // at least it is not accessible via SPI
319     public final class BloomFilteredFieldsConsumer extends FieldsConsumer {
320         private FieldsConsumer delegateFieldsConsumer;
321         private Map<FieldInfo, BloomFilter> bloomFilters = new HashMap<>();
322         private SegmentWriteState state;
323 
324         // private PostingsFormat delegatePostingsFormat;
325 
326         public BloomFilteredFieldsConsumer(FieldsConsumer fieldsConsumer,
327                                            SegmentWriteState state, PostingsFormat delegatePostingsFormat) {
328             this.delegateFieldsConsumer = fieldsConsumer;
329             // this.delegatePostingsFormat=delegatePostingsFormat;
330             this.state = state;
331         }
332 
333         // for internal use only
334         public FieldsConsumer getDelegate() {
335             return delegateFieldsConsumer;
336         }
337 
338         @Override
339         public TermsConsumer addField(FieldInfo field) throws IOException {
340             BloomFilter bloomFilter = bloomFilterFactory.createFilter(state.segmentInfo.getDocCount());
341             if (bloomFilter != null) {
342                 assert bloomFilters.containsKey(field) == false;
343                 bloomFilters.put(field, bloomFilter);
344                 return new WrappedTermsConsumer(delegateFieldsConsumer.addField(field), bloomFilter);
345             } else {
346                 // No, use the unfiltered fieldsConsumer - we are not interested in
347                 // recording any term Bitsets.
348                 return delegateFieldsConsumer.addField(field);
349             }
350         }
351 
352         @Override
353         public void close() throws IOException {
354             delegateFieldsConsumer.close();
355             // Now we are done accumulating values for these fields
356             List<Entry<FieldInfo, BloomFilter>> nonSaturatedBlooms = new ArrayList<>();
357 
358             for (Entry<FieldInfo, BloomFilter> entry : bloomFilters.entrySet()) {
359                 nonSaturatedBlooms.add(entry);
360             }
361             String bloomFileName = IndexFileNames.segmentFileName(
362                     state.segmentInfo.name, state.segmentSuffix, BLOOM_EXTENSION);
363             IndexOutput bloomOutput = null;
364             try {
365                 bloomOutput = state.directory
366                         .createOutput(bloomFileName, state.context);
367                 CodecUtil.writeHeader(bloomOutput, BLOOM_CODEC_NAME,
368                         BLOOM_CODEC_VERSION_CURRENT);
369                 // remember the name of the postings format we will delegate to
370                 bloomOutput.writeString(delegatePostingsFormat.getName());
371 
372                 // First field in the output file is the number of fields+blooms saved
373                 bloomOutput.writeInt(nonSaturatedBlooms.size());
374                 for (Entry<FieldInfo, BloomFilter> entry : nonSaturatedBlooms) {
375                     FieldInfo fieldInfo = entry.getKey();
376                     BloomFilter bloomFilter = entry.getValue();
377                     bloomOutput.writeInt(fieldInfo.number);
378                     saveAppropriatelySizedBloomFilter(bloomOutput, bloomFilter, fieldInfo);
379                 }
380                 CodecUtil.writeFooter(bloomOutput);
381             } finally {
382                 IOUtils.close(bloomOutput);
383             }
384             //We are done with large bitsets so no need to keep them hanging around
385             bloomFilters.clear();
386         }
387 
388         private void saveAppropriatelySizedBloomFilter(IndexOutput bloomOutput,
389                                                        BloomFilter bloomFilter, FieldInfo fieldInfo) throws IOException {
390 
391 //            FuzzySet rightSizedSet = bloomFilterFactory.downsize(fieldInfo,
392 //                    bloomFilter);
393 //            if (rightSizedSet == null) {
394 //                rightSizedSet = bloomFilter;
395 //            }
396 //            rightSizedSet.serialize(bloomOutput);
397             BloomFilter.serilaize(bloomFilter, bloomOutput);
398         }

402     class WrappedTermsConsumer extends TermsConsumer {
403         private TermsConsumer delegateTermsConsumer;
404         private BloomFilter bloomFilter;
405 
406         public WrappedTermsConsumer(TermsConsumer termsConsumer, BloomFilter bloomFilter) {
407             this.delegateTermsConsumer = termsConsumer;
408             this.bloomFilter = bloomFilter;
409         }
410 
411         @Override
412         public PostingsConsumer startTerm(BytesRef text) throws IOException {
413             return delegateTermsConsumer.startTerm(text);
414         }
415 
416         @Override
417         public void finishTerm(BytesRef text, TermStats stats) throws IOException {
418 
419             // Record this term in our BloomFilter
420             if (stats.docFreq > 0) {
421                 bloomFilter.put(text);
422             }
423             delegateTermsConsumer.finishTerm(text, stats);
424         }
425 
426         @Override
427         public void finish(long sumTotalTermFreq, long sumDocFreq, int docCount)
428                 throws IOException {
429             delegateTermsConsumer.finish(sumTotalTermFreq, sumDocFreq, docCount);
430         }
431 
432         @Override
433         public Comparator<BytesRef> getComparator() throws IOException {
434             return delegateTermsConsumer.getComparator();
435         }
436 
437     }

440         return this.delegatePostingsFormat;

1   package org.infinispan.client.hotrod.impl.async;
2   
3   import org.infinispan.client.hotrod.impl.ConfigurationProperties;
4   import org.infinispan.commons.executors.ExecutorFactory;
5   
6   import java.util.Properties;
7   import java.util.concurrent.ExecutorService;
8   import java.util.concurrent.LinkedBlockingQueue;
9   import java.util.concurrent.ThreadFactory;
10  import java.util.concurrent.ThreadPoolExecutor;
11  import java.util.concurrent.TimeUnit;
12  import java.util.concurrent.atomic.AtomicInteger;
13  
14  /**
15   * Default implementation for {@link org.infinispan.executors.ExecutorFactory} based on an {@link ThreadPoolExecutor}.
16   *
17   * @author Mircea.Markus@jboss.com
18   * @since 4.1
19   */
20  public class DefaultAsyncExecutorFactory implements ExecutorFactory {
21     public static final String THREAD_NAME = "HotRod-client-async-pool";
22     public static final AtomicInteger counter = new AtomicInteger(0);
23  
24     @Override
25     public ExecutorService getExecutor(Properties p) {
26        ConfigurationProperties cp = new ConfigurationProperties(p);
27        ThreadFactory tf = new ThreadFactory() {
28           @Override
29           public Thread newThread(Runnable r) {
30              Thread th = new Thread(r, THREAD_NAME + "-" + counter.getAndIncrement());
31              th.setDaemon(true);
32              return th;
33           }
34        };
35  
36        return new ThreadPoolExecutor(cp.getDefaultExecutorFactoryPoolSize(), cp.getDefaultExecutorFactoryPoolSize(),
37                                      0L, TimeUnit.MILLISECONDS,
38                                      new LinkedBlockingQueue<Runnable>(cp.getDefaultExecutorFactoryQueueSize()),
39                                      tf);
40     }
41  }

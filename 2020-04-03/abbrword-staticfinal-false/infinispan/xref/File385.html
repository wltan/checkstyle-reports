1   package org.infinispan.security;
2   
3   import static org.testng.AssertJUnit.assertEquals;
4   
5   import java.security.PrivilegedAction;
6   import java.security.PrivilegedExceptionAction;
7   
8   import javax.security.auth.Subject;
9   
10  import org.infinispan.Cache;
11  import org.infinispan.configuration.cache.CacheMode;
12  import org.infinispan.configuration.cache.ConfigurationBuilder;
13  import org.infinispan.configuration.global.GlobalConfigurationBuilder;
14  import org.infinispan.security.impl.IdentityRoleMapper;
15  import org.infinispan.test.MultipleCacheManagersTest;
16  import org.infinispan.test.TestingUtil;
17  import org.testng.annotations.AfterClass;
18  import org.testng.annotations.AfterMethod;
19  import org.testng.annotations.Test;
20  
21  @Test(groups = "functional", testName = "security.ClusteredSecureCacheTest")
22  public class ClusteredSecureCacheTest extends MultipleCacheManagersTest {
23     final static Subject ADMIN = TestingUtil.makeSubject("admin");
24  
25     @Override
26     protected void createCacheManagers() throws Throwable {
27        final GlobalConfigurationBuilder global = GlobalConfigurationBuilder.defaultClusteredBuilder();
28        final ConfigurationBuilder builder = getDefaultClusteredCacheConfig(CacheMode.REPL_SYNC);
29        global.security().authorization().enable()
30              .principalRoleMapper(new IdentityRoleMapper()).role("admin").permission(AuthorizationPermission.ALL);
31        builder.security().authorization().enable().role("admin");
32        Security.doAs(ADMIN, new PrivilegedExceptionAction<Void>() {
33           @Override
34           public Void run() throws Exception {
35              createCluster(global, builder, 2);
36              waitForClusterToForm();
37              return null;
38           }
39        });
40     }
41  
42     @Override
43     @AfterClass(alwaysRun = true)
44     protected void destroy() {
45        Security.doAs(ADMIN, new PrivilegedAction<Void>() {
46           @Override
47           public Void run() {
48              ClusteredSecureCacheTest.super.destroy();
49              return null;
50           }
51        });
52     }
53  
54     @Override
55     @AfterMethod(alwaysRun = true)
56     protected void clearContent() throws Throwable {
57        Security.doAs(ADMIN, new PrivilegedExceptionAction<Void>() {
58           @Override
59           public Void run() throws Exception {
60              try {
61                 ClusteredSecureCacheTest.super.clearContent();
62              } catch (Throwable e) {
63                 throw new Exception(e);
64              }
65              return null;
66           }
67        });
68     }
69  
70     public void testClusteredSecureCache() {
71        Security.doAs(ADMIN, new PrivilegedAction<Void>() {
72  
73           @Override
74           public Void run() {
75              Cache<String, String> cache1 = cache(0);
76              Cache<String, String> cache2 = cache(1);
77              cache1.put("a", "a");
78              cache2.put("b", "b");
79              assertEquals("a", cache2.get("a"));
80              assertEquals("b", cache1.get("b"));
81              return null;
82           }
83        });
84     }
85  }

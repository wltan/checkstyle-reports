47  /**
48   * Infinispan's implementation of an {@link ExecutorService} and {@link DistributedExecutorService}.
49   * This ExecutorService provides methods to submit tasks for an execution on a cluster of Infinispan
50   * nodes.
51   * <p>
52   *
53   *
54   * Note that due to potential task migration to another nodes every {@link Callable},
55   * {@link Runnable} and/or {@link DistributedCallable} submitted must be either {@link Serializable}
56   * or {@link Externalizable}. Also the value returned from a callable must be {@link Serializable}
57   * or {@link Externalizable}. Unfortunately if the value returned is not serializable then a
58   * {@link NotSerializableException} will be thrown.
59   *
60   * @author Vladimir Blagojevic
61   * @since 5.0
62   *
63   */
64  public class DefaultExecutorService extends AbstractExecutorService implements DistributedExecutorService {

144    /**
145     * Creates a new DefaultExecutorService given a master cache node and an ExecutorService for
146     * parallel execution of task ran on this node. All distributed task executions will be initiated
147     * from this Infinispan cache node.
148     *
149     * @param masterCacheNode
150     *           Cache node initiating distributed task
151     * @param localExecutorService
152     *           ExecutorService to run local tasks
153     * @param takeExecutorOwnership
154     *           if true {@link DistributedExecutorService#shutdown()} and
155     *           {@link DistributedExecutorService#shutdownNow()} method will shutdown
156     *           localExecutorService as well
157     *
158     */
159    public DefaultExecutorService(Cache<?, ?> masterCacheNode, ExecutorService localExecutorService,
160             boolean takeExecutorOwnership) {
161       super();
162       if (masterCacheNode == null)
163          throw new IllegalArgumentException("Can not use null cache for DefaultExecutorService");
164       else if (localExecutorService == null)
165          throw new IllegalArgumentException("Can not use null instance of ExecutorService");
166       else {
167          try {
168             if (localExecutorService.isShutdown())
169                throw new IllegalArgumentException("Can not use an instance of ExecutorService which is shutdown");
170          } catch (IllegalStateException e) {
171             if (takeExecutorOwnership) {
172                throw new IllegalArgumentException("Can not take ownership of a ManagedExecutorService");
173             }
174          }
175       }
176 
177       ensureAccessPermissions(masterCacheNode.getAdvancedCache());
178       ensureProperCacheState(masterCacheNode.getAdvancedCache());
179 
180       this.cache = masterCacheNode.getAdvancedCache();
181       ComponentRegistry registry = SecurityActions.getCacheComponentRegistry(cache);
182 
183       this.rpc = SecurityActions.getCacheRpcManager(cache);
184       this.invoker = registry.getComponent(InterceptorChain.class);
185       this.factory = registry.getComponent(CommandsFactory.class);
186       this.marshaller = registry.getComponent(StreamingMarshaller.class, CACHE_MARSHALLER);
187       this.cancellationService = registry.getComponent(CancellationService.class);
188       this.localExecutorService = localExecutorService;
189       this.takeExecutorOwnership = takeExecutorOwnership;
190       this.timeService = registry.getTimeService();
191       this.clusterDependentLogic = registry.getComponent(ClusteringDependentLogic.class);
192    }

289    /**
290     * the main mechanics of invokeAny. This was essentially copied from
291     * {@link AbstractExecutorService} doInvokeAny except that we replaced the
292     * {@link ExecutorCompletionService} with our {@link DistributedExecutionCompletionService}.
293     */
294    private <T> T doInvokeAny(Collection<? extends Callable<T>> tasks, boolean timed, long nanos)
295             throws InterruptedException, ExecutionException, TimeoutException {
296       if (tasks == null)
297          throw new NullPointerException();
298       int ntasks = tasks.size();
299       if (ntasks == 0)
300          throw new IllegalArgumentException();
301       List<Future<T>> futures = new ArrayList<Future<T>>(ntasks);
302       CompletionService<T> ecs = new DistributedExecutionCompletionService<T>(this);
303 
304       // For efficiency, especially in executors with limited
305       // parallelism, check to see if previously submitted tasks are
306       // done before submitting more of them. This interleaving
307       // plus the exception mechanics account for messiness of main
308       // loop.
309 
310       try {
311          // Record exceptions so that if we fail to obtain any
312          // result, we can throw the last exception we got.
313          ExecutionException ee = null;
314          long lastTime = (timed) ? timeService.time() : 0;
315          Iterator<? extends Callable<T>> it = tasks.iterator();
316 
317          // Start one task for sure; the rest incrementally
318          futures.add(ecs.submit(it.next()));
319          --ntasks;
320          int active = 1;
321 
322          for (;;) {
323             Future<T> f = ecs.poll();
324             if (f == null) {
325                if (ntasks > 0) {
326                   --ntasks;
327                   futures.add(ecs.submit(it.next()));
328                   ++active;
329                } else if (active == 0)
330                   break;
331                else if (timed) {
332                   f = ecs.poll(nanos, TimeUnit.NANOSECONDS);
333                   if (f == null)
334                      throw new TimeoutException();
335                   long now = timeService.time();
336                   nanos -= timeService.timeDuration(lastTime, now, TimeUnit.NANOSECONDS);
337                   lastTime = now;
338                } else
339                   f = ecs.take();
340             }
341             if (f != null) {
342                --active;
343                try {
344                   return f.get();
345                } catch (InterruptedException ie) {
346                   throw ie;
347                } catch (ExecutionException eex) {
348                   ee = eex;
349                } catch (RuntimeException rex) {
350                   ee = new ExecutionException(rex);
351                }
352             }
353          }
354 
355          if (ee == null)
356             ee = new ExecutionException() {
357                private static final long serialVersionUID = 200818694545553992L;
358             };
359          throw ee;
360 
361       } finally {
362          for (Future<T> f : futures)
363             f.cancel(true);
364       }
365    }

367    @Override
368    public void execute(Runnable command) {
369       if (!isShutdown.get()) {
370          DistributedTaskPart<Object> cmd;
371          if (command instanceof DistributedTaskPart<?>) {
372             cmd = (DistributedTaskPart<Object>) command;
373          } else if (command instanceof Serializable) {
374             cmd = (DistributedTaskPart<Object>) newTaskFor(command, null);
375          } else {
376             throw new IllegalArgumentException("Runnable command is not Serializable  " + command);
377          }
378          cmd.execute();
379       } else {
380          throw new RejectedExecutionException();
381       }
382    }

409    public <T> NotifyingFuture<T> submit(Address target, DistributedTask<T> task) {
410       if (task == null)
411          throw new NullPointerException();
412       if (target == null)
413          throw new NullPointerException();
414       List<Address> members = getMembers();
415       if (!members.contains(target)) {
416          return new NoOpFuture<>(new SuspectException("Target node " + target
417                   + " is not a cluster member, members are " + members));
418       }
419       Address me = getAddress();
420       DistributedExecuteCommand<T> c = null;
421       if (target.equals(me)) {
422          c = factory.buildDistributedExecuteCommand(clone(task.getCallable()), me, null);
423       } else {
424          c = factory.buildDistributedExecuteCommand(task.getCallable(), me, null);
425       }
426       DistributedTaskPart<T> part = createDistributedTaskPart(task, c, target, 0);
427       part.execute();
428       return part;
429    }

438    @Override
439    public <T, K> NotifyingFuture<T> submit(DistributedTask<T> task, K... input) {
440       if (task == null) throw new NullPointerException();
441 
442       if(inputKeysSpecified(input)){
443          Map<Address, List<K>> nodesKeysMap = keysToExecutionNodes(task.getTaskExecutionPolicy(), input);
444          checkExecutionPolicy(task, nodesKeysMap, input);
445          Address me = getAddress();
446          DistributedExecuteCommand<T> c = factory.buildDistributedExecuteCommand(task.getCallable(), me, Arrays.asList(input));
447          ArrayList<Address> nodes = new ArrayList<Address>(nodesKeysMap.keySet());
448          DistributedTaskPart<T> part = createDistributedTaskPart(task, c, Arrays.asList(input), selectExecutionNode(nodes), 0);
449          part.execute();
450          return part;
451       } else {
452          return submit(task.getCallable());
453       }
454    }

477          DistributedTaskPart<T> part = createDistributedTaskPart(task, c, target, 0);
478          futures.add(part);

491    @Override
492    public <T, K> List<Future<T>> submitEverywhere(DistributedTask<T> task, K... input) {
493       if (task == null) throw new NullPointerException();
494       if(inputKeysSpecified(input)) {
495          List<Future<T>> futures = new ArrayList<Future<T>>(input.length * 2);
496          Address me = getAddress();
497          Map<Address, List<K>> nodesKeysMap = keysToExecutionNodes(task.getTaskExecutionPolicy(), input);
498          checkExecutionPolicy(task, nodesKeysMap, input);
499          for (Entry<Address, List<K>> e : nodesKeysMap.entrySet()) {
500             Address target = e.getKey();
501             DistributedExecuteCommand<T> c = null;
502             if (target.equals(me)) {
503                c = factory.buildDistributedExecuteCommand(clone(task.getCallable()), me, e.getValue());
504             } else {
505                c = factory.buildDistributedExecuteCommand(task.getCallable(), me, e.getValue());
506             }
507             DistributedTaskPart<T> part = createDistributedTaskPart(task, c, e.getValue(), target, 0);
508             futures.add(part);
509             part.execute();
510          }
511          return futures;
512       } else {
513          return submitEverywhere(task);
514       }
515    }

558    protected List<Address> randomClusterMembers(final List<Address> members, int numNeeded) {
559       if(members == null || members.isEmpty())
560          throw new IllegalArgumentException("Invalid member list " + members);
561 
562       if (members.size() < numNeeded) {
563          log.cannotSelectRandomMembers(numNeeded, members);
564          numNeeded = members.size();
565       }
566       List<Address> membersCopy = new ArrayList<Address>(members);
567       List<Address> chosen = new ArrayList<Address>(numNeeded);
568       Random r = new Random();
569       while (!membersCopy.isEmpty() && numNeeded >= chosen.size()) {
570          int count = membersCopy.size();
571          Address address = membersCopy.remove(r.nextInt(count));
572          chosen.add(address);
573       }
574       return chosen;
575    }

577    protected <K> Map<Address, List<K>> keysToExecutionNodes(DistributedTaskExecutionPolicy policy, K... input) {
578       DistributionManager dm = cache.getDistributionManager();
579       Map<Address, List<K>> addressToKey = new HashMap<Address, List<K>>(input.length * 2);
580       boolean usingREPLMode = dm == null;
581       for (K key : input) {
582          Address ownerOfKey = null;
583          if (usingREPLMode) {
584             List<Address> members = new ArrayList<Address>(getMembers());
585             members =  filterMembers(policy, members);
586             // using REPL mode https://issues.jboss.org/browse/ISPN-1886
587             // since keys and values are on all nodes, lets just pick randomly
588             Collections.shuffle(members);
589             ownerOfKey = members.get(0);
590          } else {
591             // DIST mode
592             List<Address> owners = dm.locate(key);
593             List<Address> filtered = filterMembers(policy, owners);
594             if(!filtered.isEmpty()){
595                ownerOfKey = filtered.get(0);
596             } else {
597                ownerOfKey = owners.get(0);
598             }
599          }
600          List<K> keysAtNode = addressToKey.get(ownerOfKey);
601          if (keysAtNode == null) {
602             keysAtNode = new LinkedList<K>();
603             addressToKey.put(ownerOfKey, keysAtNode);
604          }
605          keysAtNode.add(key);
606       }
607       return addressToKey;
608    }

610    private List<Address> filterMembers(DistributedTaskExecutionPolicy policy, List<Address> members) {
611       NodeFilter filter = null;
612       switch (policy) {
613          case SAME_MACHINE:
614             filter = SAME_MACHINE_FILTER;
615             break;
616          case SAME_SITE:
617             filter = SAME_SITE_FILTER;
618             break;
619          case SAME_RACK:
620             filter = SAME_RACK_FILTER;
621             break;
622          case ALL:
623             filter = ALL_FILTER;
624             break;
625          default:
626             filter = ALL_FILTER;
627             break;
628       }
629       List<Address> result = new ArrayList<Address>();
630       for (Address address : members) {
631          if(address instanceof TopologyAwareAddress){
632             TopologyAwareAddress taa = (TopologyAwareAddress)address;
633             if(filter.include(taa, (TopologyAwareAddress)getAddress())){
634                result.add(address);
635             }
636          } else {
637             result.add(address);
638          }
639       }
640       return result;
641    }

659    private static class RandomNodeTaskFailoverPolicy implements DistributedTaskFailoverPolicy {
660 
661       public RandomNodeTaskFailoverPolicy() {
662          super();
663       }
664 
665       @Override
666       public Address failover(FailoverContext fc) {
667          return randomNode(fc.executionCandidates(),fc.executionFailureLocation());
668       }
669 
670       protected Address randomNode(List<Address> candidates, Address failedExecutionLocation){
671          Random r = new Random();
672          candidates.remove(failedExecutionLocation);
673          if (candidates.isEmpty())
674             throw new IllegalStateException("There are no candidates for failover: " + candidates);
675          int tIndex = r.nextInt(candidates.size());
676          return candidates.get(tIndex);
677       }
678 
679       @Override
680       public int maxFailoverAttempts() {
681          return 1;
682       }
683    }

784    /**
785     * DistributedTaskPart represents a unit of work sent to remote VM and executed there
786     *
787     *
788     * @author Mircea Markus
789     * @author Vladimir Blagojevic
790     */
791    private abstract class DistributedTaskPart<V> implements NotifyingFuture<V>, RunnableFuture<V> {
792 
793       protected final DistributedExecuteCommand<V> distCommand;
794       private final List<Object> inputKeys;
795       private final DistributedTask<V> owningTask;
796       private int failedOverCount;
797       private volatile boolean cancelled;
798 
799       protected DistributedTaskPart(List<Object> inputKeys, DistributedExecuteCommand<V> command, DistributedTask<V> task, int failedOverCount) {
800          this.inputKeys = inputKeys;
801          this.distCommand = command;
802          this.owningTask = task;
803          this.failedOverCount = failedOverCount;
804       }
805 
806       public List<Object> getInputKeys() {
807          return inputKeys;
808       }
809 
810       public DistributedExecuteCommand<V> getCommand() {
811          return distCommand;
812       }
813 
814       public DistributedTask<V> getOwningTask() {
815          return owningTask;
816       }
817 
818       public abstract Address getExecutionTarget();
819 
820       private DefaultExecutorService getOuterType() {
821          return DefaultExecutorService.this;
822       }
823 
824       public abstract void execute();
825 
826       @Override
827       public void run() {
828          //intentionally empty
829       }
830 
831       @Override
832       public boolean isCancelled() {
833          return cancelled;
834       }
835 
836       @Override
837       public V get() throws InterruptedException, ExecutionException {
838          try {
839             return innerGet(0, TimeUnit.MILLISECONDS);
840          } catch (TimeoutException e) {
841             throw new ExecutionException(e);
842          }
843       }
844 
845       @Override
846       public V get(long timeout, TimeUnit unit)
847             throws InterruptedException, ExecutionException, TimeoutException {
848          return innerGet(timeout, unit);
849       }
850 
851       protected V innerGet(long timeout, TimeUnit unit)
852             throws ExecutionException, TimeoutException, InterruptedException {
853          if (isCancelled())
854             throw new CancellationException("Task already cancelled");
855 
856          long timeoutNanos = computeTimeoutNanos(timeout, unit);
857          long endNanos = timeService.expectedEndTime(timeoutNanos, TimeUnit.NANOSECONDS);
858          try {
859             return getResult(timeoutNanos);
860          } catch (TimeoutException te) {
861             throw te;
862          } catch (Exception e) {
863             // The RPC could have finished with a org.infinispan.util.concurrent.TimeoutException right before
864             // the Future.get timeout expired. If that's the case, we want to throw a TimeoutException.
865             long remainingNanos = timeoutNanos > 0 ? timeService.remainingTime(endNanos, TimeUnit.NANOSECONDS) : timeoutNanos;
866             if (timeoutNanos > 0 && remainingNanos <= 0) {
867                if (trace) log.tracef("Distributed task timed out, throwing a TimeoutException and ignoring exception", e);
868                throw new TimeoutException();
869             }
870             boolean canFailover = failedOverCount++ < getOwningTask().getTaskFailoverPolicy().maxFailoverAttempts();
871             if (canFailover) {
872                try {
873                   return failoverExecution(e, timeoutNanos, TimeUnit.NANOSECONDS);
874                } catch (Exception failedOver) {
875                   throw wrapIntoExecutionException(failedOver);
876                }
877             } else {
878                throw wrapIntoExecutionException(e);
879             }
880          }
881       }
882 
883       protected abstract V getResult(long timeoutNanos) throws Exception;
884 
885       protected long computeTimeoutNanos(long timeout, TimeUnit unit) {
886          long taskTimeout = TimeUnit.MILLISECONDS.toNanos(getOwningTask().timeout());
887          long futureTimeout = TimeUnit.NANOSECONDS.convert(timeout, unit);
888          long actualTimeout;
889          if (taskTimeout > 0 && futureTimeout > 0) {
890             actualTimeout = Math.min(taskTimeout, futureTimeout);
891          } else {
892             actualTimeout = Math.max(taskTimeout, futureTimeout);
893          }
894          return actualTimeout;
895       }
896 
897       protected ExecutionException wrapIntoExecutionException(Exception e){
898          if (e instanceof ExecutionException) {
899             return (ExecutionException) e;
900          } else {
901             return new ExecutionException(e);
902          }
903       }
904 
905       protected V failoverExecution(final Exception cause, long timeout, TimeUnit unit)
906             throws Exception {
907          final List<Address> executionCandidates = executionCandidates(getOwningTask());
908          FailoverContext fc = new FailoverContext() {
909             @Override
910             public <K> List<K> inputKeys() {
911                return (List<K>) getInputKeys();
912             }
913 
914             @Override
915             public Address executionFailureLocation() {
916                return getExecutionTarget();
917             }
918 
919             @Override
920             public List<Address> executionCandidates() {
921                return executionCandidates;
922             }
923 
924             @Override
925             public Throwable cause() {
926                return cause;
927             }
928          };
929 
930          Address failoverTarget = getOwningTask().getTaskFailoverPolicy().failover(fc);
931          log.distributedTaskFailover(fc.executionFailureLocation(), failoverTarget, cause);
932          DistributedTaskPart<V> part = createDistributedTaskPart(owningTask, distCommand,
933                getInputKeys(), failoverTarget, failedOverCount);
934          part.execute();
935          return part.get(timeout, unit);
936       }
937 
938       protected void setCancelled() {
939          cancelled = true;
940       }
941    }

943    private class RemoteDistributedTaskPart<V> extends DistributedTaskPart<V> {
944       private final Address executionTarget;
945       private final NotifyingFutureImpl<Object> future = new NotifyingFutureImpl<Object>();
946 
947       public RemoteDistributedTaskPart(DistributedTask<V> task, DistributedExecuteCommand<V> command,
948                                  List<Object> inputKeys, Address executionTarget, int failoverCount) {
949          super(inputKeys, command, task, failoverCount);
950          if (getAddress().equals(executionTarget)) {
951             throw new IllegalArgumentException("This task should be executed as local.");
952          }
953          this.executionTarget = executionTarget;
954       }
955 
956       @Override
957       public Address getExecutionTarget() {
958          return executionTarget;
959       }
960 
961       @Override
962       public void execute() {
963          if (trace) log.tracef("Sending %s to remote execution at node %s", this, getExecutionTarget());
964          try {
965             rpc.invokeRemotelyInFuture(Collections.singletonList(getExecutionTarget()), getCommand(),
966                   rpc.getRpcOptionsBuilder(ResponseMode.SYNCHRONOUS)
967                         .timeout(getOwningTask().timeout(), TimeUnit.MILLISECONDS).build(), future);
968          } catch (Throwable e) {
969             log.remoteExecutionFailed(getExecutionTarget(), e);
970          }
971       }
972 
973       @Override
974       public boolean cancel(boolean mayInterruptIfRunning) {
975          if (!isCancelled()) {
976             CancelCommand ccc = factory.buildCancelCommandCommand(distCommand.getUUID());
977             rpc.invokeRemotely(Collections.singletonList(getExecutionTarget()), ccc, rpc.getDefaultRpcOptions(true));
978             setCancelled();
979             return future.cancel(true);
980          } else {
981             //already cancelled
982             return false;
983          }
984       }
985 
986       @Override
987       public boolean isDone() {
988          return future.isDone();
989       }
990 
991       @Override
992       protected V getResult(long timeoutNanos) throws Exception {
993          if (timeoutNanos > 0) {
994             return retrieveResult(future.get(timeoutNanos, TimeUnit.NANOSECONDS));
995          } else {
996             return retrieveResult(future.get());
997          }
998       }
999 
1000       private V retrieveResult(Object response) throws Exception {
1001          V result = null;
1002          //this is application level Exception that was raised in execution
1003          //simply rethrow it (might be good candidate for failover)
1004          if (response instanceof Exception) {
1005             throw ((Exception) response);
1006          }
1007          //these two should never happen, mark them with IllegalStateException
1008          if (response == null || !(response instanceof Map<?, ?>)) {
1009             throw new IllegalStateException("Invalid response received " + response);
1010          }
1011          Map<Address, Response> mapResult = (Map<Address, Response>) response;
1012          if (mapResult.size() == 1) {
1013             for (Entry<Address, Response> e : mapResult.entrySet()) {
1014                Response value = e.getValue();
1015                if (value instanceof SuccessfulResponse) {
1016                   result = (V) ((SuccessfulResponse) value).getResponseValue();
1017                } else {
1018                   throw new ExecutionException(new Exception(value != null ? value.toString() : "Unknown cause"));
1019                }
1020             }
1021          } else {
1022             //should never happen as we send DistributedTaskPart to one node for
1023             //execution only, therefore we should get only one response
1024             throw new IllegalStateException("Invalid response " + response);
1025          }
1026          return result;
1027       }
1028 
1029       @Override
1030       public NotifyingFuture<V> attachListener(final FutureListener<V> listener) {
1031          future.attachListener(new FutureListener<Object>() {
1032                @Override
1033                public void futureDone(Future<Object> future) {
1034                   listener.futureDone(RemoteDistributedTaskPart.this);
1035                }
1036             });
1037          return this;
1038       }
1039    }

1041    private class LocalDistributedTaskPart<V> extends DistributedTaskPart<V> {
1042       private final NotifyingFutureImpl<V> future = new NotifyingFutureImpl<V>();
1043 
1044       public LocalDistributedTaskPart(DistributedTask<V> task, DistributedExecuteCommand<V> command,
1045                List<Object> inputKeys, int failoverCount) {
1046          super(inputKeys, command, task, failoverCount);
1047       }
1048 
1049       @Override
1050       public boolean isDone() {
1051          return future.isDone();
1052       }
1053 
1054       @Override
1055       public synchronized boolean cancel(boolean mayInterruptIfRunning) {
1056          if (!isCancelled()) {
1057             CancelCommand ccc = factory.buildCancelCommandCommand(distCommand.getUUID());
1058             ccc.init(cancellationService);
1059             try {
1060                ccc.perform(null);
1061             } catch (Throwable e) {
1062                log.couldNotExecuteCancellationLocally(e.getLocalizedMessage());
1063             }
1064             setCancelled();
1065             return future.cancel(true);
1066          } else {
1067             //already cancelled
1068             return false;
1069          }
1070       }
1071 
1072       @Override
1073       protected V getResult(long timeoutNanos) throws Exception {
1074          if (timeoutNanos > 0) {
1075             return future.get(timeoutNanos, TimeUnit.NANOSECONDS);
1076          } else {
1077             return future.get();
1078          }
1079       }
1080 
1081       @Override
1082       public Address getExecutionTarget() {
1083          return getAddress();
1084       }
1085 
1086       @Override
1087       public void execute() {
1088          log.debugf("Sending %s to self", this);
1089          try {
1090             Callable<V> call = new Callable<V>() {
1091 
1092                @Override
1093                public V call() throws Exception {
1094                   return doLocalInvoke();
1095                }
1096 
1097                private V doLocalInvoke() throws Exception {
1098                   getCommand().init(cache);
1099                   DistributedTaskLifecycleService lifecycle = DistributedTaskLifecycleService.getInstance();
1100                   try {
1101                      // hook into lifecycle
1102                      lifecycle.onPreExecute(getCommand().getCallable(), cache);
1103                      cancellationService.register(Thread.currentThread(), getCommand().getUUID());
1104                      V result = getCommand().perform(null);
1105                      future.notifyDone(result);
1106                      return result;
1107                   } catch (Exception e) {
1108                      future.notifyException(e);
1109                      throw e;
1110                   } finally {
1111                      // hook into lifecycle
1112                      lifecycle.onPostExecute(getCommand().getCallable());
1113                      cancellationService.unregister(getCommand().getUUID());
1114                   }
1115                }
1116             };
1117             future.setFuture(localExecutorService.submit(call));
1118          } catch (Throwable e1) {
1119             log.localExecutionFailed(e1);
1120          }
1121       }
1122 
1123       @Override
1124       public NotifyingFuture<V> attachListener(final FutureListener<V> listener) {
1125          future.attachListener(listener);
1126          return this;
1127       }
1128    }

531    private void transactionalRemoveGroup(String groupName, EnumSet<Flag> explicitFlags, ClassLoader explicitClassLoader) {
532       final boolean onGoingTransaction = getOngoingTransaction() != null;
533       if (!onGoingTransaction) {
534          tryBegin();
535       }
536       try {
537          InvocationContext context = getInvocationContextWithImplicitTransaction(false, explicitClassLoader, UNBOUNDED);
538          Map<K, V> keys = internalGetGroup(groupName, explicitFlags, explicitClassLoader, context);
539          EnumSet<Flag> removeFlags = explicitFlags == null ? EnumSet.noneOf(Flag.class) : EnumSet.copyOf(explicitFlags);
540          removeFlags.add(IGNORE_RETURN_VALUES);
541          for (K key : keys.keySet()) {
542             removeInternal(key, removeFlags, context);
543          }
544          if (!onGoingTransaction) {
545             tryCommit();
546          }
547       } catch (RuntimeException e) {
548          if (!onGoingTransaction) {
549             tryRollback();
550          }
551          throw e;
552       }
553    }

555    private void nonTransactionalRemoveGroup(String groupName, EnumSet<Flag> explicitFlags, ClassLoader explicitClassLoader) {
556       InvocationContext context = getInvocationContextForRead(explicitClassLoader, UNBOUNDED);
557       Map<K, V> keys = internalGetGroup(groupName, explicitFlags, explicitClassLoader, context);
558       EnumSet<Flag> removeFlags = explicitFlags == null ? EnumSet.noneOf(Flag.class) : EnumSet.copyOf(explicitFlags);
559       removeFlags.add(IGNORE_RETURN_VALUES);
560       for (K key : keys.keySet()) {
561          //a new context is needed for remove since in the non-owners, the command is sent to the primary owner to be
562          //executed. If the context is already populated, it throws a ClassCastException because the wrapForRemove is
563          //not invoked.
564          remove(key, removeFlags, explicitClassLoader);
565       }
566    }

778     * not a transactional call.
779     */
780    private InvocationContext getInvocationContextWithImplicitTransaction(boolean isPutForExternalRead, ClassLoader explicitClassLoader, int keyCount) {
781       InvocationContext invocationContext;
782       boolean txInjected = false;
783       if (config.transaction().transactionMode().isTransactional() && !isPutForExternalRead) {
784          Transaction transaction = getOngoingTransaction();

786             transaction = tryBegin();

789          invocationContext = getInvocationContext(transaction, explicitClassLoader, txInjected);
790       } else {
791          invocationContext = getInvocationContextForWrite(explicitClassLoader, keyCount, isPutForExternalRead);
792       }
793       return invocationContext;
794    }

1225    final NotifyingFuture<V> putAsync(final K key, final V value, final Metadata metadata, final EnumSet<Flag> explicitFlags, final ClassLoader explicitClassLoader) {
1226       final NotifyingFutureImpl<V> result = new NotifyingFutureImpl<V>();
1227       final InvocationContext ctx = getInvocationContextWithImplicitTransactionForAsyncOps(false, explicitClassLoader, 1);
1228       Future<V> returnValue = asyncExecutor.submit(new Callable<V>() {
1229          @Override
1230          public V call() throws Exception {
1231             try {
1232                associateImplicitTransactionWithCurrentThread(ctx);
1233                V retval = putInternal(key, value, metadata, explicitFlags, ctx);
1234                try {
1235                   result.notifyDone(retval);
1236                   log.trace("Finished notifying");

1533                   try {

1536                      log.trace("Error when notifying", t);

1554    /**
1555     * Encodes the cases for an asyncGet operation in which it makes sense to actually perform the operation in sync.
1556     *
1557     * @return true if we skip the thread (performing it in sync)
1558     */
1559    private boolean asyncSkipsThread(EnumSet<Flag> flags, K key) {
1560       boolean isSkipLoader = isSkipLoader(flags);
1561       if (!isSkipLoader) {
1562          // if we can't skip the cacheloader, we really want a thread for async.
1563          return false;
1564       }
1565       if (!config.clustering().cacheMode().isDistributed()) {
1566          //in these cluster modes we won't RPC for a get, so no need to fork a thread.
1567          return true;
1568       } else if (flags != null && (flags.contains(Flag.SKIP_REMOTE_LOOKUP) || flags.contains(Flag.CACHE_MODE_LOCAL))) {
1569          //with these flags we won't RPC either
1570          return true;
1571       }
1572       //finally, we will skip the thread if the key maps to the local node
1573       return distributionManager.getLocality(key).isLocal();
1574    }

1576    private boolean isSkipLoader(EnumSet<Flag> flags) {
1577       boolean hasCacheLoaderConfig = !config.persistence().stores().isEmpty();
1578       return !hasCacheLoaderConfig
1579             || (flags != null && (flags.contains(Flag.SKIP_CACHE_LOAD) || flags.contains(Flag.SKIP_CACHE_STORE)));
1580    }

1664          return;

1668       try {
1669          transactionManager.commit();

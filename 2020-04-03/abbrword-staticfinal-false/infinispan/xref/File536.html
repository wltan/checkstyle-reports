55     @Override
56     public void init(InitializationContext ctx) {
57        this.configuration = ctx.getConfiguration();
58        this.dbFactory = newDbFactory();
59        this.ctx = ctx;
60        this.semaphore = new Semaphore(Integer.MAX_VALUE, true);
61  
62        if (this.dbFactory == null) {
63           throw log.cannotLoadlevelDBFactories(Arrays.toString(DB_FACTORY_CLASS_NAMES));
64        }
65        String dbFactoryClassName = this.dbFactory.getClass().getName();
66        if (dbFactoryClassName.equals("org.iq80.leveldb.impl.Iq80DBFactory")) {
67           log.infoUsingJavaDbFactory(dbFactoryClassName);
68        } else {
69           log.infoUsingJNIDbFactory(dbFactoryClassName);
70        }
71     }

73     protected DBFactory newDbFactory() {
74        switch (configuration.implementationType()) {
75           case JNI: {
76              return Util.getInstance(JNI_DB_FACTORY_CLASS_NAME, LevelDBStore.class.getClassLoader());
77           }
78           case JAVA: {
79              return Util.getInstance(JAVA_DB_FACTORY_CLASS_NAME, LevelDBStore.class.getClassLoader());
80           }
81           default: {
82              for (String className : DB_FACTORY_CLASS_NAMES) {
83                 try {
84                    return Util.getInstance(className, LevelDBStore.class.getClassLoader());
85                 } catch (Throwable e) {
86                    if (log.isDebugEnabled())
87                       log.debugUnableToInstantiateDbFactory(className, e);
88                 }
89              }
90           }
91        }
92        return null;
93     }

95     @Override
96     public void start()  {
97        expiryEntryQueue = new LinkedBlockingQueue<ExpiryEntry>(configuration.expiryQueueSize());
98  
99        try {
100          db = openDatabase(getQualifiedLocation(), dataDbOptions());
101          expiredDb = openDatabase(getQualifiedExpiredLocation(), expiredDbOptions());
102          stopped = false;
103       } catch (IOException e) {
104          throw new CacheConfigurationException("Unable to open database", e);
105       }
106    }

108    private String sanitizedCacheName() {
109       String cacheFileName = ctx.getCache().getName().replaceAll("[^a-zA-Z0-9-_\\.]", "_");
110       return cacheFileName;
111    }

113    private String getQualifiedLocation() {
114       return configuration.location() + sanitizedCacheName();
115    }

118       return configuration.expiredLocation() + sanitizedCacheName();

121    private Options dataDbOptions() {
122       Options options = new Options().createIfMissing(true);
123 
124       options.compressionType(CompressionType.valueOf(configuration.compressionType().name()));
125 
126       if (configuration.blockSize() != null) {
127          options.blockSize(configuration.blockSize());
128       }
129 
130       if (configuration.cacheSize() != null) {
131          options.cacheSize(configuration.cacheSize());
132       }
133 
134       return options;
135    }

137    private Options expiredDbOptions() {
138       return new Options().createIfMissing(true);
139    }

144    protected DB openDatabase(String location, Options options) throws IOException {
145       File dir = new File(location);
146 
147       // LevelDB JNI Option createIfMissing doesn't seem to work properly
148       dir.mkdirs();
149       return dbFactory.open(dir, options);
150    }

152    protected void destroyDatabase(String location) throws IOException {
153       File dir = new File(location);
154       dbFactory.destroy(dir, new Options());
155    }

157    protected DB reinitDatabase(String location, Options options) throws IOException {
158       destroyDatabase(location);
159       return openDatabase(location, options);

162    protected void reinitAllDatabases() throws IOException {
163       try {
164          semaphore.acquire(Integer.MAX_VALUE);
165       } catch (InterruptedException e) {
166          throw new PersistenceException("Cannot acquire semaphore", e);
167       }
168       try {
169          if (stopped) {
170             throw new PersistenceException("LevelDB is stopped");
171          }
172          try {
173             db.close();
174          } catch (IOException e) {
175             log.warnUnableToCloseDb(e);
176          }
177          try {
178             expiredDb.close();
179          } catch (IOException e) {
180             log.warnUnableToCloseExpiredDb(e);
181          }
182          db = reinitDatabase(getQualifiedLocation(), dataDbOptions());
183          expiredDb = reinitDatabase(getQualifiedExpiredLocation(), expiredDbOptions());
184       } finally {
185          semaphore.release(Integer.MAX_VALUE);
186       }
187    }

189    @Override
190    public void stop()  {
191       try {
192          semaphore.acquire(Integer.MAX_VALUE);
193       } catch (InterruptedException e) {
194          throw new PersistenceException("Cannot acquire semaphore", e);
195       }
196       try {
197          try {
198             db.close();
199          } catch (IOException e) {
200             log.warnUnableToCloseDb(e);
201          }
202 
203          try {
204             expiredDb.close();
205          } catch (IOException e) {
206             log.warnUnableToCloseExpiredDb(e);
207          }
208       } finally {
209          stopped = true;
210          semaphore.release(Integer.MAX_VALUE);
211       }
212    }

214    @Override
215    public void clear() {
216       long count = 0;
217       boolean destroyDatabase = false;
218       try {
219          semaphore.acquire();
220       } catch (InterruptedException e) {
221          throw new PersistenceException("Cannot acquire semaphore", e);
222       }
223       try {
224          if (stopped) {
225             throw new PersistenceException("LevelDB is stopped");
226          }
227          DBIterator it = db.iterator(new ReadOptions().fillCache(false));
228          if (configuration.clearThreshold() <= 0) {
229             try {
230                for (it.seekToFirst(); it.hasNext(); ) {
231                   Map.Entry<byte[], byte[]> entry = it.next();
232                   db.delete(entry.getKey());
233                   count++;
234 
235                   if (count > configuration.clearThreshold()) {
236                      destroyDatabase = true;
237                      break;
238                   }
239                }
240             } finally {
241                try {
242                   it.close();
243                } catch (IOException e) {
244                   log.warnUnableToCloseDbIterator(e);
245                }
246             }
247          } else {
248             destroyDatabase = true;
249          }
250       } finally {
251          semaphore.release();
252       }
253 
254       if (destroyDatabase) {
255          try {
256             reinitAllDatabases();
257          } catch (IOException e) {
258             throw new PersistenceException(e);
259          }
260       }
261    }

277    @SuppressWarnings("unchecked")
278    @Override
279    public void process(KeyFilter keyFilter, CacheLoaderTask cacheLoaderTask, Executor executor, boolean loadValues, boolean loadMetadata) {
280 
281       int batchSize = 100;
282       ExecutorAllCompletionService eacs = new ExecutorAllCompletionService(executor);
283       final TaskContext taskContext = new TaskContextImpl();
284 
285       List<Map.Entry<byte[], byte[]>> entries = new ArrayList<Map.Entry<byte[], byte[]>>(batchSize);
286       try {
287          semaphore.acquire();
288       } catch (InterruptedException e) {
289          throw new PersistenceException("Cannot acquire semaphore: CacheStore is likely stopped.", e);
290       }
291       try {
292          if (stopped) {
293             throw new PersistenceException("LevelDB is stopped");
294          }
295          DBIterator it = db.iterator(new ReadOptions().fillCache(false));
296          try {
297             for (it.seekToFirst(); it.hasNext(); ) {
298                Map.Entry<byte[], byte[]> entry = it.next();
299                entries.add(entry);
300                if (entries.size() == batchSize) {
301                   final List<Map.Entry<byte[], byte[]>> batch = entries;
302                   entries = new ArrayList<Map.Entry<byte[], byte[]>>(batchSize);
303                   submitProcessTask(cacheLoaderTask, keyFilter, eacs, taskContext, batch, loadValues, loadMetadata);
304                }
305             }
306             if (!entries.isEmpty()) {
307                submitProcessTask(cacheLoaderTask, keyFilter, eacs, taskContext, entries, loadValues, loadMetadata);
308             }
309 
310             eacs.waitUntilAllCompleted();
311             if (eacs.isExceptionThrown()) {
312                throw new PersistenceException("Execution exception!", eacs.getFirstException());
313             }
314          } catch (Exception e) {
315             throw new PersistenceException(e);
316          } finally {
317             try {
318                it.close();
319             } catch (IOException e) {
320                log.warnUnableToCloseDbIterator(e);
321             }
322          }
323       } finally {
324          semaphore.release();
325       }
326    }

328    @SuppressWarnings("unchecked")
329    private void submitProcessTask(final CacheLoaderTask cacheLoaderTask, final KeyFilter filter, CompletionService ecs,
330                                   final TaskContext taskContext, final List<Map.Entry<byte[], byte[]>> batch,
331                                   final boolean loadValues, final boolean loadMetadata) {
332       ecs.submit(new Callable<Void>() {
333          @Override
334          public Void call() throws Exception {
335             try {
336                long now = ctx.getTimeService().wallClockTime();
337                for (Map.Entry<byte[], byte[]> pair : batch) {
338                   if (taskContext.isStopped()) {break;}
339                   Object key = unmarshall(pair.getKey());
340                   if (filter == null || filter.accept(key)) {
341                      MarshalledEntry entry = loadValues || loadMetadata ? (MarshalledEntry) unmarshall(pair.getValue()) : null;
342                      boolean isExpired = entry != null && entry.getMetadata() != null && entry.getMetadata().isExpired(now);
343                      if (!isExpired) {
344                         if (!loadValues || !loadMetadata) {
345                            entry = ctx.getMarshalledEntryFactory().newMarshalledEntry(
346                                  key, loadValues ? entry.getValue() : null, loadMetadata ? entry.getMetadata() : null);
347                         }
348                         cacheLoaderTask.processEntry(entry, taskContext);
349                      }
350                   }
351                }
352             } catch (Exception e) {
353                log.errorExecutingParallelStoreTask(e);
354                throw e;
355             }
356             return null;
357          }
358       });
359    }

361    @Override
362    public boolean delete(Object key)  {
363       try {
364          byte[] keyBytes = marshall(key);
365          semaphore.acquire();
366          try {
367             if (stopped) {
368                throw new PersistenceException("LevelDB is stopped");
369             }
370             if (db.get(keyBytes) == null) {
371                return false;
372             }
373             db.delete(keyBytes);
374          } finally {
375             semaphore.release();
376          }
377          return true;
378       } catch (Exception e) {
379          throw new PersistenceException(e);
380       }
381    }

383    @Override
384    public void write(MarshalledEntry me)  {
385       try {
386          byte[] marshelledKey = marshall(me.getKey());
387          byte[] marshalledEntry = marshall(me);
388          semaphore.acquire();
389          try {
390             if (stopped) {
391                throw new PersistenceException("LevelDB is stopped");
392             }
393             db.put(marshelledKey, marshalledEntry);
394          } finally {
395             semaphore.release();
396          }
397          InternalMetadata meta = me.getMetadata();
398          if (meta != null && meta.expiryTime() > -1) {
399             addNewExpiry(me);
400          }
401       } catch (Exception e) {
402          throw new PersistenceException(e);
403       }
404    }

406    @Override
407    public MarshalledEntry load(Object key)  {
408       try {
409          byte[] marshalledEntry;
410          semaphore.acquire();
411          try {
412             if (stopped) {
413                throw new PersistenceException("LevelDB is stopped");
414             }
415             marshalledEntry = db.get(marshall(key));
416          } finally {
417             semaphore.release();
418          }
419          MarshalledEntry me = (MarshalledEntry) unmarshall(marshalledEntry);
420          if (me == null) return null;
421 
422          InternalMetadata meta = me.getMetadata();
423          if (meta != null && meta.isExpired(ctx.getTimeService().wallClockTime())) {
424             return null;
425          }
426          return me;
427       } catch (Exception e) {
428          throw new PersistenceException(e);
429       }
430    }

432    @SuppressWarnings("unchecked")
433    @Override
434    public void purge(Executor executor, PurgeListener purgeListener) {
435       try {
436          semaphore.acquire();
437       } catch (InterruptedException e) {
438          throw new PersistenceException("Cannot acquire semaphore: CacheStore is likely stopped.", e);
439       }
440       try {
441          if (stopped) {
442             throw new PersistenceException("LevelDB is stopped");
443          }
444          // Drain queue and update expiry tree
445          List<ExpiryEntry> entries = new ArrayList<ExpiryEntry>();
446          expiryEntryQueue.drainTo(entries);
447          for (ExpiryEntry entry : entries) {
448             final byte[] expiryBytes = marshall(entry.expiry);
449             final byte[] keyBytes = marshall(entry.key);
450             final byte[] existingBytes = expiredDb.get(expiryBytes);
451 
452             if (existingBytes != null) {
453                // in the case of collision make the key a List ...
454                final Object existing = unmarshall(existingBytes);
455                if (existing instanceof List) {
456                   ((List<Object>) existing).add(entry.key);
457                   expiredDb.put(expiryBytes, marshall(existing));
458                } else {
459                   List<Object> al = new ArrayList<Object>(2);
460                   al.add(existing);
461                   al.add(entry.key);
462                   expiredDb.put(expiryBytes, marshall(al));
463                }
464             } else {
465                expiredDb.put(expiryBytes, keyBytes);
466             }
467          }
468 
469          List<Long> times = new ArrayList<Long>();
470          List<Object> keys = new ArrayList<Object>();
471          DBIterator it = expiredDb.iterator(new ReadOptions().fillCache(false));
472          long now = ctx.getTimeService().wallClockTime();
473          try {
474             for (it.seekToFirst(); it.hasNext();) {
475                Map.Entry<byte[], byte[]> entry = it.next();
476 
477                Long time = (Long) unmarshall(entry.getKey());
478                if (time > now)
479                   break;
480                times.add(time);
481                Object key = unmarshall(entry.getValue());
482                if (key instanceof List)
483                   keys.addAll((List<?>) key);
484                else
485                   keys.add(key);
486             }
487 
488             for (Long time : times) {
489                expiredDb.delete(marshall(time));
490             }
491 
492             if (!keys.isEmpty())
493                log.debugf("purge (up to) %d entries", keys.size());
494             int count = 0;
495             for (Object key : keys) {
496                byte[] keyBytes = marshall(key);
497 
498                byte[] b = db.get(keyBytes);
499                if (b == null)
500                   continue;
501                MarshalledEntry me = (MarshalledEntry) ctx.getMarshaller().objectFromByteBuffer(b);
502                // TODO race condition: the entry could be updated between the get and delete!
503                if (me.getMetadata() != null && me.getMetadata().isExpired(now)) {
504                   // somewhat inefficient to FIND then REMOVE...
505                   db.delete(keyBytes);
506                   purgeListener.entryPurged(key);
507                   count++;
508                }
509 
510             }
511             if (count != 0)
512                log.debugf("purged %d entries", count);
513          } catch (Exception e) {
514             throw new PersistenceException(e);
515          } finally {
516             try {
517                it.close();
518             } catch (IOException e) {
519                log.warnUnableToCloseDbIterator(e);
520             }
521          }
522       } catch (PersistenceException e) {
523          throw e;
524       } catch (Exception e) {
525          throw new PersistenceException(e);
526       } finally {
527          semaphore.release();
528       }
529    }

531    private byte[] marshall(Object entry) throws IOException, InterruptedException {
532       return ctx.getMarshaller().objectToByteBuffer(entry);
533    }
534 
535    private Object unmarshall(byte[] bytes) throws IOException, ClassNotFoundException {
536       if (bytes == null)
537          return null;
538 
539       return ctx.getMarshaller().objectFromByteBuffer(bytes);
540    }

542    private void addNewExpiry(MarshalledEntry entry) throws IOException {
543       long expiry = entry.getMetadata().expiryTime();
544       long maxIdle = entry.getMetadata().maxIdle();
545       if (maxIdle > 0) {
546          // Coding getExpiryTime() for transient entries has the risk of
547          // being a moving target
548          // which could lead to unexpected results, hence, InternalCacheEntry
549          // calls are required
550          expiry = maxIdle + ctx.getTimeService().wallClockTime();
551       }
552       Long at = expiry;
553       Object key = entry.getKey();
554 
555       try {
556          expiryEntryQueue.put(new ExpiryEntry(at, key));
557       } catch (InterruptedException e) {
558          Thread.currentThread().interrupt(); // Restore interruption status
559       }
560    }

562    private static final class ExpiryEntry {
563       private final Long expiry;
564       private final Object key;
565 
566       private ExpiryEntry(long expiry, Object key) {
567          this.expiry = expiry;
568          this.key = key;
569       }
570 
571       @Override
572       public int hashCode() {
573          final int prime = 31;
574          int result = 1;
575          result = prime * result + ((key == null) ? 0 : key.hashCode());
576          return result;
577       }
578 
579       @Override
580       public boolean equals(Object obj) {
581          if (this == obj)
582             return true;
583          if (obj == null)
584             return false;
585          if (getClass() != obj.getClass())
586             return false;
587          ExpiryEntry other = (ExpiryEntry) obj;
588          if (key == null) {
589             if (other.key != null)
590                return false;
591          } else if (!key.equals(other.key))
592             return false;
593          return true;
594       }

6   import org.jboss.as.controller.PathElement;
7   import org.jboss.as.controller.ReloadRequiredRemoveStepHandler;
8   import org.jboss.as.controller.ReloadRequiredWriteAttributeHandler;
9   import org.jboss.as.controller.SimpleAttributeDefinition;
10  import org.jboss.as.controller.SimpleAttributeDefinitionBuilder;
11  import org.jboss.as.controller.SimpleResourceDefinition;
12  import org.jboss.as.controller.StringListAttributeDefinition;
13  import org.jboss.as.controller.registry.AttributeAccess;
14  import org.jboss.as.controller.registry.ManagementResourceRegistration;
15  import org.jboss.dmr.ModelType;
16  
17  /**
18   * AuthorizationRoleResource.
19   *
20   * @author Tristan Tarrant
21   * @since 7.0
22   */
23  public class AuthorizationRoleResource extends SimpleResourceDefinition {
24  
25      static final SimpleAttributeDefinition NAME = new SimpleAttributeDefinitionBuilder(ModelKeys.NAME, ModelType.STRING, true)
26          .setXmlName(Attribute.NAME.getLocalName())
27          .setAllowExpression(true)
28          .setAllowNull(false)
29          .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
30          .build()
31      ;
32  
33      static final StringListAttributeDefinition PERMISSIONS = new StringListAttributeDefinition.Builder(ModelKeys.PERMISSIONS)
34          .build()
35      ;
36  
37      static final AttributeDefinition[] ATTRIBUTES = new AttributeDefinition[] { NAME, PERMISSIONS };
38  
39      AuthorizationRoleResource() {
40          super(PathElement.pathElement(ModelKeys.ROLE), InfinispanExtension.getResourceDescriptionResolver(ModelKeys.CACHE_CONTAINER, ModelKeys.SECURITY, ModelKeys.AUTHORIZATION, ModelKeys.ROLE), new CacheConfigAdd(ATTRIBUTES), ReloadRequiredRemoveStepHandler.INSTANCE);
41      }
42  
43      @Override
44      public void registerAttributes(ManagementResourceRegistration registration) {
45          final OperationStepHandler writeHandler = new ReloadRequiredWriteAttributeHandler(ATTRIBUTES);
46          for (AttributeDefinition attribute: ATTRIBUTES) {
47              registration.registerReadWriteAttribute(attribute, null, writeHandler);
48          }
49      }
50  
51  }

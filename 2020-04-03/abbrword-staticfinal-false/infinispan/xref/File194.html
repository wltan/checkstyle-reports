1   package org.infinispan.configuration.cache;
2   
3   import java.util.ArrayList;
4   import java.util.Collections;
5   import java.util.HashSet;
6   import java.util.List;
7   import java.util.Set;
8   
9   import org.infinispan.commons.configuration.attributes.Attribute;
10  import org.infinispan.commons.configuration.attributes.AttributeDefinition;
11  import org.infinispan.commons.configuration.attributes.AttributeInitializer;
12  import org.infinispan.commons.configuration.attributes.AttributeSet;
13  
14  /**
15   * @author Mircea.Markus@jboss.com
16   * @since 5.2
17   */
18  public class SitesConfiguration {
19     public static final AttributeDefinition<Boolean> DISABLE_BACKUPS = AttributeDefinition.builder("disableBackups", false).immutable().build();
20     public static final AttributeDefinition<Set<String>> IN_USE_BACKUP_SITES = AttributeDefinition.builder("inUseBackupSites", null, (Class<Set<String>>)(Class<?>)Set.class)
21           .initializer(new AttributeInitializer<Set<String>>() {
22              @Override
23              public Set<String> initialize() {
24                 return new HashSet<>(2);
25              }
26           }).immutable().build();
27  
28     static final AttributeSet attributeDefinitionSet() {
29        return new AttributeSet(SitesConfiguration.class, DISABLE_BACKUPS, IN_USE_BACKUP_SITES);
30     }
31  
32     private final BackupForConfiguration backupFor;
33     private final List<BackupConfiguration> allBackups;
34     private final Attribute<Boolean> disableBackups;
35     private final Attribute<Set<String>> inUseBackupSites;
36     private final AttributeSet attributes;
37  
38     public SitesConfiguration(AttributeSet attributes, List<BackupConfiguration> allBackups, BackupForConfiguration backupFor) {
39        this.attributes = attributes.checkProtection();
40        this.allBackups = Collections.unmodifiableList(allBackups);
41        this.disableBackups = attributes.attribute(DISABLE_BACKUPS);
42        this.inUseBackupSites = attributes.attribute(IN_USE_BACKUP_SITES);
43        this.backupFor = backupFor;
44     }
45  
46     /**
47      * Returns true if this cache won't backup its data remotely.
48      * It would still accept other sites backing up data on this site.
49      */
50     public boolean disableBackups() {
51        return disableBackups.get();
52     }
53  
54     /**
55      * Returns the list of all sites where this cache might back up its data. The list of actual sites is defined by
56      * {@link #inUseBackupSites}.
57      */
58     public List<BackupConfiguration> allBackups() {
59        return allBackups;
60     }
61  
62  
63     /**
64      * Returns the list of {@link BackupConfiguration} that have {@link org.infinispan.configuration.cache.BackupConfiguration#enabled()} == true.
65      */
66     public List<BackupConfiguration> enabledBackups() {
67        List<BackupConfiguration> result = new ArrayList<>();
68        for (BackupConfiguration bc : allBackups) {
69           if (bc.enabled()) result.add(bc);
70        }
71        return result;
72     }
73  
74     /**
75      * @return information about caches that backup data into this cache.
76      */
77     public BackupForConfiguration backupFor() {
78        return backupFor;
79     }
80  
81     public BackupFailurePolicy getFailurePolicy(String siteName) {
82        for (BackupConfiguration bc : allBackups) {
83           if (bc.site().equals(siteName)) {
84              return bc.backupFailurePolicy();
85           }
86        }
87        throw new IllegalStateException("There must be a site configured for " + siteName);
88     }
89  
90     public boolean hasInUseBackup(String siteName) {
91        for (BackupConfiguration bc : allBackups) {
92           if (bc.site().equals(siteName)) {
93              return bc.enabled();
94           }
95        }
96        return false;
97     }
98  
99     public boolean hasEnabledBackups() {
100       for (BackupConfiguration bc : allBackups) {
101          if (bc.enabled()) return true;
102       }
103       return false;
104    }
105 
106    public Set<String> inUseBackupSites() {
107       return inUseBackupSites.get();
108    }
109 
110    public AttributeSet attributes() {
111       return attributes;
112    }
113 
114    @Override
115    public int hashCode() {
116       final int prime = 31;
117       int result = 1;
118       result = prime * result + ((attributes == null) ? 0 : attributes.hashCode());
119       result = prime * result + ((backupFor == null) ? 0 : backupFor.hashCode());
120       return result;
121    }
122 
123    @Override
124    public boolean equals(Object obj) {
125       if (this == obj)
126          return true;
127       if (obj == null)
128          return false;
129       if (getClass() != obj.getClass())
130          return false;
131       SitesConfiguration other = (SitesConfiguration) obj;
132       if (attributes == null) {
133          if (other.attributes != null)
134             return false;
135       } else if (!attributes.equals(other.attributes))
136          return false;
137       if (backupFor == null) {
138          if (other.backupFor != null)
139             return false;
140       } else if (!backupFor.equals(other.backupFor))
141          return false;
142       return true;
143    }
144 
145    @Override
146    public String toString() {
147       return "SitesConfiguration [backupFor=" + backupFor + ", allBackups=" + allBackups + ", attributes=" + attributes + "]";
148    }
149 }

1   package org.infinispan.configuration.cache;
2   
3   import java.util.Properties;
4   
5   import org.infinispan.commons.configuration.attributes.Attribute;
6   import org.infinispan.commons.configuration.attributes.AttributeDefinition;
7   import org.infinispan.commons.configuration.attributes.AttributeInitializer;
8   import org.infinispan.commons.configuration.attributes.AttributeSet;
9   import org.infinispan.commons.util.TypedProperties;
10  
11  public class AbstractStoreConfiguration implements StoreConfiguration {
12     public static final AttributeDefinition<Boolean> FETCH_PERSISTENT_STATE = AttributeDefinition.builder("fetchPersistentState", false).immutable().build();
13     public static final AttributeDefinition<Boolean> PURGE_ON_STARTUP = AttributeDefinition.builder("purgeOnStartup", false).immutable().build();
14     public static final AttributeDefinition<Boolean> IGNORE_MODIFICATIONS = AttributeDefinition.builder("ignoreModifications", false).immutable().build();
15     public static final AttributeDefinition<Boolean> PRELOAD = AttributeDefinition.builder("preload", false).immutable().build();
16     public static final AttributeDefinition<Boolean> SHARED = AttributeDefinition.builder("shared", false).immutable().build();
17     public static final AttributeDefinition<TypedProperties> PROPERTIES = AttributeDefinition.builder("properties", null, TypedProperties.class).initializer(new AttributeInitializer<TypedProperties>() {
18        @Override
19        public TypedProperties initialize() {
20           return new TypedProperties();
21        }
22     }).immutable().build();
23  
24     public static AttributeSet attributeDefinitionSet() {
25        return new AttributeSet(AbstractStoreConfiguration.class, FETCH_PERSISTENT_STATE, PURGE_ON_STARTUP, IGNORE_MODIFICATIONS, PRELOAD, SHARED, PROPERTIES);
26     }
27  
28     private final Attribute<Boolean> fetchPersistentState;
29     private final Attribute<Boolean> purgeOnStartup;
30     private final Attribute<Boolean> ignoreModifications;
31     private final Attribute<Boolean> preload;
32     private final Attribute<Boolean> shared;
33     private final Attribute<TypedProperties> properties;
34  
35     protected final AttributeSet attributes;
36     private final AsyncStoreConfiguration async;
37     private final SingletonStoreConfiguration singletonStore;
38  
39     /**
40      * @deprecated Use {@link AbstractStoreConfiguration#AbstractStoreConfiguration(AttributeSet, AsyncStoreConfiguration, SingletonStoreConfiguration) instead
41      */
42     @Deprecated
43     public AbstractStoreConfiguration(boolean purgeOnStartup, boolean fetchPersistentState, boolean ignoreModifications,
44           AsyncStoreConfiguration async, SingletonStoreConfiguration singletonStore, boolean preload, boolean shared, Properties properties) {
45        attributes = attributeDefinitionSet();
46        attributes.attribute(PURGE_ON_STARTUP).set(purgeOnStartup);
47        attributes.attribute(FETCH_PERSISTENT_STATE).set(fetchPersistentState);
48        attributes.attribute(IGNORE_MODIFICATIONS).set(ignoreModifications);
49        attributes.attribute(PRELOAD).set(preload);
50        attributes.attribute(SHARED).set(shared);
51        attributes.attribute(PROPERTIES).set(TypedProperties.toTypedProperties(properties));
52  
53        this.async = async;
54        this.singletonStore = singletonStore;
55        this.fetchPersistentState = attributes.attribute(FETCH_PERSISTENT_STATE);
56        this.purgeOnStartup = attributes.attribute(PURGE_ON_STARTUP);
57        this.ignoreModifications = attributes.attribute(IGNORE_MODIFICATIONS);
58        this.preload = attributes.attribute(PRELOAD);
59        this.shared = attributes.attribute(SHARED);
60        this.properties = attributes.attribute(PROPERTIES);
61     }
62  
63     public AbstractStoreConfiguration(AttributeSet attributes, AsyncStoreConfiguration async,
64           SingletonStoreConfiguration singletonStore) {
65        this.attributes = attributes.checkProtection();
66        this.async = async;
67        this.singletonStore = singletonStore;
68        this.fetchPersistentState = attributes.attribute(FETCH_PERSISTENT_STATE);
69        this.purgeOnStartup = attributes.attribute(PURGE_ON_STARTUP);
70        this.ignoreModifications = attributes.attribute(IGNORE_MODIFICATIONS);
71        this.preload = attributes.attribute(PRELOAD);
72        this.shared = attributes.attribute(SHARED);
73        this.properties = attributes.attribute(PROPERTIES);
74     }
75  
76     /**
77      * Configuration for the async cache loader. If enabled, this provides you with asynchronous
78      * writes to the cache store, giving you 'write-behind' caching.
79      */
80     @Override
81     public AsyncStoreConfiguration async() {
82        return async;
83     }
84  
85     /**
86      * SingletonStore is a delegating store used for situations when only one instance in a cluster
87      * should interact with the underlying store. The coordinator of the cluster will be responsible
88      * for the underlying CacheStore. SingletonStore is a simply facade to a real CacheStore
89      * implementation. It always delegates reads to the real CacheStore.
90      */
91     @Override
92     public SingletonStoreConfiguration singletonStore() {
93        return singletonStore;
94     }
95  
96     /**
97      * If true, purges this cache store when it starts up.
98      */
99     @Override
100    public boolean purgeOnStartup() {
101       return purgeOnStartup.get();
102    }
103 
104    @Override
105    public boolean shared() {
106       return shared.get();
107    }
108 
109    /**
110     * If true, fetch persistent state when joining a cluster. If multiple cache stores are chained,
111     * only one of them can have this property enabled. Persistent state transfer with a shared cache
112     * store does not make sense, as the same persistent store that provides the data will just end
113     * up receiving it. Therefore, if a shared cache store is used, the cache will not allow a
114     * persistent state transfer even if a cache store has this property set to true.
115     */
116    @Override
117    public boolean fetchPersistentState() {
118       return fetchPersistentState.get();
119    }
120 
121    /**
122     * If true, any operation that modifies the cache (put, remove, clear, store...etc) won't be
123     * applied to the cache store. This means that the cache store could become out of sync with the
124     * cache.
125     */
126    @Override
127    public boolean ignoreModifications() {
128       return ignoreModifications.get();
129    }
130 
131    @Override
132    public boolean preload() {
133       return preload.get();
134    }
135 
136    @Override
137    public Properties properties() {
138       return properties.get();
139    }
140 
141    public AttributeSet attributes() {
142       return attributes;
143    }
144 
145    @Override
146    public String toString() {
147       return "AbstractStoreConfiguration [attributes=" + attributes + ", async=" + async + ", singletonStore="
148             + singletonStore + "]";
149    }
150 
151    @Override
152    public int hashCode() {
153       final int prime = 31;
154       int result = 1;
155       result = prime * result + ((async == null) ? 0 : async.hashCode());
156       result = prime * result + ((attributes == null) ? 0 : attributes.hashCode());
157       result = prime * result + ((singletonStore == null) ? 0 : singletonStore.hashCode());
158       return result;
159    }
160 
161    @Override
162    public boolean equals(Object obj) {
163       if (this == obj)
164          return true;
165       if (obj == null)
166          return false;
167       if (getClass() != obj.getClass())
168          return false;
169       AbstractStoreConfiguration other = (AbstractStoreConfiguration) obj;
170       if (async == null) {
171          if (other.async != null)
172             return false;
173       } else if (!async.equals(other.async))
174          return false;
175       if (attributes == null) {
176          if (other.attributes != null)
177             return false;
178       } else if (!attributes.equals(other.attributes))
179          return false;
180       if (singletonStore == null) {
181          if (other.singletonStore != null)
182             return false;
183       } else if (!singletonStore.equals(other.singletonStore))
184          return false;
185       return true;
186    }

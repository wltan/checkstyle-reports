1   package org.infinispan.configuration.cache;
2   
3   import org.infinispan.commons.configuration.BuiltBy;
4   import org.infinispan.commons.configuration.ConfigurationFor;
5   import org.infinispan.commons.configuration.attributes.Attribute;
6   import org.infinispan.commons.configuration.attributes.AttributeDefinition;
7   import org.infinispan.commons.configuration.attributes.AttributeSet;
8   import org.infinispan.persistence.file.SingleFileStore;
9   
10  /**
11   * Defines the configuration for the single file cache store.
12   *
13   * @author Galder Zamarre√±o
14   * @since 6.0
15   */
16  @BuiltBy(SingleFileStoreConfigurationBuilder.class)
17  @ConfigurationFor(SingleFileStore.class)
18  public class SingleFileStoreConfiguration extends AbstractStoreConfiguration {
19     public static final AttributeDefinition<String> LOCATION = AttributeDefinition.builder("location", "Infinispan-SingleFileStore").immutable().build();
20     public static final AttributeDefinition<Integer> MAX_ENTRIES = AttributeDefinition.builder("maxEntries", -1).immutable().build();
21     public static final AttributeDefinition<Float> FRAGMENTATION_FACTOR = AttributeDefinition.builder("fragmentationFactor", 0.75f).immutable().build();
22     public static AttributeSet attributeDefinitionSet() {
23        return new AttributeSet(SingleFileStoreConfiguration.class, AbstractStoreConfiguration.attributeDefinitionSet(), LOCATION, MAX_ENTRIES, FRAGMENTATION_FACTOR);
24     }
25  
26     private final Attribute<String> location;
27     private final Attribute<Integer> maxEntries;
28     private final Attribute<Float> fragmentationFactor;
29  
30     public SingleFileStoreConfiguration(AttributeSet attributes, AsyncStoreConfiguration async,
31                                         SingletonStoreConfiguration singletonStore) {
32        super(attributes, async, singletonStore);
33        location = attributes.attribute(LOCATION);
34        maxEntries = attributes.attribute(MAX_ENTRIES);
35        fragmentationFactor = attributes.attribute(FRAGMENTATION_FACTOR);
36     }
37  
38     public String location() {
39        return location.get();
40     }
41  
42     public int maxEntries() {
43        return maxEntries.get();
44     }
45  
46     public float fragmentationFactor () {
47        return fragmentationFactor.get();
48     }
49  
50     @Override
51     public AttributeSet attributes() {
52        return attributes;
53     }
54  
55     @Override
56     public String toString() {
57        return "SingleFileStoreConfiguration [attributes=" + attributes + "]";
58     }
59  }

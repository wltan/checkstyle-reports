131    /**
132     * <p>
133     * Loads the specified class using the passed classloader, or, if it is <code>null</code> the Infinispan classes' classloader.
134     * </p>
135     *
136     * <p>
137     * If loadtime instrumentation via GenerateInstrumentedClassLoader is used, this class may be loaded by the bootstrap classloader.
138     * </p>
139     *
140     * @param classname name of the class to load
141     * @return the class
142     * @param userClassLoader the application classloader which should be used to load the class, or null if the class is always packaged with
143     *        Infinispan
144     * @throws ClassNotFoundException if the class cannot be loaded
145     */
146    @SuppressWarnings("unchecked")
147    public static <T> Class<T> loadClassStrict(String classname, ClassLoader userClassLoader) throws ClassNotFoundException {
148       ClassLoader[] cls = getClassLoaders(userClassLoader);
149          ClassNotFoundException e = null;
150          NoClassDefFoundError ne = null;
151          for (ClassLoader cl : cls)  {
152             if (cl == null)
153                continue;
154 
155             try {
156                return (Class<T>) Class.forName(classname, true, cl);
157             } catch (ClassNotFoundException ce) {
158                e = ce;
159             } catch (NoClassDefFoundError ce) {
160                ne = ce;
161             }
162          }
163 
164          if (e != null)
165             throw e;
166          else if (ne != null) {
167             // Before we wrap this, make sure we appropriately log this.
168             log.unableToLoadClass(classname, Arrays.toString(cls), ne);
169             throw new ClassNotFoundException(classname, ne);
170          }
171          else
172             throw new IllegalStateException();
173    }

191    private static Method getFactoryMethod(Class<?> c) {
192       for (Method m : c.getMethods()) {
193          if (m.getName().equals("getInstance") && m.getParameterTypes().length == 0 && Modifier.isStatic(m.getModifiers()))
194             return m;
195       }
196       return null;
197    }

199    /**
200     * Instantiates a class by first attempting a static <i>factory method</i> named <tt>getInstance()</tt> on the class
201     * and then falling back to an empty constructor.
202     * <p/>
203     * Any exceptions encountered are wrapped in a {@link CacheConfigurationException} and rethrown.
204     *
205     * @param clazz class to instantiate
206     * @return an instance of the class
207     */
208    public static <T> T getInstance(Class<T> clazz) {
209       try {
210          return getInstanceStrict(clazz);
211       } catch (IllegalAccessException iae) {
212          throw new CacheConfigurationException("Unable to instantiate class " + clazz.getName(), iae);
213       } catch (InstantiationException ie) {
214          throw new CacheConfigurationException("Unable to instantiate class " + clazz.getName(), ie);
215       }
216    }

218    /**
219     * Similar to {@link #getInstance(Class)} except that exceptions are propagated to the caller.
220     *
221     * @param clazz class to instantiate
222     * @return an instance of the class
223     * @throws IllegalAccessException
224     * @throws InstantiationException
225     */
226    @SuppressWarnings("unchecked")
227    public static <T> T getInstanceStrict(Class<T> clazz) throws IllegalAccessException, InstantiationException {
228       // first look for a getInstance() constructor
229       T instance = null;
230       try {
231          Method factoryMethod = getFactoryMethod(clazz);
232          if (factoryMethod != null) instance = (T) factoryMethod.invoke(null);
233       }
234       catch (Exception e) {
235          // no factory method or factory method failed.  Try a constructor.
236          instance = null;
237       }
238       if (instance == null) {
239          instance = clazz.newInstance();
240       }
241       return instance;
242    }

358     *
359     * @param is the stream to read
360     * @return the read bytes
361     * @throws java.io.IOException in case of stream read errors

363    public static byte[] readStream(InputStream is) throws IOException {
364       try {
365          ByteArrayOutputStream os = new ByteArrayOutputStream();
366          byte[] buf = new byte[1024];
367          int len;
368          while ((len = is.read(buf)) != -1) {
369             os.write(buf, 0, len);
370          }
371          return os.toByteArray();
372       } finally {
373          is.close();
374       }
375    }

475    public static <E> String toStr(Collection<E> collection) {
476       if (collection == null)
477          return "[]";
478 
479       Iterator<E> i = collection.iterator();
480       if (!i.hasNext())
481          return "[]";
482 
483       StringBuilder sb = new StringBuilder();
484       sb.append('[');
485       for (;;) {
486          E e = i.next();
487          sb.append(e == collection ? "(this Collection)" : toStr(e));
488          if (! i.hasNext())
489             return sb.append(']').toString();
490          sb.append(", ");
491       }
492    }

498    public static String printArray(byte[] array, boolean withHash) {
499       if (array == null) return "null";
500 
501       int limit = 8;
502       StringBuilder sb = new StringBuilder();
503       sb.append("[B0x");
504       if (array.length <= limit || IS_ARRAYS_DEBUG) {
505          // Convert the entire byte array
506          sb.append(toHexString(array));
507          if (withHash) {
508             sb.append(",h=");
509             sb.append(Integer.toHexString(Arrays.hashCode(array)));
510             sb.append(']');
511          }
512       } else {
513          // Pick the first 8 characters and convert that part
514          sb.append(toHexString(array, limit));
515          sb.append("..[");
516          sb.append(array.length);
517          if (withHash) {
518             sb.append("],h=");
519             sb.append(Integer.toHexString(Arrays.hashCode(array)));
520          }
521          sb.append(']');
522       }
523       return sb.toString();
524    }

530    public static String toHexString(byte input[], int limit) {
531       int i = 0;
532       if (input == null || input.length <= 0)
533          return null;
534 
535       char lookup[] = {'0', '1', '2', '3', '4', '5', '6', '7',
536                        '8', '9', 'a', 'b', 'c', 'd', 'e', 'f'};
537 
538       char[] result = new char[(input.length < limit ? input.length : limit) * 2];
539 
540       while (i < limit && i < input.length) {
541          result[2*i] = lookup[(input[i] >> 4) & 0x0F];
542          result[2*i+1] = lookup[(input[i] & 0x0F)];
543          i++;
544       }
545       return String.valueOf(result);
546    }

559    public static String threadDump() {
560       StringBuilder threadDump = new StringBuilder();
561       ThreadMXBean threadMx = ManagementFactory.getThreadMXBean();
562       if (threadMx.isObjectMonitorUsageSupported() && threadMx.isSynchronizerUsageSupported()) {
563          // Print lock info if, and only if, both object monitor usage and synchronizer usage are supported.
564           dumpThreadInfo(threadDump, true, threadMx);
565       } else {
566          dumpThreadInfo(threadDump, false, threadMx);
567       }
568       return threadDump.toString();

571    private static void dumpThreadInfo(StringBuilder threadDump, boolean withLocks, ThreadMXBean threadMx) {
572       SimpleDateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
573       String timestamp = dateFormat.format(new Date());
574       threadDump.append(timestamp);
575       threadDump.append("\nFull thread dump ");
576       threadDump.append("\n");
577 
578       if (withLocks) {
579          ThreadInfo[] threadInfos = threadMx.dumpAllThreads(true, true);
580          for (ThreadInfo threadInfo : threadInfos) {
581             printThreadInfo(threadInfo, threadDump);
582             LockInfo[] syncs = threadInfo.getLockedSynchronizers();
583             printLockInfo(syncs, threadDump);
584          }
585          threadDump.append("\n");
586       } else {
587          long[] threadIds= threadMx.getAllThreadIds();
588          ThreadInfo[] threadInfos = threadMx.getThreadInfo(threadIds, Integer.MAX_VALUE);
589          for (ThreadInfo threadInfo : threadInfos)
590             printThreadInfo(threadInfo, threadDump);
591       }
592    }

594    private static void printThreadInfo(ThreadInfo threadInfo, StringBuilder threadDump) {
595       // Print thread information
596       printThread(threadInfo, threadDump);
597       // print stack trace with locks
598       StackTraceElement[] stacktrace = threadInfo.getStackTrace();
599       MonitorInfo[] monitors = threadInfo.getLockedMonitors();
600       for (int i = 0; i < stacktrace.length; i++) {
601           StackTraceElement ste = stacktrace[i];
602           threadDump.append(INDENT + "at " + ste.toString());
603           threadDump.append("\n");
604           for (int j = 1; j < monitors.length; j++) {
605               MonitorInfo mi = monitors[j];
606               if (mi.getLockedStackDepth() == i) {
607                   threadDump.append(INDENT + "  - locked " + mi);
608                   threadDump.append("\n");
609               }
610           }
611       }
612       threadDump.append("\n");
613    }

615    private static void printLockInfo(LockInfo[] locks, StringBuilder threadDump) {
616       threadDump.append(INDENT + "Locked synchronizers: count = " + locks.length);
617       threadDump.append("\n");
618       for (LockInfo li : locks) {
619          threadDump.append(INDENT + "  - " + li);
620          threadDump.append("\n");
621       }
622       threadDump.append("\n");
623    }

625    private static void printThread(ThreadInfo threadInfo, StringBuilder threadDump) {
626       StringBuilder sb = new StringBuilder(
627             "\"" + threadInfo.getThreadName() + "\"" +
628             " nid=" + threadInfo.getThreadId() + " state=" +
629             threadInfo.getThreadState());
630       if (threadInfo.getLockName() != null && threadInfo.getThreadState() != Thread.State.BLOCKED) {
631           String[] lockInfo = threadInfo.getLockName().split("@");
632           sb.append("\n" + INDENT +"- waiting on <0x" + lockInfo[1] + "> (a " + lockInfo[0] + ")");
633           sb.append("\n" + INDENT +"- locked <0x" + lockInfo[1] + "> (a " + lockInfo[0] + ")");
634       } else if (threadInfo.getLockName() != null && threadInfo.getThreadState() == Thread.State.BLOCKED) {
635           String[] lockInfo = threadInfo.getLockName().split("@");
636           sb.append("\n" + INDENT +"- waiting to lock <0x" + lockInfo[1] + "> (a " + lockInfo[0] + ")");
637       }
638       if (threadInfo.isSuspended())
639           sb.append(" (suspended)");
640 
641       if (threadInfo.isInNative())
642           sb.append(" (running in native)");
643 
644       threadDump.append(sb.toString());
645       threadDump.append("\n");
646       if (threadInfo.getLockOwnerName() != null) {
647            threadDump.append(INDENT + " owned by " + threadInfo.getLockOwnerName() +
648                               " id=" + threadInfo.getLockOwnerId());
649            threadDump.append("\n");
650       }
651    }

703       if (type.equals(Long.class) || type.equals(long.class))
704          return Double.valueOf((Long) o);
705       else if (type.equals(Double.class) || type.equals(double.class))
706          return (Double) o;
707       else if (type.equals(Integer.class) || type.equals(int.class))
708          return Double.valueOf((Integer) o);
709       else if (type.equals(String.class))
710          return Double.valueOf((String) o);
711 
712       throw new IllegalStateException(String.format("Expected a value that can be converted into a double: type=%s, value=%s", type, o));
713    }

756    /**
757     * Returns a number such that the number is a power of two that is equal to, or greater than, the number passed in as
758     * an argument.  The smallest number returned will be 1, not 0.
759     */
760    public static int findNextHighestPowerOfTwo(int num) {
761       if (num <= 0) return 1;
762       int highestBit = Integer.highestOneBit(num);
763       return num <= highestBit ? highestBit : highestBit << 1;
764    }

766    /**
767     * A function that calculates hash code of a byte array based on its
768     * contents but using the given size parameter as deliminator for the
769     * content.
770     */
771    public static int hashCode(byte[] bytes, int size) {
772       int contentLimit = size;
773       if (size > bytes.length)
774          contentLimit = bytes.length;
775 
776       int hashCode = 1;
777       for (int i = 0; i < contentLimit; i++)
778          hashCode = 31 * hashCode + bytes[i];
779 
780       return hashCode;
781    }

793    /**
794     * Concatenates an arbitrary number of arrays returning a new array containing all elements
795     */
796    @SafeVarargs
797    public static <T> T[] arrayConcat(T[] first, T[]... rest) {
798       int totalLength = first.length;
799       for (T[] array : rest) {
800         totalLength += array.length;
801       }
802       T[] result = Arrays.copyOf(first, totalLength);
803       int offset = first.length;
804       for (T[] array : rest) {
805         System.arraycopy(array, 0, result, offset, array.length);
806         offset += array.length;
807       }
808       return result;
809     }

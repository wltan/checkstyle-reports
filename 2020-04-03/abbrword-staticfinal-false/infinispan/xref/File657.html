23  package org.jboss.as.clustering.infinispan;
24  
25  import org.infinispan.configuration.cache.CacheMode;
26  import org.jboss.as.controller.OperationFailedException;
27  import org.jboss.as.controller.persistence.ConfigurationPersistenceException;
28  import org.jboss.as.network.OutboundSocketBinding;
29  import org.jboss.logging.Messages;
30  import org.jboss.logging.annotations.Cause;
31  import org.jboss.logging.annotations.Message;
32  import org.jboss.logging.annotations.MessageBundle;
33  import org.jboss.msc.inject.InjectionException;
34  import org.jboss.msc.service.StartException;
35  
36  import java.net.UnknownHostException;
37  import java.util.Properties;
38  
39  /**
40   * InfinispanMessages
41   *
42   * logging id range: 10290 - 10299,10380 - 10389, 11000 - 11099
43   *
44   * @author <a href="mailto:jperkins@redhat.com">James R. Perkins</a>
45   */
46  @MessageBundle(projectCode = "JBAS")
47  public interface InfinispanMessages {
48  
49      /**
50       * A logger with the category of the default clustering package.
51       */
52      InfinispanMessages MESSAGES = Messages.getBundle(InfinispanMessages.class);
53  
54      /**
55       * Creates an exception indicating a failure to resolve the outbound socket binding represented by the
56       * {@code binding} parameter.
57       *
58       * @param cause the cause of the error.
59       * @param binding the outbound socket binding.
60       *
61       * @return a {@link ConfigurationPersistenceException} for the error.
62       */
63      @Message(id = 10290, value = "Could not resolve destination address for outbound socket binding named '%s'")
64      InjectionException failedToInjectSocketBinding(@Cause UnknownHostException cause, OutboundSocketBinding binding);
65  
66      @Message(id = 10291, value = "Failed to add %s %s cache to non-clustered %s cache container.")
67      StartException transportRequired(CacheMode mode, String cache, String cacheContainer);
68  
69      /**
70       * Creates an exception indicating an invalid cache store.
71       *
72       * @param cause          the cause of the error.
73       * @param cacheStoreName the name of the cache store.
74       *
75       * @return an {@link OperationFailedException} for the error.
76       */
77      @Message(id = 10292, value = "%s is not a valid cache store")
78      OperationFailedException invalidCacheStore(@Cause Throwable cause, String cacheStoreName);
79  
80      /**
81       * Creates an exception indicating an invalid cache store.
82       *
83       * @param cacheName     the name of the cache store.
84       * @param cacheContainerName the container name.
85       *
86       * @return an {@link IllegalArgumentException} for the error.
87       */
88      @Message(id = 10293, value = "%s is not a valid default cache. The %s cache container does not contain a cache with that name")
89      IllegalArgumentException invalidDefaultCache(String cacheName, String cacheContainerName);
90  
91      /**
92       * Creates an exception indicating the an executor property is invalid.
93       *
94       * @param id         the id of the property.
95       * @param properties the properties that were searched.
96       *
97       * @return an {@link IllegalStateException} for the error.
98       */
99      @Message(id = 10294, value = "No %s property was specified within the executor properties: %s")
100     IllegalStateException invalidExecutorProperty(String id, Properties properties);
101 
102     /**
103      * Creates an exception indicating the an transport property is invalid.
104      *
105      * @param id         the id of the property.
106      * @param properties the properties that were searched.
107      *
108      * @return an {@link IllegalStateException} for the error.
109      */
110     @Message(id = 10295, value = "No %s property was specified within the transport properties: %s")
111     IllegalStateException invalidTransportProperty(String id, Properties properties);
112 
113     /**
114      * Creates an exception indicating that the cache is aborting after the specified number of retries.
115      *
116      * @param cause           the cause of the error.
117      * @param numberOfRetries the number of retries.
118      *
119      * @return a {@link RuntimeException}
120      */
121     @Message(id = 10296, value = "Aborting cache operation after %d retries.")
122     RuntimeException abortingCacheOperation(@Cause Throwable cause, int numberOfRetries);
123 
124     /**
125      * Creates an exception indicating the an operation parameter is invalid.
126      *
127      * @param id         the id of the parameter.
128      * @param allowableValues the allowable values for the parameter
129      *
130      * @return the String.
131      */
132     @Message(id = 10297, value = "Invalid value for parameter %s. Allowable values: %s")
133     String invalidParameterValue(String id, String allowableValues);
134 
135     /**
136      * Creates an exception indicating the a cache store cannot be added as one already exists.
137      *
138      * @param existingStoreName the store which already exists.
139      *
140      * @return an {@link OperationFailedException} for the error.
141      */
142     @Message(id = 10298, value = "Cache store cannot be created: cache store %s is already defined")
143     OperationFailedException cacheStoreAlreadyDefined(String existingStoreName);
144 
145     /**
146      * Creates an exception indicating the a cache store cannot be added as one already exists.
147      *
148      * @param propertyKey the name of the property.
149      *
150      * @return an {@link OperationFailedException} for the error.
151      */
152     @Message(id = 10299, value = "Value for property with key %s is not defined")
153     OperationFailedException propertyValueNotDefined(String propertyKey);
154 
155     /**
156      * A message indicating that the resource could not be located.
157      *
158      * @param resourceName the name of the resource.
159      *
160      * @return the String message.
161      */
162     @Message(id = 10380, value = "Failed to locate %s")
163     String notFound(String resourceName);
164 
165     /**
166      * A message indicating that the resource could not be parsed.
167      *
168      * @param resourceName the name of the resource.
169      *
170      * @return IllegalStateException instance.
171      */
172     @Message(id = 10381, value = "Failed to parse %s")
173     IllegalStateException failedToParse(@Cause Throwable cause, String resourceName);
174 
175     /**
176      * Creates an exception indicating a singleton resource already exists.
177      *
178      * @param resourceName the name of the resource.
179      *
180      * @return an {@link OperationFailedException} for the error.
181      */
182     @Message(id = 10382, value = "Add operation failed: singleton %s already exists.")
183     OperationFailedException singletonResourceAlreadyExists(String resourceName);
184 
185     /**
186      * Creates an exception indicating unable to remove an alias from an empty list of aliases.
187      *
188      * @param aliasName the name of the alias.
189      *
190      * @return an {@link OperationFailedException} for the error.
191      */
192     @Message(id = 10383, value = "cannot remove alias % from empty list.")
193     OperationFailedException cannotRemoveAliasFromEmptyList(String aliasName);
194 
195     /**
196      * Creates an exception indicating that an attribute has been deprecated.
197      *
198      * @param attributeName the name of the deprecated attribute
199      * @return an {@link OperationFailedException} for the error
200      */
201     @Message(id = 10384, value = "Attribute '%s' has been deprecated.")
202     OperationFailedException attributeDeprecated(String attributeName);
203 
204     @Message(id = 10385, value = "Attribute 'segments' is an expression and therefore cannot be translated to legacy attribute 'virtual-nodes'. This resource will need to be ignored on that host.")
205     String virtualNodesDoesNotSupportExpressions();
206 
207     /**
208      * Creates an exception indicating the a cache loader cannot be added as one already exists.
209      *
210      * @param existingStoreName the loader which already exists.
211      * @return an {@link OperationFailedException} for the error
212      */
213     @Message(id = 11000, value = "Cache loader cannot be created: cache loader %s is already defined")
214     OperationFailedException cacheLoaderAlreadyDefined(String existingStoreName);
215 
216     /**
217      * Creates an exception indicating an invalid cache loader.
218      *
219      * @param cause the cause of the error.
220      * @param cacheLoaderName the name of the cache loader.
221      *
222      * @return an {@link IllegalArgumentException} for the error.
223      */
224     @Message(id = 11001, value = "%s is not a valid cache loader")
225     IllegalArgumentException invalidCacheLoader(@Cause Throwable cause, String cacheLoaderName);
226 
227     /**
228      * Returns a message for failed operations
229      *
230      * @param message
231      * @return
232      */
233     @Message(id = 11002, value = "Failed to invoke operation: %s")
234     String failedToInvokeOperation(String message);
235 
236     /**
237      * Creates an exception indicating an invalid compatibility marshaller.
238      *
239      * @param cause the cause of the error.
240      * @param cacheLoaderName the name of the marshaller.
241      *
242      * @return an {@link IllegalArgumentException} for the error.
243      */
244     @Message(id = 11003, value = "%s is not a valid marshaller")
245     IllegalArgumentException invalidCompatibilityMarshaller(@Cause Throwable cause, String marshallerClassName);
246 
247    /**
248     * Creates an exception indicating incompatible parameter sizes
249     *
250     * @param firstParameter  the first parameter name
251     * @param secondParameter the second parameter name
252     * @return an {@link IllegalArgumentException} for the error.
253     */
254    @Message(id = 11004, value = "Parameter %s must be the same size as parameter %s")
255    IllegalArgumentException invalidParameterSizes(String firstParameter, String secondParameter);
256 
257    /**
258     * Creates an exception indicating incompatible parameter type
259     *
260     * @param name         the parameter name
261     * @param requiredType the required type of the parameter
262     * @return an {@link IllegalArgumentException} for the error.
263     */
264    @Message(id = 11005, value = "Parameter %s must be of type %s")
265    IllegalArgumentException invalidParameterType(String name, String requiredType);
266 
267    /**
268     * Error message thrown when Subsystem can't instantiate given class.
269     */
270    @Message(id = 11006, value = "Could not instantiate class %s")
271    IllegalStateException unableToInstantiateClass(String className);
272 }

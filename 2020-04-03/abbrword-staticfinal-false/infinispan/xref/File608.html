1   package org.infinispan.scripting.impl;
2   
3   import org.infinispan.Cache;
4   import org.infinispan.commons.util.concurrent.NotifyingFuture;
5   import org.infinispan.distexec.mapreduce.MapReduceTask;
6   import org.infinispan.scripting.impl.ScriptMetadata.MetadataProperties;
7   
8   /**
9    * MapReduceRunner.
10   *
11   * @author Tristan Tarrant
12   * @since 7.2
13   */
14  public class MapReduceRunner implements ScriptRunner {
15     public static final MapReduceRunner INSTANCE = new MapReduceRunner();
16  
17     private MapReduceRunner() {
18     }
19  
20     @Override
21     public <T> NotifyingFuture<T> runScript(ScriptingManagerImpl scriptManager, ScriptMetadata metadata, CacheScriptBindings binding) {
22        MapReduceTask mapReduceTask = new MapReduceTask<>((Cache<?, ?>) binding.get(SystemBindings.CACHE.toString()));
23        mapReduceTask.mappedWith(new MapperScript<>(metadata));
24        String reducerScript = metadata.property(MetadataProperties.REDUCER);
25        if (reducerScript != null) {
26           mapReduceTask.reducedWith(new ReducerScript<>(scriptManager.getScriptMetadata(reducerScript)));
27        }
28        String combinerScript = metadata.property(MetadataProperties.COMBINER);
29        if (combinerScript != null) {
30           mapReduceTask.combinedWith(new ReducerScript<>(scriptManager.getScriptMetadata(combinerScript)));
31        }
32        String collatorScript = metadata.property(MetadataProperties.COLLATOR);
33        if (collatorScript != null) {
34           return (NotifyingFuture<T>) mapReduceTask.executeAsynchronously(new CollatorScript<>(scriptManager.getScriptMetadata(collatorScript), scriptManager));
35        } else {
36           return (NotifyingFuture<T>) mapReduceTask.executeAsynchronously();
37        }
38     }
39  
40  }

23  package org.jboss.as.clustering.infinispan.subsystem;
24  
25  import org.infinispan.util.concurrent.IsolationLevel;
26  import org.jboss.as.controller.AttributeDefinition;
27  import org.jboss.as.controller.PathElement;
28  import org.jboss.as.controller.SimpleAttributeDefinition;
29  import org.jboss.as.controller.SimpleAttributeDefinitionBuilder;
30  import org.jboss.as.controller.client.helpers.MeasurementUnit;
31  import org.jboss.as.controller.operations.validation.EnumValidator;
32  import org.jboss.as.controller.registry.AttributeAccess;
33  import org.jboss.dmr.ModelNode;
34  import org.jboss.dmr.ModelType;
35  
36  /**
37   * Resource description for the addressable resource /subsystem=infinispan/cache-container=X/cache=Y/locking=LOCKING
38   *
39   * @author Richard Achmatowicz (c) 2011 Red Hat Inc.
40   * @author Tristan Tarrant
41   */
42  public class LockingResource extends CacheChildResource {
43  
44      public static final PathElement LOCKING_PATH = PathElement.pathElement(ModelKeys.LOCKING, ModelKeys.LOCKING_NAME);
45  
46      // attributes
47      static final SimpleAttributeDefinition ACQUIRE_TIMEOUT =
48              new SimpleAttributeDefinitionBuilder(ModelKeys.ACQUIRE_TIMEOUT, ModelType.LONG, true)
49                      .setXmlName(Attribute.ACQUIRE_TIMEOUT.getLocalName())
50                      .setMeasurementUnit(MeasurementUnit.MILLISECONDS)
51                      .setAllowExpression(true)
52                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
53                      .setDefaultValue(new ModelNode().set(10000))
54                      .build();
55  
56      static final SimpleAttributeDefinition CONCURRENCY_LEVEL =
57              new SimpleAttributeDefinitionBuilder(ModelKeys.CONCURRENCY_LEVEL, ModelType.INT, true)
58                      .setXmlName(Attribute.CONCURRENCY_LEVEL.getLocalName())
59                      .setAllowExpression(true)
60                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
61                      .setDefaultValue(new ModelNode().set(32))
62                      .build();
63  
64      static final SimpleAttributeDefinition ISOLATION =
65              new SimpleAttributeDefinitionBuilder(ModelKeys.ISOLATION, ModelType.STRING, true)
66                      .setXmlName(Attribute.ISOLATION.getLocalName())
67                      .setAllowExpression(true)
68                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
69                      .setValidator(new EnumValidator<>(IsolationLevel.class, true, false))
70                      .setDefaultValue(new ModelNode().set(IsolationLevel.READ_COMMITTED.name()))
71                      .build();
72  
73      static final SimpleAttributeDefinition STRIPING =
74              new SimpleAttributeDefinitionBuilder(ModelKeys.STRIPING, ModelType.BOOLEAN, true)
75                      .setXmlName(Attribute.STRIPING.getLocalName())
76                      .setAllowExpression(true)
77                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
78                      .setDefaultValue(new ModelNode().set(false))
79                      .build();
80  
81      static final AttributeDefinition[] LOCKING_ATTRIBUTES = { ACQUIRE_TIMEOUT, CONCURRENCY_LEVEL, ISOLATION, STRIPING };
82  
83      public LockingResource(CacheResource cacheResource) {
84          super(LOCKING_PATH, ModelKeys.LOCKING, cacheResource, LOCKING_ATTRIBUTES);
85      }
86  }

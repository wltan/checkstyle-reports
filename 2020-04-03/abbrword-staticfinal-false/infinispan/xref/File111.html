35     /**
36      * Returns a set of Methods that contain the given method annotation.  This includes all public, protected, package
37      * and private methods, as well as those of superclasses.  Note that this does *not* include overridden methods.
38      *
39      * @param c              class to inspect
40      * @param annotationType the type of annotation to look for
41      * @return List of Method objects that require injection.
42      */
43     public static List<Method> getAllMethods(Class<?> c, Class<? extends Annotation> annotationType) {
44        List<Method> annotated = new LinkedList<Method>();
45        inspectRecursively(c, annotated, annotationType);
46        return annotated;
47     }

49     /**
50      * Returns a set of Methods that contain the given method annotation.  This includes all public, protected, package
51      * and private methods, but not those of superclasses and interfaces.
52      *
53      * @param c              class to inspect
54      * @param annotationType the type of annotation to look for
55      * @return List of Method objects that require injection.
56      */
57     public static List<Method> getAllMethodsShallow(Class<?> c, Class<? extends Annotation> annotationType) {
58        List<Method> annotated = new LinkedList<Method>();
59        for (Method m : c.getDeclaredMethods()) {
60           if (m.isAnnotationPresent(annotationType))
61              annotated.add(m);
62        }
63  
64        return annotated;
65     }

67     private static void getAnnotatedFieldHelper(List<Field> list, Class<?> c, Class<? extends Annotation> annotationType) {
68        Field[] declaredFields = c.getDeclaredFields();
69        for (Field field : declaredFields) {
70           if (field.isAnnotationPresent(annotationType)) {
71              list.add(field);
72           }
73        }
74     }

76     public static List<Field> getAnnotatedFields(Class<?> c, Class<? extends Annotation> annotationType) {
77        List<Field> fields = new ArrayList<Field>(4);
78        // Class could be null in the case of an interface
79        for (;c != null && !c.equals(Object.class); c = c.getSuperclass()) {
80           getAnnotatedFieldHelper(fields, c, annotationType);
81        }
82        return fields;
83     }

85     public static Method findMethod(Class<?> type, String methodName) {
86        try {
87           return type.getDeclaredMethod(methodName);
88        } catch (NoSuchMethodException e) {
89           if (type.equals(Object.class) || type.isInterface()) {
90              throw new CacheException(e);
91           }
92           return findMethod(type.getSuperclass(), methodName);
93        }
94     }

96     public static Method findMethod(Class<?> type, String methodName, Class<?>[] parameters) throws ClassNotFoundException {
97        try {
98           return type.getDeclaredMethod(methodName, parameters);
99        } catch (NoSuchMethodException e) {
100          if (type.equals(Object.class) || type.isInterface()) {
101             throw new CacheException(e);
102          }
103          return findMethod(type.getSuperclass(), methodName, parameters);
104       }
105    }

107    /**
108     * Inspects a class and its superclasses (all the way to {@link Object} for method instances that contain a given
109     * annotation. This even identifies private, package and protected methods, not just public ones.
110     *
111     * @param c
112     * @param s
113     * @param annotationType
114     */
115    private static void inspectRecursively(Class<?> c, List<Method> s, Class<? extends Annotation> annotationType) {
116 
117       for (Method m : c.getDeclaredMethods()) {
118          // don't bother if this method has already been overridden by a subclass
119          if (notFound(m, s) && m.isAnnotationPresent(annotationType)) {
120             s.add(m);
121          }
122       }
123 
124       if (!c.equals(Object.class)) {
125          if (!c.isInterface()) {
126             inspectRecursively(c.getSuperclass(), s, annotationType);
127             for (Class<?> ifc : c.getInterfaces()) inspectRecursively(ifc, s, annotationType);
128          }
129       }
130    }

132    /**
133     * Tests whether a method has already been found, i.e., overridden.
134     *
135     * @param m method to inspect
136     * @param s collection of methods found
137     * @return true a method with the same signature already exists.
138     */
139    private static boolean notFound(Method m, Collection<Method> s) {
140       for (Method found : s) {
141          if (m.getName().equals(found.getName()) &&
142                Arrays.equals(m.getParameterTypes(), found.getParameterTypes()))
143             return false;
144       }
145       return true;
146    }

148    private static Field findFieldRecursively(Class<?> c, String fieldName) {
149       Field f = null;
150       try {
151          f = c.getDeclaredField(fieldName);
152       } catch (NoSuchFieldException e) {
153          if (!c.equals(Object.class)) f = findFieldRecursively(c.getSuperclass(), fieldName);
154       }
155       return f;
156    }

158    /**
159     * Invokes a method using reflection, in an accessible manner (by using {@link Method#setAccessible(boolean)}
160     *
161     * @param instance   instance on which to execute the method
162     * @param method     method to execute
163     * @param parameters parameters
164     */
165    public static Object invokeAccessibly(Object instance, Method method, Object[] parameters) {
166       try {
167          method.setAccessible(true);
168          return method.invoke(instance, parameters);
169       } catch (InvocationTargetException e) {
170          Throwable cause = e.getCause() != null ? e.getCause() : e;
171          throw new CacheException("Unable to invoke method " + method + " on object of type " + (instance == null ? "null" : instance.getClass().getSimpleName()) +
172                                         (parameters != null ? " with parameters " + Arrays.asList(parameters) : ""), cause);
173       } catch (Exception e) {
174          throw new CacheException("Unable to invoke method " + method + " on object of type " + (instance == null ? "null" : instance.getClass().getSimpleName()) +
175                (parameters != null ? " with parameters " + Arrays.asList(parameters) : ""), e);
176       }
177    }

179    public static Method findGetterForField(Class<?> c, String fieldName) {
180       Method retval = findGetterForFieldUsingReflection(c, fieldName);
181       if (retval == null) {
182          if (!c.equals(Object.class)) {
183             if (!c.isInterface()) {
184                retval = findGetterForField(c.getSuperclass(), fieldName);
185                if (retval == null) {
186                   for (Class<?> ifc : c.getInterfaces()) {
187                      retval = findGetterForField(ifc, fieldName);
188                      if (retval != null) break;
189                   }
190                }
191             }
192          }
193       }
194       return retval;
195    }

197    private static Method findGetterForFieldUsingReflection(Class<?> c, String fieldName) {
198       for (Method m : c.getDeclaredMethods()) {
199          String name = m.getName();
200          String s = null;
201          if (name.startsWith("get")) {
202             s = name.substring(3);
203          } else if (name.startsWith("is")) {
204             s = name.substring(2);
205          }
206 
207          if (s != null && s.equalsIgnoreCase(fieldName)) {
208             return m;
209          }
210       }
211       return null;
212    }

229    public static String extractFieldName(String setterOrGetter) {
230       String field = null;
231       if (setterOrGetter.startsWith("set") || setterOrGetter.startsWith("get"))
232          field = setterOrGetter.substring(3);
233       else if (setterOrGetter.startsWith("is"))
234          field = setterOrGetter.substring(2);
235 
236       if (field != null && field.length() > 1) {
237          StringBuilder sb = new StringBuilder();
238          sb.append(Character.toLowerCase(field.charAt(0)));
239          if (field.length() > 2) sb.append(field.substring(1));
240          return sb.toString();
241       }
242       return null;
243    }

246    /**
247     * Retrieves the value of a field of an object instance via reflection
248     *
249     * @param instance  to inspect
250     * @param fieldName name of field to retrieve
251     * @return a value
252     */
253    public static Object getValue(Object instance, String fieldName) {
254       Field f = findFieldRecursively(instance.getClass(), fieldName);
255       if (f == null) throw new CacheException("Could not find field named '" + fieldName + "' on instance " + instance);
256       try {
257          f.setAccessible(true);
258          return f.get(instance);
259       } catch (IllegalAccessException iae) {
260          throw new CacheException("Cannot access field " + f, iae);
261       }
262    }

264    /**
265     * Inspects the class passed in for the class level annotation specified.  If the annotation is not available, this
266     * method recursively inspects superclasses and interfaces until it finds the required annotation.
267     * <p/>
268     * Returns null if the annotation cannot be found.
269     *
270     * @param clazz class to inspect
271     * @param ann   annotation to search for.  Must be a class-level annotation.
272     * @return the annotation instance, or null
273     */
274    @SuppressWarnings("unchecked")
275    public static <T extends Annotation> T getAnnotation(Class<?> clazz, Class<T> ann) {
276       while (true) {
277          // first check class
278          T a = clazz.getAnnotation(ann);
279          if (a != null) return a;
280 
281          // check interfaces
282          if (!clazz.isInterface()) {
283             Class<?>[] interfaces = clazz.getInterfaces();
284             for (Class<?> inter : interfaces) {
285                a = getAnnotation(inter, ann);
286                if (a != null) return a;
287             }
288          }
289 
290          // check superclasses
291          Class<?> superclass = clazz.getSuperclass();
292          if (superclass == null) return null; // no where else to look
293          clazz = superclass;
294       }
295    }

298     * Tests whether an annotation is present on a class.  The order tested is: <ul> <li>The class itself</li> <li>All
299     * implemented interfaces</li> <li>Any superclasses</li> </ul>
300     *
301     * @param clazz      class to test
302     * @param annotation annotation to look for
303     * @return true if the annotation is found, false otherwise
304     */
305    public static boolean isAnnotationPresent(Class<?> clazz, Class<? extends Annotation> annotation) {
306       return getAnnotation(clazz, annotation) != null;
307    }

309    public static Class<?>[] toClassArray(String[] typeList, ClassLoader classLoader) throws ClassNotFoundException {
310       if (typeList == null) return EMPTY_CLASS_ARRAY;
311       Class<?>[] retval = new Class[typeList.length];
312       int i = 0;
313       for (String s : typeList) retval[i++] = getClassForName(s, classLoader);
314       return retval;
315    }

317    public static Class<?> getClassForName(String name, ClassLoader cl) throws ClassNotFoundException {
318       try {
319          return Util.loadClassStrict(name, cl);
320       } catch (ClassNotFoundException cnfe) {
321          // Could be a primitive - let's check
322          for (Class<?> primitive : primitives) if (name.equals(primitive.getName())) return primitive;
323          for (Class<?> primitive : primitiveArrays) if (name.equals(primitive.getName())) return primitive;
324       }
325       throw new ClassNotFoundException("Class " + name + " cannot be found");
326    }

329       if (classes == null)
330          return EMPTY_STRING_ARRAY;

332          String[] classNames = new String[classes.length];
333          for (int i=0; i<classes.length; i++) classNames[i] = classes[i].getName();
334          return classNames;

338    public static Field getField(String fieldName, Class<?> objectClass) {
339       try {
340          return objectClass.getDeclaredField(fieldName);
341       } catch (NoSuchFieldException e) {
342          if (!objectClass.equals(Object.class)) {
343             return getField(fieldName, objectClass.getSuperclass());
344          } else {
345             return null;
346          }
347       }

350    public static <T> T unwrap(Object obj, Class<T> clazz) {
351       if (clazz != null && clazz.isAssignableFrom(obj.getClass()))
352          return clazz.cast(obj);
353 
354       throw log.unableToUnwrap(obj, clazz);
355    }
356 
357    public static <T> T unwrapAny(Class<T> clazz, Object... objs) {
358       if (clazz != null) {
359          for (Object o : objs) {
360             if (clazz.isAssignableFrom(o.getClass()))
361                return clazz.cast(o);
362          }
363       }
364 
365       throw log.unableToUnwrapAny(Arrays.toString(objs), clazz);
366    }

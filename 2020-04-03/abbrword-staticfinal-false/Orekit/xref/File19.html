43  /** This class represents the gravitational field of a celestial body.
44   * <p>The algorithm implemented in this class has been designed by
45   * Leland E. Cunningham (Lockheed Missiles and Space Company, Sunnyvale
46   * and Astronomy Department University of California, Berkeley) in
47   * his 1969 paper: <em>On the computation of the spherical harmonic
48   * terms needed during the numerical integration of the orbital motion
49   * of an artificial satellite</em> (Celestial Mechanics 2, 1970).</p>
50   * <p>
51   * Note that this class can often not be used for high degrees (say
52   * above 90) as most modern gravity fields are provided as normalized
53   * coefficients and the un-normalization process to convert these
54   * coefficients underflows at degree and order 89. This class also
55   * does not provide analytical partial derivatives (it uses finite differences
56   * to compute them) and is much slower than {@link HolmesFeatherstoneAttractionModel}
57   * (even when no derivatives are computed). For all these reasons,
58   * it is recommended to use the {@link HolmesFeatherstoneAttractionModel
59   * Holmes-Featherstone model} rather than this class.
60   * </p>
61   * <p>
62   * This class uses finite differences to compute derivatives and the steps for
63   * finite differences are initialized in the {@link
64   * #CunninghamAttractionModel(Frame, UnnormalizedSphericalHarmonicsProvider,
65   * double) constructor}.
66   * </p>
67   *
68   * @see HolmesFeatherstoneAttractionModel
69   * @author Fabien Maussion
70   * @author Luc Maisonobe
71   * @author V&eacute;ronique Pommier-Maurussane
72   */
73  
74  public class CunninghamAttractionModel extends AbstractForceModel implements TideSystemProvider {
75  
76      /** Central attraction scaling factor.
77       * <p>
78       * We use a power of 2 to avoid numeric noise introduction
79       * in the multiplications/divisions sequences.
80       * </p>
81       */
82      private static final double MU_SCALE = FastMath.scalb(1.0, 32);
83  
84      /** Drivers for force model parameters. */
85      private final ParameterDriver[] parametersDrivers;
86  
87      /** Provider for the spherical harmonics. */
88      private final UnnormalizedSphericalHarmonicsProvider provider;
89  
90      /** Central attraction coefficient. */
91      private double mu;
92  
93      /** Rotating body. */
94      private final Frame bodyFrame;
95  
96      /** Helper class computing acceleration derivatives. */
97      private Jacobianizer jacobianizer;
98  
99     /** Creates a new instance.
100    * @param centralBodyFrame rotating body frame
101    * @param provider provider for spherical harmonics
102    * @param hPosition step used for finite difference computation
103    * @since 6.0
104    */
105     public CunninghamAttractionModel(final Frame centralBodyFrame,
106                                      final UnnormalizedSphericalHarmonicsProvider provider,
107                                      final double hPosition) {
108 
109         this.parametersDrivers = new ParameterDriver[1];
110         try {
111             parametersDrivers[0] = new ParameterDriver(NewtonianAttraction.CENTRAL_ATTRACTION_COEFFICIENT,
112                                                        provider.getMu(), MU_SCALE, 0.0, Double.POSITIVE_INFINITY);
113             parametersDrivers[0].addObserver(new ParameterObserver() {
114                 /** {@inheritDoc} */
115                 @Override
116                 public void valueChanged(final double previousValue, final ParameterDriver driver) {
117                     CunninghamAttractionModel.this.mu = driver.getValue();
118                 }
119             });
120         } catch (OrekitException oe) {
121             // this should never occur as valueChanged above never throws an exception
122             throw new OrekitInternalError(oe);
123         };
124 
125         this.provider     = provider;
126         this.mu           = provider.getMu();
127         this.bodyFrame    = centralBodyFrame;
128         this.jacobianizer = new Jacobianizer(this, mu, hPosition);
129 
130     }
131 
132     /** {@inheritDoc} */
133     public TideSystem getTideSystem() {
134         return provider.getTideSystem();
135     }
136 
137     /** {@inheritDoc} */
138     public void addContribution(final SpacecraftState s, final TimeDerivativesEquations adder)
139         throws OrekitException {
140 
141         // get the position in body frame
142         final AbsoluteDate date = s.getDate();
143         final UnnormalizedSphericalHarmonics harmonics = provider.onDate(date);
144         final Transform fromBodyFrame = bodyFrame.getTransformTo(s.getFrame(), date);
145         final Transform toBodyFrame   = fromBodyFrame.getInverse();
146         final Vector3D relative = toBodyFrame.transformPosition(s.getPVCoordinates().getPosition());
147 
148         final double x = relative.getX();
149         final double y = relative.getY();
150         final double z = relative.getZ();
151 
152         final double x2 = x * x;
153         final double y2 = y * y;
154         final double z2 = z * z;
155         final double r2 = x2 + y2 + z2;
156         final double r = FastMath.sqrt(r2);
157         final double equatorialRadius = provider.getAe();
158         if (r <= equatorialRadius) {
159             throw new OrekitException(OrekitMessages.TRAJECTORY_INSIDE_BRILLOUIN_SPHERE, r);
160         }
161 
162         // define some intermediate variables
163         final double onR2 = 1 / r2;
164         final double onR3 = onR2 / r;
165         final double rEqOnR2  = equatorialRadius / r2;
166         final double rEqOnR4  = rEqOnR2 / r2;
167         final double rEq2OnR2 = equatorialRadius * rEqOnR2;
168 
169         double cmx   = -x * rEqOnR2;
170         double cmy   = -y * rEqOnR2;
171         double cmz   = -z * rEqOnR2;
172 
173         final double dx   = -2 * cmx;
174         final double dy   = -2 * cmy;
175         final double dz   = -2 * cmz;
176 
177         // intermediate variables gradients
178         // since dcy/dx = dcx/dy, dcz/dx = dcx/dz and dcz/dy = dcy/dz,
179         // we reuse the existing variables
180 
181         double dcmxdx = (x2 - y2 - z2) * rEqOnR4;
182         double dcmxdy =  dx * y * onR2;
183         double dcmxdz =  dx * z * onR2;
184         double dcmydy = (y2 - x2 - z2) * rEqOnR4;
185         double dcmydz =  dy * z * onR2;
186         double dcmzdz = (z2 - x2 - y2) * rEqOnR4;
187 
188         final double ddxdx = -2 * dcmxdx;
189         final double ddxdy = -2 * dcmxdy;
190         final double ddxdz = -2 * dcmxdz;
191         final double ddydy = -2 * dcmydy;
192         final double ddydz = -2 * dcmydz;
193         final double ddzdz = -2 * dcmzdz;
194 
195         final double donr2dx = -dx * rEqOnR2;
196         final double donr2dy = -dy * rEqOnR2;
197         final double donr2dz = -dz * rEqOnR2;
198 
199         // potential coefficients (4 per matrix)
200         double vrn  = 0.0;
201         double vin  = 0.0;
202         double vrd  = 1.0 / (equatorialRadius * r);
203         double vid  = 0.0;
204         double vrn1 = 0.0;
205         double vin1 = 0.0;
206         double vrn2 = 0.0;
207         double vin2 = 0.0;
208 
209         // gradient coefficients (4 per matrix)
210         double gradXVrn  = 0.0;
211         double gradXVin  = 0.0;
212         double gradXVrd  = -x * onR3 / equatorialRadius;
213         double gradXVid  = 0.0;
214         double gradXVrn1 = 0.0;
215         double gradXVin1 = 0.0;
216         double gradXVrn2 = 0.0;
217         double gradXVin2 = 0.0;
218 
219         double gradYVrn  = 0.0;
220         double gradYVin  = 0.0;
221         double gradYVrd  = -y * onR3 / equatorialRadius;
222         double gradYVid  = 0.0;
223         double gradYVrn1 = 0.0;
224         double gradYVin1 = 0.0;
225         double gradYVrn2 = 0.0;
226         double gradYVin2 = 0.0;
227 
228         double gradZVrn  = 0.0;
229         double gradZVin  = 0.0;
230         double gradZVrd  = -z * onR3 / equatorialRadius;
231         double gradZVid  = 0.0;
232         double gradZVrn1 = 0.0;
233         double gradZVin1 = 0.0;
234         double gradZVrn2 = 0.0;
235         double gradZVin2 = 0.0;
236 
237         // acceleration coefficients
238         double vdX = 0.0;
239         double vdY = 0.0;
240         double vdZ = 0.0;
241 
242         // start calculating
243         for (int m = 0; m <= provider.getMaxOrder(); m++) {
244 
245             double cx = cmx;
246             double cy = cmy;
247             double cz = cmz;
248 
249             double dcxdx = dcmxdx;
250             double dcxdy = dcmxdy;
251             double dcxdz = dcmxdz;
252             double dcydy = dcmydy;
253             double dcydz = dcmydz;
254             double dczdz = dcmzdz;
255 
256             for (int n = m; n <= provider.getMaxDegree(); n++) {
257 
258                 if (n == m) {
259                     // calculate the first element of the next column
260 
261                     vrn      = equatorialRadius * vrd;
262                     vin      = equatorialRadius * vid;
263 
264                     gradXVrn = equatorialRadius * gradXVrd;
265                     gradXVin = equatorialRadius * gradXVid;
266                     gradYVrn = equatorialRadius * gradYVrd;
267                     gradYVin = equatorialRadius * gradYVid;
268                     gradZVrn = equatorialRadius * gradZVrd;
269                     gradZVin = equatorialRadius * gradZVid;
270 
271                     final double tmpGradXVrd = (cx + dx) * gradXVrd - (cy + dy) * gradXVid + (dcxdx + ddxdx) * vrd - (dcxdy + ddxdy) * vid;
272                     gradXVid = (cy + dy) * gradXVrd + (cx + dx) * gradXVid + (dcxdy + ddxdy) * vrd + (dcxdx + ddxdx) * vid;
273                     gradXVrd = tmpGradXVrd;
274 
275                     final double tmpGradYVrd = (cx + dx) * gradYVrd - (cy + dy) * gradYVid + (dcxdy + ddxdy) * vrd - (dcydy + ddydy) * vid;
276                     gradYVid = (cy + dy) * gradYVrd + (cx + dx) * gradYVid + (dcydy + ddydy) * vrd + (dcxdy + ddxdy) * vid;
277                     gradYVrd = tmpGradYVrd;
278 
279                     final double tmpGradZVrd = (cx + dx) * gradZVrd - (cy + dy) * gradZVid + (dcxdz + ddxdz) * vrd - (dcydz + ddydz) * vid;
280                     gradZVid = (cy + dy) * gradZVrd + (cx + dx) * gradZVid + (dcydz + ddydz) * vrd + (dcxdz + ddxdz) * vid;
281                     gradZVrd = tmpGradZVrd;
282 
283                     final double tmpVrd = (cx + dx) * vrd - (cy + dy) * vid;
284                     vid = (cy + dy) * vrd + (cx + dx) * vid;
285                     vrd = tmpVrd;
286 
287                 } else if (n == m + 1) {
288                     // calculate the second element of the column
289                     vrn = cz * vrn1;
290                     vin = cz * vin1;
291 
292                     gradXVrn = cz * gradXVrn1 + dcxdz * vrn1;
293                     gradXVin = cz * gradXVin1 + dcxdz * vin1;
294 
295                     gradYVrn = cz * gradYVrn1 + dcydz * vrn1;
296                     gradYVin = cz * gradYVin1 + dcydz * vin1;
297 
298                     gradZVrn = cz * gradZVrn1 + dczdz * vrn1;
299                     gradZVin = cz * gradZVin1 + dczdz * vin1;
300 
301                 } else {
302                     // calculate the other elements of the column
303                     final double inv   = 1.0 / (n - m);
304                     final double coeff = n + m - 1.0;
305 
306                     vrn = (cz * vrn1 - coeff * rEq2OnR2 * vrn2) * inv;
307                     vin = (cz * vin1 - coeff * rEq2OnR2 * vin2) * inv;
308 
309                     gradXVrn = (cz * gradXVrn1 - coeff * rEq2OnR2 * gradXVrn2 + dcxdz * vrn1 - coeff * donr2dx * vrn2) * inv;
310                     gradXVin = (cz * gradXVin1 - coeff * rEq2OnR2 * gradXVin2 + dcxdz * vin1 - coeff * donr2dx * vin2) * inv;
311                     gradYVrn = (cz * gradYVrn1 - coeff * rEq2OnR2 * gradYVrn2 + dcydz * vrn1 - coeff * donr2dy * vrn2) * inv;
312                     gradYVin = (cz * gradYVin1 - coeff * rEq2OnR2 * gradYVin2 + dcydz * vin1 - coeff * donr2dy * vin2) * inv;
313                     gradZVrn = (cz * gradZVrn1 - coeff * rEq2OnR2 * gradZVrn2 + dczdz * vrn1 - coeff * donr2dz * vrn2) * inv;
314                     gradZVin = (cz * gradZVin1 - coeff * rEq2OnR2 * gradZVin2 + dczdz * vin1 - coeff * donr2dz * vin2) * inv;
315                 }
316 
317                 // increment variables
318                 cx += dx;
319                 cy += dy;
320                 cz += dz;
321 
322                 dcxdx += ddxdx;
323                 dcxdy += ddxdy;
324                 dcxdz += ddxdz;
325                 dcydy += ddydy;
326                 dcydz += ddydz;
327                 dczdz += ddzdz;
328 
329                 vrn2 = vrn1;
330                 vin2 = vin1;
331                 gradXVrn2 = gradXVrn1;
332                 gradXVin2 = gradXVin1;
333                 gradYVrn2 = gradYVrn1;
334                 gradYVin2 = gradYVin1;
335                 gradZVrn2 = gradZVrn1;
336                 gradZVin2 = gradZVin1;
337 
338                 vrn1 = vrn;
339                 vin1 = vin;
340                 gradXVrn1 = gradXVrn;
341                 gradXVin1 = gradXVin;
342                 gradYVrn1 = gradYVrn;
343                 gradYVin1 = gradYVin;
344                 gradZVrn1 = gradZVrn;
345                 gradZVin1 = gradZVin;
346 
347                 // compute the acceleration due to the Cnm and Snm coefficients
348                 // ignoring the central attraction
349                 if (n > 0) {
350                     final double cnm = harmonics.getUnnormalizedCnm(n, m);
351                     final double snm = harmonics.getUnnormalizedSnm(n, m);
352                     vdX += cnm * gradXVrn + snm * gradXVin;
353                     vdY += cnm * gradYVrn + snm * gradYVin;
354                     vdZ += cnm * gradZVrn + snm * gradZVin;
355                 }
356 
357             }
358 
359             // increment variables
360             cmx += dx;
361             cmy += dy;
362             cmz += dz;
363 
364             dcmxdx += ddxdx;
365             dcmxdy += ddxdy;
366             dcmxdz += ddxdz;
367             dcmydy += ddydy;
368             dcmydz += ddydz;
369             dcmzdz += ddzdz;
370 
371         }
372 
373         // compute acceleration in inertial frame
374         final Vector3D acceleration =
375             fromBodyFrame.transformVector(new Vector3D(mu * vdX, mu * vdY, mu * vdZ));
376         adder.addXYZAcceleration(acceleration.getX(), acceleration.getY(), acceleration.getZ());
377 
378     }

382                                                                       final FieldVector3D<DerivativeStructure> position,
383                                                                       final FieldVector3D<DerivativeStructure> velocity,
384                                                                       final FieldRotation<DerivativeStructure> rotation,
385                                                                       final DerivativeStructure mass)
386         throws OrekitException {
387         return jacobianizer.accelerationDerivatives(date, frame, position, velocity, rotation, mass);
388     }

426 	protected void send(Object object) {
427 		initializeIfNecessary();
428 		Assert.notNull(object, "request must not be null");
429 		MessageChannel channel = getRequestChannel();
430 		Assert.state(channel != null,
431 				"send is not supported, because no request channel has been configured");
432 		try {
433 			if (this.countsEnabled) {
434 				this.messageCount.incrementAndGet();
435 			}
436 			this.messagingTemplate.convertAndSend(channel, object, this.historyWritingPostProcessor);
437 		}
438 		catch (Exception e) {
439 			MessageChannel errorChan = getErrorChannel();
440 			if (errorChan != null) {
441 				this.messagingTemplate.send(errorChan, new ErrorMessage(e));
442 			}
443 			else {
444 				rethrow(e, "failed to send message");
445 			}
446 		}
447 	}

496 	@Nullable // NOSONAR
497 	private Object doSendAndReceive(Object object, boolean shouldConvert) {
498 		initializeIfNecessary();
499 		Assert.notNull(object, "request must not be null");
500 		MessageChannel channel = getRequestChannel();
501 		if (channel == null) {
502 			throw new MessagingException("No request channel available. Cannot send request message.");
503 		}
504 
505 		registerReplyMessageCorrelatorIfNecessary();
506 
507 		Object reply;
508 		Message<?> requestMessage = null;
509 		try {
510 			if (this.countsEnabled) {
511 				this.messageCount.incrementAndGet();
512 			}
513 			if (shouldConvert) {
514 				reply = this.messagingTemplate.convertSendAndReceive(channel, object, Object.class,
515 						this.historyWritingPostProcessor);
516 			}
517 			else {
518 				requestMessage = (object instanceof Message<?>)
519 						? (Message<?>) object : this.requestMapper.toMessage(object);
520 				Assert.state(requestMessage != null, () -> "request mapper resulted in no message for " + object);
521 				requestMessage = this.historyWritingPostProcessor.postProcessMessage(requestMessage);
522 				reply = this.messagingTemplate.sendAndReceive(channel, requestMessage);
523 			}
524 
525 			if (reply == null && this.errorOnTimeout) {
526 				throwMessageTimeoutException(object, "No reply received within timeout");
527 			}
528 		}
529 		catch (Exception ex) {
530 			if (logger.isDebugEnabled()) {
531 				logger.debug("failure occurred in gateway sendAndReceive: " + ex.getMessage());
532 			}
533 			reply = ex;
534 		}
535 
536 		if (reply instanceof Throwable || reply instanceof ErrorMessage) {
537 			Throwable error =
538 					reply instanceof ErrorMessage
539 							? ((ErrorMessage) reply).getPayload()
540 							: (Throwable) reply;
541 			return handleSendAndReceiveError(object, requestMessage, error, shouldConvert);
542 		}
543 		return reply;
544 	}
545 
546 	@Nullable
547 	private Object handleSendAndReceiveError(Object object, @Nullable Message<?> requestMessage, Throwable error,
548 			boolean shouldConvert) {
549 
550 		MessageChannel errorChan = getErrorChannel();
551 		if (errorChan != null) {
552 			ErrorMessage errorMessage = buildErrorMessage(requestMessage, error);
553 			Message<?> errorFlowReply = sendErrorMessageAndReceive(errorChan, errorMessage);
554 			if (errorFlowReply == null && this.errorOnTimeout) {
555 				throwMessageTimeoutException(object, "No reply received from error channel within timeout");
556 				return null; // unreachable
557 			}
558 			else {
559 				return shouldConvert && errorFlowReply != null
560 						? errorFlowReply.getPayload()
561 						: errorFlowReply;
562 			}
563 		}
564 		else {
565 			Throwable errorToReThrow = error;
566 			if (error instanceof MessagingException &&
567 					requestMessage != null && requestMessage.getHeaders().getErrorChannel() != null) {
568 				// We are in nested flow where upstream expects errors in its own errorChannel header.
569 				errorToReThrow = new MessageHandlingException(requestMessage, error);
570 			}
571 			rethrow(errorToReThrow, "gateway received checked Exception");
572 			return null; // unreachable
573 		}
574 	}

576 	@Nullable
577 	private Message<?> sendErrorMessageAndReceive(MessageChannel errorChan, ErrorMessage errorMessage) {
578 		Message<?> errorFlowReply;
579 		try {
580 			errorFlowReply = this.messagingTemplate.sendAndReceive(errorChan, errorMessage);
581 		}
582 		catch (Exception errorFlowFailure) {
583 			throw new MessagingException(errorMessage, "failure occurred in error-handling flow",
584 					errorFlowFailure);
585 		}
586 		if (errorFlowReply != null && errorFlowReply.getPayload() instanceof Throwable) {
587 			rethrow((Throwable) errorFlowReply.getPayload(), "error flow returned an Error Message");
588 		}
589 		return errorFlowReply;
590 	}

601 	protected Mono<Message<?>> sendAndReceiveMessageReactive(Object object) {
602 		initializeIfNecessary();
603 		Assert.notNull(object, "request must not be null");
604 		MessageChannel channel = getRequestChannel();
605 		if (channel == null) {
606 			throw new MessagingException("No request channel available. Cannot send request message.");
607 		}
608 
609 		registerReplyMessageCorrelatorIfNecessary();
610 
611 		return doSendAndReceiveMessageReactive(channel, object, false);
612 	}
613 
614 	private Mono<Message<?>> doSendAndReceiveMessageReactive(MessageChannel requestChannel, Object object,
615 			boolean error) {
616 
617 		return Mono.defer(() -> {
618 			Message<?> message;
619 			try {
620 				message = object instanceof Message<?>
621 						? (Message<?>) object
622 						: this.requestMapper.toMessage(object);
623 
624 				message = this.historyWritingPostProcessor.postProcessMessage(message);
625 
626 			}
627 			catch (Exception e) {
628 				throw new MessageMappingException("Cannot map to message: " + object, e);
629 			}
630 
631 			Object originalReplyChannelHeader = message.getHeaders().getReplyChannel();
632 			Object originalErrorChannelHeader = message.getHeaders().getErrorChannel();
633 
634 			MonoReplyChannel replyChan = new MonoReplyChannel();
635 
636 			Message<?> requestMessage = MutableMessageBuilder.fromMessage(message)
637 					.setReplyChannel(replyChan)
638 					.setHeader(this.messagingTemplate.getSendTimeoutHeader(), null)
639 					.setHeader(this.messagingTemplate.getReceiveTimeoutHeader(), null)
640 					.setErrorChannel(replyChan)
641 					.build();
642 
643 			sendMessageForReactiveFlow(requestChannel, requestMessage);
644 
645 			return buildReplyMono(requestMessage, replyChan.replyMono, error, originalReplyChannelHeader,
646 					originalErrorChannelHeader);
647 		});
648 	}
649 
650 	private void sendMessageForReactiveFlow(MessageChannel requestChannel, Message<?> requestMessage) {
651 		if (requestChannel instanceof ReactiveStreamsSubscribableChannel) {
652 			((ReactiveStreamsSubscribableChannel) requestChannel)
653 					.subscribeTo(Mono.just(requestMessage));
654 		}
655 		else {
656 			long sendTimeout = sendTimeout(requestMessage);
657 
658 			boolean sent =
659 					sendTimeout >= 0
660 							? requestChannel.send(requestMessage, sendTimeout)
661 							: requestChannel.send(requestMessage);
662 
663 			if (!sent) {
664 				throw new MessageDeliveryException(requestMessage,
665 						"Failed to send message to channel '" + requestChannel +
666 								"' within timeout: " + sendTimeout);
667 			}
668 		}
669 	}

671 	private Mono<Message<?>> buildReplyMono(Message<?> requestMessage, Mono<Message<?>> reply, boolean error,
672 			@Nullable Object originalReplyChannelHeader, @Nullable Object originalErrorChannelHeader) {
673 
674 		return reply
675 				.doOnSubscribe(s -> {
676 					if (!error && this.countsEnabled) {
677 						this.messageCount.incrementAndGet();
678 					}
679 				})
680 				.<Message<?>>map(replyMessage -> {
681 					if (!error && replyMessage instanceof ErrorMessage) {
682 						ErrorMessage em = (ErrorMessage) replyMessage;
683 						if (em.getPayload() instanceof MessagingException) {
684 							throw (MessagingException) em.getPayload();
685 						}
686 						else {
687 							throw new MessagingException(requestMessage, em.getPayload());
688 						}
689 					}
690 					else {
691 						return MessageBuilder.fromMessage(replyMessage)
692 								.setHeader(MessageHeaders.REPLY_CHANNEL, originalReplyChannelHeader)
693 								.setHeader(MessageHeaders.ERROR_CHANNEL, originalErrorChannelHeader)
694 								.build();
695 					}
696 				})
697 				.onErrorResume(t -> error ? Mono.error(t) : handleSendError(requestMessage, t));
698 	}
699 
700 	private Mono<Message<?>> handleSendError(Message<?> requestMessage, Throwable exception) {
701 		if (logger.isDebugEnabled()) {
702 			logger.debug("failure occurred in gateway sendAndReceiveReactive: " + exception.getMessage());
703 		}
704 		MessageChannel channel = getErrorChannel();
705 		if (channel != null) {
706 			ErrorMessage errorMessage = buildErrorMessage(requestMessage, exception);
707 			try {
708 				return doSendAndReceiveMessageReactive(channel, errorMessage, true);
709 			}
710 			catch (Exception errorFlowFailure) {
711 				throw new MessagingException(errorMessage, "failure occurred in error-handling flow",
712 						errorFlowFailure);
713 			}
714 		}
715 		else {
716 			// no errorChannel so we'll propagate
717 			throw wrapExceptionIfNecessary(exception, "gateway received checked Exception");
718 		}
719 	}

721 	private long sendTimeout(Message<?> requestMessage) {
722 		Long sendTimeout = headerToLong(requestMessage.getHeaders()
723 				.get(this.messagingTemplate.getSendTimeoutHeader()));
724 		return (sendTimeout != null ? sendTimeout : this.messagingTemplate.getSendTimeout());
725 	}

727 	@Nullable
728 	private Long headerToLong(@Nullable Object headerValue) {
729 		if (headerValue instanceof Number) {
730 			return ((Number) headerValue).longValue();
731 		}
732 		else if (headerValue instanceof String) {
733 			return Long.parseLong((String) headerValue);
734 		}
735 		else {
736 			return null;
737 		}
738 	}

777 	protected void registerReplyMessageCorrelatorIfNecessary() {
778 		MessageChannel replyChan = getReplyChannel();
779 		if (replyChan != null && this.replyMessageCorrelator == null) {
780 			synchronized (this.replyMessageCorrelatorMonitor) {
781 				if (this.replyMessageCorrelator != null) {
782 					return;
783 				}
784 				AbstractEndpoint correlator;
785 				BridgeHandler handler = new BridgeHandler();
786 				BeanFactory beanFactory = getBeanFactory();
787 				if (beanFactory != null) {
788 					handler.setBeanFactory(beanFactory);
789 				}
790 				handler.afterPropertiesSet();
791 				if (replyChan instanceof SubscribableChannel) {
792 					correlator = new EventDrivenConsumer((SubscribableChannel) replyChan, handler);
793 				}
794 				else if (replyChan instanceof PollableChannel) {
795 					PollingConsumer endpoint = new PollingConsumer((PollableChannel) replyChan, handler);
796 					if (beanFactory != null) {
797 						endpoint.setBeanFactory(beanFactory);
798 					}
799 					endpoint.setReceiveTimeout(this.replyTimeout);
800 					endpoint.afterPropertiesSet();
801 					correlator = endpoint;
802 				}
803 				else if (replyChan instanceof ReactiveStreamsSubscribableChannel) {
804 					ReactiveStreamsConsumer endpoint =
805 							new ReactiveStreamsConsumer(replyChan, (Subscriber<Message<?>>) handler);
806 					endpoint.afterPropertiesSet();
807 					correlator = endpoint;
808 				}
809 				else {
810 					throw new MessagingException("Unsupported 'replyChannel' type [" + replyChan.getClass() + "]."
811 							+ "SubscribableChannel or PollableChannel type are supported.");
812 				}
813 				this.replyMessageCorrelator = correlator;
814 			}
815 			if (isRunning()) {
816 				this.replyMessageCorrelator.start();
817 			}
818 		}
819 	}

824 			this.replyMessageCorrelator.start();

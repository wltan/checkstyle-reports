17  package org.springframework.integration.scripting.config;
18  
19  import java.util.List;
20  
21  import org.w3c.dom.Element;
22  
23  import org.springframework.beans.BeanMetadataElement;
24  import org.springframework.beans.factory.config.RuntimeBeanReference;
25  import org.springframework.beans.factory.support.BeanDefinitionBuilder;
26  import org.springframework.beans.factory.support.ManagedMap;
27  import org.springframework.beans.factory.xml.AbstractSingleBeanDefinitionParser;
28  import org.springframework.beans.factory.xml.ParserContext;
29  import org.springframework.integration.config.xml.IntegrationNamespaceUtils;
30  import org.springframework.integration.scripting.DefaultScriptVariableGenerator;
31  import org.springframework.integration.scripting.RefreshableResourceScriptSource;
32  import org.springframework.scripting.support.StaticScriptSource;
33  import org.springframework.util.CollectionUtils;
34  import org.springframework.util.StringUtils;
35  import org.springframework.util.xml.DomUtils;
36  
37  /**
38   * @author David Turanski
39   * @author Artem Bilan
40   *
41   */
42  public abstract class AbstractScriptParser extends AbstractSingleBeanDefinitionParser {
43  
44  	protected static final String LOCATION_ATTRIBUTE = "location";
45  
46  	protected static final String REFRESH_CHECK_DELAY_ATTRIBUTE = "refresh-check-delay";
47  
48  	@Override
49  	protected boolean shouldGenerateIdAsFallback() {
50  		return true;
51  	}
52  
53  	@Override
54  	protected void doParse(Element element, ParserContext parserContext, BeanDefinitionBuilder builder) {
55  		String scriptLocation = element.getAttribute(LOCATION_ATTRIBUTE);
56  		String scriptText = DomUtils.getTextValue(element);
57  		if (StringUtils.hasText(scriptLocation) == StringUtils.hasText(scriptText)) {
58  			parserContext.getReaderContext().error(
59  					"Either the 'location' attribute or inline script text must be provided, but not both.", element);
60  			return;
61  		}
62  
63  		List<Element> variableElements = DomUtils.getChildElementsByTagName(element, "variable");
64  		String scriptVariableGeneratorName = element.getAttribute("script-variable-generator");
65  
66  		if (StringUtils.hasText(scriptVariableGeneratorName) && variableElements.size() > 0) {
67  			parserContext.getReaderContext().error(
68  					"'script-variable-generator' and 'variable' sub-elements are mutually exclusive.", element);
69  			return;
70  		}
71  
72  		if (StringUtils.hasText(scriptLocation)) {
73  			builder.addConstructorArgValue(resolveScriptLocation(element, scriptLocation));
74  		}
75  		else {
76  			builder.addConstructorArgValue(new StaticScriptSource(scriptText));
77  		}
78  
79  		BeanMetadataElement scriptVariableGeneratorDef;
80  		if (!StringUtils.hasText(scriptVariableGeneratorName)) {
81  			BeanDefinitionBuilder scriptVariableGeneratorBuilder =
82  					BeanDefinitionBuilder.genericBeanDefinition(DefaultScriptVariableGenerator.class);
83  			ManagedMap<String, Object> variableMap = buildVariablesMap(element, parserContext, variableElements);
84  			if (!CollectionUtils.isEmpty(variableMap)) {
85  				scriptVariableGeneratorBuilder.addConstructorArgValue(variableMap);
86  			}
87  			scriptVariableGeneratorDef = scriptVariableGeneratorBuilder.getBeanDefinition();
88  		}
89  		else {
90  			scriptVariableGeneratorDef = new RuntimeBeanReference(scriptVariableGeneratorName);
91  		}
92  
93  		builder.addConstructorArgValue(scriptVariableGeneratorDef);
94  		postProcess(builder, element, parserContext);
95  	}
96  
97  	/**
98  	 * Subclasses may override this no-op method to provide additional configuration.
99  	 *
100 	 * @param builder The builder.
101 	 * @param element The element.
102 	 * @param parserContext The parser context.
103 	 */
104 	protected void postProcess(BeanDefinitionBuilder builder, Element element, ParserContext parserContext) {
105 	}
106 
107 	private Object resolveScriptLocation(Element element, String scriptLocation) {
108 		String refreshDelayText = element.getAttribute(REFRESH_CHECK_DELAY_ATTRIBUTE);
109 		String beanClassName = RefreshableResourceScriptSource.class.getName();
110 		BeanDefinitionBuilder resourceScriptSourceBuilder = BeanDefinitionBuilder.genericBeanDefinition(beanClassName);
111 		resourceScriptSourceBuilder.addConstructorArgValue(scriptLocation);
112 		if (StringUtils.hasText(refreshDelayText)) {
113 			resourceScriptSourceBuilder.addConstructorArgValue(refreshDelayText);
114 		}
115 		else {
116 			resourceScriptSourceBuilder.addConstructorArgValue(-1L);
117 		}
118 		return resourceScriptSourceBuilder.getBeanDefinition();
119 	}
120 
121 	private ManagedMap<String, Object> buildVariablesMap(final Element element, final ParserContext parserContext,
122 			List<Element> variableElements) {
123 
124 		@SuppressWarnings("serial")
125 		ManagedMap<String, Object> variableMap = new ManagedMap<String, Object>() {
126 
127 			@Override
128 			public Object put(String key, Object value) {
129 				if (this.containsKey(key)) {
130 					parserContext.getReaderContext().error("Duplicated variable: " + key, element);
131 				}
132 				return super.put(key, value);
133 			}
134 
135 		};
136 
137 		for (Element childElement : variableElements) {
138 			String variableName = childElement.getAttribute("name");
139 			String variableValue = childElement.getAttribute("value");
140 			String variableRef = childElement.getAttribute("ref");
141 			if (StringUtils.hasText(variableValue) == StringUtils.hasText(variableRef)) {
142 				parserContext.getReaderContext().error(
143 						"Exactly one of the 'ref' attribute or 'value' attribute, " + " is required for element "
144 								+ IntegrationNamespaceUtils.createElementDescription(element) + ".", element);
145 			}
146 			if (StringUtils.hasText(variableValue)) {
147 				variableMap.put(variableName, variableValue);
148 			}
149 			else {
150 				variableMap.put(variableName, new RuntimeBeanReference(variableRef));
151 			}
152 		}
153 
154 		String variables = element.getAttribute("variables");
155 		if (StringUtils.hasText(variables)) {
156 			String[] variablePairs = StringUtils.commaDelimitedListToStringArray(variables);
157 			for (String variablePair : variablePairs) {
158 				String[] variableValue = variablePair.split("=");
159 				if (variableValue.length != 2) {
160 					parserContext.getReaderContext().error(
161 							"Variable declarations in the 'variable' attribute must have the "
162 									+ "form 'var=value'; found : '" + variablePair + "'", element);
163 				}
164 				String variable = variableValue[0].trim();
165 				String value = variableValue[1];
166 				if (variable.endsWith("-ref")) {
167 					variable = variable.substring(0, variable.indexOf("-ref"));
168 					variableMap.put(variable, new RuntimeBeanReference(value));
169 				}
170 				else {
171 					variableMap.put(variable, value);
172 				}
173 			}
174 		}
175 
176 		return variableMap;
177 	}
178

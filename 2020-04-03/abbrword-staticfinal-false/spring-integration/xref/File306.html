64  	/**
65  	 * Obtain a value for the property from the provide object.
66  	 * Supports nested properties via period delimiter.
67  	 * @param root the object to obtain the property value
68  	 * @param propertyPath the property name to obtain a value.
69  	 * Can be nested path defined by the period.
70  	 * @return the value of the property or null
71  	 * @see DirectFieldAccessor
72  	 */
73  	public static Object getPropertyValue(Object root, String propertyPath) {
74  		Object value = null;
75  		DirectFieldAccessor accessor = new DirectFieldAccessor(root);
76  		String[] tokens = propertyPath.split("\\.");
77  		for (int i = 0; i < tokens.length; i++) {
78  			value = accessor.getPropertyValue(tokens[i]);
79  			if (value != null) {
80  				accessor = new DirectFieldAccessor(value);
81  			}
82  			else if (i == tokens.length - 1) {
83  				return null;
84  			}
85  			else {
86  				throw new IllegalArgumentException(
87  						"intermediate property '" + tokens[i] + "' is null");
88  			}
89  		}
90  		return value;
91  	}
92  
93  	/**
94  	 * Obtain a value for the property from the provide object
95  	 * and try to cast it to the provided type.
96  	 * Supports nested properties via period delimiter.
97  	 * @param root the object to obtain the property value
98  	 * @param propertyPath the property name to obtain a value.
99  	 * @param type the expected value type.
100 	 * @param <T> the expected value type.
101 	 * Can be nested path defined by the period.
102 	 * @return the value of the property or null
103 	 * @see DirectFieldAccessor
104 	 */
105 	@SuppressWarnings("unchecked")
106 	public static <T> T getPropertyValue(Object root, String propertyPath, Class<T> type) {
107 		Object value = getPropertyValue(root, propertyPath);
108 		if (value != null) {
109 			Assert.isAssignable(type, value.getClass());
110 		}
111 		return (T) value;
112 	}

114 	/**
115 	 * Create a {@link TestApplicationContext} instance
116 	 * supplied with the basic Spring Integration infrastructure.
117 	 * @return the {@link TestApplicationContext} instance
118 	 */
119 	public static TestApplicationContext createTestApplicationContext() {
120 		TestApplicationContext context = new TestApplicationContext();
121 		ErrorHandler errorHandler = new MessagePublishingErrorHandler(context);
122 		ThreadPoolTaskScheduler scheduler = createTaskScheduler(10);
123 		scheduler.setErrorHandler(errorHandler);
124 		registerBean("taskScheduler", scheduler, context);
125 		registerBean("integrationConversionService", new DefaultFormattingConversionService(), context);
126 		registerBean("errorChannel",
127 				(MessageChannel) (message, timeout) -> {
128 					LOGGER.error(message);
129 					return true;
130 				}, context);
131 		return context;
132 	}

134 	/**
135 	 * A factory for the {@link ThreadPoolTaskScheduler} instances based on the provided {@code poolSize}.
136 	 * @param poolSize the size for the {@link ThreadPoolTaskScheduler}
137 	 * @return the {@link ThreadPoolTaskScheduler} instance.
138 	 */
139 	public static ThreadPoolTaskScheduler createTaskScheduler(int poolSize) {
140 		ThreadPoolTaskScheduler scheduler = new ThreadPoolTaskScheduler();
141 		scheduler.setPoolSize(poolSize);
142 		scheduler.setRejectedExecutionHandler(new CallerRunsPolicy());
143 		scheduler.afterPropertiesSet();
144 		return scheduler;

149 		Assert.notNull(beanName, "bean name must not be null");

167 		public void registerChannel(@Nullable String channelNameArg, final MessageChannel channel) {
168 			String channelName = channelNameArg;
169 			String componentName = getComponentNameIfNamed(channel);
170 			if (componentName != null) {
171 				if (channelName == null) {
172 					channelName = componentName;
173 				}
174 				else {
175 					Assert.isTrue(componentName.equals(channelName),
176 							"channel name has already been set with a conflicting value");
177 				}
178 			}
179 			TestUtils.registerBean(channelName, channel, this);
180 		}

183 			TestUtils.registerBean(endpointName, endpoint, this);
184 		}

187 			TestUtils.registerBean(beanName, bean, this);
188 		}

190 		private String getComponentNameIfNamed(final MessageChannel channel) {
191 			Set<Class<?>> interfaces = ClassUtils.getAllInterfacesAsSet(channel);
192 			final AtomicReference<String> componentName = new AtomicReference<>();
193 			for (Class<?> intface : interfaces) {
194 				if ("org.springframework.integration.support.context.NamedComponent".equals(intface.getName())) {
195 					ReflectionUtils.doWithMethods(channel.getClass(), method -> {
196 						try {
197 							componentName.set((String) method.invoke(channel, new Object[0]));
198 						}
199 						catch (InvocationTargetException e) {
200 							throw new IllegalArgumentException(e);
201 						}
202 					}, method -> method.getName().equals("getComponentName"));
203 					break;
204 				}
205 			}
206 			return componentName.get();
207 		}
208 
209 	}

211 	/**
212 	 * @param history a message history
213 	 * @param componentName the name of a component to scan for
214 	 * @param startingIndex the index to start scanning
215 	 * @return the properties provided by the named component or null if none available
216 	 */
217 	public static Properties locateComponentInHistory(List<Properties> history, String componentName,
218 			int startingIndex) {
219 		Assert.notNull(history, "'history' must not be null");
220 		Assert.isTrue(StringUtils.hasText(componentName), "'componentName' must be provided");
221 		Assert.isTrue(startingIndex < history.size(), "'startingIndex' can not be greater then size of history");
222 		Properties component = null;
223 		for (int i = startingIndex; i < history.size(); i++) {
224 			Properties properties = history.get(i);
225 			if (componentName.equals(properties.get("name"))) {
226 				component = properties;
227 				break;
228 			}
229 		}
230 		return component;
231 	}

242 	private static class MessagePublishingErrorHandler implements ErrorHandler {
243 
244 		private final Log logger = LogFactory.getLog(this.getClass());
245 
246 		private final TestApplicationContext context;
247 
248 		MessagePublishingErrorHandler(TestApplicationContext ctx) {
249 			this.context = ctx;
250 		}
251 
252 		@Override
253 		public void handleError(Throwable throwable) {
254 			MessageChannel errorChannel = resolveErrorChannel(throwable);
255 			boolean sent = false;
256 			if (errorChannel != null) {
257 				try {
258 					sent = errorChannel.send(new ErrorMessage(throwable), 10000);
259 				}
260 				catch (Throwable errorDeliveryError) { //NOSONAR
261 					// message will be logged only
262 					if (logger.isWarnEnabled()) {
263 						logger.warn("Error message was not delivered.", errorDeliveryError);
264 					}
265 					if (errorDeliveryError instanceof Error) {
266 						throw ((Error) errorDeliveryError); // NOSONAR
267 					}
268 				}
269 			}
270 			if (!sent && logger.isErrorEnabled()) {
271 				Message<?> failedMessage =
272 						throwable instanceof MessagingException
273 								? ((MessagingException) throwable).getFailedMessage()
274 								: null;
275 				if (failedMessage != null) {
276 					logger.error("failure occurred in messaging task with message: " + failedMessage, throwable);
277 				}
278 				else {
279 					logger.error("failure occurred in messaging task", throwable);
280 				}
281 			}
282 
283 		}
284 
285 		private MessageChannel resolveErrorChannel(Throwable t) {
286 			if (t instanceof MessagingException) {
287 				Message<?> failedMessage = ((MessagingException) t).getFailedMessage();
288 				if (failedMessage == null) {
289 					return null;
290 				}
291 				Object errorChannelHeader = failedMessage.getHeaders().getErrorChannel();
292 				if (errorChannelHeader instanceof MessageChannel) {
293 					return (MessageChannel) errorChannelHeader;
294 				}
295 				else if (errorChannelHeader instanceof String) {
296 					return this.context.getBean((String) errorChannelHeader, MessageChannel.class);
297 				}
298 				else {
299 					throw new IllegalStateException("Unsupported error channel header type. " +
300 							"Expected MessageChannel or String, but actual header is [" + errorChannelHeader + "]");
301 				}
302 			}
303 			else {
304 				return null;
305 			}
306 		}
307 
308 	}

310 	public static LevelsContainer adjustLogLevels(String methodName, List<Class<?>> classes, List<String> categories,
311 			Level level) {
312 
313 		LoggerContext ctx = (LoggerContext) LogManager.getContext(false);
314 		Configuration config = ctx.getConfiguration();
315 
316 		Map<Class<?>, Level> classLevels = new HashMap<>();
317 		for (Class<?> cls : classes) {
318 			String className = cls.getName();
319 			LoggerConfig specificConfig = addLoggerConfigForCategory(config, className);
320 			classLevels.put(cls, specificConfig.getLevel());
321 			specificConfig.setLevel(level);
322 		}
323 
324 		Map<String, Level> categoryLevels = new HashMap<>();
325 		for (String category : categories) {
326 			LoggerConfig specificConfig = addLoggerConfigForCategory(config, category);
327 			categoryLevels.put(category, specificConfig.getLevel());
328 			specificConfig.setLevel(level);
329 		}
330 
331 		ctx.updateLoggers();
332 
333 		LOGGER.warn("++++++++++++++++++++++++++++ "
334 				+ "Overridden log level setting for: " + classes
335 				+ " and " + categories
336 				+ " for test " + methodName);
337 
338 		return new LevelsContainer(classLevels, categoryLevels);
339 	}
340 
341 	private static LoggerConfig addLoggerConfigForCategory(Configuration config, String category) {
342 		LoggerConfig loggerConfig = config.getLoggerConfig(category);
343 		LoggerConfig specificConfig = loggerConfig;
344 
345 		// We need a specific configuration for this logger,
346 		// otherwise we would change the level of all other loggers
347 		// having the original configuration as parent as well
348 
349 		if (!loggerConfig.getName().equals(category)) {
350 			specificConfig = new LoggerConfig(category, loggerConfig.getLevel(), true);
351 			specificConfig.setParent(loggerConfig);
352 			config.addLogger(category, specificConfig);
353 		}
354 		return specificConfig;
355 	}

357 	public static void revertLogLevels(String methodName, LevelsContainer container) {
358 		LOGGER.warn("++++++++++++++++++++++++++++ "
359 				+ "Restoring log level setting for: " + container.classLevels.keySet()
360 				+ " and " + container.categoryLevels.keySet()
361 				+ " for test " + methodName);
362 
363 		LoggerContext ctx = (LoggerContext) LogManager.getContext(false);
364 		Configuration config = ctx.getConfiguration();
365 
366 		container.classLevels.forEach((key, value) -> {
367 			LoggerConfig loggerConfig = config.getLoggerConfig(key.getName());
368 			loggerConfig.setLevel(value);
369 		});
370 
371 		container.categoryLevels.forEach((key, value) -> {
372 			LoggerConfig loggerConfig = config.getLoggerConfig(key);
373 			loggerConfig.setLevel(value);
374 		});
375 
376 		ctx.updateLoggers();
377 	}

380 
381 		private final Map<Class<?>, Level> classLevels;
382 
383 		private final Map<String, Level> categoryLevels;
384 
385 		public LevelsContainer(Map<Class<?>, Level> classLevels, Map<String, Level> categoryLevels) {
386 			this.classLevels = classLevels;

234 	private MessagingMethodInvokerHelper(Object targetObject, Class<? extends Annotation> annotationType,
235 			Method method, Class<?> expectedType, boolean canProcessMessageList) {
236 
237 		this.annotationType = annotationType;
238 		this.canProcessMessageList = canProcessMessageList;
239 		Assert.notNull(method, "method must not be null");
240 		this.method = method;
241 		this.methodName = null;
242 		this.requiresReply = expectedType != null;
243 		if (expectedType != null) {
244 			Assert.isTrue(method.getReturnType() != Void.class && method.getReturnType() != Void.TYPE,
245 					"method must have a return type");
246 			this.expectedType = TypeDescriptor.valueOf(expectedType);
247 		}
248 		else {
249 			this.expectedType = null;
250 		}
251 
252 		Assert.notNull(targetObject, "targetObject must not be null");
253 		this.targetObject = targetObject;
254 		this.handlerMethod = createHandlerMethod(this.method);
255 		this.handlerMethods = null;
256 		this.handlerMessageMethods = null;
257 		this.handlerMethodsList.add(
258 				Collections.singletonMap(this.handlerMethod.targetParameterType, this.handlerMethod));
259 		setDisplayString(targetObject, method);
260 		this.jsonObjectMapper = configureJsonObjectMapperIfAny();
261 	}

263 	private MessagingMethodInvokerHelper(Object targetObject, Class<? extends Annotation> annotationType,
264 			String methodName, Class<?> expectedType, boolean canProcessMessageList) {
265 
266 		Assert.notNull(targetObject, "targetObject must not be null");
267 		this.annotationType = annotationType;
268 		if (methodName == null) {
269 			if (targetObject instanceof Function) {
270 				this.methodName = "apply";
271 			}
272 			else if (targetObject instanceof Consumer) {
273 				this.methodName = "accept";
274 			}
275 			else {
276 				this.methodName = null;
277 			}
278 		}
279 		else {
280 			this.methodName = methodName;
281 		}
282 
283 		this.method = null;
284 
285 		this.canProcessMessageList = canProcessMessageList;
286 		this.requiresReply = expectedType != null;
287 		if (expectedType != null) {
288 			this.expectedType = TypeDescriptor.valueOf(expectedType);
289 		}
290 		else {
291 			this.expectedType = null;
292 		}
293 		this.targetObject = targetObject;
294 		Map<String, Map<Class<?>, HandlerMethod>> handlerMethodsForTarget = findHandlerMethodsForTarget();
295 		Map<Class<?>, HandlerMethod> methods = handlerMethodsForTarget.get(CANDIDATE_METHODS);
296 		Map<Class<?>, HandlerMethod> messageMethods = handlerMethodsForTarget.get(CANDIDATE_MESSAGE_METHODS);
297 		if ((methods.size() == 1 && messageMethods.isEmpty()) ||
298 				(messageMethods.size() == 1 && methods.isEmpty())) {
299 			if (methods.size() == 1) {
300 				this.handlerMethod = methods.values().iterator().next();
301 			}
302 			else {
303 				this.handlerMethod = messageMethods.values().iterator().next();
304 			}
305 		}
306 		else {
307 			this.handlerMethod = null;
308 		}
309 
310 		this.handlerMethods = methods;
311 		this.handlerMessageMethods = messageMethods;
312 		//TODO Consider to use global option to determine a precedence of methods
313 		this.handlerMethodsList.add(this.handlerMethods);
314 		this.handlerMethodsList.add(this.handlerMessageMethods);
315 
316 		setDisplayString(targetObject, methodName);
317 		this.jsonObjectMapper = configureJsonObjectMapperIfAny();
318 	}

416 	private void prepareEvaluationContext() {
417 		StandardEvaluationContext context = getEvaluationContext();
418 		Class<?> targetType = AopUtils.getTargetClass(this.targetObject);
419 		if (this.method != null) {
420 			context.registerMethodFilter(targetType,
421 					new FixedMethodFilter(ClassUtils.getMostSpecificMethod(this.method, targetType)));
422 			if (this.expectedType != null) {
423 				Assert.state(context.getTypeConverter()
424 								.canConvert(TypeDescriptor.valueOf((this.method).getReturnType()), this.expectedType),
425 						() -> "Cannot convert to expected type (" + this.expectedType + ") from " + this.method);
426 			}
427 		}
428 		else {
429 			AnnotatedMethodFilter filter = new AnnotatedMethodFilter(this.annotationType, this.methodName,
430 					this.requiresReply);
431 			Assert.state(canReturnExpectedType(filter, targetType, context.getTypeConverter()),
432 					() -> "Cannot convert to expected type (" + this.expectedType + ") from " + this.methodName);
433 			context.registerMethodFilter(targetType, filter);
434 		}
435 		context.setVariable("target", this.targetObject);
436 		try {
437 			context.registerFunction("requiredHeader",
438 					ParametersWrapper.class.getDeclaredMethod("getHeader", Map.class, String.class));
439 		}
440 		catch (NoSuchMethodException ex) {
441 			throw new IllegalStateException(ex);
442 		}
443 	}

445 	private boolean canReturnExpectedType(AnnotatedMethodFilter filter, Class<?> targetType,
446 			TypeConverter typeConverter) {
447 
448 		if (this.expectedType == null) {
449 			return true;
450 		}
451 
452 		List<Method> methods = filter.filter(Arrays.asList(ReflectionUtils.getAllDeclaredMethods(targetType)));
453 		return methods
454 				.stream()
455 				.anyMatch(candidate ->
456 						typeConverter.canConvert(TypeDescriptor.valueOf(candidate.getReturnType()), this.expectedType));
457 	}

459 	@Nullable
460 	private Object processInternal(ParametersWrapper parameters) {
461 		if (!this.initialized) {
462 			initialize();
463 		}
464 		HandlerMethod candidate = findHandlerMethodForParameters(parameters);
465 		if (candidate == null) {
466 			candidate = this.defaultHandlerMethod;
467 		}
468 		Assert.notNull(candidate, "No candidate methods found for messages.");
469 		if (!candidate.initialized) {
470 			initializeHandler(candidate);
471 		}
472 		Object result;
473 		if (this.useSpelInvoker || candidate.spelOnly) {
474 			result = invokeExpression(candidate.expression, parameters);
475 		}
476 		else {
477 			result = invokeHandlerMethod(candidate, parameters);
478 		}
479 
480 		if (result != null && this.expectedType != null) {
481 			return getEvaluationContext()
482 					.getTypeConverter()
483 					.convertValue(result, TypeDescriptor.forObject(result), this.expectedType);
484 		}
485 		else {
486 			return result;
487 		}
488 	}

490 	private synchronized void initialize() {
491 		if (isProvidedMessageHandlerFactoryBean()) {
492 			LOGGER.info("Overriding default instance of MessageHandlerMethodFactory with provided one.");
493 			this.messageHandlerMethodFactory =
494 					getBeanFactory()
495 							.getBean(
496 									this.canProcessMessageList
497 											? IntegrationContextUtils.LIST_MESSAGE_HANDLER_FACTORY_BEAN_NAME
498 											: IntegrationContextUtils.MESSAGE_HANDLER_FACTORY_BEAN_NAME,
499 									MessageHandlerMethodFactory.class);
500 		}
501 		else {
502 			configureLocalMessageHandlerFactory();
503 		}
504 
505 		prepareEvaluationContext();
506 		this.initialized = true;
507 	}

523 	private void initializeHandler(HandlerMethod candidate) {
524 		ExpressionParser parser;
525 		if (candidate.useSpelInvoker == null) {
526 			parser = EXPRESSION_PARSER_DEFAULT;
527 		}
528 		else {
529 			String compilerMode = resolveExpression(candidate.useSpelInvoker.compilerMode()).toUpperCase();
530 			parser = !StringUtils.hasText(compilerMode)
531 					? EXPRESSION_PARSER_DEFAULT
532 					: SPEL_COMPILERS.get(SpelCompilerMode.valueOf(compilerMode));
533 		}
534 		candidate.expression = parser.parseExpression(candidate.expressionString);
535 		if (!this.useSpelInvoker && !candidate.spelOnly) {
536 			candidate.setInvocableHandlerMethod(createInvocableHandlerMethod(candidate.method));
537 		}
538 		candidate.initialized = true;
539 	}

541 	private void configureLocalMessageHandlerFactory() {
542 		BeanFactory beanFactory = getBeanFactory();
543 
544 		ConfigurableCompositeMessageConverter messageConverter = new ConfigurableCompositeMessageConverter();
545 		messageConverter.setBeanFactory(beanFactory);
546 		messageConverter.afterPropertiesSet();
547 
548 		List<HandlerMethodArgumentResolver> customArgumentResolvers = new LinkedList<>();
549 		PayloadExpressionArgumentResolver payloadExpressionArgumentResolver = new PayloadExpressionArgumentResolver();
550 		PayloadsArgumentResolver payloadsArgumentResolver = new PayloadsArgumentResolver();
551 
552 		customArgumentResolvers.add(payloadExpressionArgumentResolver);
553 		customArgumentResolvers.add(new NullAwarePayloadArgumentResolver(messageConverter));
554 		customArgumentResolvers.add(payloadsArgumentResolver);
555 
556 		CollectionArgumentResolver collectionArgumentResolver = null;
557 
558 		if (this.canProcessMessageList) {
559 			collectionArgumentResolver = new CollectionArgumentResolver(true);
560 			customArgumentResolvers.add(collectionArgumentResolver);
561 		}
562 
563 		MapArgumentResolver mapArgumentResolver = new MapArgumentResolver();
564 		customArgumentResolvers.add(mapArgumentResolver);
565 		payloadExpressionArgumentResolver.setBeanFactory(beanFactory);
566 		payloadsArgumentResolver.setBeanFactory(beanFactory);
567 		mapArgumentResolver.setBeanFactory(beanFactory);
568 		if (collectionArgumentResolver != null) {
569 			collectionArgumentResolver.setBeanFactory(beanFactory);
570 		}
571 
572 		DefaultMessageHandlerMethodFactory localHandlerMethodFactory =
573 				(DefaultMessageHandlerMethodFactory) this.messageHandlerMethodFactory;
574 		localHandlerMethodFactory.setMessageConverter(messageConverter);
575 		localHandlerMethodFactory.setCustomArgumentResolvers(customArgumentResolvers);
576 		localHandlerMethodFactory.afterPropertiesSet();
577 	}

579 	private Object invokeHandlerMethod(HandlerMethod handlerMethod, ParametersWrapper parameters) {
580 		try {
581 			return handlerMethod.invoke(parameters);
582 		}
583 		catch (MethodArgumentResolutionException | MessageConversionException | IllegalStateException ex) {
584 			return processInvokeExceptionAndFallbackToExpressionIfAny(handlerMethod, parameters, ex);
585 		}
586 		catch (RuntimeException ex) { // NOSONAR no way to handle conditional catch according Sonar rules
587 			throw ex;
588 		}
589 		catch (Exception ex) {
590 			throw new IllegalStateException("HandlerMethod invocation error", ex);
591 		}
592 	}
593 
594 	private Object processInvokeExceptionAndFallbackToExpressionIfAny(HandlerMethod handlerMethod,
595 			ParametersWrapper parameters, RuntimeException ex) {
596 
597 		if (ex instanceof MessageConversionException) {
598 			if (ex.getCause() instanceof ConversionFailedException &&
599 					!(ex.getCause().getCause() instanceof ConverterNotFoundException)) {
600 				throw ex;
601 			}
602 		}
603 		else if (ex instanceof IllegalStateException && // NOSONAR complex boolean expression
604 				(!(ex.getCause() instanceof IllegalArgumentException) ||
605 						!ex.getStackTrace()[0].getClassName().equals(InvocableHandlerMethod.class.getName()) ||
606 						(!"argument type mismatch".equals(ex.getCause().getMessage()) &&
607 								// JVM generates GeneratedMethodAccessor### after several calls with less error
608 								// checking
609 								!ex.getCause().getMessage().startsWith("java.lang.ClassCastException@")))) {
610 			throw ex;
611 		}
612 
613 		return fallbackToInvokeExpression(handlerMethod, parameters);
614 	}
615 
616 	private Object fallbackToInvokeExpression(HandlerMethod handlerMethod, ParametersWrapper parameters) {
617 		Expression expression = handlerMethod.expression;
618 
619 		if (++handlerMethod.failedAttempts >= FAILED_ATTEMPTS_THRESHOLD) {
620 			handlerMethod.spelOnly = true;
621 			if (LOGGER.isInfoEnabled()) {
622 				LOGGER.info("Failed to invoke [ " + handlerMethod.invocableHandlerMethod +
623 						"] with provided arguments [ " + parameters + " ]. \n" +
624 						"Falling back to SpEL invocation for expression [ " +
625 						expression.getExpressionString() + " ]");
626 			}
627 		}
628 
629 		return invokeExpression(expression, parameters);
630 	}

644 		Throwable evaluationException = ex;
645 		if ((ex instanceof EvaluationException || ex instanceof MessageHandlingException)

703 	private Map<String, Map<Class<?>, HandlerMethod>> findHandlerMethodsForTarget() {
704 		Map<String, Map<Class<?>, HandlerMethod>> methods = new HashMap<>();
705 		Map<Class<?>, HandlerMethod> candidateMethods = new HashMap<>();
706 		Map<Class<?>, HandlerMethod> candidateMessageMethods = new HashMap<>();
707 		Map<Class<?>, HandlerMethod> fallbackMethods = new HashMap<>();
708 		Map<Class<?>, HandlerMethod> fallbackMessageMethods = new HashMap<>();
709 		AtomicReference<Class<?>> ambiguousFallbackType = new AtomicReference<>();
710 		AtomicReference<Class<?>> ambiguousFallbackMessageGenericType = new AtomicReference<>();
711 		Class<?> targetClass = getTargetClass(this.targetObject);
712 
713 		processMethodsFromTarget(candidateMethods, candidateMessageMethods, fallbackMethods, fallbackMessageMethods,
714 				ambiguousFallbackType, ambiguousFallbackMessageGenericType, targetClass);
715 
716 		if (!candidateMethods.isEmpty() || !candidateMessageMethods.isEmpty()) {
717 			methods.put(CANDIDATE_METHODS, candidateMethods);
718 			methods.put(CANDIDATE_MESSAGE_METHODS, candidateMessageMethods);
719 			return methods;
720 		}
721 
722 		if ((ambiguousFallbackType.get() != null
723 				|| ambiguousFallbackMessageGenericType.get() != null)
724 				&& ServiceActivator.class.equals(this.annotationType)) {
725 			/*
726 			 * When there are ambiguous fallback methods,
727 			 * a Service Activator can finally fallback to RequestReplyExchanger.exchange(m).
728 			 * Ambiguous means > 1 method that takes the same payload type, or > 1 method
729 			 * that takes a Message with the same generic type.
730 			 */
731 			Method frameworkMethod = obtainFrameworkMethod(targetClass);
732 			if (frameworkMethod != null) {
733 				HandlerMethod theHandlerMethod = createHandlerMethod(frameworkMethod);
734 				methods.put(CANDIDATE_METHODS, Collections.singletonMap(Object.class, theHandlerMethod));
735 				methods.put(CANDIDATE_MESSAGE_METHODS, candidateMessageMethods);
736 				return methods;
737 			}
738 		}
739 
740 		validateFallbackMethods(fallbackMethods, fallbackMessageMethods, ambiguousFallbackType,
741 				ambiguousFallbackMessageGenericType);
742 
743 		methods.put(CANDIDATE_METHODS, fallbackMethods);
744 		methods.put(CANDIDATE_MESSAGE_METHODS, fallbackMessageMethods);
745 		return methods;
746 	}

765 	private void processMethodsFromTarget(Map<Class<?>, HandlerMethod> candidateMethods,
766 			Map<Class<?>, HandlerMethod> candidateMessageMethods, Map<Class<?>, HandlerMethod> fallbackMethods,
767 			Map<Class<?>, HandlerMethod> fallbackMessageMethods, AtomicReference<Class<?>> ambiguousFallbackType,
768 			AtomicReference<Class<?>> ambiguousFallbackMessageGenericType, Class<?> targetClass) {
769 
770 		ReflectionUtils.doWithMethods(targetClass, method1 -> {
771 			boolean matchesAnnotation = false;
772 			if (this.annotationType != null && AnnotationUtils.findAnnotation(method1, this.annotationType) != null) {
773 				matchesAnnotation = true;
774 			}
775 			else if (!Modifier.isPublic(method1.getModifiers())) {
776 				return;
777 			}
778 
779 			HandlerMethod handlerMethod1 = obtainHandlerMethodIfAny(method1);
780 
781 			if (handlerMethod1 != null) {
782 				populateHandlerMethod(candidateMethods, candidateMessageMethods, fallbackMethods,
783 						fallbackMessageMethods,
784 						ambiguousFallbackType, ambiguousFallbackMessageGenericType, matchesAnnotation, handlerMethod1);
785 			}
786 
787 		}, new UniqueMethodFilter(targetClass));
788 
789 		if (candidateMethods.isEmpty() && candidateMessageMethods.isEmpty() && fallbackMethods.isEmpty()
790 				&& fallbackMessageMethods.isEmpty()) {
791 			findSingleSpecifMethodOnInterfacesIfProxy(candidateMessageMethods, candidateMethods);
792 		}
793 	}

796 	private HandlerMethod obtainHandlerMethodIfAny(Method methodToProcess) {
797 		HandlerMethod handlerMethodToUse = null;
798 		if (isMethodEligible(methodToProcess)) {
799 			try {
800 				handlerMethodToUse = createHandlerMethod(
801 						AopUtils.selectInvocableMethod(methodToProcess, ClassUtils.getUserClass(this.targetObject)));
802 			}
803 			catch (Exception e) {
804 				if (LOGGER.isDebugEnabled()) {
805 					LOGGER.debug("Method [" + methodToProcess + "] is not eligible for Message handling.", e);
806 				}
807 				return null;
808 			}

810 			if (AnnotationUtils.getAnnotation(methodToProcess, Default.class) != null) {
811 				Assert.state(this.defaultHandlerMethod == null,
812 						() -> "Only one method can be @Default, but there are more for: " + this.targetObject);
813 				this.defaultHandlerMethod = handlerMethodToUse;
814 			}
815 		}
816

818 	}

820 	private boolean isMethodEligible(Method methodToProcess) {
821 		return !(methodToProcess.isBridge() || // NOSONAR boolean complexity
822 				isMethodDefinedOnObjectClass(methodToProcess) ||
823 				methodToProcess.getDeclaringClass().equals(Proxy.class) ||
824 				(this.requiresReply && void.class.equals(methodToProcess.getReturnType())) ||
825 				(this.methodName != null && !this.methodName.equals(methodToProcess.getName())) ||

827 						ObjectUtils.containsElement(new String[] { "start", "stop", "isRunning" },

831 	private void populateHandlerMethod(Map<Class<?>, HandlerMethod> candidateMethods,
832 			Map<Class<?>, HandlerMethod> candidateMessageMethods, Map<Class<?>, HandlerMethod> fallbackMethods,
833 			Map<Class<?>, HandlerMethod> fallbackMessageMethods, AtomicReference<Class<?>> ambiguousFallbackType,
834 			AtomicReference<Class<?>> ambiguousFallbackMessageGenericType, boolean matchesAnnotation,
835 			HandlerMethod handlerMethod1) {
836 
837 		Class<?> targetParameterType = handlerMethod1.getTargetParameterType();
838 		if (matchesAnnotation || this.annotationType == null) {
839 			if (handlerMethod1.isMessageMethod()) {
840 				if (candidateMessageMethods.containsKey(targetParameterType)) {
841 					throw new IllegalArgumentException("Found more than one method match for type " +
842 							"[Message<" + targetParameterType + ">]");
843 				}
844 				candidateMessageMethods.put(targetParameterType, handlerMethod1);
845 			}
846 			else {
847 				if (candidateMethods.containsKey(targetParameterType)) {
848 					String exceptionMessage = "Found more than one method match for ";
849 					if (Void.class.equals(targetParameterType)) {
850 						exceptionMessage += "empty parameter for 'payload'";
851 					}
852 					else {
853 						exceptionMessage += "type [" + targetParameterType + "]";
854 					}
855 					throw new IllegalArgumentException(exceptionMessage);
856 				}
857 				candidateMethods.put(targetParameterType, handlerMethod1);
858 			}
859 		}
860 		else {
861 			if (handlerMethod1.isMessageMethod()) {
862 				if (fallbackMessageMethods.containsKey(targetParameterType)) {
863 					// we need to check for duplicate type matches,
864 					// but only if we end up falling back
865 					// and we'll only keep track of the first one
866 					ambiguousFallbackMessageGenericType.compareAndSet(null, targetParameterType);
867 				}
868 				fallbackMessageMethods.put(targetParameterType, handlerMethod1);
869 			}
870 			else {
871 				if (fallbackMethods.containsKey(targetParameterType)) {
872 					// we need to check for duplicate type matches,
873 					// but only if we end up falling back
874 					// and we'll only keep track of the first one
875 					ambiguousFallbackType.compareAndSet(null, targetParameterType);
876 				}
877 				fallbackMethods.put(targetParameterType, handlerMethod1);
878 			}
879 		}
880 	}

901 			Map<Class<?>, HandlerMethod> candidateMethods) {
902 		if (AopUtils.isAopProxy(this.targetObject)) {
903 			final AtomicReference<Method> targetMethod = new AtomicReference<>();
904 			final AtomicReference<Class<?>> targetClass = new AtomicReference<>();
905 			Class<?>[] interfaces = ((Advised) this.targetObject).getProxiedInterfaces();
906 			for (Class<?> clazz : interfaces) {
907 				ReflectionUtils.doWithMethods(clazz, method1 -> {

914 						targetClass.set(clazz);

926 								"[Message<" + targetParameterType + ">]");
927 					}
928 					candidateMessageMethods.put(targetParameterType, theHandlerMethod);
929 				}
930 				else {
931 					if (candidateMethods.containsKey(targetParameterType)) {
932 						String exceptionMessage = "Found more than one method match for ";
933 						if (Void.class.equals(targetParameterType)) {
934 							exceptionMessage += "empty parameter for 'payload'";
935 						}
936 						else {
937 							exceptionMessage += "type [" + targetParameterType + "]";
938 						}
939 						throw new IllegalArgumentException(exceptionMessage);
940 					}
941 					candidateMethods.put(targetParameterType, theHandlerMethod);
942 				}
943 			}
944 		}

1115 		private String generateExpression(Method method) {
1116 			StringBuilder sb =
1117 					new StringBuilder("#target.")
1118 							.append(method.getName())
1119 							.append('(');
1120 			Class<?>[] parameterTypes = method.getParameterTypes();
1121 			Annotation[][] parameterAnnotations = method.getParameterAnnotations();
1122 			boolean hasUnqualifiedMapParameter = false;
1123 			for (int i = 0; i < parameterTypes.length; i++) {
1124 				if (i != 0) {
1125 					sb.append(", ");
1126 				}
1127 				MethodParameter methodParameter = new MethodParameter(method, i);
1128 				TypeDescriptor parameterTypeDescriptor = new TypeDescriptor(methodParameter);
1129 				Class<?> parameterType = parameterTypeDescriptor.getObjectType();
1130 				Type genericParameterType = method.getGenericParameterTypes()[i];
1131 				Annotation mappingAnnotation =
1132 						MessagingAnnotationUtils.findMessagePartAnnotation(parameterAnnotations[i], true);
1133 				hasUnqualifiedMapParameter = processMethodParameterForExpression(sb, hasUnqualifiedMapParameter,
1134 						methodParameter, parameterTypeDescriptor, parameterType, genericParameterType,
1135 						mappingAnnotation);
1136 			}
1137 			sb.append(")");
1138 			if (this.targetParameterTypeDescriptor == null) {
1139 				this.targetParameterTypeDescriptor = TypeDescriptor.valueOf(Void.class);
1140 			}
1141 			return sb.toString();
1142 		}

1144 		private boolean processMethodParameterForExpression(StringBuilder sb, boolean hasUnqualifiedMapParameter,
1145 				MethodParameter methodParameter, TypeDescriptor parameterTypeDescriptor, Class<?> parameterType,
1146 				Type genericParameterType, Annotation mappingAnnotation) {
1147 
1148 			if (mappingAnnotation != null) {
1149 				processMappingAnnotationForExpression(sb, methodParameter, parameterTypeDescriptor, parameterType,
1150 						mappingAnnotation);
1151 			}
1152 			else if (parameterTypeDescriptor.isAssignableTo(MESSAGE_TYPE_DESCRIPTOR)) {
1153 				this.messageMethod = true;
1154 				sb.append("message");
1155 				setExclusiveTargetParameterType(parameterTypeDescriptor, methodParameter);
1156 			}
1157 			else if (this.canProcessMessageList &&
1158 					(parameterTypeDescriptor.isAssignableTo(MESSAGE_LIST_TYPE_DESCRIPTOR)
1159 							|| parameterTypeDescriptor.isAssignableTo(MESSAGE_ARRAY_TYPE_DESCRIPTOR))) {
1160 				sb.append("messages");
1161 				setExclusiveTargetParameterType(parameterTypeDescriptor, methodParameter);
1162 			}
1163 			else if (Collection.class.isAssignableFrom(parameterType) || parameterType.isArray()) {
1164 				addCollectionParameterForExpression(sb);
1165 				setExclusiveTargetParameterType(parameterTypeDescriptor, methodParameter);
1166 			}
1167 			else if (Iterator.class.isAssignableFrom(parameterType)) {
1168 				populateIteratorParameterForExpression(sb, genericParameterType);
1169 				setExclusiveTargetParameterType(parameterTypeDescriptor, methodParameter);
1170 			}
1171 			else if (Map.class.isAssignableFrom(parameterType)) {
1172 				Assert.isTrue(!hasUnqualifiedMapParameter,
1173 						"Found more than one Map typed parameter without any qualification. "
1174 								+ "Consider using @Payload or @Headers on at least one of the parameters.");
1175 				populateMapParameterForExpression(sb, parameterType);
1176 				return true;
1177 			}
1178 			else {
1179 				sb.append("payload");
1180 				setExclusiveTargetParameterType(parameterTypeDescriptor, methodParameter);
1181 			}
1182 			return hasUnqualifiedMapParameter;
1183 		}

1185 		private void processMappingAnnotationForExpression(StringBuilder sb, MethodParameter methodParameter,
1186 				TypeDescriptor parameterTypeDescriptor, Class<?> parameterType, Annotation mappingAnnotation) {
1187 
1188 			Class<? extends Annotation> annotationType = mappingAnnotation.annotationType();
1189 			if (annotationType.equals(Payload.class)) {
1190 				sb.append("payload");

1208 				if (StringUtils.hasText(qualifierExpression)) {
1209 					sb.append(".")
1210 							.append(qualifierExpression);
1211 				}
1212 				sb.append("]");
1213 				if (!StringUtils.hasText(qualifierExpression)) {
1214 					setExclusiveTargetParameterType(parameterTypeDescriptor, methodParameter);
1215 				}
1216 			}

1221 			}

1272 		private String determineHeaderExpression(Annotation headerAnnotation, MethodParameter methodParameter) {
1273 			methodParameter.initParameterNameDiscovery(PARAMETER_NAME_DISCOVERER);
1274 			String relativeExpression = "";
1275 			AnnotationAttributes annotationAttributes =
1276 					(AnnotationAttributes) AnnotationUtils.getAnnotationAttributes(headerAnnotation);
1277 			String valueAttribute = annotationAttributes.getString(AnnotationUtils.VALUE);
1278 			int len = valueAttribute.length();
1279 			String headerName = valueAttribute;
1280 			if (!StringUtils.hasText(valueAttribute)) {
1281 				headerName = methodParameter.getParameterName();
1282 			}
1283 			else if (len > 2 && valueAttribute.charAt(0) == '\'' && valueAttribute.charAt(len - 1) == '\'') {
1284 				headerName = valueAttribute.substring(1, len - 1);
1285 			}
1286 			else if (valueAttribute.indexOf('.') != -1) {
1287 				String[] tokens = valueAttribute.split("\\.", 2);
1288 				headerName = tokens[0];
1289 				if (StringUtils.hasText(tokens[1])) {
1290 					relativeExpression = "." + tokens[1];
1291 					this.spelOnly = true;
1292 				}
1293 			}
1294 
1295 			Assert.notNull(headerName, "Cannot determine header name. Possible reasons: -debug is "
1296 					+ "disabled or header name is not explicitly provided via @Header annotation.");
1297 			String headerRetrievalExpression = "headers['" + headerName + "']";
1298 			String fullHeaderExpression = headerRetrievalExpression + relativeExpression;
1299 			if (annotationAttributes.getBoolean("required")
1300 					&& !methodParameter.getParameterType().equals(Optional.class)) {
1301 				return "#requiredHeader(headers, '" + headerName + "')" + relativeExpression;
1302 			}
1303 			else if (!StringUtils.hasLength(relativeExpression)) {
1304 				return headerRetrievalExpression + " ?: null";
1305 			}
1306 			else {
1307 				return headerRetrievalExpression + " != null ? " + fullHeaderExpression + " : null";
1308 			}
1309 		}

1312 				MethodParameter methodParameter) {
1313 
1314 			if (this.targetParameterTypeDescriptor != null) {
1315 				throw new IneligibleMethodException("Found more than one parameter type candidate: [" +
1316 						this.targetParameterTypeDescriptor + "] and [" + targetParameterType + "]");
1317 			}
1318 			this.targetParameterTypeDescriptor = targetParameterType;
1319 			if (Message.class.isAssignableFrom(targetParameterType.getObjectType())) {
1320 				this.targetParameterType = methodParameter.nested().getNestedParameterType();
1321 			}
1322 			else {
1323 				this.targetParameterType = targetParameterType.getObjectType();
1324 			}
1325 
1326 			this.exclusiveMethodParameter = methodParameter;
1327 		}

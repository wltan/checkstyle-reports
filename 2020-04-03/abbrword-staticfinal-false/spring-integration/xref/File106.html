57  	/**
58  	 * Return an {@link ObjectMapper} if available,
59  	 * supplied with Message specific serializers and deserializers.
60  	 * Also configured to store typo info in the {@code @class} property.
61  	 * @param trustedPackages the trusted Java packages for deserialization.
62  	 * @return the mapper.
63  	 * @throws IllegalStateException if an implementation is not available.
64  	 * @since 4.3.10
65  	 */
66  	public static ObjectMapper messagingAwareMapper(String... trustedPackages) {
67  		if (JacksonPresent.isJackson2Present()) {
68  			ObjectMapper mapper = new Jackson2JsonObjectMapper().getObjectMapper();
69  
70  			mapper.setDefaultTyping(new WhitelistTypeResolverBuilder(trustedPackages));
71  
72  			GenericMessageJacksonDeserializer genericMessageDeserializer = new GenericMessageJacksonDeserializer();
73  			genericMessageDeserializer.setMapper(mapper);
74  
75  			ErrorMessageJacksonDeserializer errorMessageDeserializer = new ErrorMessageJacksonDeserializer();
76  			errorMessageDeserializer.setMapper(mapper);
77  
78  			AdviceMessageJacksonDeserializer adviceMessageDeserializer = new AdviceMessageJacksonDeserializer();
79  			adviceMessageDeserializer.setMapper(mapper);
80  
81  			MutableMessageJacksonDeserializer mutableMessageDeserializer = new MutableMessageJacksonDeserializer();
82  			mutableMessageDeserializer.setMapper(mapper);
83  
84  			SimpleModule simpleModule = new SimpleModule()
85  					.addSerializer(new MessageHeadersJacksonSerializer())
86  					.addDeserializer(GenericMessage.class, genericMessageDeserializer)
87  					.addDeserializer(ErrorMessage.class, errorMessageDeserializer)
88  					.addDeserializer(AdviceMessage.class, adviceMessageDeserializer)
89  					.addDeserializer(MutableMessage.class, mutableMessageDeserializer);
90  
91  			mapper.registerModule(simpleModule);
92  			return mapper;
93  		}
94  		else {
95  			throw new IllegalStateException("No jackson-databind.jar is present in the classpath.");
96  		}
97  	}

99  	/**
100 	 * An implementation of {@link ObjectMapper.DefaultTypeResolverBuilder}
101 	 * that wraps a default {@link TypeIdResolver} to the {@link WhitelistTypeIdResolver}.
102 	 *
103 	 * @author Rob Winch
104 	 * @author Artem Bilan
105 	 * @author Filip Hanik
106 	 * @author Gary Russell
107 	 *
108 	 * @since 4.3.11
109 	 */
110 	private static final class WhitelistTypeResolverBuilder extends ObjectMapper.DefaultTypeResolverBuilder {
111 
112 		private static final long serialVersionUID = 1L;
113 
114 		private final String[] trustedPackages;
115 
116 		WhitelistTypeResolverBuilder(String... trustedPackages) {
117 			super(ObjectMapper.DefaultTyping.NON_FINAL,
118 					//we do explicit validation in the TypeIdResolver
119 					BasicPolymorphicTypeValidator.builder()
120 							.allowIfSubType(Object.class)
121 							.build());
122 
123 			this.trustedPackages =
124 					trustedPackages != null ? Arrays.copyOf(trustedPackages, trustedPackages.length) : null;
125 
126 			init(JsonTypeInfo.Id.CLASS, null)
127 					.inclusion(JsonTypeInfo.As.PROPERTY);
128 		}
129 
130 		@Override
131 		protected TypeIdResolver idResolver(MapperConfig<?> config,
132 				JavaType baseType,
133 				PolymorphicTypeValidator subtypeValidator,
134 				Collection<NamedType> subtypes, boolean forSer, boolean forDeser) {
135 			TypeIdResolver result = super.idResolver(config, baseType, subtypeValidator, subtypes, forSer, forDeser);
136 			return new WhitelistTypeIdResolver(result, this.trustedPackages);
137 		}
138 
139 	}
140 
141 	/**
142 	 * A {@link TypeIdResolver} that delegates to an existing implementation
143 	 * and throws an IllegalStateException if the class being looked up is not whitelisted,
144 	 * does not provide an explicit mixin mappings.
145 	 *
146 	 * @author Rob Winch
147 	 * @author Artem Bilan
148 	 *
149 	 * @since 4.3.11
150 	 */
151 	private static final class WhitelistTypeIdResolver implements TypeIdResolver {
152 
153 		private static final List<String> TRUSTED_PACKAGES =
154 				Arrays.asList(
155 						"java.util",
156 						"java.lang",
157 						"org.springframework.messaging.support",
158 						"org.springframework.integration.support",
159 						"org.springframework.integration.message",
160 						"org.springframework.integration.store"
161 				);
162 
163 		private final TypeIdResolver delegate;
164 
165 		private final Set<String> trustedPackages = new LinkedHashSet<>(TRUSTED_PACKAGES);
166 
167 		WhitelistTypeIdResolver(TypeIdResolver delegate, String... trustedPackages) {
168 			this.delegate = delegate;
169 			if (trustedPackages != null) {
170 				for (String whiteListClass : trustedPackages) {
171 					if ("*".equals(whiteListClass)) {
172 						this.trustedPackages.clear();
173 						break;
174 					}
175 					else {
176 						this.trustedPackages.add(whiteListClass);
177 					}
178 				}
179 			}
180 		}
181 
182 		@Override
183 		public void init(JavaType baseType) {
184 			this.delegate.init(baseType);
185 		}
186 
187 		@Override
188 		public String idFromValue(Object value) {
189 			return this.delegate.idFromValue(value);
190 		}
191 
192 		@Override
193 		public String idFromValueAndType(Object value, Class<?> suggestedType) {
194 			return this.delegate.idFromValueAndType(value, suggestedType);
195 		}
196 
197 		@Override
198 		public String idFromBaseType() {
199 			return this.delegate.idFromBaseType();
200 		}
201 
202 		@Override
203 		public JavaType typeFromId(DatabindContext context, String id) throws IOException {
204 			DeserializationConfig config = (DeserializationConfig) context.getConfig();
205 			JavaType result = this.delegate.typeFromId(context, id);
206 
207 			Package aPackage = result.getRawClass().getPackage();
208 			if (aPackage == null || isTrustedPackage(aPackage.getName())) {
209 				return result;
210 			}
211 
212 			boolean isExplicitMixin = config.findMixInClassFor(result.getRawClass()) != null;
213 			if (isExplicitMixin) {
214 				return result;
215 			}
216 
217 			throw new IllegalArgumentException("The class with " + id + " and name of " +
218 					"" + result.getRawClass().getName() + " is not in the trusted packages: " +
219 					"" + this.trustedPackages + ". " +
220 					"If you believe this class is safe to deserialize, please provide its name or an explicit Mixin. " +
221 					"If the serialization is only done by a trusted source, you can also enable default typing.");
222 		}
223 
224 		private boolean isTrustedPackage(String packageName) {
225 			if (!this.trustedPackages.isEmpty()) {
226 				for (String trustedPackage : this.trustedPackages) {
227 					if (packageName.equals(trustedPackage) || packageName.startsWith(trustedPackage + ".")) {
228 						return true;
229 					}
230 				}
231 				return false;
232 			}
233 
234 			return true;
235 		}
236 
237 		@Override
238 		public String getDescForKnownTypeIds() {
239 			return this.delegate.getDescForKnownTypeIds();
240 		}
241 
242 		@Override
243 		public JsonTypeInfo.Id getMechanism() {
244 			return this.delegate.getMechanism();
245 		}
246 
247 	}

90  	/**
91  	 * @param queueName Must not be an empty String
92  	 * @param connectionFactory Must not be null
93  	 */
94  	public RedisQueueInboundGateway(String queueName, RedisConnectionFactory connectionFactory) {
95  		Assert.hasText(queueName, "'queueName' is required");
96  		Assert.notNull(connectionFactory, "'connectionFactory' must not be null");
97  		this.template = new RedisTemplate<>();
98  		this.template.setConnectionFactory(connectionFactory);
99  		this.template.setEnableDefaultSerializer(false);
100 		this.template.setKeySerializer(new StringRedisSerializer());
101 		this.template.afterPropertiesSet();
102 		this.boundListOperations = this.template.boundListOps(queueName);
103 	}

126 	/**
127 	 * This timeout (milliseconds) is used when retrieving elements from the queue
128 	 * specified by {@link #boundListOperations}.
129 	 * <p> If the queue does contain elements, the data is retrieved immediately. However,
130 	 * if the queue is empty, the Redis connection is blocked until either an element
131 	 * can be retrieved from the queue or until the specified timeout passes.
132 	 * <p> A timeout of zero can be used to block indefinitely. If not set explicitly
133 	 * the timeout value will default to {@code 1000}
134 	 * <p> See also: https://redis.io/commands/brpop
135 	 * @param receiveTimeout Must be non-negative. Specified in milliseconds.
136 	 */
137 	public void setReceiveTimeout(long receiveTimeout) {
138 		Assert.isTrue(receiveTimeout >= 0, "'receiveTimeout' must be >= 0.");
139 		this.receiveTimeout = receiveTimeout;
140 	}

150 	@Override
151 	protected void onInit() {
152 		super.onInit();
153 		if (!this.extractPayload) {
154 			Assert.notNull(this.serializer, "'serializer' has to be provided where 'extractPayload == false'.");
155 		}
156 		if (this.taskExecutor == null) {
157 			String beanName = getComponentName();
158 			this.taskExecutor = new SimpleAsyncTaskExecutor((beanName == null ? "" : beanName + "-")
159 					+ getComponentType());
160 		}
161 		if (!(this.taskExecutor instanceof ErrorHandlingTaskExecutor) && getBeanFactory() != null) {
162 			MessagePublishingErrorHandler errorHandler = new MessagePublishingErrorHandler();
163 			errorHandler.setBeanFactory(getBeanFactory());
164 			errorHandler.setDefaultErrorChannel(getErrorChannel());
165 			this.taskExecutor = new ErrorHandlingTaskExecutor(this.taskExecutor, errorHandler);
166 		}
167 	}

169 	@Override
170 	public String getComponentType() {
171 		return "redis:queue-inbound-gateway";
172 	}

174 	private void handlePopException(Exception e) {
175 		this.listening = false;
176 		if (this.active) {
177 			logger.error("Failed to execute listening task. Will attempt to resubmit in " + this.recoveryInterval
178 					+ " milliseconds.", e);
179 			publishException(e);
180 			sleepBeforeRecoveryAttempt();
181 		}
182 		else {
183 			logger.debug("Failed to execute listening task. " + e.getClass() + ": " + e.getMessage());
184 		}
185 	}
186 
187 	@SuppressWarnings("unchecked")
188 	private void receiveAndReply() {
189 		byte[] value;
190 		try {
191 			value = this.boundListOperations.rightPop(this.receiveTimeout, TimeUnit.MILLISECONDS);
192 		}
193 		catch (Exception e) {
194 			handlePopException(e);
195 			return;
196 		}
197 		String uuid = null;
198 		if (value != null) {
199 			if (!this.active) {
200 				this.boundListOperations.rightPush(value);
201 				return;
202 			}
203 			uuid = stringSerializer.deserialize(value);
204 			if (uuid == null) {
205 				return;
206 			}
207 			try {
208 				value = this.template.boundListOps(uuid).rightPop(this.receiveTimeout, TimeUnit.MILLISECONDS);
209 			}
210 			catch (Exception e) {
211 				handlePopException(e);
212 				return;
213 			}
214 			Message<Object> requestMessage = null;
215 			if (value != null) {
216 				if (!this.active) {
217 					this.template.boundListOps(uuid).rightPush(value);
218 					byte[] serialized = stringSerializer.serialize(uuid);
219 					if (serialized != null) {
220 						this.boundListOperations.rightPush(serialized);
221 					}
222 					return;
223 				}
224 				if (this.extractPayload) {
225 					Object payload = value;
226 					if (this.serializer != null) {
227 						payload = this.serializer.deserialize(value);
228 						if (payload == null) {
229 							return;
230 						}
231 					}
232 					requestMessage = getMessageBuilderFactory().withPayload(payload).build();
233 				}
234 				else {
235 					try {
236 						requestMessage = (Message<Object>) this.serializer.deserialize(value);
237 						if (requestMessage == null) {
238 							return;
239 						}
240 					}
241 					catch (Exception e) {
242 						throw new MessagingException("Deserialization of Message failed.", e);
243 					}
244 				}
245 				Message<?> replyMessage = sendAndReceiveMessage(requestMessage);
246 				if (replyMessage != null) {
247 					if (this.extractPayload) {
248 						value = extractReplyPayload(replyMessage);
249 					}
250 					else {
251 						if (this.serializer != null) {
252 							value = ((RedisSerializer<Object>) this.serializer).serialize(replyMessage);
253 							if (value == null) {
254 								return;
255 							}
256 						}
257 					}
258 					this.template.boundListOps(uuid + QUEUE_NAME_SUFFIX).leftPush(value);
259 				}
260 			}
261 		}
262 	}
263 
264 	@SuppressWarnings("unchecked")
265 	private byte[] extractReplyPayload(Message<?> replyMessage) {
266 		byte[] value;
267 		if (!(replyMessage.getPayload() instanceof byte[])) {
268 			if (replyMessage.getPayload() instanceof String && !this.serializerExplicitlySet) {
269 				value = stringSerializer.serialize((String) replyMessage.getPayload());
270 			}
271 			else {
272 				value = ((RedisSerializer<Object>) this.serializer).serialize(replyMessage.getPayload());
273 			}
274 		}
275 		else {
276 			value = (byte[]) replyMessage.getPayload();
277 		}
278 		return value;
279 	}

281 	@Override
282 	protected void doStart() {
283 		super.doStart();
284 		if (!this.active) {
285 			this.active = true;
286 			this.restart();
287 		}
288 	}
289 
290 	/**
291 	 * Sleep according to the specified recovery interval.
292 	 * Called between recovery attempts.
293 	 */
294 	private void sleepBeforeRecoveryAttempt() {
295 		if (this.recoveryInterval > 0) {
296 			try {
297 				Thread.sleep(this.recoveryInterval);
298 			}
299 			catch (InterruptedException e) {
300 				logger.debug("Thread interrupted while sleeping the recovery interval");
301 				Thread.currentThread().interrupt();
302 			}
303 		}
304 	}
305 
306 	private void publishException(Exception e) {
307 		if (this.applicationEventPublisher != null) {
308 			this.applicationEventPublisher.publishEvent(new RedisExceptionEvent(this, e));
309 		}
310 		else {
311 			if (logger.isDebugEnabled()) {
312 				logger.debug("No application event publisher for exception: " + e.getMessage());
313 			}
314 		}
315 	}
316 
317 	private void restart() {
318 		this.taskExecutor.execute(new ListenerTask());
319 	}
320 
321 	@Override
322 	protected void doStop(Runnable callback) {
323 		this.stopCallback = callback;
324 		doStop();
325 	}
326 
327 	@Override
328 	protected void doStop() {
329 		super.doStop();
330 		this.active = false;
331 		this.listening = false;
332 	}
333 
334 	public boolean isListening() {
335 		return this.listening;
336 	}
337 
338 	/**
339 	 * Returns the size of the Queue specified by {@link #boundListOperations}. The queue is
340 	 * represented by a Redis list. If the queue does not exist <code>0</code>
341 	 * is returned. See also https://redis.io/commands/llen
342 	 * @return Size of the queue. Never negative.
343 	 */
344 	@ManagedMetric
345 	public long getQueueSize() {
346 		Long size = this.boundListOperations.size();
347 		return size == null ? 0 : size;
348 	}
349 
350 	/**
351 	 * Clear the Redis Queue specified by {@link #boundListOperations}.
352 	 */
353 	@ManagedOperation
354 	public void clearQueue() {
355 		this.boundListOperations.getOperations().delete(this.boundListOperations.getKey());
356 	}

358 
359 	private class ListenerTask implements SchedulingAwareRunnable {
360 
361 		ListenerTask() {
362 		}
363 
364 		@Override
365 		public boolean isLongLived() {
366 			return true;
367 		}
368 
369 		@Override
370 		public void run() {
371 			try {
372 				while (RedisQueueInboundGateway.this.active) {
373 					RedisQueueInboundGateway.this.listening = true;
374 					receiveAndReply();
375 				}
376 			}
377 			finally {
378 				if (RedisQueueInboundGateway.this.active) {
379 					restart();
380 				}
381 				else if (RedisQueueInboundGateway.this.stopCallback != null) {
382 					RedisQueueInboundGateway.this.stopCallback.run();
383 					RedisQueueInboundGateway.this.stopCallback = null;
384 				}
385 			}
386 		}
387 
388 	}

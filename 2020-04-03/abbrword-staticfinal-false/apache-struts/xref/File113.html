19  package org.apache.struts2.interceptor;
20  
21  import com.opensymphony.xwork2.ActionInvocation;
22  import com.opensymphony.xwork2.TextProvider;
23  import com.opensymphony.xwork2.TextProviderFactory;
24  import com.opensymphony.xwork2.interceptor.ValidationAware;
25  import com.opensymphony.xwork2.inject.Inject;
26  import com.opensymphony.xwork2.interceptor.MethodFilterInterceptor;
27  import org.apache.logging.log4j.LogManager;
28  import org.apache.logging.log4j.Logger;
29  import org.apache.struts2.ServletActionContext;
30  import org.apache.struts2.util.TokenHelper;
31  
32  import javax.servlet.http.HttpSession;
33  
34  /**
35   * <!-- START SNIPPET: description -->
36   * <p>
37   * Ensures that only one request per token is processed. This interceptor can make sure that back buttons and double
38   * clicks don't cause un-intended side affects. For example, you can use this to prevent careless users who might double
39   * click on a "checkout" button at an online store. This interceptor uses a fairly primitive technique for when an
40   * invalid token is found: it returns the result <b>invalid.token</b>, which can be mapped in your action configuration.
41   * A more complex implementation, {@link TokenSessionStoreInterceptor}, can provide much better logic for when invalid
42   * tokens are found.
43   * </p>
44   *
45   * <p>
46   * <b>Note:</b> To set a token in your form, you should use the <b>token tag</b>. This tag is required and must be used
47   * in the forms that submit to actions protected by this interceptor. Any request that does not provide a token (using
48   * the token tag) will be processed as a request with an invalid token.
49   * </p>
50   *
51   * <p>
52   * <b>Internationalization Note:</b> The following key could be used to internationalized the action errors generated
53   * by this token interceptor
54   * </p>
55   *
56   * <ul>
57   *    <li>struts.messages.invalid.token</li>
58   * </ul>
59   *
60   * <p>
61   * <b>NOTE:</b> As this method extends off MethodFilterInterceptor, it is capable of
62   * deciding if it is applicable only to selective methods in the action class. See
63   * <code>MethodFilterInterceptor</code> for more info.
64   * </p>
65   *
66   * <!-- END SNIPPET: description -->
67   *
68   * <p><u>Interceptor parameters:</u></p>
69   *
70   * <!-- START SNIPPET: parameters -->
71   *
72   * <ul>
73   *
74   * <li>None</li>
75   *
76   * </ul>
77   *
78   * <!-- END SNIPPET: parameters -->
79   *
80   * <p><u>Extending the interceptor:</u></p>
81   *
82   * <!-- START SNIPPET: extending -->
83   * <p>
84   * While not very common for users to extend, this interceptor is extended by the {@link TokenSessionStoreInterceptor}.
85   * The {@link #handleInvalidToken}  and {@link #handleValidToken} methods are protected and available for more
86   * interesting logic, such as done with the token session interceptor.
87   * </p>
88   * <!-- END SNIPPET: extending -->
89   *
90   * <p><u>Example code:</u></p>
91   *
92   * <pre>
93   * <!-- START SNIPPET: example -->
94   *
95   * &lt;action name="someAction" class="com.examples.SomeAction"&gt;
96   *     &lt;interceptor-ref name="token"/&gt;
97   *     &lt;interceptor-ref name="basicStack"/&gt;
98   *     &lt;result name="success"&gt;good_result.ftl&lt;/result&gt;
99   * &lt;/action&gt;
100  *
101  * &lt;-- In this case, myMethod of the action class will not
102  *        get checked for invalidity of token --&gt;
103  * &lt;action name="someAction" class="com.examples.SomeAction"&gt;
104  *     &lt;interceptor-ref name="token"&gt;
105  *        &lt;param name="excludeMethods"&gt;myMethod&lt;/param&gt;
106  *     &lt;/interceptor-ref name="token"/&gt;
107  *     &lt;interceptor-ref name="basicStack"/&gt;
108  *     &lt;result name="success"&gt;good_result.ftl&lt;/result&gt;
109  * &lt;/action&gt;
110  *
111  * <!-- END SNIPPET: example -->
112  * </pre>
113  *
114  * @see TokenSessionStoreInterceptor
115  * @see TokenHelper
116  */
117 public class TokenInterceptor extends MethodFilterInterceptor {
118 
119     private static final Logger LOG = LogManager.getLogger(TokenInterceptor.class);
120 
121     public static final String INVALID_TOKEN_CODE = "invalid.token";
122 
123     private static final String INVALID_TOKEN_MESSAGE_KEY = "struts.messages.invalid.token";
124     private static final String DEFAULT_ERROR_MESSAGE = "The form has already been processed or no token was supplied, please try again.";
125 
126     private TextProvider textProvider;
127 
128     @Inject
129     public void setTextProviderFactory(TextProviderFactory textProviderFactory) {
130         this.textProvider = textProviderFactory.createInstance(getClass());
131     }
132 
133     /**
134      * @see com.opensymphony.xwork2.interceptor.MethodFilterInterceptor#doIntercept(com.opensymphony.xwork2.ActionInvocation)
135      */
136     @Override
137     protected String doIntercept(ActionInvocation invocation) throws Exception {
138         LOG.debug("Intercepting invocation to check for valid transaction token.");
139         return handleToken(invocation);
140     }
141 
142     protected String handleToken(ActionInvocation invocation) throws Exception {
143         //see WW-2902: we need to use the real HttpSession here, as opposed to the map
144         //that wraps the session, because a new wrap is created on every request
145         HttpSession session = ServletActionContext.getRequest().getSession(true);
146         synchronized (session.getId().intern()) {
147             if (!TokenHelper.validToken()) {
148                 return handleInvalidToken(invocation);
149             }
150         }
151         return handleValidToken(invocation);
152     }
153 
154     /**
155      * Determines what to do if an invalid token is provided. If the action implements {@link ValidationAware}
156      *
157      * @param invocation the action invocation where the invalid token failed
158      * @return the return code to indicate should be processed
159      * @throws Exception when any unexpected error occurs.
160      */
161     protected String handleInvalidToken(ActionInvocation invocation) throws Exception {
162         Object action = invocation.getAction();
163         String errorMessage = getErrorMessage(invocation);
164 
165         if (action instanceof ValidationAware) {
166             ((ValidationAware) action).addActionError(errorMessage);
167         } else {
168             LOG.warn(errorMessage);
169         }
170 
171         return INVALID_TOKEN_CODE;
172     }
173 
174     protected String getErrorMessage(ActionInvocation invocation) {
175         Object action = invocation.getAction();
176         if (action instanceof TextProvider) {
177             return ((TextProvider) action).getText(INVALID_TOKEN_MESSAGE_KEY, DEFAULT_ERROR_MESSAGE);
178         }
179         return textProvider.getText(INVALID_TOKEN_MESSAGE_KEY, DEFAULT_ERROR_MESSAGE);
180     }
181 
182     /**
183      * Called when a valid token is found. This method invokes the action by can be changed to do something more
184      * interesting.
185      *
186      * @param invocation the action invocation
187      * @return invocation result
188      * @throws Exception when any unexpected error occurs.
189      */
190     protected String handleValidToken(ActionInvocation invocation) throws Exception {
191         return invocation.invoke();
192     }
193 
194 }

272     private static boolean skipReportingMissingClass(@CheckForNull @DottedClassName String missing) {
273         return missing == null || missing.length() == 0 || missing.charAt(0) == '[' || missing.endsWith("package-info");

276     private static @CheckForNull
277     RepositoryLookupFailureCallback getCurrentLookupFailureCallback() {
278         AnalysisContext currentAnalysisContext2 = currentAnalysisContext();
279         if (currentAnalysisContext2 == null) {
280             return null;
281         }
282         if (currentAnalysisContext2.missingClassWarningsSuppressed) {
283             return null;

289      * file a ClassNotFoundException with the lookupFailureCallback
290      *
291      * @see #getLookupFailureCallback()
292      */
293     static public void reportMissingClass(ClassNotFoundException e) {
294         requireNonNull(e, "argument is null");
295         String missing = AbstractBugReporter.getMissingClassName(e);
296         if (skipReportingMissingClass(missing)) {
297             return;
298         }
299         if (!analyzingApplicationClass()) {
300             return;
301         }
302 
303         RepositoryLookupFailureCallback lookupFailureCallback = getCurrentLookupFailureCallback();
304         if (lookupFailureCallback != null) {
305             lookupFailureCallback.reportMissingClass(e);
306         }
307     }

314     static public boolean analyzingApplicationClass() {
315         AnalysisContext context = AnalysisContext.currentAnalysisContext();
316         if (context == null) {
317             return false;
318         }
319         ClassDescriptor clazz = context.getClassBeingAnalyzed();
320         if (clazz == null) {
321             return false;
322         }
323         return context.isApplicationClass(clazz);
324     }

331     static public void reportMissingClass(ClassDescriptor c) {
332         requireNonNull(c, "argument is null");
333         if (!analyzingApplicationClass()) {
334             return;
335         }
336         String missing = c.getDottedClassName();
337         if (missing.length() == 1) {
338             System.out.println(c);
339         }
340         if (skipReportingMissingClass(missing)) {
341             return;
342         }
343         RepositoryLookupFailureCallback lookupFailureCallback = getCurrentLookupFailureCallback();
344         if (lookupFailureCallback != null) {
345             lookupFailureCallback.reportMissingClass(c);
346         }
347     }

349     /**
350      * Report an error
351      */
352     static public void logError(String msg, Exception e) {
353         AnalysisContext currentAnalysisContext2 = currentAnalysisContext();
354         if (currentAnalysisContext2 == null) {
355             if (e instanceof NoSuchBugPattern) {
356                 return;
357             }
358             /*
359             if (false && SystemProperties.ASSERTIONS_ENABLED) {
360                 AssertionError e2 = new AssertionError("Exception logged with no analysis context");
361                 e2.initCause(e);
362                 throw e2;
363             }
364              */
365             e.printStackTrace(System.err);
366             return;
367         }
368         if (e instanceof MissingClassException) {
369             reportMissingClass(((MissingClassException) e).getClassNotFoundException());
370             return;
371         }
372         if (e instanceof edu.umd.cs.findbugs.classfile.MissingClassException) {
373             reportMissingClass(((edu.umd.cs.findbugs.classfile.MissingClassException) e).toClassNotFoundException());
374             return;
375         }
376 
377         RepositoryLookupFailureCallback lookupFailureCallback = currentAnalysisContext2.getLookupFailureCallback();
378         if (lookupFailureCallback != null) {
379             lookupFailureCallback.logError(msg, e);
380         }
381     }

465             /* ClassContext classContext =*/ analysisCache.getClassAnalysis(ClassContext.class, desc);

478     public boolean isTooBig(ClassDescriptor desc) {
479         IAnalysisCache analysisCache = Global.getAnalysisCache();
480 
481         try {
482             ClassContext classContext = analysisCache.getClassAnalysis(ClassContext.class, desc);
483             ClassData classData = analysisCache.getClassAnalysis(ClassData.class, desc);
484             if (classData.getData().length > 1000000) {
485                 return true;
486             }
487             try {
488                 JavaClass javaClass = classContext.getJavaClass();
489                 if (javaClass.getMethods().length > 1000) {
490                     return true;
491                 }
492             } catch (RuntimeException e) {
493                 AnalysisContext.logError("Error parsing class " + desc
494                         + " from " + classData.getCodeBaseEntry().getCodeBase(), e);
495                 return true;
496             }
497 
498         } catch (RuntimeException e) {
499             AnalysisContext.logError("Error getting class data for " + desc, e);
500             return true;
501         } catch (CheckedAnalysisException e) {
502             AnalysisContext.logError("Could not get class context for "  + desc, e);
503             return true;
504         }
505         return false;
506     }

522     /**
523      * This is equivalent to Repository.lookupClass() or this.lookupClass(),
524      * except it uses the original Repository instead of the current one.
525      *
526      * This can be important because URLClassPathRepository objects are closed
527      * after an analysis, so JavaClass objects obtained from them are no good on
528      * subsequent runs.
529      *
530      * @param className
531      *            the name of the class
532      * @return the JavaClass representing the class
533      * @throws ClassNotFoundException
534      */
535     public static JavaClass lookupSystemClass(@Nonnull String className) throws ClassNotFoundException {
536         // TODO: eventually we should move to our own thread-safe repository
537         // implementation
538         requireNonNull (className, "className is null");
539         if (originalRepository == null) {
540             throw new IllegalStateException("originalRepository is null");
541         }
542 
543         JavaClass clazz = originalRepository.findClass(className);
544         if(clazz != null){
545             return clazz;
546         }
547         // XXX workaround for system classes missing on Java 9
548         // Not sure if we BCEL update, but this seem to work in simple cases
549         return AnalysisContext.currentAnalysisContext().lookupClass(className);
550     }

552     /**
553      * Lookup a class's source file
554      *
555      * @param dottedClassName
556      *            the name of the class
557      * @return the source file for the class, or
558      *         {@link SourceLineAnnotation#UNKNOWN_SOURCE_FILE} if unable to
559      *         determine
560      */
561     public final String lookupSourceFile(@Nonnull @DottedClassName String dottedClassName) {
562         requireNonNull(dottedClassName, "className is null");
563         try {
564             XClass xClass = Global.getAnalysisCache().getClassAnalysis(XClass.class,
565                     DescriptorFactory.createClassDescriptorFromDottedClassName(dottedClassName));
566             String name = xClass.getSource();
567             if (name == null) {
568                 return SourceLineAnnotation.UNKNOWN_SOURCE_FILE;
569             }
570             return name;
571         } catch (CheckedAnalysisException e) {
572             return SourceLineAnnotation.UNKNOWN_SOURCE_FILE;
573         }
574     }

670      * Load an interprocedural property database.
671      *
672      * @param <DatabaseType>
673      *            actual type of the database
674      * @param <KeyType>
675      *            type of key (e.g., method or field)
676      * @param <Property>
677      *            type of properties stored in the database
678      * @param database
679      *            the empty database object
680      * @param fileName
681      *            file to load database from
682      * @param description
683      *            description of the database (for diagnostics)
684      * @return the database object, or null if the database couldn't be loaded
685      */
686     public <DatabaseType extends PropertyDatabase<KeyType, Property>, KeyType extends FieldOrMethodDescriptor, Property> DatabaseType loadPropertyDatabase(
687             DatabaseType database, String fileName, String description) {
688         try {
689             File dbFile = new File(getDatabaseInputDir(), fileName);
690             if (DEBUG) {
691                 System.out.println("Loading " + description + " from " + dbFile.getPath() + "...");
692             }
693 
694             database.readFromFile(dbFile.getPath());
695             return database;
696         } catch (IOException e) {
697             getLookupFailureCallback().logError("Error loading " + description, e);
698         } catch (PropertyDatabaseFormatException e) {
699             getLookupFailureCallback().logError("Invalid " + description, e);
700         }
701         return null;
702     }

721     public <DatabaseType extends PropertyDatabase<KeyType, Property>, KeyType extends FieldOrMethodDescriptor, Property> DatabaseType loadPropertyDatabaseFromResource(
722             DatabaseType database, String resourceName, String description) {
723         try {
724             if (DEBUG) {
725                 System.out.println("Loading default " + description + " from " + resourceName + " @ "
726                         + database.getClass().getResource(resourceName) + " ... ");
727             }
728             try(InputStream in = database.getClass().getResourceAsStream(resourceName)){
729                 if (in == null) {
730                     AnalysisContext.logError("Unable to load " + description + " from resource " + resourceName);
731                 } else {
732                     database.read(in);
733                 }
734             }
735             return database;
736         } catch (IOException e) {
737             getLookupFailureCallback().logError("Error loading " + description, e);
738         } catch (PropertyDatabaseFormatException e) {
739             getLookupFailureCallback().logError("Invalid " + description, e);
740         }
741         return null;
742     }

933     /**
934      * Lookup a class.
935      * <em>Use this method instead of Repository.lookupClass().</em>
936      *
937      * @param className
938      *            the name of the class
939      * @return the JavaClass representing the class
940      * @throws ClassNotFoundException
941      *             (but not really)
942      */
943     public JavaClass lookupClass(@Nonnull @DottedClassName String className) throws ClassNotFoundException {
944         try {
945             if (className.length() == 0) {
946                 throw new IllegalArgumentException("Class name is empty");
947             }
948             if (!ClassName.isValidClassName(className)) {
949                 throw new ClassNotFoundException("Invalid class name: " + className);
950             }
951             return Global.getAnalysisCache().getClassAnalysis(JavaClass.class,
952                     DescriptorFactory.instance().getClassDescriptor(ClassName.toSlashedClassName(className)));
953         } catch (CheckedAnalysisException e) {
954             throw new ClassNotFoundException("Class not found: " + className, e);
955         }
956     }

962     public void setAppClassList(List<ClassDescriptor> appClassCollection) {
963         // FIXME: we really should drive the progress callback here
964         HashSet<ClassDescriptor> appSet = new HashSet<>(appClassCollection);
965 
966         Collection<ClassDescriptor> allClassDescriptors = new ArrayList<>(DescriptorFactory.instance()
967                 .getAllClassDescriptors());
968         for (ClassDescriptor appClass : allClassDescriptors) {
969             try {
970                 XClass xclass = currentXFactory().getXClass(appClass);
971 
972                 if (xclass == null) {
973                     continue;
974                 }
975 
976                 // Add the application class to the database
977                 if (appSet.contains(appClass)) {
978                     getSubtypes2().addApplicationClass(xclass);
979                 } else if (xclass instanceof ClassInfo) {
980                     getSubtypes2().addClass(xclass);
981                 }
982 
983             } catch (Exception e) {
984                 AnalysisContext.logError("Unable to get XClass for " + appClass, e);
985             }
986         }
987 
988         if (true && Subtypes2.DEBUG) {
989             System.out.println(getSubtypes2().getGraph().getNumVertices() + " vertices in inheritance graph");
990         }
991     }

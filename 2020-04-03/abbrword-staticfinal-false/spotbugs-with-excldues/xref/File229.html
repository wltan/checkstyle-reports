23  package edu.umd.cs.findbugs.ba.npe;
24  
25  import edu.umd.cs.findbugs.SystemProperties;
26  
27  /**
28   * Knobs for null value analysis.
29   *
30   * @author David Hovemeyer
31   */
32  public interface IsNullValueAnalysisFeatures {
33  
34      /**
35       * Do not downgrade "null on simple path" values to "null on complex path"
36       * on control splits.
37       */
38      public static final boolean NO_SPLIT_DOWNGRADE_NSP = SystemProperties.getBoolean("inva.noSplitDowngradeNSP");
39  
40      /**
41       * Don't consider switch default cases as exception paths.
42       */
43      public static final boolean NO_SWITCH_DEFAULT_AS_EXCEPTION = SystemProperties.getBoolean("inva.noSwitchDefaultAsException");
44  
45      /**
46       * Keep track of an extra branch, so we can distinguish conditionally-null
47       * values with two branches from conditionally-null values with three or
48       * more branches.
49       */
50      public static final boolean NCP_EXTRA_BRANCH = SystemProperties.getBoolean("inva.ncpExtraBranch");
51  
52      /**
53       * If this property is true, then we assume parameters and return values can
54       * be null (but aren't definitely null).
55       */
56      public static final boolean UNKNOWN_VALUES_ARE_NSP = SystemProperties.getBoolean("findbugs.nullderef.assumensp");
57  
58      /**
59       * If this property is true, then resolution of nullness annotations is
60       * based on JSR-305 type qualifiers rather than an AnnotationDatabase.
61       */
62      public static final boolean USE_TYPE_QUALIFIERS = true; // SystemProperties.getBoolean("findbugs.npe.tq");
63  
64  }

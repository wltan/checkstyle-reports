38      private final static char[] hexDigits = "0123456789abcdef".toCharArray();
39  
40      public static String toString(byte[] b) {
41          StringBuffer sb = new StringBuffer(b.length * 3);
42          for (int i = 0; i < b.length; i++) {
43              int k = b[i] & 0xff;
44              if (i != 0) {
45                  sb.append(':');
46              }
47              sb.append(hexDigits[k >>> 4]);
48              sb.append(hexDigits[k & 0xf]);
49          }
50          return sb.toString();
51      }
52  
53      public static byte[] parse(String s) {
54          try {
55              int n = s.length();
56              ByteArrayOutputStream out = new ByteArrayOutputStream(n / 3);
57              StringReader r = new StringReader(s);
58              while (true) {
59                  int b1 = nextNibble(r);
60                  if (b1 < 0) {
61                      break;
62                  }
63                  int b2 = nextNibble(r);
64                  if (b2 < 0) {
65                      throw new RuntimeException("Invalid string " + s);
66                  }
67                  int b = (b1 << 4) | b2;
68                  out.write(b);
69              }
70              return out.toByteArray();
71          } catch (IOException e) {
72              throw new RuntimeException(e);
73          }
74      }
75  
76      public static byte[] b(String s) {
77          return parse(s);
78      }
79  
80      private static int nextNibble(StringReader r) throws IOException {
81          while (true) {
82              int ch = r.read();
83              if (ch == -1) {
84                  return -1;
85              } else if ((ch >= '0') && (ch <= '9')) {
86                  return ch - '0';
87              } else if ((ch >= 'a') && (ch <= 'f')) {
88                  return ch - 'a' + 10;
89              } else if ((ch >= 'A') && (ch <= 'F')) {
90                  return ch - 'A' + 10;
91              }
92          }
93      }

99      static class DigestTest extends Test {
100         private final String alg;
101         private final byte[] data;
102         private final byte[] digest;
103         DigestTest(String alg, byte[] data, byte[] digest) {
104             this.alg = alg;
105             this.data = data;
106             this.digest = digest;
107         }
108         void run(Provider p) throws Exception {
109             MessageDigest md;
110             if (alg.equals("MD4")) {
111                 md = sun.security.provider.MD4.getInstance();
112             } else {
113                 md = MessageDigest.getInstance(alg, p);
114             }
115             md.update(data);
116             byte[] myDigest = md.digest();
117             if (Arrays.equals(digest, myDigest) == false) {
118                 System.out.println("Digest test for " + alg + " failed:");
119                 if (data.length < 256) {
120                     System.out.println("data: " + DigestKAT.toString(data));
121                 }
122                 System.out.println("dig:  " + DigestKAT.toString(digest));
123                 System.out.println("out:  " + DigestKAT.toString(myDigest));
124                 throw new Exception("Digest test for " + alg + " failed");
125             }
126 //          System.out.println("Passed " + alg);
127         }
128     }

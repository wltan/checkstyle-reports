25   * @test
26   * @bug 6239400
27   * @summary Tests NotificationBuffer doesn't hold locks when adding listeners.
28   * @author Eamonn McManus
29   * @run clean NotificationBufferDeadlockTest
30   * @run build NotificationBufferDeadlockTest
31   * @run main NotificationBufferDeadlockTest
32   */

34  import java.lang.reflect.InvocationHandler;
35  import java.lang.reflect.Method;
36  import java.lang.reflect.Proxy;
37  import java.net.MalformedURLException;
38  import java.util.List;

44  /*
45   * Regression test for a rare but not unheard-of deadlock condition in
46   * the notification buffer support for connector servers.
47   * See bug 6239400 for the description of the bug and the example that
48   * showed it up.
49   *
50   * Here we test that, when the connector server adds its listener to an
51   * MBean, it is not holding a lock that would prevent another thread from
52   * emitting a notification (from that MBean or another one).  This is
53   * important, because we don't know how user MBeans might implement
54   * NotificationBroadcaster.addNotificationListener, and in particular we
55   * can't be sure that the method is well-behaved and can never do a
56   * blocking operation, such as attempting to acquire a lock that is also
57   * acquired when notifications are emitted.
58   *
59   * The test creates a special MBean whose addNotificationListener method
60   * does the standard addNotificationListener logic inherited
61   * from NotificationBroadcasterSupport, then
62   * creates another thread that emits a notification from the same MBean.
63   * The addNotificationListener method waits for this thread to complete.
64   * If the notification buffer logic is incorrect, then emitting the
65   * notification will attempt to acquire the lock on the buffer, but that
66   * lock is being held by the thread that called addNotificationListener,
67   * so there will be deadlock.
68   *
69   * We use this DeadlockMBean several times.  First, we create one and then
70   * add a remote listener to it.  The first time you add a remote listener
71   * through a connector server, the connector server adds its own listener
72   * to all NotificationBroadcaster MBeans.  If it holds a lock while doing
73   * this, we will see deadlock.
74   *
75   * Then we create a second DeadlockMBean.  When a new MBean is created that
76   * is a NotificationBroadcaster, the connector server adds its listener to
77   * that MBean too.  Again if it holds a lock while doing this, we will see
78   * deadlock.
79   *
80   * Finally, we do some magic with MBeanServerForwarders so that while
81   * queryNames is running (to find MBeans to which listeners must be added)
82   * we will create new MBeans.  This tests that this tricky situation is
83   * handled correctly.  It also tests the queryNames that is run when the
84   * notification buffer is being destroyed (to remove the listeners).
85   *
86   * We cause all of our test MBeans to emit exactly one notification and
87   * check that we have received exactly one notification from each MBean.
88   * If the logic for adding the notification buffer's listener is incorrect
89   * we could remove zero or two notifications from an MBean.
90   */
91  public class NotificationBufferDeadlockTest {
92      public static void main(String[] args) throws Exception {
93          System.out.println("Check no deadlock if notif sent while initial " +
94                             "remote listeners being added");
95          final String[] protos = {"rmi", "iiop", "jmxmp"};
96          for (String p : protos) {
97              try {
98                  test(p);
99              } catch (Exception e) {
100                 System.out.println("TEST FAILED: GOT EXCEPTION:");
101                 e.printStackTrace(System.out);
102                 failure = e.toString();
103             }
104         }
105         if (failure == null)
106             return;
107         else
108             throw new Exception("TEST FAILED: " + failure);
109     }
110 
111     private static void test(String proto) throws Exception {
112         System.out.println("Testing protocol " + proto);
113         MBeanServer mbs = MBeanServerFactory.newMBeanServer();
114         ObjectName testName = newName();
115         DeadlockTest test = new DeadlockTest();
116         mbs.registerMBean(test, testName);
117         JMXServiceURL url = new JMXServiceURL("service:jmx:" + proto + ":///");
118         JMXConnectorServer cs;
119         try {
120             cs =
121                 JMXConnectorServerFactory.newJMXConnectorServer(url, null, mbs);
122         } catch (MalformedURLException e) {
123             System.out.println("...protocol not supported, ignoring");
124             return;
125         }
126 
127         MBeanServerForwarder createDuringQueryForwarder = (MBeanServerForwarder)
128             Proxy.newProxyInstance(new Object() {}.getClass().getClassLoader(),
129                                    new Class[] {MBeanServerForwarder.class},
130                                    new CreateDuringQueryInvocationHandler());
131         cs.setMBeanServerForwarder(createDuringQueryForwarder);
132         cs.start();
133         JMXServiceURL addr = cs.getAddress();
134         JMXConnector cc = JMXConnectorFactory.connect(addr);
135         MBeanServerConnection mbsc = cc.getMBeanServerConnection();
136         try {
137             String fail = test(mbsc, testName);
138             if (fail != null)
139                 System.out.println("FAILED: " + fail);
140             failure = fail;
141         } finally {
142             cc.close();
143             cs.stop();
144         }
145     }
146 
147     private static String test(MBeanServerConnection mbsc,
148                                ObjectName testName) throws Exception {
149 
150         NotificationListener dummyListener = new NotificationListener() {
151             public void handleNotification(Notification n, Object h) {
152             }
153         };
154         thisFailure = null;
155         mbsc.addNotificationListener(testName, dummyListener, null, null);
156         if (thisFailure != null)
157             return thisFailure;
158         ObjectName newName = newName();
159         mbsc.createMBean(DeadlockTest.class.getName(), newName);
160         if (thisFailure != null)
161             return thisFailure;
162         Set<ObjectName> names =
163             mbsc.queryNames(new ObjectName("d:type=DeadlockTest,*"), null);
164         System.out.printf("...found %d test MBeans\n", names.size());
165 
166         sources.clear();
167         countListener = new MyListener(names.size());
168 
169         for (ObjectName name : names)
170             mbsc.addNotificationListener(name, countListener, null, null);
171         if (thisFailure != null)
172             return thisFailure;
173         for (ObjectName name : names)
174             mbsc.invoke(name, "send", null, null);
175 
176         if (!countListener.waiting(MAX_WAITING_TIME)) {
177             return "did not get " + names.size() + " notifs as expected\n";
178         }
179 
180         if (!sources.containsAll(names))
181             return "missing names: " + sources;
182         return thisFailure;
183     }

185     public static interface DeadlockTestMBean {
186         public void send();
187     }
188 
189     public static class DeadlockTest extends NotificationBroadcasterSupport
190             implements DeadlockTestMBean {
191         @Override
192         public void addNotificationListener(NotificationListener listener,
193                                             NotificationFilter filter,
194                                             Object handback) {
195             super.addNotificationListener(listener, filter, handback);
196             Thread t = new Thread() {
197                 @Override
198                 public void run() {
199                     Notification n =
200                         new Notification("type", DeadlockTest.this, 0L);
201                     DeadlockTest.this.sendNotification(n);
202                 }
203             };
204             t.start();
205             try {
206                 t.join(5000L);
207             } catch (Exception e) {
208                 thisFailure = "Join exception: " + e;
209             }
210             if (t.isAlive())
211                 thisFailure = "Deadlock detected";
212         }
213 
214         public void send() {
215             sendNotification(new Notification(TESTING_TYPE, DeadlockTest.this, 1L));
216         }
217     }

219     private static class CreateDuringQueryInvocationHandler
220             implements InvocationHandler {
221         public Object invoke(Object proxy, Method m, Object[] args)
222                 throws Throwable {
223             if (m.getName().equals("setMBeanServer")) {
224                 mbs = (MBeanServer) args[0];
225                 return null;
226             }
227             createMBeanIfQuery(m);
228             Object ret = m.invoke(mbs, args);
229             createMBeanIfQuery(m);
230             return ret;
231         }
232 
233         private void createMBeanIfQuery(Method m) throws InterruptedException {
234             if (m.getName().equals("queryNames")) {
235                 Thread t = new Thread() {
236                     public void run() {
237                         try {
238                             mbs.createMBean(DeadlockTest.class.getName(),
239                                             newName());
240                         } catch (Exception e) {
241                             e.printStackTrace();
242                             thisFailure = e.toString();
243                         }
244                     }
245                 };
246                 t.start();
247                 t.join(5000);
248                 if (t.isAlive())
249                     failure = "Query deadlock detected";
250             }
251         }
252 
253         private MBeanServer mbs;
254     }

256     private static synchronized ObjectName newName() {
257         try {
258             return new ObjectName("d:type=DeadlockTest,instance=" +
259                                   ++nextNameIndex);
260         } catch (MalformedObjectNameException e) {
261             throw new IllegalArgumentException("bad ObjectName", e);
262         }
263     }
264 
265     private static class MyListener implements NotificationListener {
266         public MyListener(int waitNB) {
267             this.waitNB= waitNB;
268         }
269 
270         public void handleNotification(Notification n, Object h) {
271             System.out.println("MyListener got: "+n.getSource()+" "+n.getType());
272 
273             synchronized(this) {
274                 if (TESTING_TYPE.equals(n.getType())) {
275                     sources.add((ObjectName) n.getSource());
276 
277                     if (sources.size() == waitNB) {
278                         this.notifyAll();
279                     }
280                 }
281             }
282         }
283 
284         public boolean waiting(long timeout) {
285             final long startTime = System.currentTimeMillis();
286             long toWait = timeout;
287 
288             synchronized(this) {
289                 while(sources.size() < waitNB && toWait > 0) {
290                     try {
291                         this.wait(toWait);
292                     } catch (InterruptedException ire) {
293                         break;
294                     }
295 
296                     toWait = timeout -
297                         (System.currentTimeMillis() - startTime);
298                 }
299             }
300 
301             return sources.size() == waitNB;
302         }
303 
304         private final int waitNB;
305     }

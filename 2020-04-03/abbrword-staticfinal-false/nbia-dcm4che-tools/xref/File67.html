100     public void setConceptNameCodes(String conceptNames) {
101         if ( conceptNames == null || conceptNames.trim().length() < 1 ) {
102             resetToDefaultCodes();
103         } else {
104             conceptNameCodeLists = parseConceptNameCodeString( conceptNames );
105         }
106     }

108     public String getConceptNameCodes() {
109         String assign, grpID, code;
110         Map<String, List<Dataset>> map;
111         TreeMap<String, Map<String, List<String>>> resultMap =new TreeMap<String, Map<String, List<String>>>();
112         List<String> cuidList;
113         Map<String,List<String>> cuidMap;
114         for ( Map.Entry entry : conceptNameCodeLists.entrySet() ) {
115             assign = (String) entry.getKey();
116             map = (Map<String, List<Dataset>>) entry.getValue();
117             for ( Map.Entry entryCuidCode : map.entrySet() ) {
118                 grpID = (String)entryCuidCode.getKey();
119                 for ( Dataset ds : (List<Dataset>) entryCuidCode.getValue() ) {
120                     code = toCodeString(ds);
121                     cuidMap = resultMap.get(code);
122                     if ( cuidMap == null ) {
123                         cuidMap = new TreeMap<String, List<String>>();
124                         resultMap.put( code, cuidMap);
125                     }
126                     cuidList = cuidMap.get(assign);
127                     if ( cuidList == null ) {
128                         cuidList = new ArrayList<String>();
129                         cuidMap.put(assign, cuidList);
130                     }
131                     if ( grpID != null )
132                         cuidList.add(grpID);
133                 }
134             }
135         }
136         StringBuffer sb = new StringBuffer();
137         for ( Map.Entry entry : resultMap.entrySet() ) {
138             code = (String)entry.getKey();
139             sb.append(code).append(':');
140             for ( Map.Entry entryAssign : ((Map<String, List<String>>) entry.getValue()).entrySet() ) {
141                 sb.append(entryAssign.getKey());
142                 cuidList = (List<String>)entryAssign.getValue();
143                 if ( cuidList.size() > 0 ) {
144                     sb.append('(');
145                     for ( String id : cuidList ) {
146                         sb.append(id).append(',');
147                     }
148                     sb.setLength(sb.length()-1);
149                     sb.append(')');
150                 }
151                 sb.append(',');
152             }
153             sb.setLength(sb.length()-1); 
154             sb.append(System.getProperty("line.separator", "\n"));
155         }
156         return sb.toString();
157     }
158     private void resetToDefaultCodes() {
159         conceptNameCodeLists.clear();
160         HashMap<String, List<Dataset>> mapAll = new HashMap<String, List<Dataset>>();
161         HashMap<String, List<Dataset>> mapR = new HashMap<String, List<Dataset>>();
162         mapR.put(null, ConceptNameCodeConfig.getDefaultRadiologyConceptNameCodes());
163         mapAll.put(null, ConceptNameCodeConfig.getDefaultRadiologyConceptNameCodes());
164         conceptNameCodeLists.put(ALL, mapAll);
165         conceptNameCodeLists.put(RADIOLOGY, mapR);
166         HashMap<String, List<Dataset>> mapC = new HashMap<String, List<Dataset>>();
167         mapC.put(null, ConceptNameCodeConfig.getDefaultCardiologyConceptNameCodes());
168         mapAll.get(null).addAll(ConceptNameCodeConfig.getDefaultCardiologyConceptNameCodes());
169         conceptNameCodeLists.put(CARDIOLOGY, mapC);
170     }

172     /**
173      * 
174      * Map<summary_id, Map<cuid_grp_id, List<code_dataset>> 
175      * @param conceptNames Configuration String.
176      * @return
177      */
178     public Map<String, Map<String, List<Dataset>>> parseConceptNameCodeString( String conceptNames ) {
179         Map<String, Map<String, List<Dataset>>> map = new HashMap<String, Map<String, List<Dataset>>>();
180         if ( conceptNames != null && conceptNames.trim().length() > 0 ) {
181             StringTokenizer stCode, stAssign;
182             String line, assign;
183             Dataset code;
184             int pos, pos1;
185             for ( StringTokenizer stLine = new StringTokenizer(conceptNames, "\r\n;") ; stLine.hasMoreTokens() ; ) {
186                 line = stLine.nextToken();
187                 pos = line.indexOf(':');
188                 stCode = new StringTokenizer( line.substring(0,pos), "^");
189                 code = createCodeDS( stCode.nextToken(), 
190                         stCode.hasMoreTokens() ? stCode.nextToken():"LN",
191                                 stCode.hasMoreTokens() ? stCode.nextToken():null );
192 
193                 for ( stAssign = new StringTokenizer( line.substring(++pos), ",") ; stAssign.hasMoreTokens() ; ) {
194                     assign = stAssign.nextToken();
195                     if ( (pos1 = assign.indexOf('(')) != -1) {
196                         String cuidGrps = assign.substring(++pos1,assign.length()-1);
197                         assign = assign.substring(0, --pos1);
198                         for ( StringTokenizer stCuidGrp = new StringTokenizer( cuidGrps , ",") ; 
199                         stCuidGrp.hasMoreTokens() ; ) {
200                             addAssign(assign, stCuidGrp.nextToken(), code, map);
201                         }
202                     } else {
203                         addAssign(assign, null, code, map);
204                     }
205                 }
206             }
207             return map;
208         }
209         return null;
210 
211     }
212 
213     private void addAssign(String assign, String cuidGrpId, Dataset code,
214             Map<String, Map<String, List<Dataset>>> map) {
215         Map<String, List<Dataset>> map1 = map.get(assign);
216         if ( map1==null) {
217             map1 = new HashMap<String, List<Dataset>>();
218             map.put(assign, map1);
219         }
220         List<Dataset> l = map1.get(cuidGrpId);
221         if ( l == null ) {
222             l = new ArrayList<Dataset>();
223             map1.put(cuidGrpId, l);
224         }
225         l.add(code);
226     }

229     public String toCodeString(Dataset ds) {
230         StringBuffer sb = new StringBuffer();
231         sb.append(ds.getString(Tags.CodeValue)).append('^');
232         sb.append(ds.getString(Tags.CodingSchemeDesignator));
233         if (ds.getString(Tags.CodeMeaning) != null )
234             sb.append('^').append(ds.getString(Tags.CodeMeaning));
235         return sb.toString();
236     }

266     private static Dataset createCodeDS( String value, String design, String meaning ) {
267         Dataset ds = factory.newDataset();
268         ds.putSH(Tags.CodeValue, value);
269         ds.putSH(Tags.CodingSchemeDesignator, design);
270         if (meaning != null) {
271             ds.putLO(Tags.CodeMeaning, meaning);
272         }
273         return ds;
274     }

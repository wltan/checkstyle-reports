38  /**
39   * Global information about the source code for an application. Currently, this
40   * object contains a map of source line information for fields and classes
41   * (items we don't get line number information for directly in classfiles), and
42   * also source line information for methods that don't appear directly in
43   * classfiles, such as abstract and native methods.
44   *
45   * @author David Hovemeyer
46   */
47  public class SourceInfoMap {
48      static class FieldDescriptor implements Comparable<FieldDescriptor> {
49          String className;
50  
51          String fieldName;
52  
53          public FieldDescriptor(String className, String fieldName) {
54              this.className = className;
55              this.fieldName = fieldName;
56          }
57  
58          @Override
59          public String toString() {
60              return className + "." + fieldName;
61          }
62  
63          /*
64           * (non-Javadoc)
65           *
66           * @see java.lang.Comparable#compareTo(T)
67           */
68          @Override
69          public int compareTo(FieldDescriptor o) {
70              int cmp = className.compareTo(o.className);
71              if (cmp != 0) {
72                  return cmp;
73              }
74              return fieldName.compareTo(o.fieldName);
75          }
76  
77          /*
78           * (non-Javadoc)
79           *
80           * @see java.lang.Object#hashCode()
81           */
82          @Override
83          public int hashCode() {
84              return 1277 * className.hashCode() + fieldName.hashCode();
85          }
86  
87          /*
88           * (non-Javadoc)
89           *
90           * @see java.lang.Object#equals(java.lang.Object)
91           */
92          @Override
93          public boolean equals(Object obj) {
94              if (obj == null || obj.getClass() != this.getClass()) {
95                  return false;
96              }
97              FieldDescriptor other = (FieldDescriptor) obj;
98              return className.equals(other.className) && fieldName.equals(other.fieldName);
99          }
100     }

102     static class MethodDescriptor implements Comparable<MethodDescriptor> {
103         private final String className;
104 
105         private final String methodName;
106 
107         private final String methodSignature;
108 
109         public MethodDescriptor(String className, String methodName, String methodSignature) {
110             this.className = className;
111             this.methodName = methodName;
112             this.methodSignature = methodSignature;
113         }
114 
115         @Override
116         public String toString() {
117             return className + "." + methodName + ":" + methodSignature;
118         }
119 
120         /*
121          * (non-Javadoc)
122          *
123          * @see java.lang.Comparable#compareTo(T)
124          */
125         @Override
126         public int compareTo(MethodDescriptor o) {
127             int cmp;
128             if ((cmp = className.compareTo(o.className)) != 0) {
129                 return cmp;
130             }
131             if ((cmp = methodName.compareTo(o.methodName)) != 0) {
132                 return cmp;
133             }
134             return methodSignature.compareTo(o.methodSignature);
135         }
136 
137         /*
138          * (non-Javadoc)
139          *
140          * @see java.lang.Object#hashCode()
141          */
142         @Override
143         public int hashCode() {
144             return 1277 * className.hashCode() + 37 * methodName.hashCode() + methodSignature.hashCode();
145         }
146 
147         /*
148          * (non-Javadoc)
149          *
150          * @see java.lang.Object#equals(java.lang.Object)
151          */
152         @Override
153         public boolean equals(Object obj) {
154             if (obj == null || obj.getClass() != this.getClass()) {
155                 return false;
156             }
157             MethodDescriptor other = (MethodDescriptor) obj;
158             return className.equals(other.className) && methodName.equals(other.methodName)
159                     && methodSignature.equals(other.methodSignature);
160         }
161     }

163     /**
164      * A range of source lines.
165      */
166     public static class SourceLineRange {
167         private final Integer start, end;
168 
169         /**
170          * Constructor for a single line.
171          */
172         public SourceLineRange(@Nonnull Integer line) {
173             this.start = this.end = line;
174         }
175 
176         /**
177          * Constructor for a range of lines.
178          *
179          * @param start
180          *            start line in range
181          * @param end
182          *            end line in range
183          */
184         public SourceLineRange(@Nonnull Integer start, @Nonnull Integer end) {
185             this.start = start;
186             this.end = end;
187         }
188 
189         /**
190          * @return Returns the start.
191          */
192         public @Nonnull
193         Integer getStart() {
194             return start;
195         }
196 
197         /**
198          * @return Returns the end.
199          */
200         public @Nonnull
201         Integer getEnd() {
202             return end;
203         }
204 
205         /*
206          * (non-Javadoc)
207          *
208          * @see java.lang.Object#toString()
209          */
210         @Override
211         public String toString() {
212             return start + (start.equals(end) ? "" : "-" + end);
213         }
214     }

328     private static final Pattern DIGITS = Pattern.compile("^[0-9]+$");
329 
330     /**
331      * Read source info from given InputStream. The stream is guaranteed to be
332      * closed.
333      *
334      * @param inputStream
335      *            the InputStream
336      * @throws IOException
337      *             if an I/O error occurs, or if the format is invalid
338      */
339     public void read(@WillClose InputStream inputStream) throws IOException {
340         BufferedReader reader = new BufferedReader(Util.getReader(inputStream));
341 
342         int lineNumber = 0;
343         try {
344             String line;
345             int lparen;
346             String version;
347 
348             while ((line = reader.readLine()) != null) {
349                 ++lineNumber;
350 
351                 if (lineNumber == 1) {
352                     if (DEBUG) {
353                         System.out.println("First line: " + line);
354                     }
355                     // Try to parse the version number string from the first
356                     // line.
357                     // null means that the line does not appear to be a version
358                     // number.
359                     version = parseVersionNumber(line);
360                     if (version != null) {
361                         // Check to see if version is supported.
362                         // Only 1.0 supported for now.
363                         if (!"1.0".equals(version)) {
364                             throw new IOException("Unsupported sourceInfo version " + version);
365                         }
366 
367                         // Version looks good. Skip to next line of file.
368                         continue;
369                     }
370                 }
371 
372                 StringTokenizer tokenizer = new StringTokenizer(line, ",");
373 
374                 String className = tokenizer.nextToken();
375                 String next = tokenizer.nextToken();
376                 if (DIGITS.matcher(next).matches()) {
377                     // Line number for class
378                     SourceLineRange range = createRange(next, tokenizer.nextToken());
379                     classLineMap.put(className, range);
380                     if (DEBUG) {
381                         System.out.println("class:" + className + "," + range);
382                     }
383                 } else if ((lparen = next.indexOf('(')) >= 0) {
384                     // Line number for method
385                     String methodName = next.substring(0, lparen);
386                     String methodSignature = next.substring(lparen);
387 
388                     if ("init^".equals(methodName)) {
389                         methodName = "<init>";
390                     } else if ("clinit^".equals(methodName)) {
391                         methodName = "<clinit>";
392                     }
393 
394                     SourceLineRange range = createRange(tokenizer.nextToken(), tokenizer.nextToken());
395                     methodLineMap.put(new MethodDescriptor(className, methodName, methodSignature), range);
396                     if (DEBUG) {
397                         System.out.println("method:" + methodName + methodSignature + "," + range);
398                     }
399                 } else {
400                     // Line number for field
401                     String fieldName = next;
402                     SourceLineRange range = createRange(tokenizer.nextToken(), tokenizer.nextToken());
403                     fieldLineMap.put(new FieldDescriptor(className, fieldName), range);
404                     if (DEBUG) {
405                         System.out.println("field:" + className + "," + fieldName + "," + range);
406                     }
407                 }
408 
409                 // Note: we could complain if there are more tokens,
410                 // but instead we'll just ignore them.
411             }
412         } catch (NoSuchElementException e) {
413             IOException ioe = new IOException("Invalid syntax in source info file at line " + lineNumber);
414             ioe.initCause(e);
415             throw ioe;
416         } finally {
417             try {
418                 reader.close();
419             } catch (IOException e) {
420                 // ignore
421             }
422         }
423     }

425     /**
426      * Parse the sourceInfo version string.
427      *
428      * @param line
429      *            the first line of the sourceInfo file
430      * @return the version number constant, or null if the line does not appear
431      *         to be a version string
432      */
433     private static String parseVersionNumber(String line) {
434         StringTokenizer tokenizer = new StringTokenizer(line, " \t");
435 
436         if (!expect(tokenizer, "sourceInfo") || !expect(tokenizer, "version") || !tokenizer.hasMoreTokens()) {
437             return null;
438         }
439 
440         return tokenizer.nextToken();
441     }
442 
443     /**
444      * Expect a particular token string to be returned by the given
445      * StringTokenizer.
446      *
447      * @param tokenizer
448      *            the StringTokenizer
449      * @param token
450      *            the expectedToken
451      * @return true if the expected token was returned, false if not
452      */
453     private static boolean expect(StringTokenizer tokenizer, String token) {
454         if (!tokenizer.hasMoreTokens()) {
455             return false;
456         }
457         String s = tokenizer.nextToken();
458         if (DEBUG) {
459             System.out.println("token=" + s);
460         }
461         return s.equals(token);
462     }

464     private static SourceLineRange createRange(String start, String end) {
465         return new SourceLineRange(Integer.valueOf(start), Integer.valueOf(end));
466     }

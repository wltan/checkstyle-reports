106         private static String arguments(String signature) {
107             int i = signature.indexOf('(');
108             if (i == -1) {
109                 return signature;
110             }
111             return signature.substring(0,i+1);
112         }

114         @Override
115         public ClassInfo build() {
116             AnalysisContext context = AnalysisContext.currentAnalysisContext();
117             FieldInfo fields[];
118             MethodInfo methods[];
119             if (fieldInfoList.size() == 0) {
120                 fields = FieldInfo.EMPTY_ARRAY;
121             } else {
122                 fields = fieldInfoList.toArray(new FieldInfo[fieldInfoList.size()]);
123             }
124 
125 
126             for(MethodInfo m : methodInfoList) {
127                 if (m.isBridge() && !bridgedSignatures.containsKey(m))  {
128 
129                     if (DEBUG) {
130                         System.out.println("Have bridge method:" + m);
131                     }
132                     for(MethodInfo to : methodInfoList) {
133                         if (m != to) {
134                             if (!to.isBridge()
135                                     && m.getName().equals(to.getName())
136                                     && arguments( m.getSignature()).equals(arguments(to.getSignature()))) {
137                                 if (DEBUG) {
138                                     System.out.println("  to method:" + to);
139                                 }
140                                 bridgedSignatures.put(m, to.getSignature());
141                             }
142 
143                         }
144                     } // end  for(MethodInfo to
145                 } // end  if (m.isBridge()
146             } // end  for(MethodInfo m : methodInfoList)
147 
148 
149 
150             for (Map.Entry<MethodInfo, String> e : bridgedSignatures.entrySet()) {
151                 MethodInfo method = e.getKey();
152                 String signature = e.getValue();
153                 for (MethodInfo m : methodInfoList) {
154                     if (m.getName().equals(method.getName()) && m.getSignature().equals(signature)) {
155                         if (DEBUG) {
156                             System.out.println("Found bridge method:");
157                             System.out.println("  " + method);
158                             if (!method.getAnnotations().isEmpty()) {
159                                 System.out.println("    " + method.getAnnotations());
160                             }
161                             System.out.println("  " + m);
162                             if (!m.getAnnotations().isEmpty()) {
163                                 System.out.println("    " + m.getAnnotations());
164                             }
165                         }
166                         context.setBridgeMethod(method, m);
167 
168                     }
169                 }
170 
171             } // end for
172 
173             if (methodInfoList.size() == 0) {
174                 methods = MethodInfo.EMPTY_ARRAY;
175             } else {
176                 methods = methodInfoList.toArray(new MethodInfo[methodInfoList.size()]);
177             }
178 
179             return new ClassInfo(classDescriptor, classSourceSignature, superclassDescriptor, interfaceDescriptorList,
180                     codeBaseEntry, accessFlags, source, majorVersion, minorVersion, referencedClassDescriptorList,
181                     calledClassDescriptors, classAnnotations, fields, methods, immediateEnclosingClass, usesConcurrency,
182                     hasStubs);
183         }

239     private MethodInfo[] computeMethodsInCallOrder() {
240         final Map<String, MethodInfo> map = new HashMap<>();
241 
242         for (MethodInfo m : xMethods) {
243             map.put(m.getName() + m.getSignature() + m.isStatic(), m);
244         }
245         final MultiMap<MethodInfo, MethodInfo> multiMap = SelfMethodCalls.getSelfCalls(getClassDescriptor(), map);
246         OutEdges2<MethodInfo> edges1 = new OutEdges2<MethodInfo>() {
247 
248             @Override
249             public Collection<MethodInfo> getOutEdges(MethodInfo method) {
250                 return multiMap.get(method);
251             }
252 
253             @Override
254             public int score(MethodInfo e) {
255                 return e.getMethodCallCount();
256             }
257         };
258         List<MethodInfo> result = TopologicalSort.sortByCallGraph(Arrays.asList(xMethods), edges1);
259         assert xMethods.length == result.size();
260         return result.toArray(new MethodInfo[result.size()]);
261     }

328     @Override
329     public XMethod findMethod(String methodName, String methodSig, boolean isStatic) {
330         int hash = FieldOrMethodDescriptor.getNameSigHashCode(methodName, methodSig);
331         for (MethodInfo mInfo : xMethods) {
332             if (mInfo.getNameSigHashCode() == hash && mInfo.getName().equals(methodName)
333                     && mInfo.getSignature().equals(methodSig) && mInfo.isStatic() == isStatic) {
334                 return mInfo;
335             }
336         }
337         return null;
338     }

341     public XMethod findMethod(MethodDescriptor descriptor) {
342         if (!descriptor.getClassDescriptor().equals(this)) {
343             throw new IllegalArgumentException();
344         }
345         return findMatchingMethod(descriptor);
346     }

353     @Override
354     public XField findField(String name, String signature, boolean isStatic) {
355         int hash = FieldOrMethodDescriptor.getNameSigHashCode(name, signature);
356         for (FieldInfo fInfo : xFields) {
357             if (fInfo.getNameSigHashCode() == hash && fInfo.getName().equals(name) && fInfo.getSignature().equals(signature)
358                     && fInfo.isStatic() == isStatic) {
359                 return fInfo;
360             }
361         }
362         try {
363             if (getSuperclassDescriptor() == null) {
364                 return null;
365             }
366             XClass superClass = Global.getAnalysisCache().getClassAnalysis(XClass.class, getSuperclassDescriptor());
367             XField result = superClass.findField(name, signature, isStatic);
368             if (result != null) {
369                 return result;
370             }
371             if (!isStatic) {
372                 return null;
373             }
374             ClassDescriptor[] interfaces = getInterfaceDescriptorList();
375             for (ClassDescriptor implementedInterface : interfaces) {
376                 superClass = Global.getAnalysisCache().getClassAnalysis(XClass.class, implementedInterface);
377                 result = superClass.findField(name, signature, isStatic);
378                 if (result != null) {
379                     return result;
380                 }
381             }
382             return null;
383         } catch (CheckedAnalysisException e) {
384             return null;
385         }
386     }

393     @Override
394     public String getPackageName() {
395         String dottedClassName = getClassDescriptor().toDottedClassName();
396         int lastDot = dottedClassName.lastIndexOf('.');
397         if (lastDot < 0) {
398             return "";
399         } else {
400             return dottedClassName.substring(0, lastDot);
401         }
402     }

404     public String getSlashedPackageName() {
405         String slashedClassName = getClassDescriptor().getClassName();
406         int lastSlash = slashedClassName.lastIndexOf('/');
407         if (lastSlash < 0) {

409         } else {
410             return slashedClassName.substring(0, lastSlash);

429     /**
430      * Destructively add an annotation to the object. In general, this is not a
431      * great idea, since it could cause the same class to appear to have
432      * different annotations at different times. However, this method is
433      * necessary for "built-in" annotations that FindBugs adds to system
434      * classes. As long as we add such annotations early enough that nobody will
435      * notice, we should be ok.
436      *
437      * @param annotationValue
438      *            an AnnotationValue to add to the class
439      */
440     public void addAnnotation(AnnotationValue annotationValue) {
441         HashMap<ClassDescriptor, AnnotationValue> updatedMap = new HashMap<>(classAnnotations);
442         updatedMap.put(annotationValue.getAnnotationClass(), annotationValue);
443         classAnnotations = Util.immutableMap(updatedMap);
444     }

446     @Override
447     public ElementType getElementType() {
448         if (getClassName().endsWith("package-info")) {
449             return ElementType.PACKAGE;
450         } else if (isAnnotation()) {
451             return ElementType.ANNOTATION_TYPE;
452         }
453         return ElementType.TYPE;

473     public @CheckForNull
474     AnnotatedObject getContainingScope0() {
475         try {
476             if (immediateEnclosingClass != null) {
477                 return Global.getAnalysisCache().getClassAnalysis(XClass.class, getImmediateEnclosingClass());
478             }
479             if (getClassName().endsWith("package-info")) {
480                 return null;
481             }
482             ClassDescriptor p = DescriptorFactory.createClassDescriptor(getSlashedPackageName() + "/" + "package-info");
483             return Global.getAnalysisCache().getClassAnalysis(XClass.class, p);
484         } catch (CheckedAnalysisException e) {
485             return null;
486         }
487     }

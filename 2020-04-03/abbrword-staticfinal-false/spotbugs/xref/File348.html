19  package edu.umd.cs.findbugs.detect;
20  
21  import java.util.Arrays;
22  import java.util.List;
23  
24  import org.apache.bcel.Const;
25  import org.apache.bcel.classfile.Code;
26  import org.apache.bcel.classfile.JavaClass;
27  import org.apache.bcel.classfile.Signature;
28  
29  import edu.umd.cs.findbugs.BugAccumulator;
30  import edu.umd.cs.findbugs.BugInstance;
31  import edu.umd.cs.findbugs.BugReporter;
32  import edu.umd.cs.findbugs.OpcodeStack;
33  import edu.umd.cs.findbugs.ba.ClassContext;
34  import edu.umd.cs.findbugs.bcel.OpcodeStackDetector;
35  import edu.umd.cs.findbugs.classfile.MethodDescriptor;
36  
37  /**
38   * equals and hashCode are blocking methods on URL's. Warn about invoking equals
39   * or hashCode on them, or defining Set or Maps with them as keys.
40   */
41  public class URLProblems extends OpcodeStackDetector {
42  
43      private static final MethodDescriptor URL_EQUALS = new MethodDescriptor("java/net/URL", "equals", "(Ljava/lang/Object;)Z");
44      private static final MethodDescriptor URL_HASHCODE = new MethodDescriptor("java/net/URL", "hashCode", "()I");
45  
46      final static String[] BAD_SIGNATURES = { "Hashtable<Ljava/net/URL", "Map<Ljava/net/URL", "Set<Ljava/net/URL" };
47  
48      // Must be sorted
49      private static final String[] HASHSET_KEY_METHODS = {"add", "contains", "remove"};
50      private static final String[] HASHMAP_KEY_METHODS = {"containsKey", "get", "remove"};
51      private static final String[] HASHMAP_TWO_ARG_KEY_METHODS = {"put"};
52  
53      private static final List<MethodDescriptor> methods = Arrays.asList(URL_EQUALS, URL_HASHCODE,
54              new MethodDescriptor("", "add", "(Ljava/lang/Object;)Z"),
55              new MethodDescriptor("", "contains", "(Ljava/lang/Object;)Z"),
56              new MethodDescriptor("", "remove", "(Ljava/lang/Object;)Z"),
57              new MethodDescriptor("", "containsKey", "(Ljava/lang/Object;)Z"),
58              new MethodDescriptor("", "get", "(Ljava/lang/Object;)Ljava/lang/Object;"),
59              new MethodDescriptor("", "remove", "(Ljava/lang/Object;)Ljava/lang/Object;"),
60              new MethodDescriptor("", "put", "(Ljava/lang/Object;Ljava/lang/Object;)Ljava/lang/Object;"));
61  
62      final private BugReporter bugReporter;
63  
64      final private BugAccumulator accumulator;
65  
66      private boolean hasInterestingMethodCalls;
67  
68      public URLProblems(BugReporter bugReporter) {
69          this.bugReporter = bugReporter;
70          this.accumulator = new BugAccumulator(bugReporter);
71      }
72  
73      @Override
74      public void visitClassContext(ClassContext classContext) {
75          this.hasInterestingMethodCalls = hasInterestingMethod(classContext.getJavaClass().getConstantPool(), methods);
76          super.visitClassContext(classContext);
77      }
78  
79      @Override
80      public void visitAfter(JavaClass obj) {
81          accumulator.reportAccumulatedBugs();
82      }
83  
84      @Override
85      public void visit(Code obj) {
86          if(this.hasInterestingMethodCalls) {
87              super.visit(obj);
88          }
89      }
90  
91      @Override
92      public void visit(Signature obj) {
93          String sig = obj.getSignature();
94          for (String s : BAD_SIGNATURES) {
95              if (sig.indexOf(s) >= 0) {
96                  if (visitingField()) {
97                      bugReporter.reportBug(new BugInstance(this, "DMI_COLLECTION_OF_URLS", HIGH_PRIORITY).addClass(this)
98                              .addVisitedField(this));
99                  } else if (visitingMethod()) {
100                     bugReporter.reportBug(new BugInstance(this, "DMI_COLLECTION_OF_URLS", HIGH_PRIORITY).addClassAndMethod(this));
101                 } else {
102                     bugReporter.reportBug(new BugInstance(this, "DMI_COLLECTION_OF_URLS", HIGH_PRIORITY).addClass(this).addClass(
103                             this));
104                 }
105             }
106         }
107     }
108 
109     void check(String className, String[] methodNames, int target, int url) {
110         if (Arrays.binarySearch(methodNames, getNameConstantOperand()) < 0) {
111             return;
112         }
113         if (stack.getStackDepth() <= target) {
114             return;
115         }
116         OpcodeStack.Item targetItem = stack.getStackItem(target);
117         OpcodeStack.Item urlItem = stack.getStackItem(url);
118         if (!"Ljava/net/URL;".equals(urlItem.getSignature())) {
119             return;
120         }
121         if (!targetItem.getSignature().equals(className)) {
122             return;
123         }
124         accumulator.accumulateBug(new BugInstance(this, "DMI_COLLECTION_OF_URLS", HIGH_PRIORITY).addClassAndMethod(this)
125                 .addCalledMethod(this), this);
126     }
127 
128     @Override
129     public void sawOpcode(int seen) {
130 
131         if (seen == Const.INVOKEVIRTUAL || seen == Const.INVOKEINTERFACE) {
132             check("Ljava/util/HashSet;", HASHSET_KEY_METHODS, 1, 0);
133             check("Ljava/util/HashMap;", HASHMAP_KEY_METHODS, 1, 0);
134             check("Ljava/util/HashMap;", HASHMAP_TWO_ARG_KEY_METHODS, 2, 1);
135         }
136 
137         if (seen == Const.INVOKEVIRTUAL && (getMethodDescriptorOperand().equals(URL_EQUALS)
138                 || getMethodDescriptorOperand().equals(URL_HASHCODE))) {
139             accumulator.accumulateBug(
140                     new BugInstance(this, "DMI_BLOCKING_METHODS_ON_URL", HIGH_PRIORITY).addClassAndMethod(this)
141                     .addCalledMethod(this), this);
142         }
143     }
144 }

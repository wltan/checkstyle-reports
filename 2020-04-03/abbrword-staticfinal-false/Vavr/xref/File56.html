51      public void shouldSolveProblem8() {
52          assertThat(largestProductOfConsecutives(4, _1000_DIGITS_NUMBER)).isEqualTo(5_832);
53          assertThat(largestProductOfConsecutives(13, _1000_DIGITS_NUMBER)).isEqualTo(23_514_624_000L);

56      private static long largestProductOfConsecutives(int sizeOfConsecutive, String num) {
57          return digits(num)
58                  .sliding(sizeOfConsecutive)
59                  .map(Seq::product)
60                  .max().get().longValue();
61      }
62  
63      private static List<Integer> digits(String num) {
64          return List.of(num.split("")).map(s -> Character.digit(s.charAt(0), 10));
65      }

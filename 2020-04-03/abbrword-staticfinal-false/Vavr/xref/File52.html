238     @Test
239     public void shouldSupportMultithreading() {
240         final AtomicBoolean isEvaluated = new AtomicBoolean();
241         final AtomicBoolean lock = new AtomicBoolean();
242         final Lazy<Integer> lazy = Lazy.of(() -> {
243             while (lock.get()) {
244                 Try.run(() -> Thread.sleep(300));
245             }
246             return 1;
247         });
248         new Thread(() -> {
249             Try.run(() -> Thread.sleep(100));
250             new Thread(() -> {
251                 Try.run(() -> Thread.sleep(100));
252                 lock.set(false);
253             }).start();
254             isEvaluated.compareAndSet(false, lazy.isEvaluated());
255             lazy.get();
256         }).start();
257         assertThat(isEvaluated.get()).isFalse();
258         assertThat(lazy.get()).isEqualTo(1);
259     }

261     @Test
262     @SuppressWarnings({ "StatementWithEmptyBody", "rawtypes" })
263     public void shouldBeConsistentFromMultipleThreads() throws Exception {
264         for (int i = 0; i < 100; i++) {
265             final AtomicBoolean canProceed = new AtomicBoolean(false);
266             final Vector<CompletableFuture<Void>> futures = Vector.range(0, 10).map(j -> {
267                 final AtomicBoolean isEvaluated = new AtomicBoolean(false);
268                 final Integer expected = ((j % 2) == 1) ? null : j;
269                 Lazy<Integer> lazy = Lazy.of(() -> {
270                     assertThat(isEvaluated.getAndSet(true)).isFalse();
271                     return expected;
272                 });
273                 return Tuple.of(lazy, expected);
274             }).flatMap(t -> range(0, 5).map(j -> runAsync(() -> {
275                         while (!canProceed.get()) { /* busy wait */ }
276                         assertThat(t._1.get()).isEqualTo(t._2);
277                     }))
278             );
279 
280             final CompletableFuture all = CompletableFuture.allOf(futures.toJavaList().toArray(new CompletableFuture<?>[0]));
281             canProceed.set(true);
282             all.join();
283         }
284     }

355 /**
356  * Lazy can't be empty. It is a placeholder for an existing value, but only evaluated when needed.
357  * In order to re-use existing unit tests, that are valid for all Value implementations,
358  * we provide here an _imaginary_ empty Lazy implementation, called 'undefined'.
359  * <p>
360  * Note: It is no good idea to leak it outside of the test scope into the core library (otherwise Undefined will be the new null).
361  */
362 final class Undefined<T> implements Value<T>, Serializable {
363 
364     private static final long serialVersionUID = 1L;
365 
366     static final Undefined<?> INSTANCE = new Undefined<>();
367 
368     private Lazy<T> prototype = Lazy.of(() -> null);
369 
370     private Undefined() {
371     }
372 
373     @Override
374     public T get() {
375         throw new NoSuchElementException();
376     }
377 
378     @Override
379     public boolean isAsync() {
380         return prototype.isAsync();
381     }
382 
383     @Override
384     public boolean isEmpty() {
385         return true;
386     }
387 
388     @Override
389     public boolean isLazy() {
390         return prototype.isLazy();
391     }
392 
393     @Override
394     public boolean isSingleValued() {
395         return prototype.isSingleValued();
396     }
397 
398     @Override
399     public Value<T> peek(Consumer<? super T> action) {
400         return this;
401     }
402 
403     @Override
404     public String stringPrefix() {
405         return null;
406     }
407 
408     @Override
409     public Iterator<T> iterator() {
410         return Iterator.empty();
411     }
412 
413     @SuppressWarnings("unchecked")
414     @Override
415     public <U> Value<U> map(Function<? super T, ? extends U> mapper) {
416         return (Value<U>) this;
417     }
418 
419     @Override
420     public boolean equals(Object o) {
421         return o == INSTANCE;
422     }
423 
424     @Override
425     public int hashCode() {
426         return 1;
427     }
428 
429     @Override
430     public String toString() {
431         return "Lazy()";
432     }
433 
434     /**
435      * Instance control for object serialization.
436      *
437      * @return The singleton instance of Undefined.
438      * @see java.io.Serializable
439      */
440     private Object readResolve() {
441         return INSTANCE;
442     }

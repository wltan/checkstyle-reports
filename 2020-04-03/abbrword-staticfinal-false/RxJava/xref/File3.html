18  import java.util.HashSet;
19  import java.util.List;

22  import rx.annotations.Experimental;

24  /**
25   * @warn javadoc class description missing
26   */
27  public final class Exceptions {
28      private Exceptions() {
29  
30      }
31  
32      /**
33       * Convenience method to throw a {@code RuntimeException} and {@code Error} directly
34       * or wrap any other exception type into a {@code RuntimeException}.
35       * @param t the exception to throw directly or wrapped
36       * @return because {@code propagate} itself throws an exception or error, this is a sort of phantom return
37       *         value; {@code propagate} does not actually return anything
38       */
39      public static RuntimeException propagate(Throwable t) {
40          /*
41           * The return type of RuntimeException is a trick for code to be like this:
42           * 
43           * throw Exceptions.propagate(e);
44           * 
45           * Even though nothing will return and throw via that 'throw', it allows the code to look like it
46           * so it's easy to read and understand that it will always result in a throw.
47           */
48          if (t instanceof RuntimeException) {
49              throw (RuntimeException) t;
50          } else if (t instanceof Error) {
51              throw (Error) t;
52          } else {
53              throw new RuntimeException(t);
54          }
55      }
56      /**
57       * Throws a particular {@code Throwable} only if it belongs to a set of "fatal" error varieties. These
58       * varieties are as follows:
59       * <ul>
60       * <li>{@link OnErrorNotImplementedException}</li>
61       * <li>{@link OnErrorFailedException}</li>
62       * <li>{@code StackOverflowError}</li>
63       * <li>{@code VirtualMachineError}</li>
64       * <li>{@code ThreadDeath}</li>
65       * <li>{@code LinkageError}</li>
66       * </ul>
67       * This can be useful if you are writing an operator that calls user-supplied code, and you want to
68       * notify subscribers of errors encountered in that code by calling their {@code onError} methods, but only
69       * if the errors are not so catastrophic that such a call would be futile, in which case you simply want to
70       * rethrow the error.
71       *
72       * @param t
73       *         the {@code Throwable} to test and perhaps throw
74       * @see <a href="https://github.com/ReactiveX/RxJava/issues/748#issuecomment-32471495">RxJava: StackOverflowError is swallowed (Issue #748)</a>
75       */
76      public static void throwIfFatal(Throwable t) {
77          if (t instanceof OnErrorNotImplementedException) {
78              throw (OnErrorNotImplementedException) t;
79          } else if (t instanceof OnErrorFailedException) {
80              Throwable cause = ((OnErrorFailedException) t).getCause();
81              if (cause instanceof RuntimeException) {
82                  throw (RuntimeException) cause;
83              } else {
84                  throw (OnErrorFailedException) t;
85              }
86          }
87          // values here derived from https://github.com/ReactiveX/RxJava/issues/748#issuecomment-32471495
88          else if (t instanceof StackOverflowError) {
89              throw (StackOverflowError) t;
90          } else if (t instanceof VirtualMachineError) {
91              throw (VirtualMachineError) t;
92          } else if (t instanceof ThreadDeath) {
93              throw (ThreadDeath) t;
94          } else if (t instanceof LinkageError) {
95              throw (LinkageError) t;
96          }
97      }
98  
99      private static final int MAX_DEPTH = 25;
100 
101     /**
102      * Adds a {@code Throwable} to a causality-chain of Throwables, as an additional cause (if it does not
103      * already appear in the chain among the causes).
104      *
105      * @param e
106      *         the {@code Throwable} at the head of the causality chain
107      * @param cause
108      *         the {@code Throwable} you want to add as a cause of the chain
109      */
110     public static void addCause(Throwable e, Throwable cause) {
111         Set<Throwable> seenCauses = new HashSet<Throwable>();
112 
113         int i = 0;
114         while (e.getCause() != null) {
115             if (i++ >= MAX_DEPTH) {
116                 // stack too deep to associate cause
117                 return;
118             }
119             e = e.getCause();
120             if (seenCauses.contains(e.getCause())) {
121                 break;
122             } else {
123                 seenCauses.add(e.getCause());
124             }
125         }
126         // we now have 'e' as the last in the chain
127         try {
128             e.initCause(cause);
129         } catch (Throwable t) {
130             // ignore
131             // the javadocs say that some Throwables (depending on how they're made) will never
132             // let me call initCause without blowing up even if it returns null
133         }
134     }
135 
136     /**
137      * Get the {@code Throwable} at the end of the causality-chain for a particular {@code Throwable}
138      *
139      * @param e
140      *         the {@code Throwable} whose final cause you are curious about
141      * @return the last {@code Throwable} in the causality-chain of {@code e} (or a "Stack too deep to get
142      *         final cause" {@code RuntimeException} if the chain is too long to traverse)
143      */
144     public static Throwable getFinalCause(Throwable e) {
145         int i = 0;
146         while (e.getCause() != null) {
147             if (i++ >= MAX_DEPTH) {
148                 // stack too deep to get final cause
149                 return new RuntimeException("Stack too deep to get final cause");
150             }
151             e = e.getCause();
152         }
153         return e;
154     }
155     /**
156      * Throws a single or multiple exceptions contained in the collection, wrapping it into
157      * {@code CompositeException} if necessary.
158      * @param exceptions the collection of exceptions. If null or empty, no exception is thrown.
159      * If the collection contains a single exception, that exception is either thrown as-is or wrapped into a
160      * CompositeException. Multiple exceptions are wrapped into a CompositeException.
161      */
162     @Experimental
163     public static void throwIfAny(List<? extends Throwable> exceptions) {
164         if (exceptions != null && !exceptions.isEmpty()) {
165             if (exceptions.size() == 1) {
166                 Throwable t = exceptions.get(0);
167                 // had to manually inline propagate because some tests attempt StackOverflowError 
168                 // and can't handle it with the stack space remaining
169                 if (t instanceof RuntimeException) {
170                     throw (RuntimeException) t;
171                 } else if (t instanceof Error) {
172                     throw (Error) t;
173                 } else {
174                     throw new RuntimeException(t);
175                 }
176             }
177             throw new CompositeException(
178                     "Multiple exceptions", exceptions);
179         }
180     }

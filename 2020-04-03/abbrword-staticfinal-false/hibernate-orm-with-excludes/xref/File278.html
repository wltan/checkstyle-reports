55  public abstract class AbstractStandardBasicType<T>
56  		implements BasicType, StringRepresentableType<T>, XmlRepresentableType<T> {
57  
58  	private static final Size DEFAULT_SIZE = new Size( 19, 2, 255, Size.LobMultiplier.NONE ); // to match legacy behavior
59  	private final Size dictatedSize = new Size();
60  
61  	// Don't use final here.  Need to initialize after-the-fact
62  	// by DynamicParameterizedTypes.
63  	private SqlTypeDescriptor sqlTypeDescriptor;
64  	private JavaTypeDescriptor<T> javaTypeDescriptor;
65  	// sqlTypes need always to be in sync with sqlTypeDescriptor
66  	private int[] sqlTypes;
67  
68  	public AbstractStandardBasicType(SqlTypeDescriptor sqlTypeDescriptor, JavaTypeDescriptor<T> javaTypeDescriptor) {
69  		this.sqlTypeDescriptor = sqlTypeDescriptor;
70  		this.sqlTypes = new int[] { sqlTypeDescriptor.getSqlType() };
71  		this.javaTypeDescriptor = javaTypeDescriptor;
72  	}
73  
74  	public T fromString(String string) {
75  		return javaTypeDescriptor.fromString( string );
76  	}
77  
78  	public String toString(T value) {
79  		return javaTypeDescriptor.toString( value );
80  	}
81  
82  	public T fromStringValue(String xml) throws HibernateException {
83  		return fromString( xml );
84  	}
85  
86  	public String toXMLString(T value, SessionFactoryImplementor factory) throws HibernateException {
87  		return toString( value );
88  	}
89  
90  	public T fromXMLString(String xml, Mapping factory) throws HibernateException {
91  		return StringHelper.isEmpty( xml ) ? null : fromStringValue( xml );
92  	}
93  
94  	protected MutabilityPlan<T> getMutabilityPlan() {
95  		return javaTypeDescriptor.getMutabilityPlan();
96  	}
97  
98  	protected T getReplacement(T original, T target, SessionImplementor session) {
99  		if ( !isMutable() ) {
100 			return original;
101 		}
102 		else if ( isEqual( original, target ) ) {
103 			return original;
104 		}
105 		else {
106 			return deepCopy( original );
107 		}
108 	}
109 
110 	public boolean[] toColumnNullness(Object value, Mapping mapping) {
111 		return value == null ? ArrayHelper.FALSE : ArrayHelper.TRUE;
112 	}
113 
114 	public String[] getRegistrationKeys() {
115 		return registerUnderJavaType()
116 				? new String[] { getName(), javaTypeDescriptor.getJavaTypeClass().getName() }
117 				: new String[] { getName() };
118 	}
119 
120 	protected boolean registerUnderJavaType() {
121 		return false;
122 	}
123 
124 	protected static Size getDefaultSize() {
125 		return DEFAULT_SIZE;
126 	}
127 
128 	protected Size getDictatedSize() {
129 		return dictatedSize;
130 	}

356 	@SuppressWarnings({ "unchecked" })
357 	public Object replace(
358 			Object original,
359 			Object target,
360 			SessionImplementor session,
361 			Object owner,
362 			Map copyCache,
363 			ForeignKeyDirection foreignKeyDirection) {
364 		return ForeignKeyDirection.FOREIGN_KEY_FROM_PARENT == foreignKeyDirection
365 				? getReplacement( (T) original, (T) target, session )
366 				: target;
367 	}

369 	// TODO : have SessionImplementor extend WrapperOptions
370 	private WrapperOptions getOptions(final SessionImplementor session) {
371 		return new WrapperOptions() {
372 			public boolean useStreamForLobBinding() {
373 				return Environment.useStreamsForBinary()
374 						|| session.getFactory().getDialect().useInputStreamToInsertBlob();
375 			}
376 
377 			public LobCreator getLobCreator() {
378 				return Hibernate.getLobCreator( session );
379 			}
380 
381 			public SqlTypeDescriptor remapSqlTypeDescriptor(SqlTypeDescriptor sqlTypeDescriptor) {
382 				final SqlTypeDescriptor remapped = sqlTypeDescriptor.canBeRemapped()
383 						? session.getFactory().getDialect().remapSqlTypeDescriptor( sqlTypeDescriptor )
384 						: sqlTypeDescriptor;
385 				return remapped == null ? sqlTypeDescriptor : remapped;
386 			}
387 		};

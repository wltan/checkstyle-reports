32  /**
33   * Task to provide functional testing under Ant, with a fairly complex workflow of:
34   *
35   * <ul>
36   * <li>Conditional execution</li>
37   * <li>Application to start</li>
38   * <li>A probe to "waitfor" before running tests</li>
39   * <li>A tests sequence</li>
40   * <li>A reporting sequence that runs after the tests have finished</li>
41   * <li>A "teardown" clause that runs after the rest.</li>
42   * <li>Automated termination of the program it executes, if a timeout is not met</li>
43   * <li>Checking of a failure property and automatic raising of a fault
44   *     (with the text in failureText)
45   * if test shutdown and reporting succeeded</li>
46   *  </ul>
47   *
48   * The task is designed to be framework neutral; it will work with JUnit,
49   *  TestNG and other test frameworks That can be
50   * executed from Ant. It bears a resemblance to the FunctionalTest task from
51   * SmartFrog, as the attribute names were
52   * chosen to make migration easier. However, this task benefits from the
53   * ability to tweak Ant's internals, and so
54   * simplify the workflow, and from the experience of using the SmartFrog task.
55   * No code has been shared.
56   *
57   * @since Ant 1.8
58   */
59

386     /**
387      * Run the functional test sequence.
388      * <p/>
389      * This is a fairly complex workflow -what is going on is that we try to clean up
390      * no matter how the run ended, and to retain the innermost exception that got thrown
391      * during cleanup. That is, if teardown fails after the tests themselves failed, it is the
392      * test failing that is more important.
393      * @throws BuildException if something was caught during the run or teardown.
394      */
395     public void execute() throws BuildException {
396 
397         //validation
398         validateTask(setup, "setup");
399         validateTask(application, "application");
400         validateTask(tests, "tests");
401         validateTask(reporting, "reporting");
402         validateTask(teardown, "teardown");
403 
404         //check the condition
405         //and bail out if it is defined but not true
406         if (condition != null && !condition.eval()) {
407             //we are skipping the test
408             log(SKIPPING_TESTS);
409             return;
410         }
411 
412         long timeoutMillis = timeout * timeoutUnitMultiplier;
413 
414         //set up the application to run in a separate thread
415         Parallel applicationRun = newParallel(timeoutMillis);
416         //with a worker which we can use to manage it
417         WorkerAnt worker = new WorkerAnt(applicationRun, null);
418         if (application != null) {
419             applicationRun.addTask(application);
420         }
421 
422         //The test run consists of the block followed by the tests.
423         long testRunTimeout = 0;
424         Sequential testRun = new Sequential();
425         bind(testRun);
426         if (block != null) {
427             //waitfor is not a task, it needs to be adapted
428             TaskAdapter ta = new TaskAdapter(block);
429             ta.bindToOwner(this);
430             validateTask(ta, "block");
431             testRun.addTask(ta);
432             //add the block time to the total test run timeout
433             testRunTimeout = block.calculateMaxWaitMillis();
434         }
435 
436         //add the tests and more delay
437         if (tests != null) {
438             testRun.addTask(tests);
439             testRunTimeout += timeoutMillis;
440         }
441         //add the reporting and more delay
442         if (reporting != null) {
443             testRun.addTask(reporting);
444             testRunTimeout += timeoutMillis;
445         }
446 
447         //wrap this in a parallel purely to set up timeouts for the
448         //test run
449         timedTests = newParallel(testRunTimeout, testRun);
450 
451         try {
452             //run any setup task
453             if (setup != null) {
454                 Parallel setupRun = newParallel(timeoutMillis, setup);
455                 setupRun.execute();
456             }
457             //start the worker thread and leave it running
458             worker.start();
459             //start the probe+test sequence
460             timedTests.execute();
461         } catch (BuildException e) {
462             //Record the exception and continue
463             testException = e;
464         } finally {
465             //teardown always runs; its faults are filed away
466             if (teardown != null) {
467                 try {
468                     Parallel teardownRun = newParallel(timeoutMillis, teardown);
469                     teardownRun.execute();
470                 } catch (BuildException e) {
471                     teardownException = e;
472                 }
473             }
474         }
475 
476         //we get here whether or not the tests/teardown have thrown a BuildException.
477         //do a forced shutdown of the running application, before processing the faults
478 
479         try {
480             //wait for the worker to have finished
481             long shutdownTimeMillis = shutdownTime * shutdownUnitMultiplier;
482             worker.waitUntilFinished(shutdownTimeMillis);
483             if (worker.isAlive()) {
484                 //then, if it is still running, interrupt it a second time.
485                 log(APPLICATION_FORCIBLY_SHUT_DOWN, Project.MSG_WARN);
486                 worker.interrupt();
487                 worker.waitUntilFinished(shutdownTimeMillis);
488             }
489         } catch (InterruptedException e) {
490             //success, something interrupted the shutdown. There may be a leaked
491             //worker;
492             log(SHUTDOWN_INTERRUPTED, e, Project.MSG_VERBOSE);
493         }
494         applicationException = worker.getBuildException();
495 
496         //Now faults are analysed
497 
498         processExceptions();
499     }
500 
501     /**
502      * Now faults are analysed.
503      * <p> The priority is
504      * <ol>
505      * <li>testexceptions, except those indicating a build timeout when the application itself
506      failed.<br>
507      (because often it is the application fault that is more interesting than the probe
508      failure, which is usually triggered by the application not starting
509      </li><li>
510      Application exceptions (above test timeout exceptions)
511      </li><li>
512      Teardown exceptions -except when they are being ignored
513      </li><li>
514      Test failures as indicated by the failure property
515      </li></ol>
516 
517      */
518     protected void processExceptions() {
519         taskException = testException;
520 
521         //look for an application fault
522         if (applicationException != null) {
523             if (taskException == null || taskException instanceof BuildTimeoutException) {
524                 taskException = applicationException;
525             } else {
526                 ignoringThrowable(APPLICATION_EXCEPTION, applicationException);
527             }
528         }
529 
530         //now look for teardown faults, which may be ignored
531         if (teardownException != null) {
532             if (taskException == null && failOnTeardownErrors) {
533                 taskException = teardownException;
534             } else {
535                 //don't let the cleanup exception get in the way of any other failure
536                 ignoringThrowable(TEARDOWN_EXCEPTION, teardownException);
537             }
538         }
539 
540         //now, analyse the tests
541         if (failureProperty != null
542              && getProject().getProperty(failureProperty) != null) {
543             //we've failed
544             log(failureMessage);
545             if (taskException == null) {
546                 taskException = new BuildException(failureMessage);
547             }
548         }
549 
550         //at this point taskException is null or not.
551         //if not, throw the exception
552         if (taskException != null) {
553             throw taskException;
554         }
555     }
556 
557     /**
558      * log that we are ignoring something rather than rethrowing it.
559      * @param type name of exception
560      * @param thrown what was thrown
561      */
562     protected void ignoringThrowable(String type, Throwable thrown) {
563         log(type + ": " + thrown.toString(),
564                 thrown,
565                 Project.MSG_WARN);
566     }

568     private static class NestedCondition extends ConditionBase implements Condition {
569         public boolean eval() {
570             if (countConditions() != 1) {
571                 throw new BuildException(
572                     "A single nested condition is required.");
573             }
574             return ((Condition) (getConditions().nextElement())).eval();
575         }

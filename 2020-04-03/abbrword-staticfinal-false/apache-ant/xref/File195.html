42  /**
43   * <p>Collects all failing test <i>cases</i> and creates a new JUnit test class containing
44   * a suite() method which calls these failed tests.</p>
45   * <p>Having classes <i>A</i> ... <i>D</i> with each several testcases you could earn a new
46   * test class like
47   * <pre>
48   * // generated on: 2007.08.06 09:42:34,555
49   * import junit.framework.*;
50   * public class FailedTests extends TestCase {
51   *     public FailedTests(String testname) {
52   *         super(testname);
53   *     }
54   *     public static Test suite() {
55   *         TestSuite suite = new TestSuite();
56   *         suite.addTest( new B("test04") );
57   *         suite.addTest( new org.D("test10") );
58   *         return suite;
59   *     }
60   * }
61   * </pre>
62   *
63   * Because each running test case gets its own formatter, we collect
64   * the failing test cases in a static list. Because we dont have a finalizer
65   * method in the formatters "lifecycle", we register this formatter as
66   * BuildListener and generate the new java source on taskFinished event.
67   *
68   * @since Ant 1.8.0
69   */
70  public class FailureRecorder extends ProjectComponent implements JUnitResultFormatter, BuildListener {
71  
72      /**
73       * This is the name of a magic System property ({@value}). The value of this
74       * <b>System</b> property should point to the location where to store the
75       * generated class (without suffix).
76       * Default location and name is defined in DEFAULT_CLASS_LOCATION.
77       * @see #DEFAULT_CLASS_LOCATION
78       */
79      public static final String MAGIC_PROPERTY_CLASS_LOCATION
80          = "ant.junit.failureCollector";
81  
82      /** Default location and name for the generated JUnit class file,
83       *  in the temp directory + FailedTests */
84      public static final String DEFAULT_CLASS_LOCATION
85          = System.getProperty("java.io.tmpdir") + "FailedTests";
86  
87      /** Prefix for logging. {@value} */
88      private static final String LOG_PREFIX = "    [junit]";
89  
90      /** Class names of failed tests without duplicates. */
91      private static SortedSet/*<TestInfos>*/ failedTests = new TreeSet();
92  
93      /** A writer for writing the generated source to. */
94      private BufferedWriter writer;
95  
96      /**
97       * Location and name of the generated JUnit class.
98       * Lazy instantiated via getLocationName().
99       */
100     private static String locationName;
101 
102     /**
103      * Returns the (lazy evaluated) location for the collector class.
104      * Order for evaluation: System property > Ant property > default value
105      * @return location for the collector class
106      * @see #MAGIC_PROPERTY_CLASS_LOCATION
107      * @see #DEFAULT_CLASS_LOCATION
108      */
109     private String getLocationName() {
110         if (locationName == null) {
111             String syspropValue = System.getProperty(MAGIC_PROPERTY_CLASS_LOCATION);
112             String antpropValue = getProject().getProperty(MAGIC_PROPERTY_CLASS_LOCATION);
113 
114             if (syspropValue != null) {
115                 locationName = syspropValue;
116                 verbose("System property '" + MAGIC_PROPERTY_CLASS_LOCATION + "' set, so use "
117                         + "its value '" + syspropValue + "' as location for collector class.");
118             } else if (antpropValue != null) {
119                 locationName = antpropValue;
120                 verbose("Ant property '" + MAGIC_PROPERTY_CLASS_LOCATION + "' set, so use "
121                         + "its value '" + antpropValue + "' as location for collector class.");
122             } else {
123                 locationName = DEFAULT_CLASS_LOCATION;
124                 verbose("System property '" + MAGIC_PROPERTY_CLASS_LOCATION + "' not set, so use "
125                         + "value as location for collector class: '"
126                         + DEFAULT_CLASS_LOCATION + "'");
127             }
128 
129             File locationFile = new File(locationName);
130             if (!locationFile.isAbsolute()) {
131                 File f = new File(getProject().getBaseDir(), locationName);
132                 locationName = f.getAbsolutePath();
133                 verbose("Location file is relative (" + locationFile + ")"
134                         + " use absolute path instead (" + locationName + ")");
135             }
136         }
137 
138         return locationName;
139     }
140 
141     /**
142      * This method is called by the Ant runtime by reflection. We use the project reference for
143      * registration of this class as BuildListener.
144      *
145      * @param project
146      *            project reference
147      */
148     public void setProject(Project project) {
149         // store project reference for logging
150         super.setProject(project);
151         // check if already registered
152         boolean alreadyRegistered = false;
153         Vector allListeners = project.getBuildListeners();
154         final int size = allListeners.size();
155         for (int i = 0; i < size; i++) {
156             Object listener = allListeners.get(i);
157             if (listener instanceof FailureRecorder) {
158                 alreadyRegistered = true;
159                 break;
160             }
161         }
162         // register if needed
163         if (!alreadyRegistered) {
164             verbose("Register FailureRecorder (@" + this.hashCode() + ") as BuildListener");
165             project.addBuildListener(this);
166         }
167     }

246     // ===== "Templates" for generating the JUnit class =====
247 
248     private void writeJavaClass() {
249         try {
250             File sourceFile = new File((getLocationName() + ".java"));
251             verbose("Write collector class to '" + sourceFile.getAbsolutePath() + "'");
252 
253             if (sourceFile.exists() && !sourceFile.delete()) {
254                 throw new IOException("could not delete " + sourceFile);
255             }
256             writer = new BufferedWriter(new FileWriter(sourceFile));
257 
258             createClassHeader();
259             createSuiteMethod();
260             createClassFooter();
261 
262         } catch (IOException e) {
263             e.printStackTrace();
264         } finally {
265             FileUtils.close(writer);
266         }
267     }
268 
269     private void createClassHeader() throws IOException {
270         String className = getLocationName().replace('\\', '/');
271         if (className.indexOf('/') > -1) {
272             className = className.substring(className.lastIndexOf('/') + 1);
273         }
274         SimpleDateFormat sdf = new SimpleDateFormat("yyyy.MM.dd HH:mm:ss,SSS");
275         writer.write("// generated on: ");
276         writer.write(sdf.format(new Date()));
277         writer.newLine();
278         writer.write("import junit.framework.*;");
279         writer.newLine();
280         writer.write("public class ");
281         writer.write(className);
282         // If this class does not extend TC, Ant doesnt run these
283         writer.write(" extends TestCase {");
284         writer.newLine();
285         // standard String-constructor
286         writer.write("    public ");
287         writer.write(className);
288         writer.write("(String testname) {");
289         writer.newLine();
290         writer.write("        super(testname);");
291         writer.newLine();
292         writer.write("    }");
293         writer.newLine();
294     }
295 
296     private void createSuiteMethod() throws IOException {
297         writer.write("    public static Test suite() {");
298         writer.newLine();
299         writer.write("        TestSuite suite = new TestSuite();");
300         writer.newLine();
301         for (Iterator iter = failedTests.iterator(); iter.hasNext();) {
302             TestInfos testInfos = (TestInfos) iter.next();
303             writer.write("        suite.addTest(");
304             writer.write(String.valueOf(testInfos));
305             writer.write(");");
306             writer.newLine();
307         }
308         writer.write("        return suite;");
309         writer.newLine();
310         writer.write("    }");
311         writer.newLine();
312     }
313 
314     private void createClassFooter() throws IOException {
315         writer.write("}");
316         writer.newLine();
317     }

337     /**
338      * TestInfos holds information about a given test for later use.
339      */
340     public static class TestInfos implements Comparable {
341 
342         /** The class name of the test. */
343         private final String className;
344 
345         /** The method name of the testcase. */
346         private final String methodName;
347 
348         /**
349          * This constructor extracts the needed information from the given test.
350          * @param test Test to analyze
351          */
352         public TestInfos(Test test) {
353             className = test.getClass().getName();
354             String _methodName = test.toString();
355             methodName = _methodName.substring(0, _methodName.indexOf('('));
356         }
357 
358         /**
359          * This String-Representation can directly be used for instantiation of
360          * the JUnit testcase.
361          * @return the string representation.
362          * @see java.lang.Object#toString()
363          * @see FailureRecorder#createSuiteMethod()
364          */
365         public String toString() {
366             return "new " + className + "(\"" + methodName + "\")";
367         }
368 
369         /**
370          * The SortedMap needs comparable elements.
371          * @param other the object to compare to.
372          * @return the result of the comparison.
373          * @see java.lang.Comparable#compareTo
374          * @see SortedSet#comparator()
375          */
376         public int compareTo(Object other) {
377             if (other instanceof TestInfos) {
378                 TestInfos otherInfos = (TestInfos) other;
379                 return toString().compareTo(otherInfos.toString());
380             } else {
381                 return -1;
382             }
383         }
384         public boolean equals(Object obj) {
385             return obj instanceof TestInfos && toString().equals(obj.toString());
386         }
387         public int hashCode() {
388             return toString().hashCode();
389         }
390     }

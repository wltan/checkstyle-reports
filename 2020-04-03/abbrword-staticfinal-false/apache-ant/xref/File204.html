48  /**
49   * Prints XML output of the test to a specified Writer.
50   *
51   * @see FormatterElement
52   */

54  public class XMLJUnitResultFormatter implements JUnitResultFormatter, XMLConstants, IgnoredTestListener {
55  
56      private static final double ONE_SECOND = 1000.0;
57  
58      /** constant for unnnamed testsuites/cases */
59      private static final String UNKNOWN = "unknown";
60  
61      private static DocumentBuilder getDocumentBuilder() {
62          try {
63              return DocumentBuilderFactory.newInstance().newDocumentBuilder();
64          } catch (Exception exc) {
65              throw new ExceptionInInitializerError(exc);
66          }
67      }
68  
69      /**
70       * The XML document.
71       */
72      private Document doc;
73      /**
74       * The wrapper for the whole testsuite.
75       */
76      private Element rootElement;
77      /**
78       * Element for the current test.
79       * 
80       * The keying of this map is a bit of a hack: tests are keyed by caseName(className) since
81       * the Test we get for Test-start isn't the same as the Test we get during test-assumption-fail,
82       * so we can't easily match Test objects without manually iterating over all keys and checking
83       * individual fields.
84       */
85      private Hashtable<String, Element> testElements = new Hashtable<String, Element>();
86      /**
87       * tests that failed.
88       */
89      private Hashtable failedTests = new Hashtable();
90      /**
91       * Tests that were skipped.
92       */
93      private Hashtable<String, Test> skippedTests = new Hashtable<String, Test>();
94      /**
95       * Tests that were ignored. See the note above about the key being a bit of a hack.
96       */
97      private Hashtable<String, Test> ignoredTests = new Hashtable<String, Test>();
98      /**
99       * Timing helper.
100      */
101     private Hashtable<String, Long> testStarts = new Hashtable<String, Long>();
102     /**
103      * Where to write the log to.
104      */
105     private OutputStream out;
106

108     public XMLJUnitResultFormatter() {

126     /**
127      * The whole testsuite started.
128      * @param suite the testsuite.
129      */
130     public void startTestSuite(JUnitTest suite) {
131         doc = getDocumentBuilder().newDocument();
132         rootElement = doc.createElement(TESTSUITE);
133         String n = suite.getName();
134         rootElement.setAttribute(ATTR_NAME, n == null ? UNKNOWN : n);
135 
136         //add the timestamp
137         final String timestamp = DateUtils.format(new Date(),
138                 DateUtils.ISO8601_DATETIME_PATTERN);
139         rootElement.setAttribute(TIMESTAMP, timestamp);
140         //and the hostname.
141         rootElement.setAttribute(HOSTNAME, getHostname());
142 
143         // Output properties
144         Element propsElement = doc.createElement(PROPERTIES);
145         rootElement.appendChild(propsElement);
146         Properties props = suite.getProperties();
147         if (props != null) {
148             Enumeration e = props.propertyNames();
149             while (e.hasMoreElements()) {
150                 String name = (String) e.nextElement();
151                 Element propElement = doc.createElement(PROPERTY);
152                 propElement.setAttribute(ATTR_NAME, name);
153                 propElement.setAttribute(ATTR_VALUE, props.getProperty(name));
154                 propsElement.appendChild(propElement);
155             }
156         }
157     }
158 
159     /**
160      * get the local hostname
161      * @return the name of the local host, or "localhost" if we cannot work it out
162      */
163     private String getHostname()  {
164         try {
165             return InetAddress.getLocalHost().getHostName();
166         } catch (UnknownHostException e) {
167             return "localhost";
168         }
169     }
170 
171     /**
172      * The whole testsuite ended.
173      * @param suite the testsuite.
174      * @throws BuildException on error.
175      */
176     public void endTestSuite(JUnitTest suite) throws BuildException {
177         rootElement.setAttribute(ATTR_TESTS, "" + suite.runCount());
178         rootElement.setAttribute(ATTR_FAILURES, "" + suite.failureCount());
179         rootElement.setAttribute(ATTR_ERRORS, "" + suite.errorCount());
180         rootElement.setAttribute(ATTR_SKIPPED, "" + suite.skipCount());
181         rootElement.setAttribute(
182             ATTR_TIME, "" + (suite.getRunTime() / ONE_SECOND));
183         if (out != null) {
184             Writer wri = null;
185             try {
186                 wri = new BufferedWriter(new OutputStreamWriter(out, "UTF8"));
187                 wri.write("<?xml version=\"1.0\" encoding=\"UTF-8\" ?>\n");
188                 (new DOMElementWriter()).write(rootElement, wri, 0, "  ");
189             } catch (IOException exc) {
190                 throw new BuildException("Unable to write log file", exc);
191             } finally {
192                 if (wri != null) {
193                     try {
194                         wri.flush();
195                     } catch (IOException ex) {
196                         // ignore
197                     }
198                 }
199                 if (out != System.out && out != System.err) {
200                     FileUtils.close(wri);
201                 }
202             }
203         }
204     }

206     /**
207      * Interface TestListener.
208      *
209      * <p>A new Test is started.
210      * @param t the test.
211      */
212     public void startTest(Test t) {
213         testStarts.put(createDescription(t), System.currentTimeMillis());
214     }
215 
216     private static String createDescription(Test test) throws BuildException {
217         return JUnitVersionHelper.getTestCaseName(test) + "(" + JUnitVersionHelper.getTestCaseClassName(test) + ")";
218 	}
219 
220     /**
221      * Interface TestListener.
222      *
223      * <p>A Test is finished.
224      * @param test the test.
225      */
226     public void endTest(Test test) {
227         String testDescription = createDescription(test);
228 
229         // Fix for bug #5637 - if a junit.extensions.TestSetup is
230         // used and throws an exception during setUp then startTest
231         // would never have been called
232         if (!testStarts.containsKey(testDescription)) {
233             startTest(test);
234         }
235         Element currentTest;
236         if (!failedTests.containsKey(test) && !skippedTests.containsKey(testDescription) && !ignoredTests.containsKey(testDescription)) {
237             currentTest = doc.createElement(TESTCASE);
238             String n = JUnitVersionHelper.getTestCaseName(test);
239             currentTest.setAttribute(ATTR_NAME,
240                                      n == null ? UNKNOWN : n);
241             // a TestSuite can contain Tests from multiple classes,
242             // even tests with the same name - disambiguate them.
243             currentTest.setAttribute(ATTR_CLASSNAME,
244                     JUnitVersionHelper.getTestCaseClassName(test));
245             rootElement.appendChild(currentTest);
246             testElements.put(createDescription(test), currentTest);
247         } else {
248             currentTest = testElements.get(testDescription);
249         }
250 
251         Long l = testStarts.get(createDescription(test));
252         currentTest.setAttribute(ATTR_TIME,
253             "" + ((System.currentTimeMillis() - l) / ONE_SECOND));
254     }

256     /**
257      * Interface TestListener for JUnit &lt;= 3.4.

259      * <p>A Test failed.
260      * @param test the test.
261      * @param t the exception.

264         formatError(FAILURE, test, t);
265     }

270      * <p>A Test failed.

272      * @param t the assertion.

274     public void addFailure(Test test, AssertionFailedError t) {
275         addFailure(test, (Throwable) t);
276     }

278     /**
279      * Interface TestListener.
280      *
281      * <p>An error occurred while running the test.
282      * @param test the test.
283      * @param t the error.
284      */
285     public void addError(Test test, Throwable t) {
286         formatError(ERROR, test, t);
287     }

289     private void formatError(String type, Test test, Throwable t) {
290         if (test != null) {
291             endTest(test);
292             failedTests.put(test, test);
293         }
294 
295         Element nested = doc.createElement(type);
296         Element currentTest;
297         if (test != null) {
298             currentTest = testElements.get(createDescription(test));
299         } else {
300             currentTest = rootElement;
301         }
302 
303         currentTest.appendChild(nested);
304 
305         String message = t.getMessage();
306         if (message != null && message.length() > 0) {
307             nested.setAttribute(ATTR_MESSAGE, t.getMessage());
308         }
309         nested.setAttribute(ATTR_TYPE, t.getClass().getName());
310 
311         String strace = JUnitTestRunner.getFilteredTrace(t);
312         Text trace = doc.createTextNode(strace);
313         nested.appendChild(trace);
314     }
315 
316     private void formatOutput(String type, String output) {
317         Element nested = doc.createElement(type);
318         rootElement.appendChild(nested);
319         nested.appendChild(doc.createCDATASection(output));
320     }
321 
322     public void testIgnored(Test test) {
323         formatSkip(test, JUnitVersionHelper.getIgnoreMessage(test));
324         if (test != null) {
325             ignoredTests.put(createDescription(test), test);
326         }
327     }
328 
329 
330     public void formatSkip(Test test, String message) {
331         if (test != null) {
332             endTest(test);
333         }
334 
335         Element nested = doc.createElement("skipped");
336 
337         if (message != null) {
338             nested.setAttribute("message", message);
339         }
340 
341         Element currentTest;
342         if (test != null) {
343             currentTest = testElements.get(createDescription(test));
344         } else {
345             currentTest = rootElement;
346         }
347 
348         currentTest.appendChild(nested);
349 
350     }
351 
352     public void testAssumptionFailure(Test test, Throwable failure) {
353         formatSkip(test, failure.getMessage());
354         skippedTests.put(createDescription(test), test);

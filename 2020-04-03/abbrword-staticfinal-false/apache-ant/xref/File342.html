18  package task;
19  import org.apache.tools.ant.Task;
20  import org.apache.tools.ant.BuildException;
21  import org.apache.tools.ant.util.FileUtils;
22  import java.io.File;
23  import java.io.BufferedInputStream;
24  import java.io.FileInputStream;
25  import java.io.FileOutputStream;
26  import java.io.InputStream;
27  import java.io.OutputStream;
28  
29  /**
30   * Base class for the uuencode/decode test tasks.
31   */
32  abstract public class BaseTask extends Task {
33      private final static FileUtils FILE_UTILS = FileUtils.getFileUtils();
34      private File inFile;
35      private File outFile;
36  
37      public void setInFile(File inFile) {
38          this.inFile = inFile;
39      }
40      protected File getInFile() {
41          return inFile;
42      }
43      public void setOutFile(File outFile) {
44          this.outFile = outFile;
45      }
46      protected File getOutFile() {
47          return outFile;
48      }
49      public void execute() {
50          assertAttribute(inFile, "inFile");
51          assertAttribute(outFile, "outFile");
52          InputStream inputStream = null;
53          OutputStream outputStream = null;
54          try {
55              inputStream = new BufferedInputStream(
56                  new FileInputStream(getInFile()));
57              outputStream = new FileOutputStream(getOutFile());
58              doit(inputStream, outputStream);
59          } catch (Exception ex) {
60              throw new BuildException(ex);
61          } finally {
62              FILE_UTILS.close(inputStream);
63              FILE_UTILS.close(outputStream);
64          }
65      }
66      
67      abstract protected void doit(
68          InputStream is, OutputStream os) throws Exception;
69      
70      private void assertAttribute(File file, String attributeName) {
71          if (file == null) {
72              throw new BuildException("Required attribute " + attributeName
73                                       + " not set");
74          }
75      }
76  }

142     /**
143      * Actually run a test, finding all input files (and corresponding
144      * goldfile)
145      */
146     private void doTest(String msg, boolean keepRoot, boolean collapse,
147                         boolean semantic, boolean include, boolean localRoot) throws IOException {
148         Enumeration iter =
149             getFiles(new File(System.getProperty("root"), "src/etc/testcases/taskdefs/xmlproperty/inputs"));
150         while (iter.hasMoreElements()) {
151             File inputFile = (File) iter.nextElement();
152             // What's the working directory?  If local, then its the
153             // folder of the input file.  Otherwise, its the "current" dir..
154             File workingDir;
155             if ( localRoot ) {
156                 workingDir = inputFile.getParentFile();
157             } else {
158                 workingDir = FILE_UTILS.resolveFile(new File("."), ".");
159             }
160 
161 
162             File propertyFile = getGoldfile(inputFile, keepRoot, collapse,
163                                             semantic, include, localRoot);
164             if (!propertyFile.exists()) {
165 //                    System.out.println("Skipping as "
166 //                                       + propertyFile.getAbsolutePath()
167 //                                       + ") doesn't exist.");
168                 continue;
169             }
170 
171             //                System.out.println(msg + " (" + propertyFile.getName() + ") in (" + workingDir + ")");
172 
173             Project p = new Project();
174 
175             XmlProperty xmlproperty = new XmlProperty();
176             xmlproperty.setProject(p);
177             xmlproperty.setFile(inputFile);
178 
179             xmlproperty.setKeeproot(keepRoot);
180             xmlproperty.setCollapseAttributes(collapse);
181             xmlproperty.setSemanticAttributes(semantic);
182             xmlproperty.setIncludeSemanticAttribute(include);
183             xmlproperty.setRootDirectory(workingDir);
184 
185             // Set a property on the project to make sure that loading
186             // a property with the same name from an xml file will
187             // *not* change it.
188             p.setNewProperty("override.property.test", "foo");
189 
190             xmlproperty.execute();
191 
192             Properties props = new Properties();
193             props.load(new FileInputStream(propertyFile));
194 
195             //printProperties(p.getProperties());
196 
197             ensureProperties(msg, inputFile, workingDir, p, props);
198             ensureReferences(msg, inputFile, p.getReferences());
199 
200         }
201     }

203     /**
204      * Make sure every property loaded from the goldfile was also
205      * read from the XmlProperty.  We could try and test the other way,
206      * but some other properties may get set in the XmlProperty due
207      * to generic Project/Task configuration.
208      */
209     private static void ensureProperties (String msg, File inputFile,
210                                           File workingDir, Project p,
211                                           Properties properties) {
212         Hashtable xmlproperties = p.getProperties();
213         // Every key identified by the Properties must have been loaded.
214         Enumeration propertyKeyEnum = properties.propertyNames();
215         while(propertyKeyEnum.hasMoreElements()){
216             String currentKey = propertyKeyEnum.nextElement().toString();
217             String assertMsg = msg + "-" + inputFile.getName()
218                 + " Key=" + currentKey;
219 
220             String propertyValue = properties.getProperty(currentKey);
221 
222             String xmlValue = (String)xmlproperties.get(currentKey);
223 
224             if (propertyValue.startsWith("ID.")) {
225                 // The property is an id's thing -- either a property
226                 // or a path.  We need to make sure
227                 // that the object was created with the given id.
228                 // We don't have an adequate way of testing the actual
229                 // *value* of the Path object, though...
230                 String id = currentKey;
231                 Object obj = p.getReferences().get(id);
232 
233                 if ( obj == null ) {
234                     fail(assertMsg + " Object ID does not exist.");
235                 }
236 
237                 // What is the property supposed to be?
238                 propertyValue =
239                     propertyValue.substring(3, propertyValue.length());
240                 if (propertyValue.equals("path")) {
241                     if (!(obj instanceof Path)) {
242                         fail(assertMsg + " Path ID is a "
243                              + obj.getClass().getName());
244                     }
245                 } else {
246                     assertEquals(assertMsg, propertyValue, obj.toString());
247                 }
248 
249             } else {
250 
251                 if (propertyValue.startsWith("FILE.")) {
252                     // The property is the name of a file.  We are testing
253                     // a location attribute, so we need to resolve the given
254                     // file name in the provided folder.
255                     String fileName =
256                         propertyValue.substring(5, propertyValue.length());
257                     File f = new File(workingDir, fileName);
258                     propertyValue = f.getAbsolutePath();
259                 }
260 
261                 assertEquals(assertMsg, propertyValue, xmlValue);
262             }
263 
264         }
265     }

279     /**
280      * Ensure all references loaded by the project are valid.
281      */
282     private static void ensureReferences (String msg, File inputFile,
283                                           Hashtable references) {
284         Enumeration referenceKeyEnum = references.keys();
285         while(referenceKeyEnum.hasMoreElements()){
286             String currentKey = referenceKeyEnum.nextElement().toString();
287             Object currentValue = references.get(currentKey);
288 
289             if (currentValue instanceof Path) {
290             } else if (currentValue instanceof String) {
291             } else {
292                 if( ! currentKey.startsWith("ant.") ) {
293                     fail(msg + "-" + inputFile.getName() + " Key="
294                          + currentKey + " is not a recognized type.");
295                 }
296             }
297         }
298     }

300     /**
301      * Munge the name of the input file to find an appropriate goldfile,
302      * based on hardwired naming conventions.
303      */
304     private static File getGoldfile (File input, boolean keepRoot,
305                                      boolean collapse, boolean semantic,
306                                      boolean include, boolean localRoot) {
307         // Substitute .xml with .properties
308         String baseName = input.getName().toLowerCase();
309         if (baseName.endsWith(".xml")) {
310             baseName = baseName.substring(0, baseName.length() - 4)
311                 + ".properties";
312         }
313 
314         File dir = input.getParentFile().getParentFile();
315 
316         String goldFileFolder = "goldfiles/";
317 
318         if (keepRoot) {
319             goldFileFolder += "keeproot-";
320         } else {
321             goldFileFolder += "nokeeproot-";
322         }
323 
324         if (semantic) {
325             goldFileFolder += "semantic-";
326             if (include) {
327                 goldFileFolder += "include-";
328             }
329         } else {
330             if (collapse) {
331                 goldFileFolder += "collapse-";
332             } else {
333                 goldFileFolder += "nocollapse-";
334             }
335         }
336 
337         return new File(dir, goldFileFolder + baseName);
338     }

340     /**
341      * Retrieve a list of xml files in the specified folder
342      * and below.
343      */
344     private static Enumeration getFiles (final File startingDir) {
345         Vector result = new Vector();
346         getFiles(startingDir, result);
347         return result.elements();
348     }
349 
350     /**
351      * Collect a list of xml files in the specified folder
352      * and below.
353      */
354     private static void getFiles (final File startingDir, Vector collect) {
355         FileFilter filter = new FileFilter() {
356             public boolean accept (File file) {
357                 if (file.isDirectory()) {
358                     return true;
359                 } else {
360                     return (file.getPath().indexOf("taskdefs") > 0 &&
361                             file.getPath().toLowerCase().endsWith(".xml") );
362                 }
363             }
364         };
365 
366         File[] files = startingDir.listFiles(filter);
367         for (int i=0;i<files.length;i++) {
368             File f = files[i];
369             if (!f.isDirectory()) {
370                 collect.addElement(f);
371             } else {
372                 getFiles(f, collect);
373             }
374         }
375     }

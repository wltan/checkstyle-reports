15  package com.google.common.eventbus;
16  
17  import static com.google.common.base.Preconditions.checkNotNull;
18  
19  import com.google.common.collect.Queues;
20  import java.util.Iterator;
21  import java.util.Queue;
22  import java.util.concurrent.ConcurrentLinkedQueue;
23  
24  /**
25   * Handler for dispatching events to subscribers, providing different event ordering guarantees that
26   * make sense for different situations.
27   *
28   * <p><b>Note:</b> The dispatcher is orthogonal to the subscriber's {@code Executor}. The dispatcher
29   * controls the order in which events are dispatched, while the executor controls how (i.e. on which
30   * thread) the subscriber is actually called when an event is dispatched to it.
31   *
32   * @author Colin Decker
33   */
34  abstract class Dispatcher {
35  
36    /**
37     * Returns a dispatcher that queues events that are posted reentrantly on a thread that is already
38     * dispatching an event, guaranteeing that all events posted on a single thread are dispatched to
39     * all subscribers in the order they are posted.
40     *
41     * <p>When all subscribers are dispatched to using a <i>direct</i> executor (which dispatches on
42     * the same thread that posts the event), this yields a breadth-first dispatch order on each
43     * thread. That is, all subscribers to a single event A will be called before any subscribers to
44     * any events B and C that are posted to the event bus by the subscribers to A.
45     */
46    static Dispatcher perThreadDispatchQueue() {
47      return new PerThreadQueuedDispatcher();
48    }
49  
50    /**
51     * Returns a dispatcher that queues events that are posted in a single global queue. This behavior
52     * matches the original behavior of AsyncEventBus exactly, but is otherwise not especially useful.
53     * For async dispatch, an {@linkplain #immediate() immediate} dispatcher should generally be
54     * preferable.
55     */
56    static Dispatcher legacyAsync() {
57      return new LegacyAsyncDispatcher();
58    }
59  
60    /**
61     * Returns a dispatcher that dispatches events to subscribers immediately as they're posted
62     * without using an intermediate queue to change the dispatch order. This is effectively a
63     * depth-first dispatch order, vs. breadth-first when using a queue.
64     */
65    static Dispatcher immediate() {
66      return ImmediateDispatcher.INSTANCE;
67    }
68  
69    /** Dispatches the given {@code event} to the given {@code subscribers}. */
70    abstract void dispatch(Object event, Iterator<Subscriber> subscribers);
71  
72    /** Implementation of a {@link #perThreadDispatchQueue()} dispatcher. */
73    private static final class PerThreadQueuedDispatcher extends Dispatcher {
74  
75      // This dispatcher matches the original dispatch behavior of EventBus.
76  
77      /** Per-thread queue of events to dispatch. */
78      private final ThreadLocal<Queue<Event>> queue =
79          new ThreadLocal<Queue<Event>>() {
80            @Override
81            protected Queue<Event> initialValue() {
82              return Queues.newArrayDeque();
83            }
84          };
85  
86      /** Per-thread dispatch state, used to avoid reentrant event dispatching. */
87      private final ThreadLocal<Boolean> dispatching =
88          new ThreadLocal<Boolean>() {
89            @Override
90            protected Boolean initialValue() {
91              return false;
92            }
93          };
94  
95      @Override
96      void dispatch(Object event, Iterator<Subscriber> subscribers) {
97        checkNotNull(event);
98        checkNotNull(subscribers);
99        Queue<Event> queueForThread = queue.get();
100       queueForThread.offer(new Event(event, subscribers));
101 
102       if (!dispatching.get()) {
103         dispatching.set(true);
104         try {
105           Event nextEvent;
106           while ((nextEvent = queueForThread.poll()) != null) {
107             while (nextEvent.subscribers.hasNext()) {
108               nextEvent.subscribers.next().dispatchEvent(nextEvent.event);
109             }
110           }
111         } finally {
112           dispatching.remove();
113           queue.remove();
114         }
115       }
116     }
117 
118     private static final class Event {
119       private final Object event;
120       private final Iterator<Subscriber> subscribers;
121 
122       private Event(Object event, Iterator<Subscriber> subscribers) {
123         this.event = event;
124         this.subscribers = subscribers;
125       }
126     }
127   }
128 
129   /** Implementation of a {@link #legacyAsync()} dispatcher. */
130   private static final class LegacyAsyncDispatcher extends Dispatcher {
131 
132     // This dispatcher matches the original dispatch behavior of AsyncEventBus.
133     //
134     // We can't really make any guarantees about the overall dispatch order for this dispatcher in
135     // a multithreaded environment for a couple reasons:
136     //
137     // 1. Subscribers to events posted on different threads can be interleaved with each other
138     //    freely. (A event on one thread, B event on another could yield any of
139     //    [a1, a2, a3, b1, b2], [a1, b2, a2, a3, b2], [a1, b2, b3, a2, a3], etc.)
140     // 2. It's possible for subscribers to actually be dispatched to in a different order than they
141     //    were added to the queue. It's easily possible for one thread to take the head of the
142     //    queue, immediately followed by another thread taking the next element in the queue. That
143     //    second thread can then dispatch to the subscriber it took before the first thread does.
144     //
145     // All this makes me really wonder if there's any value in queueing here at all. A dispatcher
146     // that simply loops through the subscribers and dispatches the event to each would actually
147     // probably provide a stronger order guarantee, though that order would obviously be different
148     // in some cases.
149 
150     /** Global event queue. */
151     private final ConcurrentLinkedQueue<EventWithSubscriber> queue =
152         Queues.newConcurrentLinkedQueue();
153 
154     @Override
155     void dispatch(Object event, Iterator<Subscriber> subscribers) {
156       checkNotNull(event);
157       while (subscribers.hasNext()) {
158         queue.add(new EventWithSubscriber(event, subscribers.next()));
159       }
160 
161       EventWithSubscriber e;
162       while ((e = queue.poll()) != null) {
163         e.subscriber.dispatchEvent(e.event);
164       }
165     }
166 
167     private static final class EventWithSubscriber {
168       private final Object event;
169       private final Subscriber subscriber;
170 
171       private EventWithSubscriber(Object event, Subscriber subscriber) {
172         this.event = event;
173         this.subscriber = subscriber;
174       }
175     }
176   }
177 
178   /** Implementation of {@link #immediate()}. */
179   private static final class ImmediateDispatcher extends Dispatcher {
180     private static final ImmediateDispatcher INSTANCE = new ImmediateDispatcher();
181 
182     @Override
183     void dispatch(Object event, Iterator<Subscriber> subscribers) {
184       checkNotNull(event);
185       while (subscribers.hasNext()) {
186         subscribers.next().dispatchEvent(event);
187       }
188     }
189   }
190 }

81      public Class<?> loadClass(String className) throws ClassNotFoundException {
82          return loadClass(className, getContextClassLoader());

85      /**
86       * @param className A non-null fully qualified class name.
87       * @param codebase  A non-null, space-separated list of URL strings.
88       */
89      public Class<?> loadClass(String className, String codebase)
90              throws ClassNotFoundException, MalformedURLException {
91  
92          ClassLoader parent = getContextClassLoader();
93          ClassLoader cl =
94                  URLClassLoader.newInstance(getUrlArray(codebase), parent);
95  
96          return loadClass(className, cl);
97      }

111     /*
112      * Returns a JNDI property from the system properties. Returns
113      * null if the property is not set, or if there is no permission
114      * to read it.
115      */
116     String getJndiProperty(int i) {
117         PrivilegedAction<String> act = () -> {
118             try {
119                 return System.getProperty(PROPS[i]);
120             } catch (SecurityException e) {
121                 return null;
122             }
123         };
124         return AccessController.doPrivileged(act);
125     }
126 
127     /*
128      * Reads each property in PROPS from the system properties, and
129      * returns their values -- in order -- in an array.  For each
130      * unset property, the corresponding array element is set to null.
131      * Returns null if there is no permission to call System.getProperties().
132      */
133     String[] getJndiProperties() {
134         PrivilegedAction<Properties> act = () -> {
135             try {
136                 return System.getProperties();
137             } catch (SecurityException e) {
138                 return null;
139             }
140         };
141         Properties sysProps = AccessController.doPrivileged(act);
142         if (sysProps == null) {
143             return null;
144         }
145         String[] jProps = new String[PROPS.length];
146         for (int i = 0; i < PROPS.length; i++) {
147             jProps[i] = sysProps.getProperty(PROPS[i]);
148         }
149         return jProps;
150     }

152     private static String resolveName(Class<?> c, String name) {
153         if (name == null) {
154             return name;
155         }
156         if (!name.startsWith("/")) {
157             while (c.isArray()) {
158                 c = c.getComponentType();
159             }
160             String baseName = c.getName();
161             int index = baseName.lastIndexOf('.');
162             if (index != -1) {
163                 name = baseName.substring(0, index).replace('.', '/')
164                     +"/"+name;
165             }
166         } else {
167             name = name.substring(1);
168         }
169         return name;
170     }

172     /*
173      * Returns the resource of a given name associated with a particular
174      * class (never null), or null if none can be found.
175      */
176     InputStream getResourceAsStream(Class<?> c, String name) {
177         PrivilegedAction<InputStream> act = () -> {
178             try {
179                 Module m = c.getModule();
180                 return c.getModule().getResourceAsStream(resolveName(c,name));
181              } catch (IOException x) {
182                  return null;
183              }
184         };
185         return AccessController.doPrivileged(act);
186     }

188     /*
189      * Returns an input stream for a file in <java.home>/conf,
190      * or null if it cannot be located or opened.
191      *
192      * @param filename  The file name, sans directory.
193      */
194     InputStream getJavaHomeConfStream(String filename) {
195         PrivilegedAction<InputStream> act = () -> {
196             try {
197                 String javahome = System.getProperty("java.home");
198                 if (javahome == null) {
199                     return null;
200                 }
201                 String pathname = javahome + File.separator +
202                         "conf" + File.separator + filename;
203                 return new FileInputStream(pathname);
204             } catch (Exception e) {
205                 return null;
206             }
207         };
208         return AccessController.doPrivileged(act);
209     }

211     /*
212      * Returns an enumeration (never null) of InputStreams of the
213      * resources of a given name associated with a particular class
214      * loader.  Null represents the bootstrap class loader in some
215      * Java implementations.
216      */
217     NamingEnumeration<InputStream> getResources(ClassLoader cl,
218                                                 String name) throws IOException {
219         Enumeration<URL> urls;
220         PrivilegedExceptionAction<Enumeration<URL>> act = () ->
221                 (cl == null)
222                         ? ClassLoader.getSystemResources(name)
223                         : cl.getResources(name);
224         try {
225             urls = AccessController.doPrivileged(act);
226         } catch (PrivilegedActionException e) {
227             throw (IOException) e.getException();
228         }
229         return new InputStreamEnumeration(urls);
230     }

233     /**
234      * Package private.
235      * <p>
236      * This internal method returns Thread Context Class Loader (TCCL),
237      * if null, returns the system Class Loader.
238      * <p>
239      * Please don't expose this method as public.
240      * @throws SecurityException if the class loader is not accessible
241      */
242     ClassLoader getContextClassLoader() {
243 
244         PrivilegedAction<ClassLoader> act = () -> {
245             ClassLoader loader = Thread.currentThread().getContextClassLoader();
246             if (loader == null) {
247                 // Don't use bootstrap class loader directly!
248                 loader = ClassLoader.getSystemClassLoader();
249             }
250             return loader;
251         };
252         return AccessController.doPrivileged(act);
253     }

255     private static URL[] getUrlArray(String codebase)
256             throws MalformedURLException {
257         // Parse codebase into separate URLs
258         StringTokenizer parser = new StringTokenizer(codebase);
259         List<String> list = new ArrayList<>();
260         while (parser.hasMoreTokens()) {
261             list.add(parser.nextToken());
262         }
263         String[] url = new String[list.size()];
264         for (int i = 0; i < url.length; i++) {
265             url[i] = list.get(i);
266         }
267 
268         URL[] urlArray = new URL[url.length];
269         for (int i = 0; i < urlArray.length; i++) {
270             urlArray[i] = new URL(url[i]);
271         }
272         return urlArray;
273     }

275     /**
276      * Given an enumeration of URLs, an instance of this class represents
277      * an enumeration of their InputStreams.  Each operation on the URL
278      * enumeration is performed within a doPrivileged block.
279      * This is used to enumerate the resources under a foreign codebase.
280      * This class is not MT-safe.
281      */
282     private class InputStreamEnumeration implements
283             NamingEnumeration<InputStream> {
284 
285         private final Enumeration<URL> urls;
286 
287         private InputStream nextElement;
288 
289         InputStreamEnumeration(Enumeration<URL> urls) {
290             this.urls = urls;
291         }
292 
293         /*
294          * Returns the next InputStream, or null if there are no more.
295          * An InputStream that cannot be opened is skipped.
296          */
297         private InputStream getNextElement() {
298             PrivilegedAction<InputStream> act = () -> {
299                 while (urls.hasMoreElements()) {
300                     try {
301                         return urls.nextElement().openStream();
302                     } catch (IOException e) {
303                         // skip this URL
304                     }
305                 }
306                 return null;
307             };
308             return AccessController.doPrivileged(act);
309         }
310 
311         public boolean hasMore() {
312             if (nextElement != null) {
313                 return true;
314             }
315             nextElement = getNextElement();
316             return (nextElement != null);
317         }
318 
319         public boolean hasMoreElements() {
320             return hasMore();
321         }
322 
323         public InputStream next() {
324             if (hasMore()) {
325                 InputStream res = nextElement;
326                 nextElement = null;
327                 return res;
328             } else {
329                 throw new NoSuchElementException();
330             }
331         }
332 
333         public InputStream nextElement() {
334             return next();
335         }
336 
337         public void close() {
338         }
339     }

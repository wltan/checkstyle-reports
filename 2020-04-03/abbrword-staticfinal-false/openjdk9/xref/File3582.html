43      private static void check(String encodedBlob) throws Exception {
44          byte[] blob = new byte[encodedBlob.length() * 2];
45          for (int i = 0; i < blob.length; ) {
46              final char ch = encodedBlob.charAt(i / 2);
47              blob[i++] = (byte) (ch >> 8);
48              blob[i++] = (byte) ch;
49          }
50          KeyStore store = KeyStore.getInstance("PKCS12");
51          store.load(new ByteArrayInputStream(blob), new char[0]);
52          if (!store.aliases().nextElement().equals("test"))
53              throw new Exception("test alias not found");
54          KeyStore.PrivateKeyEntry e =
55              (KeyStore.PrivateKeyEntry) store.getEntry("test",
56                      new KeyStore.PasswordProtection(new char[0]));
57          X509Certificate cert = (X509Certificate) e.getCertificateChain()[0];
58          if (!cert.getSubjectDN().toString().equals("CN=Test Key"))
59              throw new Exception("invalid certificate subject DN");
60          RSAPrivateCrtKey key = (RSAPrivateCrtKey) e.getPrivateKey();
61          if (!key.getPublicExponent().equals(BigInteger.valueOf(65537)))
62              throw new Exception("invalid public exponent");
63      }

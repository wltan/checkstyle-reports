24  package org.hibernate.ejb.criteria.basic;
25  
26  import java.math.BigDecimal;
27  import java.math.BigInteger;
28  import java.util.List;
29  import javax.persistence.EntityManager;
30  import javax.persistence.criteria.CriteriaBuilder;
31  import javax.persistence.criteria.CriteriaQuery;
32  import javax.persistence.criteria.Root;
33  
34  import org.junit.Assert;
35  import org.junit.Test;
36  
37  import org.hibernate.dialect.SybaseASE157Dialect;
38  import org.hibernate.dialect.SybaseASE15Dialect;
39  import org.hibernate.ejb.metamodel.AbstractMetamodelSpecificTest;
40  import org.hibernate.ejb.metamodel.Product;
41  import org.hibernate.ejb.metamodel.Product_;
42  import org.hibernate.testing.SkipForDialect;
43  import org.hibernate.testing.TestForIssue;
44  @SkipForDialect( value = SybaseASE15Dialect.class)
45  public class CastTest extends AbstractMetamodelSpecificTest {
46  	private static final int QUANTITY = 2;
47  
48  	@Test
49  	@TestForIssue( jiraKey = "HHH-5755" )
50  	public void testCastToString() {
51  		EntityManager em = getOrCreateEntityManager();
52  		em.getTransaction().begin();
53  		Product product = new Product();
54  		product.setId( "product1" );
55  		product.setPrice( 1.23d );
56  		product.setQuantity( QUANTITY );
57  		product.setPartNumber( ((long)Integer.MAX_VALUE) + 1 );
58  		product.setRating( 1.999f );
59  		product.setSomeBigInteger( BigInteger.valueOf( 987654321 ) );
60  		product.setSomeBigDecimal( BigDecimal.valueOf( 987654.321 ) );
61  		em.persist( product );
62  		em.getTransaction().commit();
63  		em.close();
64  
65  		em = getOrCreateEntityManager();
66  		em.getTransaction().begin();
67  		CriteriaBuilder builder = em.getCriteriaBuilder();
68  		CriteriaQuery<Product> criteria = builder.createQuery( Product.class );
69  		Root<Product> root = criteria.from( Product.class );
70  		criteria.where( builder.equal(root.get(Product_.quantity).as( String.class ), builder.literal(String.valueOf(QUANTITY))) );
71  		List<Product> result = em.createQuery( criteria ).getResultList();
72  		Assert.assertEquals( 1, result.size() );
73  		em.getTransaction().commit();
74  		em.close();
75  
76  		em = getOrCreateEntityManager();
77  		em.getTransaction().begin();
78  		em.createQuery( "delete Product" ).executeUpdate();
79  		em.getTransaction().commit();
80  		em.close();
81  	}
82  }

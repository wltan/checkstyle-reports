28  import org.hibernate.cfg.Configuration;
29  import org.hibernate.engine.spi.SessionFactoryImplementor;
30  import org.hibernate.envers.configuration.AuditConfiguration;
31  import org.hibernate.event.service.spi.EventListenerRegistry;
32  import org.hibernate.event.spi.EventType;
33  import org.hibernate.integrator.spi.Integrator;
34  import org.hibernate.internal.CoreMessageLogger;
35  import org.hibernate.internal.util.config.ConfigurationHelper;
36  import org.hibernate.metamodel.source.MetadataImplementor;
37  import org.hibernate.service.classloading.spi.ClassLoaderService;
38  import org.hibernate.service.spi.SessionFactoryServiceRegistry;
39  
40  /**
41   * Provides integration for Envers into Hibernate, which mainly means registering the proper event listeners.
42   *
43   * @author Steve Ebersole
44   */
45  public class EnversIntegrator implements Integrator {
46  	private static final CoreMessageLogger LOG = Logger.getMessageLogger( CoreMessageLogger.class, EnversIntegrator.class.getName() );
47  
48  	public static final String AUTO_REGISTER = "hibernate.listeners.envers.autoRegister";
49      private AuditConfiguration enversConfiguration;
50  
51  	@Override
52  	public void integrate(
53  			Configuration configuration,
54  			SessionFactoryImplementor sessionFactory,
55  			SessionFactoryServiceRegistry serviceRegistry) {
56  		final boolean autoRegister = ConfigurationHelper.getBoolean( AUTO_REGISTER, configuration.getProperties(), true );
57  		if ( !autoRegister ) {
58  			LOG.debug( "Skipping Envers listener auto registration" );
59  			return;
60  		}
61  
62  		EventListenerRegistry listenerRegistry = serviceRegistry.getService( EventListenerRegistry.class );
63  		listenerRegistry.addDuplicationStrategy( EnversListenerDuplicationStrategy.INSTANCE );
64  
65  		enversConfiguration = AuditConfiguration.getFor( configuration, serviceRegistry.getService( ClassLoaderService.class ) );
66  
67          if (enversConfiguration.getEntCfg().hasAuditedEntities()) {
68  		    listenerRegistry.appendListeners( EventType.POST_DELETE, new EnversPostDeleteEventListenerImpl( enversConfiguration ) );
69  		    listenerRegistry.appendListeners( EventType.POST_INSERT, new EnversPostInsertEventListenerImpl( enversConfiguration ) );
70  		    listenerRegistry.appendListeners( EventType.POST_UPDATE, new EnversPostUpdateEventListenerImpl( enversConfiguration ) );
71  		    listenerRegistry.appendListeners( EventType.POST_COLLECTION_RECREATE, new EnversPostCollectionRecreateEventListenerImpl( enversConfiguration ) );
72  		    listenerRegistry.appendListeners( EventType.PRE_COLLECTION_REMOVE, new EnversPreCollectionRemoveEventListenerImpl( enversConfiguration ) );
73  		    listenerRegistry.appendListeners( EventType.PRE_COLLECTION_UPDATE, new EnversPreCollectionUpdateEventListenerImpl( enversConfiguration ) );
74          }
75  	}
76  
77  	@Override
78  	public void disintegrate(SessionFactoryImplementor sessionFactory, SessionFactoryServiceRegistry serviceRegistry) {
79  		if ( enversConfiguration != null ) {
80  			enversConfiguration.destroy();
81  		}
82  	}
83  
84  	/**
85  	 * {@inheritDoc}
86  	 *
87  	 * @see org.hibernate.integrator.spi.Integrator#integrate(org.hibernate.metamodel.source.MetadataImplementor, org.hibernate.engine.spi.SessionFactoryImplementor, org.hibernate.service.spi.SessionFactoryServiceRegistry)
88  	 */
89  	@Override
90  	public void integrate( MetadataImplementor metadata,
91  	                       SessionFactoryImplementor sessionFactory,
92  	                       SessionFactoryServiceRegistry serviceRegistry ) {
93  	    // TODO: implement
94  	}
95  }

63      @Test
64      public void testNaif() throws OrekitException, UnsupportedEncodingException, IOException {
65          Utils.setDataRoot("regular-data");
66          final Frame refFrame = FramesFactory.getICRF();
67          final TimeScale tdb = TimeScalesFactory.getTDB();
68          final InputStream inEntry = getClass().getResourceAsStream("/naif/DE431-ephemeris-NAIF.txt");
69          BufferedReader reader = new BufferedReader(new InputStreamReader(inEntry, "UTF-8"));
70          for (String line = reader.readLine(); line != null; line = reader.readLine()) {
71              line = line.trim();
72              if (!line.isEmpty() && !line.startsWith("#")) {
73  
74                  // extract reference data from Naif
75                  String[] fields = line.split("\\s+");
76                  final AbsoluteDate date1 = new AbsoluteDate(fields[0], tdb);
77                  final AbsoluteDate date2 = new AbsoluteDate(AbsoluteDate.J2000_EPOCH,
78                                                              Double.parseDouble(fields[1]),
79                                                              tdb);
80                  String name       = fields[2];
81                  final String barycenter = fields[3];
82                  final Vector3D pRef     = new Vector3D(Double.parseDouble(fields[4]) * 1000.0,
83                                                         Double.parseDouble(fields[5]) * 1000.0,
84                                                         Double.parseDouble(fields[6]) * 1000.0);
85                  final Vector3D vRef     = new Vector3D(Double.parseDouble(fields[7]) * 1000.0,
86                                                         Double.parseDouble(fields[8]) * 1000.0,
87                                                         Double.parseDouble(fields[9]) * 1000.0);
88  
89                  // check position-velocity
90                  Assert.assertEquals("BARYCENTER", barycenter);
91                  if (name.equals("EARTH")) {
92                      name = "EARTH-MOON BARYCENTER";
93                  }
94                  Assert.assertEquals(0.0, date2.durationFrom(date1), 8.0e-5);
95                  final PVCoordinates pv = CelestialBodyFactory.getBody(name).getPVCoordinates(date2,
96                                                                                               refFrame);
97                                                    
98                  Assert.assertEquals(0.0, Vector3D.distance(pRef, pv.getPosition()), 15.0);
99                  Assert.assertEquals(0.0, Vector3D.distance(vRef, pv.getVelocity()), 1.0e-5);
100             }
101         }
102     }

190     private void testPOCoordinate(final int year, final int month, final int day,
191                                   final int targetNumber, final int centerNumber,
192                                   final int coordinateNumber, final double coordinateValue,
193                                   final double threshold)
194         throws OrekitException {
195         final AbsoluteDate date = new AbsoluteDate(year,  month, day, TimeScalesFactory.getTDB());
196         final CelestialBody target = getBody(targetNumber);
197         final CelestialBody center = getBody(centerNumber);
198         if (target != null && center != null) {
199             final PVCoordinates relativePV =
200                             new PVCoordinates(center.getPVCoordinates(date, FramesFactory.getICRF()),
201                                               target.getPVCoordinates(date, FramesFactory.getICRF()));
202             Assert.assertEquals(coordinateValue, getCoordinate(coordinateNumber, relativePV), threshold);
203         }
204     }

239     public double getCoordinate(final int number, final PVCoordinates pv) {
240         switch (number) {
241             case 1 :
242                 return pv.getPosition().getX() / Constants.JPL_SSD_ASTRONOMICAL_UNIT;
243             case 2 :
244                 return pv.getPosition().getY() / Constants.JPL_SSD_ASTRONOMICAL_UNIT;
245             case 3 :
246                 return pv.getPosition().getZ() / Constants.JPL_SSD_ASTRONOMICAL_UNIT;
247             case 4 :
248                 return pv.getVelocity().getX() * Constants.JULIAN_DAY / Constants.JPL_SSD_ASTRONOMICAL_UNIT;
249             case 5 :
250                 return pv.getVelocity().getY() * Constants.JULIAN_DAY / Constants.JPL_SSD_ASTRONOMICAL_UNIT;
251             case 6 :
252                 return pv.getVelocity().getZ() * Constants.JULIAN_DAY / Constants.JPL_SSD_ASTRONOMICAL_UNIT;
253             default :
254                 return Double.NaN;
255         }

324     @Test
325     public void testFrameShift() throws OrekitException {
326         Utils.setDataRoot("regular-data");
327         final Frame moon  = CelestialBodyFactory.getMoon().getBodyOrientedFrame();
328         final Frame earth = CelestialBodyFactory.getEarth().getBodyOrientedFrame();
329         final AbsoluteDate date0 = new AbsoluteDate(1969, 06, 25, TimeScalesFactory.getTDB());
330 
331         for (double t = 0; t < Constants.JULIAN_DAY; t += 3600) {
332             final AbsoluteDate date = date0.shiftedBy(t);
333             final Transform transform = earth.getTransformTo(moon, date);
334             for (double dt = -10; dt < 10; dt += 0.125) {
335                 final Transform shifted  = transform.shiftedBy(dt);
336                 final Transform computed = earth.getTransformTo(moon, transform.getDate().shiftedBy(dt));
337                 final Transform error    = new Transform(computed.getDate(), computed, shifted.getInverse());
338                 Assert.assertEquals(0.0, error.getTranslation().getNorm(),   100.0);
339                 Assert.assertEquals(0.0, error.getVelocity().getNorm(),       20.0);
340                 Assert.assertEquals(0.0, error.getRotation().getAngle(),    4.0e-8);
341                 Assert.assertEquals(0.0, error.getRotationRate().getNorm(), 8.0e-10);
342             }
343         }
344     }

346     @Test
347     public void testPropagationVsEphemeris() throws OrekitException {
348 
349         Utils.setDataRoot("regular-data");
350 
351         //Creation of the celestial bodies of the solar system
352         final CelestialBody sun     = CelestialBodyFactory.getSun();
353         final CelestialBody mercury = CelestialBodyFactory.getMercury();
354         final CelestialBody venus   = CelestialBodyFactory.getVenus();
355         final CelestialBody earth   = CelestialBodyFactory.getEarth();  
356         final CelestialBody mars    = CelestialBodyFactory.getMars();
357         final CelestialBody jupiter = CelestialBodyFactory.getJupiter();
358         final CelestialBody saturn  = CelestialBodyFactory.getSaturn();
359         final CelestialBody uranus  = CelestialBodyFactory.getUranus();
360         final CelestialBody neptune = CelestialBodyFactory.getNeptune();
361         final CelestialBody pluto   = CelestialBodyFactory.getPluto();
362 
363         //Starting and end dates
364         final AbsoluteDate startingDate = new AbsoluteDate(2000, 1, 2, TimeScalesFactory.getUTC());
365         AbsoluteDate endDate = startingDate.shiftedBy(30 * Constants.JULIAN_DAY);
366 
367         final Frame icrf = FramesFactory.getICRF();
368 
369         // fake orbit around negligible point mass at solar system barycenter
370         double negligibleMu = 1.0e-3;
371         SpacecraftState initialState = new SpacecraftState(new CartesianOrbit(venus.getPVCoordinates(startingDate, icrf),
372                                                            icrf, startingDate, negligibleMu));
373 
374         //Creation of the numerical propagator
375         final double[][] tol = NumericalPropagator.tolerances(1000, initialState.getOrbit(), OrbitType.CARTESIAN);
376         AbstractIntegrator dop1 = new DormandPrince853Integrator(1.0, 1.0e5, tol[0], tol[1]);
377         NumericalPropagator propag = new NumericalPropagator(dop1);
378         propag.setOrbitType(OrbitType.CARTESIAN);
379         propag.setInitialState(initialState);
380         propag.setMu(negligibleMu);
381 
382         //Creation of the ForceModels
383         propag.addForceModel(new BodyAttraction(sun)); 
384         propag.addForceModel(new BodyAttraction(mercury));
385         propag.addForceModel(new BodyAttraction(earth)); 
386         propag.addForceModel(new BodyAttraction(mars)); 
387         propag.addForceModel(new BodyAttraction(jupiter));
388         propag.addForceModel(new BodyAttraction(saturn));
389         propag.addForceModel(new BodyAttraction(uranus));
390         propag.addForceModel(new BodyAttraction(neptune));
391         propag.addForceModel(new BodyAttraction(pluto)); 
392 
393         // checks are done within the step handler
394         propag.setMasterMode(1000.0, new OrekitFixedStepHandler() {
395             public void init(SpacecraftState s0, AbsoluteDate t) {
396             }
397             public void handleStep(SpacecraftState currentState, boolean isLast)
398                 throws OrekitException {
399                 // propagated position should remain within 1400m of ephemeris for one month
400                 Vector3D propagatedP = currentState.getPVCoordinates(icrf).getPosition();
401                 Vector3D ephemerisP  = venus.getPVCoordinates(currentState.getDate(), icrf).getPosition();
402                 Assert.assertEquals(0, Vector3D.distance(propagatedP, ephemerisP), 1400.0);
403             }
404         });
405 
406         propag.propagate(startingDate,endDate);
407 
408     }

410     private static class BodyAttraction extends AbstractForceModel {
411 
412         /** Suffix for parameter name for attraction coefficient enabling jacobian processing. */
413         public static final String ATTRACTION_COEFFICIENT_SUFFIX = " attraction coefficient";
414 
415         /** Drivers for force model parameters. */
416         private final ParameterDriver[] parametersDrivers;
417 
418         /** The body to consider. */
419         private final CelestialBody body;
420 
421         /** Local value for body attraction coefficient. */
422         private double gm;
423 
424         /** Simple constructor.
425          * @param body the third body to consider
426          * (ex: {@link org.orekit.bodies.CelestialBodyFactory#getSun()} or
427          * {@link org.orekit.bodies.CelestialBodyFactory#getMoon()})
428          */
429         public BodyAttraction(final CelestialBody body) {
430             this.parametersDrivers = new ParameterDriver[1];
431             try {
432                 parametersDrivers[0] = new ParameterDriver(body.getName() + ATTRACTION_COEFFICIENT_SUFFIX,
433                                                            body.getGM(), 1.0e-5 * body.getGM(),
434                                                            0.0, Double.POSITIVE_INFINITY);
435                 parametersDrivers[0].addObserver(new ParameterObserver() {
436                     /** {@inheritDoc} */
437                     @Override
438                     public void valueChanged(double previousValue, final ParameterDriver driver) {
439                         BodyAttraction.this.gm = driver.getValue();
440                     }
441                 });
442             } catch (OrekitException oe) {
443                 // this should never occur as valueChanged above never throws an exception
444                 throw new OrekitInternalError(oe);
445             };
446             this.body = body;
447             this.gm   = body.getGM();
448         }
449 
450         /** {@inheritDoc} */
451         public void addContribution(final SpacecraftState s, final TimeDerivativesEquations adder)
452             throws OrekitException {
453 
454             // compute bodies separation vectors and squared norm
455             final Vector3D centralToBody = body.getPVCoordinates(s.getDate(), s.getFrame()).getPosition();
456             final Vector3D satToBody     = centralToBody.subtract(s.getPVCoordinates().getPosition());
457             final double r2Sat           = satToBody.getNormSq();
458 
459             // compute relative acceleration
460             final Vector3D gamma =
461                 new Vector3D(gm / (r2Sat * FastMath.sqrt(r2Sat)), satToBody);
462 
463             // add contribution to the ODE second member
464             adder.addXYZAcceleration(gamma.getX(), gamma.getY(), gamma.getZ());
465 
466         }
467 
468         /** {@inheritDoc} */
469         public FieldVector3D<DerivativeStructure> accelerationDerivatives(final AbsoluteDate date, final Frame frame,
470                                                                           final FieldVector3D<DerivativeStructure> position,
471                                                                           final FieldVector3D<DerivativeStructure> velocity,
472                                                                           final FieldRotation<DerivativeStructure> rotation,
473                                                                           final DerivativeStructure mass)
474             throws OrekitException {
475 
476             // compute bodies separation vectors and squared norm
477             final Vector3D centralToBody    = body.getPVCoordinates(date, frame).getPosition();
478             final FieldVector3D<DerivativeStructure> satToBody = position.subtract(centralToBody).negate();
479             final DerivativeStructure r2Sat = satToBody.getNormSq();
480 
481             // compute absolute acceleration
482             return new FieldVector3D<DerivativeStructure>(r2Sat.pow(-1.5).multiply(gm), satToBody);
483 
484         }
485 
486         /** {@inheritDoc} */
487         public FieldVector3D<DerivativeStructure> accelerationDerivatives(final SpacecraftState s, final String paramName)
488             throws OrekitException {
489 
490             complainIfNotSupported(paramName);
491 
492             // compute bodies separation vectors and squared norm
493             final Vector3D centralToBody = body.getPVCoordinates(s.getDate(), s.getFrame()).getPosition();
494             final Vector3D satToBody     = centralToBody.subtract(s.getPVCoordinates().getPosition());
495             final double r2Sat           = Vector3D.dotProduct(satToBody, satToBody);
496 
497             final DerivativeStructure gmds = new DerivativeStructure(1, 1, 0, gm);
498 
499             // compute relative acceleration

533     private void checkKepler(final PVCoordinatesProvider orbiting, final CelestialBody central,
534                              final AbsoluteDate start, final double a, final double epsilon)
535         throws OrekitException {
536 
537         // set up Keplerian orbit of orbiting body around central body
538         Orbit orbit = new KeplerianOrbit(orbiting.getPVCoordinates(start, central.getInertiallyOrientedFrame()),
539                                          central.getInertiallyOrientedFrame(),start, central.getGM());
540         KeplerianPropagator propagator = new KeplerianPropagator(orbit);
541         Assert.assertEquals(a, orbit.getA(), 0.02 * a);
542         double duration = FastMath.min(50 * Constants.JULIAN_DAY, 0.01 * orbit.getKeplerianPeriod());
543 
544         double max = 0;
545         for (AbsoluteDate date = start; date.durationFrom(start) < duration; date = date.shiftedBy(duration / 100)) {
546             PVCoordinates ephemPV = orbiting.getPVCoordinates(date, central.getInertiallyOrientedFrame());
547             PVCoordinates keplerPV = propagator.propagate(date).getPVCoordinates();
548             Vector3D error = keplerPV.getPosition().subtract(ephemPV.getPosition());
549             max = FastMath.max(max, error.getNorm());
550         }
551         Assert.assertTrue(max < epsilon * a);
552     }

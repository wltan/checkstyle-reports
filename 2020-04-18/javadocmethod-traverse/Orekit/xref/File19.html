35  /**
36   * This class handles circular orbital parameters.
37  
38   * <p>
39   * The parameters used internally are the circular elements which can be
40   * related to keplerian elements as follows:
41   *   <ul>
42   *     <li>a</li>
43   *     <li>e<sub>x</sub> = e cos(ω)</li>
44   *     <li>e<sub>y</sub> = e sin(ω)</li>
45   *     <li>i</li>
46   *     <li>Ω</li>
47   *     <li>α<sub>v</sub> = v + ω</li>
48   *   </ul>
49   * where Ω stands for the Right Ascension of the Ascending Node and
50   * α<sub>v</sub> stands for the true latitude argument
51   *
52   * <p>
53   * The conversion equations from and to keplerian elements given above hold only
54   * when both sides are unambiguously defined, i.e. when orbit is neither equatorial
55   * nor circular. When orbit is circular (but not equatorial), the circular
56   * parameters are still unambiguously defined whereas some keplerian elements
57   * (more precisely ω and Ω) become ambiguous. When orbit is equatorial,
58   * neither the keplerian nor the circular parameters can be defined unambiguously.
59   * {@link EquinoctialOrbit equinoctial orbits} is the recommended way to represent
60   * orbits.
61   * </p>
62   * <p>
63   * The instance <code>CircularOrbit</code> is guaranteed to be immutable.
64   * </p>
65   * @see    Orbit
66   * @see    KeplerianOrbit
67   * @see    CartesianOrbit
68   * @see    EquinoctialOrbit
69   * @author Luc Maisonobe
70   * @author Fabien Maussion
71   * @author V&eacute;ronique Pommier-Maurussane
72   */
73

101     /** Creates a new instance.
102      * @param a  semi-major axis (m)
103      * @param ex e cos(ω), first component of circular eccentricity vector
104      * @param ey e sin(ω), second component of circular eccentricity vector
105      * @param i inclination (rad)
106      * @param raan right ascension of ascending node (Ω, rad)
107      * @param alpha  an + ω, mean, eccentric or true latitude argument (rad)
108      * @param type type of latitude argument
109      * @param frame the frame in which are defined the parameters
110      * (<em>must</em> be a {@link Frame#isPseudoInertial pseudo-inertial frame})
111      * @param date date of the orbital parameters
112      * @param mu central attraction coefficient (m³/s²)
113      * @exception IllegalArgumentException if eccentricity is equal to 1 or larger or
114      * if frame is not a {@link Frame#isPseudoInertial pseudo-inertial frame}
115      */
116     public CircularOrbit(final double a, final double ex, final double ey,
117                          final double i, final double raan,
118                          final double alpha, final PositionAngle type,
119                          final Frame frame, final AbsoluteDate date, final double mu)
120         throws IllegalArgumentException {
121         super(frame, date, mu);
122         if (ex * ex + ey * ey >= 1.0) {
123             throw new OrekitIllegalArgumentException(OrekitMessages.HYPERBOLIC_ORBIT_NOT_HANDLED_AS,
124                                                      getClass().getName());
125         }
126         this.a    =  a;
127         this.ex   = ex;
128         this.ey   = ey;
129         this.i    = i;
130         this.raan = raan;
131 
132         switch (type) {
133             case MEAN :
134                 this.alphaV = eccentricToTrue(meanToEccentric(alpha));
135                 break;
136             case ECCENTRIC :
137                 this.alphaV = eccentricToTrue(alpha);
138                 break;
139             case TRUE :
140                 this.alphaV = alpha;
141                 break;
142             default :
143                 throw new OrekitInternalError(null);
144         }
145 
146         serializePV = false;
147 
148     }

200     /** Constructor from cartesian parameters.
201      *
202      * <p> The acceleration provided in {@code pvCoordinates} is accessible using
203      * {@link #getPVCoordinates()} and {@link #getPVCoordinates(Frame)}. All other methods
204      * use {@code mu} and the position to compute the acceleration, including
205      * {@link #shiftedBy(double)} and {@link #getPVCoordinates(AbsoluteDate, Frame)}.
206      *
207      * @param pvCoordinates the {@link PVCoordinates} in inertial frame
208      * @param frame the frame in which are defined the {@link PVCoordinates}
209      * (<em>must</em> be a {@link Frame#isPseudoInertial pseudo-inertial frame})
210      * @param mu central attraction coefficient (m³/s²)
211      * @exception IllegalArgumentException if frame is not a {@link
212      * Frame#isPseudoInertial pseudo-inertial frame}
213      */
214     public CircularOrbit(final TimeStampedPVCoordinates pvCoordinates, final Frame frame, final double mu)
215         throws IllegalArgumentException {
216         super(pvCoordinates, frame, mu);
217 
218         // compute semi-major axis
219         final Vector3D pvP = pvCoordinates.getPosition();
220         final Vector3D pvV = pvCoordinates.getVelocity();
221         final double r  = pvP.getNorm();
222         final double V2 = pvV.getNormSq();
223         final double rV2OnMu = r * V2 / mu;
224 
225         if (rV2OnMu > 2) {
226             throw new OrekitIllegalArgumentException(OrekitMessages.HYPERBOLIC_ORBIT_NOT_HANDLED_AS,
227                                                      getClass().getName());
228         }
229 
230         a = r / (2 - rV2OnMu);
231 
232         // compute inclination
233         final Vector3D momentum = pvCoordinates.getMomentum();
234         i = Vector3D.angle(momentum, Vector3D.PLUS_K);
235 
236         // compute right ascension of ascending node
237         final Vector3D node  = Vector3D.crossProduct(Vector3D.PLUS_K, momentum);
238         raan = FastMath.atan2(node.getY(), node.getX());
239 
240         // 2D-coordinates in the canonical frame
241         final double cosRaan = FastMath.cos(raan);
242         final double sinRaan = FastMath.sin(raan);
243         final double cosI    = FastMath.cos(i);
244         final double sinI    = FastMath.sin(i);
245         final double xP      = pvP.getX();
246         final double yP      = pvP.getY();
247         final double zP      = pvP.getZ();
248         final double x2      = (xP * cosRaan + yP * sinRaan) / a;
249         final double y2      = ((yP * cosRaan - xP * sinRaan) * cosI + zP * sinI) / a;
250 
251         // compute eccentricity vector
252         final double eSE    = Vector3D.dotProduct(pvP, pvV) / FastMath.sqrt(mu * a);
253         final double eCE    = rV2OnMu - 1;
254         final double e2     = eCE * eCE + eSE * eSE;
255         final double f      = eCE - e2;
256         final double g      = FastMath.sqrt(1 - e2) * eSE;
257         final double aOnR   = a / r;
258         final double a2OnR2 = aOnR * aOnR;
259         ex = a2OnR2 * (f * x2 + g * y2);
260         ey = a2OnR2 * (f * y2 - g * x2);
261 
262         // compute latitude argument
263         final double beta = 1 / (1 + FastMath.sqrt(1 - ex * ex - ey * ey));
264         alphaV = eccentricToTrue(FastMath.atan2(y2 + ey + eSE * beta * ex, x2 + ex - eSE * beta * ey));
265 
266         serializePV = true;
267 
268     }

291     /** Constructor from any kind of orbital parameters.
292      * @param op orbital parameters to copy
293      */
294     public CircularOrbit(final Orbit op) {
295         super(op.getFrame(), op.getDate(), op.getMu());
296         a    = op.getA();
297         i    = op.getI();
298         raan = FastMath.atan2(op.getHy(), op.getHx());
299         final double cosRaan = FastMath.cos(raan);
300         final double sinRaan = FastMath.sin(raan);
301         final double equiEx = op.getEquinoctialEx();
302         final double equiEy = op.getEquinoctialEy();
303         ex   = equiEx * cosRaan + equiEy * sinRaan;
304         ey   = equiEy * cosRaan - equiEx * sinRaan;
305         this.alphaV = op.getLv() - raan;
306         serializePV = false;
307     }

370     /** Get the eccentric latitude argument.
371      * @return E + ω eccentric latitude argument (rad)
372      */
373     public double getAlphaE() {
374         final double epsilon   = FastMath.sqrt(1 - ex * ex - ey * ey);
375         final double cosAlphaV = FastMath.cos(alphaV);
376         final double sinAlphaV = FastMath.sin(alphaV);
377         return alphaV + 2 * FastMath.atan((ey * cosAlphaV - ex * sinAlphaV) /
378                                       (epsilon + 1 + ex * cosAlphaV + ey * sinAlphaV));
379     }

381     /** Computes the true latitude argument from the eccentric latitude argument.
382      * @param alphaE = E + ω eccentric latitude argument (rad)
383      * @return the true latitude argument.
384      */
385     private double eccentricToTrue(final double alphaE) {
386         final double epsilon   = FastMath.sqrt(1 - ex * ex - ey * ey);
387         final double cosAlphaE = FastMath.cos(alphaE);
388         final double sinAlphaE = FastMath.sin(alphaE);
389         return alphaE + 2 * FastMath.atan((ex * sinAlphaE - ey * cosAlphaE) /
390                                       (epsilon + 1 - ex * cosAlphaE - ey * sinAlphaE));
391     }

393     /** Get the mean latitude argument.
394      * @return M + ω mean latitude argument (rad)
395      */
396     public double getAlphaM() {
397         final double alphaE = getAlphaE();
398         return alphaE - ex * FastMath.sin(alphaE) + ey * FastMath.cos(alphaE);
399     }

401     /** Computes the eccentric latitude argument from the mean latitude argument.
402      * @param alphaM = M + ω  mean latitude argument (rad)
403      * @return the eccentric latitude argument.
404      */
405     private double meanToEccentric(final double alphaM) {
406         // Generalization of Kepler equation to circular parameters
407         // with alphaE = PA + E and
408         //      alphaM = PA + M = alphaE - ex.sin(alphaE) + ey.cos(alphaE)
409         double alphaE        = alphaM;
410         double shift         = 0.0;
411         double alphaEMalphaM = 0.0;
412         double cosAlphaE     = FastMath.cos(alphaE);
413         double sinAlphaE     = FastMath.sin(alphaE);
414         int    iter          = 0;
415         do {
416             final double f2 = ex * sinAlphaE - ey * cosAlphaE;
417             final double f1 = 1.0 - ex * cosAlphaE - ey * sinAlphaE;
418             final double f0 = alphaEMalphaM - f2;
419 
420             final double f12 = 2.0 * f1;
421             shift = f0 * f12 / (f1 * f12 - f0 * f2);
422 
423             alphaEMalphaM -= shift;
424             alphaE         = alphaM + alphaEMalphaM;
425             cosAlphaE      = FastMath.cos(alphaE);
426             sinAlphaE      = FastMath.sin(alphaE);
427 
428         } while ((++iter < 50) && (FastMath.abs(shift) > 1.0e-12));
429 
430         return alphaE;
431 
432     }

466     /** {@inheritDoc} */
467     protected TimeStampedPVCoordinates initPVCoordinates() {
468 
469         // get equinoctial parameters
470         final double equEx = getEquinoctialEx();
471         final double equEy = getEquinoctialEy();
472         final double hx = getHx();
473         final double hy = getHy();
474         final double lE = getLE();
475 
476         // inclination-related intermediate parameters
477         final double hx2   = hx * hx;
478         final double hy2   = hy * hy;
479         final double factH = 1. / (1 + hx2 + hy2);
480 
481         // reference axes defining the orbital plane
482         final double ux = (1 + hx2 - hy2) * factH;
483         final double uy =  2 * hx * hy * factH;
484         final double uz = -2 * hy * factH;
485 
486         final double vx = uy;
487         final double vy = (1 - hx2 + hy2) * factH;
488         final double vz =  2 * hx * factH;
489 
490         // eccentricity-related intermediate parameters
491         final double exey = equEx * equEy;
492         final double ex2  = equEx * equEx;
493         final double ey2  = equEy * equEy;
494         final double e2   = ex2 + ey2;
495         final double eta  = 1 + FastMath.sqrt(1 - e2);
496         final double beta = 1. / eta;
497 
498         // eccentric latitude argument
499         final double cLe    = FastMath.cos(lE);
500         final double sLe    = FastMath.sin(lE);
501         final double exCeyS = equEx * cLe + equEy * sLe;
502 
503         // coordinates of position and velocity in the orbital plane
504         final double x      = a * ((1 - beta * ey2) * cLe + beta * exey * sLe - equEx);
505         final double y      = a * ((1 - beta * ex2) * sLe + beta * exey * cLe - equEy);
506 
507         final double factor = FastMath.sqrt(getMu() / a) / (1 - exCeyS);
508         final double xdot   = factor * (-sLe + beta * equEy * exCeyS);
509         final double ydot   = factor * ( cLe - beta * equEx * exCeyS);
510 
511         final Vector3D position =
512             new Vector3D(x * ux + y * vx, x * uy + y * vy, x * uz + y * vz);
513         final double r2         = position.getNormSq();
514         final Vector3D velocity =
515             new Vector3D(xdot * ux + ydot * vx, xdot * uy + ydot * vy, xdot * uz + ydot * vz);
516         final Vector3D acceleration = new Vector3D(-getMu() / (r2 * FastMath.sqrt(r2)), position);
517         return new TimeStampedPVCoordinates(getDate(), position, velocity, acceleration);
518 
519     }

529     /** {@inheritDoc}
530      * <p>
531      * The interpolated instance is created by polynomial Hermite interpolation
532      * on circular elements, without derivatives (which means the interpolation
533      * falls back to Lagrange interpolation only).
534      * </p>
535      * <p>
536      * As this implementation of interpolation is polynomial, it should be used only
537      * with small samples (about 10-20 points) in order to avoid <a
538      * href="http://en.wikipedia.org/wiki/Runge%27s_phenomenon">Runge's phenomenon</a>
539      * and numerical problems (including NaN appearing).
540      * </p>
541      * <p>
542      * If orbit interpolation on large samples is needed, using the {@link
543      * org.orekit.propagation.analytical.Ephemeris} class is a better way than using this
544      * low-level interpolation. The Ephemeris class automatically handles selection of
545      * a neighboring sub-sample with a predefined number of point from a large global sample
546      * in a thread-safe way.
547      * </p>
548      */
549     public CircularOrbit interpolate(final AbsoluteDate date, final Collection<Orbit> sample) {
550 
551         // set up an interpolator
552         final HermiteInterpolator interpolator = new HermiteInterpolator();
553 
554         // add sample points
555         AbsoluteDate previousDate = null;
556         double previousRAAN   = Double.NaN;
557         double previousAlphaM = Double.NaN;
558         for (final Orbit orbit : sample) {
559             final CircularOrbit circ = (CircularOrbit) OrbitType.CIRCULAR.convertType(orbit);
560             final double continuousRAAN;
561             final double continuousAlphaM;
562             if (previousDate == null) {
563                 continuousRAAN   = circ.getRightAscensionOfAscendingNode();
564                 continuousAlphaM = circ.getAlphaM();
565             } else {
566                 final double dt       = circ.getDate().durationFrom(previousDate);
567                 final double keplerAM = previousAlphaM + circ.getKeplerianMeanMotion() * dt;
568                 continuousRAAN   = MathUtils.normalizeAngle(circ.getRightAscensionOfAscendingNode(), previousRAAN);
569                 continuousAlphaM = MathUtils.normalizeAngle(circ.getAlphaM(), keplerAM);
570             }
571             previousDate   = circ.getDate();
572             previousRAAN   = continuousRAAN;
573             previousAlphaM = continuousAlphaM;
574             interpolator.addSamplePoint(circ.getDate().durationFrom(date),
575                                         new double[] {
576                                             circ.getA(),
577                                             circ.getCircularEx(),
578                                             circ.getCircularEy(),
579                                             circ.getI(),
580                                             continuousRAAN,
581                                             continuousAlphaM
582                                         });
583         }
584 
585         // interpolate
586         final double[] interpolated = interpolator.value(0);
587 
588         // build a new interpolated instance
589         return new CircularOrbit(interpolated[0], interpolated[1], interpolated[2],
590                                  interpolated[3], interpolated[4], interpolated[5],
591                                  PositionAngle.MEAN, getFrame(), date, getMu());
592 
593     }

595     /** {@inheritDoc} */
596     protected double[][] computeJacobianMeanWrtCartesian() {
597 
598         final double[][] jacobian = new double[6][6];
599 
600         // compute various intermediate parameters
601         final PVCoordinates pvc = getPVCoordinates();
602         final Vector3D position = pvc.getPosition();
603         final Vector3D velocity = pvc.getVelocity();
604 
605         final double x          = position.getX();
606         final double y          = position.getY();
607         final double z          = position.getZ();
608         final double vx         = velocity.getX();
609         final double vy         = velocity.getY();
610         final double vz         = velocity.getZ();
611         final double pv         = Vector3D.dotProduct(position, velocity);
612         final double r2         = position.getNormSq();
613         final double r          = FastMath.sqrt(r2);
614         final double v2         = velocity.getNormSq();
615 
616         final double mu         = getMu();
617         final double oOsqrtMuA  = 1 / FastMath.sqrt(mu * a);
618         final double rOa        = r / a;
619         final double aOr        = a / r;
620         final double aOr2       = a / r2;
621         final double a2         = a * a;
622 
623         final double ex2        = ex * ex;
624         final double ey2        = ey * ey;
625         final double e2         = ex2 + ey2;
626         final double epsilon    = FastMath.sqrt(1 - e2);
627         final double beta       = 1 / (1 + epsilon);
628 
629         final double eCosE      = 1 - rOa;
630         final double eSinE      = pv * oOsqrtMuA;
631 
632         final double cosI       = FastMath.cos(i);
633         final double sinI       = FastMath.sin(i);
634         final double cosRaan    = FastMath.cos(raan);
635         final double sinRaan    = FastMath.sin(raan);
636 
637         // da
638         fillHalfRow(2 * aOr * aOr2, position, jacobian[0], 0);
639         fillHalfRow(2 * a2 / mu, velocity, jacobian[0], 3);
640 
641         // differentials of the normalized momentum
642         final Vector3D danP = new Vector3D(v2, position, -pv, velocity);
643         final Vector3D danV = new Vector3D(r2, velocity, -pv, position);
644         final double recip  = 1 / pvc.getMomentum().getNorm();
645         final double recip2 = recip * recip;
646         final Vector3D dwXP = new Vector3D(recip, new Vector3D(  0,  vz, -vy), -recip2 * sinRaan * sinI, danP);
647         final Vector3D dwYP = new Vector3D(recip, new Vector3D(-vz,   0,  vx),  recip2 * cosRaan * sinI, danP);
648         final Vector3D dwZP = new Vector3D(recip, new Vector3D( vy, -vx,   0), -recip2 * cosI,           danP);
649         final Vector3D dwXV = new Vector3D(recip, new Vector3D(  0,  -z,   y), -recip2 * sinRaan * sinI, danV);
650         final Vector3D dwYV = new Vector3D(recip, new Vector3D(  z,   0,  -x),  recip2 * cosRaan * sinI, danV);
651         final Vector3D dwZV = new Vector3D(recip, new Vector3D( -y,   x,   0), -recip2 * cosI,           danV);
652 
653         // di
654         fillHalfRow(sinRaan * cosI, dwXP, -cosRaan * cosI, dwYP, -sinI, dwZP, jacobian[3], 0);
655         fillHalfRow(sinRaan * cosI, dwXV, -cosRaan * cosI, dwYV, -sinI, dwZV, jacobian[3], 3);
656 
657         // dRaan
658         fillHalfRow(sinRaan / sinI, dwYP, cosRaan / sinI, dwXP, jacobian[4], 0);
659         fillHalfRow(sinRaan / sinI, dwYV, cosRaan / sinI, dwXV, jacobian[4], 3);
660 
661         // orbital frame: (p, q, w) p along ascending node, w along momentum
662         // the coordinates of the spacecraft in this frame are: (u, v, 0)
663         final double u     =  x * cosRaan + y * sinRaan;
664         final double cv    = -x * sinRaan + y * cosRaan;
665         final double v     = cv * cosI + z * sinI;
666 
667         // du
668         final Vector3D duP = new Vector3D(cv * cosRaan / sinI, dwXP,
669                                           cv * sinRaan / sinI, dwYP,
670                                           1, new Vector3D(cosRaan, sinRaan, 0));
671         final Vector3D duV = new Vector3D(cv * cosRaan / sinI, dwXV,
672                                           cv * sinRaan / sinI, dwYV);
673 
674         // dv
675         final Vector3D dvP = new Vector3D(-u * cosRaan * cosI / sinI + sinRaan * z, dwXP,
676                                           -u * sinRaan * cosI / sinI - cosRaan * z, dwYP,
677                                           cv, dwZP,
678                                           1, new Vector3D(-sinRaan * cosI, cosRaan * cosI, sinI));
679         final Vector3D dvV = new Vector3D(-u * cosRaan * cosI / sinI + sinRaan * z, dwXV,
680                                           -u * sinRaan * cosI / sinI - cosRaan * z, dwYV,
681                                           cv, dwZV);
682 
683         final Vector3D dc1P = new Vector3D(aOr2 * (2 * eSinE * eSinE + 1 - eCosE) / r2, position,
684                                             -2 * aOr2 * eSinE * oOsqrtMuA, velocity);
685         final Vector3D dc1V = new Vector3D(-2 * aOr2 * eSinE * oOsqrtMuA, position,
686                                             2 / mu, velocity);
687         final Vector3D dc2P = new Vector3D(aOr2 * eSinE * (eSinE * eSinE - (1 - e2)) / (r2 * epsilon), position,
688                                             aOr2 * (1 - e2 - eSinE * eSinE) * oOsqrtMuA / epsilon, velocity);
689         final Vector3D dc2V = new Vector3D(aOr2 * (1 - e2 - eSinE * eSinE) * oOsqrtMuA / epsilon, position,
690                                             eSinE / (mu * epsilon), velocity);
691 
692         final double cof1   = aOr2 * (eCosE - e2);
693         final double cof2   = aOr2 * epsilon * eSinE;
694         final Vector3D dexP = new Vector3D(u, dc1P,  v, dc2P, cof1, duP,  cof2, dvP);
695         final Vector3D dexV = new Vector3D(u, dc1V,  v, dc2V, cof1, duV,  cof2, dvV);
696         final Vector3D deyP = new Vector3D(v, dc1P, -u, dc2P, cof1, dvP, -cof2, duP);
697         final Vector3D deyV = new Vector3D(v, dc1V, -u, dc2V, cof1, dvV, -cof2, duV);
698         fillHalfRow(1, dexP, jacobian[1], 0);
699         fillHalfRow(1, dexV, jacobian[1], 3);
700         fillHalfRow(1, deyP, jacobian[2], 0);
701         fillHalfRow(1, deyV, jacobian[2], 3);
702 
703         final double cle = u / a + ex - eSinE * beta * ey;
704         final double sle = v / a + ey + eSinE * beta * ex;
705         final double m1  = beta * eCosE;
706         final double m2  = 1 - m1 * eCosE;
707         final double m3  = (u * ey - v * ex) + eSinE * beta * (u * ex + v * ey);
708         final double m4  = -sle + cle * eSinE * beta;
709         final double m5  = cle + sle * eSinE * beta;
710         fillHalfRow((2 * m3 / r + aOr * eSinE + m1 * eSinE * (1 + m1 - (1 + aOr) * m2) / epsilon) / r2, position,
711                     (m1 * m2 / epsilon - 1) * oOsqrtMuA, velocity,
712                     m4, dexP, m5, deyP, -sle / a, duP, cle / a, dvP,
713                     jacobian[5], 0);
714         fillHalfRow((m1 * m2 / epsilon - 1) * oOsqrtMuA, position,
715                     (2 * m3 + eSinE * a + m1 * eSinE * r * (eCosE * beta * 2 - aOr * m2) / epsilon) / mu, velocity,
716                     m4, dexV, m5, deyV, -sle / a, duV, cle / a, dvV,
717                     jacobian[5], 3);
718 
719         return jacobian;
720 
721     }

723     /** {@inheritDoc} */
724     protected double[][] computeJacobianEccentricWrtCartesian() {
725 
726         // start by computing the Jacobian with mean angle
727         final double[][] jacobian = computeJacobianMeanWrtCartesian();
728 
729         // Differentiating the Kepler equation aM = aE - ex sin aE + ey cos aE leads to:
730         // daM = (1 - ex cos aE - ey sin aE) daE - sin aE dex + cos aE dey
731         // which is inverted and rewritten as:
732         // daE = a/r daM + sin aE a/r dex - cos aE a/r dey
733         final double alphaE = getAlphaE();
734         final double cosAe  = FastMath.cos(alphaE);
735         final double sinAe  = FastMath.sin(alphaE);
736         final double aOr    = 1 / (1 - ex * cosAe - ey * sinAe);
737 
738         // update longitude row
739         final double[] rowEx = jacobian[1];
740         final double[] rowEy = jacobian[2];
741         final double[] rowL  = jacobian[5];
742         for (int j = 0; j < 6; ++j) {
743             rowL[j] = aOr * (rowL[j] + sinAe * rowEx[j] - cosAe * rowEy[j]);
744         }
745 
746         return jacobian;
747 
748     }

750     /** {@inheritDoc} */
751     protected double[][] computeJacobianTrueWrtCartesian() {
752 
753         // start by computing the Jacobian with eccentric angle
754         final double[][] jacobian = computeJacobianEccentricWrtCartesian();
755 
756         // Differentiating the eccentric latitude equation
757         // tan((aV - aE)/2) = [ex sin aE - ey cos aE] / [sqrt(1-ex^2-ey^2) + 1 - ex cos aE - ey sin aE]
758         // leads to
759         // cT (daV - daE) = cE daE + cX dex + cY dey
760         // with
761         // cT = [d^2 + (ex sin aE - ey cos aE)^2] / 2
762         // d  = 1 + sqrt(1-ex^2-ey^2) - ex cos aE - ey sin aE
763         // cE = (ex cos aE + ey sin aE) (sqrt(1-ex^2-ey^2) + 1) - ex^2 - ey^2
764         // cX =  sin aE (sqrt(1-ex^2-ey^2) + 1) - ey + ex (ex sin aE - ey cos aE) / sqrt(1-ex^2-ey^2)
765         // cY = -cos aE (sqrt(1-ex^2-ey^2) + 1) + ex + ey (ex sin aE - ey cos aE) / sqrt(1-ex^2-ey^2)
766         // which can be solved to find the differential of the true latitude
767         // daV = (cT + cE) / cT daE + cX / cT deX + cY / cT deX
768         final double alphaE    = getAlphaE();
769         final double cosAe     = FastMath.cos(alphaE);
770         final double sinAe     = FastMath.sin(alphaE);
771         final double eSinE     = ex * sinAe - ey * cosAe;
772         final double ecosE     = ex * cosAe + ey * sinAe;
773         final double e2        = ex * ex + ey * ey;
774         final double epsilon   = FastMath.sqrt(1 - e2);
775         final double onePeps   = 1 + epsilon;
776         final double d         = onePeps - ecosE;
777         final double cT        = (d * d + eSinE * eSinE) / 2;
778         final double cE        = ecosE * onePeps - e2;
779         final double cX        = ex * eSinE / epsilon - ey + sinAe * onePeps;
780         final double cY        = ey * eSinE / epsilon + ex - cosAe * onePeps;
781         final double factorLe  = (cT + cE) / cT;
782         final double factorEx  = cX / cT;
783         final double factorEy  = cY / cT;
784 
785         // update latitude row
786         final double[] rowEx = jacobian[1];
787         final double[] rowEy = jacobian[2];
788         final double[] rowA = jacobian[5];
789         for (int j = 0; j < 6; ++j) {
790             rowA[j] = factorLe * rowA[j] + factorEx * rowEx[j] + factorEy * rowEy[j];
791         }
792 
793         return jacobian;
794 
795     }

797     /** {@inheritDoc} */
798     public void addKeplerContribution(final PositionAngle type, final double gm,
799                                       final double[] pDot) {
800         final double oMe2;
801         final double ksi;
802         final double n = FastMath.sqrt(gm / a) / a;
803         switch (type) {
804             case MEAN :
805                 pDot[5] += n;
806                 break;
807             case ECCENTRIC :
808                 oMe2  = 1 - ex * ex - ey * ey;
809                 ksi   = 1 + ex * FastMath.cos(alphaV) + ey * FastMath.sin(alphaV);
810                 pDot[5] += n * ksi / oMe2;
811                 break;
812             case TRUE :
813                 oMe2  = 1 - ex * ex - ey * ey;
814                 ksi   = 1 + ex * FastMath.cos(alphaV) + ey * FastMath.sin(alphaV);
815                 pDot[5] += n * ksi * ksi / (oMe2 * FastMath.sqrt(oMe2));
816                 break;
817             default :
818                 throw new OrekitInternalError(null);
819         }
820     }

849         private double[] d;

854         /** Simple constructor.
855          * @param orbit instance to serialize
856          */
857         private DTO(final CircularOrbit orbit) {
858 
859             final AbsoluteDate date = orbit.getDate();
860 
861             // decompose date
862             final double epoch  = FastMath.floor(date.durationFrom(AbsoluteDate.J2000_EPOCH));
863             final double offset = date.durationFrom(AbsoluteDate.J2000_EPOCH.shiftedBy(epoch));
864 
865             if (orbit.serializePV) {
866                 final TimeStampedPVCoordinates pv = orbit.getPVCoordinates();
867                 this.d = new double[] {
868                     epoch, offset, orbit.getMu(),
869                     orbit.a, orbit.ex, orbit.ey,
870                     orbit.i, orbit.raan, orbit.alphaV,
871                     pv.getPosition().getX(),     pv.getPosition().getY(),     pv.getPosition().getZ(),
872                     pv.getVelocity().getX(),     pv.getVelocity().getY(),     pv.getVelocity().getZ(),
873                     pv.getAcceleration().getX(), pv.getAcceleration().getY(), pv.getAcceleration().getZ(),
874                 };
875             } else {
876                 this.d = new double[] {
877                     epoch, offset, orbit.getMu(),
878                     orbit.a, orbit.ex, orbit.ey,
879                     orbit.i, orbit.raan, orbit.alphaV
880                 };
881             }
882 
883             this.frame = orbit.getFrame();
884 
885         }

38  /** This class holds cartesian orbital parameters.
39  
40   * <p>
41   * The parameters used internally are the cartesian coordinates:
42   *   <ul>
43   *     <li>x</li>
44   *     <li>y</li>
45   *     <li>z</li>
46   *     <li>xDot</li>
47   *     <li>yDot</li>
48   *     <li>zDot</li>
49   *   </ul>
50   * contained in {@link PVCoordinates}.
51   *
52  
53   * <p>
54   * Note that the implementation of this class delegates all non-cartesian related
55   * computations ({@link #getA()}, {@link #getEquinoctialEx()}, ...) to an underlying
56   * instance of the {@link EquinoctialOrbit} class. This implies that using this class
57   * only for analytical computations which are always based on non-cartesian
58   * parameters is perfectly possible but somewhat sub-optimal.
59   * </p>
60   * <p>
61   * The instance <code>CartesianOrbit</code> is guaranteed to be immutable.
62   * </p>
63   * @see    Orbit
64   * @see    KeplerianOrbit
65   * @see    CircularOrbit
66   * @see    EquinoctialOrbit
67   * @author Luc Maisonobe
68   * @author Guylaine Prat
69   * @author Fabien Maussion
70   * @author V&eacute;ronique Pommier-Maurussane
71   */
72  public class CartesianOrbit extends Orbit {

148     /** {@inheritDoc} */
149     public double getA() {
150         // lazy evaluation of semi-major axis
151         final double r  = getPVCoordinates().getPosition().getNorm();
152         final double V2 = getPVCoordinates().getVelocity().getNormSq();
153         return r / (2 - r * V2 / getMu());
154     }

156     /** {@inheritDoc} */
157     public double getE() {
158         final Vector3D pvP   = getPVCoordinates().getPosition();
159         final Vector3D pvV   = getPVCoordinates().getVelocity();
160         final double rV2OnMu = pvP.getNorm() * pvV.getNormSq() / getMu();
161         final double eSE     = Vector3D.dotProduct(pvP, pvV) / FastMath.sqrt(getMu() * getA());
162         final double eCE     = rV2OnMu - 1;
163         return FastMath.sqrt(eCE * eCE + eSE * eSE);
164     }

183     /** {@inheritDoc} */
184     public double getHx() {
185         final Vector3D w = getPVCoordinates().getMomentum().normalize();
186         // Check for equatorial retrograde orbit
187         if (((w.getX() * w.getX() + w.getY() * w.getY()) == 0) && w.getZ() < 0) {
188             return Double.NaN;
189         }
190         return -w.getY() / (1 + w.getZ());
191     }

193     /** {@inheritDoc} */
194     public double getHy() {
195         final Vector3D w = getPVCoordinates().getMomentum().normalize();
196         // Check for equatorial retrograde orbit
197         if (((w.getX() * w.getX() + w.getY() * w.getY()) == 0) && w.getZ() < 0) {
198             return Double.NaN;
199         }
200         return  w.getX() / (1 + w.getZ());
201     }

233     /** {@inheritDoc}
234      * <p>
235      * The interpolated instance is created by polynomial Hermite interpolation
236      * ensuring velocity remains the exact derivative of position.
237      * </p>
238      * <p>
239      * As this implementation of interpolation is polynomial, it should be used only
240      * with small samples (about 10-20 points) in order to avoid <a
241      * href="http://en.wikipedia.org/wiki/Runge%27s_phenomenon">Runge's phenomenon</a>
242      * and numerical problems (including NaN appearing).
243      * </p>
244      * <p>
245      * If orbit interpolation on large samples is needed, using the {@link
246      * org.orekit.propagation.analytical.Ephemeris} class is a better way than using this
247      * low-level interpolation. The Ephemeris class automatically handles selection of
248      * a neighboring sub-sample with a predefined number of point from a large global sample
249      * in a thread-safe way.
250      * </p>
251      */
252     public CartesianOrbit interpolate(final AbsoluteDate date, final Collection<Orbit> sample) {
253         final List<TimeStampedPVCoordinates> datedPV = new ArrayList<TimeStampedPVCoordinates>(sample.size());
254         for (final Orbit o : sample) {
255             datedPV.add(new TimeStampedPVCoordinates(o.getDate(),
256                                                      o.getPVCoordinates().getPosition(),
257                                                      o.getPVCoordinates().getVelocity(),
258                                                      o.getPVCoordinates().getAcceleration()));
259         }
260         final TimeStampedPVCoordinates interpolated =
261                 TimeStampedPVCoordinates.interpolate(date, CartesianDerivativesFilter.USE_PVA, datedPV);
262         return new CartesianOrbit(interpolated, getFrame(), date, getMu());
263     }

265     /** Compute shifted position and velocity in elliptic case.
266      * @param dt time shift
267      * @return shifted position and velocity
268      */
269     private PVCoordinates shiftPVElliptic(final double dt) {
270 
271         // preliminary computation
272         final Vector3D pvP   = getPVCoordinates().getPosition();
273         final Vector3D pvV   = getPVCoordinates().getVelocity();
274         final double r       = pvP.getNorm();
275         final double rV2OnMu = r * pvV.getNormSq() / getMu();
276         final double a       = getA();
277         final double eSE     = Vector3D.dotProduct(pvP, pvV) / FastMath.sqrt(getMu() * a);
278         final double eCE     = rV2OnMu - 1;
279         final double e2      = eCE * eCE + eSE * eSE;
280 
281         // we can use any arbitrary reference 2D frame in the orbital plane
282         // in order to simplify some equations below, we use the current position as the u axis
283         final Vector3D u     = pvP.normalize();
284         final Vector3D v     = Vector3D.crossProduct(getPVCoordinates().getMomentum(), u).normalize();
285 
286         // the following equations rely on the specific choice of u explained above,
287         // some coefficients that vanish to 0 in this case have already been removed here
288         final double ex      = (eCE - e2) * a / r;
289         final double ey      = -FastMath.sqrt(1 - e2) * eSE * a / r;
290         final double beta    = 1 / (1 + FastMath.sqrt(1 - e2));
291         final double thetaE0 = FastMath.atan2(ey + eSE * beta * ex, r / a + ex - eSE * beta * ey);
292         final double thetaM0 = thetaE0 - ex * FastMath.sin(thetaE0) + ey * FastMath.cos(thetaE0);
293 
294         // compute in-plane shifted eccentric argument
295         final double thetaM1 = thetaM0 + getKeplerianMeanMotion() * dt;
296         final double thetaE1 = meanToEccentric(thetaM1, ex, ey);
297         final double cTE     = FastMath.cos(thetaE1);
298         final double sTE     = FastMath.sin(thetaE1);
299 
300         // compute shifted in-plane cartesian coordinates
301         final double exey   = ex * ey;
302         final double exCeyS = ex * cTE + ey * sTE;
303         final double x      = a * ((1 - beta * ey * ey) * cTE + beta * exey * sTE - ex);
304         final double y      = a * ((1 - beta * ex * ex) * sTE + beta * exey * cTE - ey);
305         final double factor = FastMath.sqrt(getMu() / a) / (1 - exCeyS);
306         final double xDot   = factor * (-sTE + beta * ey * exCeyS);
307         final double yDot   = factor * ( cTE - beta * ex * exCeyS);
308 
309         final Vector3D shiftedP = new Vector3D(x, u, y, v);
310         final double   r2       = x * x + y * y;
311         final Vector3D shiftedV = new Vector3D(xDot, u, yDot, v);
312         final Vector3D shiftedA = new Vector3D(-getMu() / (r2 * FastMath.sqrt(r2)), shiftedP);
313         return new PVCoordinates(shiftedP, shiftedV, shiftedA);
314 
315     }

317     /** Compute shifted position and velocity in hyperbolic case.
318      * @param dt time shift
319      * @return shifted position and velocity
320      */
321     private PVCoordinates shiftPVHyperbolic(final double dt) {
322 
323         final PVCoordinates pv = getPVCoordinates();
324         final Vector3D pvP   = pv.getPosition();
325         final Vector3D pvV   = pv.getVelocity();
326         final Vector3D pvM   = pv.getMomentum();
327         final double r       = pvP.getNorm();
328         final double rV2OnMu = r * pvV.getNormSq() / getMu();
329         final double a       = getA();
330         final double muA     = getMu() * a;
331         final double e       = FastMath.sqrt(1 - Vector3D.dotProduct(pvM, pvM) / muA);
332         final double sqrt    = FastMath.sqrt((e + 1) / (e - 1));
333 
334         // compute mean anomaly
335         final double eSH     = Vector3D.dotProduct(pvP, pvV) / FastMath.sqrt(-muA);
336         final double eCH     = rV2OnMu - 1;
337         final double H0      = FastMath.log((eCH + eSH) / (eCH - eSH)) / 2;
338         final double M0      = e * FastMath.sinh(H0) - H0;
339 
340         // find canonical 2D frame with p pointing to perigee
341         final double v0      = 2 * FastMath.atan(sqrt * FastMath.tanh(H0 / 2));
342         final Vector3D p     = new Rotation(pvM, v0, RotationConvention.FRAME_TRANSFORM).applyTo(pvP).normalize();
343         final Vector3D q     = Vector3D.crossProduct(pvM, p).normalize();
344 
345         // compute shifted eccentric anomaly
346         final double M1      = M0 + getKeplerianMeanMotion() * dt;
347         final double H1      = meanToHyperbolicEccentric(M1, e);
348 
349         // compute shifted in-plane cartesian coordinates
350         final double cH     = FastMath.cosh(H1);
351         final double sH     = FastMath.sinh(H1);
352         final double sE2m1  = FastMath.sqrt((e - 1) * (e + 1));
353 
354         // coordinates of position and velocity in the orbital plane
355         final double x      = a * (cH - e);
356         final double y      = -a * sE2m1 * sH;
357         final double factor = FastMath.sqrt(getMu() / -a) / (e * cH - 1);
358         final double xDot   = -factor * sH;
359         final double yDot   =  factor * sE2m1 * cH;
360 
361         final Vector3D shiftedP = new Vector3D(x, p, y, q);
362         final double   r2       = x * x + y * y;
363         final Vector3D shiftedV = new Vector3D(xDot, p, yDot, q);
364         final Vector3D shiftedA = new Vector3D(-getMu() / (r2 * FastMath.sqrt(r2)), shiftedP);
365         return new PVCoordinates(shiftedP, shiftedV, shiftedA);
366 
367     }

369     /** Computes the eccentric in-plane argument from the mean in-plane argument.
370      * @param thetaM = mean in-plane argument (rad)
371      * @param ex first component of eccentricity vector
372      * @param ey second component of eccentricity vector
373      * @return the eccentric in-plane argument.
374      */
375     private double meanToEccentric(final double thetaM, final double ex, final double ey) {
376         // Generalization of Kepler equation to in-plane parameters
377         // with thetaE = eta + E and
378         //      thetaM = eta + M = thetaE - ex.sin(thetaE) + ey.cos(thetaE)
379         // and eta being counted from an arbitrary reference in the orbital plane
380         double thetaE        = thetaM;
381         double thetaEMthetaM = 0.0;
382         int    iter          = 0;
383         do {
384             final double cosThetaE = FastMath.cos(thetaE);
385             final double sinThetaE = FastMath.sin(thetaE);
386 
387             final double f2 = ex * sinThetaE - ey * cosThetaE;
388             final double f1 = 1.0 - ex * cosThetaE - ey * sinThetaE;
389             final double f0 = thetaEMthetaM - f2;
390 
391             final double f12 = 2.0 * f1;
392             final double shift = f0 * f12 / (f1 * f12 - f0 * f2);
393 
394             thetaEMthetaM -= shift;
395             thetaE         = thetaM + thetaEMthetaM;
396 
397             if (FastMath.abs(shift) <= 1.0e-12) {
398                 return thetaE;
399             }
400 
401         } while (++iter < 50);
402 
403         throw new MathIllegalStateException(LocalizedCoreFormats.CONVERGENCE_FAILED);
404 
405     }

407     /** Computes the hyperbolic eccentric anomaly from the mean anomaly.
408      * <p>
409      * The algorithm used here for solving hyperbolic Kepler equation is
410      * Danby's iterative method (3rd order) with Vallado's initial guess.
411      * </p>
412      * @param M mean anomaly (rad)
413      * @param ecc eccentricity
414      * @return the hyperbolic eccentric anomaly
415      */
416     private double meanToHyperbolicEccentric(final double M, final double ecc) {
417 
418         // Resolution of hyperbolic Kepler equation for keplerian parameters
419 
420         // Initial guess
421         double H;
422         if (ecc < 1.6) {
423             if ((-FastMath.PI < M && M < 0.) || M > FastMath.PI) {
424                 H = M - ecc;
425             } else {
426                 H = M + ecc;
427             }
428         } else {
429             if (ecc < 3.6 && FastMath.abs(M) > FastMath.PI) {
430                 H = M - FastMath.copySign(ecc, M);
431             } else {
432                 H = M / (ecc - 1.);
433             }
434         }
435 
436         // Iterative computation
437         int iter = 0;
438         do {
439             final double f3  = ecc * FastMath.cosh(H);
440             final double f2  = ecc * FastMath.sinh(H);
441             final double f1  = f3 - 1.;
442             final double f0  = f2 - H - M;
443             final double f12 = 2. * f1;
444             final double d   = f0 / f12;
445             final double fdf = f1 - d * f2;
446             final double ds  = f0 / fdf;
447 
448             final double shift = f0 / (fdf + ds * ds * f3 / 6.);
449 
450             H -= shift;
451 
452             if (FastMath.abs(shift) <= 1.0e-12) {
453                 return H;
454             }
455 
456         } while (++iter < 50);
457 
458         throw new MathIllegalStateException(OrekitMessages.UNABLE_TO_COMPUTE_HYPERBOLIC_ECCENTRIC_ANOMALY,
459                                             iter);
460     }

462     @Override
463     public void getJacobianWrtCartesian(final PositionAngle type, final double[][] jacobian) {
464         // this is the fastest way to set the 6x6 identity matrix
465         for (int i = 0; i < 6; i++) {
466             for (int j = 0; j < 6; j++) {
467                 jacobian[i][j] = 0;
468             }
469             jacobian[i][i] = 1;
470         }
471     }

490     /** {@inheritDoc} */
491     public void addKeplerContribution(final PositionAngle type, final double gm,
492                                       final double[] pDot) {
493 
494         final PVCoordinates pv = getPVCoordinates();
495 
496         // position derivative is velocity
497         final Vector3D velocity = pv.getVelocity();
498         pDot[0] += velocity.getX();
499         pDot[1] += velocity.getY();
500         pDot[2] += velocity.getZ();
501 
502         // velocity derivative is Newtonian acceleration
503         final Vector3D position = pv.getPosition();
504         final double r2         = position.getNormSq();
505         final double coeff      = -gm / (r2 * FastMath.sqrt(r2));
506         pDot[3] += coeff * position.getX();
507         pDot[4] += coeff * position.getY();
508         pDot[5] += coeff * position.getZ();
509 
510     }

531     /** Internal class used only for serialization. */
532     private static class DTO implements Serializable {
533 
534         /** Serializable UID. */
535         private static final long serialVersionUID = 20140723L;
536 
537         /** Double values. */
538         private double[] d;
539 
540         /** Frame in which are defined the orbital parameters. */
541         private final Frame frame;
542 
543         /** Simple constructor.
544          * @param orbit instance to serialize
545          */
546         private DTO(final CartesianOrbit orbit) {
547 
548             final TimeStampedPVCoordinates pv = orbit.getPVCoordinates();
549 
550             // decompose date
551             final double epoch  = FastMath.floor(pv.getDate().durationFrom(AbsoluteDate.J2000_EPOCH));
552             final double offset = pv.getDate().durationFrom(AbsoluteDate.J2000_EPOCH.shiftedBy(epoch));
553 
554             this.d = new double[] {
555                 epoch, offset, orbit.getMu(),
556                 pv.getPosition().getX(),     pv.getPosition().getY(),     pv.getPosition().getZ(),
557                 pv.getVelocity().getX(),     pv.getVelocity().getY(),     pv.getVelocity().getZ(),
558                 pv.getAcceleration().getX(), pv.getAcceleration().getY(), pv.getAcceleration().getZ(),
559             };
560 
561             this.frame = orbit.getFrame();
562 
563         }
564 
565         /** Replace the deserialized data transfer object with a {@link CartesianOrbit}.
566          * @return replacement {@link CartesianOrbit}
567          */
568         private Object readResolve() {
569             return new CartesianOrbit(new TimeStampedPVCoordinates(AbsoluteDate.J2000_EPOCH.shiftedBy(d[0]).shiftedBy(d[1]),
570                                                                    new Vector3D(d[3], d[ 4], d[ 5]),
571                                                                    new Vector3D(d[6], d[ 7], d[ 8]),
572                                                                    new Vector3D(d[9], d[10], d[11])),
573                                       frame, d[2]);
574         }
575 
576     }

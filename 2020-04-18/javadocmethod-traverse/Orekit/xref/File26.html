36  /** This class handles the state for one {@link EventDetector
37   * event detector} during integration steps.
38   *
39   * <p>This class is heavily based on the class with the same name from the
40   * Hipparchus library. The changes performed consist in replacing
41   * raw types (double and double arrays) with space dynamics types
42   * ({@link AbsoluteDate}, {@link SpacecraftState}).</p>
43   * <p>Each time the propagator proposes a step, the event detector
44   * should be checked. This class handles the state of one detector
45   * during one propagation step, with references to the state at the
46   * end of the preceding step. This information is used to determine if
47   * the detector should trigger an event or not during the proposed
48   * step (and hence the step should be reduced to ensure the event
49   * occurs at a bound rather than inside the step).</p>
50   * @author Luc Maisonobe
51   * @param <T> class type for the generic version
52   */
53  public class EventState<T extends EventDetector> implements Serializable {

134     /** Initialize event handler at the start of a propagation.
135      * <p>
136      * This method is called once at the start of the propagation. It
137      * may be used by the event handler to initialize some internal data
138      * if needed.

140      * @param s0 initial state
141      * @param t target time for the integration

144         detector.init(s0, t);
145         lastT = AbsoluteDate.PAST_INFINITY;
146         lastG = Double.NaN;

149     /** Compute the value of the switching function.
150      * This function must be continuous (at least in its roots neighborhood),
151      * as the integrator will need to find its roots to locate the events.
152      * @param s the current state information: date, kinematics, attitude
153      * @return value of the switching function
154      * @exception OrekitException if some specific error occurs
155      */
156     private double g(final SpacecraftState s) throws OrekitException {
157         if (!s.getDate().equals(lastT)) {
158             lastT = s.getDate();
159             lastG = detector.g(s);
160         }
161         return lastG;
162     }

164     /** Reinitialize the beginning of the step.
165      * @param interpolator interpolator valid for the current step
166      * @exception OrekitException if the event detector
167      * value cannot be evaluated at the beginning of the step
168      */
169     public void reinitializeBegin(final OrekitStepInterpolator interpolator)
170         throws OrekitException {
171         forward = interpolator.isForward();
172         final SpacecraftState s0 = interpolator.getPreviousState();
173         this.t0 = s0.getDate();
174         g0 = g(s0);
175         while (g0 == 0) {
176             // extremely rare case: there is a zero EXACTLY at interval start
177             // we will use the sign slightly after step beginning to force ignoring this zero
178             // try moving forward by half a convergence interval
179             final double dt = (forward ? 0.5 : -0.5) * detector.getThreshold();
180             AbsoluteDate startDate = t0.shiftedBy(dt);
181             // if convergence is too small move an ulp
182             if (t0.equals(startDate)) {
183                 startDate = nextAfter(startDate);
184             }
185             t0 = startDate;
186             g0 = g(interpolator.getInterpolatedState(t0));
187         }
188         g0Positive = g0 > 0;
189         // "last" event was increasing
190         increasing = g0Positive;
191     }

193     /** Evaluate the impact of the proposed step on the event detector.
194      * @param interpolator step interpolator for the proposed step
195      * @return true if the event detector triggers an event before
196      * the end of the proposed step (this implies the step should be
197      * rejected)
198      * @exception OrekitException if the switching function
199      * cannot be evaluated
200      * @exception MathRuntimeException if an event cannot be located
201      */
202     public boolean evaluateStep(final OrekitStepInterpolator interpolator)
203         throws OrekitException, MathRuntimeException {
204 
205         forward = interpolator.isForward();
206         final SpacecraftState s1 = interpolator.getCurrentState();
207         final AbsoluteDate t1 = s1.getDate();
208         final double dt = t1.durationFrom(t0);
209         if (FastMath.abs(dt) < detector.getThreshold()) {
210             // we cannot do anything on such a small step, don't trigger any events
211             return false;
212         }
213         // number of points to check in the current step
214         final int n = FastMath.max(1, (int) FastMath.ceil(FastMath.abs(dt) / detector.getMaxCheckInterval()));
215         final double h = dt / n;
216 
217 
218         AbsoluteDate ta = t0;
219         double ga = g0;
220         for (int i = 0; i < n; ++i) {
221 
222             // evaluate handler value at the end of the substep
223             final AbsoluteDate tb = (i == n - 1) ? t1 : t0.shiftedBy((i + 1) * h);
224             final double gb = g(interpolator.getInterpolatedState(tb));
225 
226             // check events occurrence
227             if (gb == 0.0 || (g0Positive ^ (gb > 0))) {
228                 // there is a sign change: an event is expected during this step
229                 if (findRoot(interpolator, ta, ga, tb, gb)) {
230                     return true;
231                 }
232             } else {
233                 // no sign change: there is no event for now
234                 ta = tb;
235                 ga = gb;
236             }
237 
238         }
239 
240         // no event during the whole step
241         pendingEvent     = false;
242         pendingEventTime = null;
243         return false;
244 
245     }

247     /**
248      * Find a root in a bracketing interval.
249      *
250      * <p> When calling this method one of the following must be true. Either ga == 0, gb
251      * == 0, (ga < 0  and gb > 0), or (ga > 0 and gb < 0).
252      *
253      * @param interpolator that covers the interval.
254      * @param ta           earliest possible time for root.
255      * @param ga           g(ta).
256      * @param tb           latest possible time for root.
257      * @param gb           g(tb).
258      * @return if a zero crossing was found.
259      * @throws OrekitException if the event detector throws one
260      */
261     private boolean findRoot(final OrekitStepInterpolator interpolator,
262                              final AbsoluteDate ta, final double ga,
263                              final AbsoluteDate tb, final double gb)
264             throws OrekitException {
265         // check there appears to be a root in [ta, tb]
266         check(ga == 0.0 || gb == 0.0 || (ga > 0.0 && gb < 0.0) || (ga < 0.0 && gb > 0.0));
267 
268         final double convergence = detector.getThreshold();
269         final int maxIterationCount = detector.getMaxIterationCount();
270         final BracketedUnivariateSolver<UnivariateFunction> solver =
271                 new BracketingNthOrderBrentSolver(0, convergence, 0, 5);
272 
273         // event time, just at or before the actual root.
274         AbsoluteDate beforeRootT = null;
275         double beforeRootG = Double.NaN;
276         // time on the other side of the root.
277         // Initialized the the loop below executes once.
278         AbsoluteDate afterRootT = ta;
279         double afterRootG = 0.0;
280 
281         // check for some conditions that the root finders don't like
282         // these conditions cannot not happen in the loop below
283         // the ga == 0.0 case is handled by the loop below
284         if (ta.equals(tb)) {
285             // both non-zero but times are the same. Probably due to reset state
286             beforeRootT = ta;
287             beforeRootG = ga;
288             afterRootT = shiftedBy(beforeRootT, convergence);
289             afterRootG = g(interpolator.getInterpolatedState(afterRootT));
290         } else if (ga != 0.0 && gb == 0.0) {
291             // hard: ga != 0.0 and gb == 0.0
292             // look past gb by up to convergence to find next sign
293             // throw an exception if g(t) = 0.0 in [tb, tb + convergence]
294             beforeRootT = tb;
295             beforeRootG = gb;
296             afterRootT = shiftedBy(beforeRootT, convergence);
297             afterRootG = g(interpolator.getInterpolatedState(afterRootT));
298         } else if (ga != 0.0) {
299             final double newGa = g(interpolator.getInterpolatedState(ta));
300             if (ga > 0 != newGa > 0) {
301                 // both non-zero, step sign change at ta, possibly due to reset state
302                 beforeRootT = ta;
303                 beforeRootG = newGa;
304                 afterRootT = minTime(shiftedBy(beforeRootT, convergence), tb);
305                 afterRootG = g(interpolator.getInterpolatedState(afterRootT));
306             }
307         }
308 
309         // loop to skip through "fake" roots, i.e. where g(t) = g'(t) = 0.0
310         // executed once if we didn't hit a special case above
311         AbsoluteDate loopT = ta;
312         double loopG = ga;
313         while ((afterRootG == 0.0 || afterRootG > 0.0 == g0Positive) &&
314                 strictlyAfter(afterRootT, tb)) {
315             if (loopG == 0.0) {
316                 // ga == 0.0 and gb may or may not be 0.0
317                 // handle the root at ta first
318                 beforeRootT = loopT;
319                 beforeRootG = loopG;
320                 afterRootT = minTime(shiftedBy(beforeRootT, convergence), tb);
321                 afterRootG = g(interpolator.getInterpolatedState(afterRootT));
322             } else {
323                 // both non-zero, the usual case, use a root finder.
324                 try {
325                     // time zero for evaluating the function f. Needs to be final
326                     final AbsoluteDate fT0 = loopT;
327                     final UnivariateFunction f = dt -> {
328                         try {
329                             return g(interpolator.getInterpolatedState(fT0.shiftedBy(dt)));
330                         } catch (OrekitException oe) {
331                             throw new OrekitExceptionWrapper(oe);
332                         }
333                     };
334                     // tb as a double for use in f
335                     final double tbDouble = tb.durationFrom(fT0);
336                     if (forward) {
337                         final Interval interval =
338                                 solver.solveInterval(maxIterationCount, f, 0, tbDouble);
339                         beforeRootT = fT0.shiftedBy(interval.getLeftAbscissa());
340                         beforeRootG = interval.getLeftValue();
341                         afterRootT = fT0.shiftedBy(interval.getRightAbscissa());
342                         afterRootG = interval.getRightValue();
343                     } else {
344                         final Interval interval =
345                                 solver.solveInterval(maxIterationCount, f, tbDouble, 0);
346                         beforeRootT = fT0.shiftedBy(interval.getRightAbscissa());
347                         beforeRootG = interval.getRightValue();
348                         afterRootT = fT0.shiftedBy(interval.getLeftAbscissa());
349                         afterRootG = interval.getLeftValue();
350                     }
351                 } catch (OrekitExceptionWrapper oew) {
352                     throw oew.getException();
353                 }
354             }
355             // tolerance is set to less than 1 ulp
356             // assume tolerance is 1 ulp
357             if (beforeRootT.equals(afterRootT)) {
358                 afterRootT = nextAfter(afterRootT);
359                 afterRootG = g(interpolator.getInterpolatedState(afterRootT));
360             }
361             // check loop is making some progress
362             check((forward && afterRootT.compareTo(beforeRootT) > 0) ||
363                   (!forward && afterRootT.compareTo(beforeRootT) < 0));
364             // setup next iteration
365             loopT = afterRootT;
366             loopG = afterRootG;
367         }
368 
369         // figure out the result of root finding, and return accordingly
370         if (afterRootG == 0.0 || afterRootG > 0.0 == g0Positive) {
371             // loop gave up and didn't find any crossing within this step
372             return false;
373         } else {
374             // real crossing
375             check(beforeRootT != null && !Double.isNaN(beforeRootG));
376             // variation direction, with respect to the integration direction
377             increasing = !g0Positive;
378             pendingEventTime = beforeRootT;
379             stopTime = beforeRootG == 0.0 ? beforeRootT : afterRootT;
380             pendingEvent = true;
381             afterEvent = afterRootT;
382             afterG = afterRootG;
383 
384             // check increasing set correctly
385             check(afterG > 0 == increasing);
386             check(increasing == gb >= ga);
387 
388             return true;
389         }
390 
391     }

393     /**
394      * Get the next number after the given number in the current propagation direction.
395      *
396      * @param t input time
397      * @return t +/- 1 ulp depending on the direction.
398      */
399     private AbsoluteDate nextAfter(final AbsoluteDate t) {
400         return t.shiftedBy(forward ? +Precision.EPSILON : -Precision.EPSILON);
401     }

412     /**
413      * Try to accept the current history up to the given time.
414      *
415      * <p> It is not necessary to call this method before calling {@link
416      * #doEvent(SpacecraftState)} with the same state. It is necessary to call this
417      * method before you call {@link #doEvent(SpacecraftState)} on some other event
418      * detector.
419      *
420      * @param state        to try to accept.
421      * @param interpolator to use to find the new root, if any.
422      * @return if the event detector has an event it has not detected before that is on or
423      * before the same time as {@code state}. In other words {@code false} means continue
424      * on while {@code true} means stop and handle my event first.
425      * @exception OrekitException if the g function throws one
426      */
427     public boolean tryAdvance(final SpacecraftState state,
428                               final OrekitStepInterpolator interpolator)
429         throws OrekitException {
430         // check this is only called before a pending event.
431         check(!(pendingEvent && strictlyAfter(pendingEventTime, state.getDate())));
432 
433         final AbsoluteDate t = state.getDate();
434 
435         // just found an event and we know the next time we want to search again
436         if (strictlyAfter(t, earliestTimeConsidered)) {
437             return false;
438         }
439 
440         final double g = g(state);
441         final boolean positive = g > 0;
442 
443         // check for new root, pendingEventTime may be null if there is not pending event
444         if ((g == 0.0 && t.equals(pendingEventTime)) || positive == g0Positive) {
445             // at a root we already found, or g function has expected sign
446             t0 = t;
447             g0 = g; // g0Positive is the same
448             return false;
449         } else {
450             // found a root we didn't expect -> find precise location
451             return findRoot(interpolator, t0, g0, t, g);
452         }
453     }

455     /**
456      * Notify the user's listener of the event. The event occurs wholly within this method
457      * call including a call to {@link EventDetector#resetState(SpacecraftState)}
458      * if necessary.
459      *
460      * @param state the state at the time of the event. This must be at the same time as
461      *              the current value of {@link #getEventDate()}.
462      * @return the user's requested action and the new state if the action is {@link
463      * org.orekit.propagation.events.handlers.EventHandler.Action#RESET_STATE Action.RESET_STATE}.
464      * Otherwise the new state is {@code state}. The stop time indicates what time propagation
465      * should stop if the action is {@link
466      * org.orekit.propagation.events.handlers.EventHandler.Action#STOP Action.STOP}.
467      * This guarantees the integration will stop on or after the root, so that integration
468      * may be restarted safely.
469      * @exception OrekitException if the event detector throws one
470      */
471     public EventOccurrence doEvent(final SpacecraftState state)
472         throws OrekitException {
473         // check event is pending and is at the same time
474         check(pendingEvent);
475         check(state.getDate().equals(this.pendingEventTime));
476 
477         final EventHandler.Action action = detector.eventOccurred(state, increasing == forward);
478         final SpacecraftState newState;
479         if (action == EventHandler.Action.RESET_STATE) {
480             newState = detector.resetState(state);
481         } else {
482             newState = state;
483         }
484         // clear pending event
485         pendingEvent     = false;
486         pendingEventTime = null;
487         // setup for next search
488         earliestTimeConsidered = afterEvent;
489         t0 = afterEvent;
490         g0 = afterG;
491         g0Positive = increasing;
492         // check g0Positive set correctly
493         check(g0 == 0.0 || g0Positive == (g0 > 0));
494         return new EventOccurrence(action, newState, stopTime);
495     }

497     /**
498      * Shift a time value along the current integration direction: {@link #forward}.
499      *
500      * @param t     the time to shift.
501      * @param delta the amount to shift.
502      * @return t + delta if forward, else t - delta. If the result has to be rounded it
503      * will be rounded to be before the true value of t + delta.
504      */
505     private AbsoluteDate shiftedBy(final AbsoluteDate t, final double delta) {
506         if (forward) {
507             final AbsoluteDate ret = t.shiftedBy(delta);
508             if (ret.durationFrom(t) > delta) {
509                 return ret.shiftedBy(-Precision.EPSILON);
510             } else {
511                 return ret;
512             }
513         } else {
514             final AbsoluteDate ret = t.shiftedBy(-delta);
515             if (t.durationFrom(ret) > delta) {
516                 return ret.shiftedBy(+Precision.EPSILON);
517             } else {
518                 return ret;
519             }
520         }
521     }

523     /**
524      * Get the time that happens first along the current propagation direction: {@link
525      * #forward}.
526      *
527      * @param a first time
528      * @param b second time
529      * @return min(a, b) if forward, else max (a, b)
530      */
531     private AbsoluteDate minTime(final AbsoluteDate a, final AbsoluteDate b) {
532         return (forward ^ (a.compareTo(b) > 0)) ? a : b;
533     }
534 
535     /**
536      * Check the ordering of two times.
537      *
538      * @param t1 the first time.
539      * @param t2 the second time.
540      * @return true if {@code t2} is strictly after {@code t1} in the propagation
541      * direction.
542      */
543     private boolean strictlyAfter(final AbsoluteDate t1, final AbsoluteDate t2) {
544         if (t1 == null || t2 == null) {
545             return false;
546         } else {
547             return forward ? t1.compareTo(t2) < 0 : t2.compareTo(t1) < 0;
548         }
549     }

551     /**
552      * Same as keyword assert, but throw a {@link MathRuntimeException}.
553      *
554      * @param condition to check
555      * @throws MathRuntimeException if {@code condition} is false.
556      */
557     private void check(final boolean condition) throws MathRuntimeException {
558         if (!condition) {
559             throw new OrekitInternalError(null);
560         }
561     }

563     /**
564      * Class to hold the data related to an event occurrence that is needed to decide how
565      * to modify integration.
566      */
567     public static class EventOccurrence {
568 
569         /** User requested action. */
570         private final EventHandler.Action action;
571         /** New state for a reset action. */
572         private final SpacecraftState newState;
573         /** The time to stop propagation if the action is a stop event. */
574         private final AbsoluteDate stopDate;
575 
576         /**
577          * Create a new occurrence of an event.
578          *
579          * @param action   the user requested action.
580          * @param newState for a reset event. Should be the current state unless the
581          *                 action is {@link Action#RESET_STATE}.
582          * @param stopDate to stop propagation if the action is {@link Action#STOP}. Used
583          *                 to move the stop time to just after the root.
584          */
585         EventOccurrence(final EventHandler.Action action,
586                         final SpacecraftState newState,
587                         final AbsoluteDate stopDate) {
588             this.action = action;
589             this.newState = newState;
590             this.stopDate = stopDate;
591         }
592 
593         /**
594          * Get the user requested action.
595          *
596          * @return the action.
597          */
598         public EventHandler.Action getAction() {
599             return action;
600         }
601 
602         /**
603          * Get the new state for a reset action.
604          *
605          * @return the new state.
606          */
607         public SpacecraftState getNewState() {
608             return newState;
609         }
610 
611         /**
612          * Get the new time for a stop action.
613          *
614          * @return when to stop propagation.
615          */
616         public AbsoluteDate getStopDate() {
617             return stopDate;
618         }
619 
620     }

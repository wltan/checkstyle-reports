252    @Override
253    protected final Transaction suspendIfNeeded() {
254       if (transactionManager == null) {
255          return null;
256       }
257 
258       try {
259          switch (transactionManager.getStatus()) {
260             case Status.STATUS_ACTIVE:
261             case Status.STATUS_NO_TRANSACTION:
262                return null;
263             case Status.STATUS_MARKED_ROLLBACK:
264             case Status.STATUS_PREPARED:
265             case Status.STATUS_COMMITTED:
266             case Status.STATUS_ROLLEDBACK:
267             case Status.STATUS_UNKNOWN:
268             case Status.STATUS_PREPARING:
269             case Status.STATUS_COMMITTING:
270             case Status.STATUS_ROLLING_BACK:
271             default:
272                //suspend in default and in unknown status to be safer
273                return transactionManager.suspend();
274          }
275       } catch (Exception e) {
276          if (log.isTraceEnabled()) {
277             log.trace("An error occurred while trying to suspend a transaction.", e);
278          }
279          return null;
280       }
281    }
282 
283    @Override
284    protected final void resumeIfNeeded(Transaction transaction) {
285       if (transaction == null || transactionManager == null) {
286          return;
287       }
288       try {
289          transactionManager.resume(transaction);
290       } catch (Exception e) {
291          if (log.isTraceEnabled()) {
292             log.tracef(e, "An error occurred while trying to resume a suspended transaction. tx=%s", transaction);
293          }
294       }
295    }

298    public void notifyCacheEntryCreated(K key, V value, boolean pre,
299          InvocationContext ctx, FlagAffectedCommand command) {
300       if (!cacheEntryCreatedListeners.isEmpty()) {
301          EventImpl<K, V> e = EventImpl.createEvent(cache, CACHE_ENTRY_CREATED);
302          configureEvent(e, key, value, pre, ctx, command, null, null);
303          boolean isLocalNodePrimaryOwner = clusteringDependentLogic.localNodeIsPrimaryOwner(key);
304          boolean sendEvents = !ctx.isInTxScope();
305          try {
306             for (CacheEntryListenerInvocation<K, V> listener : cacheEntryCreatedListeners) {
307                listener.invoke(e, isLocalNodePrimaryOwner);
308             }
309             if (sendEvents) {
310                eventManager.sendEvents();
311                sendEvents = false;
312             }
313          } finally {
314             if (sendEvents) {
315                eventManager.dropEvents();
316             }
317          }
318       }
319    }

370    private void configureEvent(EventImpl<K, V> e, K key, V value, boolean pre, InvocationContext ctx,
371                                FlagAffectedCommand command, V previousValue, Metadata previousMetadata) {
372       if (typeConverter != null) {
373          key = (K) typeConverter.unboxKey(key);
374          value = (V) typeConverter.unboxValue(value);
375          previousValue = (V) typeConverter.unboxValue(previousValue);
376       }
377 
378       e.setOriginLocal(ctx.isOriginLocal());
379       e.setValue(pre ? previousValue : value);
380       e.setPre(pre);
381       e.setOldValue(previousValue);
382       e.setOldMetadata(previousMetadata);
383       CacheEntry entry = ctx.lookupEntry(key);
384       if (entry != null) {
385          e.setMetadata(entry.getMetadata());
386       } else if (command instanceof ClearCommand) {
387          e.setMetadata(previousMetadata);
388       }
389       Set<Flag> flags;
390       if (command != null && (flags = command.getFlags()) != null && flags.contains(Flag.COMMAND_RETRY)) {
391          e.setCommandRetried(true);
392       }
393       e.setKey(key);
394       setTx(ctx, e);
395    }

410    @Override
411    public void notifyCacheEntriesEvicted(Collection<InternalCacheEntry<? extends K, ? extends V>> entries, InvocationContext ctx, FlagAffectedCommand command) {
412       if (!entries.isEmpty()) {
413          if (isNotificationAllowed(command, cacheEntriesEvictedListeners)) {
414             EventImpl<K, V> e = EventImpl.createEvent(cache, CACHE_ENTRY_EVICTED);
415             Map<K, V> evictedKeysAndValues = transformCollectionToMap(entries,
416                new InfinispanCollections.MapMakerFunction<K, V, InternalCacheEntry<? extends K, ? extends V>>() {
417                   @Override
418                   public Map.Entry<K, V> transform(final InternalCacheEntry<? extends K, ? extends V> input) {
419                      return new Map.Entry<K, V>() {
420                         @Override
421                         public K getKey() {
422                           return input.getKey();
423                         }
424 
425                         @Override
426                         public V getValue() {
427                           return input.getValue();
428                         }
429 
430                         @Override
431                         public V setValue(V value) {
432                           throw new UnsupportedOperationException();
433                         }
434                      };
435                   }
436                }
437             );
438 
439             e.setEntries(evictedKeysAndValues);
440             for (CacheEntryListenerInvocation<K, V> listener : cacheEntriesEvictedListeners) listener.invoke(e);
441          }
442 
443          // For backward compat
444          if (isNotificationAllowed(command, cacheEntryEvictedListeners)) {
445             for (InternalCacheEntry<? extends K, ? extends V> ice : entries) {
446                EventImpl<K, V> e = EventImpl.createEvent(cache, CACHE_ENTRY_EVICTED);
447                e.setKey(ice.getKey());
448                e.setValue(ice.getValue());
449                boolean isLocalNodePrimaryOwner = clusteringDependentLogic.localNodeIsPrimaryOwner(ice.getKey());
450                for (CacheEntryListenerInvocation<K, V> listener : cacheEntryEvictedListeners) listener.invoke(e, isLocalNodePrimaryOwner);
451             }
452          }
453       }
454    }

538    @Override
539    public void notifyTransactionCompleted(GlobalTransaction transaction, boolean successful, InvocationContext ctx) {
540       if (!transactionCompletedListeners.isEmpty()) {
541          boolean isOriginLocal = ctx.isOriginLocal();
542          EventImpl<K, V> e = EventImpl.createEvent(cache, TRANSACTION_COMPLETED);
543          e.setOriginLocal(isOriginLocal);
544          e.setTransactionId(transaction);
545          e.setTransactionSuccessful(successful);
546          for (CacheEntryListenerInvocation<K, V> listener : transactionCompletedListeners) listener.invoke(e);
547          if (ctx.isInTxScope()) {
548             if (successful) {
549                eventManager.sendEvents();
550             } else {
551                eventManager.dropEvents();
552             }
553          }
554       }
555    }

604    @Override
605    public void notifyClusterListeners(Collection<? extends CacheEntryEvent<K, V>> events, UUID uuid) {
606       for (CacheEntryEvent<K, V> event : events) {
607          if (event.isPre()) {
608             throw new IllegalArgumentException("Events for cluster listener should never be pre change");
609          }
610          switch (event.getType()) {
611             case CACHE_ENTRY_MODIFIED:
612                for (CacheEntryListenerInvocation<K, V> listener : cacheEntryModifiedListeners) {
613                   if (listener.isClustered() && uuid.equals(listener.getIdentifier())) {
614                      // We force invocation, since it means the owning node passed filters already and they
615                      // already converted so don't run converter either
616                      listener.invokeNoChecks(event, false, true);
617                   }
618                }
619                break;
620             case CACHE_ENTRY_CREATED:
621                for (CacheEntryListenerInvocation<K, V> listener : cacheEntryCreatedListeners) {
622                   if (listener.isClustered() && uuid.equals(listener.getIdentifier())) {
623                      // We force invocation, since it means the owning node passed filters already and they
624                      // already converted so don't run converter either
625                      listener.invokeNoChecks(event, false, true);
626                   }
627                }
628                break;
629             case CACHE_ENTRY_REMOVED:
630                for (CacheEntryListenerInvocation<K, V> listener : cacheEntryRemovedListeners) {
631                   if (listener.isClustered() && uuid.equals(listener.getIdentifier())) {
632                      // We force invocation, since it means the owning node passed filters already and they
633                      // already converted so don't run converter either
634                      listener.invokeNoChecks(event, false, true);
635                   }
636                }
637                break;
638             default:
639                throw new IllegalArgumentException("Unexpected event type encountered!");
640          }
641       }
642    }

705    /**
706     * Adds the listener using the provided filter converter and class loader.  The provided builder is used to add
707     * additional configuration including (clustered, onlyPrimary & identifier) which can be used after this method
708     * is completed to see what values were used in the addition of this listener
709     * @param listener
710     * @param filter
711     * @param converter
712     * @param classLoader
713     * @param <C>
714     * @return
715     */
716    public <C> void addListener(Object listener, CacheEventFilter<? super K, ? super V> filter,
717                                            CacheEventConverter<? super K, ? super V, C> converter, ClassLoader classLoader) {
718       final Listener l = testListenerClassValidity(listener.getClass());
719       final UUID generatedId = UUID.randomUUID();
720       final CacheMode cacheMode = config.clustering().cacheMode();
721 
722       FilterIndexingServiceProvider indexingProvider = null;
723       boolean foundMethods = false;
724       if (filter instanceof IndexedFilter) {
725          IndexedFilter indexedFilter = (IndexedFilter) filter;
726          indexingProvider = findIndexingServiceProvider(indexedFilter);
727          if (indexingProvider != null) {
728             DelegatingCacheInvocationBuilder builder = new DelegatingCacheInvocationBuilder(indexingProvider);
729             builder
730                   .setIncludeCurrentState(l.includeCurrentState())
731                   .setClustered(l.clustered())
732                   .setOnlyPrimary(l.clustered() ? cacheMode.isDistributed() : l.primaryOnly())
733                   .setObservation(l.clustered() ? Listener.Observation.POST : l.observation())
734                   .setFilter(filter)
735                   .setConverter(converter)
736                   .setIdentifier(generatedId)
737                   .setClassLoader(classLoader);
738             foundMethods = validateAndAddListenerInvocations(listener, builder);
739             builder.registerListenerInvocations();
740          }
741       }
742       if (indexingProvider == null) {
743          CacheInvocationBuilder builder = new CacheInvocationBuilder();
744          builder
745                .setIncludeCurrentState(l.includeCurrentState())
746                .setClustered(l.clustered())
747                .setOnlyPrimary(l.clustered() ? cacheMode.isDistributed() : l.primaryOnly())
748                .setObservation(l.clustered() ? Listener.Observation.POST : l.observation())
749                .setFilter(filter)
750                .setConverter(converter)
751                .setIdentifier(generatedId)
752                .setClassLoader(classLoader);
753          foundMethods = validateAndAddListenerInvocations(listener, builder);
754       }
755 
756       if (foundMethods && l.clustered()) {
757          if (l.observation() == Listener.Observation.PRE) {
758             throw log.clusterListenerRegisteredWithOnlyPreEvents(listener.getClass());
759          } else if (cacheMode.isInvalidation()) {
760             throw new UnsupportedOperationException("Cluster listeners cannot be used with Invalidation Caches!");
761          } else if (cacheMode.isDistributed()) {
762             clusterListenerIDs.put(listener, generatedId);
763             EmbeddedCacheManager manager = cache.getCacheManager();
764             Address ourAddress = manager.getAddress();
765 
766             List<Address> members = manager.getMembers();
767             // If we are the only member don't even worry about sending listeners
768             if (members != null && members.size() > 1) {
769                DistributedExecutionCompletionService decs = new DistributedExecutionCompletionService(distExecutorService);
770 
771                if (log.isTraceEnabled()) {
772                   log.tracef("Replicating cluster listener to other nodes %s for cluster listener with id %s",
773                              members, generatedId);
774                }
775                Callable callable = new ClusterListenerReplicateCallable(generatedId, ourAddress, filter, converter, l.sync());
776                for (Address member : members) {
777                   if (!member.equals(ourAddress)) {
778                      decs.submit(member, callable);
779                   }
780                }
781 
782                for (int i = 0; i < members.size() - 1; ++i) {
783                   try {
784                      decs.take().get();
785                   } catch (InterruptedException e) {
786                      throw new CacheListenerException(e);
787                   } catch (ExecutionException e) {
788                      throw new CacheListenerException(e);
789                   }
790                }
791 
792                int extraCount = 0;
793                // If anyone else joined since we sent these we have to send the listeners again, since they may have queried
794                // before the other nodes got the new listener
795                List<Address> membersAfter = manager.getMembers();
796                for (Address member : membersAfter) {
797                   if (!members.contains(member) && !member.equals(ourAddress)) {
798                      if (log.isTraceEnabled()) {
799                         log.tracef("Found additional node %s that joined during replication of cluster listener with id %s",
800                                    member, generatedId);
801                      }
802                      extraCount++;
803                      decs.submit(member, callable);
804                   }
805                }
806 
807                for (int i = 0; i < extraCount; ++i) {
808                   try {
809                      decs.take().get();
810                   } catch (InterruptedException e) {
811                      throw new CacheListenerException(e);
812                   } catch (ExecutionException e) {
813                      throw new CacheListenerException(e);
814                   }
815                }
816             }
817          }
818       }
819 
820       // If we have a segment listener handler, it means we have to do initial state
821       QueueingSegmentListener handler = segmentHandler.remove(generatedId);
822       if (handler != null) {
823          if (log.isTraceEnabled()) {
824             log.tracef("Listener %s requests initial state for cache", generatedId);
825          }
826          final KeyValueFilter usedFilter;
827          final Converter usedConverter;
828          if (filter instanceof CacheEventFilterConverter && (filter == converter || converter == null)) {
829             usedFilter = new CacheEventFilterConverterAsKeyValueFilterConverter((CacheEventFilterConverter) filter);
830             usedConverter = null;
831          } else {
832             usedFilter = filter == null ? null : new CacheEventFilterAsKeyValueFilter(filter);
833             usedConverter = converter == null ? null : new CacheEventConverterAsConverter(converter);
834          }
835          try (CloseableIterator<CacheEntry<K, C>> iterator = entryRetriever.retrieveEntries(usedFilter, usedConverter, null, handler)) {
836             while (iterator.hasNext()) {
837                CacheEntry<K, C> entry = iterator.next();
838                // Mark the key as processed and see if we had a concurrent update
839                Object value = handler.markKeyAsProcessing(entry.getKey());
840                if (value == BaseQueueingSegmentListener.REMOVED) {
841                   // Don't process this value if we had a concurrent remove
842                   continue;
843                }
844                raiseEventForInitialTransfer(generatedId, entry, l.clustered());
845 
846                handler.notifiedKey(entry.getKey());
847             }
848          }
849 
850          Set<CacheEntry> entries = handler.findCreatedEntries();
851 
852          for (CacheEntry entry : entries) {
853             raiseEventForInitialTransfer(generatedId, entry, l.clustered());
854          }
855 
856          if (log.isTraceEnabled()) {
857             log.tracef("Listener %s initial state for cache completed", generatedId);
858          }
859 
860          handler.transferComplete();
861       }
862    }

879    private void raiseEventForInitialTransfer(UUID identifier, CacheEntry entry, boolean clustered) {
880       EventImpl preEvent;
881       if (clustered) {
882          // In clustered mode we only send post event
883          preEvent = null;
884       } else {
885          preEvent = EventImpl.createEvent(cache, CACHE_ENTRY_CREATED);
886          preEvent.setKey(entry.getKey());
887          preEvent.setPre(true);
888       }
889 
890       EventImpl postEvent = EventImpl.createEvent(cache, CACHE_ENTRY_CREATED);
891       postEvent.setKey(entry.getKey());
892       postEvent.setValue(entry.getValue());
893       postEvent.setMetadata(entry.getMetadata());
894       postEvent.setPre(false);
895 
896       for (CacheEntryListenerInvocation<K, V> invocation : cacheEntryCreatedListeners) {
897          // Now notify all our methods of the creates
898          if (invocation.getIdentifier() == identifier) {
899             if (preEvent != null) {
900                // Non clustered notifications are done twice
901                invocation.invokeNoChecks(preEvent, true, true);
902             }
903             invocation.invokeNoChecks(postEvent, true, true);
904          }
905       }
906    }

990       @Override
991       public CacheEntryListenerInvocation<K, V> build() {
992          ListenerInvocation<Event<K, V>> invocation = new ListenerInvocationImpl(target, method, sync, classLoader,
993                                                                                  subject);
994 
995          wireFilterAndConverterDependencies(filter, converter);
996 
997          // If we are dealing with clustered events that forces the cluster listener to only use primary only else we would
998          // have duplicate events
999          CacheEntryListenerInvocation<K, V> returnValue;
1000 
1001          if (includeCurrentState) {
1002             // If it is a clustered listener and distributed cache we can do some extra optimizations
1003             if (clustered) {
1004                QueueingSegmentListener handler = segmentHandler.get(identifier);
1005                if (handler == null) {
1006                   if (config.clustering().cacheMode().isDistributed()) {

1009                      handler = new QueueingAllSegmentListener(entryFactory);
1010                   }
1011                   QueueingSegmentListener currentQueue = segmentHandler.putIfAbsent(identifier, handler);
1012                   if (currentQueue != null) {
1013                      handler = currentQueue;
1014                   }
1015                }
1016                returnValue = new ClusteredListenerInvocation<K, V>(invocation, handler, filter, converter, annotation,
1017                                                                    onlyPrimary, identifier, sync, observation);
1018             } else {
1019 //               TODO: this is removed until non cluster listeners are supported
1020 //               QueueingSegmentListener handler = segmentHandler.get(identifier);
1021 //               if (handler == null) {
1022 //                  handler = new QueueingAllSegmentListener();
1023 //                  QueueingSegmentListener currentQueue = segmentHandler.putIfAbsent(identifier, handler);

1025 //                     handler = currentQueue;
1026 //                  }

1030                returnValue = new BaseCacheEntryListenerInvocation(invocation, filter, converter, annotation,

1183          if (resultingEvent != null) {
1184             invokeNoChecks(resultingEvent, false, filterAndConvert);
1185          }
1186       }

1188       /**
1189        * This is the entry point for remote listener events being fired
1190        * @param event
1191        * @param skipQueue
1192        */
1193       @Override
1194       public void invokeNoChecks(CacheEntryEvent<K, V> event, boolean skipQueue, boolean skipConverter) {
1195          // We run the converter first, this way the converter doesn't have to run serialized when enqueued and also
1196          // the handler doesn't have to worry about it
1197          CacheEntryEvent<K, V> eventToUse;
1198          if (!skipConverter) {
1199             eventToUse = convertValue(converter, event);
1200          } else {
1201             eventToUse = event;
1202          }
1203 
1204          if (skipQueue) {
1205             invocation.invoke(eventToUse);
1206          } else {
1207             doRealInvocation(eventToUse);
1208          }
1209       }

1219       protected CacheEntryEvent<K, V> shouldInvoke(CacheEntryEvent<K, V> event, boolean isLocalNodePrimaryOwner) {
1220          if (onlyPrimary && !isLocalNodePrimaryOwner) return null;
1221          if (event instanceof EventImpl) {
1222             EventImpl<K, V> eventImpl = (EventImpl<K, V>)event;
1223             if (!shouldInvoke(event)) return null;
1224             EventType eventType;
1225             // Only use the filter if it was provided and we have an event that we can filter properly
1226             if (filter != null && (eventType = getEvent(eventImpl)) != null) {
1227                if (filterAndConvert) {
1228                   Object newValue = ((CacheEventFilterConverter)filter).filterAndConvert(eventImpl.getKey(),
1229                         eventImpl.getOldValue(), eventImpl.getOldMetadata(), eventImpl.getValue(),
1230                         eventImpl.getMetadata(), eventType);
1231                   if (newValue != null) {
1232                      EventImpl<K, V> clone = eventImpl.clone();
1233                      clone.setValue((V) newValue);
1234                      return clone;
1235                   } else {
1236                      return null;
1237                   }
1238                } else if (!filter.accept(eventImpl.getKey(), eventImpl.getOldValue(), eventImpl.getOldMetadata(),
1239                                  eventImpl.getValue(), eventImpl.getMetadata(), eventType)) {
1240                   return null;
1241                }
1242             }
1243          }
1244          return event;
1245       }

1247       // We can't currently filter events that don't implement CacheEntryEvent or CACHE_ENTRY_EVICTED events.  Basically
1248       // events that have a single key value pair only
1249       private EventType getEvent(EventImpl<K, V> event) {
1250          switch (event.getType()) {
1251             case CACHE_ENTRY_ACTIVATED:
1252             case CACHE_ENTRY_CREATED:
1253             case CACHE_ENTRY_INVALIDATED:
1254             case CACHE_ENTRY_LOADED:
1255             case CACHE_ENTRY_MODIFIED:
1256             case CACHE_ENTRY_PASSIVATED:
1257             case CACHE_ENTRY_REMOVED:
1258             case CACHE_ENTRY_VISITED:
1259                return new EventType(event.isCommandRetried(), event.isPre(), event.getType());
1260             default:
1261                return null;
1262          }
1263       }

1300       protected CacheEntryEvent<K, V> convertValue(CacheEventConverter<? super K, ? super V, ?> converter, CacheEntryEvent<K, V> event) {
1301          CacheEntryEvent<K, V> returnedEvent;
1302          if (converter != null) {
1303             if (event instanceof EventImpl) {
1304                // This is a bit hacky to let the C type be passed in for the V type
1305                EventImpl<K, V> eventImpl = (EventImpl<K, V>)event;
1306                Object newValue = converter.convert(eventImpl.getKey(), (V) eventImpl.getOldValue(),
1307                                  eventImpl.getOldMetadata(), (V) eventImpl.getValue(),
1308                                  eventImpl.getMetadata(), null);
1309                if (newValue != eventImpl.getValue()) {
1310                   EventImpl<K, V> clone = eventImpl.clone();
1311                   clone.setValue((V) newValue);
1312                   returnedEvent = clone;
1313                } else {
1314                   returnedEvent = eventImpl;
1315                }
1316             } else {
1317                throw new IllegalArgumentException("Provided event should be org.infinispan.notifications.cachelistener.event.impl.EventImpl " +
1318                                                         "when a converter is being used!");
1319             }
1320          } else {
1321             returnedEvent = event;
1322          }
1323          return returnedEvent;
1324       }

1332    @Override
1333    public void removeListener(Object listener) {
1334       super.removeListener(listener);
1335       UUID id = clusterListenerIDs.remove(listener);
1336       if (id != null) {
1337          List<Future<?>> futures = distExecutorService.submitEverywhere(new ClusterListenerRemoveCallable(id));
1338          for (Future<?> future : futures) {
1339             try {
1340                future.get();
1341             } catch (InterruptedException e) {
1342                throw new CacheListenerException(e);
1343             } catch (ExecutionException e) {
1344                throw new CacheListenerException(e);
1345             }
1346          }
1347       }
1348    }

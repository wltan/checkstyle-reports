24  /**
25   * Invoke a ClusteredQueryCommand on the cluster, including on own node.
26   * 
27   * @author Israel Lacerra <israeldl@gmail.com>
28   * @author Sanne Grinovero <sanne@infinispan.org> (C) 2011 Red Hat Inc.
29   * @since 5.1
30   */
31  public class ClusteredQueryInvoker {
32  
33     private final RpcManager rpcManager;
34     private final Cache<?, ?> localCacheInstance;
35     private final Address myAddress;
36     private final ExecutorService asyncExecutor;
37     private final RpcOptions rpcOptions;
38  
39     ClusteredQueryInvoker(Cache<?, ?> localCacheInstance, ExecutorService asyncExecutor) {
40        this.asyncExecutor = asyncExecutor;
41        this.rpcManager = localCacheInstance.getAdvancedCache().getComponentRegistry().getLocalComponent(RpcManager.class);
42        this.localCacheInstance = localCacheInstance;
43        this.myAddress = localCacheInstance.getAdvancedCache().getRpcManager().getAddress();
44        this.rpcOptions = rpcManager.getRpcOptionsBuilder(ResponseMode.SYNCHRONOUS).timeout(10000, TimeUnit.MILLISECONDS).build();
45     }
46  
47     /**
48      * Retrieves the value (using doc index) in a remote query instance
49      * 
50      * @param doc
51      *           Doc index of the value on remote query
52      * @param address
53      *           Address of the node who has the value
54      * @param queryId
55      *           Id of the query
56      * @return The value of index doc of the query with queryId on node at address
57      */
58     public Object getValue(int doc, Address address, UUID queryId) {
59        ClusteredQueryCommand clusteredQuery = ClusteredQueryCommand.retrieveKeyFromLazyQuery(
60                 localCacheInstance, queryId, doc);
61  
62        if (address.equals(myAddress)) {
63           Future<QueryResponse> localResponse = localInvoke(clusteredQuery);
64           try {
65              return localResponse.get().getFetchedValue();
66           } catch (InterruptedException e) {
67              throw new SearchException("interrupted while searching locally", e);
68           } catch (ExecutionException e) {
69              throw new SearchException("Exception while searching locally", e);
70           }
71        } else {
72           List<Address> addresss = new ArrayList<Address>(1);
73           addresss.add(address);
74  
75           Map<Address, Response> responses = rpcManager.invokeRemotely(addresss, clusteredQuery, rpcOptions);
76           List<QueryResponse> objects = cast(responses);
77           return objects.get(0).getFetchedValue();
78        }
79     }
80  
81     /**
82      * Broadcast this ClusteredQueryCommand to all cluster nodes. The command will be also invoked on
83      * local node.
84      * 
85      * @param clusteredQuery
86      * @return A list with all responses
87      */
88     public List<QueryResponse> broadcast(ClusteredQueryCommand clusteredQuery) {
89        // invoke on own node
90        Future<QueryResponse> localResponse = localInvoke(clusteredQuery);
91        Map<Address, Response> responses = rpcManager.invokeRemotely(null, clusteredQuery, rpcOptions);
92  
93        List<QueryResponse> objects = cast(responses);
94        final QueryResponse localReturnValue;
95        try {
96           localReturnValue = localResponse.get();
97        } catch (InterruptedException e1) {
98           throw new SearchException("interrupted while searching locally", e1);
99        } catch (ExecutionException e1) {
100          throw new SearchException("Exception while searching locally", e1);
101       }
102       objects.add(localReturnValue);
103       return objects;
104    }
105 
106    private Future<QueryResponse> localInvoke(ClusteredQueryCommand clusteredQuery) {
107       ClusteredQueryCallable clusteredQueryCallable = new ClusteredQueryCallable(clusteredQuery,
108                localCacheInstance);
109       return asyncExecutor.submit(clusteredQueryCallable);
110    }
111 
112    private List<QueryResponse> cast(Map<Address, Response> responses) {
113       List<QueryResponse> objects = new LinkedList<QueryResponse>();
114       for (Entry<Address, Response> pair : responses.entrySet()) {
115          Response resp = pair.getValue();
116          if (resp instanceof SuccessfulResponse) {
117             QueryResponse response = (QueryResponse) ((SuccessfulResponse) resp).getResponseValue();
118             objects.add(response);
119          } else {
120             throw new SearchException("Unexpected response: " + resp);
121          }
122       }
123 
124       return objects;
125    }
126 
127    /**
128     * Created to call a ClusteredQueryCommand on own node.
129     * 
130     * @author Israel Lacerra <israeldl@gmail.com>
131     * @since 5.1
132     */
133    private static final class ClusteredQueryCallable implements Callable<QueryResponse> {
134 
135       private final ClusteredQueryCommand clusteredQuery;
136 
137       private final Cache<?, ?> localInstance;
138 
139       ClusteredQueryCallable(ClusteredQueryCommand clusteredQuery, Cache<?, ?> localInstance) {
140          this.clusteredQuery = clusteredQuery;
141          this.localInstance = localInstance;
142       }
143 
144       @Override
145       public QueryResponse call() throws Exception {
146          try {
147             return clusteredQuery.perform(localInstance);
148          } catch (Throwable e) {
149             e.printStackTrace();
150             throw new Exception(e.getMessage());
151          }
152       }
153 
154    }

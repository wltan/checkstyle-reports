84      /**
85       * Trigger loading of a configuration by talking to the remote resource.
86       * @return The initialized configuration
87       * @throws Exception If anything goes wrong.
88       */
89      public Configuration loadResourceConfiguration() throws Exception {
90  
91          Configuration config = new Configuration();
92  
93          /*
94           * Grouped definitions get a special treatment, as they may have a special property
95           * that will be evaluated to look at a child resource or a special attribute or such
96           */
97          List<PropertyGroupDefinition> gdef = configurationDefinition.getGroupDefinitions();
98          for (PropertyGroupDefinition pgDef : gdef) {
99              loadHandleGroup(config, pgDef);
100         }
101         /*
102          * Now handle the non-grouped properties
103          */
104         List<PropertyDefinition> nonGroupdedDefs = configurationDefinition.getNonGroupedProperties();
105         Operation op = new ReadResource(address);
106         if (includeRuntime) {
107             op.addAdditionalProperty("include-runtime", "true"); // Include runtime values for attributes
108         }
109         //        op.addAdditionalProperty("recursive", "true"); // Also get sub-resources
110         loadHandleProperties(config, nonGroupdedDefs, op);
111 
112         return config;
113     }

115     /**
116      * Handle a set of grouped properties. The name of the group tells us how to deal with it:
117      * <ul>
118      *     <li>attribute: read the passed attribute of the resource</li>
119      *     <li>children:  read the children of the given child-type</li>
120      * </ul>
121      * @param config Configuration to return
122      * @param groupDefinition Definition of this group
123      * @throws Exception If anything goes wrong
124      */
125     private void loadHandleGroup(Configuration config, PropertyGroupDefinition groupDefinition) throws Exception {
126         Operation operation = null;
127         String groupName = groupDefinition.getName();
128         if (groupName.startsWith("attribute:")) {
129             String attr = groupName.substring("attribute:".length());
130             operation = new ReadAttribute(address, attr);
131         } else if (groupName.startsWith("children:")) {
132             String type = groupName.substring("children:".length());
133             if (type.contains(":")) {
134 
135                 // If the third part ends with a ?, we fill this config prop with the resource name from the path
136                 String tmp = type.substring(type.indexOf(":") + 1);
137                 if (tmp.endsWith("+-")) {
138                     nameFromPathProperty = tmp.substring(0, tmp.length() - 2);
139                 }
140 
141                 // We need the type for reading
142                 type = type.substring(0, type.indexOf(":"));
143 
144             }
145             operation = new ReadChildrenResources(address, type);
146             operation.addAdditionalProperty("recursive", "true");
147         } else if (groupName.startsWith("child:")) {
148             String subPath = groupName.substring("child:".length());
149             if (!subPath.contains("="))
150                 throw new IllegalArgumentException("subPath of 'child:' expression has no =");
151 
152             if (subPath.contains(":")) {
153                 subPath = subPath.substring(0,subPath.indexOf(':'));
154             }
155             boolean includeRuntime=false;
156             if (subPath.endsWith("*")) {
157                 subPath = subPath.substring(0,subPath.length()-1);
158                 includeRuntime=true;
159             }
160 
161             Address address1 = new Address(address);
162             address1.addSegment(subPath);
163             operation = new ReadResource(address1);
164             if (includeRuntime)
165                 ((ReadResource)operation).includeRuntime(true);
166         } else {//no special handling of <c:groups> details required.
167             //Just assume normal group operations aggregation semantics, so retrieve entries and load.
168             ReadResource op = new ReadResource(address);
169 // Added these lines to allow for runtime attributes to be returned
170             if (includeRuntime) {
171                op.includeRuntime(true);
172             }
173 // Added these lines to allow for runtime attributes to be returned
174             List<PropertyDefinition> listedDefs = configurationDefinition.getPropertiesInGroup(groupName);
175             loadHandleProperties(config, listedDefs, op);
176             return;
177         }
178         List<PropertyDefinition> listedDefs = configurationDefinition.getPropertiesInGroup(groupName);
179         loadHandleProperties(config, listedDefs, operation);
180 
181     }

183     private void loadHandleProperties(Configuration config, List<PropertyDefinition> definitions, Operation op)
184         throws Exception {
185         if (definitions.size() == 0)
186             return;
187 
188         Result operationResult = connection.execute(op);
189         if (!operationResult.isSuccess()) {
190             // check if the definition is needed -- required==true or not. Only complain for required ones
191             if (definitions.size()==1 && definitions.get(0).isRequired()==false) {
192                 return;
193             }
194 
195             throw new IOException("Operation " + op + " failed: " + operationResult.getFailureDescription());
196         }
197 
198         if (operationResult.getResult() instanceof List) {
199             PropertyList propertyList = loadHandlePropertyList((PropertyDefinitionList) definitions.get(0),
200                 operationResult.getResult());
201 
202             if (propertyList != null)
203                 config.put(propertyList);
204             return;
205         }
206 
207         Map<String, Object> results = (Map<String, Object>) operationResult.getResult();
208 
209         for (PropertyDefinition propDef : definitions) {
210 
211             /*
212              * We may have a mismatch for groups where the <c:group name="children:*"> child is a list of maps
213              * but the result is actually the contained map
214              */
215             String propertyName = propDef.getName();
216             if (propDef instanceof PropertyDefinitionList && (propertyName.startsWith("*"))) {
217                 propDef = ((PropertyDefinitionList) propDef).getMemberDefinition();
218 
219                 if (!(propDef instanceof PropertyDefinitionMap)) {
220                     log.error("Embedded child is not a map");
221                     return;
222                 }
223                 // Now we are at map level which matches the operations results
224 
225                 PropertyList list = new PropertyList(propertyName);
226                 if (results != null) {
227                     for (Map.Entry<String, Object> entry : results.entrySet()) {
228                         Object val = entry.getValue();
229                         String key = entry.getKey();
230 
231                         PropertyMap propertyMap = loadHandlePropertyMap((PropertyDefinitionMap) propDef, val, key);
232 
233                         if (nameFromPathProperty != null) {
234                             // We need to fill that property as well
235                             PropertySimple ps = new PropertySimple(nameFromPathProperty, key);
236                             propertyMap.put(ps);
237                         }
238 
239                         if (propertyMap != null)
240                             list.add(propertyMap);
241                     }
242                 }
243 
244 
245                 config.put(list);
246             } else if (propDef instanceof PropertyDefinitionMap && (propertyName.startsWith("*"))) {
247                 // a map in a group with name child:type=name
248                 Map<String, Object> mapResult = results;
249                 PropertyMap propertyMap = loadHandlePropertyMap((PropertyDefinitionMap) propDef, mapResult,
250                     propertyName);
251                 config.put(propertyMap);
252 
253             } else { // standard case
254 
255                 Object valueObject;
256 
257                 //strip :number from properties
258                 if (propertyName.contains(":")) {
259                     try {
260                         Integer.parseInt(propertyName.substring(propertyName.lastIndexOf(':') + 1));
261                         propertyName = propertyName.substring(0, propertyName.lastIndexOf(':'));
262                     } catch (Exception e) {
263                         //do nothing, this means the property name does not end with :number, so nothing needs to be stripped
264                     }
265                 }
266 
267                 if (propertyName.endsWith(":expr") || propertyName.endsWith(":collapsed")
268                     || propertyName.endsWith(":nullable")) {
269                     String realName = propertyName.substring(0, propertyName.indexOf(":"));
270                     valueObject = results.get(realName);
271                 } else {
272                     valueObject = results.get(propertyName);
273                 }
274 
275                 if (propDef instanceof PropertyDefinitionSimple) {
276 
277                     PropertySimple value = loadHandlePropertySimple((PropertyDefinitionSimple) propDef, valueObject);
278                     if (value != null)
279                         config.put(value);
280                 }
281 
282                 else if (propDef instanceof PropertyDefinitionList) {
283                     PropertyList propertyList = loadHandlePropertyList((PropertyDefinitionList) propDef, valueObject);
284 
285                     if (propertyList != null)
286                         config.put(propertyList);
287                 } else if (propDef instanceof PropertyDefinitionMap) {
288                     PropertyMap propertyMap = loadHandlePropertyMap((PropertyDefinitionMap) propDef, valueObject, null);
289 
290                     if (propertyMap != null)
291                         config.put(propertyMap);
292                 }
293             }
294         }
295     }

297     PropertySimple loadHandlePropertySimple(PropertyDefinitionSimple propDef, Object valueObject) {
298         PropertySimple propertySimple;
299 
300         String name = propDef.getName();
301         if (valueObject != null) {
302             // Property is non-null -> return it.
303 
304             if (valueObject instanceof Map) { // If this is a map and no single type, get the EXPRESSION_VALUE
305                 Object o = ((Map) valueObject).get("EXPRESSION_VALUE");
306                 propertySimple = new PropertySimple(name, o);
307             } else {
308                 propertySimple = new PropertySimple(name, valueObject);
309             }
310         } else {
311             // property is null? Check if it is required
312             if (propDef.isRequired()) {
313                 String defaultValue = ((PropertyDefinitionSimple) propDef).getDefaultValue();
314                 propertySimple = new PropertySimple(name, defaultValue);
315             } else { // Not required and null -> return null
316                 propertySimple = new PropertySimple(name, null);
317             }
318         }
319         return propertySimple;
320 
321     }

323     /**
324      * Handle a Map of ...
325      *
326      * @param propDef Definition of the map
327      * @param valueObject the objects to put into the map
328      * @param optionalEntryName
329      * @return the populated map
330      */
331     PropertyMap loadHandlePropertyMap(PropertyDefinitionMap propDefMap, Object valueObject, String optionalEntryName) {
332         if (valueObject == null)
333             return null;
334 
335         String propDefName = propDefMap.getName();
336         PropertyMap propertyMap = new PropertyMap(propDefName);
337         String specialNameProp = null;
338         if (propDefName.startsWith("*:")) {
339             specialNameProp = propDefName.substring(2);
340             PropertySimple additionalNameProperty = new PropertySimple(specialNameProp, optionalEntryName);
341             propertyMap.put(additionalNameProperty);
342         }
343 
344         List<PropertyDefinition> propDefs = propDefMap.getPropertyDefinitions();
345 
346         if (propDefName.endsWith(":collapsed")) {
347             // The result is a map of {" a" : " b" }, while the propdef is in the form
348             // map { prop {name=a } , prop {name=b) }
349             if (propDefs.size()<1 || propDefs.size() > 2)
350                 throw new IllegalArgumentException("Collapsed map [" + propDefName + "] needs 1 or 2 entries and not "
351                     + propDefs.size());
352 
353             Iterator<PropertyDefinition> iterator = propDefs.iterator();
354             String name = iterator.next().getName();
355             Map<String, String> objects = (Map<String, String>) valueObject;
356             String val = objects.keySet().iterator().next();
357             PropertySimple ps = new PropertySimple(name, val);
358             propertyMap.put(ps);
359             if (iterator.hasNext()) {
360                 name = iterator.next().getName();
361                 val = objects.values().iterator().next();
362                 ps = new PropertySimple(name, val);
363                 propertyMap.put(ps);
364             }
365 
366             return propertyMap;
367         }
368 
369         Map<String, Object> objects = (Map<String, Object>) valueObject;
370         for (PropertyDefinition propDef : propDefs) {
371             String key = propDef.getName();
372             if (key.equals(specialNameProp)) // Skip over specialName prop, as we have processed that already.
373                 continue;
374 
375             // special case: if the key is "*", we just pick the first element
376             Object o;
377             if (key.equals("*")) {
378                 o = objects.entrySet().iterator().next().getValue();
379             } else if (key.endsWith(":expr")) {
380                 // TODO we need to check te
381                 String tmp = key.substring(0, key.indexOf(":"));
382                 o = objects.get(tmp);
383             } else {
384                 o = objects.get(key);
385             }
386 
387             Property property;
388             if (propDef instanceof PropertyDefinitionSimple)
389                 property = loadHandlePropertySimple((PropertyDefinitionSimple) propDef, o);
390             else if (propDef instanceof PropertyDefinitionList)
391                 property = loadHandlePropertyList((PropertyDefinitionList) propDef, o);
392             else if (propDef instanceof PropertyDefinitionMap)
393                 property = loadHandlePropertyMap((PropertyDefinitionMap) propDef, o, null);
394             else
395                 throw new IllegalArgumentException("Unknown property type in map property [" + propDefName + "]");
396 
397             if (property != null)
398                 propertyMap.put(property);
399             else {
400                 if (log.isDebugEnabled())
401                     log.debug("Property " + key + " was null");
402             }
403 
404         }
405 
406         return propertyMap;
407     }

409     /**
410      * Handle a List of ...
411      * @param propDef Definition of this list
412      * @param valueObject The objects to put into the list
413      * @return the property that describes the list.
414      */
415     PropertyList loadHandlePropertyList(PropertyDefinitionList propDef, Object valueObject) {
416         String propertyName = propDef.getName();
417         PropertyList propertyList = new PropertyList(propertyName);
418         PropertyDefinition memberDefinition = propDef.getMemberDefinition();
419         if (memberDefinition == null)
420             throw new IllegalArgumentException("Member definition for property [" + propertyName + "] was null");
421 
422         if (valueObject == null) {
423             //            System.out.println("vo null");
424             return null;
425         }
426 
427         Collection<Object> objects;
428         if (valueObject instanceof List)
429             objects = (List<Object>) valueObject;
430         else /*if (valueObject instanceof Map)*/{
431             objects = ((Map) valueObject).values();
432         }
433 
434         if (memberDefinition instanceof PropertyDefinitionSimple) {
435             for (Object obj : objects) {
436                 PropertySimple property = loadHandlePropertySimple((PropertyDefinitionSimple) memberDefinition, obj);
437                 if (property != null)
438                     propertyList.add(property);
439             }
440         } else if (memberDefinition instanceof PropertyDefinitionMap) {
441             for (Object obj : objects) {
442                 Map<String, Object> map = (Map<String, Object>) obj;
443 
444                 PropertyMap propertyMap = loadHandlePropertyMap((PropertyDefinitionMap) propDef.getMemberDefinition(),
445                     map, null);
446                 if (propertyMap != null)
447                     propertyList.add(propertyMap);
448             }
449         }
450         // TODO List of lists ?
451         return propertyList;
452     }

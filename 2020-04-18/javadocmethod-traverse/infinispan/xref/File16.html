49     /**
50      * Returns the contents of a specific node of given element name, provided a certain attribute exists and is set to
51      * value. E.g., if you have a {@link Element} which represents the following XML snippet:
52      * <pre>
53      *   &lt;ItemQuantity Colour="Red"&gt;100&lt;/ItemQuantity&gt;
54      *   &lt;ItemQuantity Colour="Blue"&gt;30&lt;/ItemQuantity&gt;
55      *   &lt;ItemQuantity Colour="Black"&gt;10&lt;/ItemQuantity&gt;
56      * <pre>
57      * <p/>
58      * The following results could be expected:
59      * </p>
60      * <pre>
61      *    getTagContents(element, "Red", "ItemQuantity", "Colour"); // 100
62      *    getTagContents(element, "Black", "ItemQuantity", "Colour"); // 10
63      *    getTagContents(element, "Blah", "ItemQuantity", "Colour"); // null
64      *    getTagContents(element, "Red", "Blah", "Colour"); // null
65      *    getTagContents(element, "Black", "ItemQuantity", "Blah"); // null
66      * </pre>
67      * <p/>
68      * None of the parameters should be null - otherwise the method may throw a NullPointerException.
69      * </p>
70      *
71      * @param elem          - element to search through.
72      * @param value         - expected value to match against
73      * @param elementName   - element name
74      * @param attributeName - attribute name of the element that would contain the expected value.
75      * @return the contents of the matched element, or null if not found/matched
76      */
77     public static String getTagContents(Element elem, String value, String elementName, String attributeName) {
78        NodeList list = elem.getElementsByTagName(elementName);
79  
80        for (int s = 0; s < list.getLength(); s++) {
81           org.w3c.dom.Node node = list.item(s);
82           if (node.getNodeType() != org.w3c.dom.Node.ELEMENT_NODE)
83              continue;
84  
85           Element element = (Element) node;
86           String name = element.getAttribute(attributeName);
87           if (name.equals(value)) {
88              return getElementContent(element, true);
89           }
90        }
91        return null;
92     }

94     /**
95      * Retrieves the value of a given attribute for the first encountered instance of a tag in an element. <p/> E.g., if
96      * you have a {@link Element} which represents the following XML snippet: </p>
97      * <pre>
98      *   &lt;ItemQuantity Colour="Red"&gt;100&lt;/ItemQuantity&gt;
99      *   &lt;ItemQuantity Colour="Blue"&gt;30&lt;/ItemQuantity&gt;
100     *   &lt;ItemQuantity Colour="Black"&gt;10&lt;/ItemQuantity&gt;
101     * <pre>
102     * <p/>
103     * The following results could be expected:
104     * </p>
105     * <pre>
106     *    getAttributeValue(element, "ItemQuantity", "Colour"); // "Red"
107     *    getTagContents(element, "Blah", "Colour"); // null
108     *    getTagContents(element, "ItemQuantity", "Blah"); // null
109     * </pre>
110     * None of the parameters should be null - otherwise the method may throw a NullPointerException.
111     *
112     * @param elem          - element to search through.
113     * @param elementName   - element name
114     * @param attributeName - attribute name of the element that would contain the expected value.
115     * @return the contents of the matched attribute, or null if not found/matched
116     */
117    public static String getAttributeValue(Element elem, String elementName, String attributeName) {
118       NodeList list = elem.getElementsByTagName(elementName);
119 
120       for (int s = 0; s < list.getLength(); s++) {
121          org.w3c.dom.Node node = list.item(s);
122          if (node.getNodeType() != org.w3c.dom.Node.ELEMENT_NODE)
123             continue;
124 
125          Element element = (Element) node;
126          String value = element.getAttribute(attributeName);
127          return value == null ? null : StringPropertyReplacer.replaceProperties(value);
128 
129       }
130       return null;
131    }

133    /**
134     * Returns a named sub-element of the current element passed in.
135     * <p/>
136     * None of the parameters should be null - otherwise the method may throw a NullPointerException.
137     *
138     * @param element        - element to search through.
139     * @param subElementName - the name of a sub element to look for
140     * @return the first matching sub element, if found, or null otherwise.
141     */
142    public static Element getSubElement(Element element, String subElementName) {
143       NodeList nl = element.getChildNodes();
144       for (int i = 0; i < nl.getLength(); i++) {
145          Node node = nl.item(i);
146          if (node.getNodeType() == Node.ELEMENT_NODE && subElementName.equals(((Element) node).getTagName())) {
147             return (Element) node;
148          }
149       }
150 
151       if (log.isDebugEnabled()) log.debugf("getSubElement(): Does not exist for %s", subElementName);
152       return null;
153    }

155    /**
156     * Reads the contents of the element passed in.
157     * <p/>
158     * None of the parameters should be null - otherwise the method may throw a NullPointerException.
159     *
160     * @param element - element to search through.
161     * @param trim    - if true, whitespace is trimmed before returning
162     * @return the contents of the element passed in.  Will return an empty String if the element is empty.
163     */
164    public static String getElementContent(Element element, boolean trim) {
165       NodeList nl = element.getChildNodes();
166       StringBuilder attributeText = new StringBuilder();
167       for (int i = 0; i < nl.getLength(); i++) {
168          Node n = nl.item(i);
169          if (n instanceof Text) {
170             attributeText.append(StringPropertyReplacer.replaceProperties(((Text) n).getData()));
171          }
172       } // end of for ()
173       if (trim)
174          return attributeText.toString().trim();
175       return attributeText.toString();
176    }

178    /**
179     * Reads the contents of the first occurrence of elementName under the given element, trimming results of whitespace.
180     * <p/>
181     * None of the parameters should be null - otherwise the method may throw a NullPointerException.
182     *
183     * @param element     - element to search through.
184     * @param elementName - name of the element to find within the element passed in
185     * @return may return an empty String of not found.
186     */
187    public static String readStringContents(Element element, String elementName) {
188       NodeList nodes = element.getElementsByTagName(elementName);
189       if (nodes.getLength() > 0) {
190          Node node = nodes.item(0);
191          Element ne = (Element) node;
192          NodeList nl2 = ne.getChildNodes();
193          Node node2 = nl2.item(0);
194          if (node2 != null) {
195             String value = node2.getNodeValue();
196             if (value == null)
197                return "";
198             return StringPropertyReplacer.replaceProperties(value.trim());
199          } else {
200             return "";
201          }
202       } else {
203          return "";
204       }
205    }

207    /**
208     * Escapes backslashes ('\') with additional backslashes in a given String, returning a new, escaped String.
209     *
210     * @param value String to escape.   Cannot be null.
211     * @return escaped String.  Never is null.
212     */
213    public static String escapeBackslashes(String value) {
214       StringBuilder buf = new StringBuilder(value);
215       for (int looper = 0; looper < buf.length(); looper++) {
216          char curr = buf.charAt(looper);
217          char next = 0;
218          if (looper + 1 < buf.length())
219             next = buf.charAt(looper + 1);
220 
221          if (curr == '\\') {
222             if (next != '\\') {           // only if not already escaped
223                buf.insert(looper, '\\');  // escape backslash
224             }
225             looper++;                    // skip past extra backslash (either the one we added or existing)
226          }
227       }
228       return buf.toString();
229    }

231    /**
232     * Reads the contents of a named sub element within a given element, and attempts to parse the contents as a Java
233     * properties file.
234     * <p/>
235     * E.g., if you have a {@link Element} which represents the following XML snippet:
236     * <p/>
237     * <pre>
238     *   &lt;props&gt;
239     *       my.attrib.1 = blah
240     *       my.attrib.2 = blahblah
241     *   &lt;/props&gt;
242     * <pre>
243     * <p/>
244     * The following results could be expected:
245     * <p/>
246     * <pre>
247     *    Properties p = readPropertiesContents(element, "props");
248     *    p.getProperty("my.attrib.1"); // blah
249     *    p.getProperty("my.attrib.2"); // blahblah
250     * </pre>
251     * None of the parameters should be null - otherwise the method may throw a NullPointerException.
252     *
253     * @param element     - element to search through.
254     * @param elementName - name of the element to find within the element passed in
255     * @return a {@link Properties} object, never null.
256     * @throws IOException if unable to parse the contents of the element
257     */
258    public static Properties readPropertiesContents(Element element, String elementName) {
259       String stringContents = readStringContents(element, elementName);
260       if (stringContents == null) return new Properties();
261       stringContents = escapeBackslashes(stringContents);
262       ByteArrayInputStream is;
263       Properties properties;
264       try {
265          is = new ByteArrayInputStream(stringContents.trim().getBytes("ISO8859_1"));
266          properties = new Properties();
267          properties.load(is);
268          is.close();
269       }
270       catch (IOException e) {
271          log.errorReadingProperties(e);
272          throw new CacheConfigurationException("Exception occured while reading properties from XML document", e);
273       }
274       return properties;
275    }

311     * Converts a String representing an XML snippet into an {@link org.w3c.dom.Element}.
312     *
313     * @param xml snippet as a string
314     * @return a DOM Element
315     * @throws Exception if unable to parse the String or if it doesn't contain valid XML.
316     */
317    public static Element stringToElement(String xml) throws Exception {
318       ByteArrayInputStream bais = new ByteArrayInputStream(xml.getBytes("utf8"));
319       DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
320       DocumentBuilder builder = factory.newDocumentBuilder();
321       Document d = builder.parse(bais);
322       bais.close();
323       return d.getDocumentElement();
324    }

340    /**
341     * Returns the root element of a given input stream
342     *
343     * @param is stream to parse
344     * @return XML DOM element, or null if unable to parse stream
345     */
346    public static Element getDocumentRoot(InputStream is) {
347       Document doc;
348       try {
349          InputSource xmlInp = new InputSource(is);
350 
351          DocumentBuilderFactory docBuilderFactory = DocumentBuilderFactory.newInstance();
352          docBuilderFactory.setNamespaceAware(true);
353          DocumentBuilder parser = docBuilderFactory.newDocumentBuilder();
354          doc = parser.parse(xmlInp);
355          Element root = doc.getDocumentElement();
356          root.normalize();
357          return root;
358       }
359       catch (SAXParseException err) {
360          log.configuratorSAXParseError(err);
361       }
362       catch (SAXException e) {
363          log.configuratorSAXError(e);
364       }
365       catch (Exception pce) {
366          log.configuratorError(pce);
367       }
368       return null;
369    }

392    public static Object valueConverter(@SuppressWarnings("rawtypes") Class klass, String value) {
393       if (klass == Integer.class) {
394          return Integer.valueOf(value);
395       } else if (klass == Long.class) {
396          return Long.valueOf(value);
397       } else if (klass == Boolean.class) {
398          return Boolean.valueOf(value);
399       } else if (klass == String.class) {
400          return value;
401       } else if (klass == Float.class) {
402          return Float.valueOf(value);
403       } else if (klass == Double.class) {
404          return Double.valueOf(value);
405       } else if (klass .isEnum()) {
406          return Enum.valueOf(klass, value);
407       } else {
408          throw new CacheConfigurationException("Cannot convert "+ value + " to type " + klass.getName());
409       }
410    }

412    public static Map<Object, Object> setAttributes(AttributeSet attributes, Map<?, ?> attribs, boolean isXmlAttribs, boolean failOnMissingAttribute) {
413       Map<Object, Object> ignoredAttribs = new HashMap<Object, Object>();
414       for(Entry<?, ?> entry : attribs.entrySet()) {
415          String name = (String) entry.getKey();
416          if (attributes.contains(name)) {
417             Attribute<Object> attribute = attributes.attribute(name);
418             attribute.set(valueConverter(attribute.getAttributeDefinition().getType(), (String) entry.getValue()));
419          } else if (failOnMissingAttribute) {
420             throw new CacheConfigurationException("Couldn't find an attribute named [" + name + "] on attribute set [" + attributes.getName() + "]");
421          } else {
422             ignoredAttribs.put(name, entry.getValue());
423          }
424       }
425       return ignoredAttribs;
426    }

428    public static Map<Object, Object> setValues(Object target, Map<?, ?> attribs, boolean isXmlAttribs, boolean failOnMissingSetter) {
429       Class<?> objectClass = target.getClass();
430       Map<Object, Object> ignoredAttribs = new HashMap<Object, Object>();
431       // go thru simple string setters first.
432       for (Map.Entry<?, ?> entry : attribs.entrySet()) {
433          String propName = (String) entry.getKey();
434          String setter = BeanUtils.setterName(propName);
435          String fluentSetter = BeanUtils.fluentSetterName(propName);
436 
437          try {
438             Method method;
439             if (isXmlAttribs) {
440                method = objectClass.getMethod(setter, Element.class);
441                method.invoke(target, entry.getValue());
442             } else {
443                method = objectClass.getMethod(setter, String.class);
444                method.invoke(target, entry.getValue());
445             }
446 
447             continue;
448          }
449          catch (NoSuchMethodException me) {
450             // this is ok, but certainly log this as a warning
451             // this is hugely noisy!
452             //if (log.isWarnEnabled()) log.warn("Unrecognised attribute " + propName + ".  Please check your configuration.  Ignoring!!");
453          }
454          catch (Exception e) {
455             throw new CacheConfigurationException("Unable to invoke setter " + setter + " on " + objectClass, e);
456          }
457 
458          boolean setterFound = false;
459          // if we get here, we could not find a String or Element setter.
460          for (Method m : objectClass.getMethods()) {
461             if (setter.equals(m.getName()) || fluentSetter.equals(m.getName())) {
462                Class<?> paramTypes[] = m.getParameterTypes();
463                if (paramTypes.length != 1) {
464                   log.tracef("Rejecting setter %s on class %s due to incorrect number of parameters", m, objectClass);
465                   continue; // try another param with the same name.
466                }
467 
468                Class<?> parameterType = paramTypes[0];
469                PropertyEditor editor = PropertyEditorManager.findEditor(parameterType);
470                if (editor == null) {
471                   throw new CacheConfigurationException("Couldn't find a property editor for parameter type " + parameterType);
472                }
473 
474                editor.setAsText((String) attribs.get(propName));
475 
476                Object parameter = editor.getValue();
477                //if (log.isDebugEnabled()) log.debug("Invoking setter method: " + setter + " with parameter \"" + parameter + "\" of type " + parameter.getClass());
478 
479                try {
480                   m.invoke(target, parameter);
481                   setterFound = true;
482                   break;
483                }
484                catch (Exception e) {
485                   throw new CacheConfigurationException("Unable to invoke setter " + setter + " on " + objectClass, e);
486                }
487             }
488          }
489          // Skip hot rod properties ...
490          if (!setterFound && !propName.startsWith("infinispan.client.hotrod"))
491             if (failOnMissingSetter) {
492                throw new CacheConfigurationException("Couldn't find a setter named [" + setter + "] which takes a single parameter, for parameter " + propName + " on class [" + objectClass + "]");
493             } else {
494                ignoredAttribs.put(propName, attribs.get(propName));
495             }
496       }
497       return ignoredAttribs;
498    }

506    public static Properties extractProperties(Element source) {
507       TypedProperties p = new TypedProperties();
508       NodeList list = source.getElementsByTagName("property");
509       if (list == null) return null;
510       // loop through attributes
511       for (int loop = 0; loop < list.getLength(); loop++) {
512          Node node = list.item(loop);
513          if (node.getNodeType() != Node.ELEMENT_NODE) continue;
514 
515          // for each element (attribute) ...
516          Element element = (Element) node;
517          String name = element.getAttribute("name");
518          String valueStr = element.getAttribute("value");
519 
520          if (valueStr.length() > 0) {
521             valueStr = valueStr.trim();
522             valueStr = StringPropertyReplacer.replaceProperties(valueStr);
523             p.put(name, valueStr);
524          }
525       }
526       return p.isEmpty() ? null : p;
527    }

529    public static String toString(Element e) {
530       try {
531          TransformerFactory tfactory = TransformerFactory.newInstance();
532          Transformer xform = tfactory.newTransformer();
533          Source src = new DOMSource(e);
534          java.io.StringWriter writer = new StringWriter();
535          Result result = new javax.xml.transform.stream.StreamResult(writer);
536          xform.transform(src, result);
537          return writer.toString();
538       }
539       catch (Exception ex) {
540          return "Unable to convert to string: " + ex.toString();
541       }

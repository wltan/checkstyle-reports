412     /**
413      * return previous opcode;
414      *
415      * @param offset
416      *            0 for current opcode, 1 for one before that, etc.
417      */
418     public int getPrevOpcode(int offset) {
419         if (offset < 0) {
420             throw new IllegalArgumentException("offset (" + offset + ") must be nonnegative");
421         }
422         if (offset >= prevOpcode.length || offset > sizePrevOpcodeBuffer) {
423             return Const.NOP;
424         }
425         int pos = currentPosInPrevOpcodeBuffer - offset;
426         if (pos < 0) {
427             pos += prevOpcode.length;
428         }
429         return prevOpcode[pos];
430     }

493     private static void sortByOffset(int[] switchOffsets, int[] switchLabels) {
494         int npairs = switchOffsets.length;
495         // Sort by offset
496         for (int j = 0; j < npairs; j++) {
497             int min = j;
498             for (int k = j + 1; k < npairs; k++) {
499                 if (switchOffsets[min] > switchOffsets[k]) {
500                     min = k;
501                 }
502             }
503             if (min > j) {
504                 int tmp = switchOffsets[min];
505                 switchOffsets[min] = switchOffsets[j];
506                 switchOffsets[j] = tmp;
507                 tmp = switchLabels[min];
508                 switchLabels[min] = switchLabels[j];
509                 switchLabels[j] = tmp;
510             }
511         }
512     }

535     @Override
536     public void visit(Code obj) {
537         //        if (getXMethod().usesInvokeDynamic()) {
538         //            AnalysisContext.currentAnalysisContext().analysisSkippedDueToInvokeDynamic(getXMethod());
539         //            return;
540         //        }
541         sizePrevOpcodeBuffer = 0;
542         currentPosInPrevOpcodeBuffer = prevOpcode.length - 1;
543 
544         int switchLow = 1000000;
545         int switchHigh = -1000000;
546         codeBytes = obj.getCode();
547         DataInputStream byteStream = new DataInputStream(new ByteArrayInputStream(codeBytes));
548 
549         lineNumberTable = obj.getLineNumberTable();
550 
551         try {
552             for (int i = 0; i < codeBytes.length;) {
553                 resetState();
554                 PC = i;
555                 opcodeIsWide = false;
556                 opcode = byteStream.readUnsignedByte();
557 
558                 sizePrevOpcodeBuffer++;
559                 currentPosInPrevOpcodeBuffer++;
560                 if (currentPosInPrevOpcodeBuffer >= prevOpcode.length) {
561                     currentPosInPrevOpcodeBuffer = 0;
562                 }
563                 prevOpcode[currentPosInPrevOpcodeBuffer] = opcode;
564                 i++;
565                 // System.out.println(OPCODE_NAMES[opCode]);
566                 int byteStreamArgCount = Const.getNoOfOperands(opcode);
567                 if (byteStreamArgCount == Const.UNPREDICTABLE) {
568 
569                     if (opcode == Const.LOOKUPSWITCH) {
570                         int pad = 4 - (i & 3);
571                         if (pad == 4) {
572                             pad = 0;
573                         }
574                         int count = pad;
575                         while (count > 0) {
576                             count -= byteStream.skipBytes(count);
577                         }
578                         i += pad;
579                         defaultSwitchOffset = byteStream.readInt();
580                         branchOffset = defaultSwitchOffset;
581                         branchTarget = branchOffset + PC;
582                         i += 4;
583                         int npairs = byteStream.readInt();
584                         i += 4;
585                         switchOffsets = new int[npairs];
586                         switchLabels = new int[npairs];
587                         for (int o = 0; o < npairs; o++) {
588                             switchLabels[o] = byteStream.readInt();
589                             switchOffsets[o] = byteStream.readInt();
590                             i += 8;
591                         }
592                         sortByOffset(switchOffsets, switchLabels);
593                     } else if (opcode == Const.TABLESWITCH) {
594                         int pad = 4 - (i & 3);
595                         if (pad == 4) {
596                             pad = 0;
597                         }
598                         int count = pad;
599                         while (count > 0) {
600                             count -= byteStream.skipBytes(count);
601                         }
602                         i += pad;
603                         defaultSwitchOffset = byteStream.readInt();
604                         branchOffset = defaultSwitchOffset;
605                         branchTarget = branchOffset + PC;
606                         i += 4;
607                         switchLow = byteStream.readInt();
608                         i += 4;
609                         switchHigh = byteStream.readInt();
610                         i += 4;
611                         int npairs = switchHigh - switchLow + 1;
612                         switchOffsets = new int[npairs];
613                         switchLabels = new int[npairs];
614                         for (int o = 0; o < npairs; o++) {
615                             switchLabels[o] = o + switchLow;
616                             switchOffsets[o] = byteStream.readInt();
617                             i += 4;
618                         }
619                         sortByOffset(switchOffsets, switchLabels);
620                     } else if (opcode == Const.WIDE) {
621                         opcodeIsWide = true;
622                         opcode = byteStream.readUnsignedByte();
623                         i++;
624                         switch (opcode) {
625                         case Const.ILOAD:
626                         case Const.FLOAD:
627                         case Const.ALOAD:
628                         case Const.LLOAD:
629                         case Const.DLOAD:
630                         case Const.ISTORE:
631                         case Const.FSTORE:
632                         case Const.ASTORE:
633                         case Const.LSTORE:
634                         case Const.DSTORE:
635                         case Const.RET:
636                             registerOperand = byteStream.readUnsignedShort();
637                             i += 2;
638                             break;
639                         case Const.IINC:
640                             registerOperand = byteStream.readUnsignedShort();
641                             i += 2;
642                             intConstant = byteStream.readShort();
643                             i += 2;
644                             break;
645                         default:
646                             throw new IllegalStateException(String.format("bad wide bytecode %d: %s" , opcode, Const.getOpcodeName(opcode)));
647                         }
648                     } else {
649                         throw new IllegalStateException(String.format("bad unpredicatable bytecode %d: %s" , opcode, Const.getOpcodeName(opcode)));
650                     }
651                 } else {
652                     if (byteStreamArgCount < 0) {
653                         throw new IllegalStateException(String.format("bad length for bytecode %d: %s" , opcode, Const.getOpcodeName(opcode)));
654                     }
655                     for (int k = 0; k < Const.getOperandTypeCount(opcode); k++) {
656 
657                         int v;
658                         int t = Const.getOperandType(opcode, k);
659                         int m = MEANING_OF_OPERANDS[opcode][k];
660                         boolean unsigned = (m == M_CP || m == M_R || m == M_UINT);
661                         switch (t) {
662                         case Const.T_BYTE:
663                             if (unsigned) {
664                                 v = byteStream.readUnsignedByte();
665                             } else {
666                                 v = byteStream.readByte();
667                             }
668                             /*
669                              * System.out.print("Read byte " + v);
670                              * System.out.println(" with meaning" + m);
671                              */
672                             i++;
673                             break;
674                         case Const.T_SHORT:
675                             if (unsigned) {
676                                 v = byteStream.readUnsignedShort();
677                             } else {
678                                 v = byteStream.readShort();
679                             }
680                             i += 2;
681                             break;
682                         case Const.T_INT:
683                             v = byteStream.readInt();
684                             i += 4;
685                             break;
686                         default:
687                             throw new IllegalStateException();
688                         }
689                         switch (m) {
690                         case M_BR:
691                             branchOffset = v;
692                             branchTarget = v + PC;
693                             branchFallThrough = i;
694                             break;
695                         case M_CP:
696                             constantRefOperand = getConstantPool().getConstant(v);
697                             if (constantRefOperand instanceof ConstantClass) {
698                                 ConstantClass clazz = (ConstantClass) constantRefOperand;
699                                 classConstantOperand = getStringFromIndex(clazz.getNameIndex());
700                                 referencedClass = DescriptorFactory.createClassDescriptor(classConstantOperand);
701 
702                             } else if (constantRefOperand instanceof ConstantInteger) {
703                                 intConstant = ((ConstantInteger) constantRefOperand).getBytes();
704                             } else if (constantRefOperand instanceof ConstantLong) {
705                                 longConstant = ((ConstantLong) constantRefOperand).getBytes();
706                             } else if (constantRefOperand instanceof ConstantFloat) {
707                                 floatConstant = ((ConstantFloat) constantRefOperand).getBytes();
708                             } else if (constantRefOperand instanceof ConstantDouble) {
709                                 doubleConstant = ((ConstantDouble) constantRefOperand).getBytes();
710                             } else if (constantRefOperand instanceof ConstantString) {
711                                 int s = ((ConstantString) constantRefOperand).getStringIndex();
712 
713                                 stringConstantOperand = getStringFromIndex(s);
714                             } else if (constantRefOperand instanceof ConstantInvokeDynamic) {
715                                 ConstantInvokeDynamic id = (ConstantInvokeDynamic) constantRefOperand;
716                                 ConstantNameAndType sig = (ConstantNameAndType) getConstantPool().getConstant(
717                                         id.getNameAndTypeIndex());
718                                 nameConstantOperand = getStringFromIndex(sig.getNameIndex());
719                                 sigConstantOperand = getStringFromIndex(sig.getSignatureIndex());
720                             } else if (constantRefOperand instanceof ConstantCP) {
721                                 ConstantCP cp = (ConstantCP) constantRefOperand;
722                                 ConstantClass clazz = (ConstantClass) getConstantPool().getConstant(cp.getClassIndex());
723                                 classConstantOperand = getStringFromIndex(clazz.getNameIndex());
724                                 referencedClass = DescriptorFactory.createClassDescriptor(classConstantOperand);
725                                 referencedXClass = null;
726                                 ConstantNameAndType sig = (ConstantNameAndType) getConstantPool().getConstant(
727                                         cp.getNameAndTypeIndex());
728                                 nameConstantOperand = getStringFromIndex(sig.getNameIndex());
729                                 sigConstantOperand = getStringFromIndex(sig.getSignatureIndex());
730                                 refConstantOperand = null;
731                             }
732                             break;
733                         case M_R:
734                             registerOperand = v;
735                             break;
736                         case M_UINT:
737                         case M_INT:
738                             intConstant = v;
739                             break;
740                         case M_PAD:
741                             break;
742                         default:
743                             throw new IllegalStateException("Unexpecting meaning " + m);
744                         }
745                     }
746 
747                 }
748                 switch (opcode) {
749                 case Const.IINC:
750                     isRegisterLoad = true;
751                     isRegisterStore = true;
752                     break;
753                 case Const.ILOAD_0:
754                 case Const.ILOAD_1:
755                 case Const.ILOAD_2:
756                 case Const.ILOAD_3:
757                     registerOperand = opcode - Const.ILOAD_0;
758                     isRegisterLoad = true;
759                     break;
760 
761                 case Const.ALOAD_0:
762                 case Const.ALOAD_1:
763                 case Const.ALOAD_2:
764                 case Const.ALOAD_3:
765                     registerOperand = opcode - Const.ALOAD_0;
766                     isRegisterLoad = true;
767                     break;
768 
769                 case Const.FLOAD_0:
770                 case Const.FLOAD_1:
771                 case Const.FLOAD_2:
772                 case Const.FLOAD_3:
773                     registerOperand = opcode - Const.FLOAD_0;
774                     isRegisterLoad = true;
775                     break;
776 
777                 case Const.DLOAD_0:
778                 case Const.DLOAD_1:
779                 case Const.DLOAD_2:
780                 case Const.DLOAD_3:
781                     registerOperand = opcode - Const.DLOAD_0;
782                     isRegisterLoad = true;
783                     break;
784 
785                 case Const.LLOAD_0:
786                 case Const.LLOAD_1:
787                 case Const.LLOAD_2:
788                 case Const.LLOAD_3:
789                     registerOperand = opcode - Const.LLOAD_0;
790                     isRegisterLoad = true;
791                     break;
792                 case Const.ILOAD:
793                 case Const.FLOAD:
794                 case Const.ALOAD:
795                 case Const.LLOAD:
796                 case Const.DLOAD:
797                     isRegisterLoad = true;
798                     break;
799 
800                 case Const.ISTORE_0:
801                 case Const.ISTORE_1:
802                 case Const.ISTORE_2:
803                 case Const.ISTORE_3:
804                     registerOperand = opcode - Const.ISTORE_0;
805                     isRegisterStore = true;
806                     break;
807 
808                 case Const.ASTORE_0:
809                 case Const.ASTORE_1:
810                 case Const.ASTORE_2:
811                 case Const.ASTORE_3:
812                     registerOperand = opcode - Const.ASTORE_0;
813                     isRegisterStore = true;
814                     break;
815 
816                 case Const.FSTORE_0:
817                 case Const.FSTORE_1:
818                 case Const.FSTORE_2:
819                 case Const.FSTORE_3:
820                     registerOperand = opcode - Const.FSTORE_0;
821                     isRegisterStore = true;
822                     break;
823 
824                 case Const.DSTORE_0:
825                 case Const.DSTORE_1:
826                 case Const.DSTORE_2:
827                 case Const.DSTORE_3:
828                     registerOperand = opcode - Const.DSTORE_0;
829                     isRegisterStore = true;
830                     break;
831 
832                 case Const.LSTORE_0:
833                 case Const.LSTORE_1:
834                 case Const.LSTORE_2:
835                 case Const.LSTORE_3:
836                     registerOperand = opcode - Const.LSTORE_0;
837                     isRegisterStore = true;
838                     break;
839                 case Const.ISTORE:
840                 case Const.FSTORE:
841                 case Const.ASTORE:
842                 case Const.LSTORE:
843                 case Const.DSTORE:
844                     isRegisterStore = true;
845                     break;
846                 default:
847                     break;
848                 }
849 
850                 switch (opcode) {
851                 case Const.ILOAD:
852                 case Const.FLOAD:
853                 case Const.ALOAD:
854                 case Const.LLOAD:
855                 case Const.DLOAD:
856                     // registerKind = opcode - ILOAD;
857                     break;
858                 case Const.ISTORE:
859                 case Const.FSTORE:
860                 case Const.ASTORE:
861                 case Const.LSTORE:
862                 case Const.DSTORE:
863                     // registerKind = opcode - ISTORE;
864                     break;
865                 case Const.RET:
866                     // registerKind = R_REF;
867                     break;
868                 case Const.GETSTATIC:
869                 case Const.PUTSTATIC:
870                     refFieldIsStatic = true;
871                     break;
872                 case Const.GETFIELD:
873                 case Const.PUTFIELD:
874                     refFieldIsStatic = false;
875                     break;
876                 default:
877                     break;
878                 }
879 
880                 nextPC = i;
881                 if (beforeOpcode(opcode)) {
882                     sawOpcode(opcode);
883                 }
884                 afterOpcode(opcode);
885 
886                 if (opcode == Const.TABLESWITCH) {
887                     sawInt(switchLow);
888                     sawInt(switchHigh);
889                     //                    int prevOffset = i - PC;
890                     for (int o = 0; o <= switchHigh - switchLow; o++) {
891                         sawBranchTo(switchOffsets[o] + PC);
892                         //                        prevOffset = switchOffsets[o];
893                     }
894                     sawBranchTo(defaultSwitchOffset + PC);
895                 } else if (opcode == Const.LOOKUPSWITCH) {
896                     sawInt(switchOffsets.length);
897                     //                    int prevOffset = i - PC;
898                     for (int o = 0; o < switchOffsets.length; o++) {
899                         sawBranchTo(switchOffsets[o] + PC);
900                         //                        prevOffset = switchOffsets[o];
901                         sawInt(switchLabels[o]);
902                     }
903                     sawBranchTo(defaultSwitchOffset + PC);
904                 } else {
905                     for (int k = 0; k < Const.getOperandTypeCount(opcode); k++) {
906                         int m = MEANING_OF_OPERANDS[opcode][k];
907                         switch (m) {
908                         case M_BR:
909                             sawBranchTo(branchOffset + PC);
910                             break;
911                         case M_CP:
912                             if (constantRefOperand instanceof ConstantInteger) {
913                                 sawInt(intConstant);
914                             } else if (constantRefOperand instanceof ConstantLong) {
915                                 sawLong(longConstant);
916                             } else if (constantRefOperand instanceof ConstantFloat) {
917                                 sawFloat(floatConstant);
918                             } else if (constantRefOperand instanceof ConstantDouble) {
919                                 sawDouble(doubleConstant);
920                             } else if (constantRefOperand instanceof ConstantString) {
921                                 sawString(stringConstantOperand);
922                             } else if (constantRefOperand instanceof ConstantFieldref) {
923                                 sawField();
924                             } else if (constantRefOperand instanceof ConstantMethodref) {
925                                 sawMethod();
926                             } else if (constantRefOperand instanceof ConstantInterfaceMethodref) {
927                                 sawIMethod();
928                             } else if (constantRefOperand instanceof ConstantClass) {
929                                 sawClass();
930                             }
931                             break;
932                         case M_R:
933                             sawRegister(registerOperand);
934                             break;
935                         case M_INT:
936                             sawInt(intConstant);
937                             break;
938                         default:
939                             break;
940                         }
941                     }
942                 }
943             }
944         } catch (IOException e) {
945             AnalysisContext.logError("Error while dismantling bytecode", e);
946             assert false;
947         }
948 
949         try {
950             byteStream.close();
951         } catch (IOException e) {
952             assert false;
953         }
954     }

1006     public void printOpCode(int seen) {
1007         System.out.print("  " + this.getClass().getSimpleName() + ": [" + formatter.format(getPC()) + "]  " + Const.getOpcodeName(seen));
1008         if ((seen == Const.INVOKEVIRTUAL) || (seen == Const.INVOKESPECIAL) || (seen == Const.INVOKEINTERFACE) || (seen == Const.INVOKESTATIC)) {
1009             System.out.print("   " + getClassConstantOperand() + "." + getNameConstantOperand() + " " + getSigConstantOperand());
1010         } else if (seen == Const.LDC || seen == Const.LDC_W || seen == Const.LDC2_W) {
1011             Constant c = getConstantRefOperand();
1012             if (c instanceof ConstantString) {
1013                 System.out.print("   \"" + getStringConstantOperand() + "\"");
1014             } else if (c instanceof ConstantClass) {
1015                 System.out.print("   " + getClassConstantOperand());
1016             } else {
1017                 System.out.print("   " + c);
1018             }
1019         } else if ((seen == Const.ALOAD) || (seen == Const.ASTORE)) {
1020             System.out.print("   " + getRegisterOperand());
1021         } else if ((seen == Const.GOTO) || (seen == Const.GOTO_W) || (seen == Const.IF_ACMPEQ) || (seen == Const.IF_ACMPNE) || (seen == Const.IF_ICMPEQ)
1022                 || (seen == Const.IF_ICMPGE) || (seen == Const.IF_ICMPGT) || (seen == Const.IF_ICMPLE) || (seen == Const.IF_ICMPLT)
1023                 || (seen == Const.IF_ICMPNE) || (seen == Const.IFEQ) || (seen == Const.IFGE) || (seen == Const.IFGT) || (seen == Const.IFLE) || (seen == Const.IFLT)
1024                 || (seen == Const.IFNE) || (seen == Const.IFNONNULL) || (seen == Const.IFNULL)) {
1025             System.out.print("   " + getBranchTarget());
1026         } else if ((seen == Const.NEW) || (seen == Const.INSTANCEOF)) {
1027             System.out.print("   " + getClassConstantOperand());
1028         } else if ((seen == Const.TABLESWITCH) || (seen == Const.LOOKUPSWITCH)) {
1029             System.out.print("    [");
1030             int switchPC = getPC();
1031             int[] offsets = getSwitchOffsets();
1032             for (int offset : offsets) {
1033                 System.out.print((switchPC + offset) + ",");
1034             }
1035             System.out.print((switchPC + getDefaultSwitchOffset()) + "]");
1036         }
1037 
1038         System.out.println();
1039     }

1082     public static boolean areOppositeBranches(int opcode1, int opcode2) {
1083         if (!isBranch(opcode1)) {
1084             throw new IllegalArgumentException(Const.getOpcodeName(opcode1) + " isn't a branch");
1085         }
1086         if (!isBranch(opcode2)) {
1087             throw new IllegalArgumentException(Const.getOpcodeName(opcode2) + " isn't a branch");
1088         }
1089         switch (opcode1) {
1090         case Const.IF_ACMPEQ:
1091         case Const.IF_ACMPNE:
1092         case Const.IF_ICMPEQ:
1093         case Const.IF_ICMPNE:
1094         case Const.IF_ICMPLT:
1095         case Const.IF_ICMPLE:
1096         case Const.IF_ICMPGT:
1097         case Const.IF_ICMPGE:
1098         case Const.IFNE:
1099         case Const.IFEQ:
1100         case Const.IFLT:
1101         case Const.IFLE:
1102         case Const.IFGT:
1103         case Const.IFGE:
1104             return ((opcode1 + 1) ^ 1) == opcode2 + 1;
1105         case Const.IFNONNULL:
1106             return opcode2 == Const.IFNULL;
1107         case Const.IFNULL:
1108             return opcode2 == Const.IFNONNULL;
1109         default:
1110             return false;

1112         }

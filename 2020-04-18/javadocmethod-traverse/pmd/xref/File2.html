4   package net.sourceforge.pmd.maven.stubs;
5   
6   import java.io.File;
7   import java.util.ArrayList;
8   import java.util.List;
9   
10  import org.apache.maven.artifact.Artifact;
11  import org.apache.maven.model.Build;
12  import org.apache.maven.model.Model;
13  import org.apache.maven.model.io.xpp3.MavenXpp3Reader;
14  import org.apache.maven.plugin.testing.stubs.MavenProjectStub;
15  import org.codehaus.plexus.util.ReaderFactory;
16  
17  public class ProjectStub extends MavenProjectStub {
18      /**
19       * Default constructor
20       */
21      public ProjectStub() {
22          MavenXpp3Reader pomReader = new MavenXpp3Reader();
23          Model model;
24          try {
25              model = pomReader.read(ReaderFactory.newXmlReader(new File(getBasedir(), "pom.xml")));
26              setModel(model);
27          } catch (Exception e) {
28              throw new RuntimeException(e);
29          }
30  
31          setGroupId(model.getGroupId());
32          setArtifactId(model.getArtifactId());
33          setVersion(model.getVersion());
34          setName(model.getName());
35          setUrl(model.getUrl());
36          setPackaging(model.getPackaging());
37  
38          Build build = new Build();
39          build.setFinalName(model.getArtifactId());
40          build.setDirectory(getBasedir() + "/target");
41          build.setSourceDirectory(getBasedir() + "/src/main/java");
42          build.setOutputDirectory(getBasedir() + "/target/classes");
43          build.setTestSourceDirectory(getBasedir() + "/src/test/java");
44          build.setTestOutputDirectory(getBasedir() + "/target/test-classes");
45          setBuild(build);
46  
47          List<String> compileSourceRoots = new ArrayList<String>();
48          compileSourceRoots.add(getBasedir() + "/src/main/java");
49          setCompileSourceRoots(compileSourceRoots);
50  
51          List<String> testCompileSourceRoots = new ArrayList<String>();
52          testCompileSourceRoots.add(getBasedir() + "/src/test/java");
53          setTestCompileSourceRoots(testCompileSourceRoots);
54      }
55  
56      /** {@inheritDoc} */
57      public File getBasedir() {
58          return new File(super.getBasedir() + "/target/unit/sample-pmd/");
59      }
60  
61      @Override
62      public List<Artifact> getRuntimeArtifacts() {
63          return new ArrayList<Artifact>();
64      }
65  }

50  /**
51   * Asynchronous wrapper around SSLEngine. send and receive is fully non
52   * blocking. When handshaking is required, a thread is created to perform
53   * the handshake and application level sends do not take place during this time.
54   *
55   * Is implemented using queues and functions operating on the receiving end
56   * of each queue.
57   *
58   * Application writes to:
59   *        ||
60   *        \/
61   *     appOutputQ
62   *        ||
63   *        \/
64   * appOutputQ read by "upperWrite" method which does SSLEngine.wrap
65   * and does async write to PlainHttpConnection
66   *
67   * Reading side is as follows
68   * --------------------------
69   *
70   * "upperRead" method reads off channelInputQ and calls SSLEngine.unwrap and
71   * when decrypted data is returned, it is passed to the user's Consumer<ByteBuffer>
72   *        /\
73   *        ||
74   *     channelInputQ
75   *        /\
76   *        ||
77   * "asyncReceive" method puts buffers into channelInputQ. It is invoked from
78   * OP_READ events from the selector.
79   *
80   * Whenever handshaking is required, the doHandshaking() method is called
81   * which creates a thread to complete the handshake. It takes over the
82   * channelInputQ from upperRead, and puts outgoing packets on channelOutputQ.
83   * Selector events are delivered to asyncReceive and lowerWrite as normal.
84   *
85   * Errors
86   *
87   * Any exception thrown by the engine or channel, causes all Queues to be closed
88   * the channel to be closed, and the error is reported to the user's
89   * Consumer<Throwable>
90   */
91  class AsyncSSLDelegate implements ExceptionallyCloseable, AsyncConnection {

124     AsyncSSLDelegate(HttpConnection lowerOutput, HttpClientImpl client, String[] alpn)
125     {
126         SSLContext context = client.sslContext();
127         engine = context.createSSLEngine();
128         engine.setUseClientMode(true);
129         SSLParameters sslp = client.sslParameters()
130                                    .orElseGet(context::getSupportedSSLParameters);
131         sslParameters = Utils.copySSLParameters(sslp);
132         if (alpn != null) {
133             Log.logSSL("AsyncSSLDelegate: Setting application protocols: " + Arrays.toString(alpn));
134             sslParameters.setApplicationProtocols(alpn);
135         } else {
136             Log.logSSL("AsyncSSLDelegate: no applications set!");
137         }
138         logParams(sslParameters);
139         engine.setSSLParameters(sslParameters);
140         this.lowerOutput = lowerOutput;
141         this.client = client;
142         this.channelInputQ = new Queue<>();
143         this.channelInputQ.registerPutCallback(this::upperRead);
144         this.alpn = alpn;
145     }

184     /**
185      * Attempts to wrap buffers from appOutputQ and place them on the
186      * channelOutputQ for writing. If handshaking is happening, then the
187      * process stalls and last buffers taken off the appOutputQ are put back
188      * into it until handshaking completes.
189      *
190      * This same method is called to try and resume output after a blocking
191      * handshaking operation has completed.
192      */
193     private void upperWrite(ByteBufferReference[] refs, AsyncWriteQueue delayCallback) {
194         // currently delayCallback is not used. Use it when it's needed to execute handshake in another thread.
195         try {
196             ByteBuffer[] buffers = ByteBufferReference.toBuffers(refs);
197             int bytes = Utils.remaining(buffers);
198             while (bytes > 0) {
199                 EngineResult r = wrapBuffers(buffers);
200                 int bytesProduced = r.bytesProduced();
201                 int bytesConsumed = r.bytesConsumed();
202                 bytes -= bytesConsumed;
203                 if (bytesProduced > 0) {
204                     lowerOutput.writeAsync(new ByteBufferReference[]{r.destBuffer});
205                 }
206 
207                 // The code below can be uncommented to shake out
208                 // the implementation by inserting random delays and trigger
209                 // handshake in the SelectorManager thread (upperRead)
210 
211                 // int sleep = random.nextInt(100);
212                 // if (sleep > 20) {
213                 //   Thread.sleep(sleep);
214                 // }
215 
216                 // handshaking is happening or is needed
217                 if (r.handshaking()) {
218                     Log.logTrace("Write: needs handshake");
219                     doHandshakeNow("Write");
220                 }
221             }
222             ByteBufferReference.clear(refs);
223         } catch (Throwable t) {
224             closeExceptionally(t);
225             errorHandler.accept(t);
226         }
227     }

241     private void startHandshake(String tag) {
242         Runnable run = () -> {
243             try {
244                 doHandshakeNow(tag);
245             } catch (Throwable t) {
246                 Log.logTrace("{0}: handshake failed: {1}", tag, t);
247                 closeExceptionally(t);
248                 errorHandler.accept(t);
249             }
250         };
251         client.executor().execute(run);
252     }
253 
254     private void doHandshakeNow(String tag)
255         throws IOException, InterruptedException
256     {
257         handshaker.acquire();
258         try {
259             channelInputQ.disableCallback();
260             lowerOutput.flushAsync();
261             Log.logTrace("{0}: Starting handshake...", tag);
262             doHandshakeImpl();
263             Log.logTrace("{0}: Handshake completed", tag);
264             // don't unblock the channel here, as we aren't sure yet, whether ALPN
265             // negotiation succeeded. Caller will call enableCallback() externally
266         } finally {
267             handshaker.release();
268         }
269     }

275      /**
276      * Executes entire handshake in calling thread.
277      * Returns after handshake is completed or error occurs
278      */
279     private void doHandshakeImpl() throws IOException {
280         engine.beginHandshake();
281         while (true) {
282             SSLEngineResult.HandshakeStatus status = engine.getHandshakeStatus();
283             switch(status) {
284                 case NEED_TASK: {
285                     List<Runnable> tasks = obtainTasks();
286                     for (Runnable task : tasks) {
287                         task.run();
288                     }
289                 } break;
290                 case NEED_WRAP:
291                     handshakeWrapAndSend();
292                     break;
293                 case NEED_UNWRAP: case NEED_UNWRAP_AGAIN:
294                     handshakeReceiveAndUnWrap();
295                     break;
296                 case FINISHED:
297                     return;
298                 case NOT_HANDSHAKING:
299                     return;
300                 default:
301                     throw new InternalError("Unexpected Handshake Status: "
302                                              + status);
303             }
304         }
305     }

380     EngineResult handshakeWrapAndSend() throws IOException {
381         EngineResult r = wrapBuffer(Utils.EMPTY_BYTEBUFFER);
382         if (r.bytesProduced() > 0) {
383             lowerOutput.writeAsync(new ByteBufferReference[]{r.destBuffer});
384             lowerOutput.flushAsync();
385         }
386         return r;
387     }

389     // called during handshaking. It blocks until a complete packet
390     // is available, unwraps it and returns.
391     void handshakeReceiveAndUnWrap() throws IOException {
392         ByteBufferReference ref = channelInputQ.take();
393         while (true) {
394             // block waiting for input
395             EngineResult r = unwrapBuffer(ref.get());
396             SSLEngineResult.Status status = r.status();
397             if (status == BUFFER_UNDERFLOW) {
398                 // wait for another buffer to arrive
399                 ByteBufferReference ref1 = channelInputQ.take();
400                 ref = combine (ref, ref1);
401                 continue;
402             }
403             // OK
404             // theoretically possible we could receive some user data
405             if (r.bytesProduced() > 0) {
406                 asyncReceiver.accept(r.destBuffer);
407             } else {
408                 r.destBuffer.clear();
409             }
410             // it is also possible that a delegated task could be needed
411             // even though they are handled in the calling function
412             if (r.handshakeStatus() == NEED_TASK) {
413                 obtainTasks().stream().forEach((task) -> task.run());
414             }
415 
416             if (!ref.get().hasRemaining()) {
417                 ref.clear();
418                 return;
419             }
420         }
421     }

449     EngineResult wrapBuffers(ByteBuffer[] src) throws SSLException {
450         ByteBufferReference dst = getNetBuffer();
451         while (true) {
452             SSLEngineResult sslResult = engine.wrap(src, dst.get());
453             switch (sslResult.getStatus()) {
454                 case BUFFER_OVERFLOW:
455                     // Shouldn't happen. We allocated buffer with packet size
456                     // get it again if net buffer size was changed
457                     dst = getNetBuffer();
458                     break;
459                 case CLOSED:
460                 case OK:
461                     dst.get().flip();
462                     return new EngineResult(sslResult, dst);
463                 case BUFFER_UNDERFLOW:
464                     // Shouldn't happen.  Doesn't returns when wrap()
465                     // underflow handled externally
466                     return new EngineResult(sslResult);
467                 default:
468                     assert false;
469             }
470         }
471     }
472 
473     EngineResult unwrapBuffer(ByteBuffer srcbuf) throws IOException {
474         ByteBufferReference dst = getAppBuffer();
475         while (true) {
476             SSLEngineResult sslResult = engine.unwrap(srcbuf, dst.get());
477             switch (sslResult.getStatus()) {
478                 case BUFFER_OVERFLOW:
479                     // may happen only if app size buffer was changed.
480                     // get it again if app buffer size changed
481                     dst = getAppBuffer();
482                     break;
483                 case CLOSED:
484                     doClosure();
485                     throw new IOException("Engine closed");
486                 case BUFFER_UNDERFLOW:
487                     dst.clear();
488                     return new EngineResult(sslResult);
489                 case OK:
490                      dst.get().flip();
491                      return new EngineResult(sslResult, dst);
492             }
493         }
494     }

518     public void upperRead() {
519         ByteBufferReference src;
520         ByteBufferReference next = null;
521         synchronized (reader) {
522             try {
523                 src = pollInput();
524                 if (src == null) {
525                     return;
526                 }
527                 while (true) {
528                     EngineResult r = unwrapBuffer(src.get());
529                     switch (r.result.getStatus()) {
530                         case BUFFER_UNDERFLOW:
531                             // Buffer too small. Need to combine with next buf
532                             next = pollInput(next);
533                             if (next == null) {
534                                 // no data available.
535                                 // push buffer back until more data available
536                                 channelInputQ.pushback(src);
537                                 return;
538                             } else {
539                                 src = shift(src, next);
540                                 if (!next.get().hasRemaining()) {
541                                     next.clear();
542                                     next = null;
543                                 }
544                             }
545                             break;
546                         case OK:
547                             // check for any handshaking work
548                             if (r.handshaking()) {
549                                 // handshaking is happening or is needed
550                                 // so we put the buffer back on Q to process again
551                                 // later.
552                                 Log.logTrace("Read: needs handshake");
553                                 channelInputQ.pushback(src);
554                                 startHandshake("Read");
555                                 return;
556                             }
557                             asyncReceiver.accept(r.destBuffer);
558                     }
559                     if (src.get().hasRemaining()) {
560                         continue;
561                     }
562                     src.clear();
563                     src = pollInput(next);
564                     next = null;
565                     if (src == null) {
566                         return;
567                     }
568                 }
569             } catch (Throwable t) {
570                 closeExceptionally(t);
571                 errorHandler.accept(t);
572             }
573         }
574     }

576     ByteBufferReference shift(ByteBufferReference ref1, ByteBufferReference ref2) {
577         ByteBuffer buf1 = ref1.get();
578         if (buf1.capacity() < engine.getSession().getPacketBufferSize()) {
579             ByteBufferReference newRef = getNetBuffer();
580             ByteBuffer newBuf = newRef.get();
581             newBuf.put(buf1);
582             buf1 = newBuf;
583             ref1.clear();
584             ref1 = newRef;
585         } else {
586             buf1.compact();
587         }
588         ByteBuffer buf2 = ref2.get();
589         Utils.copy(buf2, buf1, Math.min(buf1.remaining(), buf2.remaining()));
590         buf1.flip();
591         return ref1;
592     }

595     ByteBufferReference combine(ByteBufferReference ref1, ByteBufferReference ref2) {
596         ByteBuffer buf1 = ref1.get();
597         ByteBuffer buf2 = ref2.get();
598         int avail1 = buf1.capacity() - buf1.remaining();
599         if (buf2.remaining() < avail1) {
600             buf1.compact();
601             buf1.put(buf2);
602             buf1.flip();
603             ref2.clear();
604             return ref1;
605         }
606         int newsize = buf1.remaining() + buf2.remaining();
607         ByteBuffer newbuf = ByteBuffer.allocate(newsize); // getting rid of buffer pools
608         newbuf.put(buf1);
609         newbuf.put(buf2);
610         newbuf.flip();
611         ref1.clear();
612         ref2.clear();
613         return ByteBufferReference.of(newbuf);
614     }

620     static void logParams(SSLParameters p) {
621         if (!Log.ssl()) {
622             return;
623         }
624 
625         if (p == null) {
626             Log.logSSL("SSLParameters: Null params");
627             return;
628         }
629 
630         final StringBuilder sb = new StringBuilder("SSLParameters:");
631         final List<Object> params = new ArrayList<>();
632         if (p.getCipherSuites() != null) {
633             for (String cipher : p.getCipherSuites()) {
634                 sb.append("\n    cipher: {")
635                   .append(params.size()).append("}");
636                 params.add(cipher);
637             }
638         }
639 
640         // SSLParameters.getApplicationProtocols() can't return null
641         // JDK 8 EXCL START
642         for (String approto : p.getApplicationProtocols()) {
643             sb.append("\n    application protocol: {")
644               .append(params.size()).append("}");
645             params.add(approto);
646         }
647         // JDK 8 EXCL END
648 
649         if (p.getProtocols() != null) {
650             for (String protocol : p.getProtocols()) {
651                 sb.append("\n    protocol: {")
652                   .append(params.size()).append("}");
653                 params.add(protocol);
654             }
655         }
656 
657         if (p.getServerNames() != null) {
658             for (SNIServerName sname : p.getServerNames()) {
659                  sb.append("\n    server name: {")
660                   .append(params.size()).append("}");
661                 params.add(sname.toString());
662             }
663         }
664         sb.append('\n');
665 
666         Log.logSSL(sb.toString(), params.toArray());
667     }

96      /**
97       * Spin-waits until s.hasQueuedThread(t) becomes true.
98       */
99      void waitForQueuedThread(PublicSemaphore s, Thread t) {
100         long startTime = System.nanoTime();
101         while (!s.hasQueuedThread(t)) {
102             if (millisElapsedSince(startTime) > LONG_DELAY_MS)
103                 throw new AssertionFailedError("timed out");
104             Thread.yield();
105         }
106         assertTrue(s.hasQueuedThreads());
107         assertTrue(t.isAlive());
108     }

110     /**
111      * Spin-waits until s.hasQueuedThreads() becomes true.
112      */
113     void waitForQueuedThreads(Semaphore s) {
114         long startTime = System.nanoTime();
115         while (!s.hasQueuedThreads()) {
116             if (millisElapsedSince(startTime) > LONG_DELAY_MS)
117                 throw new AssertionFailedError("timed out");
118             Thread.yield();
119         }
120     }

122     enum AcquireMethod {
123         acquire() {
124             void acquire(Semaphore s) throws InterruptedException {
125                 s.acquire();
126             }
127         },
128         acquireN() {
129             void acquire(Semaphore s, int permits) throws InterruptedException {
130                 s.acquire(permits);
131             }
132         },
133         acquireUninterruptibly() {
134             void acquire(Semaphore s) {
135                 s.acquireUninterruptibly();
136             }
137         },
138         acquireUninterruptiblyN() {
139             void acquire(Semaphore s, int permits) {
140                 s.acquireUninterruptibly(permits);
141             }
142         },
143         tryAcquire() {
144             void acquire(Semaphore s) {
145                 assertTrue(s.tryAcquire());
146             }
147         },
148         tryAcquireN() {
149             void acquire(Semaphore s, int permits) {
150                 assertTrue(s.tryAcquire(permits));
151             }
152         },
153         tryAcquireTimed() {
154             void acquire(Semaphore s) throws InterruptedException {
155                 assertTrue(s.tryAcquire(2 * LONG_DELAY_MS, MILLISECONDS));
156             }
157         },
158         tryAcquireTimedN {
159             void acquire(Semaphore s, int permits) throws InterruptedException {
160                 assertTrue(s.tryAcquire(permits, 2 * LONG_DELAY_MS, MILLISECONDS));
161             }
162         };
163 
164         // Intentionally meta-circular
165 
166         /** Acquires 1 permit. */
167         void acquire(Semaphore s) throws InterruptedException {
168             acquire(s, 1);
169         }
170         /** Acquires the given number of permits. */
171         void acquire(Semaphore s, int permits) throws InterruptedException {
172             for (int i = 0; i < permits; i++)
173                 acquire(s);
174         }
175     }

255     public void testInterruptible(boolean fair, final AcquireMethod acquirer) {
256         final PublicSemaphore s = new PublicSemaphore(0, fair);
257         final Semaphore pleaseInterrupt = new Semaphore(0, fair);
258         Thread t = newStartedThread(new CheckedRunnable() {
259             public void realRun() {
260                 // Interrupt before acquire
261                 Thread.currentThread().interrupt();
262                 try {
263                     acquirer.acquire(s);
264                     shouldThrow();
265                 } catch (InterruptedException success) {}
266 
267                 // Interrupt during acquire
268                 try {
269                     acquirer.acquire(s);
270                     shouldThrow();
271                 } catch (InterruptedException success) {}
272 
273                 // Interrupt before acquire(N)
274                 Thread.currentThread().interrupt();
275                 try {
276                     acquirer.acquire(s, 3);
277                     shouldThrow();
278                 } catch (InterruptedException success) {}
279 
280                 pleaseInterrupt.release();
281 
282                 // Interrupt during acquire(N)
283                 try {
284                     acquirer.acquire(s, 3);
285                     shouldThrow();
286                 } catch (InterruptedException success) {}
287             }});
288 
289         waitForQueuedThread(s, t);
290         t.interrupt();
291         await(pleaseInterrupt);
292         waitForQueuedThread(s, t);
293         t.interrupt();
294         awaitTermination(t);
295     }

297     /**
298      * acquireUninterruptibly(), acquireUninterruptibly(N) are
299      * uninterruptible
300      */
301     public void testUninterruptible_acquireUninterruptibly()       { testUninterruptible(false, AcquireMethod.acquireUninterruptibly); }
302     public void testUninterruptible_acquireUninterruptibly_fair()  { testUninterruptible(true,  AcquireMethod.acquireUninterruptibly); }
303     public void testUninterruptible_acquireUninterruptiblyN()      { testUninterruptible(false, AcquireMethod.acquireUninterruptiblyN); }
304     public void testUninterruptible_acquireUninterruptiblyN_fair() { testUninterruptible(true,  AcquireMethod.acquireUninterruptiblyN); }
305     public void testUninterruptible(boolean fair, final AcquireMethod acquirer) {
306         final PublicSemaphore s = new PublicSemaphore(0, fair);
307         final Semaphore pleaseInterrupt = new Semaphore(-1, fair);
308 
309         Thread t1 = newStartedThread(new CheckedRunnable() {
310             public void realRun() throws InterruptedException {
311                 // Interrupt before acquire
312                 pleaseInterrupt.release();
313                 Thread.currentThread().interrupt();
314                 acquirer.acquire(s);
315                 assertTrue(Thread.interrupted());
316             }});
317 
318         Thread t2 = newStartedThread(new CheckedRunnable() {
319             public void realRun() throws InterruptedException {
320                 // Interrupt during acquire
321                 pleaseInterrupt.release();
322                 acquirer.acquire(s);
323                 assertTrue(Thread.interrupted());
324             }});
325 
326         await(pleaseInterrupt);
327         waitForQueuedThread(s, t1);
328         waitForQueuedThread(s, t2);
329         t2.interrupt();
330 
331         assertThreadStaysAlive(t1);
332         assertTrue(t2.isAlive());
333 
334         s.release(2);
335 
336         awaitTermination(t1);
337         awaitTermination(t2);
338     }

341      * hasQueuedThreads reports whether there are waiting threads
342      */
343     public void testHasQueuedThreads()      { testHasQueuedThreads(false); }
344     public void testHasQueuedThreads_fair() { testHasQueuedThreads(true); }
345     public void testHasQueuedThreads(boolean fair) {
346         final PublicSemaphore lock = new PublicSemaphore(1, fair);
347         assertFalse(lock.hasQueuedThreads());
348         lock.acquireUninterruptibly();
349         Thread t1 = newStartedThread(new InterruptedLockRunnable(lock));
350         waitForQueuedThread(lock, t1);
351         assertTrue(lock.hasQueuedThreads());
352         Thread t2 = newStartedThread(new InterruptibleLockRunnable(lock));
353         waitForQueuedThread(lock, t2);
354         assertTrue(lock.hasQueuedThreads());
355         t1.interrupt();
356         awaitTermination(t1);
357         assertTrue(lock.hasQueuedThreads());
358         lock.release();
359         awaitTermination(t2);
360         assertFalse(lock.hasQueuedThreads());
361     }

409         assertTrue(lock.getQueuedThreads().isEmpty());

474     /**
475      * a reserialized semaphore has same number of permits and
476      * fairness, but no queued threads
477      */
478     public void testSerialization()      { testSerialization(false); }
479     public void testSerialization_fair() { testSerialization(true); }
480     public void testSerialization(boolean fair) {
481         try {
482             Semaphore s = new Semaphore(3, fair);
483             s.acquire();
484             s.acquire();
485             s.release();
486 
487             Semaphore clone = serialClone(s);
488             assertEquals(fair, s.isFair());
489             assertEquals(fair, clone.isFair());
490             assertEquals(2, s.availablePermits());
491             assertEquals(2, clone.availablePermits());
492             clone.acquire();
493             clone.acquire();
494             clone.release();
495             assertEquals(2, s.availablePermits());
496             assertEquals(1, clone.availablePermits());
497             assertFalse(s.hasQueuedThreads());
498             assertFalse(clone.hasQueuedThreads());
499         } catch (InterruptedException e) { threadUnexpectedException(e); }
500 
501         {
502             PublicSemaphore s = new PublicSemaphore(0, fair);
503             Thread t = newStartedThread(new InterruptibleLockRunnable(s));
504             // waitForQueuedThreads(s); // suffers from "flicker", so ...
505             waitForQueuedThread(s, t);  // ... we use this instead
506             PublicSemaphore clone = serialClone(s);
507             assertEquals(fair, s.isFair());
508             assertEquals(fair, clone.isFair());
509             assertEquals(0, s.availablePermits());
510             assertEquals(0, clone.availablePermits());
511             assertTrue(s.hasQueuedThreads());
512             assertFalse(clone.hasQueuedThreads());
513             s.release();
514             awaitTermination(t);
515             assertFalse(s.hasQueuedThreads());
516             assertFalse(clone.hasQueuedThreads());
517         }
518     }

584     public void testReleaseAcquireDifferentThreads(boolean fair,
585                                                    final AcquireMethod acquirer) {
586         final Semaphore s = new Semaphore(0, fair);
587         final int rounds = 4;
588         long startTime = System.nanoTime();
589         Thread t = newStartedThread(new CheckedRunnable() {
590             public void realRun() throws InterruptedException {
591                 for (int i = 0; i < rounds; i++) {
592                     assertFalse(s.hasQueuedThreads());
593                     if (i % 2 == 0)
594                         acquirer.acquire(s);
595                     else
596                         acquirer.acquire(s, 3);
597                 }}});
598 
599         for (int i = 0; i < rounds; i++) {
600             while (! (s.availablePermits() == 0 && s.hasQueuedThreads()))
601                 Thread.yield();
602             assertTrue(t.isAlive());
603             if (i % 2 == 0)
604                 s.release();
605             else
606                 s.release(3);
607         }
608         awaitTermination(t);
609         assertEquals(0, s.availablePermits());
610         assertTrue(millisElapsedSince(startTime) < LONG_DELAY_MS);
611     }

613     /**
614      * fair locks are strictly FIFO
615      */
616     public void testFairLocksFifo() {
617         final PublicSemaphore s = new PublicSemaphore(1, true);
618         final CountDownLatch pleaseRelease = new CountDownLatch(1);
619         Thread t1 = newStartedThread(new CheckedRunnable() {
620             public void realRun() throws InterruptedException {
621                 // Will block; permits are available, but not three
622                 s.acquire(3);
623             }});
624 
625         waitForQueuedThread(s, t1);
626 
627         Thread t2 = newStartedThread(new CheckedRunnable() {
628             public void realRun() throws InterruptedException {
629                 // Will fail, even though 1 permit is available
630                 assertFalse(s.tryAcquire(0L, MILLISECONDS));
631                 assertFalse(s.tryAcquire(1, 0L, MILLISECONDS));
632 
633                 // untimed tryAcquire will barge and succeed
634                 assertTrue(s.tryAcquire());
635                 s.release(2);
636                 assertTrue(s.tryAcquire(2));
637                 s.release();
638 
639                 pleaseRelease.countDown();
640                 // Will queue up behind t1, even though 1 permit is available
641                 s.acquire();
642             }});
643 
644         await(pleaseRelease);
645         waitForQueuedThread(s, t2);
646         s.release(2);
647         awaitTermination(t1);
648         assertTrue(t2.isAlive());
649         s.release();
650         awaitTermination(t2);
651     }

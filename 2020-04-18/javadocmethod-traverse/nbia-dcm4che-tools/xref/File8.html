137         }

156     public void ejbPostCreate(Dataset ds, PatientLocal patient)
157     throws CreateException {
158         setPatient(patient);
159         try {
160             setScheduledWorkItemCode(CodeBean.valueOf(codeHome,
161                         ds.getItem(Tags.ScheduledWorkitemCodeSeq)));
162             CodeBean.addCodesTo(codeHome, 
163                         ds.get(Tags.ScheduledProcessingApplicationsCodeSeq),
164                     getScheduledProcessingApplicationsCodes());
165             CodeBean.addCodesTo(codeHome,
166                         ds.get(Tags.ScheduledStationNameCodeSeq),
167                     getScheduledStationNameCodes());
168             CodeBean.addCodesTo(codeHome,
169                         ds.get(Tags.ScheduledStationClassCodeSeq),
170                     getScheduledStationClassCodes());
171             CodeBean.addCodesTo(codeHome,
172                         ds.get(Tags.ScheduledStationGeographicLocationCodeSeq),
173                     getScheduledStationGeographicLocationCodes());
174             updateScheduledHumanPerformers(null,
175                     ds.get(Tags.ScheduledHumanPerformersSeq));
176             updateRefRequests(null, ds.get(Tags.RefRequestSeq));
177             updateRelatedPS(null, ds.get(Tags.RelatedProcedureStepSeq));
178             // TODO Tags.ReplacedProcedureStepSeq not yet defined
179             // updateReplacedPS(ds.get(Tags.ReplacedProcedureStepSeq)
180         } catch (Exception e) {
181             throw new EJBException(e);
182         }
183         LOG.info(prompt("Created UPS[pk="));
184         if (LOG.isDebugEnabled()) {
185             LOG.debug(ds);
186         }
187     }

199     private void updateScheduledHumanPerformers(DcmElement oldPerformers,
200             DcmElement newPerformers)
201             throws CreateException, FinderException {
202         if (!equals(oldPerformers, newPerformers)) {
203             Collection<CodeLocal> c = getScheduledHumanPerformerCodes();
204             c.clear();
205             if (newPerformers != null)
206                 for (int i = 0, n = newPerformers.countItems(); i < n; i++) {
207                     DcmElement codeSq = newPerformers.getItem(i).get(Tags.HumanPerformerCodeSeq);
208                     Dataset code;
209                     if (codeSq != null && (code = codeSq.getItem()) != null)
210                         c.add(CodeBean.valueOf(codeHome, code));
211                 }
212         }
213     }
214     private void updateRefRequests(DcmElement oldRequests,
215             DcmElement newRequests) throws CreateException, RemoveException {
216         if (!equals(oldRequests, newRequests)) {
217             Collection<UPSRequestLocal> c = getRefRequests();
218             for (UPSRequestLocal refRequest :
219                 c.toArray(new UPSRequestLocal[c.size()])) {
220                 refRequest.remove();
221             }
222             if (newRequests != null) {
223                 UPSLocal ups = (UPSLocal) ejbctx.getEJBLocalObject();
224                 for (int i = 0, n = newRequests.countItems(); i < n; i++)
225                     c.add(rqHome.create(newRequests.getItem(i), ups));
226             }
227         }
228     }
229 
230     private void updateRelatedPS(DcmElement oldRelPS, DcmElement newRelPS)
231             throws CreateException, RemoveException {
232         if (!equals(oldRelPS, newRelPS)) {
233             Collection<UPSRelatedPSLocal> c = getRelatedProcedureSteps();
234             for (UPSRelatedPSLocal relatedPS :
235                 c.toArray(new UPSRelatedPSLocal[c.size()])) {
236                 relatedPS.remove();
237             }
238             if (newRelPS != null) {
239                 UPSLocal ups = (UPSLocal) ejbctx.getEJBLocalObject();
240                 for (int i = 0, n = newRelPS.countItems(); i < n; i++)
241                     c.add(relPSHome.create(newRelPS.getItem(i), ups));
242             }
243         }
244     }
245 
246     private void updateReplacedPS(DcmElement oldReplPS, DcmElement newReplPS)
247             throws CreateException, RemoveException {
248         if (!equals(oldReplPS, newReplPS)) {
249             Collection<UPSReplacedPSLocal> c = getReplacedProcedureSteps();
250             for (UPSReplacedPSLocal replacedPS :
251                 c.toArray(new UPSReplacedPSLocal[c.size()])) {
252                 replacedPS.remove();
253             }
254             UPSLocal ups = (UPSLocal) ejbctx.getEJBLocalObject();
255             for (int i = 0, n = newReplPS.countItems(); i < n; i++)
256                 c.add(replPSHome.create(newReplPS.getItem(i), ups));
257         }
258     }

513     /**
514      * @ejb.interface-method
515      */
516     public boolean updateState(int newState) {
517         if (getStateAsInt() == newState)
518             return false;
519         setStateAsInt(newState);
520         Dataset ds = getAttributes();
521         ds.putCS(Tags.UPSState, UPSState.toString(newState));
522         setEncodedAttributes(DatasetUtils.toByteArray(ds,
523                 UIDs.DeflatedExplicitVRLittleEndian));
524         return true;
525     }

527     /**
528      * @ejb.interface-method
529      */
530     public void setAttributes(Dataset ds) {
531         setSopInstanceUID(ds.getString(Tags.SOPInstanceUID));
532         setAdmissionID(ds.getString(Tags.AdmissionID));
533         Dataset issuer = ds.getItem(Tags.IssuerOfAdmissionIDSeq);
534         if (issuer != null) {
535             setIssuerOfAdmissionIDLocalNamespaceEntityID(
536                     issuer.getString(Tags.LocalNamespaceEntityID));
537             setIssuerOfAdmissionIDUniversialEntityID(
538                     issuer.getString(Tags.UniversalEntityID));
539         }
540         setProcedureStepLabel(ds.getString(Tags.ProcedureStepLabel));
541         setWorklistLabel(ds.getString(Tags.WorklistLabel));
542         setStateAsInt(UPSState.toInt(ds.getString(Tags.UPSState)));
543         setPriorityAsInt(Priority.toInt(ds.getString(Tags.SPSPriority)));
544         setScheduledStartDateTime(toTimestamp(ds.getDate(Tags.SPSStartDateAndTime)));
545         setExpectedCompletionDateTime(
546                 toTimestamp(ds.getDate(Tags.ExpectedCompletionDateAndTime)));
547         AttributeFilter filter = AttributeFilter.getExcludePatientAttributeFilter();
548         setEncodedAttributes(DatasetUtils.toByteArray(filter.filter(ds),
549                 filter.getTransferSyntaxUID()));
550     }

552     /**
553      * @ejb.interface-method
554      */
555     public void updateAttributes(Dataset newAttrs) {
556         Dataset ds = getAttributes();
557         try {
558             updateWorkitemCode(ds.getItem(Tags.ScheduledWorkitemCodeSeq),
559                     newAttrs.getItem(Tags.ScheduledWorkitemCodeSeq));
560             updateScheduledProcessingApplicationsCodes(
561                     ds.get(Tags.ScheduledProcessingApplicationsCodeSeq),
562                     newAttrs.get(Tags.ScheduledProcessingApplicationsCodeSeq));
563             updateScheduledStationNameCodes(
564                     ds.get(Tags.ScheduledStationNameCodeSeq),
565                     newAttrs.get(Tags.ScheduledStationNameCodeSeq));
566             updateScheduledStationClassCodes(
567                     ds.get(Tags.ScheduledStationClassCodeSeq),
568                     newAttrs.get(Tags.ScheduledStationClassCodeSeq));
569             updateScheduledStationGeographicLocationCodes(
570                     ds.get(Tags.ScheduledStationGeographicLocationCodeSeq),
571                     newAttrs.get(Tags.ScheduledStationGeographicLocationCodeSeq));
572             updateScheduledHumanPerformers(
573                     ds.get(Tags.ScheduledHumanPerformersSeq),
574                     newAttrs.get(Tags.ScheduledHumanPerformersSeq));
575             updateRefRequests(ds.get(Tags.RefRequestSeq),
576                     newAttrs.get(Tags.RefRequestSeq));
577             updateRelatedPS(ds.get(Tags.RelatedProcedureStepSeq),
578                     newAttrs.get(Tags.RelatedProcedureStepSeq));
579         } catch (Exception e) {
580             throw new EJBException(e);
581         }
582         ds.putAll(newAttrs, Dataset.REPLACE_ITEMS);

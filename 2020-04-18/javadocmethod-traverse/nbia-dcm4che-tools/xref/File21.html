172 	private String getCUIDs(Map map) {
173 		if ( map == null || map.isEmpty() ) return "";
174 		StringBuffer sb = new StringBuffer( map.size() << 5);//StringBuffer initial size: nrOfUIDs x 32
175 		Iterator iter = map.keySet().iterator();
176 		while ( iter.hasNext() ) {
177 			sb.append(iter.next()).append(System.getProperty("line.separator", "\n"));
178 		}
179 		return sb.toString();
180 	}

182 	private Map setCUIDs( String sopCuids ) {
183         StringTokenizer st = new StringTokenizer(sopCuids, "\r\n;");
184         String uid,name;
185         Map map = new TreeMap();
186         int i = 0;
187         char ch;
188         while ( st.hasMoreTokens() ) {
189         	uid = st.nextToken().trim();
190     		name = uid;
191     		ch = uid.charAt(0);
192         	if ( ch >= '0' && ch <= '9' ) {
193         		if ( ! UIDs.isValid(uid) ) 
194         			throw new IllegalArgumentException("UID "+uid+" isn't a valid UID!");
195         	} else {
196         		uid = UIDs.forName( name );
197         	}
198         	map.put(name,uid);
199         }
200 		return map;
201 	}

259 	public Dataset getKeyObject(String iuids) throws RemoteException, FinderException, HomeFactoryException, CreateException {
260     	StringTokenizer st = new StringTokenizer(iuids,",|");
261     	List l = new ArrayList();
262     	while ( st.hasMoreTokens() ) {
263     		l.add(st.nextToken());
264     	}
265     	return getKeyObject(l,null, null);
266     }
267     /**
268      * Creates a new Key Object Selection Dataset.
269      * <p>
270      * The content sequence will contain all items from <code>contentItems</code> and references 
271      * to all instances given in iuids.
272      * <p>
273      * If instances from differet studies (but same patient) are given, the dataset will contain the Identical Document sequence with
274      * references to copies of this Key Object in the approbiate study. (new generated Series and SOP Instance UID).
275      * This Sequence can be used to create the copies of the Key object.
276      * <p>
277      * If instances from different patients are given, an IllegalArgumentException will be thrown.
278      *  
279      * @param iuids List of instances referenced by this Key Object.
280      * @param rootInfo Dataset with Attributes for root node. (e.g. Concept Name code sequence for title)
281      * @param contentItems List of Datasets. each dataset is an item of the Content Sequence.
282      * 
283      * @return The Key Object Selection Dataset.
284      * 
285      * @throws RemoteException
286      * @throws FinderException
287      * @throws HomeFactoryException
288      * @throws CreateException
289      */
290 	public Dataset getKeyObject(Collection iuids, Dataset rootInfo, Collection contentItems) throws RemoteException, FinderException, HomeFactoryException, CreateException{
291     	Collection col = lookupContentManager().getSOPInstanceRefMacros(iuids);
292     	Dataset ds;
293     	if ( ! col.isEmpty() ) {
294     		String studyIUID = ((Dataset)col.iterator().next()).getString(Tags.StudyInstanceUID);
295         	ds = newKeyObject( studyIUID, rootInfo );
296         	addPatInfo( ds, studyIUID );
297         	addContentSequence(ds, contentItems, col);
298         	addReqProcEvidenceSequence(ds,col);
299     		
300     	} else {
301     		ds = dof.newDataset();
302     	}
303     	if ( log.isDebugEnabled() ) {
304     		log.debug("Key Object Dataset:"); log.debug(ds);
305     	}
306     	return ds;
307     }

309     private Dataset newKeyObject(String studyIUID, Dataset rootInfo) {
310     	UIDGenerator uidGenerator = UIDGenerator.getInstance();
311 		Dataset ds = dof.newDataset();
312     	if ( rootInfo != null ) {
313     		ds.putAll(rootInfo);
314     	}
315         ds.putUI(Tags.StudyInstanceUID, studyIUID != null ? studyIUID : uidGenerator.createUID());
316     	String seriesIUID = uidGenerator.createUID();
317         ds.putUI(Tags.SeriesInstanceUID, seriesIUID );
318         ds.putUI(Tags.SOPClassUID, UIDs.KeyObjectSelectionDocument);
319         ds.putUI(Tags.SOPInstanceUID, uidGenerator.createUID());
320         ds.putCS(Tags.Modality, "KO");
321         ds.putIS(Tags.InstanceNumber, 1);
322         ds.putDA(Tags.ContentDate, new Date());
323         ds.putTM(Tags.ContentTime, new Date());
324         ds.putCS(Tags.ValueType, "CONTAINER");
325     	Dataset tmplDS = ds.putSQ(Tags.ContentTemplateSeq).addNewItem();
326     	tmplDS.putCS(Tags.TemplateIdentifier, "2010");
327     	tmplDS.putCS(Tags.MappingResource, "DCMR");
328         ds.putSQ(Tags.RefPPSSeq);//Type 2 Attr
329 	return ds;
330     }

333 	 * @param ds
334 	 * @param col
335 	 */
336 	private void addContentSequence(Dataset ds, Collection conceptNames, Collection refSopInstances) {
337     	DcmElement sq = ds.putSQ(Tags.ContentSeq);
338     	if ( conceptNames != null ) {
339     		for ( Iterator iter = conceptNames.iterator() ; iter.hasNext() ; ) {
340     			sq.addItem( (Dataset) iter.next() );
341     		}
342     	}
343     	Dataset sopInstRef, refSerItem, refSopItem, item;
344     	DcmElement refSerSq,refSopSq;
345     	for ( Iterator iter = refSopInstances.iterator() ; iter.hasNext() ; ) {
346     		sopInstRef = (Dataset) iter.next();
347     		refSerSq = sopInstRef.get(Tags.RefSeriesSeq);
348     		for ( int i = 0,len = refSerSq.countItems() ; i < len ; i++ ) {
349     			refSerItem = refSerSq.getItem(i);
350     			refSopSq = refSerItem.get(Tags.RefSOPSeq);
351     			for ( int j=0,n=refSopSq.countItems() ; j < n ; j++ ) {
352     				refSopItem = refSopSq.getItem(j);
353     				item = sq.addNewItem();
354     				item.putCS(Tags.RelationshipType, "CONTAINS");
355     				item.putCS(Tags.ValueType, getValueType( refSopItem.getString(Tags.RefSOPClassUID)));
356     				Dataset sopDS = item.putSQ(Tags.RefSOPSeq).addNewItem();
357     				sopDS.putUI(Tags.RefSOPClassUID, refSopItem.getString(Tags.RefSOPClassUID));
358     				sopDS.putUI(Tags.RefSOPInstanceUID, refSopItem.getString(Tags.RefSOPInstanceUID));
359     			}
360     		}
361     	}
362 	}

375 	 * @param ds
376 	 * @param col
377 	 * @throws CreateException
378 	 * @throws HomeFactoryException
379 	 * @throws FinderException
380 	 * @throws RemoteException
381 	 */
382 	private void addReqProcEvidenceSequence(Dataset ds, Collection col) throws RemoteException, FinderException, HomeFactoryException, CreateException {
383 		DcmElement sq = ds.putSQ(Tags.CurrentRequestedProcedureEvidenceSeq);
384 		if ( col.isEmpty() ) return;
385 		Iterator iter = col.iterator();
386 		sq.addItem( (Dataset) iter.next() );
387 		if ( iter.hasNext() ) {
388 			DcmElement identicalSq = ds.putSQ(Tags.IdenticalDocumentsSeq);
389 			Dataset studyDS;
390 			String patID = ds.getString(Tags.PatientID);
391 			String patID1;
392 			for (  ; iter.hasNext() ; ) { //loop only if this key object selection contains instances in different studies
393 				studyDS = (Dataset) iter.next();
394 				patID1 = lookupContentManager().getPatientForStudy(studyDS.getString(Tags.StudyInstanceUID)).getString(Tags.PatientID);
395 				if ( ! patID.equals(patID1)) {
396 					throw new IllegalArgumentException("Instances of different patients are not allowed!");
397 				}
398 				sq.addItem( studyDS );//add to ReqProcEvidenceSequence
399 				addSOPInstanceRef(identicalSq, studyDS);//add copy reference to identical document sequence
400 			}
401 		}
402 	}

416 		instDS.putUI( Tags.RefSOPInstanceUID, UIDGenerator.getInstance().createUID() );
417 		instDS.putUI( Tags.RefSOPClassUID, UIDs.KeyObjectSelectionDocument );
418 	}

445                 new String[] { String.class.getName(), Dataset.class.getName() } );

450     protected void writeFile(FileDTO fileDTO, Dataset dataset) throws Exception{
451         File file = FileUtils.toFile(fileDTO.getDirectoryPath(), fileDTO.getFilePath());
452         String tsUID = UIDs.ExplicitVRLittleEndian;
453 
454         boolean deleteFile = true;
455         try{
456             log.info("M-WRITE file: " + file);
457             MessageDigest md = MessageDigest.getInstance("MD5");
458             DigestOutputStream dos = new DigestOutputStream(
459                     new FileOutputStream(file), md);
460             BufferedOutputStream out = 
461                 new BufferedOutputStream(dos, new byte[bufferSize]);
462             try {
463                 DcmEncodeParam encParam = DcmEncodeParam.valueOf(tsUID);
464                 dataset.writeFile(out, encParam);
465             }finally{
466                 out.close();
467             }
468             fileDTO.setFileMd5(md.digest());
469             fileDTO.setFileSize(file.length());
470             fileDTO.setFileTsuid(tsUID);
471             importFile(fileDTO, dataset);
472             deleteFile = false;
473         } finally {
474             if (deleteFile) {
475                 log.info("M-DELETE file:" + file);
476                 if (!file.delete()) {
477                     log.error("Failed to delete " + file);
478                 }
479             }
480         }
481     }

484         server.invoke(storeScpServiceName, "importFile",
485                 new Object[] { fileDTO, dataset, null, Boolean.TRUE },
486                 new String[] { FileDTO.class.getName(), Dataset.class.getName(),
487                         String.class.getName(), boolean.class.getName() });
488     }

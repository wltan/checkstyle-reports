59  /**
60   * Message Splitter that uses an {@link XPathExpression} to split a
61   * {@link Document}, {@link java.io.File} or {@link String} payload into a {@link NodeList}.
62   * The return value will be either Strings or {@link Node}s depending on the
63   * received payload type. Additionally, node types will be converted to
64   * Documents if the 'createDocuments' property is set to <code>true</code>.
65   *
66   * @author Jonas Partner
67   * @author Mark Fisher
68   * @author Artem Bilan
69   * @author Gary Russell
70   */
71  public class XPathMessageSplitter extends AbstractMessageSplitter {

111 	/**
112 	 * Construct an instance based on the provided xpath expression, namespaces and
113 	 * {@link TransformerFactory}.
114 	 * @param expression the xpath expression for splitting.
115 	 * @param namespaces the XML namespace for parsing.
116 	 * @param transformerFactory the {@link TransformerFactory}
117 	 * for parsing and building documents.
118 	 * @since 4.3.19
119 	 */
120 	public XPathMessageSplitter(String expression, Map<String, String> namespaces,
121 			TransformerFactory transformerFactory) {
122 
123 		this(XPathExpressionFactory.createXPathExpression(expression, namespaces), transformerFactory);
124 
125 		XPath xpath = XPathFactory.newInstance().newXPath();
126 		SimpleNamespaceContext namespaceContext = new SimpleNamespaceContext();
127 		namespaceContext.setBindings(namespaces);
128 		xpath.setNamespaceContext(namespaceContext);
129 		try {
130 			this.jaxpExpression = xpath.compile(expression);
131 		}
132 		catch (XPathExpressionException e) {
133 			throw new org.springframework.xml.xpath.XPathParseException(
134 					"Could not compile [" + expression + "] to a XPathExpression: " + e.getMessage(), e);
135 		}
136 	}

200 	@Override
201 	protected void doInit() {
202 		super.doInit();
203 		if (this.returnIterator && this.jaxpExpression == null) {
204 			logger.info("The 'iterator' option isn't available for an external XPathExpression. Will be ignored");
205 			this.returnIterator = false;
206 		}
207 	}
208 
209 	@Override
210 	protected Object splitMessage(Message<?> message) {
211 		try {
212 			Object payload = message.getPayload();
213 			Object result;
214 			if (payload instanceof Node) {
215 				result = splitNode((Node) payload);
216 			}
217 			else {
218 				Document document = this.xmlPayloadConverter.convertToDocument(payload);
219 				Assert.notNull(document, () -> "unsupported payload type [" + payload.getClass().getName() + "]");
220 				result = splitDocument(document);
221 			}
222 			return result;
223 		}
224 		catch (ParserConfigurationException ex) {
225 			throw new MessageConversionException(message, "failed to create DocumentBuilder", ex);
226 		}
227 		catch (Exception ex) {
228 			throw IntegrationUtils.wrapInHandlingExceptionIfNecessary(message,
229 					() -> "Failed to split Message payload in the [" + this + ']', ex);
230 		}
231 	}
232 
233 	@Override
234 	protected int obtainSizeIfPossible(Iterator<?> iterator) {
235 		Iterator<?> theIterator = iterator;
236 
237 		if (iterator instanceof TransformFunctionIterator) {
238 			theIterator = ((TransformFunctionIterator) iterator).delegate;
239 		}
240 
241 		if (theIterator instanceof NodeListIterator) {
242 			return ((NodeListIterator) theIterator).nodeList.getLength();
243 		}
244 
245 		return 0;
246 	}
247 
248 	@SuppressWarnings("unchecked")
249 	private Object splitDocument(Document document) throws ParserConfigurationException, TransformerException {
250 		Object nodes = splitNode(document);
251 		final Transformer transformer;
252 		synchronized (this.transformerFactory) {
253 			transformer = this.transformerFactory.newTransformer();
254 		}
255 		if (this.outputProperties != null) {
256 			transformer.setOutputProperties(this.outputProperties);
257 		}
258 
259 		if (nodes instanceof List) {
260 			List<Node> items = (List<Node>) nodes;
261 			List<String> splitStrings = new ArrayList<>(items.size());
262 			for (Node nodeFromList : items) {
263 				StringResult result = new StringResult();
264 				transformer.transform(new DOMSource(nodeFromList), result);
265 				splitStrings.add(result.toString());
266 			}
267 			return splitStrings;
268 		}
269 		else {
270 			return new TransformFunctionIterator((Iterator<Node>) nodes,
271 					node -> {
272 						StringResult result = new StringResult();
273 						try {
274 							transformer.transform(new DOMSource(node), result);
275 						}
276 						catch (TransformerException e) {
277 							throw new IllegalStateException("failed to create DocumentBuilder", e);
278 						}
279 						return result.toString();
280 					});
281 		}
282 	}
283 
284 	private Object splitNode(Node node) throws ParserConfigurationException {
285 		if (this.returnIterator) {
286 			try {
287 				NodeList nodeList = (NodeList) this.jaxpExpression.evaluate(node, XPathConstants.NODESET);
288 				return new NodeListIterator(nodeList);
289 			}
290 			catch (XPathExpressionException e) {
291 				throw new XPathException("Could not evaluate XPath expression:" + e.getMessage(), e);
292 			}
293 		}
294 		else {
295 			List<Node> nodeList = this.xpathExpression.evaluateAsNodeList(node);
296 			if (this.createDocuments) {
297 				return convertNodesToDocuments(nodeList);
298 			}
299 			return nodeList;
300 		}
301 	}
302 
303 	private List<Node> convertNodesToDocuments(List<Node> nodes) throws ParserConfigurationException {
304 		DocumentBuilder documentBuilder = getNewDocumentBuilder();
305 		List<Node> documents = new ArrayList<>(nodes.size());
306 		for (Node node : nodes) {
307 			Document document = convertNodeToDocument(documentBuilder, node);
308 			documents.add(document);
309 		}
310 		return documents;
311 	}
312 
313 	private Document convertNodeToDocument(DocumentBuilder documentBuilder, Node node) {
314 		Document document = documentBuilder.newDocument();
315 		document.appendChild(document.importNode(node, true));
316 		return document;
317 	}
318 
319 	private DocumentBuilder getNewDocumentBuilder() throws ParserConfigurationException {
320 		synchronized (this.documentBuilderFactoryMonitor) {
321 			return this.documentBuilderFactory.newDocumentBuilder();
322 		}
323 	}
324 
325 	private final class NodeListIterator implements Iterator<Node> {
326 
327 		private final DocumentBuilder documentBuilder;
328 
329 		private final NodeList nodeList;
330 
331 		private int index;
332 
333 		NodeListIterator(NodeList nodeList) throws ParserConfigurationException {
334 			this.nodeList = nodeList;
335 			if (XPathMessageSplitter.this.createDocuments) {
336 				this.documentBuilder = getNewDocumentBuilder();
337 			}
338 			else {
339 				this.documentBuilder = null;
340 			}
341 		}
342 
343 		@Override
344 		public boolean hasNext() {
345 			return this.index < this.nodeList.getLength();
346 		}
347 
348 		@Override
349 		public Node next() {
350 			if (!hasNext()) {
351 				return null;
352 			}
353 
354 			Node node = this.nodeList.item(this.index++);
355 			if (this.documentBuilder != null) {
356 				node = convertNodeToDocument(this.documentBuilder, node);
357 			}
358 			return node;
359 		}
360 
361 		@Override
362 		public void remove() {
363 			throw new UnsupportedOperationException("Operation not supported");
364 		}
365 
366 	}

368 	private static final class TransformFunctionIterator extends FunctionIterator<Node, String> {
369 
370 		private final Iterator<Node> delegate;
371 
372 		TransformFunctionIterator(Iterator<Node> delegate, Function<? super Node, ? extends String> function) {
373 			super(null, delegate, function);
374 			this.delegate = delegate;
375 		}

377 	}

49  /**
50   * Registry of subscribers to a single event bus.
51   *
52   * @author Colin Decker
53   */
54  final class SubscriberRegistry {
55  
56    /**
57     * All registered subscribers, indexed by event type.
58     *
59     * <p>The {@link CopyOnWriteArraySet} values make it easy and relatively lightweight to get an
60     * immutable snapshot of all current subscribers to an event without any locking.
61     */
62    private final ConcurrentMap<Class<?>, CopyOnWriteArraySet<Subscriber>> subscribers =
63        Maps.newConcurrentMap();
64  
65    /** The event bus this registry belongs to. */
66    @Weak private final EventBus bus;
67  
68    SubscriberRegistry(EventBus bus) {
69      this.bus = checkNotNull(bus);
70    }
71  
72    /** Registers all subscriber methods on the given listener object. */
73    void register(Object listener) {
74      Multimap<Class<?>, Subscriber> listenerMethods = findAllSubscribers(listener);
75  
76      for (Entry<Class<?>, Collection<Subscriber>> entry : listenerMethods.asMap().entrySet()) {
77        Class<?> eventType = entry.getKey();
78        Collection<Subscriber> eventMethodsInListener = entry.getValue();
79  
80        CopyOnWriteArraySet<Subscriber> eventSubscribers = subscribers.get(eventType);
81  
82        if (eventSubscribers == null) {
83          CopyOnWriteArraySet<Subscriber> newSet = new CopyOnWriteArraySet<>();
84          eventSubscribers =
85              MoreObjects.firstNonNull(subscribers.putIfAbsent(eventType, newSet), newSet);
86        }
87  
88        eventSubscribers.addAll(eventMethodsInListener);
89      }
90    }
91  
92    /** Unregisters all subscribers on the given listener object. */
93    void unregister(Object listener) {
94      Multimap<Class<?>, Subscriber> listenerMethods = findAllSubscribers(listener);
95  
96      for (Entry<Class<?>, Collection<Subscriber>> entry : listenerMethods.asMap().entrySet()) {
97        Class<?> eventType = entry.getKey();
98        Collection<Subscriber> listenerMethodsForType = entry.getValue();
99  
100       CopyOnWriteArraySet<Subscriber> currentSubscribers = subscribers.get(eventType);
101       if (currentSubscribers == null || !currentSubscribers.removeAll(listenerMethodsForType)) {
102         // if removeAll returns true, all we really know is that at least one subscriber was
103         // removed... however, barring something very strange we can assume that if at least one
104         // subscriber was removed, all subscribers on listener for that event type were... after
105         // all, the definition of subscribers on a particular class is totally static
106         throw new IllegalArgumentException(
107             "missing event subscriber for an annotated method. Is " + listener + " registered?");
108       }
109 
110       // don't try to remove the set if it's empty; that can't be done safely without a lock
111       // anyway, if the set is empty it'll just be wrapping an array of length 0
112     }
113   }
114 
115   @VisibleForTesting
116   Set<Subscriber> getSubscribersForTesting(Class<?> eventType) {
117     return MoreObjects.firstNonNull(subscribers.get(eventType), ImmutableSet.<Subscriber>of());
118   }
119 
120   /**
121    * Gets an iterator representing an immutable snapshot of all subscribers to the given event at
122    * the time this method is called.
123    */
124   Iterator<Subscriber> getSubscribers(Object event) {
125     ImmutableSet<Class<?>> eventTypes = flattenHierarchy(event.getClass());
126 
127     List<Iterator<Subscriber>> subscriberIterators =
128         Lists.newArrayListWithCapacity(eventTypes.size());
129 
130     for (Class<?> eventType : eventTypes) {
131       CopyOnWriteArraySet<Subscriber> eventSubscribers = subscribers.get(eventType);
132       if (eventSubscribers != null) {
133         // eager no-copy snapshot
134         subscriberIterators.add(eventSubscribers.iterator());
135       }
136     }
137 
138     return Iterators.concat(subscriberIterators.iterator());
139   }
140 
141   /**
142    * A thread-safe cache that contains the mapping from each class to all methods in that class and
143    * all super-classes, that are annotated with {@code @Subscribe}. The cache is shared across all
144    * instances of this class; this greatly improves performance if multiple EventBus instances are
145    * created and objects of the same class are registered on all of them.
146    */
147   private static final LoadingCache<Class<?>, ImmutableList<Method>> subscriberMethodsCache =
148       CacheBuilder.newBuilder()
149           .weakKeys()
150           .build(
151               new CacheLoader<Class<?>, ImmutableList<Method>>() {
152                 @Override
153                 public ImmutableList<Method> load(Class<?> concreteClass) throws Exception {
154                   return getAnnotatedMethodsNotCached(concreteClass);
155                 }
156               });
157 
158   /**
159    * Returns all subscribers for the given listener grouped by the type of event they subscribe to.
160    */
161   private Multimap<Class<?>, Subscriber> findAllSubscribers(Object listener) {
162     Multimap<Class<?>, Subscriber> methodsInListener = HashMultimap.create();
163     Class<?> clazz = listener.getClass();
164     for (Method method : getAnnotatedMethods(clazz)) {
165       Class<?>[] parameterTypes = method.getParameterTypes();
166       Class<?> eventType = parameterTypes[0];
167       methodsInListener.put(eventType, Subscriber.create(bus, listener, method));
168     }
169     return methodsInListener;
170   }
171 
172   private static ImmutableList<Method> getAnnotatedMethods(Class<?> clazz) {
173     return subscriberMethodsCache.getUnchecked(clazz);
174   }
175 
176   private static ImmutableList<Method> getAnnotatedMethodsNotCached(Class<?> clazz) {
177     Set<? extends Class<?>> supertypes = TypeToken.of(clazz).getTypes().rawTypes();
178     Map<MethodIdentifier, Method> identifiers = Maps.newHashMap();
179     for (Class<?> supertype : supertypes) {
180       for (Method method : supertype.getDeclaredMethods()) {
181         if (method.isAnnotationPresent(Subscribe.class) && !method.isSynthetic()) {
182           // TODO(cgdecker): Should check for a generic parameter type and error out
183           Class<?>[] parameterTypes = method.getParameterTypes();
184           checkArgument(
185               parameterTypes.length == 1,
186               "Method %s has @Subscribe annotation but has %s parameters."
187                   + "Subscriber methods must have exactly 1 parameter.",
188               method,
189               parameterTypes.length);
190 
191           MethodIdentifier ident = new MethodIdentifier(method);
192           if (!identifiers.containsKey(ident)) {
193             identifiers.put(ident, method);
194           }
195         }
196       }
197     }
198     return ImmutableList.copyOf(identifiers.values());
199   }
200 
201   /** Global cache of classes to their flattened hierarchy of supertypes. */
202   private static final LoadingCache<Class<?>, ImmutableSet<Class<?>>> flattenHierarchyCache =
203       CacheBuilder.newBuilder()
204           .weakKeys()
205           .build(
206               new CacheLoader<Class<?>, ImmutableSet<Class<?>>>() {
207                 // <Class<?>> is actually needed to compile
208                 @SuppressWarnings("RedundantTypeArguments")
209                 @Override
210                 public ImmutableSet<Class<?>> load(Class<?> concreteClass) {
211                   return ImmutableSet.<Class<?>>copyOf(
212                       TypeToken.of(concreteClass).getTypes().rawTypes());
213                 }
214               });
215 
216   /**
217    * Flattens a class's type hierarchy into a set of {@code Class} objects including all
218    * superclasses (transitively) and all interfaces implemented by these superclasses.
219    */
220   @VisibleForTesting
221   static ImmutableSet<Class<?>> flattenHierarchy(Class<?> concreteClass) {
222     try {
223       return flattenHierarchyCache.getUnchecked(concreteClass);
224     } catch (UncheckedExecutionException e) {
225       throw Throwables.propagate(e.getCause());
226     }
227   }
228 
229   private static final class MethodIdentifier {
230 
231     private final String name;
232     private final List<Class<?>> parameterTypes;
233 
234     MethodIdentifier(Method method) {
235       this.name = method.getName();
236       this.parameterTypes = Arrays.asList(method.getParameterTypes());
237     }
238 
239     @Override
240     public int hashCode() {
241       return Objects.hashCode(name, parameterTypes);
242     }
243 
244     @Override
245     public boolean equals(@Nullable Object o) {
246       if (o instanceof MethodIdentifier) {
247         MethodIdentifier ident = (MethodIdentifier) o;
248         return name.equals(ident.name) && parameterTypes.equals(ident.parameterTypes);
249       }
250       return false;
251     }
252   }

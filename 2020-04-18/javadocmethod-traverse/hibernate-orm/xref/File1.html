43  /**
44   * Provides the basic templating of how instrumentation should occur.
45   *
46   * @author Steve Ebersole
47   */
48  public abstract class AbstractInstrumenter implements Instrumenter {
49  	private static final int ZIP_MAGIC = 0x504B0304;
50  	private static final int CLASS_MAGIC = 0xCAFEBABE;
51  	
52  	protected final Logger logger;
53  	protected final Options options;
54  
55  	/**
56  	 * Creates the basic instrumentation strategy.
57  	 *
58  	 * @param logger The bridge to the environment's logging system.
59  	 * @param options User-supplied options.
60  	 */
61  	public AbstractInstrumenter(Logger logger, Options options) {
62  		this.logger = logger;
63  		this.options = options;
64  	}
65  
66  	/**
67  	 * Given the bytecode of a java class, retrieve the descriptor for that class.
68  	 *
69  	 * @param byecode The class bytecode.
70  	 *
71  	 * @return The class's descriptor
72  	 *
73  	 * @throws Exception Indicates problems access the bytecode.
74  	 */
75  	protected abstract ClassDescriptor getClassDescriptor(byte[] byecode) throws Exception;
76  
77  	/**
78  	 * Create class transformer for the class.
79  	 *
80  	 * @param descriptor The descriptor of the class to be instrumented.
81  	 * @param classNames The names of all classes to be instrumented; the "pipeline" if you will.
82  	 *
83  	 * @return The transformer for the given class; may return null to indicate that transformation should
84  	 * be skipped (ala already instrumented).
85  	 */
86  	protected abstract ClassTransformer getClassTransformer(ClassDescriptor descriptor, Set classNames);
87  
88  	/**
89  	 * The main instrumentation entry point.  Given a set of files, perform instrumentation on each discovered class
90  	 * file.
91  	 *
92  	 * @param files The files.
93  	 */
94  	public void execute(Set<File> files) {
95  		Set<String> classNames = new HashSet<String>();
96  
97  		if ( options.performExtendedInstrumentation() ) {
98  			logger.debug( "collecting class names for extended instrumentation determination" );
99  			try {
100 				for ( Object file1 : files ) {
101 					final File file = (File) file1;
102 					collectClassNames( file, classNames );
103 				}
104 			}
105 			catch ( ExecutionException ee ) {
106 				throw ee;
107 			}
108 			catch ( Exception e ) {
109 				throw new ExecutionException( e );
110 			}
111 		}
112 
113 		logger.info( "starting instrumentation" );
114 		try {
115 			for ( File file : files ) {
116 				processFile( file, classNames );
117 			}
118 		}
119 		catch ( ExecutionException ee ) {
120 			throw ee;
121 		}
122 		catch ( Exception e ) {
123 			throw new ExecutionException( e );
124 		}
125 	}
126 
127 	/**
128 	 * Extract the names of classes from file, adding them to the classNames collection.
129 	 * <p/>
130 	 * IMPL NOTE : file here may be either a class file or a jar.  If a jar, all entries in the jar file are
131 	 * processed.
132 	 *
133 	 * @param file The file from which to extract class metadata (descriptor).
134 	 * @param classNames The collected class name collection.
135 	 *
136 	 * @throws Exception indicates problems accessing the file or its contents.
137 	 */
138 	private void collectClassNames(File file, final Set<String> classNames) throws Exception {
139 	    if ( isClassFile( file ) ) {
140 			byte[] bytes = ByteCodeHelper.readByteCode( file );
141 			ClassDescriptor descriptor = getClassDescriptor( bytes );
142 		    classNames.add( descriptor.getName() );
143 	    }
144 	    else if ( isJarFile( file ) ) {
145 		    ZipEntryHandler collector = new ZipEntryHandler() {
146 			    public void handleEntry(ZipEntry entry, byte[] byteCode) throws Exception {
147 					if ( !entry.isDirectory() ) {
148 						// see if the entry represents a class file
149 						DataInputStream din = new DataInputStream( new ByteArrayInputStream( byteCode ) );
150 						if ( din.readInt() == CLASS_MAGIC ) {
151 				            classNames.add( getClassDescriptor( byteCode ).getName() );
152 						}
153 					}
154 			    }
155 		    };
156 			ZipFileProcessor processor = new ZipFileProcessor( collector );
157 		    processor.process( file );
158 	    }
159 	}
160 
161 	/**
162 	 * Does this file represent a compiled class?
163 	 *
164 	 * @param file The file to check.
165 	 *
166 	 * @return True if the file is a class; false otherwise.
167 	 *
168 	 * @throws IOException Indicates problem access the file.
169 	 */
170 	protected final boolean isClassFile(File file) throws IOException {
171         return checkMagic( file, CLASS_MAGIC );
172     }
173 
174 	/**
175 	 * Does this file represent a zip file of some format?
176 	 *
177 	 * @param file The file to check.
178 	 *
179 	 * @return True if the file is n archive; false otherwise.
180 	 *
181 	 * @throws IOException Indicates problem access the file.
182 	 */
183     protected final boolean isJarFile(File file) throws IOException {
184         return checkMagic(file, ZIP_MAGIC);
185     }
186 
187 	protected final boolean checkMagic(File file, long magic) throws IOException {
188         DataInputStream in = new DataInputStream( new FileInputStream( file ) );
189         try {
190             int m = in.readInt();
191             return magic == m;
192         }
193         finally {
194             in.close();
195         }
196     }
197 
198 	/**
199 	 * Actually process the file by applying instrumentation transformations to any classes it contains.
200 	 * <p/>
201 	 * Again, just like with {@link #collectClassNames} this method can handle both class and archive files.
202 	 *
203 	 * @param file The file to process.
204 	 * @param classNames The 'pipeline' of classes to be processed.  Only actually populated when the user
205 	 * specifies to perform {@link org.hibernate.bytecode.buildtime.spi.Instrumenter.Options#performExtendedInstrumentation() extended} instrumentation.
206 	 *
207 	 * @throws Exception Indicates an issue either access files or applying the transformations.
208 	 */
209 	protected void processFile(File file, Set<String> classNames) throws Exception {
210 	    if ( isClassFile( file ) ) {
211 			logger.debug( "processing class file : " + file.getAbsolutePath() );
212 	        processClassFile( file, classNames );
213 	    }
214 	    else if ( isJarFile( file ) ) {
215 			logger.debug( "processing jar file : " + file.getAbsolutePath() );
216 	        processJarFile( file, classNames );
217 	    }
218 	    else {
219 		    logger.debug( "ignoring file : " + file.getAbsolutePath() );
220 	    }
221 	}
222 
223 	/**
224 	 * Process a class file.  Delegated to from {@link #processFile} in the case of a class file.
225 	 *
226 	 * @param file The class file to process.
227 	 * @param classNames The 'pipeline' of classes to be processed.  Only actually populated when the user
228 	 * specifies to perform {@link org.hibernate.bytecode.buildtime.spi.Instrumenter.Options#performExtendedInstrumentation() extended} instrumentation.
229 	 *
230 	 * @throws Exception Indicates an issue either access files or applying the transformations.
231 	 */
232 	protected void processClassFile(File file, Set<String> classNames) throws Exception {
233 		byte[] bytes = ByteCodeHelper.readByteCode( file );
234 		ClassDescriptor descriptor = getClassDescriptor( bytes );
235 		ClassTransformer transformer = getClassTransformer( descriptor, classNames );
236 		if ( transformer == null ) {
237 			logger.debug( "no trasformer for class file : " + file.getAbsolutePath() );
238 			return;
239 		}
240 
241 		logger.info( "processing class : " + descriptor.getName() + ";  file = " + file.getAbsolutePath() );
242 		byte[] transformedBytes = transformer.transform(
243 				getClass().getClassLoader(),
244 				descriptor.getName(),
245 				null,
246 				null,
247 				descriptor.getBytes()
248 		);
249 
250 		OutputStream out = new FileOutputStream( file );
251 		try {
252 			out.write( transformedBytes );
253 			out.flush();
254 		}
255 		finally {
256 			try {
257 				out.close();
258 			}
259 			catch ( IOException ignore) {
260 				// intentionally empty
261 			}
262 		}
263 	}
264 
265 	/**
266 	 * Process an archive file.  Delegated to from {@link #processFile} in the case of an archive file.
267 	 *
268 	 * @param file The archive file to process.
269 	 * @param classNames The 'pipeline' of classes to be processed.  Only actually populated when the user
270 	 * specifies to perform {@link org.hibernate.bytecode.buildtime.spi.Instrumenter.Options#performExtendedInstrumentation() extended} instrumentation.
271 	 *
272 	 * @throws Exception Indicates an issue either access files or applying the transformations.
273 	 */
274 	protected void processJarFile(final File file, final Set<String> classNames) throws Exception {
275         File tempFile = File.createTempFile(
276 		        file.getName(),
277 		        null,
278 		        new File( file.getAbsoluteFile().getParent() )
279         );
280 
281         try {
282 			FileOutputStream fout = new FileOutputStream( tempFile, false );
283 			try {
284 				final ZipOutputStream out = new ZipOutputStream( fout );
285 				ZipEntryHandler transformer = new ZipEntryHandler() {
286 					public void handleEntry(ZipEntry entry, byte[] byteCode) throws Exception {
287 								logger.debug( "starting zip entry : " + entry.toString() );
288 								if ( !entry.isDirectory() ) {
289 									// see if the entry represents a class file
290 									DataInputStream din = new DataInputStream( new ByteArrayInputStream( byteCode ) );
291 									if ( din.readInt() == CLASS_MAGIC ) {
292 										ClassDescriptor descriptor = getClassDescriptor( byteCode );
293 										ClassTransformer transformer = getClassTransformer( descriptor, classNames );
294 										if ( transformer == null ) {
295 											logger.debug( "no transformer for zip entry :  " + entry.toString() );
296 										}
297 										else {
298 											logger.info( "processing class : " + descriptor.getName() + ";  entry = " + file.getAbsolutePath() );
299 											byteCode = transformer.transform(
300 													getClass().getClassLoader(),
301 													descriptor.getName(),
302 													null,
303 													null,
304 													descriptor.getBytes()
305 											);
306 										}
307 									}
308 									else {
309 										logger.debug( "ignoring zip entry : " + entry.toString() );
310 									}
311 								}
312 
313 								ZipEntry outEntry = new ZipEntry( entry.getName() );
314 								outEntry.setMethod( entry.getMethod() );
315 								outEntry.setComment( entry.getComment() );
316 								outEntry.setSize( byteCode.length );
317 
318 								if ( outEntry.getMethod() == ZipEntry.STORED ){
319 									CRC32 crc = new CRC32();
320 									crc.update( byteCode );
321 									outEntry.setCrc( crc.getValue() );
322 									outEntry.setCompressedSize( byteCode.length );
323 								}
324 								out.putNextEntry( outEntry );
325 								out.write( byteCode );
326 								out.closeEntry();
327 					}
328 				};
329 				ZipFileProcessor processor = new ZipFileProcessor( transformer );
330 				processor.process( file );
331 				out.close();
332 			}
333 			finally{
334 				fout.close();
335 			}
336 
337             if ( file.delete() ) {
338 	            File newFile = new File( tempFile.getAbsolutePath() );
339                 if( !newFile.renameTo( file ) ) {
340 	                throw new IOException( "can not rename " + tempFile + " to " + file );
341                 }
342             }
343             else {
344 	            throw new IOException( "can not delete " + file );
345             }
346         }
347         finally {
348 	        if ( ! tempFile.delete() ) {
349 				logger.info( "Unable to cleanup temporary jar file : " + tempFile.getAbsolutePath() );
350 			}
351         }
352 	}

354 	/**
355 	 * Allows control over what exactly to transform.
356 	 */
357 	protected class CustomFieldFilter implements FieldFilter {
358 		private final ClassDescriptor descriptor;
359 		private final Set classNames;
360 
361 		public CustomFieldFilter(ClassDescriptor descriptor, Set classNames) {
362 			this.descriptor = descriptor;
363 			this.classNames = classNames;
364 		}
365 
366 		public boolean shouldInstrumentField(String className, String fieldName) {
367 			if ( descriptor.getName().equals( className ) ) {
368 				logger.trace( "accepting transformation of field [" + className + "." + fieldName + "]" );
369 				return true;
370 			}
371 			else {
372 				logger.trace( "rejecting transformation of field [" + className + "." + fieldName + "]" );
373 				return false;
374 			}
375 		}
376 
377 		public boolean shouldTransformFieldAccess(
378 				String transformingClassName,
379 				String fieldOwnerClassName,
380 				String fieldName) {
381 			if ( descriptor.getName().equals( fieldOwnerClassName ) ) {
382 				logger.trace( "accepting transformation of field access [" + fieldOwnerClassName + "." + fieldName + "]" );
383 				return true;
384 			}
385 			else if ( options.performExtendedInstrumentation() && classNames.contains( fieldOwnerClassName ) ) {
386 				logger.trace( "accepting extended transformation of field access [" + fieldOwnerClassName + "." + fieldName + "]" );
387 				return true;
388 			}
389 			else {
390 				logger.trace( "rejecting transformation of field access [" + fieldOwnerClassName + "." + fieldName + "]; caller = " + transformingClassName  );
391 				return false;
392 			}
393 		}
394 	}

396 	/**
397 	 * General strategy contract for handling entries in an archive file.
398 	 */
399 	private static interface ZipEntryHandler {
400 		/**
401 		 * Apply strategy to the given archive entry.
402 		 *
403 		 * @param entry The archive file entry.
404 		 * @param byteCode The bytes making up the entry
405 		 *
406 		 * @throws Exception Problem handling entry
407 		 */
408 		public void handleEntry(ZipEntry entry, byte[] byteCode) throws Exception;
409 	}
410 
411 	/**
412 	 * Applies {@link ZipEntryHandler} strategies to the entries of an archive file.
413 	 */
414 	private static class ZipFileProcessor {
415 		private final ZipEntryHandler entryHandler;
416 
417 		public ZipFileProcessor(ZipEntryHandler entryHandler) {
418 			this.entryHandler = entryHandler;
419 		}
420 
421 		public void process(File file) throws Exception {
422 			ZipInputStream zip = new ZipInputStream( new FileInputStream( file ) );
423 
424 			try {
425 				ZipEntry entry;
426 				while ( (entry = zip.getNextEntry()) != null ) {
427 					byte bytes[] = ByteCodeHelper.readByteCode( zip );
428 					entryHandler.handleEntry( entry, bytes );
429 					zip.closeEntry();
430 				}
431             }
432             finally {
433 	            zip.close();
434             }
435 		}
436 	}

32  /**
33   * {@link FuncHereContext} extends {@link XPathContext} for supplying context
34   * for the <CODE>here()</CODE> function. The here() function needs to know
35   * <I>where</I> in an XML instance the XPath text string appeared. This can be
36   * in {@link org.w3c.dom.Text}, {@link org.w3c.dom.Attr}ibutes and {@ProcessingInstrinction} nodes. The
37   * correct node must be supplied to the constructor of {@link FuncHereContext}.
38   * The supplied Node MUST contain the XPath which is to be executed.
39   *
40   * <PRE>
41   * From: Scott_Boag\@lotus.com
42   * To: Christian Geuer-Pollmann <maillist\@nue.et-inf.uni-siegen.de>
43   * CC: xalan-dev@xml.apache.org
44   * Subject: Re: Cleanup of XPathContext & definition of XSLTContext
45   * Date: Tue, 21 Aug 2001 18:36:24 -0400
46   *
47   * > My point is to say to get this baby to run, the XPath must have a
48   * > possibility to retrieve the information where itself occured in a
49   * > document.
50   *
51   * It sounds to me like you have to derive an XMLSigContext from the
52   * XPathContext?
53   *
54   * > and supplied the Node which contains the xpath string as "owner". Question:
55   * > Is this the correct use of the owner object? It works, but I don't know
56   * > whether this is correct from the xalan-philosophy...
57   *
58   * Philosophically it's fine.  The owner is the TransformerImpl if XPath is
59   * running under XSLT.  If it is not running under XSLT, it can be whatever
60   * you want.
61   *
62   * -scott
63   * </PRE>
64   *
65   * @author $Author: mullan $
66   * @see com.sun.org.apache.xml.internal.security.transforms.implementations.FuncHere
67   * @see com.sun.org.apache.xml.internal.security.utils.XPathFuncHereAPI
68   * @see <A HREF="http://www.w3.org/Signature/Drafts/xmldsig-core/Overview.html#function-here">XML Signature - The here() function</A>
69   */
70  public class FuncHereContext extends XPathContext {
71  
72     /**
73      * This constuctor is disabled because if we use the here() function we
74      * <I>always</I> need to know in which node the XPath occured.
75      */
76     private FuncHereContext() {}
77  
78     /**
79      * Constructor FuncHereContext
80      *
81      * @param owner
82      */
83     public FuncHereContext(Node owner) {
84        super(owner);
85     }
86  
87     /**
88      * Constructor FuncHereContext
89      *
90      * @param owner
91      * @param xpathContext
92      */
93     public FuncHereContext(Node owner, XPathContext xpathContext) {
94  
95        super(owner);
96  
97        try {
98           super.m_dtmManager = xpathContext.getDTMManager();
99        } catch (IllegalAccessError iae) {
100          throw new IllegalAccessError(I18n.translate("endorsed.jdk1.4.0")
101                                       + " Original message was \""
102                                       + iae.getMessage() + "\"");
103       }
104    }
105 
106    /**
107     * Constructor FuncHereContext
108     *
109     * @param owner
110     * @param previouslyUsed
111     */
112    public FuncHereContext(Node owner, CachedXPathAPI previouslyUsed) {
113 
114       super(owner);
115 
116       try {
117          super.m_dtmManager = previouslyUsed.getXPathContext().getDTMManager();
118       } catch (IllegalAccessError iae) {
119          throw new IllegalAccessError(I18n.translate("endorsed.jdk1.4.0")
120                                       + " Original message was \""
121                                       + iae.getMessage() + "\"");
122       }
123    }
124 
125    /**
126     * Constructor FuncHereContext
127     *
128     * @param owner
129     * @param dtmManager
130     */
131    public FuncHereContext(Node owner, DTMManager dtmManager) {
132 
133       super(owner);
134 
135       try {
136          super.m_dtmManager = dtmManager;
137       } catch (IllegalAccessError iae) {
138          throw new IllegalAccessError(I18n.translate("endorsed.jdk1.4.0")
139                                       + " Original message was \""
140                                       + iae.getMessage() + "\"");
141       }
142    }
143 }

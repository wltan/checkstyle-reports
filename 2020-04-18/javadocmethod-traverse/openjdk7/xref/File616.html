31  package sun.security.krb5.internal.tools;
32  
33  import java.io.File;
34  import sun.security.krb5.*;
35  import sun.security.krb5.internal.*;
36  import sun.security.krb5.internal.ccache.*;
37  import java.io.IOException;
38  import java.util.Arrays;
39  import javax.security.auth.kerberos.KerberosPrincipal;

43  /**
44   * Kinit tool for obtaining Kerberos v5 tickets.
45   *
46   * @author Yanni Zhang
47   * @author Ram Marti
48   */
49  public class Kinit {
50  
51      private KinitOptions options;
52      private static final boolean DEBUG = Krb5.DEBUG;
53  
54      /**
55       * The main method is used to accept user command line input for ticket
56       * request.
57       * <p>
58       * Usage: kinit [-A] [-f] [-p] [-c cachename] [[-k [-t keytab_file_name]]
59       * [principal] [password]
60       * <ul>
61       * <li>    -A        do not include addresses
62       * <li>    -f        forwardable
63       * <li>    -p        proxiable
64       * <li>    -c        cache name (i.e., FILE://c:\temp\mykrb5cc)
65       * <li>    -k        use keytab
66       * <li>    -t        keytab file name
67       * <li>    principal the principal name (i.e., duke@java.sun.com)
68       * <li>    password  the principal's Kerberos password
69       * </ul>
70       * <p>
71       * Use java sun.security.krb5.tools.Kinit -help to bring up help menu.
72       * <p>
73       * We currently support only file-based credentials cache to
74       * store the tickets obtained from the KDC.
75       * By default, for all Unix platforms a cache file named
76       * /tmp/krb5cc_&lt;uid&gt will be generated. The &lt;uid&gt is the
77       * numeric user identifier.
78       * For all other platforms, a cache file named
79       * &lt;USER_HOME&gt/krb5cc_&lt;USER_NAME&gt would be generated.
80       * <p>
81       * &lt;USER_HOME&gt is obtained from <code>java.lang.System</code>
82       * property <i>user.home</i>.
83       * &lt;USER_NAME&gt is obtained from <code>java.lang.System</code>
84       * property <i>user.name</i>.
85       * If &lt;USER_HOME&gt is null the cache file would be stored in
86       * the current directory that the program is running from.
87       * &lt;USER_NAME&gt is operating system's login username.
88       * It could be different from user's principal name.
89       *</p>
90       *<p>
91       * For instance, on Windows NT, it could be
92       * c:\winnt\profiles\duke\krb5cc_duke, in
93       * which duke is the &lt;USER_NAME&gt, and c:\winnt\profile\duke is the
94       * &lt;USER_HOME&gt.
95       *<p>
96       * A single user could have multiple principal names,
97       * but the primary principal of the credentials cache could only be one,
98       * which means one cache file could only store tickets for one
99       * specific user principal. If the user switches
100      * the principal name at the next Kinit, the cache file generated for the
101      * new ticket would overwrite the old cache file by default.
102      * To avoid overwriting, you need to specify
103      * a different cache file name when you request a
104      * new ticket.
105      *</p>
106      *<p>
107      * You can specify the location of the cache file by using the -c option
108      *
109      */
110 
111     public static void main(String[] args) {
112         try {
113             Kinit self = new Kinit(args);
114         }
115         catch (Exception e) {
116             String msg = null;
117             if (e instanceof KrbException) {
118                 msg = ((KrbException)e).krbErrorMessage() + " " +
119                     ((KrbException)e).returnCodeMessage();
120             } else  {
121                 msg = e.getMessage();
122             }
123             if (msg != null) {
124                 System.err.println("Exception: " + msg);
125             } else {
126                 System.out.println("Exception: " + e);
127             }
128             e.printStackTrace();
129             System.exit(-1);
130         }
131         return;
132     }
133 
134     /**
135      * Constructs a new Kinit object.
136      * @param args array of ticket request options.
137      * Avaiable options are: -f, -p, -c, principal, password.
138      * @exception IOException if an I/O error occurs.
139      * @exception RealmException if the Realm could not be instantiated.
140      * @exception KrbException if error occurs during Kerberos operation.
141      */
142     private Kinit(String[] args)
143         throws IOException, RealmException, KrbException {
144         if (args == null || args.length == 0) {
145             options = new KinitOptions();
146         } else {
147             options = new KinitOptions(args);
148         }
149         String princName = null;
150         PrincipalName principal = options.getPrincipal();
151         if (principal != null) {
152             princName = principal.toString();
153         }
154         KrbAsReqBuilder builder;
155         if (DEBUG) {
156             System.out.println("Principal is " + principal);
157         }
158         char[] psswd = options.password;
159         boolean useKeytab = options.useKeytabFile();
160         if (!useKeytab) {
161             if (princName == null) {
162                 throw new IllegalArgumentException
163                     (" Can not obtain principal name");
164             }
165             if (psswd == null) {
166                 System.out.print("Password for " + princName + ":");
167                 System.out.flush();
168                 psswd = Password.readPassword(System.in);
169                 if (DEBUG) {
170                     System.out.println(">>> Kinit console input " +
171                         new String(psswd));
172                 }
173             }
174             builder = new KrbAsReqBuilder(principal, psswd);
175         } else {
176             if (DEBUG) {
177                 System.out.println(">>> Kinit using keytab");
178             }
179             if (princName == null) {
180                 throw new IllegalArgumentException
181                     ("Principal name must be specified.");
182             }
183             String ktabName = options.keytabFileName();
184             if (ktabName != null) {
185                 if (DEBUG) {
186                     System.out.println(
187                                        ">>> Kinit keytab file name: " + ktabName);
188                 }
189             }
190 
191             builder = new KrbAsReqBuilder(principal, ktabName == null
192                     ? KeyTab.getInstance()
193                     : KeyTab.getInstance(new File(ktabName)));
194         }
195 
196         KDCOptions opt = new KDCOptions();
197         setOptions(KDCOptions.FORWARDABLE, options.forwardable, opt);
198         setOptions(KDCOptions.PROXIABLE, options.proxiable, opt);
199         builder.setOptions(opt);
200         String realm = options.getKDCRealm();
201         if (realm == null) {
202             realm = Config.getInstance().getDefaultRealm();
203         }
204 
205         if (DEBUG) {
206             System.out.println(">>> Kinit realm name is " + realm);
207         }
208 
209         PrincipalName sname = new PrincipalName("krbtgt" + "/" + realm,
210                                         PrincipalName.KRB_NT_SRV_INST);
211         sname.setRealm(realm);
212         builder.setTarget(sname);
213 
214         if (DEBUG) {
215             System.out.println(">>> Creating KrbAsReq");
216         }
217 
218         if (options.getAddressOption())
219             builder.setAddresses(HostAddresses.getLocalAddresses());
220 
221         builder.action();
222 
223         sun.security.krb5.internal.ccache.Credentials credentials =
224             builder.getCCreds();
225         builder.destroy();
226 
227         // we always create a new cache and store the ticket we get
228         CredentialsCache cache =
229             CredentialsCache.create(principal, options.cachename);
230         if (cache == null) {
231            throw new IOException("Unable to create the cache file " +
232                                  options.cachename);
233         }
234         cache.update(credentials);
235         cache.save();
236 
237         if (options.password == null) {
238             // Assume we're running interactively
239             System.out.println("New ticket is stored in cache file " +
240                                options.cachename);
241          } else {
242              Arrays.fill(options.password, '0');
243          }
244 
245         // clear the password
246         if (psswd != null) {
247             Arrays.fill(psswd, '0');
248         }
249         options = null; // release reference to options
250     }
251 
252     private static void setOptions(int flag, int option, KDCOptions opt) {
253         switch (option) {
254         case 0:
255             break;
256         case -1:
257             opt.set(flag, false);
258             break;
259         case 1:
260             opt.set(flag, true);
261         }
262     }

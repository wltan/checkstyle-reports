19  package com.opensymphony.xwork2.util;
20  
21  import org.apache.struts2.StrutsException;
22  
23  import java.io.File;
24  import java.io.FileInputStream;
25  import java.io.IOException;
26  import java.net.URI;
27  import java.net.URISyntaxException;
28  import java.net.URL;
29  import java.net.URLClassLoader;
30  import java.util.Collections;
31  import java.util.HashMap;
32  import java.util.Vector;
33  import java.util.zip.ZipEntry;
34  import java.util.zip.ZipInputStream;
35  
36  /**
37   * This class is an utility class that will search through the classpath
38   * for files whose names match the given pattern. The filename is tested
39   * using the given implementation of {@link com.opensymphony.xwork2.util.PatternMatcher} by default it
40   * uses {@link com.opensymphony.xwork2.util.WildcardHelper}
41   */
42  public class ClassPathFinder {
43  
44      /**
45       * The String pattern to test against.
46       */
47      private String pattern;
48  
49      private int[] compiledPattern;
50  
51      /**
52       * The PatternMatcher implementation to use
53       */
54      private PatternMatcher<int[]> patternMatcher = new WildcardHelper();
55  
56      private Vector<String> compared = new Vector<>();
57  
58      /**
59       * @return the pattern in use
60       */
61      public String getPattern() {
62          return pattern;
63      }
64  
65      /**
66       * @param pattern the String pattern for comparing filenames
67       */
68      public void setPattern(String pattern) {
69          this.pattern = pattern;
70      }
71  
72      /**
73       * Builds a {@link java.util.Vector} containing Strings which each name a file
74       * who's name matches the pattern set by setPattern(String). The classpath is
75       * searched recursively, so use with caution.
76       *
77       * @return Vector&lt;String&gt; containing matching filenames
78       */
79      public Vector<String> findMatches() {
80          Vector<String> matches = new Vector<>();
81          URL[] parentUrls = getClassLoaderURLs();
82          compiledPattern = patternMatcher.compilePattern(pattern);
83          for (URL url : parentUrls) {
84              if (!"file".equals(url.getProtocol())) {
85                  continue;
86              }
87              URI entryURI;
88              try {
89                  entryURI = url.toURI();
90              } catch (URISyntaxException e) {
91                  continue;
92              }
93  
94              // debug docker build for JDK 9+
95              if (entryURI.getRawQuery() != null) {
96                  throw new StrutsException("Currently URI with query component isn't supported: " + entryURI.toString());
97              }
98  
99              File entry = new File(entryURI);
100             if (entry.isFile() && entry.toString().endsWith(".jar")) {
101                 try(ZipInputStream zip = new ZipInputStream(new FileInputStream(entry))) {
102                     for (ZipEntry zipEntry = zip.getNextEntry(); zipEntry != null; zipEntry = zip.getNextEntry()) {
103                         boolean doesMatch = patternMatcher.match(new HashMap<String, String>(), zipEntry.getName(), compiledPattern);
104                         if (doesMatch) {
105                             matches.add(zipEntry.getName());
106                         }
107                     }
108                 } catch (IOException e) {
109                     e.printStackTrace();
110                 }
111             } else {
112                 Vector<String> results = checkEntries(entry.list(), entry, "");
113                 if (results != null) {
114                     matches.addAll(results);
115                 }
116             }
117         }
118         return matches;
119     }
120 
121     private Vector<String> checkEntries(String[] entries, File parent, String prefix) {
122 
123         if (entries == null) {
124             return null;
125         }
126 
127         Vector<String> matches = new Vector<>();
128         for (String listEntry : entries) {
129             File tempFile;
130             if (!"".equals(prefix)) {
131                 tempFile = new File(parent, prefix + "/" + listEntry);
132             } else {
133                 tempFile = new File(parent, listEntry);
134             }
135             if (tempFile.isDirectory() &&
136                 !(".".equals(listEntry) || "..".equals(listEntry))) {
137                 if (!"".equals(prefix)) {
138                     matches.addAll(checkEntries(tempFile.list(), parent, prefix + "/" + listEntry));
139                 } else {
140                     matches.addAll(checkEntries(tempFile.list(), parent, listEntry));
141                 }
142             } else {
143 
144                 String entryToCheck;
145                 if ("".equals(prefix)) {
146                     entryToCheck = listEntry;
147                 } else {
148                     entryToCheck = prefix + "/" + listEntry;
149                 }
150 
151                 if (compared.contains(entryToCheck)) {
152                     continue;
153                 } else {
154                     compared.add(entryToCheck);
155                 }
156 
157                 boolean doesMatch = patternMatcher.match(new HashMap<String, String>(), entryToCheck, compiledPattern);
158                 if (doesMatch) {
159                     matches.add(entryToCheck);
160                 }
161             }
162         }
163         return matches;
164     }
165 
166     /**
167      * @param patternMatcher the PatternMatcher implementation to use when comparing filenames
168      */
169     public void setPatternMatcher(PatternMatcher<int[]> patternMatcher) {
170         this.patternMatcher = patternMatcher;
171     }
172 
173     private URL[] getClassLoaderURLs() {
174         URL[] urls;
175         ClassLoader loader = Thread.currentThread().getContextClassLoader();
176 
177         if (!(loader instanceof URLClassLoader)) {
178             loader = ClassPathFinder.class.getClassLoader();
179         }
180 
181         if (loader instanceof URLClassLoader) {
182             urls = ((URLClassLoader) loader).getURLs();
183         } else {    //jdk9 or later
184             try {
185                 urls = Collections.list(loader.getResources("")).toArray(new URL[0]);
186             } catch (IOException e) {
187                 throw new StrutsException("unable to get ClassLoader URLs", e);
188             }
189         }
190 
191         return urls;
192     }
193 }

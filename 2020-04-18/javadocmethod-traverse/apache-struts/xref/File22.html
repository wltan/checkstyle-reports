45  /**
46   * <p>
47   * This class is the default unknown handler for all of the Convention
48   * plugin integration with XWork. This handles instances when a URL doesn't
49   * have an action associated with it but does a viable result and also the
50   * instance where an action returns a result code that isn't already
51   * configured but there is a viable result for the code.
52   * </p>
53   *
54   * <p>
55   * This class also handles all of the index actions using redirects
56   * and actions in nested packages. For example, if there is an action
57   * <strong>/foo/index</strong> and the URL <strong>/foo</strong> is used,
58   * this will render the index action in the /foo namespace.
59   * </p>
60   */
61  public class ConventionUnknownHandler implements UnknownHandler {

78      /**
79       * Constructs the unknown handler.
80       *
81       * @param configuration            The XWork configuration.
82       * @param objectFactory            The XWork object factory used to create result instances.
83       * @param servletContext           The servlet context used to help build the action configurations.
84       * @param container                The Xwork container
85       * @param defaultParentPackageName The default XWork package that the unknown handler will use as
86       *                                 the parent package for new actions and results.
87       * @param redirectToSlash          A boolean parameter that controls whether or not this will handle
88       *                                 unknown actions in the same manner as Apache, Tomcat and other web servers. This
89       *                                 handling will send back a redirect for URLs such as /foo to /foo/ if there doesn't
90       *                                 exist an action that responds to /foo.
91       * @param nameSeparator            The character used as word separator in the action names. "-" by default
92       */
93      @Inject
94      public ConventionUnknownHandler(Configuration configuration, ObjectFactory objectFactory,
95                                      ServletContext servletContext, Container container,
96                                      @Inject(ConventionConstants.CONVENTION_DEFAULT_PARENT_PACKAGE) String defaultParentPackageName,
97                                      @Inject(ConventionConstants.CONVENTION_REDIRECT_TO_SLASH) String redirectToSlash,
98                                      @Inject(ConventionConstants.CONVENTION_ACTION_NAME_SEPARATOR) String nameSeparator) {
99          this.configuration = configuration;
100         this.objectFactory = objectFactory;
101         this.servletContext = servletContext;
102         this.resultMapBuilder = container.getInstance(ResultMapBuilder.class, container.getInstance(String.class, ConventionConstants.CONVENTION_RESULT_MAP_BUILDER));
103         this.conventionsService = container.getInstance(ConventionsService.class, container.getInstance(String.class, ConventionConstants.CONVENTION_CONVENTIONS_SERVICE));
104         this.defaultParentPackageName = defaultParentPackageName;
105         this.nameSeparator = nameSeparator;
106 
107         this.parentPackage = configuration.getPackageConfig(defaultParentPackageName);
108         if (parentPackage == null) {
109             throw new ConfigurationException("Unknown default parent package [" + defaultParentPackageName + "]");
110         }
111 
112         this.redirectToSlash = Boolean.parseBoolean(redirectToSlash);
113 
114         allowedMethods = TextParseUtil.commaDelimitedStringToSet("execute,input,back,cancel,browse,index");
115     }

117     public ActionConfig handleUnknownAction(String namespace, String actionName)
118             throws StrutsException {
119         // Strip the namespace if it is just a slash
120         if (namespace == null || "/".equals(namespace)) {
121             namespace = "";
122         }
123 
124         Map<String, ResultTypeConfig> resultsByExtension = conventionsService.getResultTypesByExtension(parentPackage);
125         String pathPrefix = determinePath(null, namespace);
126         ActionConfig actionConfig = null;
127 
128         // Try /idx/action.jsp if actionName is not empty, otherwise it will just be /.jsp
129         if (!actionName.equals("")) {
130             Resource resource = findResource(resultsByExtension, pathPrefix, actionName);
131             if (resource != null) {
132                 actionConfig = buildActionConfig(resource.path, resultsByExtension.get(resource.ext));
133             }
134         }
135 
136         if (actionConfig == null) {
137             Resource resource = findResource(resultsByExtension, pathPrefix, actionName, "/index");
138 
139             // If the URL is /foo and there is an action we can redirect to, send the redirect to /foo/.
140             // However, if that action is not in the same namespace, it is the default, so I'm not going
141             // to return that.
142             if (!actionName.equals("") && redirectToSlash) {
143                 ResultTypeConfig redirectResultTypeConfig = parentPackage.getAllResultTypeConfigs().get("redirect");
144                 String redirectNamespace = namespace + "/" + actionName;
145                  LOG.trace("Checking if there is an action named index in the namespace {}", redirectNamespace);
146 
147                 actionConfig = configuration.getRuntimeConfiguration().getActionConfig(redirectNamespace, "index");
148                 if (actionConfig != null) {
149                     LOG.trace("Found action config");
150 
151                     PackageConfig packageConfig = configuration.getPackageConfig(actionConfig.getPackageName());
152                     if (redirectNamespace.equals(packageConfig.getNamespace())) {
153                         LOG.trace("Action is not a default - redirecting");
154                         return buildActionConfig(redirectNamespace + "/", redirectResultTypeConfig);
155                     }
156 
157                     LOG.trace("Action was a default - NOT redirecting");
158                 }
159 
160                 if (resource != null) {
161                     return buildActionConfig(redirectNamespace + "/", redirectResultTypeConfig);
162                 }
163             }
164 
165             if (resource != null) {
166                 // Otherwise, if the URL is /foo or /foo/ look for index pages in /foo/
167                 actionConfig = buildActionConfig(resource.path, resultsByExtension.get(resource.ext));
168             }
169 
170             // try to find index action in current namespace or in default one
171             if (actionConfig == null) {
172                 LOG.trace("Looking for action named [index] in namespace [{}] or in default namespace", namespace);
173                 actionConfig = configuration.getRuntimeConfiguration().getActionConfig(namespace, "index");
174             }
175         }
176 
177         return actionConfig;
178     }

180     /**
181      * Finds a resource using the given path parts and all of the extensions in the map.
182      *
183      * @param resultsByExtension Map of extension to result type config objects.
184      * @param parts              The parts of the resource.
185      * @return The resource path or null.
186      */
187     protected Resource findResource(Map<String, ResultTypeConfig> resultsByExtension, String... parts) {
188         for (String ext : resultsByExtension.keySet()) {
189             String canonicalPath = canonicalize(string(parts) + "." + ext);
190             LOG.trace("Checking for {}", canonicalPath);
191 
192             try {
193                 if (servletContext.getResource(canonicalPath) != null) {
194                     return new Resource(canonicalPath, ext);
195                 }
196             } catch (MalformedURLException e) {
197                 LOG.error("Unable to parse path to the web application resource {} skipping...", canonicalPath);
198             }
199         }
200 
201         return null;
202     }
203 
204     protected String canonicalize(final String path) {
205         if (path == null) {
206             return null;
207         }
208 
209         return path.replaceAll("/+", "/");
210     }

212     protected ActionConfig buildActionConfig(String path, ResultTypeConfig resultTypeConfig) {
213         Map<String, ResultConfig> results = new HashMap<>();
214         HashMap<String, String> params = new HashMap<>();
215         if (resultTypeConfig.getParams() != null) {
216             params.putAll(resultTypeConfig.getParams());
217         }
218         params.put(resultTypeConfig.getDefaultResultParam(), path);
219 
220         PackageConfig pkg = configuration.getPackageConfig(defaultParentPackageName);
221         List<InterceptorMapping> interceptors = InterceptorBuilder.constructInterceptorReference(pkg, pkg.getFullDefaultInterceptorRef(), Collections.emptyMap(), null, objectFactory);
222         ResultConfig config = new ResultConfig.Builder(Action.SUCCESS, resultTypeConfig.getClassName()).
223                 addParams(params).build();
224         results.put(Action.SUCCESS, config);
225 
226         return new ActionConfig.Builder(defaultParentPackageName, "execute", ActionSupport.class.getName()).
227                 addInterceptors(interceptors).
228                 addResultConfigs(results).
229                 addAllowedMethod(pkg.getGlobalAllowedMethods()).
230                 build();
231     }

233     private Result scanResultsByExtension(String ns, String actionName, String pathPrefix,
234                                           String resultCode, ActionContext actionContext) {
235         Map<String, ResultTypeConfig> resultsByExtension = conventionsService.getResultTypesByExtension(parentPackage);
236         Result result = null;
237         for (String ext : resultsByExtension.keySet()) {
238             if (LOG.isTraceEnabled()) {
239                 String fqan = ns + "/" + actionName;
240                 LOG.trace("Trying to locate the correct result for the FQ action [{}]"
241                         + " with an file extension of [{}] in the directory [{}] " + "and a result code of [{}]",
242                         fqan, ext, pathPrefix, resultCode);
243             }
244 
245             String path = string(pathPrefix, actionName, nameSeparator, resultCode, ".", ext);
246             result = findResult(path, resultCode, ext, actionContext, resultsByExtension);
247             if (result != null) {
248                 break;
249             }
250 
251             path = string(pathPrefix, actionName, ".", ext);
252             result = findResult(path, resultCode, ext, actionContext, resultsByExtension);
253             if (result != null) {
254                 break;
255             }
256 
257             // Issue #6 - Scan for result-code as page name
258             path = string(pathPrefix, resultCode, ".", ext);
259             result = findResult(path, resultCode, ext, actionContext, resultsByExtension);
260             if (result != null) {
261                 break;
262             }
263 
264         }
265         return result;
266     }

268     public Result handleUnknownResult(ActionContext actionContext, String actionName,
269                                       ActionConfig actionConfig, String resultCode) throws StrutsException {
270 
271         PackageConfig pkg = configuration.getPackageConfig(actionConfig.getPackageName());
272         String ns = pkg.getNamespace();
273         String pathPrefix = determinePath(actionConfig, ns);
274 
275         Result result = scanResultsByExtension(ns, actionName, pathPrefix, resultCode, actionContext);
276 
277         if (result == null) {
278             // Try /idx/action/index.jsp
279             Map<String, ResultTypeConfig> resultsByExtension = conventionsService.getResultTypesByExtension(pkg);
280             for (String ext : resultsByExtension.keySet()) {
281                 if (LOG.isTraceEnabled()) {
282                     String fqan = ns + "/" + actionName;
283                     LOG.trace("Checking for [{}/index.{}]", fqan, ext);
284                 }
285 
286                 String path = string(pathPrefix, actionName, "/index", nameSeparator, resultCode, ".", ext);
287                 result = findResult(path, resultCode, ext, actionContext, resultsByExtension);
288                 if (result != null) {
289                     break;
290                 }
291 
292                 path = string(pathPrefix, actionName, "/index.", ext);
293                 result = findResult(path, resultCode, ext, actionContext, resultsByExtension);
294                 if (result != null) {
295                     break;
296                 }
297             }
298         }
299 
300         if (result == null && resultCode != null) {
301             //try to find an action to chain to. If the source action is "foo" and
302             //the result is "bar", we will try to find an action called "foo-bar"
303             //in the same package
304             String chainedTo = actionName + nameSeparator + resultCode;
305             ActionConfig chainedToConfig = pkg.getActionConfigs().get(chainedTo);
306             if (chainedToConfig != null) {
307                 LOG.trace("Action [{}] used as chain result for [{}] and result [{}]", chainedTo, actionName, resultCode);
308                 ResultTypeConfig chainResultType = pkg.getAllResultTypeConfigs().get("chain");
309                 result = buildResult(chainedTo, resultCode, chainResultType, actionContext);
310             }
311         }
312 
313         return result;
314     }

316     protected Result findResult(String path, String resultCode, String ext, ActionContext actionContext,
317                                 Map<String, ResultTypeConfig> resultsByExtension) {
318         try {
319             LOG.trace("Checking ServletContext for {}", path);
320 
321             URL resource = servletContext.getResource(path);
322             if (resource != null && resource.getPath().endsWith(path)) {
323                 LOG.trace("Found resource {}", resource);
324                 return buildResult(path, resultCode, resultsByExtension.get(ext), actionContext);
325             }
326 
327             LOG.trace("Checking ClassLoader for {}", path);
328 
329             String classLoaderPath = path.startsWith("/") ? path.substring(1, path.length()) : path;
330             resource = ClassLoaderUtil.getResource(classLoaderPath, getClass());
331             if (resource != null && resource.getPath().endsWith(classLoaderPath)) {
332                 LOG.trace("Found resource {}", resource);
333                 return buildResult(path, resultCode, resultsByExtension.get(ext), actionContext);
334             }
335         } catch (MalformedURLException e) {
336             LOG.error("Unable to parse template path: {} skipping...", path);
337         }
338 
339         return null;
340     }

342     protected Result buildResult(String path, String resultCode, ResultTypeConfig config, ActionContext invocationContext) {
343         String resultClass = config.getClassName();
344 
345         Map<String, String> params = new LinkedHashMap<>();
346         if (config.getParams() != null) {
347             params.putAll(config.getParams());
348         }
349         params.put(config.getDefaultResultParam(), path);
350 
351         ResultConfig resultConfig = new ResultConfig.Builder(resultCode, resultClass).addParams(params).build();
352         try {
353             return objectFactory.buildResult(resultConfig, invocationContext.getContextMap());
354         } catch (Exception e) {
355             throw new StrutsException("Unable to build convention result", e, resultConfig);
356         }
357     }

359     protected String string(String... parts) {
360         StringBuilder sb = new StringBuilder();
361         for (String part : parts) {
362             sb.append(part);
363         }
364         return sb.toString();
365     }

367     /**
368      * Determines the result path prefix that the request URL is for, minus the action name. This includes
369      * the base result location and the namespace, with all the slashes handled.
370      *
371      * @param actionConfig (Optional) The might be a ConventionActionConfig, from which we can get the
372      *                     default base result location of that specific action.
373      * @param namespace    The current URL namespace.
374      * @return The path prefix and never null.
375      */
376     protected String determinePath(ActionConfig actionConfig, String namespace) {
377         String finalPrefix = conventionsService.determineResultPath(actionConfig);
378 
379         if (!finalPrefix.endsWith("/")) {
380             finalPrefix += "/";
381         }
382 
383         if (StringUtils.equals(namespace,"/" )) {
384             namespace = "";
385         }
386 
387         if (namespace.length() > 0) {
388             if (namespace.startsWith("/")) {
389                 namespace = namespace.substring(1);
390             }
391 
392             if (!namespace.endsWith("/")) {
393                 namespace += "/";
394             }
395         }
396 
397         return finalPrefix + namespace;
398     }

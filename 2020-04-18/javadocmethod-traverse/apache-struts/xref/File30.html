78      private final void print(String name, String value, PrintWriter w) {
79          if (value != null) {
80              w.print(name + " = {\n\t");
81              w.print(value);
82              w.print("\n}\n");
83          }

86      public String toString() {
87          StringWriter sw = new StringWriter();
88          PrintWriter out = new PrintWriter(sw);
89          print("tlibversion", tlibversion, out);
90          print("jspversion", jspversion, out);
91          print("shortname", shortname, out);
92          print("urn", urn, out);
93          print("info", info, out);
94          print("uri", uri, out);
95          print("tagLibraryValidator", "" + tagLibraryValidator, out);
96  
97          for (int i = 0; i < tags.length; i++)
98              out.println(tags[i].toString());
99  
100         for (int i = 0; i < tagFiles.length; i++)
101             out.println(tagFiles[i].toString());
102 
103         for (int i = 0; i < functions.length; i++)
104             out.println(functions[i].toString());
105 
106         return sw.toString();
107     }

109     // XXX FIXME
110     // resolveRelativeUri and/or getResourceAsStream don't seem to properly
111     // handle relative paths when dealing when home and getDocBase are set
112     // the following is a workaround until these problems are resolved.
113     private InputStream getResourceAsStream(String uri)
114             throws FileNotFoundException {
115         try {
116             // see if file exists on the filesystem first
117             String real = ctxt.getRealPath(uri);
118             if (real == null) {
119                 return ctxt.getResourceAsStream(uri);
120             } else {
121                 return new FileInputStream(real);
122             }
123         } catch (FileNotFoundException ex) {
124             // if file not found on filesystem, get the resource through
125             // the context
126             return ctxt.getResourceAsStream(uri);
127         }
128 
129     }

131     /**
132      * Constructor.

134     public TagLibraryInfoImpl(JspCompilationContext ctxt, ParserController pc, PageInfo pi,
135             String prefix, String uriIn, String[] location, ErrorDispatcher err)
136             throws JasperException {
137         super(prefix, uriIn);
138 
139         this.ctxt = ctxt;
140         this.parserController = pc;
141         this.pi = pi;
142         this.err = err;
143         InputStream in = null;
144         JarFile jarFile = null;
145 
146         if (location == null) {
147             // The URI points to the TLD itself or to a JAR file in which the
148             // TLD is stored
149             location = generateTLDLocation(uri, ctxt);
150         }
151 
152         try {
153             if (!location[0].endsWith("jar")) {
154                 // Location points to TLD file
155                 try {
156                     in = getResourceAsStream(location[0]);
157                     if (in == null) {
158                         throw new FileNotFoundException(location[0]);
159                     }
160                 } catch (FileNotFoundException ex) {
161                     err.jspError("jsp.error.file.not.found", location[0]);
162                 }
163 
164                 parseTLD(ctxt, location[0], in, null);
165                 // Add TLD to dependency list
166                 PageInfo pageInfo = ctxt.createCompiler().getPageInfo();
167                 if (pageInfo != null) {
168                     pageInfo.addDependant(location[0]);
169                 }
170             } else {
171                 // Tag library is packaged in JAR file
172                 try {
173                     URL jarFileUrl = new URL("jar:" + location[0] + "!/");
174                     JarURLConnection conn = (JarURLConnection) jarFileUrl
175                             .openConnection();
176                     conn.setUseCaches(false);
177                     conn.connect();
178                     jarFile = conn.getJarFile();
179                     ZipEntry jarEntry = jarFile.getEntry(location[1]);
180                     in = jarFile.getInputStream(jarEntry);
181                     parseTLD(ctxt, location[0], in, jarFileUrl);
182                 } catch (Exception ex) {
183                     err.jspError("jsp.error.tld.unable_to_read", location[0],
184                             location[1], ex.toString());
185                 }
186             }
187         } finally {
188             if (in != null) {
189                 try {
190                     in.close();
191                 } catch (Throwable t) {
192                 }
193             }
194             if (jarFile != null) {
195                 try {
196                     jarFile.close();
197                 } catch (Throwable t) {
198                 }
199             }
200         }
201 
202     }

209     /*
210      * @param ctxt The JSP compilation context @param uri The TLD's uri @param
211      * in The TLD's input stream @param jarFileUrl The JAR file containing the
212      * TLD, or null if the tag library is not packaged in a JAR
213      */
214     private void parseTLD(JspCompilationContext ctxt, String uri,
215             InputStream in, URL jarFileUrl) throws JasperException {
216         Vector tagVector = new Vector();
217         Vector tagFileVector = new Vector();
218         Hashtable functionTable = new Hashtable();
219 
220         // Create an iterator over the child elements of our <taglib> element
221         ParserUtils pu = new ParserUtils();
222         TreeNode tld = pu.parseXMLDocument(uri, in);
223 
224         // Check to see if the <taglib> root element contains a 'version'
225         // attribute, which was added in JSP 2.0 to replace the <jsp-version>
226         // subelement
227         this.jspversion = tld.findAttribute("version");
228 
229         // Process each child element of our <taglib> element
230         Iterator list = tld.findChildren();
231 
232         while (list.hasNext()) {
233             TreeNode element = (TreeNode) list.next();
234             String tname = element.getName();
235 
236             if ("tlibversion".equals(tname) // JSP 1.1
237                     || "tlib-version".equals(tname)) { // JSP 1.2
238                 this.tlibversion = element.getBody();
239             } else if ("jspversion".equals(tname)
240                     || "jsp-version".equals(tname)) {
241                 this.jspversion = element.getBody();
242             } else if ("shortname".equals(tname) || "short-name".equals(tname))
243                 this.shortname = element.getBody();
244             else if ("uri".equals(tname))
245                 this.urn = element.getBody();
246             else if ("info".equals(tname) || "description".equals(tname))
247                 this.info = element.getBody();
248             else if ("validator".equals(tname))
249                 this.tagLibraryValidator = createValidator(element);
250             else if ("tag".equals(tname))
251                 tagVector.addElement(createTagInfo(element, jspversion));
252             else if ("tag-file".equals(tname)) {
253                 TagFileInfo tagFileInfo = createTagFileInfo(element, uri,
254                         jarFileUrl);
255                 tagFileVector.addElement(tagFileInfo);
256             } else if ("function".equals(tname)) { // JSP2.0
257                 FunctionInfo funcInfo = createFunctionInfo(element);
258                 String funcName = funcInfo.getName();
259                 if (functionTable.containsKey(funcName)) {
260                     err.jspError("jsp.error.tld.fn.duplicate.name", funcName,
261                             uri);
262 
263                 }
264                 functionTable.put(funcName, funcInfo);
265             } else if ("display-name".equals(tname) || // Ignored elements
266                     "small-icon".equals(tname) || "large-icon".equals(tname)
267                     || "listener".equals(tname)) {
268                 ;
269             } else if ("taglib-extension".equals(tname)) {
270                 // Recognized but ignored
271             } else {
272                 if (log.isWarnEnabled()) {
273                     log.warn(Localizer.getMessage(
274                             "jsp.warning.unknown.element.in.taglib", tname));
275                 }
276             }
277 
278         }
279 
280         if (tlibversion == null) {
281             err.jspError("jsp.error.tld.mandatory.element.missing",
282                     "tlib-version");
283         }
284         if (jspversion == null) {
285             err.jspError("jsp.error.tld.mandatory.element.missing",
286                     "jsp-version");
287         }
288 
289         this.tags = new TagInfo[tagVector.size()];
290         tagVector.copyInto(this.tags);
291 
292         this.tagFiles = new TagFileInfo[tagFileVector.size()];
293         tagFileVector.copyInto(this.tagFiles);
294 
295         this.functions = new FunctionInfo[functionTable.size()];
296         int i = 0;
297         Enumeration enumeration = functionTable.elements();
298         while (enumeration.hasMoreElements()) {
299             this.functions[i++] = (FunctionInfo) enumeration.nextElement();
300         }
301     }

303     /*
304      * @param uri The uri of the TLD @param ctxt The compilation context
305      * 
306      * @return String array whose first element denotes the path to the TLD. If
307      * the path to the TLD points to a jar file, then the second element denotes
308      * the name of the TLD entry in the jar file, which is hardcoded to
309      * META-INF/taglib.tld.
310      */
311     private String[] generateTLDLocation(String uri, JspCompilationContext ctxt)
312             throws JasperException {
313 
314         int uriType = TldLocationsCache.uriType(uri);
315         if (uriType == TldLocationsCache.ABS_URI) {
316             err.jspError("jsp.error.taglibDirective.absUriCannotBeResolved",
317                     uri);
318         } else if (uriType == TldLocationsCache.NOROOT_REL_URI) {
319             uri = ctxt.resolveRelativeUri(uri);
320         }
321 
322         String[] location = new String[2];
323         location[0] = uri;
324         if (location[0].endsWith("jar")) {
325             URL url = null;
326             try {
327                 url = ctxt.getResource(location[0]);
328             } catch (Exception ex) {
329                 err.jspError("jsp.error.tld.unable_to_get_jar", location[0], ex
330                         .toString());
331             }
332             if (url == null) {
333                 err.jspError("jsp.error.tld.missing_jar", location[0]);
334             }
335             location[0] = url.toString();
336             location[1] = "META-INF/taglib.tld";
337         }
338 
339         return location;
340     }

342     private TagInfo createTagInfo(TreeNode elem, String jspVersion)
343             throws JasperException {
344 
345         String tagName = null;
346         String tagClassName = null;
347         String teiClassName = null;
348 
349         /*
350          * Default body content for JSP 1.2 tag handlers (<body-content> has
351          * become mandatory in JSP 2.0, because the default would be invalid for
352          * simple tag handlers)
353          */
354         String bodycontent = "JSP";
355 
356         String info = null;
357         String displayName = null;
358         String smallIcon = null;
359         String largeIcon = null;
360         boolean dynamicAttributes = false;
361 
362         Vector attributeVector = new Vector();
363         Vector variableVector = new Vector();
364         Iterator list = elem.findChildren();
365         while (list.hasNext()) {
366             TreeNode element = (TreeNode) list.next();
367             String tname = element.getName();
368 
369             if ("name".equals(tname)) {
370                 tagName = element.getBody();
371             } else if ("tagclass".equals(tname) || "tag-class".equals(tname)) {
372                 tagClassName = element.getBody();
373             } else if ("teiclass".equals(tname) || "tei-class".equals(tname)) {
374                 teiClassName = element.getBody();
375             } else if ("bodycontent".equals(tname)
376                     || "body-content".equals(tname)) {
377                 bodycontent = element.getBody();
378             } else if ("display-name".equals(tname)) {
379                 displayName = element.getBody();
380             } else if ("small-icon".equals(tname)) {
381                 smallIcon = element.getBody();
382             } else if ("large-icon".equals(tname)) {
383                 largeIcon = element.getBody();
384             } else if ("icon".equals(tname)) {
385                 TreeNode icon = element.findChild("small-icon");
386                 if (icon != null) {
387                     smallIcon = icon.getBody();
388                 }
389                 icon = element.findChild("large-icon");
390                 if (icon != null) {
391                     largeIcon = icon.getBody();
392                 }
393             } else if ("info".equals(tname) || "description".equals(tname)) {
394                 info = element.getBody();
395             } else if ("variable".equals(tname)) {
396                 variableVector.addElement(createVariable(element));
397             } else if ("attribute".equals(tname)) {
398                 attributeVector
399                         .addElement(createAttribute(element, jspVersion));
400             } else if ("dynamic-attributes".equals(tname)) {
401                 dynamicAttributes = JspUtil.booleanValue(element.getBody());
402             } else if ("example".equals(tname)) {
403                 // Ignored elements
404             } else if ("tag-extension".equals(tname)) {
405                 // Ignored
406             } else {
407                 if (log.isWarnEnabled()) {
408                     log.warn(Localizer.getMessage(
409                             "jsp.warning.unknown.element.in.tag", tname));
410                 }
411             }
412         }
413 
414         TagExtraInfo tei = null;
415         if (teiClassName != null && !teiClassName.equals("")) {
416             try {
417                 Class teiClass = ctxt.getClassLoader().loadClass(teiClassName);
418                 tei = (TagExtraInfo) teiClass.newInstance();
419             } catch (Exception e) {
420                 err.jspError("jsp.error.teiclass.instantiation", teiClassName,
421                         e);
422             }
423         }
424 
425         TagAttributeInfo[] tagAttributeInfo = new TagAttributeInfo[attributeVector
426                 .size()];
427         attributeVector.copyInto(tagAttributeInfo);
428 
429         TagVariableInfo[] tagVariableInfos = new TagVariableInfo[variableVector
430                 .size()];
431         variableVector.copyInto(tagVariableInfos);
432 
433         TagInfo taginfo = new TagInfo(tagName, tagClassName, bodycontent, info,
434                 this, tei, tagAttributeInfo, displayName, smallIcon, largeIcon,
435                 tagVariableInfos, dynamicAttributes);
436         return taginfo;
437     }

439     /*
440      * Parses the tag file directives of the given TagFile and turns them into a
441      * TagInfo.
442      * 
443      * @param elem The <tag-file> element in the TLD @param uri The location of
444      * the TLD, in case the tag file is specified relative to it @param jarFile
445      * The JAR file, in case the tag file is packaged in a JAR
446      * 
447      * @return TagInfo correspoding to tag file directives
448      */
449     private TagFileInfo createTagFileInfo(TreeNode elem, String uri,
450             URL jarFileUrl) throws JasperException {
451 
452         String name = null;
453         String path = null;
454 
455         Iterator list = elem.findChildren();
456         while (list.hasNext()) {
457             TreeNode child = (TreeNode) list.next();
458             String tname = child.getName();
459             if ("name".equals(tname)) {
460                 name = child.getBody();
461             } else if ("path".equals(tname)) {
462                 path = child.getBody();
463             } else if ("example".equals(tname)) {
464                 // Ignore <example> element: Bugzilla 33538
465             } else if ("tag-extension".equals(tname)) {
466                 // Ignore <tag-extension> element: Bugzilla 33538
467             } else if ("icon".equals(tname) 
468                     || "display-name".equals(tname) 
469                     || "description".equals(tname)) {
470                 // Ignore these elements: Bugzilla 38015
471             } else {
472                 if (log.isWarnEnabled()) {
473                     log.warn(Localizer.getMessage(
474                             "jsp.warning.unknown.element.in.tagfile", tname));
475                 }
476             }
477         }
478 
479         if (path.startsWith("/META-INF/tags")) {
480             // Tag file packaged in JAR
481             // See https://issues.apache.org/bugzilla/show_bug.cgi?id=46471
482             // This needs to be removed once all the broken code that depends on
483             // it has been removed
484             ctxt.setTagFileJarUrl(path, jarFileUrl);
485         } else if (!path.startsWith("/WEB-INF/tags")) {
486             err.jspError("jsp.error.tagfile.illegalPath", path);
487         }
488 
489         TagInfo tagInfo = TagFileProcessor.parseTagFileDirectives(
490                 parserController, name, path, jarFileUrl, this);
491         return new TagFileInfo(name, path, tagInfo);
492     }

494     TagAttributeInfo createAttribute(TreeNode elem, String jspVersion) {
495         String name = null;
496         String type = null;
497         String expectedType = null;
498         String methodSignature = null;
499         boolean required = false, rtexprvalue = false, reqTime = false, isFragment = false, deferredValue = false, deferredMethod = false;
500 
501         Iterator list = elem.findChildren();
502         while (list.hasNext()) {
503             TreeNode element = (TreeNode) list.next();
504             String tname = element.getName();
505 
506             if ("name".equals(tname)) {
507                 name = element.getBody();
508             } else if ("required".equals(tname)) {
509                 String s = element.getBody();
510                 if (s != null)
511                     required = JspUtil.booleanValue(s);
512             } else if ("rtexprvalue".equals(tname)) {
513                 String s = element.getBody();
514                 if (s != null)
515                     rtexprvalue = JspUtil.booleanValue(s);
516             } else if ("type".equals(tname)) {
517                 type = element.getBody();
518                 if ("1.2".equals(jspVersion)
519                         && (type.equals("Boolean") || type.equals("Byte")
520                                 || type.equals("Character")
521                                 || type.equals("Double")
522                                 || type.equals("Float")
523                                 || type.equals("Integer")
524                                 || type.equals("Long") || type.equals("Object")
525                                 || type.equals("Short") || type
526                                 .equals("String"))) {
527                     type = "java.lang." + type;
528                 }
529             } else if ("fragment".equals(tname)) {
530                 String s = element.getBody();
531                 if (s != null) {
532                     isFragment = JspUtil.booleanValue(s);
533                 }
534             } else if ("deferred-value".equals(tname)) {
535                 deferredValue = true;
536                 type = "javax.el.ValueExpression";
537                 TreeNode child = element.findChild("type");
538                 if (child != null) {
539                     expectedType = child.getBody();
540                     if (expectedType != null) {
541                         expectedType = expectedType.trim();
542                     }
543                 } else {
544                     expectedType = "java.lang.Object";
545                 }
546             } else if ("deferred-method".equals(tname)) {
547                 deferredMethod = true;
548                 type = "javax.el.MethodExpression";
549                 TreeNode child = element.findChild("method-signature");
550                 if (child != null) {
551                     methodSignature = child.getBody();
552                     if (methodSignature != null) {
553                         methodSignature = methodSignature.trim();
554                     }
555                 } else {
556                     methodSignature = "java.lang.Object method()";
557                 }
558             } else if ("description".equals(tname) || // Ignored elements
559             false) {
560                 ;
561             } else {
562                 if (log.isWarnEnabled()) {
563                     log.warn(Localizer.getMessage(
564                             "jsp.warning.unknown.element.in.attribute", tname));
565                 }
566             }
567         }
568 
569         if (isFragment) {
570             /*
571              * According to JSP.C-3 ("TLD Schema Element Structure - tag"),
572              * 'type' and 'rtexprvalue' must not be specified if 'fragment' has
573              * been specified (this will be enforced by validating parser).
574              * Also, if 'fragment' is TRUE, 'type' is fixed at
575              * javax.servlet.jsp.tagext.JspFragment, and 'rtexprvalue' is fixed
576              * at true. See also JSP.8.5.2.
577              */
578             type = "javax.servlet.jsp.tagext.JspFragment";
579             rtexprvalue = true;
580         }
581 
582         if (!rtexprvalue && type == null) {
583             // According to JSP spec, for static values (those determined at
584             // translation time) the type is fixed at java.lang.String.
585             type = "java.lang.String";
586         }
587         
588         return new TagAttributeInfo(name, required, type, rtexprvalue,
589                 isFragment, null, deferredValue, deferredMethod, expectedType,
590                 methodSignature);
591     }

593     TagVariableInfo createVariable(TreeNode elem) {
594         String nameGiven = null;
595         String nameFromAttribute = null;
596         String className = "java.lang.String";
597         boolean declare = true;
598         int scope = VariableInfo.NESTED;
599 
600         Iterator list = elem.findChildren();
601         while (list.hasNext()) {
602             TreeNode element = (TreeNode) list.next();
603             String tname = element.getName();
604             if ("name-given".equals(tname))
605                 nameGiven = element.getBody();
606             else if ("name-from-attribute".equals(tname))
607                 nameFromAttribute = element.getBody();
608             else if ("variable-class".equals(tname))
609                 className = element.getBody();
610             else if ("declare".equals(tname)) {
611                 String s = element.getBody();
612                 if (s != null)
613                     declare = JspUtil.booleanValue(s);
614             } else if ("scope".equals(tname)) {
615                 String s = element.getBody();
616                 if (s != null) {
617                     if ("NESTED".equals(s)) {
618                         scope = VariableInfo.NESTED;
619                     } else if ("AT_BEGIN".equals(s)) {
620                         scope = VariableInfo.AT_BEGIN;
621                     } else if ("AT_END".equals(s)) {
622                         scope = VariableInfo.AT_END;
623                     }
624                 }
625             } else if ("description".equals(tname) || // Ignored elements
626             false) {
627             } else {
628                 if (log.isWarnEnabled()) {
629                     log.warn(Localizer.getMessage(
630                             "jsp.warning.unknown.element.in.variable", tname));
631                 }
632             }
633         }
634         return new TagVariableInfo(nameGiven, nameFromAttribute, className,
635                 declare, scope);
636     }

638     private TagLibraryValidator createValidator(TreeNode elem)
639             throws JasperException {
640 
641         String validatorClass = null;
642         Map initParams = new Hashtable();
643 
644         Iterator list = elem.findChildren();
645         while (list.hasNext()) {
646             TreeNode element = (TreeNode) list.next();
647             String tname = element.getName();
648             if ("validator-class".equals(tname))
649                 validatorClass = element.getBody();
650             else if ("init-param".equals(tname)) {
651                 String[] initParam = createInitParam(element);
652                 initParams.put(initParam[0], initParam[1]);
653             } else if ("description".equals(tname) || // Ignored elements
654             false) {
655             } else {
656                 if (log.isWarnEnabled()) {
657                     log.warn(Localizer.getMessage(
658                             "jsp.warning.unknown.element.in.validator", tname));
659                 }
660             }
661         }
662 
663         TagLibraryValidator tlv = null;
664         if (validatorClass != null && !validatorClass.equals("")) {
665             try {
666                 Class tlvClass = ctxt.getClassLoader()
667                         .loadClass(validatorClass);
668                 tlv = (TagLibraryValidator) tlvClass.newInstance();
669             } catch (Exception e) {
670                 err.jspError("jsp.error.tlvclass.instantiation",
671                         validatorClass, e);
672             }
673         }
674         if (tlv != null) {
675             tlv.setInitParameters(initParams);
676         }
677         return tlv;
678     }

680     String[] createInitParam(TreeNode elem) {
681         String[] initParam = new String[2];
682 
683         Iterator list = elem.findChildren();
684         while (list.hasNext()) {
685             TreeNode element = (TreeNode) list.next();
686             String tname = element.getName();
687             if ("param-name".equals(tname)) {
688                 initParam[0] = element.getBody();
689             } else if ("param-value".equals(tname)) {
690                 initParam[1] = element.getBody();
691             } else if ("description".equals(tname)) {
692                  // Do nothing
693             } else {
694                 if (log.isWarnEnabled()) {
695                     log.warn(Localizer.getMessage(
696                             "jsp.warning.unknown.element.in.initParam", tname));
697                 }
698             }
699         }
700         return initParam;
701     }

703     FunctionInfo createFunctionInfo(TreeNode elem) {
704 
705         String name = null;
706         String klass = null;
707         String signature = null;
708 
709         Iterator list = elem.findChildren();
710         while (list.hasNext()) {
711             TreeNode element = (TreeNode) list.next();
712             String tname = element.getName();
713 
714             if ("name".equals(tname)) {
715                 name = element.getBody();
716             } else if ("function-class".equals(tname)) {
717                 klass = element.getBody();
718             } else if ("function-signature".equals(tname)) {
719                 signature = element.getBody();
720             } else if ("display-name".equals(tname) || // Ignored elements
721                     "small-icon".equals(tname) || "large-icon".equals(tname)
722                     || "description".equals(tname) || "example".equals(tname)) {
723             } else {
724                 if (log.isWarnEnabled()) {
725                     log.warn(Localizer.getMessage(
726                             "jsp.warning.unknown.element.in.function", tname));
727                 }
728             }
729         }
730 
731         return new FunctionInfo(name, klass, signature);
732     }

746     /**
747      * Translation-time validation of the XML document associated with the JSP
748      * page. This is a convenience method on the associated TagLibraryValidator
749      * class.
750      * 
751      * @param thePage
752      *            The JSP page object
753      * @return A string indicating whether the page is valid or not.
754      */
755     public ValidationMessage[] validate(PageData thePage) {
756         TagLibraryValidator tlv = getTagLibraryValidator();
757         if (tlv == null)
758             return null;
759 
760         String uri = getURI();
761         if (uri.startsWith("/")) {
762             uri = URN_JSPTLD + uri;
763         }
764 
765         return tlv.validate(getPrefixString(), uri, thePage);
766     }

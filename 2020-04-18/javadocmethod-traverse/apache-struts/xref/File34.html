41  /**
42   * Struts2 to GXP adapter. Can be used to write a GXP or create a
43   * {@link MarkupClosure}. Pulls GXP parameters from Struts2 value stack.
44   *
45   * @author Bob Lee
46   */
47  public abstract class AbstractGxp<T extends MarkupClosure> {
48  
49      ValueStackFactory valueStackFactory;
50      Map defaultValues = new HashMap();
51      List<Param> params;
52      Class gxpClass;
53      Method writeMethod;
54      Method getGxpClosureMethod;
55      boolean hasBodyParam;
56  
57      protected AbstractGxp(Class gxpClass) {
58          this(gxpClass, lookupMethodByName(gxpClass, "write"), lookupMethodByName(gxpClass, "getGxpClosure"));
59      }
60  
61      protected AbstractGxp(Class gxpClass, Method writeMethod, Method getGxpClosureMethod) {
62          this.gxpClass = gxpClass;
63          this.writeMethod = writeMethod;
64          this.getGxpClosureMethod = getGxpClosureMethod;
65          this.params = lookupParams();
66      }
67  
68      /**
69       * Writes GXP. Pulls GXP parameters from Struts2's value stack.
70       *
71       * @param out Appendable
72       * @param gxpContext GXP content
73       */
74      public void write(Appendable out, GxpContext gxpContext) {
75          write(out, gxpContext, null);
76      }
77  
78      /**
79       * Writes GXP. Pulls GXP parameters from Struts2's value stack.
80       *
81       * @param out Appendable
82       * @param gxpContext GXP content
83       * @param overrides parameter map pushed onto the value stack
84       */
85      protected void write(Appendable out, GxpContext gxpContext, Map overrides) {
86          Object[] args = getArgs(out, gxpContext, overrides);
87  
88          try {
89              writeMethod.invoke(getGxpInstance(), args);
90          } catch (Exception e) {
91              throw new RuntimeException(createDebugString(args, e), e);
92          }
93      }
94  
95      protected Object[] getArgs(Appendable out, GxpContext gxpContext, Map overrides) {
96          List<Object> argList = getArgListFromValueStack(overrides);
97          Object[] args = new Object[argList.size() + 2];
98          args[0] = out;
99          args[1] = gxpContext;
100         int index = 2;
101         for (Iterator<Object> i = argList.iterator(); i.hasNext(); index++) {
102             args[index] = i.next();
103         }
104         return args;
105     }
106 
107     /**
108      * @return the object on which to call the write and getGxpClosure methods. If
109      *         the methods are static, this can return {@code null}
110      */
111     protected Object getGxpInstance() {
112         return null;
113     }
114 
115     /**
116      * @return Creates GXP closure. Pulls GXP parameters from Struts 2 value stack.
117      */
118     public T getGxpClosure() {
119         return getGxpClosure(null, null);
120     }
121 
122     /**
123      * @return  Creates GXP closure. Pulls GXP parameters from Struts 2 value stack.
124      *
125      * @param body   is pushed onto the stack if this GXP has a
126      *               {@link MarkupClosure} (or subclass) parameter named "body".
127      * @param params comes first on the value stack.
128      */
129     @SuppressWarnings("unchecked")
130     protected T getGxpClosure(T body, Map params) {
131         final Map overrides = getOverrides(body, params);
132 
133         Object[] args = getArgListFromValueStack(overrides).toArray();
134 
135         try {
136             return (T) getGxpClosureMethod.invoke(getGxpInstance(), args);
137         } catch (IllegalArgumentException e) {
138             throw new RuntimeException(createDebugString(args, e), e);
139         } catch (IllegalAccessException e) {
140             throw new RuntimeException(createDebugString(args, e), e);
141         } catch (InvocationTargetException e) {
142             throw new RuntimeException(createDebugString(args, e), e);
143         }
144     }
145 
146     protected Map getOverrides(T body, Map params) {
147         final Map overrides = new HashMap();
148         if (hasBodyParam && body != null) {
149             overrides.put(Param.BODY_PARAM_NAME, body);
150         }
151         if (params != null) {
152             overrides.putAll(params);
153         }
154         return overrides;
155     }
156 
157     /**
158      * Iterates over GXP parameters, pulls value from value stack for each
159      * parameter, and appends the values to an argument list which will
160      * be passed to a method on a GXP.
161      *
162      * @param overrides parameter map pushed onto the value stack
163      *
164      * @return list of arguments
165      */
166     List getArgListFromValueStack(Map overrides) {
167 
168         ValueStack valueStack = valueStackFactory.createValueStack(ActionContext.getContext().getValueStack());
169 
170         // add default values to the bottom of the stack. if no action provides
171         // a getter for a param, the default value will be used.
172         valueStack.getRoot().add(this.defaultValues);
173 
174         // push override parameters onto the stack.
175         if (overrides != null && !overrides.isEmpty()) {
176             valueStack.push(overrides);
177         }
178 
179         List args = new ArrayList(params.size());
180         for (Param param : getParams()) {
181             try {
182                 args.add(valueStack.findValue(param.getName(), param.getType()));
183             } catch (Exception e) {
184                 throw new RuntimeException("Exception while finding '" + param.getName() + "'.", e);
185             }
186         }
187 
188         return args;
189     }
190 
191     /**
192      * Combines parameter names and types into <code>Param</code> objects.
193      *
194      * @return parameter list
195      */
196     List<Param> lookupParams() {
197         List<Param> params = new ArrayList<Param>();
198 
199         List<String> parameterNames = lookupParameterNames();
200         List<Class<?>> parameterTypes = lookupParameterTypes();
201         Iterator<Class<?>> parameterTypeIterator = parameterTypes.iterator();
202 
203         // If there are more parameter names than parameter types it means that we are
204         // using instantiable GXPs and there are 1 or more constructor parameters.
205         // Constructor params will always be first in the list, so just drop an appropriate
206         // number of elements from the beginning of the list.
207         if (parameterNames.size() > parameterTypes.size()) {
208             parameterNames = parameterNames.subList(parameterNames.size() - parameterTypes.size(), parameterNames.size());
209         }
210 
211         for (String name : parameterNames) {
212             Class paramType = parameterTypeIterator.next();
213             Param param = new Param(gxpClass, name, paramType);
214             params.add(param);
215 
216             if (param.isBody()) {
217                 hasBodyParam = true;
218             }
219 
220             if (param.isOptional()) {
221                 defaultValues.put(param.getName(), param.getDefaultValue());
222             }
223         }
224 
225         this.defaultValues = Collections.unmodifiableMap(this.defaultValues);
226         return Collections.unmodifiableList(params);
227     }
228 
229     /**
230      * @return  list of parameter types.
231      */
232     List<Class<?>> lookupParameterTypes() {
233         List<Class<?>> parameterTypes = Arrays.asList(writeMethod.getParameterTypes());
234         // skip the first two, gxp_out and gxp_context. they are for internal use.
235         return parameterTypes.subList(2, parameterTypes.size());
236     }
237 
238     /**
239      * @return  list of parameter names.
240      */
241     List<String> lookupParameterNames() {
242         try {
243             return (List<String>) gxpClass.getMethod("getArgList").invoke(null);
244         } catch (Exception e) {
245             throw new RuntimeException(e);
246         }
247     }
248 
249     /**
250      * @param clazz class
251      * @param name method name for lookup
252      *
253      * @return  first method with the given name. Should not be used if the
254      * method is overloaded.
255      */
256     protected static Method lookupMethodByName(Class clazz, String name) {
257         Method[] methods = clazz.getMethods();
258         for (int i = 0; i < methods.length; i++) {
259             if (methods[i].getName().equals(name)) {
260                 return methods[i];
261             }
262         }
263         throw new RuntimeException("No " + name + "(...) method found for "
264                 + clazz.getName() + ".");
265     }

278     /**
279      * @param gxpPath Gxp path
280      *
281      * @return  generated GXP class given an absolute path to a GXP file.
282      * The current implementation assumes that the GXP and generated Java source
283      * file share the same name with different extensions.
284      */
285     @VisibleForTesting
286     public static Class getGxpClassForPath(String gxpPath) {
287         int offset = (gxpPath.charAt(0) == '/') ? 1 : 0;
288         String className = gxpPath.substring(offset, gxpPath.length() - 4).replace('/', '.');
289         try {
290             return getClassLoader().loadClass(className);
291         } catch (ClassNotFoundException e) {
292             throw new RuntimeException(e);
293         }
294     }
295 
296     static ClassLoader getClassLoader() {
297         ClassLoader loader = Thread.currentThread().getContextClassLoader();
298         return (loader == null) ? ClassLoader.getSystemClassLoader() : loader;
299     }

301     /**
302      * @param args arguments
303      * @param exception Exception
304      *
305      * @return debug String which can be tacked onto an exception.
306      */
307     String createDebugString(Object[] args, Exception exception) {
308         StringBuffer buffer = new StringBuffer();
309         printExceptionTraceToBuffer(exception, buffer);
310         buffer.append("\nException in GXP: ").append(gxpClass.getName()).append(". Params:");
311         int index = 2;
312         for (Param param : getParams()) {
313             try {
314                 Object arg = args[index++];
315                 String typesMatch = "n/a (null)";
316                 if (arg != null) {
317                     if (doesArgumentTypeMatchParamType(param, arg)) {
318                         typesMatch = "YES";
319                     } else {
320                         typesMatch = "NO";
321                     }
322                 }
323                 buffer.append("\n  ")
324                         .append(param.toString())
325                         .append(" = ")
326                         .append(arg)
327                         .append("; ")
328                         .append("[types match? ")
329                         .append(typesMatch)
330                         .append("]");
331             } catch (Exception e) {
332                 buffer.append(" >Error getting information for param # ").append(index).append("< ");
333             }
334         }
335         buffer.append("\nStack trace: ");
336         return buffer.toString();
337     }
338 
339     private void printExceptionTraceToBuffer(Exception e,
340                                              StringBuffer buffer) {
341         StringWriter out = new StringWriter();
342         e.printStackTrace(new PrintWriter(out));
343         buffer.append(out.getBuffer().toString());
344     }
345 
346     private boolean doesArgumentTypeMatchParamType(Param param, Object arg) {
347         Class paramType = param.getType();
348         Class<? extends Object> argClass = arg.getClass();
349 
350         // TODO(jpelly): Handle all primitive unwrapping (ie, Boolean --> boolean).
351         if (boolean.class.equals(paramType) && Boolean.class.equals(argClass)) {
352             return true;
353         } else if (char.class.equals(paramType) && Character.class.equals(argClass)) {
354             return true;
355         }
356         return paramType.isAssignableFrom(argClass);
357     }

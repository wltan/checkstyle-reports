32  /**
33   * Moves a file or directory to a new file or directory.
34   * By default, the
35   * destination file is overwritten if it already exists.
36   * When <i>overwrite</i> is
37   * turned off, then files are only moved if the source file is
38   * newer than the destination file, or when the destination file does
39   * not exist.
40   *
41   * <p>Source files and directories are only deleted when the file or
42   * directory has been copied to the destination successfully.  Filtering
43   * also works.</p>
44   *
45   * <p>This implementation is based on Arnout Kuiper's initial design
46   * document, the following mailing list discussions, and the
47   * copyfile/copydir tasks.</p>
48   *
49   * @since Ant 1.2
50   *
51   * @ant.task category="filesystem"
52   */
53  public class Move extends Copy {
54  
55      private boolean performGc = Os.isFamily("windows");
56  
57      /**
58       * Constructor of object.
59       * This sets the forceOverwrite attribute of the Copy parent class
60       * to true.
61       *
62       */
63      public Move() {
64          super();
65          setOverwrite(true);
66      }
67  
68      /**
69       * Whether to perform a garbage collection before retrying a failed delete.
70       *
71       * <p>This may be required on Windows (where it is set to true by
72       * default) but also on other operating systems, for example when
73       * deleting directories from an NFS share.</p>
74       *
75       * @since Ant 1.8.3
76       */
77      public void setPerformGcOnFailedDelete(boolean b) {
78          performGc = b;
79      }
80  
81      /** {@inheritDoc}. */
82      protected void validateAttributes() throws BuildException {
83          if (file != null && file.isDirectory()) {
84              if ((destFile != null && destDir != null)
85                  || (destFile == null && destDir == null)) {
86                  throw new BuildException("One and only one of tofile and todir must be set.");
87              }
88              destFile = destFile == null ? new File(destDir, file.getName()) : destFile;
89              destDir = destDir == null ? destFile.getParentFile() : destDir;
90  
91              completeDirMap.put(file, destFile);
92              file = null;
93          } else {
94              super.validateAttributes();
95          }
96      }

102     /**
103      * Override copy's doFileOperations to move the files instead of copying them.
104      */
105     protected void doFileOperations() {
106         //Attempt complete directory renames, if any, first.
107         if (completeDirMap.size() > 0) {
108             for (Iterator fromDirs = completeDirMap.keySet().iterator(); fromDirs.hasNext();) {
109                 File fromDir = (File) fromDirs.next();
110                 File toDir = (File) completeDirMap.get(fromDir);
111                 boolean renamed = false;
112                 try {
113                     log("Attempting to rename dir: " + fromDir + " to " + toDir, verbosity);
114                     renamed = renameFile(fromDir, toDir, filtering, forceOverwrite);
115                 } catch (IOException ioe) {
116                     String msg = "Failed to rename dir " + fromDir
117                             + " to " + toDir + " due to " + ioe.getMessage();
118                     throw new BuildException(msg, ioe, getLocation());
119                 }
120                 if (!renamed) {
121                     FileSet fs = new FileSet();
122                     fs.setProject(getProject());
123                     fs.setDir(fromDir);
124                     addFileset(fs);
125                     DirectoryScanner ds = fs.getDirectoryScanner(getProject());
126                     String[] files = ds.getIncludedFiles();
127                     String[] dirs = ds.getIncludedDirectories();
128                     scan(fromDir, toDir, files, dirs);
129                 }
130             }
131         }
132         int moveCount = fileCopyMap.size();
133         if (moveCount > 0) {   // files to move
134             log("Moving " + moveCount + " file" + ((moveCount == 1) ? "" : "s")
135                     + " to " + destDir.getAbsolutePath());
136 
137             for (Iterator fromFiles = fileCopyMap.keySet().iterator(); fromFiles.hasNext();) {
138                 String fromFile = (String) fromFiles.next();
139                 File f = new File(fromFile);
140                 boolean selfMove = false;
141                 if (f.exists()) { //Is this file still available to be moved?
142                     String[] toFiles = (String[]) fileCopyMap.get(fromFile);
143                     for (int i = 0; i < toFiles.length; i++) {
144                         String toFile = (String) toFiles[i];
145 
146                         if (fromFile.equals(toFile)) {
147                             log("Skipping self-move of " + fromFile, verbosity);
148                             selfMove = true;
149 
150                             // if this is the last time through the loop then
151                             // move will not occur, but that's what we want
152                             continue;
153                         }
154                         File d = new File(toFile);
155                         if ((i + 1) == toFiles.length && !selfMove) {
156                             // Only try to move if this is the last mapped file
157                             // and one of the mappings isn't to itself
158                             moveFile(f, d, filtering, forceOverwrite);
159                         } else {
160                             copyFile(f, d, filtering, forceOverwrite);
161                         }
162                     }
163                 }
164             }
165         }
166 
167         if (includeEmpty) {
168             int createCount = 0;
169             for (Iterator fromDirNames = dirCopyMap.keySet().iterator(); fromDirNames.hasNext();) {
170                 String fromDirName = (String) fromDirNames.next();
171                 String[] toDirNames = (String[]) dirCopyMap.get(fromDirName);
172                 boolean selfMove = false;
173                 for (int i = 0; i < toDirNames.length; i++) {
174                     if (fromDirName.equals(toDirNames[i])) {
175                         log("Skipping self-move of " + fromDirName, verbosity);
176                         selfMove = true;
177                         continue;
178                     }
179                     File d = new File(toDirNames[i]);
180                     if (!d.exists()) {
181                         if (!(d.mkdirs() || d.exists())) {
182                             log("Unable to create directory "
183                                     + d.getAbsolutePath(), Project.MSG_ERR);
184                         } else {
185                             createCount++;
186                         }
187                     }
188                 }
189                 File fromDir = new File(fromDirName);
190                 if (!selfMove && okToDelete(fromDir)) {
191                     deleteDir(fromDir);
192                 }
193             }
194             if (createCount > 0) {
195                 log("Moved " + dirCopyMap.size()
196                         + " empty director"
197                         + (dirCopyMap.size() == 1 ? "y" : "ies")
198                         + " to " + createCount
199                         + " empty director"
200                         + (createCount == 1 ? "y" : "ies") + " under "
201                         + destDir.getAbsolutePath());
202             }
203         }
204     }

206     /**
207      * Try to move the file via a rename, but if this fails or filtering
208      * is enabled, copy the file then delete the sourceFile.
209      */
210     private void moveFile(File fromFile, File toFile, boolean filtering, boolean overwrite) {
211         boolean moved = false;
212         try {
213             log("Attempting to rename: " + fromFile + " to " + toFile, verbosity);
214             moved = renameFile(fromFile, toFile, filtering, forceOverwrite);
215         } catch (IOException ioe) {
216             String msg = "Failed to rename " + fromFile
217                 + " to " + toFile + " due to " + ioe.getMessage();
218             throw new BuildException(msg, ioe, getLocation());
219         }
220 
221         if (!moved) {
222             copyFile(fromFile, toFile, filtering, overwrite);
223             if (!getFileUtils().tryHardToDelete(fromFile, performGc)) {
224                 throw new BuildException("Unable to delete " + "file "
225                         + fromFile.getAbsolutePath());
226             }
227         }
228     }

230     /**
231      * Copy fromFile to toFile.
232      * @param fromFile
233      * @param toFile
234      * @param filtering
235      * @param overwrite
236      */
237     private void copyFile(File fromFile, File toFile, boolean filtering, boolean overwrite) {
238         try {
239             log("Copying " + fromFile + " to " + toFile, verbosity);
240 
241             FilterSetCollection executionFilters = new FilterSetCollection();
242             if (filtering) {
243                 executionFilters.addFilterSet(getProject().getGlobalFilterSet());
244             }
245             for (Iterator filterIter = getFilterSets().iterator(); filterIter.hasNext();) {
246                 executionFilters.addFilterSet((FilterSet) filterIter.next());
247             }
248             getFileUtils().copyFile(fromFile, toFile, executionFilters,
249                                     getFilterChains(),
250                                     forceOverwrite,
251                                     getPreserveLastModified(),
252                                     /* append: */ false,
253                                     getEncoding(),
254                                     getOutputEncoding(),
255                                     getProject(), getForce());
256         } catch (IOException ioe) {
257             String msg = "Failed to copy " + fromFile
258                     + " to " + toFile + " due to " + ioe.getMessage();
259             throw new BuildException(msg, ioe, getLocation());
260         }
261     }

263     /**
264      * Its only ok to delete a directory tree if there are no files in it.
265      * @param d the directory to check
266      * @return true if a deletion can go ahead
267      */
268     protected boolean okToDelete(File d) {
269         String[] list = d.list();
270         if (list == null) {
271             return false;
272         }     // maybe io error?
273 
274         for (int i = 0; i < list.length; i++) {
275             String s = list[i];
276             File f = new File(d, s);
277             if (f.isDirectory()) {
278                 if (!okToDelete(f)) {
279                     return false;
280                 }
281             } else {
282                 return false;   // found a file
283             }
284         }
285         return true;
286     }

288     /**
289      * Go and delete the directory tree.
290      * @param d the directory to delete
291      */
292     protected void deleteDir(File d) {
293         deleteDir(d, false);
294     }
295 
296     /**
297      * Go and delete the directory tree.
298      * @param d the directory to delete
299      * @param deleteFiles whether to delete files
300      */
301     protected void deleteDir(File d, boolean deleteFiles) {
302         String[] list = d.list();
303         if (list == null) {
304             return;
305         }      // on an io error list() can return null
306 
307         for (int i = 0; i < list.length; i++) {
308             String s = list[i];
309             File f = new File(d, s);
310             if (f.isDirectory()) {
311                 deleteDir(f);
312             } else if (deleteFiles && !getFileUtils().tryHardToDelete(f,
313                                                                       performGc)) {
314                 throw new BuildException("Unable to delete file " + f.getAbsolutePath());
315             } else {
316                 throw new BuildException("UNEXPECTED ERROR - The file "
317                         + f.getAbsolutePath() + " should not exist!");
318             }
319         }
320         log("Deleting directory " + d.getAbsolutePath(), verbosity);
321         if (!getFileUtils().tryHardToDelete(d, performGc)) {
322             throw new BuildException("Unable to delete directory " + d.getAbsolutePath());
323         }
324     }

326     /**
327      * Attempts to rename a file from a source to a destination.
328      * If overwrite is set to true, this method overwrites existing file
329      * even if the destination file is newer.  Otherwise, the source file is
330      * renamed only if the destination file is older than it.
331      * Method then checks if token filtering is used.  If it is, this method
332      * returns false assuming it is the responsibility to the copyFile method.
333      *
334      * @param sourceFile the file to rename
335      * @param destFile   the destination file
336      * @param filtering  if true, filtering is in operation, file will
337      *                   be copied/deleted instead of renamed
338      * @param overwrite  if true force overwrite even if destination file
339      *                   is newer than source file
340      * @return true if the file was renamed
341      * @exception IOException if an error occurs
342      * @exception BuildException if an error occurs
343      */
344     protected boolean renameFile(File sourceFile, File destFile, boolean filtering,
345                                  boolean overwrite) throws IOException, BuildException {
346         if (destFile.isDirectory() || filtering || getFilterSets().size() > 0
347                 || getFilterChains().size() > 0) {
348             return false;
349         }
350 
351         // identical logic lives in ResourceUtils.copyResource():
352         if (destFile.isFile() && !destFile.canWrite()) {
353             if (!getForce()) {
354                 throw new IOException("can't replace read-only destination "
355                                       + "file " + destFile);
356             } else if (!getFileUtils().tryHardToDelete(destFile)) {
357                 throw new IOException("failed to delete read-only "
358                                       + "destination file " + destFile);
359             }
360         }
361 
362         // identical logic lives in FileUtils.rename():
363         File parent = destFile.getParentFile();
364         if (parent != null && !parent.exists()) {
365             parent.mkdirs();
366         } else if (destFile.isFile()) {
367             sourceFile = getFileUtils().normalize(sourceFile.getAbsolutePath()).getCanonicalFile();
368             destFile = getFileUtils().normalize(destFile.getAbsolutePath());
369             if (destFile.getAbsolutePath().equals(sourceFile.getAbsolutePath())) {
370                 //no point in renaming a file to its own canonical version...
371                 log("Rename of " + sourceFile + " to " + destFile
372                     + " is a no-op.", Project.MSG_VERBOSE);
373                 return true;
374             }
375             if (!(getFileUtils().areSame(sourceFile, destFile)
376                   || getFileUtils().tryHardToDelete(destFile, performGc))) {
377                 throw new BuildException("Unable to remove existing file " + destFile);
378             }
379         }
380         return sourceFile.renameTo(destFile);
381     }

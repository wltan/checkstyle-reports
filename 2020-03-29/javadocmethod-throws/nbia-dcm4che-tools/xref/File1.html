104     public void setSessionContext(SessionContext ctx) {
105         Context jndiCtx = null;
106         try {
107             jndiCtx = new InitialContext();
108             patHome = (PatientLocalHome) jndiCtx
109                     .lookup("java:comp/env/ejb/Patient");
110             studyHome = (StudyLocalHome) jndiCtx
111                     .lookup("java:comp/env/ejb/Study");
112             seriesHome = (SeriesLocalHome) jndiCtx
113                     .lookup("java:comp/env/ejb/Series");
114             instHome = (InstanceLocalHome) jndiCtx
115                     .lookup("java:comp/env/ejb/Instance");
116             mppsHome = (MPPSLocalHome) jndiCtx
117             .lookup("java:comp/env/ejb/MPPS");
118             mwlHome = (MWLItemLocalHome) jndiCtx
119             .lookup("java:comp/env/ejb/MWLItem");
120         } catch (NamingException e) {
121             throw new EJBException(e);
122         } finally {
123             if (jndiCtx != null) {
124                 try {
125                     jndiCtx.close();
126                 } catch (NamingException ignore) {
127                 }
128             }
129         }
130     }

139     /**
140      * @ejb.interface-method
141      */
142     public boolean modifyAttributes(Dataset attrs, Date time, String system,
143             String reason, boolean updateOriginalAttributesSeq,
144             int entityNotFoundErrorCode) throws DcmServiceException {
145         try {
146             String level = attrs.getString(Tags.QueryRetrieveLevel);
147             if (level == null)
148                 throw new IllegalArgumentException(
149                         "Missing Query/Retrieve Level");
150             if ("IMAGE".equals(level))
151                 return updateInstanceAttrs(attrs, time, system, reason,
152                         updateOriginalAttributesSeq);
153             if ("SERIES".equals(level))
154                 return updateSeriesAttrs(attrs, time, system, reason,
155                         updateOriginalAttributesSeq);
156             if ("STUDY".equals(level))
157                 return updateStudyAttrs(attrs, time, system, reason,
158                         updateOriginalAttributesSeq);
159             throw new IllegalArgumentException(
160                     "Illegal Query/Retrieve Level: " + level);
161         } catch (IllegalArgumentException e) {
162             throw new DcmServiceException(
163                     Status.DataSetDoesNotMatchSOPClassError, e.getMessage());
164         } catch (ObjectNotFoundException e) {
165             if (entityNotFoundErrorCode != 0) {
166                 throw new DcmServiceException(entityNotFoundErrorCode,
167                         "No entity with specified uid found");
168             }
169             log.info("No entity with specified uid found - ignore update:");
170             log.debug(attrs);
171             return false;
172         } catch (FinderException e) {
173             throw new EJBException(e);
174         }
175     }
176 
177     private boolean updateStudyAttrs(Dataset attrs, Date time, String system,
178             String reason, boolean updateOriginalAttributesSeq)
179             throws FinderException {
180         String styiuid = attrs.getString(Tags.StudyInstanceUID);
181         if (styiuid ==  null)
182             throw new IllegalArgumentException("Missing Study Instance UID");
183         StudyLocal study = studyHome.findByStudyIuid(styiuid);
184         Dataset origAttrs = updateOriginalAttributesSeq
185                 ? DcmObjectFactory.getInstance().newDataset()
186                 : null;
187         if (!study.updateAttributes(attrs, origAttrs))
188             return false;
189         if (updateOriginalAttributesSeq)
190             updateOriginalAttributesSeq(study, time, system, reason, origAttrs);
191         return true;
192     }
193 
194     private boolean updateSeriesAttrs(Dataset attrs, Date time, String system,
195             String reason, boolean updateOriginalAttributesSeq)
196             throws FinderException {
197         String seriuid = attrs.getString(Tags.SeriesInstanceUID);
198         if (seriuid ==  null)
199             throw new IllegalArgumentException("Missing Series Instance UID");
200         SeriesLocal series = seriesHome.findBySeriesIuid(seriuid);
201         Dataset origAttrs = updateOriginalAttributesSeq
202                 ? DcmObjectFactory.getInstance().newDataset()
203                 : null;
204         if (!series.updateAttributes(attrs, true, origAttrs))
205             return false;
206         if (updateOriginalAttributesSeq)
207             updateOriginalAttributesSeq(series, time, system, reason, origAttrs);
208         return true;
209     }
210 
211     private boolean updateInstanceAttrs(Dataset attrs, Date time,
212             String system, String reason, boolean updateOriginalAttributesSeq)
213             throws FinderException {
214         String sopiuid = attrs.getString(Tags.SOPInstanceUID);
215         if (sopiuid == null)
216             throw new IllegalArgumentException("Missing SOP Instance UID");
217         InstanceLocal inst = instHome.findBySopIuid(sopiuid);
218         Dataset origAttrs = updateOriginalAttributesSeq
219                 ? DcmObjectFactory.getInstance().newDataset()
220                 : null;
221         if (!inst.updateAttributes(attrs, origAttrs))
222             return false;
223         if (updateOriginalAttributesSeq)
224             updateOriginalAttributesSeq(inst, time, system, reason, origAttrs);
225         return true;
226     }

228     /**
229      * @throws CreateException 
230      * @throws RemoveException 
231      * @ejb.interface-method
232      */
233     public boolean moveStudyToPatient(Dataset attrs, PatientMatching matching, boolean create) throws FinderException, RemoveException, CreateException {
234         String[] suids = attrs.getStrings(Tags.StudyInstanceUID);
235         if (suids == null || suids.length == 0) {
236             throw new IllegalArgumentException("Missing Study Instance UID for moveStudyToPatient");
237         }
238         StudyLocal study = studyHome.findByStudyIuid(suids[0]);
239         PatientLocal pat = this.getPatient(attrs, matching, create);
240         if (pat != null && !pat.isIdentical(study.getPatient())) {
241             for (int i = 0 ; i < suids.length ; i++) {
242                 pat.getMpps().addAll(mppsHome.findByStudyIuid(suids[i]));
243                 pat.getMwlItems().addAll(mwlHome.findByStudyIuid(suids[i]));
244                 pat.getStudies().add(i==0 ? study : studyHome.findByStudyIuid(suids[i]));
245             }
246             return true;
247         }
248         return false;
249     }
250 
251     private void updateOriginalAttributesSeq(StudyLocal study,
252             Date time, String system, String reason, Dataset origAttrs) {
253         for (Iterator iter = study.getSeries().iterator(); iter.hasNext();)
254             updateOriginalAttributesSeq((SeriesLocal) iter.next(), 
255                     time, system, reason, origAttrs);
256     }
257 
258     private void updateOriginalAttributesSeq(SeriesLocal series,
259             Date time, String system, String reason, Dataset origAttrs) {
260         for (Iterator iter = series.getInstances().iterator(); iter.hasNext();)
261             updateOriginalAttributesSeq((InstanceLocal) iter.next(),
262                     time, system, reason, origAttrs);
263     }
264 
265     private void updateOriginalAttributesSeq(InstanceLocal inst,
266             Date time, String system, String reason, Dataset origAttrs) {
267         Dataset attrs = inst.getAttributes(false);
268         DcmElement origAttrsSeq = attrs.get(Tags.OriginalAttributesSeq);
269         if (origAttrsSeq == null)
270             origAttrsSeq = attrs.putSQ(Tags.OriginalAttributesSeq);
271         Dataset origAttrsItem = origAttrsSeq.addNewItem();
272         origAttrsItem.putLO(Tags.SourceOfPreviousValues);
273         origAttrsItem.putDT(Tags.AttributeModificationDatetime, time);
274         origAttrsItem.putLO(Tags.ModifyingSystem, system);
275         origAttrsItem.putCS(Tags.ReasonForTheAttributeModification, reason);
276         origAttrsItem.putSQ(Tags.ModifiedAttributesSeq).addItem(origAttrs);
277         inst.setAttributes(attrs);
278     }

281     private PatientLocal getPatient(Dataset attrs, PatientMatching matching, boolean create) 
282             throws FinderException, NonUniquePatientException, RemoveException, CreateException {
283         PatientLocal pat;
284         try {
285             return patHome.selectPatient(attrs, matching, true);
286         } catch (ObjectNotFoundException onfe) {
287             if (create) {
288                 try {
289                     pat = patHome.create(attrs);
290                     // Check if patient record was also inserted by concurrent thread
291                     try {
292                         return patHome.selectPatient(attrs, matching, true);
293                     } catch (NonUniquePatientException nupe) {
294                         pat.remove();
295                         pat = patHome.selectPatient(attrs, matching, true);
296                     } catch (ObjectNotFoundException onfe2) {
297                         // Just inserted Patient not found because of missing value
298                         // of attribute configured as required for Patient Matching
299                         return pat;
300                     }
301                  } catch (CreateException ce) {
302                     // Check if patient record was inserted by concurrent thread
303                     // with unique index on (pat_id, pat_id_issuer)
304                      try {
305                          pat = patHome.selectPatient(attrs, matching, true);
306                      } catch (ObjectNotFoundException onfe2) {
307                          throw ce;
308                      }
309                 }
310             } else {
311                 throw onfe;
312             }
313         } catch (NonUniquePatientException nupe) {
314             if (create)
315                 return patHome.create(attrs);
316             else
317                 throw nupe;
318         }
319         return pat;
320     }

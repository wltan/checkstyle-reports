41  /**
42   * RSA Cipher implementation class. We currently only support
43   * PKCS#1 v1.5 padding on top of CKM_RSA_PKCS.
44   *
45   * @author  Andreas Sterbenz
46   * @since   1.5
47   */
48  final class P11RSACipher extends CipherSpi {
49  
50      // minimum length of PKCS#1 v1.5 padding
51      private final static int PKCS1_MIN_PADDING_LENGTH = 11;
52  
53      // constant byte[] of length 0
54      private final static byte[] B0 = new byte[0];
55  
56      // mode constant for public key encryption
57      private final static int MODE_ENCRYPT = 1;
58      // mode constant for private key decryption
59      private final static int MODE_DECRYPT = 2;
60      // mode constant for private key encryption (signing)
61      private final static int MODE_SIGN    = 3;
62      // mode constant for public key decryption (verifying)
63      private final static int MODE_VERIFY  = 4;
64  
65      // padding type constant for NoPadding
66      private final static int PAD_NONE = 1;
67      // padding type constant for PKCS1Padding
68      private final static int PAD_PKCS1 = 2;
69  
70      // token instance
71      private final Token token;
72  
73      // algorithm name (always "RSA")
74      private final String algorithm;
75  
76      // mechanism id
77      private final long mechanism;
78  
79      // associated session, if any
80      private Session session;
81  
82      // mode, one of MODE_* above
83      private int mode;
84  
85      // padding, one of PAD_* above
86      private int padType;
87  
88      private byte[] buffer;
89      private int bufOfs;
90  
91      // key, if init() was called
92      private P11Key p11Key;
93  
94      // flag indicating whether an operation is initialized
95      private boolean initialized;
96  
97      // maximum input data size allowed
98      // for decryption, this is the length of the key
99      // for encryption, length of the key minus minimum padding length
100     private int maxInputSize;
101 
102     // maximum output size. this is the length of the key
103     private int outputSize;
104 
105     P11RSACipher(Token token, String algorithm, long mechanism)
106             throws PKCS11Exception {
107         super();
108         this.token = token;
109         this.algorithm = "RSA";
110         this.mechanism = mechanism;
111     }

113     // modes do not make sense for RSA, but allow ECB
114     // see JCE spec
115     protected void engineSetMode(String mode) throws NoSuchAlgorithmException {
116         if (mode.equalsIgnoreCase("ECB") == false) {
117             throw new NoSuchAlgorithmException("Unsupported mode " + mode);
118         }
119     }

121     protected void engineSetPadding(String padding)
122             throws NoSuchPaddingException {
123         String lowerPadding = padding.toLowerCase(Locale.ENGLISH);
124         if (lowerPadding.equals("pkcs1padding")) {
125             padType = PAD_PKCS1;
126         } else if (lowerPadding.equals("nopadding")) {
127             padType = PAD_NONE;
128         } else {
129             throw new NoSuchPaddingException("Unsupported padding " + padding);
130         }
131     }

185     private void implInit(int opmode, Key key) throws InvalidKeyException {
186         cancelOperation();
187         p11Key = P11KeyFactory.convertKey(token, key, algorithm);
188         boolean encrypt;
189         if (opmode == Cipher.ENCRYPT_MODE) {
190             encrypt = true;
191         } else if (opmode == Cipher.DECRYPT_MODE) {
192             encrypt = false;
193         } else if (opmode == Cipher.WRAP_MODE) {
194             if (p11Key.isPublic() == false) {
195                 throw new InvalidKeyException
196                                 ("Wrap has to be used with public keys");
197             }
198             // No further setup needed for C_Wrap(). We'll initialize later if
199             // we can't use C_Wrap().
200             return;
201         } else if (opmode == Cipher.UNWRAP_MODE) {
202             if (p11Key.isPrivate() == false) {
203                 throw new InvalidKeyException
204                                 ("Unwrap has to be used with private keys");
205             }
206             // No further setup needed for C_Unwrap(). We'll initialize later
207             // if we can't use C_Unwrap().
208             return;
209         } else {
210             throw new InvalidKeyException("Unsupported mode: " + opmode);
211         }
212         if (p11Key.isPublic()) {
213             mode = encrypt ? MODE_ENCRYPT : MODE_VERIFY;
214         } else if (p11Key.isPrivate()) {
215             mode = encrypt ? MODE_SIGN : MODE_DECRYPT;
216         } else {
217             throw new InvalidKeyException("Unknown key type: " + p11Key);
218         }
219         int n = (p11Key.keyLength() + 7) >> 3;
220         outputSize = n;
221         buffer = new byte[n];
222         maxInputSize = ((padType == PAD_PKCS1 && encrypt) ?
223                             (n - PKCS1_MIN_PADDING_LENGTH) : n);
224         try {
225             initialize();
226         } catch (PKCS11Exception e) {
227             throw new InvalidKeyException("init() failed", e);
228         }
229     }
230 
231     private void cancelOperation() {
232         token.ensureValid();
233         if (initialized == false) {
234             return;
235         }
236         initialized = false;
237         if ((session == null) || (token.explicitCancel == false)) {
238             return;
239         }
240         if (session.hasObjects() == false) {
241             session = token.killSession(session);
242             return;
243         }
244         try {
245             PKCS11 p11 = token.p11;
246             int inLen = maxInputSize;
247             int outLen = buffer.length;
248             switch (mode) {
249             case MODE_ENCRYPT:
250                 p11.C_Encrypt
251                         (session.id(), buffer, 0, inLen, buffer, 0, outLen);
252                 break;
253             case MODE_DECRYPT:
254                 p11.C_Decrypt
255                         (session.id(), buffer, 0, inLen, buffer, 0, outLen);
256                 break;
257             case MODE_SIGN:
258                 byte[] tmpBuffer = new byte[maxInputSize];
259                 p11.C_Sign
260                         (session.id(), tmpBuffer);
261                 break;
262             case MODE_VERIFY:
263                 p11.C_VerifyRecover
264                         (session.id(), buffer, 0, inLen, buffer, 0, outLen);
265                 break;
266             default:
267                 throw new ProviderException("internal error");
268             }
269         } catch (PKCS11Exception e) {
270             // XXX ensure this always works, ignore error
271         }
272     }
273 
274     private void ensureInitialized() throws PKCS11Exception {
275         token.ensureValid();
276         if (initialized == false) {
277             initialize();
278         }
279     }
280 
281     private void initialize() throws PKCS11Exception {
282         if (session == null) {
283             session = token.getOpSession();
284         }
285         PKCS11 p11 = token.p11;
286         CK_MECHANISM ckMechanism = new CK_MECHANISM(mechanism);
287         switch (mode) {
288         case MODE_ENCRYPT:
289             p11.C_EncryptInit(session.id(), ckMechanism, p11Key.keyID);
290             break;
291         case MODE_DECRYPT:
292             p11.C_DecryptInit(session.id(), ckMechanism, p11Key.keyID);
293             break;
294         case MODE_SIGN:
295             p11.C_SignInit(session.id(), ckMechanism, p11Key.keyID);
296             break;
297         case MODE_VERIFY:
298             p11.C_VerifyRecoverInit(session.id(), ckMechanism, p11Key.keyID);
299             break;
300         default:
301             throw new AssertionError("internal error");
302         }
303         bufOfs = 0;
304         initialized = true;
305     }
306 
307     private void implUpdate(byte[] in, int inOfs, int inLen) {
308         try {
309             ensureInitialized();
310         } catch (PKCS11Exception e) {
311             throw new ProviderException("update() failed", e);
312         }
313         if ((inLen == 0) || (in == null)) {
314             return;
315         }
316         if (bufOfs + inLen > maxInputSize) {
317             bufOfs = maxInputSize + 1;
318             return;
319         }
320         System.arraycopy(in, inOfs, buffer, bufOfs, inLen);
321         bufOfs += inLen;
322     }
323 
324     private int implDoFinal(byte[] out, int outOfs, int outLen)
325             throws BadPaddingException, IllegalBlockSizeException {
326         if (bufOfs > maxInputSize) {
327             throw new IllegalBlockSizeException("Data must not be longer "
328                 + "than " + maxInputSize + " bytes");
329         }
330         try {
331             ensureInitialized();
332             PKCS11 p11 = token.p11;
333             int n;
334             switch (mode) {
335             case MODE_ENCRYPT:
336                 n = p11.C_Encrypt
337                         (session.id(), buffer, 0, bufOfs, out, outOfs, outLen);
338                 break;
339             case MODE_DECRYPT:
340                 n = p11.C_Decrypt
341                         (session.id(), buffer, 0, bufOfs, out, outOfs, outLen);
342                 break;
343             case MODE_SIGN:
344                 byte[] tmpBuffer = new byte[bufOfs];
345                 System.arraycopy(buffer, 0, tmpBuffer, 0, bufOfs);
346                 tmpBuffer = p11.C_Sign(session.id(), tmpBuffer);
347                 if (tmpBuffer.length > outLen) {
348                     throw new BadPaddingException("Output buffer too small");
349                 }
350                 System.arraycopy(tmpBuffer, 0, out, outOfs, tmpBuffer.length);
351                 n = tmpBuffer.length;
352                 break;
353             case MODE_VERIFY:
354                 n = p11.C_VerifyRecover
355                         (session.id(), buffer, 0, bufOfs, out, outOfs, outLen);
356                 break;
357             default:
358                 throw new ProviderException("internal error");
359             }
360             return n;
361         } catch (PKCS11Exception e) {
362             throw (BadPaddingException)new BadPaddingException
363                 ("doFinal() failed").initCause(e);
364         } finally {
365             initialized = false;
366             session = token.releaseSession(session);
367         }
368     }

370     // see JCE spec
371     protected byte[] engineUpdate(byte[] in, int inOfs, int inLen) {
372         implUpdate(in, inOfs, inLen);
373         return B0;
374     }

376     // see JCE spec
377     protected int engineUpdate(byte[] in, int inOfs, int inLen,
378             byte[] out, int outOfs) throws ShortBufferException {
379         implUpdate(in, inOfs, inLen);
380         return 0;
381     }

383     // see JCE spec
384     protected byte[] engineDoFinal(byte[] in, int inOfs, int inLen)
385             throws IllegalBlockSizeException, BadPaddingException {
386         implUpdate(in, inOfs, inLen);
387         int n = implDoFinal(buffer, 0, buffer.length);
388         byte[] out = new byte[n];
389         System.arraycopy(buffer, 0, out, 0, n);
390         return out;
391     }
392 
393     // see JCE spec
394     protected int engineDoFinal(byte[] in, int inOfs, int inLen,
395             byte[] out, int outOfs) throws ShortBufferException,
396             IllegalBlockSizeException, BadPaddingException {
397         implUpdate(in, inOfs, inLen);
398         return implDoFinal(out, outOfs, out.length - outOfs);
399     }
400 
401     private byte[] doFinal() throws BadPaddingException,
402             IllegalBlockSizeException {
403         byte[] t = new byte[2048];
404         int n = implDoFinal(t, 0, t.length);
405         byte[] out = new byte[n];
406         System.arraycopy(t, 0, out, 0, n);
407         return out;
408     }

410     // see JCE spec
411     protected byte[] engineWrap(Key key) throws InvalidKeyException,
412             IllegalBlockSizeException {
413         String keyAlg = key.getAlgorithm();
414         P11Key sKey = null;
415         try {
416             // The conversion may fail, e.g. trying to wrap an AES key on
417             // a token that does not support AES, or when the key size is
418             // not within the range supported by the token.
419             sKey = P11SecretKeyFactory.convertKey(token, key, keyAlg);
420         } catch (InvalidKeyException ike) {
421             byte[] toBeWrappedKey = key.getEncoded();
422             if (toBeWrappedKey == null) {
423                 throw new InvalidKeyException
424                         ("wrap() failed, no encoding available", ike);
425             }
426             // Directly encrypt the key encoding when key conversion failed
427             implInit(Cipher.ENCRYPT_MODE, p11Key);
428             implUpdate(toBeWrappedKey, 0, toBeWrappedKey.length);
429             try {
430                 return doFinal();
431             } catch (BadPaddingException bpe) {
432                 // should not occur
433                 throw new InvalidKeyException("wrap() failed", bpe);
434             } finally {
435                 // Restore original mode
436                 implInit(Cipher.WRAP_MODE, p11Key);
437             }
438         }
439         Session s = null;
440         try {
441             s = token.getOpSession();
442             return token.p11.C_WrapKey(s.id(), new CK_MECHANISM(mechanism),
443                 p11Key.keyID, sKey.keyID);
444         } catch (PKCS11Exception e) {
445             throw new InvalidKeyException("wrap() failed", e);
446         } finally {
447             token.releaseSession(s);
448         }
449     }

451     // see JCE spec
452     protected Key engineUnwrap(byte[] wrappedKey, String algorithm,
453             int type) throws InvalidKeyException, NoSuchAlgorithmException {
454         if (algorithm.equals("TlsRsaPremasterSecret")) {
455             // the instance variable "session" has been initialized for
456             // decrypt mode, so use a local variable instead.
457             Session s = null;
458             try {
459                 s = token.getObjSession();
460                 long keyType = CKK_GENERIC_SECRET;
461                 CK_ATTRIBUTE[] attributes = new CK_ATTRIBUTE[] {
462                     new CK_ATTRIBUTE(CKA_CLASS, CKO_SECRET_KEY),
463                     new CK_ATTRIBUTE(CKA_KEY_TYPE, keyType),
464                 };
465                 attributes = token.getAttributes
466                     (O_IMPORT, CKO_SECRET_KEY, keyType, attributes);
467                 long keyID = token.p11.C_UnwrapKey(s.id(),
468                         new CK_MECHANISM(mechanism), p11Key.keyID, wrappedKey,
469                         attributes);
470                 return P11Key.secretKey(s, keyID, algorithm, 48 << 3,
471                         attributes);
472             } catch (PKCS11Exception e) {
473                 throw new InvalidKeyException("unwrap() failed", e);
474             } finally {
475                 token.releaseSession(s);
476             }
477         }
478         // XXX implement unwrap using C_Unwrap() for all keys
479         implInit(Cipher.DECRYPT_MODE, p11Key);
480         if (wrappedKey.length > maxInputSize) {
481             throw new InvalidKeyException("Key is too long for unwrapping");
482         }
483         implUpdate(wrappedKey, 0, wrappedKey.length);
484         try {
485             byte[] encoded = doFinal();
486             return ConstructKeys.constructKey(encoded, algorithm, type);
487         } catch (BadPaddingException e) {
488             // should not occur
489             throw new InvalidKeyException("Unwrapping failed", e);
490         } catch (IllegalBlockSizeException e) {
491             // should not occur, handled with length check above
492             throw new InvalidKeyException("Unwrapping failed", e);
493         }
494     }
495 
496     // see JCE spec
497     protected int engineGetKeySize(Key key) throws InvalidKeyException {
498         int n = P11KeyFactory.convertKey(token, key, algorithm).keyLength();
499         return n;
500     }

503 final class ConstructKeys {
504     /**
505      * Construct a public key from its encoding.
506      *
507      * @param encodedKey the encoding of a public key.
508      *
509      * @param encodedKeyAlgorithm the algorithm the encodedKey is for.
510      *
511      * @return a public key constructed from the encodedKey.
512      */
513     private static final PublicKey constructPublicKey(byte[] encodedKey,
514             String encodedKeyAlgorithm)
515             throws InvalidKeyException, NoSuchAlgorithmException {
516         try {
517             KeyFactory keyFactory =
518                 KeyFactory.getInstance(encodedKeyAlgorithm);
519             X509EncodedKeySpec keySpec = new X509EncodedKeySpec(encodedKey);
520             return keyFactory.generatePublic(keySpec);
521         } catch (NoSuchAlgorithmException nsae) {
522             throw new NoSuchAlgorithmException("No installed providers " +
523                                                "can create keys for the " +
524                                                encodedKeyAlgorithm +
525                                                "algorithm", nsae);
526         } catch (InvalidKeySpecException ike) {
527             throw new InvalidKeyException("Cannot construct public key", ike);
528         }
529     }
530 
531     /**
532      * Construct a private key from its encoding.
533      *
534      * @param encodedKey the encoding of a private key.
535      *
536      * @param encodedKeyAlgorithm the algorithm the wrapped key is for.
537      *
538      * @return a private key constructed from the encodedKey.
539      */
540     private static final PrivateKey constructPrivateKey(byte[] encodedKey,
541             String encodedKeyAlgorithm) throws InvalidKeyException,
542             NoSuchAlgorithmException {
543         try {
544             KeyFactory keyFactory =
545                 KeyFactory.getInstance(encodedKeyAlgorithm);
546             PKCS8EncodedKeySpec keySpec = new PKCS8EncodedKeySpec(encodedKey);
547             return keyFactory.generatePrivate(keySpec);
548         } catch (NoSuchAlgorithmException nsae) {
549             throw new NoSuchAlgorithmException("No installed providers " +
550                                                "can create keys for the " +
551                                                encodedKeyAlgorithm +
552                                                "algorithm", nsae);
553         } catch (InvalidKeySpecException ike) {
554             throw new InvalidKeyException("Cannot construct private key", ike);
555         }
556     }
557 
558     /**
559      * Construct a secret key from its encoding.
560      *
561      * @param encodedKey the encoding of a secret key.
562      *
563      * @param encodedKeyAlgorithm the algorithm the secret key is for.
564      *
565      * @return a secret key constructed from the encodedKey.
566      */
567     private static final SecretKey constructSecretKey(byte[] encodedKey,
568             String encodedKeyAlgorithm) {
569         return new SecretKeySpec(encodedKey, encodedKeyAlgorithm);
570     }
571 
572     static final Key constructKey(byte[] encoding, String keyAlgorithm,
573             int keyType) throws InvalidKeyException, NoSuchAlgorithmException {
574         switch (keyType) {
575         case Cipher.SECRET_KEY:
576             return constructSecretKey(encoding, keyAlgorithm);
577         case Cipher.PRIVATE_KEY:
578             return constructPrivateKey(encoding, keyAlgorithm);
579         case Cipher.PUBLIC_KEY:
580             return constructPublicKey(encoding, keyAlgorithm);
581         default:
582             throw new InvalidKeyException("Unknown keytype " + keyType);
583         }
584     }

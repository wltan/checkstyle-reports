24  /*
25    @test
26    @bug       4685768
27    @summary   Tests that it's possible to manually request focus on a disabled component.
28    @author    Anton Tarasov: area=awt.focus
29    @library   ../../regtesthelpers
30    @build     Util
31    @run       main RequestFocusToDisabledCompTest
32  */
33  
34  import java.awt.Robot;
35  import javax.swing.*;
36  import java.awt.*;
37  import java.awt.event.*;
38  import java.applet.Applet;
39  import test.java.awt.regtesthelpers.Util;
40  
41  public class RequestFocusToDisabledCompTest extends Applet {
42      Robot robot;
43      JFrame frame = new JFrame("Frame");
44      JButton b0 = new JButton("b0");
45      JButton b1 = new JButton("b1");
46  
47      public static void main(String[] args) {
48          RequestFocusToDisabledCompTest app = new RequestFocusToDisabledCompTest();
49          app.init();
50          app.start();
51      }
52  
53      public void init() {
54          robot = Util.createRobot();
55          frame.add(b0);
56          frame.add(b1);
57          frame.setLayout(new FlowLayout());
58          frame.pack();
59  
60          b1.setEnabled(false);
61      }
62  
63      public void start() {
64          Util.showWindowWait(frame);
65  
66          if (!b0.hasFocus()) {
67              // Request focus on b0.
68              if (!Util.focusComponent(b0, 2000)) {
69                  throw new TestErrorException("couldn't focus " + b0);
70              }
71          }
72  
73          // Try to request focus on b1.
74          if (!Util.focusComponent(b1, 2000)) {
75              throw new TestFailedException("focus wasn't requested on disabled " + b1);
76          }
77          System.out.println("Test passed.");
78      }
79  }
80  
81  /**
82   * Thrown when the behavior being verified is found wrong.
83   */
84  class TestFailedException extends RuntimeException {
85      TestFailedException(String msg) {
86          super("Test failed: " + msg);
87      }
88  }
89  
90  /**
91   * Thrown when an error not related to the behavior being verified is encountered.
92   */
93  class TestErrorException extends RuntimeException {
94      TestErrorException(String msg) {
95          super("Unexpected error: " + msg);
96      }
97  }

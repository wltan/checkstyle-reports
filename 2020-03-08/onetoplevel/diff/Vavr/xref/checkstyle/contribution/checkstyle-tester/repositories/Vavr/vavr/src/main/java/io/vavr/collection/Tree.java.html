35      /**
36       * Returns a {@link java.util.stream.Collector} which may be used in conjunction with
37       * {@link java.util.stream.Stream#collect(java.util.stream.Collector)} to obtain a {@link Tree}.
38       *
39       * @param <T> Component type of the Tree.
40       * @return A io.vavr.collection.Tree Collector.
41       */
42      static <T> Collector<T, ArrayList<T>, Tree<T>> collector() {
43          final Supplier<ArrayList<T>> supplier = ArrayList::new;
44          final BiConsumer<ArrayList<T>, T> accumulator = ArrayList::add;
45          final BinaryOperator<ArrayList<T>> combiner = (left, right) -> {
46              left.addAll(right);
47              return left;
48          };
49          final Function<ArrayList<T>, Tree<T>> finisher = Tree::ofAll;
50          return Collector.of(supplier, accumulator, combiner, finisher);
51      }

149             final io.vavr.collection.List<T> list = io.vavr.collection.List.ofAll(iterable);
150             return list.isEmpty() ? Empty.instance() : new Node<>(list.head(), list.tail().map(Tree::of));
151         }

302     /**
303      * Traverses this tree in a specific order.
304      *
305      * @param order the tree traversal order
306      * @return A sequence of nodes.
307      * @throws java.lang.NullPointerException if order is null
308      */
309     default Seq<Node<T>> traverse(Order order) {
310         Objects.requireNonNull(order, "order is null");
311         if (isEmpty()) {
312             return Stream.empty();
313         } else {
314             final Node<T> node = (Node<T>) this;
315             switch (order) {
316                 case PRE_ORDER:
317                     return TreeModule.traversePreOrder(node);
318                 case IN_ORDER:
319                     return TreeModule.traverseInOrder(node);
320                 case POST_ORDER:
321                     return TreeModule.traversePostOrder(node);
322                 case LEVEL_ORDER:
323                     return TreeModule.traverseLevelOrder(node);
324                 default:
325                     throw new IllegalStateException("Unknown order: " + order.name());
326             }
327         }
328     }

352     /**
353      * Counts the number of branches of this tree. The empty tree and a leaf have no branches.
354      *
355      * @return The number of branches of this tree.
356      */
357     default int branchCount() {
358         if (isEmpty() || isLeaf()) {
359             return 0;
360         } else {
361             return getChildren().foldLeft(1, (count, child) -> count + child.branchCount());
362         }
363     }
364 
365     /**
366      * Counts the number of leaves of this tree. The empty tree has no leaves.
367      *
368      * @return The number of leaves of this tree.
369      */
370     default int leafCount() {
371         if (isEmpty()) {
372             return 0;
373         } else if (isLeaf()) {
374             return 1;
375         } else {
376             return getChildren().foldLeft(0, (count, child) -> count + child.leafCount());
377         }
378     }

846         private void drawAux(String indent, StringBuilder builder) {
847             builder.append(value);
848             for (io.vavr.collection.List<Node<T>> it = children; !it.isEmpty(); it = it.tail()) {
849                 final boolean isLast = it.tail().isEmpty();
850                 builder.append('\n')
851                         .append(indent)
852                         .append(isLast ? "└──" : "├──");
853                 it.head().drawAux(indent + (isLast ? "   " : "│  "), builder);
854             }

861             } else {

1103     static <T, U> Tree<U> flatMap(Node<T> node, Function<? super T, ? extends Iterable<? extends U>> mapper) {

1122     // Idea:
1123     // Traverse (depth-first) until a match is found, then stop and rebuild relevant parts of the tree.
1124     // If not found, return the same tree instance.
1125     static <T> Node<T> replace(Node<T> node, T currentElement, T newElement) {
1126         if (Objects.equals(node.getValue(), currentElement)) {
1127             return new Node<>(newElement, node.getChildren());
1128         } else {
1129             for (Node<T> child : node.getChildren()) {
1130                 final Node<T> newChild = replace(child, currentElement, newElement);
1131                 final boolean found = newChild != child;
1132                 if (found) {
1133                     final io.vavr.collection.List<Node<T>> newChildren = node.getChildren().replace(child, newChild);
1134                     return new Node<>(node.getValue(), newChildren);
1135                 }
1136             }
1137             return node;
1138         }
1139     }

1146     static <T> Stream<Node<T>> traverseInOrder(Node<T> node) {
1147         if (node.isLeaf()) {
1148             return Stream.of(node);
1149         } else {
1150             final io.vavr.collection.List<Node<T>> children = node.getChildren();
1151             return children
1152                     .tail()
1153                     .foldLeft(Stream.<Node<T>> empty(), (acc, child) -> acc.appendAll(traverseInOrder(child)))
1154                     .prepend(node)
1155                     .prependAll(traverseInOrder(children.head()));
1156         }

1161                 .getChildren()
1162                 .foldLeft(Stream.<Node<T>> empty(), (acc, child) -> acc.appendAll(traversePostOrder(child)))
1163                 .append(node);

1166     static <T> Stream<Node<T>> traverseLevelOrder(Node<T> node) {
1167         Stream<Node<T>> result = Stream.empty();
1168         final java.util.Queue<Node<T>> queue = new java.util.LinkedList<>();
1169         queue.add(node);
1170         while (!queue.isEmpty()) {
1171             final Node<T> next = queue.remove();
1172             result = result.prepend(next);
1173             queue.addAll(next.getChildren().toJavaList());
1174         }
1175         return result.reverse();
1176     }

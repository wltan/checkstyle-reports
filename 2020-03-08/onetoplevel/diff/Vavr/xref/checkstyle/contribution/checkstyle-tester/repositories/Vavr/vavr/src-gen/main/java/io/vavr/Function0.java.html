13  import static io.vavr.Function0Module.sneakyThrow;
14  
15  import io.vavr.control.Option;
16  import io.vavr.control.Try;
17  import java.util.Objects;
18  import java.util.function.Function;
19  import java.util.function.Supplier;
20  
21  /**
22   * Represents a function with no arguments.
23   *
24   * @param <R> return type of the function
25   * @author Daniel Dietrich
26   */
27  @FunctionalInterface
28  public interface Function0<R> extends Lambda<R>, Supplier<R> {
29  
30      /**
31       * The <a href="https://docs.oracle.com/javase/8/docs/api/index.html">serial version uid</a>.
32       */
33      long serialVersionUID = 1L;
34  
35      /**
36       * Creates a {@code Function0} based on
37       * <ul>
38       * <li><a href="https://docs.oracle.com/javase/tutorial/java/javaOO/methodreferences.html">method reference</a></li>
39       * <li><a href="https://docs.oracle.com/javase/tutorial/java/javaOO/lambdaexpressions.html#syntax">lambda expression</a></li>
40       * </ul>
41       *
42       * Examples (w.l.o.g. referring to Function1):
43       * <pre><code>// using a lambda expression
44       * Function1&lt;Integer, Integer&gt; add1 = Function1.of(i -&gt; i + 1);
45       *
46       * // using a method reference (, e.g. Integer method(Integer i) { return i + 1; })
47       * Function1&lt;Integer, Integer&gt; add2 = Function1.of(this::method);
48       *
49       * // using a lambda reference
50       * Function1&lt;Integer, Integer&gt; add3 = Function1.of(add1::apply);
51       * </code></pre>
52       * <p>
53       * <strong>Caution:</strong> Reflection loses type information of lambda references.
54       * <pre><code>// type of a lambda expression
55       * Type&lt;?, ?&gt; type1 = add1.getType(); // (Integer) -&gt; Integer
56       *
57       * // type of a method reference
58       * Type&lt;?, ?&gt; type2 = add2.getType(); // (Integer) -&gt; Integer
59       *
60       * // type of a lambda reference
61       * Type&lt;?, ?&gt; type3 = add3.getType(); // (Object) -&gt; Object
62       * </code></pre>
63       *
64       * @param methodReference (typically) a method reference, e.g. {@code Type::method}
65       * @param <R> return type
66       * @return a {@code Function0}
67       */
68      static <R> Function0<R> of(Function0<R> methodReference) {
69          return methodReference;
70      }
71  
72      /**
73       * Lifts the given {@code partialFunction} into a total function that returns an {@code Option} result.
74       *
75       * @param partialFunction a function that is not defined for all values of the domain (e.g. by throwing)
76       * @param <R> return type
77       * @return a function that applies arguments to the given {@code partialFunction} and returns {@code Some(result)}
78       *         if the function is defined for the given arguments, and {@code None} otherwise.
79       */
80      @SuppressWarnings("RedundantTypeArguments")
81      static <R> Function0<Option<R>> lift(Supplier<? extends R> partialFunction) {
82          return () -> Try.<R>of(partialFunction::get).toOption();
83      }
84  
85      /**
86       * Lifts the given {@code partialFunction} into a total function that returns an {@code Try} result.
87       *
88       * @param partialFunction a function that is not defined for all values of the domain (e.g. by throwing)
89       * @param <R> return type
90       * @return a function that applies arguments to the given {@code partialFunction} and returns {@code Success(result)}
91       *         if the function is defined for the given arguments, and {@code Failure(throwable)} otherwise.
92       */
93      static <R> Function0<Try<R>> liftTry(Supplier<? extends R> partialFunction) {
94          return () -> Try.of(partialFunction::get);
95      }
96  
97      /**
98       * Narrows the given {@code Supplier<? extends R>} to {@code Function0<R>}
99       *
100      * @param f A {@code Function0}
101      * @param <R> return type
102      * @return the given {@code f} instance as narrowed type {@code Function0<R>}
103      */
104     @SuppressWarnings("unchecked")
105     static <R> Function0<R> narrow(Supplier<? extends R> f) {
106         return (Function0<R>) f;
107     }
108 
109     /**
110      * Applies this function to no arguments and returns the result.
111      *
112      * @return the result of function application
113      * 
114      */
115     R apply();
116 
117     /**
118      * Implementation of {@linkplain java.util.function.Supplier#get()}, just calls {@linkplain #apply()}.
119      *
120      * @return the result of {@code apply()}
121      */
122     @Override
123     default R get() {
124         return apply();
125     }
126 
127     @Override
128     default int arity() {
129         return 0;
130     }
131 
132     /**
133      * Returns a function that always returns the constant
134      * value that you give in parameter.
135      *
136 
137      * @param <R> the result type
138      * @param value the value to be returned
139      * @return a function always returning the given value
140      */
141     static <R> Function0<R> constant(R value) {
142         return () -> value;
143     }
144 
145     @Override
146     default Function0<R> curried() {
147         return this;
148     }
149 
150     @Override
151     default Function1<Tuple0, R> tupled() {
152         return t -> apply();
153     }
154 
155     @Override
156     default Function0<R> reversed() {
157         return this;
158     }
159 
160     @Override
161     default Function0<R> memoized() {
162         if (isMemoized()) {
163             return this;
164         } else {
165             return (Function0<R> & Memoized) Lazy.of(this::apply)::get;
166         }
167     }
168 
169     /**
170      * Returns a composed function that first applies this Function0 to the given argument and then applies
171      * {@linkplain Function} {@code after} to the result.
172      *
173      * @param <V> return type of after
174      * @param after the function applied after this
175      * @return a function composed of this and after
176      * @throws NullPointerException if after is null
177      */
178     default <V> Function0<V> andThen(Function<? super R, ? extends V> after) {
179         Objects.requireNonNull(after, "after is null");
180         return () -> after.apply(apply());
181     }
182 
183 }
184 
185 interface Function0Module {
186 
187     // DEV-NOTE: we do not plan to expose this as public API
188     @SuppressWarnings("unchecked")
189     static <T extends Throwable, R> R sneakyThrow(Throwable t) throws T {
190         throw (T) t;
191     }
192 }

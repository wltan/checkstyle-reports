42      /**
43       * Creates a {@code CheckedFunction6} based on
44       * <ul>
45       * <li><a href="https://docs.oracle.com/javase/tutorial/java/javaOO/methodreferences.html">method reference</a></li>
46       * <li><a href="https://docs.oracle.com/javase/tutorial/java/javaOO/lambdaexpressions.html#syntax">lambda expression</a></li>
47       * </ul>
48       *
49       * Examples (w.l.o.g. referring to Function1):
50       * <pre><code>// using a lambda expression
51       * Function1&lt;Integer, Integer&gt; add1 = Function1.of(i -&gt; i + 1);
52       *
53       * // using a method reference (, e.g. Integer method(Integer i) { return i + 1; })
54       * Function1&lt;Integer, Integer&gt; add2 = Function1.of(this::method);
55       *
56       * // using a lambda reference
57       * Function1&lt;Integer, Integer&gt; add3 = Function1.of(add1::apply);
58       * </code></pre>
59       * <p>
60       * <strong>Caution:</strong> Reflection loses type information of lambda references.
61       * <pre><code>// type of a lambda expression
62       * Type&lt;?, ?&gt; type1 = add1.getType(); // (Integer) -&gt; Integer
63       *
64       * // type of a method reference
65       * Type&lt;?, ?&gt; type2 = add2.getType(); // (Integer) -&gt; Integer
66       *
67       * // type of a lambda reference
68       * Type&lt;?, ?&gt; type3 = add3.getType(); // (Object) -&gt; Object
69       * </code></pre>
70       *
71       * @param methodReference (typically) a method reference, e.g. {@code Type::method}
72       * @param <R> return type
73       * @param <T1> 1st argument
74       * @param <T2> 2nd argument
75       * @param <T3> 3rd argument
76       * @param <T4> 4th argument
77       * @param <T5> 5th argument
78       * @param <T6> 6th argument
79       * @return a {@code CheckedFunction6}
80       */
81      static <T1, T2, T3, T4, T5, T6, R> CheckedFunction6<T1, T2, T3, T4, T5, T6, R> of(CheckedFunction6<T1, T2, T3, T4, T5, T6, R> methodReference) {
82          return methodReference;
83      }

85      /**
86       * Lifts the given {@code partialFunction} into a total function that returns an {@code Option} result.
87       *
88       * @param partialFunction a function that is not defined for all values of the domain (e.g. by throwing)
89       * @param <R> return type
90       * @param <T1> 1st argument
91       * @param <T2> 2nd argument
92       * @param <T3> 3rd argument
93       * @param <T4> 4th argument
94       * @param <T5> 5th argument
95       * @param <T6> 6th argument
96       * @return a function that applies arguments to the given {@code partialFunction} and returns {@code Some(result)}
97       *         if the function is defined for the given arguments, and {@code None} otherwise.
98       */
99      @SuppressWarnings("RedundantTypeArguments")
100     static <T1, T2, T3, T4, T5, T6, R> Function6<T1, T2, T3, T4, T5, T6, Option<R>> lift(CheckedFunction6<? super T1, ? super T2, ? super T3, ? super T4, ? super T5, ? super T6, ? extends R> partialFunction) {
101         return (t1, t2, t3, t4, t5, t6) -> Try.<R>of(() -> partialFunction.apply(t1, t2, t3, t4, t5, t6)).toOption();
102     }

122     /**
123      * Narrows the given {@code CheckedFunction6<? super T1, ? super T2, ? super T3, ? super T4, ? super T5, ? super T6, ? extends R>} to {@code CheckedFunction6<T1, T2, T3, T4, T5, T6, R>}

133      * @return the given {@code f} instance as narrowed type {@code CheckedFunction6<T1, T2, T3, T4, T5, T6, R>}

137         return (CheckedFunction6<T1, T2, T3, T4, T5, T6, R>) f;

252     @Override
253     default CheckedFunction6<T1, T2, T3, T4, T5, T6, R> memoized() {
254         if (isMemoized()) {
255             return this;
256         } else {
257             final Map<Tuple6<T1, T2, T3, T4, T5, T6>, R> cache = new HashMap<>();
258             return (CheckedFunction6<T1, T2, T3, T4, T5, T6, R> & Memoized) (t1, t2, t3, t4, t5, t6)
259                     -> Memoized.of(cache, Tuple.of(t1, t2, t3, t4, t5, t6), t -> Try.of(() -> apply(t1, t2, t3, t4, t5, t6)).get());
260         }
261     }

263     /**
264      * Return a composed function that first applies this CheckedFunction6 to the given arguments and in case of throwable
265      * try to get value from {@code recover} function with same arguments and throwable information.
266      *
267      * @param recover the function applied in case of throwable
268      * @return a function composed of this and recover
269      * @throws NullPointerException if recover is null
270      */
271     default Function6<T1, T2, T3, T4, T5, T6, R> recover(Function<? super Throwable, ? extends Function6<? super T1, ? super T2, ? super T3, ? super T4, ? super T5, ? super T6, ? extends R>> recover) {
272         Objects.requireNonNull(recover, "recover is null");
273         return (t1, t2, t3, t4, t5, t6) -> {
274             try {
275                 return this.apply(t1, t2, t3, t4, t5, t6);
276             } catch (Throwable throwable) {
277                 final Function6<? super T1, ? super T2, ? super T3, ? super T4, ? super T5, ? super T6, ? extends R> func = recover.apply(throwable);
278                 Objects.requireNonNull(func, () -> "recover return null for " + throwable.getClass() + ": " + throwable.getMessage());
279                 return func.apply(t1, t2, t3, t4, t5, t6);
280             }
281         };
282     }

284     /**
285      * Returns an unchecked function that will <em>sneaky throw</em> if an exceptions occurs when applying the function.
286      *
287      * @return a new Function6 that throws a {@code Throwable}.
288      */
289     default Function6<T1, T2, T3, T4, T5, T6, R> unchecked() {
290         return (t1, t2, t3, t4, t5, t6) -> {
291             try {
292                 return apply(t1, t2, t3, t4, t5, t6);
293             } catch(Throwable t) {
294                 return sneakyThrow(t);
295             }
296         };
297     }

299     /**
300      * Returns a composed function that first applies this CheckedFunction6 to the given argument and then applies
301      * {@linkplain CheckedFunction1} {@code after} to the result.
302      *
303      * @param <V> return type of after
304      * @param after the function applied after this
305      * @return a function composed of this and after
306      * @throws NullPointerException if after is null
307      */
308     default <V> CheckedFunction6<T1, T2, T3, T4, T5, T6, V> andThen(CheckedFunction1<? super R, ? extends V> after) {
309         Objects.requireNonNull(after, "after is null");
310         return (t1, t2, t3, t4, t5, t6) -> after.apply(apply(t1, t2, t3, t4, t5, t6));
311     }

13  import static io.vavr.CheckedFunction2Module.sneakyThrow;
14  
15  import io.vavr.control.Option;
16  import io.vavr.control.Try;
17  import java.util.HashMap;
18  import java.util.Map;
19  import java.util.Objects;
20  import java.util.function.BiFunction;
21  import java.util.function.Function;

23  /**
24   * Represents a function with two arguments.
25   *
26   * @param <T1> argument 1 of the function
27   * @param <T2> argument 2 of the function
28   * @param <R> return type of the function
29   * @author Daniel Dietrich
30   */
31  @FunctionalInterface
32  public interface CheckedFunction2<T1, T2, R> extends Lambda<R> {
33  
34      /**
35       * The <a href="https://docs.oracle.com/javase/8/docs/api/index.html">serial version uid</a>.
36       */
37      long serialVersionUID = 1L;
38  
39      /**
40       * Creates a {@code CheckedFunction2} based on
41       * <ul>
42       * <li><a href="https://docs.oracle.com/javase/tutorial/java/javaOO/methodreferences.html">method reference</a></li>
43       * <li><a href="https://docs.oracle.com/javase/tutorial/java/javaOO/lambdaexpressions.html#syntax">lambda expression</a></li>
44       * </ul>
45       *
46       * Examples (w.l.o.g. referring to Function1):
47       * <pre><code>// using a lambda expression
48       * Function1&lt;Integer, Integer&gt; add1 = Function1.of(i -&gt; i + 1);
49       *
50       * // using a method reference (, e.g. Integer method(Integer i) { return i + 1; })
51       * Function1&lt;Integer, Integer&gt; add2 = Function1.of(this::method);
52       *
53       * // using a lambda reference
54       * Function1&lt;Integer, Integer&gt; add3 = Function1.of(add1::apply);
55       * </code></pre>
56       * <p>
57       * <strong>Caution:</strong> Reflection loses type information of lambda references.
58       * <pre><code>// type of a lambda expression
59       * Type&lt;?, ?&gt; type1 = add1.getType(); // (Integer) -&gt; Integer
60       *
61       * // type of a method reference
62       * Type&lt;?, ?&gt; type2 = add2.getType(); // (Integer) -&gt; Integer
63       *
64       * // type of a lambda reference
65       * Type&lt;?, ?&gt; type3 = add3.getType(); // (Object) -&gt; Object
66       * </code></pre>
67       *
68       * @param methodReference (typically) a method reference, e.g. {@code Type::method}
69       * @param <R> return type
70       * @param <T1> 1st argument
71       * @param <T2> 2nd argument
72       * @return a {@code CheckedFunction2}
73       */
74      static <T1, T2, R> CheckedFunction2<T1, T2, R> of(CheckedFunction2<T1, T2, R> methodReference) {
75          return methodReference;
76      }
77  
78      /**
79       * Lifts the given {@code partialFunction} into a total function that returns an {@code Option} result.
80       *
81       * @param partialFunction a function that is not defined for all values of the domain (e.g. by throwing)
82       * @param <R> return type
83       * @param <T1> 1st argument
84       * @param <T2> 2nd argument
85       * @return a function that applies arguments to the given {@code partialFunction} and returns {@code Some(result)}
86       *         if the function is defined for the given arguments, and {@code None} otherwise.
87       */
88      @SuppressWarnings("RedundantTypeArguments")
89      static <T1, T2, R> Function2<T1, T2, Option<R>> lift(CheckedFunction2<? super T1, ? super T2, ? extends R> partialFunction) {
90          return (t1, t2) -> Try.<R>of(() -> partialFunction.apply(t1, t2)).toOption();
91      }
92  
93      /**
94       * Lifts the given {@code partialFunction} into a total function that returns an {@code Try} result.
95       *
96       * @param partialFunction a function that is not defined for all values of the domain (e.g. by throwing)
97       * @param <R> return type
98       * @param <T1> 1st argument
99       * @param <T2> 2nd argument
100      * @return a function that applies arguments to the given {@code partialFunction} and returns {@code Success(result)}
101      *         if the function is defined for the given arguments, and {@code Failure(throwable)} otherwise.
102      */
103     static <T1, T2, R> Function2<T1, T2, Try<R>> liftTry(CheckedFunction2<? super T1, ? super T2, ? extends R> partialFunction) {
104         return (t1, t2) -> Try.of(() -> partialFunction.apply(t1, t2));
105     }
106 
107     /**
108      * Narrows the given {@code CheckedFunction2<? super T1, ? super T2, ? extends R>} to {@code CheckedFunction2<T1, T2, R>}
109      *
110      * @param f A {@code CheckedFunction2}
111      * @param <R> return type
112      * @param <T1> 1st argument
113      * @param <T2> 2nd argument
114      * @return the given {@code f} instance as narrowed type {@code CheckedFunction2<T1, T2, R>}
115      */
116     @SuppressWarnings("unchecked")
117     static <T1, T2, R> CheckedFunction2<T1, T2, R> narrow(CheckedFunction2<? super T1, ? super T2, ? extends R> f) {
118         return (CheckedFunction2<T1, T2, R>) f;
119     }
120 
121     /**
122      * Applies this function to two arguments and returns the result.
123      *
124      * @param t1 argument 1
125      * @param t2 argument 2
126      * @return the result of function application
127      * @throws Throwable if something goes wrong applying this function to the given arguments
128      */
129     R apply(T1 t1, T2 t2) throws Throwable;
130 
131     /**
132      * Applies this function partially to one argument.
133      *
134      * @param t1 argument 1
135      * @return a partial application of this function
136      */
137     default CheckedFunction1<T2, R> apply(T1 t1) {
138         return (T2 t2) -> apply(t1, t2);
139     }
140 
141     @Override
142     default int arity() {
143         return 2;
144     }
145 
146     /**
147      * Returns a function that always returns the constant
148      * value that you give in parameter.
149      *
150      * @param <T1> generic parameter type 1 of the resulting function
151      * @param <T2> generic parameter type 2 of the resulting function
152      * @param <R> the result type
153      * @param value the value to be returned
154      * @return a function always returning the given value
155      */
156     static <T1, T2, R> CheckedFunction2<T1, T2, R> constant(R value) {
157         return (t1, t2) -> value;
158     }
159 
160     @Override
161     default Function1<T1, CheckedFunction1<T2, R>> curried() {
162         return t1 -> t2 -> apply(t1, t2);
163     }
164 
165     @Override
166     default CheckedFunction1<Tuple2<T1, T2>, R> tupled() {
167         return t -> apply(t._1, t._2);
168     }
169 
170     @Override
171     default CheckedFunction2<T2, T1, R> reversed() {
172         return (t2, t1) -> apply(t1, t2);
173     }
174 
175     @Override
176     default CheckedFunction2<T1, T2, R> memoized() {
177         if (isMemoized()) {
178             return this;
179         } else {
180             final Map<Tuple2<T1, T2>, R> cache = new HashMap<>();
181             return (CheckedFunction2<T1, T2, R> & Memoized) (t1, t2)
182                     -> Memoized.of(cache, Tuple.of(t1, t2), t -> Try.of(() -> apply(t1, t2)).get());
183         }
184     }
185 
186     /**
187      * Return a composed function that first applies this CheckedFunction2 to the given arguments and in case of throwable
188      * try to get value from {@code recover} function with same arguments and throwable information.
189      *
190      * @param recover the function applied in case of throwable
191      * @return a function composed of this and recover
192      * @throws NullPointerException if recover is null
193      */
194     default Function2<T1, T2, R> recover(Function<? super Throwable, ? extends BiFunction<? super T1, ? super T2, ? extends R>> recover) {
195         Objects.requireNonNull(recover, "recover is null");
196         return (t1, t2) -> {
197             try {
198                 return this.apply(t1, t2);
199             } catch (Throwable throwable) {
200                 final BiFunction<? super T1, ? super T2, ? extends R> func = recover.apply(throwable);
201                 Objects.requireNonNull(func, () -> "recover return null for " + throwable.getClass() + ": " + throwable.getMessage());
202                 return func.apply(t1, t2);
203             }
204         };
205     }
206 
207     /**
208      * Returns an unchecked function that will <em>sneaky throw</em> if an exceptions occurs when applying the function.
209      *
210      * @return a new Function2 that throws a {@code Throwable}.
211      */
212     default Function2<T1, T2, R> unchecked() {
213         return (t1, t2) -> {
214             try {
215                 return apply(t1, t2);
216             } catch(Throwable t) {
217                 return sneakyThrow(t);
218             }
219         };
220     }
221 
222     /**
223      * Returns a composed function that first applies this CheckedFunction2 to the given argument and then applies
224      * {@linkplain CheckedFunction1} {@code after} to the result.
225      *
226      * @param <V> return type of after
227      * @param after the function applied after this
228      * @return a function composed of this and after
229      * @throws NullPointerException if after is null
230      */
231     default <V> CheckedFunction2<T1, T2, V> andThen(CheckedFunction1<? super R, ? extends V> after) {
232         Objects.requireNonNull(after, "after is null");
233         return (t1, t2) -> after.apply(apply(t1, t2));
234     }
235 
236 }
237 
238 interface CheckedFunction2Module {
239 
240     // DEV-NOTE: we do not plan to expose this as public API
241     @SuppressWarnings("unchecked")
242     static <T extends Throwable, R> R sneakyThrow(Throwable t) throws T {
243         throw (T) t;
244     }
245 }

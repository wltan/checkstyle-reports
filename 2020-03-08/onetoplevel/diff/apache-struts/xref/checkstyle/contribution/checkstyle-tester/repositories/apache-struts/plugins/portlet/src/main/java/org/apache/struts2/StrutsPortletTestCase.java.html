19  package org.apache.struts2;
20  
21  import com.opensymphony.xwork2.ActionContext;
22  import org.apache.logging.log4j.Logger;
23  import org.apache.logging.log4j.LogManager;
24  import org.apache.struts2.portlet.PortletConstants;
25  import org.apache.struts2.portlet.PortletPhase;
26  import org.springframework.mock.web.portlet.MockPortletContext;
27  import org.springframework.mock.web.portlet.MockPortletRequest;
28  import org.springframework.mock.web.portlet.MockPortletResponse;
29  import org.springframework.mock.web.portlet.MockPortletSession;
30  import org.springframework.mock.web.portlet.MockStateAwareResponse;
31  
32  import javax.portlet.PortletMode;
33  import java.util.HashMap;
34  import java.util.Map;
35  
36  /**
37   * Base class used to test action in portlet environment
38   */
39  public abstract class StrutsPortletTestCase extends StrutsTestCase {
40  
41      private static final Logger LOG = LogManager.getLogger(StrutsPortletTestCase.class);
42  
43      protected MockPortletSession portletSession;
44      protected MockPortletRequest portletRequest;
45      protected MockPortletResponse portletResponse;
46      protected MockContext portletContext;
47  
48      @Override
49      protected void initActionContext(ActionContext actionContext) {
50          super.initActionContext(actionContext);
51          initPortletContext(actionContext);
52      }
53  
54      protected void initPortletContext(ActionContext actionContext) {
55          LOG.debug("Initializing mock portlet environment");
56          portletContext = new MockContext();
57          portletContext.setMajorVersion(getMajorVersion());
58          actionContext.put(StrutsStatics.STRUTS_PORTLET_CONTEXT, portletContext);
59  
60          portletRequest = new MockPortletRequest(portletContext);
61          portletResponse = new MockStateAwareResponse();
62          portletSession = new MockPortletSession();
63          portletRequest.setSession(portletSession);
64          actionContext.setSession(createSession());
65          actionContext.put(PortletConstants.REQUEST, portletRequest);
66          actionContext.put(PortletConstants.RESPONSE, portletResponse);
67          actionContext.put(PortletConstants.MODE_NAMESPACE_MAP, new HashMap<PortletMode, String>());
68          actionContext.put(PortletConstants.PHASE, PortletPhase.EVENT_PHASE);
69      }
70  
71      /**
72       * Override to define version of your portlet environment
73       *
74       * @return portlet version
75       */
76      protected int getMajorVersion() {
77          return 2;
78      }
79  
80      /**
81       * Override to create your own session
82       *
83       * @return Map with session parameters
84       */
85      private Map<String, Object> createSession() {
86          return new HashMap<String, Object>(portletRequest.getPortletSession().getAttributeMap());
87      }
88  
89  }
90  
91  /**
92   * Simple workaround to define Portlet version
93   */
94  class MockContext extends MockPortletContext {
95  
96      private int majorVersion;
97  
98      @Override
99      public int getMajorVersion() {
100         return majorVersion;
101     }
102 
103     public void setMajorVersion(int majorVersion) {
104         this.majorVersion = majorVersion;
105     }
106 
107 }

33  /**
34   * This set of classes/interfaces (K/I/C) is specially designed to expose a
35   * bug in the JVM where it did not find some overloaded methods in some
36   * specific situations. (fixed by hotspot changeset ffb9316fd9ed)
37   */
38  interface K {
39      int bbb(Long l);
40  }
41  
42  interface I extends K {
43      default void aaa() {}
44      default void aab() {}
45      default void aac() {}
46  
47      default int bbb(Integer i) { return 22; }
48      default int bbb(Float f) { return 33; }
49      default int bbb(Long l) { return 44; }
50      default int bbb(Double d) { return 55; }
51      default int bbb(String s) { return 66; }
52  
53      default void caa() {}
54      default void cab() {}
55      default void cac() {}
56  }
57  
58  class C implements I {}
59  
60  public class DefaultMethodRegressionTests {
61  
62      @Test(groups = "vm")
63      public void testLostOverloadedMethod() {
64          C c = new C();
65          assertEquals(c.bbb(new Integer(1)), 22);
66          assertEquals(c.bbb(new Float(1.1)), 33);
67          assertEquals(c.bbb(new Long(1L)), 44);
68          assertEquals(c.bbb(new Double(0.01)), 55);
69          assertEquals(c.bbb(new String("")), 66);
70      }

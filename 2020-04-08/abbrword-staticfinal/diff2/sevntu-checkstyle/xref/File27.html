33  /**
34   * Prevents using wildcards as return type of methods.
35   * <p>
36   * <i>Joshua Bloch, "Effective Java (2nd edition)" Item 28: page 137 :</i>
37   * </p>
38   * <p>
39   * "Do not use wildcard types as return types. Rather than providing
40   * additional flexibility for your users,
41   * it would force them to use wildcard types in client code. Properly used,
42   * wildcard types are nearly invisible to users of a class. They cause methods
43   * to accept the parameters they should accept and reject those they should
44   * reject. If the user of a class has to think about wildcard types, there is
45   * probably something wrong with the classâ€™s API."
46   * Attention: some JDK classes have public methods with "?"(wildcard) in return type
47   * so it might not always possible to avoid wildcards in return type, as they do not demand user
48   * to bother about it (invisible for user or method). So suppressions should be used.
49   * </p>
50   * <p>
51   * Examples:
52   * <a href="https://docs.oracle.com/javase/8/docs/api/java/util/stream/Collectors.html">
53   * JDK Collectors</a>, so usage
54   * of methods that return wildcard could force user customizations over Collectors use wildcard in
55   * public methods
56   * </p>
57   * <pre>{@code
58   * // custom util method, wildcard come from Collectors.toList()
59   * public <T> Collector<T, ?, T> singleResult(Function<? super Iterable<T>, T> collector) {
60   *   return Collectors.collectingAndThen(Collectors.toList(), collected -> collected.get(0));
61   * }
62   * }</pre>
63   * <p>If suppressions become too wide spread and annoying it might be reasonable to update Check
64   * with option to ignore wildcard if used with another type (not alone).
65   * </p>
66   * @author <a href='mailto:barataliba@gmail.com'>Baratali Izmailov</a>
67   * @since 1.9.0
68   */
69  public class ForbidWildcardAsReturnTypeCheck extends AbstractCheck {
70  
71      /**
72       * Key for error message.
73       */
74      public static final String MSG_KEY = "forbid.wildcard.as.return.type";
75      /**
76       * Token of 'extends' keyword in bounded wildcard.
77       */
78      private static final int WILDCARD_EXTENDS_IDENT =
79              TokenTypes.TYPE_UPPER_BOUNDS;
80      /**
81       * Token of 'super' keyword in bounded wildcard.
82       */
83      private static final int WILDCARD_SUPER_IDENT =
84              TokenTypes.TYPE_LOWER_BOUNDS;

86      /** {@link Deprecated Deprecated} annotation name. */
87      private static final String DEPRECATED = "Deprecated";

89      /** Fully-qualified {@link Deprecated Deprecated} annotation name. */
90      private static final String FQ_DEPRECATED = "java.lang." + DEPRECATED;
91  
92      /** {@link Override Override} annotation name. */
93      private static final String OVERRIDE = "Override";
94  
95      /** Fully-qualified {@link Override Override} annotation name. */
96      private static final String FQ_OVERRIDE = "java.lang." + OVERRIDE;

102     /**
103      * Check methods with 'public' modifier.
104      */
105     private boolean checkPublicMethods = true;
106     /**
107      * Check methods with 'protected' modifier.

110     /**
111      * Check methods with 'package' modifier.

113     private boolean checkPackageMethods = true;
114     /**
115      * Check methods with 'private' modifier.

117     private boolean checkPrivateMethods;

121     private boolean checkOverrideMethods;
122     /**
123      * Check methods with @Deprecated annotation.

125     private boolean checkDeprecatedMethods;
126     /**
127      * Allow wildcard with 'super'. Example: "? super T"
128      */
129     private boolean allowReturnWildcardWithSuper;
130     /**
131      * Allow wildcard with 'extends'. Example: "? extends T"
132      */
133     private boolean allowReturnWildcardWithExtends;
134     /**
135      * Ignore regexp for return type class names.
136      */
137     private Pattern returnTypeClassNamesIgnoreRegex = Pattern.compile(
138             "^(Comparator|Comparable)$");

230     @Override
231     public void visitToken(DetailAST methodDefAst) {
232         final String methodScope = getVisibilityScope(methodDefAst);
233         if (isCheckableMethodScope(methodScope)
234                 && (checkOverrideMethods
235                         || !AnnotationUtil.containsAnnotation(methodDefAst, OVERRIDE)
236                             && !AnnotationUtil.containsAnnotation(methodDefAst, FQ_OVERRIDE))
237                 && (checkDeprecatedMethods
238                         || !AnnotationUtil.containsAnnotation(methodDefAst, DEPRECATED)
239                             && !AnnotationUtil.containsAnnotation(methodDefAst,
240                                 FQ_DEPRECATED))) {
241             final List<DetailAST> wildcardTypeArguments =
242                     getWildcardArgumentsAsMethodReturnType(methodDefAst);
243             if (!wildcardTypeArguments.isEmpty()
244                     && !isIgnoreCase(methodDefAst, wildcardTypeArguments)) {
245                 log(methodDefAst, MSG_KEY);
246             }
247         }
248     }

250     /**
251      * Checks if the method scope is defined as one of the types to check.
252      * @param methodScope The string version of the scope.
253      * @return {@code true} if the method should be checked.
254      */
255     private boolean isCheckableMethodScope(String methodScope) {
256         return checkPublicMethods && "public".equals(methodScope)
257                 || checkPrivateMethods && "private".equals(methodScope)
258                 || checkProtectedMethods && "protected".equals(methodScope)
259                 || checkPackageMethods && "package".equals(methodScope);
260     }
261 
262     /**
263      * Returns the visibility scope of method.
264      * @param methodDefAst DetailAST of method definition.
265      * @return one of "public", "private", "protected", "package"
266      */
267     private static String getVisibilityScope(DetailAST methodDefAst) {
268         String result = "package";
269         if (isInsideInterfaceDefinition(methodDefAst)) {
270             result = "public";
271         }
272         else {
273             final String[] visibilityScopeModifiers = {"public", "private",
274                 "protected", };
275             final Set<String> methodModifiers = getModifiers(methodDefAst);
276             for (final String modifier : visibilityScopeModifiers) {
277                 if (methodModifiers.contains(modifier)) {
278                     result = modifier;
279                     break;
280                 }
281             }
282         }
283         return result;
284     }

286     /**
287      * Verify that method definition is inside interface definition.
288      * @param methodDefAst DetailAST of method definition.
289      * @return true if method definition is inside interface definition.
290      */
291     private static boolean isInsideInterfaceDefinition(DetailAST methodDefAst) {
292         boolean result = false;
293         final DetailAST objBlock = methodDefAst.getParent();
294         final DetailAST interfaceDef = objBlock.getParent();
295         if (interfaceDef.getType() == TokenTypes.INTERFACE_DEF) {
296             result = true;
297         }
298         return result;
299     }

301     /**
302      * Returns the set of modifier Strings for a METHOD_DEF AST.
303      * @param methodDefAst AST for a method definition
304      * @return the set of modifier Strings for aMethodDefAST
305      */
306     private static Set<String> getModifiers(DetailAST methodDefAst) {
307         final DetailAST modifiersAst = methodDefAst.getFirstChild();
308         final Set<String> modifiersSet = new HashSet<>();
309         DetailAST modifierAst = modifiersAst.getFirstChild();
310         while (modifierAst != null) {
311             modifiersSet.add(modifierAst.getText());
312             modifierAst = modifierAst.getNextSibling();
313         }
314         return modifiersSet;
315     }

317     /**
318      * Get identifier of aAST.
319      * @param ast
320      *        DetailAST instance
321      * @return identifier of aAST, null if AST does not have identifier.
322      */
323     private static String getIdentifier(final DetailAST ast) {
324         final DetailAST identifier = ast.findFirstToken(TokenTypes.IDENT);
325         return identifier.getText();
326     }

328     /**
329      * Receive list of arguments(AST nodes) which have wildcard.
330      * @param methodDefAst
331      *        DetailAST of method definition.
332      * @return list of arguments which have wildcard.
333      */
334     private static List<DetailAST> getWildcardArgumentsAsMethodReturnType(DetailAST methodDefAst) {
335         final List<DetailAST> result = new LinkedList<>();
336         final DetailAST methodTypeAst =
337                 methodDefAst.findFirstToken(TokenTypes.TYPE);
338         final DetailAST[] methodTypeArgumentTokens =
339                 getGenericTypeArguments(methodTypeAst);
340         for (DetailAST typeArgumentAst: methodTypeArgumentTokens) {
341             if (hasChildToken(typeArgumentAst, TokenTypes.WILDCARD_TYPE)) {
342                 result.add(typeArgumentAst);
343             }
344         }
345         return result;
346     }
347 
348     /**
349      * Get all type arguments of TypeAST.
350      * @param typeAst
351      *        DetailAST of type definition.
352      * @return array of type arguments.
353      */
354     private static DetailAST[] getGenericTypeArguments(DetailAST typeAst) {
355         DetailAST[] result = EMPTY_DETAILAST_ARRAY;
356         if (hasChildToken(typeAst, TokenTypes.TYPE_ARGUMENTS)) {
357             final DetailAST typeArguments = typeAst
358                     .findFirstToken(TokenTypes.TYPE_ARGUMENTS);
359             final int argumentsCount = typeArguments
360                     .getChildCount(TokenTypes.TYPE_ARGUMENT);
361             result = new DetailAST[argumentsCount];
362             DetailAST firstTypeArgument = typeArguments
363                     .findFirstToken(TokenTypes.TYPE_ARGUMENT);
364             int counter = 0;
365             while (firstTypeArgument != null) {
366                 if (firstTypeArgument.getType() == TokenTypes.TYPE_ARGUMENT) {
367                     result[counter] = firstTypeArgument;
368                     counter++;
369                 }
370                 firstTypeArgument = firstTypeArgument.getNextSibling();
371             }
372         }
373         return result;
374     }

376     /**
377      * Verify that aAST has token of aTokenType type.
378      * @param ast
379      *        DetailAST instance.
380      * @param tokenType
381      *        one of TokenTypes
382      * @return true if aAST has token of given type, or false otherwise.
383      */
384     private static boolean hasChildToken(DetailAST ast, int tokenType) {
385         return ast.findFirstToken(tokenType) != null;
386     }

388     /**
389      * Verify that method with wildcards as return type is allowed by current
390      * check configuration.
391      * @param methodDefAst DetailAST of method definition.
392      * @param wildcardTypeArguments list of wildcard type arguments.
393      * @return true if method is allowed by current check configuration.
394      */
395     private boolean isIgnoreCase(DetailAST methodDefAst,
396             List<DetailAST> wildcardTypeArguments) {
397         boolean result = false;
398         if (matchesIgnoreClassNames(methodDefAst)) {
399             result = true;
400         }
401         else {
402             final boolean hasExtendsWildcardAsReturnType =
403                     hasBoundedWildcardAsReturnType(wildcardTypeArguments,
404                             WILDCARD_EXTENDS_IDENT);
405             final boolean hasSuperWildcardAsReturnType =
406                     hasBoundedWildcardAsReturnType(wildcardTypeArguments,
407                             WILDCARD_SUPER_IDENT);
408             final boolean hasOnlyExtendsWildcardAsReturnType =
409                     hasExtendsWildcardAsReturnType
410                     && !hasSuperWildcardAsReturnType;
411             final boolean hasOnlySuperWildcardAsReturnType =
412                     hasSuperWildcardAsReturnType
413                     && !hasExtendsWildcardAsReturnType;
414             final boolean hasBoundedWildcardAsReturnType =
415                     hasExtendsWildcardAsReturnType
416                     || hasSuperWildcardAsReturnType;
417             final boolean isAllowedBoundedWildcards =
418                     allowReturnWildcardWithExtends
419                     && allowReturnWildcardWithSuper;
420             result = isAllowedBoundedWildcards
421                             && hasBoundedWildcardAsReturnType
422                     || allowReturnWildcardWithExtends
423                             && hasOnlyExtendsWildcardAsReturnType
424                     || allowReturnWildcardWithSuper
425                             && hasOnlySuperWildcardAsReturnType;
426         }
427         return result;
428     }

430     /**
431      * Verify that method's return type name matches ignore regexp.
432      * @param methodDefAst DetailAST of method.
433      * @return true if aMethodDefAST's name matches ignore regexp.
434      *      false otherwise.
435      */
436     private boolean matchesIgnoreClassNames(DetailAST methodDefAst) {
437         final DetailAST methodTypeAst =
438                 methodDefAst.findFirstToken(TokenTypes.TYPE);
439         final String typeIdentifier = getIdentifier(methodTypeAst);
440         return returnTypeClassNamesIgnoreRegex
441                 .matcher(typeIdentifier).matches();
442     }
443 
444     /**
445      * Verify that method has bounded wildcard in type arguments list.
446      * @param typeArgumentsList list of type arguments.
447      * @param boundedWildcardType type of bounded wildcard.
448      * @return true if aTypeArgumentsList contains bounded wildcard.
449      */
450     private static boolean hasBoundedWildcardAsReturnType(
451             final List<DetailAST> typeArgumentsList, int boundedWildcardType) {
452         boolean result = false;
453         for (DetailAST typeArgumentAst: typeArgumentsList) {
454             if (hasChildToken(typeArgumentAst, boundedWildcardType)) {
455                 result = true;
456                 break;
457             }
458         }
459         return result;
460     }

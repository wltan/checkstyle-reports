22  import java.util.concurrent.atomic.AtomicInteger;
23  
24  import org.springframework.context.ApplicationEventPublisher;
25  import org.springframework.integration.context.OrderlyShutdownCapable;
26  import org.springframework.integration.gateway.MessagingGatewaySupport;

28  import org.springframework.integration.ip.tcp.connection.AbstractClientConnectionFactory;

36  import org.springframework.integration.ip.tcp.connection.TcpSender;

39  import org.springframework.messaging.support.ErrorMessage;
40  import org.springframework.util.Assert;

42  /**
43   * Inbound Gateway using a server connection factory - threading is controlled by the
44   * factory. For java.net connections, each socket can process only one message at a time.
45   * For java.nio connections, messages may be multiplexed but the client will need to
46   * provide correlation logic. If the client is a {@link TcpOutboundGateway} multiplexing
47   * is not used, but multiple concurrent connections can be used if the connection factory uses
48   * single-use connections. For true asynchronous bi-directional communication, a pair of
49   * inbound / outbound channel adapters should be used.
50   *
51   * @author Gary Russell
52   * @author Artem Bilan
53   *
54   * @since 2.0
55   */
56  public class TcpInboundGateway extends MessagingGatewaySupport implements
57  		TcpListener, TcpSender, ClientModeCapable, OrderlyShutdownCapable {
58  
59  	/**
60  	 * A default retry interval in milliseconds - {@value #DEFAULT_RETRY_INTERVAL}.
61  	 */
62  	public static final long DEFAULT_RETRY_INTERVAL = 60000;
63  
64  	private final Map<String, TcpConnection> connections = new ConcurrentHashMap<>();
65  
66  	private final AtomicInteger activeCount = new AtomicInteger();
67  
68  	private AbstractServerConnectionFactory serverConnectionFactory;
69  
70  	private AbstractClientConnectionFactory clientConnectionFactory;
71  
72  	private boolean isClientMode;
73  
74  	private boolean isSingleUse;
75  
76  	private long retryInterval = DEFAULT_RETRY_INTERVAL;
77  
78  	private volatile boolean active;
79  
80  	private volatile ClientModeConnectionManager clientModeConnectionManager;
81  
82  	private volatile ScheduledFuture<?> scheduledFuture;
83  
84  	private volatile boolean shuttingDown;
85  
86  	@Override
87  	public boolean onMessage(Message<?> message) {
88  		boolean isErrorMessage = message instanceof ErrorMessage;
89  		try {
90  			if (this.shuttingDown) {
91  				if (logger.isInfoEnabled()) {
92  					logger.info("Inbound message ignored; shutting down; " + message.toString());
93  				}
94  			}
95  			else {
96  				if (isErrorMessage) {
97  					/*
98  					 * Socket errors are sent here so they can be conveyed to any waiting thread.
99  					 * There's not one here; simply ignore.
100 					 */
101 					return false;
102 				}
103 				this.activeCount.incrementAndGet();
104 				try {
105 					return doOnMessage(message);
106 				}
107 				finally {
108 					this.activeCount.decrementAndGet();
109 				}
110 			}
111 			return false;
112 		}
113 		finally {
114 			String connectionId = (String) message.getHeaders().get(IpHeaders.CONNECTION_ID);
115 			if (connectionId != null && !isErrorMessage && this.isSingleUse) {
116 				if (this.serverConnectionFactory != null) {
117 					this.serverConnectionFactory.closeConnection(connectionId);
118 				}
119 				else {
120 					this.clientConnectionFactory.closeConnection(connectionId);
121 				}
122 			}
123 		}
124 	}
125 
126 	private boolean doOnMessage(Message<?> message) {
127 		Message<?> reply = sendAndReceiveMessage(message);
128 		if (reply == null) {
129 			if (logger.isDebugEnabled()) {
130 				logger.debug("null reply received for " + message + " nothing to send");
131 			}
132 			return false;
133 		}
134 		String connectionId = (String) message.getHeaders().get(IpHeaders.CONNECTION_ID);
135 		TcpConnection connection = null;
136 		if (connectionId != null) {
137 			connection = this.connections.get(connectionId);
138 		}
139 		if (connection == null) {
140 			publishNoConnectionEvent(message, connectionId);
141 			logger.error("Connection not found when processing reply " + reply + " for " + message);
142 			return false;
143 		}
144 		try {
145 			connection.send(reply);
146 		}
147 		catch (Exception e) {
148 			logger.error("Failed to send reply", e);
149 		}
150 		return false;
151 	}
152 
153 	private void publishNoConnectionEvent(Message<?> message, String connectionId) {
154 		AbstractConnectionFactory cf =
155 				this.serverConnectionFactory != null
156 						? this.serverConnectionFactory
157 						: this.clientConnectionFactory;
158 		ApplicationEventPublisher applicationEventPublisher = cf.getApplicationEventPublisher();
159 		if (applicationEventPublisher != null) {
160 			applicationEventPublisher.publishEvent(
161 					new TcpConnectionFailedCorrelationEvent(this, connectionId,
162 							new MessagingException(message, "Connection not found to process reply.")));
163 		}
164 	}
165 
166 	/**
167 	 * @return true if the associated connection factory is listening.
168 	 */
169 	public boolean isListening() {
170 		return this.serverConnectionFactory != null && this.serverConnectionFactory.isListening();
171 	}
172 
173 	/**
174 	 * Must be {@link AbstractClientConnectionFactory} or {@link AbstractServerConnectionFactory}.
175 	 * @param connectionFactory the Connection Factory
176 	 */
177 	public void setConnectionFactory(AbstractConnectionFactory connectionFactory) {
178 		Assert.notNull(connectionFactory, "Connection factory must not be null");
179 		if (connectionFactory instanceof AbstractServerConnectionFactory) {
180 			this.serverConnectionFactory = (AbstractServerConnectionFactory) connectionFactory;
181 		}
182 		else if (connectionFactory instanceof AbstractClientConnectionFactory) {
183 			this.clientConnectionFactory = (AbstractClientConnectionFactory) connectionFactory;
184 		}
185 		else {
186 			throw new IllegalArgumentException("Connection factory must be either an " +
187 					"AbstractServerConnectionFactory or an AbstractClientConnectionFactory");
188 		}
189 		connectionFactory.registerListener(this);
190 		connectionFactory.registerSender(this);
191 		this.isSingleUse = connectionFactory.isSingleUse();
192 	}
193 
194 	@Override
195 	public void addNewConnection(TcpConnection connection) {
196 		this.connections.put(connection.getConnectionId(), connection);
197 	}
198 
199 	@Override
200 	public void removeDeadConnection(TcpConnection connection) {
201 		this.connections.remove(connection.getConnectionId());
202 	}
203 
204 	@Override
205 	public String getComponentType() {
206 		return "ip:tcp-inbound-gateway";
207 	}
208 
209 	@Override
210 	protected void onInit() {
211 		super.onInit();
212 		if (this.isClientMode) {
213 			Assert.notNull(this.clientConnectionFactory,
214 					"For client-mode, connection factory must be type='client'");
215 			Assert.isTrue(!this.clientConnectionFactory.isSingleUse(),
216 					"For client-mode, connection factory must have single-use='false'");
217 		}
218 	}
219 
220 	@Override // protected by super#lifecycleLock
221 	protected void doStart() {
222 		super.doStart();
223 		if (!this.active) {
224 			this.active = true;
225 			this.shuttingDown = false;
226 			if (this.serverConnectionFactory != null) {
227 				this.serverConnectionFactory.start();
228 			}
229 			if (this.clientConnectionFactory != null) {
230 				this.clientConnectionFactory.start();
231 			}
232 			if (this.isClientMode) {
233 				ClientModeConnectionManager manager =
234 						new ClientModeConnectionManager(this.clientConnectionFactory);
235 				this.clientModeConnectionManager = manager;
236 				Assert.state(getTaskScheduler() != null, "Client mode requires a task scheduler");
237 				this.scheduledFuture = getTaskScheduler().scheduleAtFixedRate(manager, this.retryInterval);
238 			}
239 		}
240 	}
241 
242 	@Override // protected by super#lifecycleLock
243 	protected void doStop() {
244 		super.doStop();
245 		if (this.active) {
246 			this.active = false;
247 			if (this.scheduledFuture != null) {
248 				this.scheduledFuture.cancel(true);
249 			}
250 			this.clientModeConnectionManager = null;
251 			if (this.clientConnectionFactory != null) {
252 				this.clientConnectionFactory.stop();
253 			}
254 			if (this.serverConnectionFactory != null) {
255 				this.serverConnectionFactory.stop();
256 			}
257 		}
258 	}
259

265 		return this.isClientMode;
266 	}

269 	 * @param isClientMode

271 	 */
272 	public void setClientMode(boolean isClientMode) {
273 		this.isClientMode = isClientMode;
274 	}
275 
276 	/**
277 	 * @return the retryInterval
278 	 */
279 	public long getRetryInterval() {
280 		return this.retryInterval;
281 	}
282 
283 	/**
284 	 * Configure a retry interval.
285 	 * Defaults to {@link #DEFAULT_RETRY_INTERVAL}.
286 	 * @param retryInterval the retryInterval to set
287 	 */
288 	public void setRetryInterval(long retryInterval) {
289 		this.retryInterval = retryInterval;
290 	}
291 
292 	@Override
293 	public boolean isClientModeConnected() {
294 		if (this.isClientMode && this.clientModeConnectionManager != null) {
295 			return this.clientModeConnectionManager.isConnected();
296 		}
297 		else {
298 			return false;
299 		}
300 	}
301 
302 	@Override
303 	public void retryConnection() {
304 		if (this.active && this.isClientMode && this.clientModeConnectionManager != null) {
305 			this.clientModeConnectionManager.run();
306 		}
307 	}
308 
309 	@Override
310 	public int beforeShutdown() {
311 		this.shuttingDown = true;
312 		return this.activeCount.get();
313 	}
314 
315 	@Override
316 	public int afterShutdown() {
317 		this.stop();
318 		return this.activeCount.get();
319 	}

58  /**
59   * The abstract MongoDB {@link AbstractMessageGroupStore} implementation to provide configuration for common options

68  		implements InitializingBean, ApplicationContextAware {

71  
72  	protected final Log logger = LogFactory.getLog(getClass()); // NOSONAR - final

75  			new UnsupportedOperationException("The operation isn't implemented for this class.");

89  	public AbstractConfigurableMongoDbMessageStore(MongoTemplate mongoTemplate, String collectionName) {
90  		Assert.notNull(mongoTemplate, "'mongoTemplate' must not be null");
91  		Assert.hasText(collectionName, "'collectionName' must not be empty");
92  		this.collectionName = collectionName;
93  		this.mongoTemplate = mongoTemplate;
94  		this.mongoDbFactory = null;
95  	}

101 	public AbstractConfigurableMongoDbMessageStore(MongoDatabaseFactory mongoDbFactory,
102 			MappingMongoConverter mappingMongoConverter, String collectionName) {
103 		Assert.notNull(mongoDbFactory, "'mongoDbFactory' must not be null");
104 		Assert.hasText(collectionName, "'collectionName' must not be empty");
105 		this.collectionName = collectionName;
106 		this.mongoDbFactory = mongoDbFactory;
107 		this.mappingMongoConverter = mappingMongoConverter;
108 	}

131 	@Override
132 	public void afterPropertiesSet() {
133 		if (this.mongoTemplate == null) {
134 			if (this.mappingMongoConverter == null) {
135 				this.mappingMongoConverter = new MappingMongoConverter(new DefaultDbRefResolver(this.mongoDbFactory),
136 						new MongoMappingContext());
137 				this.mappingMongoConverter.setApplicationContext(this.applicationContext);
138 				List<Object> customConverters = new ArrayList<>();
139 				customConverters.add(new MessageToBinaryConverter());
140 				customConverters.add(new BinaryToMessageConverter());
141 				this.mappingMongoConverter.setCustomConversions(new MongoCustomConversions(customConverters));
142 				this.mappingMongoConverter.afterPropertiesSet();
143 			}
144 			this.mongoTemplate = new MongoTemplate(this.mongoDbFactory, this.mappingMongoConverter);
145 		}
146 
147 		this.messageBuilderFactory = IntegrationUtils.getMessageBuilderFactory(this.applicationContext);
148 
149 		IndexOperations indexOperations = this.mongoTemplate.indexOps(this.collectionName);
150 
151 		indexOperations.ensureIndex(new Index(MessageDocumentFields.MESSAGE_ID, Sort.Direction.ASC));
152 
153 		indexOperations.ensureIndex(new Index(MessageDocumentFields.GROUP_ID, Sort.Direction.ASC)
154 				.on(MessageDocumentFields.MESSAGE_ID, Sort.Direction.ASC)
155 				.unique());
156 
157 		indexOperations.ensureIndex(new Index(MessageDocumentFields.GROUP_ID, Sort.Direction.ASC)
158 				.on(MessageDocumentFields.LAST_MODIFIED_TIME, Sort.Direction.DESC)
159 				.on(MessageDocumentFields.SEQUENCE, Sort.Direction.DESC));
160 	}

162 	public Message<?> getMessage(UUID id) {
163 		Assert.notNull(id, "'id' must not be null");
164 		Query query = Query.query(Criteria.where(MessageDocumentFields.MESSAGE_ID).is(id));
165 		MessageDocument document = this.mongoTemplate.findOne(query, MessageDocument.class, this.collectionName);
166 		return document != null ? document.getMessage() : null;
167 	}
168 
169 	public MessageMetadata getMessageMetadata(UUID id) {
170 		Assert.notNull(id, "'id' must not be null");
171 		Query query = Query.query(Criteria.where(MessageDocumentFields.MESSAGE_ID).is(id));
172 		MessageDocument document = this.mongoTemplate.findOne(query, MessageDocument.class, this.collectionName);
173 		if (document != null) {
174 			MessageMetadata messageMetadata = new MessageMetadata(id);
175 			messageMetadata.setTimestamp(document.getCreatedTime());
176 			return messageMetadata;
177 		}
178 		else {
179 			return null;
180 		}
181 	}

183 	@Override
184 	public void removeMessageGroup(Object groupId) {
185 		this.mongoTemplate.remove(groupIdQuery(groupId), this.collectionName);
186 	}
187 
188 	@Override
189 	public int messageGroupSize(Object groupId) {
190 		long lCount = this.mongoTemplate.count(groupIdQuery(groupId), this.collectionName);
191 		Assert.isTrue(lCount <= Integer.MAX_VALUE, "Message count is out of Integer's range");
192 		return (int) lCount;
193 	}
194 
195 	/**
196 	 * Perform MongoDB {@code INC} operation for the document, which contains the {@link MessageDocument}
197 	 * {@code sequence}, and return the new incremented value for the new {@link MessageDocument}.
198 	 * The {@link #SEQUENCE_NAME} document is created on demand.
199 	 * @return the next sequence value.
200 	 */
201 	protected int getNextId() {
202 		Query query = Query.query(Criteria.where("_id").is(SEQUENCE_NAME));
203 		query.fields().include(MessageDocumentFields.SEQUENCE);
204 		return (Integer) this.mongoTemplate.findAndModify(query,
205 				new Update().inc(MessageDocumentFields.SEQUENCE, 1),
206 				FindAndModifyOptions.options().returnNew(true).upsert(true),
207 				Map.class, this.collectionName)
208 				.get(MessageDocumentFields.SEQUENCE); // NOSONAR - never returns null
209 	}
210 
211 	protected void addMessageDocument(final MessageDocument document) {
212 		if (document.getGroupCreatedTime() == 0) {
213 			document.setGroupCreatedTime(System.currentTimeMillis());
214 		}
215 		document.setCreatedTime(System.currentTimeMillis());
216 		try {
217 			this.mongoTemplate.insert(document, this.collectionName);
218 		}
219 		catch (DuplicateKeyException e) {
220 			if (this.logger.isDebugEnabled()) {
221 				this.logger.debug("The Message with id [" + document.getMessageId() + "] already exists.\n" +
222 						"Ignoring INSERT and SELECT existing...");
223 			}
224 		}
225 	}
226 
227 	protected static Query groupIdQuery(Object groupId) {
228 		return Query.query(Criteria.where(MessageDocumentFields.GROUP_ID).is(groupId));
229 	}

17  package org.springframework.integration.context;
18  
19  import java.io.IOException;
20  import java.util.Properties;
21  
22  import org.springframework.beans.factory.config.PropertiesFactoryBean;
23  import org.springframework.core.io.Resource;
24  import org.springframework.core.io.support.PathMatchingResourcePatternResolver;
25  import org.springframework.core.io.support.ResourcePatternResolver;
26  
27  /**
28   * Utility class to encapsulate infrastructure Integration properties constants and
29   * their default values from resources 'META-INF/spring.integration.default.properties'.
30   *
31   * @author Artem Bilan
32   *
33   * @since 3.0
34   */
35  public final class IntegrationProperties {
36  
37  	public static final String INTEGRATION_PROPERTIES_PREFIX = "spring.integration.";
38  
39  	/**
40  	 * Specifies whether to allow create automatically {@link org.springframework.integration.channel.DirectChannel}
41  	 * beans for non-declared channels or not.
42  	 */
43  	public static final String CHANNELS_AUTOCREATE = INTEGRATION_PROPERTIES_PREFIX + "channels.autoCreate";
44  
45  	/**
46  	 * Specifies the value for {@link org.springframework.integration.dispatcher.UnicastingDispatcher#maxSubscribers}
47  	 * in case of point-to-point channels (e.g. {@link org.springframework.integration.channel.ExecutorChannel}),
48  	 * if the attribute {@code max-subscribers} isn't configured on the channel component.
49  	 */
50  	public static final String CHANNELS_MAX_UNICAST_SUBSCRIBERS = INTEGRATION_PROPERTIES_PREFIX + "channels.maxUnicastSubscribers";
51  
52  	/**
53  	 * Specifies the value for {@link org.springframework.integration.dispatcher.BroadcastingDispatcher#maxSubscribers}
54  	 * in case of point-to-point channels (e.g. {@link org.springframework.integration.channel.PublishSubscribeChannel}),
55  	 * if the attribute {@code max-subscribers} isn't configured on the channel component.
56  	 */
57  	public static final String CHANNELS_MAX_BROADCAST_SUBSCRIBERS = INTEGRATION_PROPERTIES_PREFIX + "channels.maxBroadcastSubscribers";
58  
59  	/**
60  	 * Specifies the value of {@link org.springframework.scheduling.concurrent.ThreadPoolTaskScheduler#poolSize}
61  	 * for the {@code taskScheduler} bean initialized by the Integration infrastructure.
62  	 */
63  	public static final String TASK_SCHEDULER_POOL_SIZE = INTEGRATION_PROPERTIES_PREFIX + "taskScheduler.poolSize";
64  
65  	/**
66  	 * Specifies the value of {@link org.springframework.messaging.core.GenericMessagingTemplate#throwExceptionOnLateReply}.
67  	 */
68  	public static final String THROW_EXCEPTION_ON_LATE_REPLY = INTEGRATION_PROPERTIES_PREFIX + "messagingTemplate.throwExceptionOnLateReply";
69  
70  	/**
71  	 * Specifies the value of {@link org.springframework.integration.support.DefaultMessageBuilderFactory#readOnlyHeaders}.
72  	 */
73  	public static final String READ_ONLY_HEADERS = INTEGRATION_PROPERTIES_PREFIX + "readOnly.headers";
74  
75  	/**
76  	 * Specifies the value of {@link org.springframework.integration.endpoint.AbstractEndpoint#autoStartup}.
77  	 */
78  	public static final String ENDPOINTS_NO_AUTO_STARTUP = INTEGRATION_PROPERTIES_PREFIX + "endpoints.noAutoStartup";
79  
80  	private static Properties defaults;
81  
82  	static {
83  		String resourcePattern = "classpath*:META-INF/spring.integration.default.properties";
84  		try {
85  			ResourcePatternResolver resourceResolver = new PathMatchingResourcePatternResolver(IntegrationProperties.class.getClassLoader());
86  			Resource[] defaultResources = resourceResolver.getResources(resourcePattern);
87  
88  			PropertiesFactoryBean propertiesFactoryBean = new PropertiesFactoryBean();
89  			propertiesFactoryBean.setLocations(defaultResources);
90  			propertiesFactoryBean.afterPropertiesSet();
91  			defaults = propertiesFactoryBean.getObject();
92  		}
93  		catch (IOException e) {
94  			throw new IllegalStateException("Can't load '" + resourcePattern + "' resources.", e);
95  		}
96  	}
97  
98  	/**
99  	 * @return {@link Properties} with default values for Integration properties
100 	 *         from resources 'META-INF/spring.integration.default.properties'.
101 	 */
102 	public static Properties defaults() {
103 		return defaults;
104 	}
105 
106 	/**
107 	 * Build the bean property definition expression to resolve the value
108 	 * from Integration properties within the bean building phase.
109 	 *
110 	 * @param key the Integration property key.
111 	 * @return the bean property definition expression.
112 	 * @throws IllegalArgumentException if provided {@code key} isn't an Integration property.
113 	 */
114 	public static String getExpressionFor(String key) {
115 		if (defaults.containsKey(key)) {
116 			return "#{T(org.springframework.integration.context.IntegrationContextUtils).getIntegrationProperties(beanFactory).getProperty('" + key + "')}";
117 		}
118 		else {
119 			throw new IllegalArgumentException("The provided key [" + key + "] isn't the one of Integration properties: " + defaults.keySet());
120 		}
121 	}
122 
123 	private IntegrationProperties() {
124 	}
125 
126 }

17  package org.springframework.integration.channel;
18  
19  import java.util.Comparator;
20  import java.util.concurrent.PriorityBlockingQueue;
21  import java.util.concurrent.atomic.AtomicLong;
22  
23  import org.springframework.integration.StaticMessageHeaderAccessor;
24  import org.springframework.integration.store.MessageGroupQueue;
25  import org.springframework.integration.store.PriorityCapableChannelMessageStore;
26  import org.springframework.integration.util.UpperBound;
27  import org.springframework.lang.Nullable;
28  import org.springframework.messaging.Message;
29  import org.springframework.messaging.MessageHeaders;
30  
31  /**
32   * A message channel that prioritizes messages based on a {@link Comparator}.
33   * The default comparator is based upon the message header's 'priority'.
34   *
35   * @author Mark Fisher
36   * @author Oleg Zhurakousky
37   * @author Gary Russell
38   * @author Artem Bilan
39   */
40  public class PriorityChannel extends QueueChannel {
41  
42  	/**
43  	 * PriorityBlockingQueue#DEFAULT_INITIAL_CAPACITY is private
44  	 */
45  	private static final int DEFAULT_INITIAL_CAPACITY = 11;
46  
47  	private final UpperBound upperBound;
48  
49  	private final AtomicLong sequenceCounter = new AtomicLong();
50  
51  	private final boolean useMessageStore;
52  
53  	/**
54  	 * Create a channel with an unbounded queue. Message priority will be
55  	 * based on the value of {@link StaticMessageHeaderAccessor#getPriority(Message)}.
56  	 */
57  	public PriorityChannel() {
58  		this(0, null);
59  	}
60  
61  	/**
62  	 * Create a channel with the specified queue capacity. Message priority
63  	 * will be based upon the value of {@link StaticMessageHeaderAccessor#getPriority(Message)}.
64  	 * @param capacity The queue capacity.
65  	 */
66  	public PriorityChannel(int capacity) {
67  		this(capacity, null);
68  	}
69  
70  	/**
71  	 * Create a channel with an unbounded queue. Message priority will be
72  	 * determined by the provided {@link Comparator}. If the comparator
73  	 * is <code>null</code>, the priority will be based upon the value of
74  	 * {@link StaticMessageHeaderAccessor#getPriority(Message)}.
75  	 * @param comparator The comparator.
76  	 */
77  	public PriorityChannel(Comparator<Message<?>> comparator) {
78  		this(0, comparator);
79  	}
80  
81  	/**
82  	 * Create a channel with the specified queue capacity. If the capacity
83  	 * is a non-positive value, the queue will be unbounded. Message priority
84  	 * will be determined by the provided {@link Comparator}. If the comparator
85  	 * is <code>null</code>, the priority will be based upon the value of
86  	 * {@link StaticMessageHeaderAccessor#getPriority(Message)}.
87  	 * @param capacity The capacity.
88  	 * @param comparator The comparator.
89  	 */
90  	public PriorityChannel(int capacity, @Nullable Comparator<Message<?>> comparator) {
91  		super(new PriorityBlockingQueue<>(DEFAULT_INITIAL_CAPACITY, new SequenceFallbackComparator(comparator)));
92  		this.upperBound = new UpperBound(capacity);
93  		this.useMessageStore = false;
94  	}
95  
96  	/**
97  	 * Create a channel based on the provided {@link PriorityCapableChannelMessageStore}
98  	 * and group id for message store operations.
99  	 * @param messageGroupStore the {@link PriorityCapableChannelMessageStore} to use.
100 	 * @param groupId to group message for this channel in the message store.
101 	 * @since 5.0
102 	 */
103 	public PriorityChannel(PriorityCapableChannelMessageStore messageGroupStore, Object groupId) {
104 		this(new MessageGroupQueue(messageGroupStore, groupId));
105 	}
106 
107 	/**
108 	 * Create a channel based on the provided {@link MessageGroupQueue}.
109 	 * @param messageGroupQueue the {@link MessageGroupQueue} to use.
110 	 * @since 5.0
111 	 */
112 	public PriorityChannel(MessageGroupQueue messageGroupQueue) {
113 		super(messageGroupQueue);
114 		this.upperBound = new UpperBound(0);
115 		this.useMessageStore = true;
116 	}
117 
118 	@Override
119 	public int getRemainingCapacity() {
120 		return this.upperBound.availablePermits();
121 	}
122 
123 	@Override
124 	protected boolean doSend(Message<?> message, long timeout) {
125 		if (!this.upperBound.tryAcquire(timeout)) {
126 			return false;
127 		}
128 		if (!this.useMessageStore) {
129 			return super.doSend(new MessageWrapper(message), 0);
130 		}
131 		else {
132 			return super.doSend(message, 0);
133 		}
134 	}
135 
136 	@Override
137 	protected Message<?> doReceive(long timeout) {
138 		Message<?> message = super.doReceive(timeout);
139 		if (message != null) {
140 			if (!this.useMessageStore) {
141 				message = ((MessageWrapper) message).getRootMessage();
142 			}
143 			this.upperBound.release();
144 		}
145 		return message;
146 	}
147 
148 	private static final class SequenceFallbackComparator implements Comparator<Message<?>> {
149 
150 		private final Comparator<Message<?>> targetComparator;
151 
152 		SequenceFallbackComparator(@Nullable Comparator<Message<?>> targetComparator) {
153 			this.targetComparator = targetComparator;
154 		}
155 
156 		@Override
157 		public int compare(Message<?> message1, Message<?> message2) {
158 			int compareResult;
159 			if (this.targetComparator != null) {
160 				compareResult = this.targetComparator.compare(message1, message2);
161 			}
162 			else {
163 				Integer priority1 = StaticMessageHeaderAccessor.getPriority(message1);
164 				Integer priority2 = StaticMessageHeaderAccessor.getPriority(message2);
165 
166 				priority1 = priority1 != null ? priority1 : 0;
167 				priority2 = priority2 != null ? priority2 : 0;
168 				compareResult = priority2.compareTo(priority1);
169 			}
170 
171 			if (compareResult == 0) {
172 				Long sequence1 = ((MessageWrapper) message1).getSequence();
173 				Long sequence2 = ((MessageWrapper) message2).getSequence();
174 				compareResult = sequence1.compareTo(sequence2);
175 			}
176 			return compareResult;
177 		}
178 
179 	}
180 
181 	//we need this because of INT-2508
182 	private final class MessageWrapper implements Message<Object> {
183 
184 		private final Message<?> rootMessage;
185 
186 		private final long sequence;
187 
188 		MessageWrapper(Message<?> rootMessage) {
189 			this.rootMessage = rootMessage;
190 			this.sequence = PriorityChannel.this.sequenceCounter.incrementAndGet();
191 		}
192 
193 		public Message<?> getRootMessage() {
194 			return this.rootMessage;
195 		}
196 
197 		@Override
198 		public MessageHeaders getHeaders() {
199 			return this.rootMessage.getHeaders();
200 		}
201 
202 		@Override
203 		public Object getPayload() {
204 			return this.rootMessage.getPayload();
205 		}
206 
207 		long getSequence() {
208 			return this.sequence;
209 		}
210 
211 	}

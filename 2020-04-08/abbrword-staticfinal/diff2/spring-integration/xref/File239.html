82  /**
83   * <p>
84   * MBean exporter for Spring Integration components in an existing application. Add an instance of this as a bean
85   * definition in the same context as the components you need to monitor and all message channels and message handlers
86   * will be exposed.
87   * </p>
88   * <p>
89   * Channels will report metrics on send and receive (counts, rates, errors) and handlers will report metrics on
90   * execution duration. Channels will be registered under their name (bean id), if explicit, or the last part of their
91   * internal name (e.g. "nullChannel") if registered by the framework. A handler that is attached to an endpoint will be
92   * registered with the endpoint name (bean id) if there is one, otherwise under the name of the input channel. Handler
93   * object names contain a <code>bean</code> key that reports the source of the name: "endpoint" if the name is the
94   * endpoint id; "anonymous" if it is the input channel; and "handler" as a fallback, where the object name is just the
95   * <code>toString()</code> of the handler.
96   * </p>
97   * <p>
98   * This component is itself an MBean, reporting attributes concerning the names and object names of the channels and
99   * handlers. It doesn't register itself to avoid conflicts with the standard <code>&lt;context:mbean-export/&gt;</code>
100  * from Spring (which should therefore be used any time you need to expose those features).
101  * </p>
102  *
103  * @author Dave Syer
104  * @author Helena Edelson
105  * @author Oleg Zhurakousky
106  * @author Gary Russell
107  * @author Artem Bilan
108  * @author Meherzad Lahewala
109  */
110 @ManagedResource
111 @SuppressWarnings("deprecation")

113 		implements ApplicationContextAware, DestructionAwareBeanPostProcessor {

192 	/**
193 	 * Set the array of simple patterns for component names to register (defaults to '*').
194 	 * The pattern is applied to all components before they are registered, looking for a
195 	 * match on the 'name' property of the ObjectName. A MessageChannel and a
196 	 * MessageHandler (for instance) can share a name because they have a different type,
197 	 * so in that case they would either both be included or both excluded. Since version
198 	 * 4.2, a leading '!' negates the pattern match ('!foo*' means don't export components
199 	 * where the name matches the pattern 'foo*'). For components with names that match
200 	 * multiple patterns, the first pattern wins.
201 	 * @param componentNamePatterns the patterns.
202 	 */
203 	public void setComponentNamePatterns(String[] componentNamePatterns) {
204 		Assert.notEmpty(componentNamePatterns, "componentNamePatterns must not be empty");
205 		this.componentNamePatterns = Arrays.copyOf(componentNamePatterns, componentNamePatterns.length);
206 	}

224 		populateMessageHandlers();

234 			registerEndpoints();

236 			if (this.applicationContext

256 	private void populateMessageHandlers() {
257 		Map<String, org.springframework.integration.support.management.MessageHandlerMetrics> messageHandlers =
258 				this.applicationContext
259 					.getBeansOfType(org.springframework.integration.support.management.MessageHandlerMetrics.class);
260 		for (Entry<String, org.springframework.integration.support.management.MessageHandlerMetrics> entry
261 				: messageHandlers.entrySet()) {
262 
263 			String beanName = entry.getKey();
264 			org.springframework.integration.support.management.MessageHandlerMetrics bean = entry.getValue();
265 			if (this.handlerInAnonymousWrapper(bean) != null) {
266 				if (logger.isDebugEnabled()) {
267 					logger.debug("Skipping " + beanName + " because it wraps another handler");
268 				}
269 				continue;
270 			}
271 			// If the handler is proxied, we have to extract the target to expose as an MBean.
272 			// The MetadataMBeanInfoAssembler does not support JDK dynamic proxies.
273 			org.springframework.integration.support.management.MessageHandlerMetrics monitor =
274 					(org.springframework.integration.support.management.MessageHandlerMetrics) extractTarget(bean);
275 			this.handlers.add(monitor);
276 		}
277 	}

327 	@Override
328 	public Object postProcessAfterInitialization(Object bean, String beanName) throws BeansException {
329 		if (this.singletonsInstantiated) {
330 			try {
331 				if (bean instanceof org.springframework.integration.support.management.MessageChannelMetrics) {
332 					org.springframework.integration.support.management.MessageChannelMetrics monitor =
333 						(org.springframework.integration.support.management.MessageChannelMetrics) extractTarget(bean);
334 					this.channels.add(monitor);
335 					registerChannel(monitor);
336 					this.runtimeBeans.add(bean);
337 				}
338 				else if (bean instanceof MessageProducer && bean instanceof Lifecycle) {
339 					registerProducer((MessageProducer) bean);
340 					this.runtimeBeans.add(bean);
341 				}
342 				else if (bean instanceof AbstractEndpoint) {
343 					postProcessAbstractEndpoint(bean);
344 				}
345 			}
346 			catch (Exception e) {
347 				logger.error("Could not register an MBean for: " + beanName, e);
348 			}
349 		}
350 		return bean;
351 	}

353 	private void postProcessAbstractEndpoint(Object bean) {
354 		if (bean instanceof IntegrationConsumer) {
355 			IntegrationConsumer integrationConsumer = (IntegrationConsumer) bean;
356 			MessageHandler handler = integrationConsumer.getHandler();
357 			if (handler instanceof org.springframework.integration.support.management.MessageHandlerMetrics) {
358 				org.springframework.integration.support.management.MessageHandlerMetrics messageHandlerMetrics =
359 						(org.springframework.integration.support.management.MessageHandlerMetrics) extractTarget(handler);
360 				registerHandler(messageHandlerMetrics);
361 				this.handlers.add(messageHandlerMetrics);
362 				this.runtimeBeans.add(messageHandlerMetrics);
363 				return;
364 			}
365 		}
366 		else if (bean instanceof SourcePollingChannelAdapter) {
367 			SourcePollingChannelAdapter pollingChannelAdapter = (SourcePollingChannelAdapter) bean;
368 			MessageSource<?> messageSource = pollingChannelAdapter.getMessageSource();
369 			if (messageSource instanceof org.springframework.integration.support.management.MessageSourceMetrics) {
370 				org.springframework.integration.support.management.MessageSourceMetrics messageSourceMetrics =
371 						(org.springframework.integration.support.management.MessageSourceMetrics)
372 							extractTarget(messageSource);
373 				registerSource(messageSourceMetrics);
374 				this.sources.add(messageSourceMetrics);
375 				this.runtimeBeans.add(messageSourceMetrics);
376 				return;
377 			}
378 		}
379 
380 		registerEndpoint((AbstractEndpoint) bean);
381 		this.runtimeBeans.add(bean);
382 	}

400 	@Override
401 	public void postProcessBeforeDestruction(Object bean, String beanName) throws BeansException {
402 		if (this.runtimeBeans.remove(bean)) {
403 			ObjectName objectName = this.objectNames.remove(bean);
404 			if (objectName != null) {
405 				doUnregister(objectName);
406 				if (bean instanceof AbstractEndpoint) {
407 					this.endpointNames.remove(((AbstractEndpoint) bean).getComponentName());
408 				}
409 				else {
410 
411 					this.endpointsByMonitor.remove(bean);
412 					if (bean instanceof org.springframework.integration.support.management.MessageChannelMetrics) {
413 						this.channels.remove(bean);
414 						this.allChannelsByName.remove(((NamedComponent) bean).getComponentName());
415 					}
416 					else if (bean instanceof org.springframework.integration.support.management.MessageHandlerMetrics) {
417 						this.handlers.remove(bean);
418 						this.endpointNames.remove(((NamedComponent) bean).getComponentName());
419 					}
420 					else if (bean instanceof org.springframework.integration.support.management.MessageSourceMetrics) {
421 						this.sources.remove(bean);
422 						this.endpointNames.remove(((NamedComponent) bean).getComponentName());
423 						String managedName =
424 								((org.springframework.integration.support.management.MessageSourceMetrics) bean)
425 									.getManagedName();
426 						this.allSourcesByName.remove(managedName);
427 					}
428 				}
429 			}
430 			else if (bean instanceof MessageProducer && bean instanceof Lifecycle) {
431 				this.inboundLifecycleMessageProducers.remove(bean);
432 			}
433 		}
434 	}

436 	private MessageHandler handlerInAnonymousWrapper(final Object bean) {
437 		if (bean != null && bean.getClass().isAnonymousClass()) {
438 			final AtomicReference<MessageHandler> wrapped = new AtomicReference<>();
439 			ReflectionUtils.doWithFields(bean.getClass(), field -> {
440 				field.setAccessible(true);
441 				Object handler = field.get(bean);
442 				if (handler instanceof MessageHandler) {
443 					wrapped.set((MessageHandler) handler);
444 				}
445 			}, field -> wrapped.get() == null && field.getName().startsWith("val$"));
446 			return wrapped.get();
447 		}
448 		else {
449 			return null;
450 		}
451 	}

453 	/**
454 	 * Copy of private method in super class. Needed so we can avoid using the bean factory to extract the bean again,
455 	 * and risk it being a proxy (which it almost certainly is by now).
456 	 * @param bean the bean instance to register
457 	 * @param beanKey the bean name or human readable version if auto-generated
458 	 * @return the JMX object name of the MBean that was registered
459 	 */
460 	private ObjectName registerBeanInstance(Object bean, String beanKey) {
461 		try {
462 			ObjectName objectName = getObjectName(bean, beanKey);
463 			Object mbeanToExpose = null;
464 			if (isMBean(bean.getClass())) {
465 				mbeanToExpose = bean;
466 			}
467 			else {
468 				DynamicMBean adaptedBean = adaptMBeanIfPossible(bean);
469 				if (adaptedBean != null) {
470 					mbeanToExpose = adaptedBean;
471 				}
472 			}
473 			if (mbeanToExpose != null) {
474 				if (logger.isInfoEnabled()) {
475 					logger.info("Located MBean '" + beanKey + "': registering with JMX server as MBean [" + objectName
476 							+ "]");
477 				}
478 				doRegister(mbeanToExpose, objectName);
479 			}
480 			else {
481 				if (logger.isInfoEnabled()) {
482 					logger.info("Located managed bean '" + beanKey + "': registering with JMX server as MBean ["
483 							+ objectName + "]");
484 				}
485 				ModelMBean mbean = createAndConfigureMBean(bean, beanKey);
486 				doRegister(mbean, objectName);
487 				// injectNotificationPublisherIfNecessary(bean, mbean, objectName);
488 			}
489 			return objectName;
490 		}
491 		catch (JMException e) {
492 			throw new UnableToRegisterMBeanException("Unable to register MBean [" + bean + "] with key '" + beanKey
493 					+ "'", e);
494 		}
495 	}

511 	/**
512 	 * Shutdown active components.
513 	 * @param howLong The time to wait in total for all activities to complete
514 	 * in milliseconds.
515 	 */
516 	@ManagedOperation
517 	public void stopActiveComponents(long howLong) {
518 		if (!this.shuttingDown.compareAndSet(false, true)) {
519 			logger.error("Shutdown already in process");
520 			return;
521 		}
522 		this.shutdownDeadline = System.currentTimeMillis() + howLong;
523 		try {
524 			logger.debug("Running shutdown");
525 			doShutdown();
526 		}
527 		catch (Exception e) {
528 			logger.error("Orderly shutdown failed", e);
529 		}
530 	}
531 
532 	/**
533 	 * Perform orderly shutdown - called or executed from
534 	 * {@link #stopActiveComponents(long)}.
535 	 */
536 	private void doShutdown() {
537 		try {
538 			orderlyShutdownCapableComponentsBefore();
539 			stopActiveChannels();
540 			stopMessageSources();
541 			stopInboundMessageProducers();
542 			// Wait any remaining time for messages to quiesce
543 			long timeLeft = this.shutdownDeadline - System.currentTimeMillis();
544 			if (timeLeft > 0) {
545 				try {
546 					Thread.sleep(timeLeft);
547 				}
548 				catch (InterruptedException e) {
549 					Thread.currentThread().interrupt();
550 					logger.error("Interrupted while waiting for quiesce");
551 				}
552 			}
553 			orderlyShutdownCapableComponentsAfter();
554 		}
555 		finally {
556 			this.shuttingDown.set(false);
557 		}
558 	}

561 	 * Stops all message sources - may cause interrupts.
562 	 */
563 	@ManagedOperation
564 	public void stopMessageSources() {
565 		for (org.springframework.integration.support.management.MessageSourceMetrics sourceMetrics
566 				: this.allSourcesByName.values()) {
567 
568 			if (sourceMetrics instanceof Lifecycle) {
569 				if (logger.isInfoEnabled()) {
570 					logger.info("Stopping message source " + sourceMetrics);
571 				}
572 				((Lifecycle) sourceMetrics).stop();
573 			}
574 			else {
575 				if (logger.isInfoEnabled()) {
576 					logger.info("Message source " + sourceMetrics + " cannot be stopped");
577 				}
578 			}
579 		}
580 	}

584 	 * - may cause interrupts.

598 	@ManagedOperation

825 	private void registerEndpoint(AbstractEndpoint endpoint) {
826 		String beanKey;
827 		String name = endpoint.getComponentName();
828 		String source;
829 		if (name.startsWith('_' + IntegrationConfigUtils.BASE_PACKAGE)) {
830 			name = getInternalComponentName(name);
831 			source = "internal";
832 		}
833 		else {
834 			source = "endpoint";
835 		}
836 		if (matches(this.componentNamePatterns, name)) {
837 			if (this.endpointNames.contains(name)) {
838 				int count = 0;
839 				String unique = name + "#" + count;
840 				while (this.endpointNames.contains(unique)) {
841 					unique = name + "#" + (++count);
842 				}
843 				name = unique;
844 			}
845 			this.endpointNames.add(name);
846 			beanKey = getEndpointBeanKey(name, source);
847 			ObjectName objectName = registerBeanInstance(new ManagedEndpoint(endpoint), beanKey);
848 			this.objectNames.put(endpoint, objectName);
849 			if (logger.isInfoEnabled()) {
850 				logger.info("Registered endpoint without MessageSource: " + objectName);
851 			}
852 		}
853 	}

856 	 * Simple pattern match against the supplied patterns; also supports negated ('!')
857 	 * patterns. First match wins (positive or negative).
858 	 * @param patterns the patterns.
859 	 * @param name the name to match.
860 	 * @return true if positive match, false if no match or negative match.
861 	 */
862 	private boolean matches(String[] patterns, String name) {
863 		Boolean match = PatternMatchUtils.smartMatch(name, patterns);
864 		return match == null ? false : match;

932 	@SuppressWarnings("unlikely-arg-type")
933 	private org.springframework.integration.support.management.MessageHandlerMetrics enhanceHandlerMonitor(
934 			org.springframework.integration.support.management.MessageHandlerMetrics monitor) {
935 
936 		if (monitor.getManagedName() != null && monitor.getManagedType() != null) {
937 			return monitor;
938 		}
939 
940 		// Assignment algorithm and bean id, with bean id pulled reflectively out of enclosing endpoint if possible
941 		String[] names = this.applicationContext.getBeanNamesForType(IntegrationConsumer.class);
942 
943 		String name = null;
944 		String endpointName = null;
945 		String source = "endpoint";
946 		IntegrationConsumer endpoint = null;
947 
948 		for (String beanName : names) {
949 			endpoint = this.applicationContext.getBean(beanName, IntegrationConsumer.class);
950 			try {
951 				MessageHandler handler = endpoint.getHandler();
952 				if (handler.equals(monitor) ||
953 						extractTarget(handlerInAnonymousWrapper(handler)).equals(monitor)) {
954 					name = beanName;
955 					endpointName = beanName;
956 					break;
957 				}
958 			}
959 			catch (Exception e) {
960 				logger.trace("Could not get handler from bean = " + beanName);
961 				endpoint = null;
962 			}
963 		}
964 
965 		org.springframework.integration.support.management.MessageHandlerMetrics messageHandlerMetrics =
966 				buildMessageHandlerMetrics(monitor, name, source, endpoint);
967 		if (endpointName != null) {
968 			this.endpointsByMonitor.put(messageHandlerMetrics, endpointName);
969 		}
970 		return messageHandlerMetrics;
971 	}

973 	private org.springframework.integration.support.management.MessageHandlerMetrics buildMessageHandlerMetrics(
974 			org.springframework.integration.support.management.MessageHandlerMetrics monitor,
975 			String name, String source, IntegrationConsumer endpoint) {
976 
977 		org.springframework.integration.support.management.MessageHandlerMetrics result = monitor;
978 		String managedType = source;
979 		String managedName = name;
980 
981 		if (managedName != null && managedName.startsWith('_' + IntegrationConfigUtils.BASE_PACKAGE)) {
982 			managedName = getInternalComponentName(managedName);
983 			managedType = "internal";
984 		}
985 		if (managedName != null && name.startsWith(IntegrationConfigUtils.BASE_PACKAGE)) {
986 			MessageChannel inputChannel = endpoint.getInputChannel();
987 			if (inputChannel != null) {
988 				managedName = buildAnonymousManagedName(this.anonymousHandlerCounters, inputChannel);
989 				managedType = "anonymous";
990 			}
991 		}
992 
993 		if (endpoint instanceof Lifecycle) {
994 			result = wrapMessageHandlerInLifecycleMetrics(monitor, (Lifecycle) endpoint);
995 		}
996 
997 		if (managedName == null) {
998 			if (monitor instanceof NamedComponent) {
999 				managedName = ((NamedComponent) monitor).getComponentName();
1000 			}
1001 			if (managedName == null) {
1002 				managedName = monitor.toString();
1003 			}
1004 			managedType = "handler";
1005 		}
1006 
1007 		result.setManagedType(managedType);
1008 		result.setManagedName(managedName);
1009 		return result;
1010 	}

1012 	private String buildAnonymousManagedName(Map<Object, AtomicLong> anonymousCache, MessageChannel messageChannel) {
1013 		AtomicLong count = anonymousCache.computeIfAbsent(messageChannel, (key) -> new AtomicLong());
1014 		long total = count.incrementAndGet();
1015 		/*
1016 		 * Short hack to makes sure object names are unique if more than one endpoint has the same input
1017 		 * channel
1018 		 */
1019 
1020 		String channelName =
1021 				messageChannel instanceof NamedComponent
1022 						? ((NamedComponent) messageChannel).getBeanName()
1023 						: messageChannel.toString();
1024 
1025 		return channelName + (total > 1 ? "#" + total : "");
1026 	}

1063 	private org.springframework.integration.support.management.MessageSourceMetrics enhanceSourceMonitor(
1064 			org.springframework.integration.support.management.MessageSourceMetrics monitor) {
1065 
1066 		if (monitor.getManagedName() != null) {
1067 			return monitor;
1068 		}
1069 
1070 		String endpointName = null;
1071 		String source = "endpoint";
1072 		AbstractEndpoint endpoint = getEndpointForMonitor(monitor);
1073 
1074 		if (endpoint != null) {
1075 			endpointName = endpoint.getBeanName();
1076 		}
1077 		if (endpointName != null && endpointName.startsWith('_' + IntegrationConfigUtils.BASE_PACKAGE)) {
1078 			endpointName = getInternalComponentName(endpointName);
1079 			source = "internal";
1080 		}
1081 
1082 		org.springframework.integration.support.management.MessageSourceMetrics messageSourceMetrics =
1083 				buildMessageSourceMetricsIfAny(monitor, endpointName, source, endpoint);
1084 		if (endpointName != null) {
1085 			this.endpointsByMonitor.put(messageSourceMetrics, endpointName);
1086 		}
1087 		return messageSourceMetrics;

1090 	@SuppressWarnings("unlikely-arg-type")
1091 	private AbstractEndpoint getEndpointForMonitor(
1092 			org.springframework.integration.support.management.MessageSourceMetrics monitor) {
1093 
1094 		for (AbstractEndpoint endpoint : this.applicationContext.getBeansOfType(AbstractEndpoint.class).values()) {
1095 			Object target = null;
1096 			if (monitor instanceof MessagingGatewaySupport && endpoint.equals(monitor)) {
1097 				target = monitor;
1098 			}
1099 			else if (endpoint instanceof SourcePollingChannelAdapter) {
1100 				target = ((SourcePollingChannelAdapter) endpoint).getMessageSource();
1101 			}
1102 			if (monitor.equals(target)) {
1103 				return endpoint;
1104 			}
1105 		}
1106 		return null;
1107 	}

1109 	private org.springframework.integration.support.management.MessageSourceMetrics buildMessageSourceMetricsIfAny(
1110 			org.springframework.integration.support.management.MessageSourceMetrics monitor, String name,
1111 			String source, Object endpoint) {
1112 
1113 		org.springframework.integration.support.management.MessageSourceMetrics result = monitor;
1114 		String managedType = source;
1115 		String managedName = name;
1116 
1117 		if (managedName != null && managedName.startsWith(IntegrationConfigUtils.BASE_PACKAGE)) {
1118 			Object target = endpoint;
1119 			if (endpoint instanceof Advised) {
1120 				TargetSource targetSource = ((Advised) endpoint).getTargetSource();
1121 				try {
1122 					target = targetSource.getTarget();
1123 				}
1124 				catch (Exception e) {
1125 					logger.error("Could not get handler from bean = " + managedName);
1126 				}
1127 			}
1128 
1129 			MessageChannel outputChannel = null;
1130 			if (target instanceof MessagingGatewaySupport) {
1131 				outputChannel = ((MessagingGatewaySupport) target).getRequestChannel();
1132 			}
1133 			else if (target instanceof SourcePollingChannelAdapter) {
1134 				outputChannel = ((SourcePollingChannelAdapter) target).getOutputChannel();
1135 			}
1136 
1137 			if (outputChannel != null) {
1138 				managedName = buildAnonymousManagedName(this.anonymousSourceCounters, outputChannel);
1139 				managedType = "anonymous";
1140 			}
1141 		}
1142 
1143 		if (endpoint instanceof Lifecycle) {
1144 			result = wrapMessageSourceInLifecycleMetrics(result, endpoint);
1145 		}
1146 
1147 		if (managedName == null) {
1148 			managedName = result.toString();
1149 			managedType = "source";
1150 		}
1151 
1152 		result.setManagedType(managedType);
1153 		result.setManagedName(managedName);
1154 		return result;
1155 	}

1171 			}

1174 			if (monitor instanceof MessageSourceManagement) {
1175 				result = new LifecycleMessageSourceManagement((Lifecycle) endpoint,
1176 						(MessageSourceManagement) monitor);
1177 			}

1179 				result = new org.springframework.integration.support.management.LifecycleMessageSourceMetrics(
1180 						(Lifecycle) endpoint, monitor);
1181 			}
1182 		}

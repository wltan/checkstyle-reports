122 	@SuppressWarnings("unchecked")
123 	public AbstractMethodAnnotationPostProcessor(ConfigurableListableBeanFactory beanFactory) {
124 		Assert.notNull(beanFactory, "'beanFactory' must not be null");
125 		this.messageHandlerAttributes.add(SEND_TIMEOUT_ATTRIBUTE);
126 		this.beanFactory = beanFactory;
127 		this.conversionService = this.beanFactory.getConversionService() != null
128 				? this.beanFactory.getConversionService()
129 				: DefaultConversionService.getSharedInstance();
130 		this.channelResolver = ChannelResolverUtils.getChannelResolver(beanFactory);
131 		this.annotationType =
132 				(Class<T>) GenericTypeResolver.resolveTypeArgument(this.getClass(),
133 						MethodAnnotationPostProcessor.class);
134 		Disposables disposablesBean = null;
135 		try {
136 			disposablesBean = beanFactory.getBean(Disposables.class);
137 		}
138 		catch (@SuppressWarnings("unused") Exception e) {
139 			// NOSONAR - only for test cases
140 		}
141 		this.disposables = disposablesBean;
142 	}

145 	@Override
146 	public Object postProcess(Object bean, String beanName, Method method, List<Annotation> annotations) {
147 		Object sourceHandler = null;
148 		if (beanAnnotationAware() && AnnotatedElementUtils.isAnnotated(method, Bean.class.getName())) {
149 			if (!this.beanFactory.containsBeanDefinition(resolveTargetBeanName(method))) {
150 				this.logger.debug("Skipping endpoint creation; perhaps due to some '@Conditional' annotation.");
151 				return null;
152 			}
153 			else {
154 				sourceHandler = resolveTargetBeanFromMethodWithBeanAnnotation(method);
155 			}
156 		}
157 
158 		MessageHandler handler = createHandler(bean, method, annotations);
159 
160 		if (!(handler instanceof ReactiveMessageHandlerAdapter)) {
161 			orderable(method, handler);
162 			producerOrRouter(annotations, handler);
163 
164 			if (!handler.equals(sourceHandler)) {
165 				handler = registerHandlerBean(beanName, method, handler);
166 			}
167 
168 			handler = annotated(method, handler);
169 			handler = adviceChain(beanName, annotations, handler);
170 		}
171 
172 		AbstractEndpoint endpoint = createEndpoint(handler, method, annotations);
173 		if (endpoint != null) {
174 			return endpoint;
175 		}
176 		else {
177 			return handler;
178 		}
179 	}

181 	private MessageHandler registerHandlerBean(String beanName, Method method, final MessageHandler handler) {
182 		MessageHandler handlerBean = handler;
183 		String handlerBeanName = generateHandlerBeanName(beanName, method);
184 		if (handlerBean instanceof ReplyProducingMessageHandlerWrapper
185 				&& StringUtils.hasText(MessagingAnnotationUtils.endpointIdValue(method))) {
186 			handlerBeanName = handlerBeanName + ".wrapper";
187 		}
188 		if (handlerBean instanceof IntegrationObjectSupport) {
189 			((IntegrationObjectSupport) handlerBean).setComponentName(
190 					handlerBeanName.substring(0,
191 							handlerBeanName.indexOf(IntegrationConfigUtils.HANDLER_ALIAS_SUFFIX)));
192 		}
193 		this.beanFactory.registerSingleton(handlerBeanName, handler);
194 		handlerBean = (MessageHandler) this.beanFactory.initializeBean(handlerBean, handlerBeanName);
195 		if (handlerBean instanceof DisposableBean && this.disposables != null) {
196 			this.disposables.add((DisposableBean) handlerBean);
197 		}
198 		return handlerBean;
199 	}

202 	private void orderable(Method method, MessageHandler handler) {
203 		if (handler instanceof Orderable) {
204 			Order orderAnnotation = AnnotationUtils.findAnnotation(method, Order.class);
205 			if (orderAnnotation != null) {
206 				((Orderable) handler).setOrder(orderAnnotation.value());
207 			}
208 		}
209 	}

212 	private void producerOrRouter(List<Annotation> annotations, MessageHandler handler) {
213 		if (handler instanceof AbstractMessageProducingHandler || handler instanceof AbstractMessageRouter) {
214 			String sendTimeout = MessagingAnnotationUtils.resolveAttribute(annotations, "sendTimeout", String.class);
215 			if (sendTimeout != null) {
216 				String resolvedValue = this.beanFactory.resolveEmbeddedValue(sendTimeout);
217 				if (resolvedValue != null) {
218 					long value = Long.parseLong(resolvedValue);
219 					if (handler instanceof AbstractMessageProducingHandler) {
220 						((AbstractMessageProducingHandler) handler).setSendTimeout(value);
221 					}
222 					else {
223 						((AbstractMessageRouter) handler).setSendTimeout(value);
224 					}
225 				}
226 			}
227 		}
228 	}

231 	private MessageHandler annotated(Method method, MessageHandler handlerArg) {
232 		MessageHandler handler = handlerArg;
233 		if (AnnotatedElementUtils.isAnnotated(method, IdempotentReceiver.class.getName())
234 				&& !AnnotatedElementUtils.isAnnotated(method, Bean.class.getName())) {
235 
236 			String[] interceptors =
237 					AnnotationUtils.getAnnotation(method, IdempotentReceiver.class).value(); // NOSONAR never null
238 			for (String interceptor : interceptors) {
239 				DefaultBeanFactoryPointcutAdvisor advisor = new DefaultBeanFactoryPointcutAdvisor();
240 				advisor.setAdviceBeanName(interceptor);
241 				NameMatchMethodPointcut pointcut = new NameMatchMethodPointcut();
242 				pointcut.setMappedName("handleMessage");
243 				advisor.setPointcut(pointcut);
244 				advisor.setBeanFactory(this.beanFactory);
245 
246 				if (handler instanceof Advised) {
247 					((Advised) handler).addAdvisor(advisor);
248 				}
249 				else {
250 					ProxyFactory proxyFactory = new ProxyFactory(handler);
251 					proxyFactory.addAdvisor(advisor);
252 					handler = (MessageHandler) proxyFactory.getProxy(this.beanFactory.getBeanClassLoader());
253 				}
254 			}
255 		}
256 		return handler;
257 	}

260 	private MessageHandler adviceChain(String beanName, List<Annotation> annotations, MessageHandler handlerArg) {
261 		MessageHandler handler = handlerArg;
262 		List<Advice> adviceChain = extractAdviceChain(beanName, annotations);
263 
264 		if (!CollectionUtils.isEmpty(adviceChain) && handler instanceof AbstractReplyProducingMessageHandler) {
265 			((AbstractReplyProducingMessageHandler) handler).setAdviceChain(adviceChain);
266 		}
267 
268 		if (!CollectionUtils.isEmpty(adviceChain)) {
269 			for (Advice advice : adviceChain) {
270 				if (advice instanceof HandleMessageAdvice) {
271 					NameMatchMethodPointcutAdvisor handlerAdvice = new NameMatchMethodPointcutAdvisor(advice);
272 					handlerAdvice.addMethodName("handleMessage");
273 					if (handler instanceof Advised) {
274 						((Advised) handler).addAdvisor(handlerAdvice);
275 					}
276 					else {
277 						ProxyFactory proxyFactory = new ProxyFactory(handler);
278 						proxyFactory.addAdvisor(handlerAdvice);
279 						handler = (MessageHandler) proxyFactory.getProxy(this.beanFactory.getBeanClassLoader());
280 					}
281 				}
282 			}
283 		}
284 		return handler;
285 	}

287 	@Override
288 	public boolean shouldCreateEndpoint(Method method, List<Annotation> annotations) {
289 		String inputChannel = MessagingAnnotationUtils.resolveAttribute(annotations, getInputChannelAttribute(),
290 				String.class);
291 		boolean createEndpoint = StringUtils.hasText(inputChannel);
292 		if (!createEndpoint && beanAnnotationAware()) {
293 			boolean isBean = AnnotatedElementUtils.isAnnotated(method, Bean.class.getName());
294 			Assert.isTrue(!isBean, "A channel name in '" + getInputChannelAttribute() + "' is required when " +
295 					this.annotationType + " is used on '@Bean' methods.");
296 		}
297 		return createEndpoint;
298 	}

308 	protected List<Advice> extractAdviceChain(String beanName, List<Annotation> annotations) {
309 		List<Advice> adviceChain = null;
310 		String[] adviceChainNames = MessagingAnnotationUtils.resolveAttribute(annotations, ADVICE_CHAIN_ATTRIBUTE,
311 				String[].class);
312 		/*
313 		 * Note: we don't merge advice chain contents; if the directAnnotation has a non-empty
314 		 * attribute, it wins. You cannot "remove" an advice chain from a meta-annotation
315 		 * by setting an empty array on the custom annotation.
316 		 */
317 		if (adviceChainNames != null && adviceChainNames.length > 0) {
318 			adviceChain = new ArrayList<>();
319 			for (String adviceChainName : adviceChainNames) {
320 				Object adviceChainBean = this.beanFactory.getBean(adviceChainName);
321 				if (adviceChainBean instanceof Advice) {
322 					adviceChain.add((Advice) adviceChainBean);
323 				}
324 				else if (adviceChainBean instanceof Advice[]) {
325 					Collections.addAll(adviceChain, (Advice[]) adviceChainBean);
326 				}
327 				else if (adviceChainBean instanceof Collection) {
328 					@SuppressWarnings("unchecked")
329 					Collection<Advice> adviceChainEntries = (Collection<Advice>) adviceChainBean;
330 					adviceChain.addAll(adviceChainEntries);
331 				}
332 				else {
333 					throw new IllegalArgumentException("Invalid advice chain type:" +
334 							adviceChainName.getClass().getName() + " for bean '" + beanName + "'");
335 				}
336 			}
337 		}
338 		return adviceChain;
339 	}

341 	protected AbstractEndpoint createEndpoint(MessageHandler handler, @SuppressWarnings("unused") Method method,
342 			List<Annotation> annotations) {
343 
344 		AbstractEndpoint endpoint = null;
345 		String inputChannelName = MessagingAnnotationUtils.resolveAttribute(annotations, getInputChannelAttribute(),
346 				String.class);
347 		if (StringUtils.hasText(inputChannelName)) {
348 			MessageChannel inputChannel;
349 			try {
350 				inputChannel = this.channelResolver.resolveDestination(inputChannelName);
351 			}
352 			catch (DestinationResolutionException e) {
353 				if (e.getCause() instanceof NoSuchBeanDefinitionException) {
354 					inputChannel = new DirectChannel();
355 					this.beanFactory.registerSingleton(inputChannelName, inputChannel);
356 					inputChannel = (MessageChannel) this.beanFactory.initializeBean(inputChannel, inputChannelName);
357 					if (this.disposables != null) {
358 						this.disposables.add((DisposableBean) inputChannel);
359 					}
360 				}
361 				else {
362 					throw e;
363 				}
364 			}
365 			Assert.notNull(inputChannel, "failed to resolve inputChannel '" + inputChannelName + "'");
366 
367 			endpoint = doCreateEndpoint(handler, inputChannel, annotations);
368 		}
369 		return endpoint;
370 	}
371 
372 	protected AbstractEndpoint doCreateEndpoint(MessageHandler handler, MessageChannel inputChannel,
373 			List<Annotation> annotations) {
374 		AbstractEndpoint endpoint;
375 		if (inputChannel instanceof PollableChannel) {
376 			PollingConsumer pollingConsumer = new PollingConsumer((PollableChannel) inputChannel, handler);
377 			configurePollingEndpoint(pollingConsumer, annotations);
378 			endpoint = pollingConsumer;
379 		}
380 		else {
381 			Poller[] pollers = MessagingAnnotationUtils.resolveAttribute(annotations, "poller", Poller[].class);
382 			Assert.state(ObjectUtils.isEmpty(pollers), "A '@Poller' should not be specified for Annotation-based " +
383 					"endpoint, since '" + inputChannel + "' is a SubscribableChannel (not pollable).");
384 			if (inputChannel instanceof Publisher) {
385 				if (handler instanceof ReactiveMessageHandlerAdapter) {
386 					endpoint = new ReactiveStreamsConsumer(inputChannel,
387 							((ReactiveMessageHandlerAdapter) handler).getDelegate());
388 				}
389 				else {
390 					endpoint = new ReactiveStreamsConsumer(inputChannel, handler);
391 				}
392 			}
393 			else {
394 				endpoint = new EventDrivenConsumer((SubscribableChannel) inputChannel, handler);
395 			}
396 		}
397 		return endpoint;
398 	}

400 	protected void configurePollingEndpoint(AbstractPollingEndpoint pollingEndpoint, List<Annotation> annotations) {
401 		PollerMetadata pollerMetadata;
402 		Poller[] pollers = MessagingAnnotationUtils.resolveAttribute(annotations, "poller", Poller[].class);
403 		if (!ObjectUtils.isEmpty(pollers)) {
404 			Assert.state(pollers.length == 1,
405 					"The 'poller' for an Annotation-based endpoint can have only one '@Poller'.");
406 			Poller poller = pollers[0];
407 
408 			String ref = poller.value();
409 			String triggerRef = poller.trigger();
410 			String executorRef = poller.taskExecutor();
411 			String fixedDelayValue = this.beanFactory.resolveEmbeddedValue(poller.fixedDelay());
412 			String fixedRateValue = this.beanFactory.resolveEmbeddedValue(poller.fixedRate());
413 			String maxMessagesPerPollValue = this.beanFactory.resolveEmbeddedValue(poller.maxMessagesPerPoll());
414 			String cron = this.beanFactory.resolveEmbeddedValue(poller.cron());
415 			String errorChannel = this.beanFactory.resolveEmbeddedValue(poller.errorChannel());
416 			String receiveTimeout = this.beanFactory.resolveEmbeddedValue(poller.receiveTimeout());
417 
418 			if (StringUtils.hasText(ref)) {
419 				Assert.state(!StringUtils.hasText(triggerRef)
420 								&& !StringUtils.hasText(executorRef)
421 								&& !StringUtils.hasText(cron)
422 								&& !StringUtils.hasText(fixedDelayValue)
423 								&& !StringUtils.hasText(fixedRateValue)
424 								&& !StringUtils.hasText(maxMessagesPerPollValue), // NOSONAR boolean complexity
425 						"The '@Poller' 'ref' attribute is mutually exclusive with other attributes.");
426 				pollerMetadata = this.beanFactory.getBean(ref, PollerMetadata.class);
427 			}
428 			else {
429 				pollerMetadata = configurePoller(pollingEndpoint, triggerRef, executorRef, fixedDelayValue,
430 						fixedRateValue, maxMessagesPerPollValue, cron, errorChannel, receiveTimeout);
431 			}
432 		}
433 		else {
434 			pollerMetadata = PollerMetadata.getDefaultPollerMetadata(this.beanFactory);
435 			Assert.notNull(pollerMetadata, "No poller has been defined for Annotation-based endpoint, " +
436 					"and no default poller is available within the context.");
437 		}
438 		pollingEndpoint.setTaskExecutor(pollerMetadata.getTaskExecutor());
439 		pollingEndpoint.setTrigger(pollerMetadata.getTrigger());
440 		pollingEndpoint.setAdviceChain(pollerMetadata.getAdviceChain());
441 		pollingEndpoint.setMaxMessagesPerPoll(pollerMetadata.getMaxMessagesPerPoll());
442 		pollingEndpoint.setErrorHandler(pollerMetadata.getErrorHandler());
443 		if (pollingEndpoint instanceof PollingConsumer) {
444 			((PollingConsumer) pollingEndpoint).setReceiveTimeout(pollerMetadata.getReceiveTimeout());
445 		}
446 		pollingEndpoint.setTransactionSynchronizationFactory(pollerMetadata.getTransactionSynchronizationFactory());
447 	}

450 	private PollerMetadata configurePoller(AbstractPollingEndpoint pollingEndpoint, String triggerRef,
451 			String executorRef, String fixedDelayValue, String fixedRateValue, String maxMessagesPerPollValue,
452 			String cron, String errorChannel, String receiveTimeout) {
453 
454 		PollerMetadata pollerMetadata;
455 		pollerMetadata = new PollerMetadata();
456 		if (StringUtils.hasText(maxMessagesPerPollValue)) {
457 			pollerMetadata.setMaxMessagesPerPoll(Long.parseLong(maxMessagesPerPollValue));
458 		}
459 		else if (pollingEndpoint instanceof SourcePollingChannelAdapter) {
460 			// SPCAs default to 1 message per poll
461 			pollerMetadata.setMaxMessagesPerPoll(1);
462 		}
463 
464 		if (StringUtils.hasText(executorRef)) {
465 			pollerMetadata.setTaskExecutor(this.beanFactory.getBean(executorRef, TaskExecutor.class));
466 		}
467 
468 		trigger(triggerRef, fixedDelayValue, fixedRateValue, cron, pollerMetadata);
469 
470 		if (StringUtils.hasText(errorChannel)) {
471 			MessagePublishingErrorHandler errorHandler = new MessagePublishingErrorHandler();
472 			errorHandler.setDefaultErrorChannelName(errorChannel);
473 			errorHandler.setBeanFactory(this.beanFactory);
474 			pollerMetadata.setErrorHandler(errorHandler);
475 		}
476 
477 		if (StringUtils.hasText(receiveTimeout)) {
478 			pollerMetadata.setReceiveTimeout(Long.parseLong(receiveTimeout));
479 		}
480 		return pollerMetadata;
481 	}

484 	private void trigger(String triggerRef, String fixedDelayValue, String fixedRateValue, String cron,
485 			PollerMetadata pollerMetadata) {
486 		Trigger trigger = null;
487 		if (StringUtils.hasText(triggerRef)) {
488 			Assert.state(!StringUtils.hasText(cron) && !StringUtils.hasText(fixedDelayValue)
489 							&& !StringUtils.hasText(fixedRateValue),
490 					"The '@Poller' 'trigger' attribute is mutually exclusive with other attributes.");
491 			trigger = this.beanFactory.getBean(triggerRef, Trigger.class);
492 		}
493 		else if (StringUtils.hasText(cron)) {
494 			Assert.state(!StringUtils.hasText(fixedDelayValue) && !StringUtils.hasText(fixedRateValue),
495 					"The '@Poller' 'cron' attribute is mutually exclusive with other attributes.");
496 			trigger = new CronTrigger(cron);
497 		}
498 		else if (StringUtils.hasText(fixedDelayValue)) {
499 			Assert.state(!StringUtils.hasText(fixedRateValue),
500 					"The '@Poller' 'fixedDelay' attribute is mutually exclusive with other attributes.");
501 			trigger = new PeriodicTrigger(Long.parseLong(fixedDelayValue));
502 		}
503 		else if (StringUtils.hasText(fixedRateValue)) {
504 			trigger = new PeriodicTrigger(Long.parseLong(fixedRateValue));
505 			((PeriodicTrigger) trigger).setFixedRate(true);
506 		}
507 		//'Trigger' can be null. 'PollingConsumer' does fallback to the 'new PeriodicTrigger(10)'.
508 		pollerMetadata.setTrigger(trigger);
509 	}

511 	protected String generateHandlerBeanName(String originalBeanName, Method method) {
512 		String name = MessagingAnnotationUtils.endpointIdValue(method);
513 		if (!StringUtils.hasText(name)) {
514 			String baseName = originalBeanName + "." + method.getName() + "."
515 					+ org.springframework.util.ClassUtils.getShortNameAsProperty(this.annotationType);
516 			name = baseName;
517 			int count = 1;
518 			while (this.beanFactory.containsBean(name)) {
519 				name = baseName + "#" + (++count);
520 			}
521 		}
522 		return name + IntegrationConfigUtils.HANDLER_ALIAS_SUFFIX;
523 	}

539 	protected String resolveTargetBeanName(Method method) {
540 		String id = method.getName();
541 		String[] names = AnnotationUtils.getAnnotation(method, Bean.class).name(); // NOSONAR never null
542 		if (!ObjectUtils.isEmpty(names)) {
543 			id = names[0];
544 		}
545 		return id;
546 	}

548 	@SuppressWarnings("unchecked")
549 	protected <H> H extractTypeIfPossible(@Nullable Object targetObject, Class<H> expectedType) {
550 		if (targetObject == null) {
551 			return null;
552 		}
553 		if (expectedType.isAssignableFrom(targetObject.getClass())) {
554 			return (H) targetObject;
555 		}
556 		if (targetObject instanceof Advised) {
557 			TargetSource targetSource = ((Advised) targetObject).getTargetSource();
558 			try {
559 				return extractTypeIfPossible(targetSource.getTarget(), expectedType);
560 			}
561 			catch (Exception e) {
562 				throw new IllegalStateException(e);
563 			}
564 		}
565 		return null;
566 	}

568 	protected void checkMessageHandlerAttributes(String handlerBeanName, List<Annotation> annotations) {
569 		for (String attribute : this.messageHandlerAttributes) {
570 			for (Annotation annotation : annotations) {
571 				Object value = AnnotationUtils.getValue(annotation, attribute);
572 				if (MessagingAnnotationUtils.hasValue(value)) {
573 					throw new BeanDefinitionValidationException("The MessageHandler [" + handlerBeanName +
574 							"] can not be populated because of ambiguity with annotation attributes " +
575 							this.messageHandlerAttributes + " which are not allowed when an integration annotation " +
576 							"is used with a @Bean definition for a MessageHandler." +
577 							"\nThe attribute causing the ambiguity is: [" + attribute + "]." +
578 							"\nUse the appropriate setter on the MessageHandler directly when configuring an " +
579 							"endpoint this way.");
580 				}
581 			}
582 		}
583 	}

589 	@Nullable
590 	protected MessageProcessor<?> buildLambdaMessageProcessorForBeanMethod(Method method, Object target) {
591 		if ((target instanceof Function || target instanceof Consumer) && ClassUtils.isLambda(target.getClass())
592 				|| ClassUtils.isKotlinFaction1(target.getClass())) {
593 
594 			ResolvableType methodReturnType = ResolvableType.forMethodReturnType(method);
595 			Class<?> expectedPayloadType = methodReturnType.getGeneric(0).toClass();
596 			return new LambdaMessageProcessor(target, expectedPayloadType);
597 		}
598 		else {
599 			return null;
600 		}
601 	}

54  /**
55   * Adapter that receives Messages from an AMQP Queue, converts them into
56   * Spring Integration Messages, and sends the results to a Message Channel.
57   * If a reply Message is received, it will be converted and sent back to
58   * the AMQP 'replyTo'.
59   *
60   * @author Mark Fisher
61   * @author Artem Bilan
62   * @author Gary Russell
63   *
64   * @since 2.1
65   */

109 	private AmqpInboundGateway(AbstractMessageListenerContainer listenerContainer, AmqpTemplate amqpTemplate,
110 			boolean amqpTemplateExplicitlySet) {
111 		Assert.notNull(listenerContainer, "listenerContainer must not be null");
112 		Assert.notNull(amqpTemplate, "'amqpTemplate' must not be null");
113 		Assert.isNull(listenerContainer.getMessageListener(),
114 				"The listenerContainer provided to an AMQP inbound Gateway " +
115 						"must not have a MessageListener configured since " +
116 						"the adapter needs to configure its own listener implementation.");
117 		this.messageListenerContainer = listenerContainer;
118 		this.messageListenerContainer.setAutoStartup(false);
119 		this.amqpTemplate = amqpTemplate;
120 		this.amqpTemplateExplicitlySet = amqpTemplateExplicitlySet;
121 		if (this.amqpTemplateExplicitlySet && this.amqpTemplate instanceof RabbitTemplate) {
122 			this.templateMessageConverter = ((RabbitTemplate) this.amqpTemplate).getMessageConverter();
123 		}
124 		setErrorMessageStrategy(new AmqpMessageHeaderErrorMessageStrategy());
125 	}

129 	 * Specify the {@link MessageConverter} to convert request and reply to/from {@link Message}.
130 	 * If the {@link #amqpTemplate} is explicitly set, this {@link MessageConverter}
131 	 * isn't populated there. You must configure that external {@link #amqpTemplate}.
132 	 * @param messageConverter the {@link MessageConverter} to use.

135 		Assert.notNull(messageConverter, "MessageConverter must not be null");
136 		this.amqpMessageConverter = messageConverter;
137 		if (!this.amqpTemplateExplicitlySet) {
138 			((RabbitTemplate) this.amqpTemplate).setMessageConverter(messageConverter);
139 			this.templateMessageConverter = messageConverter;
140 		}
141 	}

213 	/**
214 	 * When mapping headers for the outbound (reply) message, determine whether the headers are
215 	 * mapped before the message is converted, or afterwards. This only affects headers
216 	 * that might be added by the message converter. When false, the converter's headers
217 	 * win; when true, any headers added by the converter will be overridden (if the
218 	 * source message has a header that maps to those headers). You might wish to set this
219 	 * to true, for example, when using a
220 	 * {@link org.springframework.amqp.support.converter.SimpleMessageConverter} with a
221 	 * String payload that contains json; the converter will set the content type to
222 	 * {@code text/plain} which can be overridden to {@code application/json} by setting
223 	 * the {@link AmqpHeaders#CONTENT_TYPE} message header. Default: false.
224 	 * @param replyHeadersMappedLast true if reply headers are mapped after conversion.
225 	 * @since 5.1.9
226 	 */
227 	public void setReplyHeadersMappedLast(boolean replyHeadersMappedLast) {
228 		this.replyHeadersMappedLast = replyHeadersMappedLast;
229 	}

236 	@Override
237 	protected void onInit() {
238 		if (this.retryTemplate != null) {
239 			Assert.state(getErrorChannel() == null, "Cannot have an 'errorChannel' property when a 'RetryTemplate' is "
240 					+ "provided; use an 'ErrorMessageSendingRecoverer' in the 'recoveryCallback' property to "
241 					+ "send an error message when retries are exhausted");
242 		}
243 		Listener messageListener = new Listener();
244 		this.messageListenerContainer.setMessageListener(messageListener);
245 		this.messageListenerContainer.afterPropertiesSet();
246 		if (!this.amqpTemplateExplicitlySet) {
247 			((RabbitTemplate) this.amqpTemplate).afterPropertiesSet();
248 		}
249 		super.onInit();
250 		if (this.retryTemplate != null && getErrorChannel() != null) {
251 			logger.warn("Usually, when using a RetryTemplate you should use an ErrorMessageSendingRecoverer and not "
252 					+ "provide an errorChannel. Using an errorChannel could defeat retry and will receive an error "
253 					+ "message for each delivery attempt.");
254 		}
255 	}

269 	/**
270 	 * If there's a retry template, it will set the attributes holder via the listener. If
271 	 * there's no retry template, but there's an error channel, we create a new attributes
272 	 * holder here. If an attributes holder exists (by either method), we set the
273 	 * attributes for use by the
274 	 * {@link org.springframework.integration.support.ErrorMessageStrategy}.
275 	 * @param amqpMessage the AMQP message to use.
276 	 * @param message the Spring Messaging message to use.
277 	 * @since 4.3.10
278 	 */
279 	private void setAttributesIfNecessary(Message amqpMessage, org.springframework.messaging.Message<?> message) {
280 		boolean needHolder = getErrorChannel() != null && this.retryTemplate == null;
281 		boolean needAttributes = needHolder || this.retryTemplate != null;
282 		if (needHolder) {
283 			ATTRIBUTES_HOLDER.set(ErrorMessageUtils.getAttributeAccessor(null, null));
284 		}
285 		if (needAttributes) {
286 			AttributeAccessor attributes = this.retryTemplate != null
287 					? RetrySynchronizationManager.getContext()
288 					: ATTRIBUTES_HOLDER.get();
289 			if (attributes != null) {
290 				attributes.setAttribute(ErrorMessageUtils.INPUT_MESSAGE_CONTEXT_KEY, message);
291 				attributes.setAttribute(AmqpMessageHeaderErrorMessageStrategy.AMQP_RAW_MESSAGE, amqpMessage);
292 			}
293 		}
294 	}
295 
296 	@Override
297 	protected AttributeAccessor getErrorMessageAttributes(org.springframework.messaging.Message<?> message) {
298 		AttributeAccessor attributes = ATTRIBUTES_HOLDER.get();
299 		if (attributes == null) {
300 			return super.getErrorMessageAttributes(message);
301 		}
302 		else {
303 			return attributes;
304 		}
305 	}

307 	protected class Listener implements ChannelAwareMessageListener {
308 
309 		@SuppressWarnings("unchecked")
310 		@Override
311 		public void onMessage(final Message message, final Channel channel) {
312 			if (AmqpInboundGateway.this.retryTemplate == null) {
313 				try {
314 					org.springframework.messaging.Message<Object> converted = convert(message, channel);
315 					if (converted != null) {
316 						process(message, converted);
317 					}
318 				}
319 				finally {
320 					ATTRIBUTES_HOLDER.remove();
321 				}
322 			}
323 			else {
324 				org.springframework.messaging.Message<Object> converted = convert(message, channel);
325 				if (converted != null) {
326 					AmqpInboundGateway.this.retryTemplate.execute(context -> {
327 								StaticMessageHeaderAccessor.getDeliveryAttempt(converted).incrementAndGet();
328 								process(message, converted);
329 								return null;
330 							},
331 							(RecoveryCallback<Object>) AmqpInboundGateway.this.recoveryCallback);
332 				}
333 			}
334 		}
335 
336 		private org.springframework.messaging.Message<Object> convert(Message message, Channel channel) {
337 			Map<String, Object> headers;
338 			Object payload;
339 			boolean isManualAck =
340 					AmqpInboundGateway.this.messageListenerContainer.getAcknowledgeMode() == AcknowledgeMode.MANUAL;
341 			try {
342 				if (AmqpInboundGateway.this.batchingStrategy.canDebatch(message.getMessageProperties())) {
343 					List<Object> payloads = new ArrayList<>();
344 					AmqpInboundGateway.this.batchingStrategy.deBatch(message, fragment -> payloads
345 							.add(AmqpInboundGateway.this.amqpMessageConverter.fromMessage(fragment)));
346 					payload = payloads;
347 				}
348 				else {
349 					payload = AmqpInboundGateway.this.amqpMessageConverter.fromMessage(message);
350 				}
351 				headers = AmqpInboundGateway.this.headerMapper.toHeadersFromRequest(message.getMessageProperties());
352 				if (isManualAck) {
353 					headers.put(AmqpHeaders.DELIVERY_TAG, message.getMessageProperties().getDeliveryTag());
354 					headers.put(AmqpHeaders.CHANNEL, channel);
355 				}
356 				if (AmqpInboundGateway.this.retryTemplate != null) {
357 					headers.put(IntegrationMessageHeaderAccessor.DELIVERY_ATTEMPT, new AtomicInteger());
358 				}
359 				if (AmqpInboundGateway.this.bindSourceMessage) {
360 					headers.put(IntegrationMessageHeaderAccessor.SOURCE_DATA, message);
361 				}
362 			}
363 			catch (RuntimeException e) {
364 				MessageChannel errorChannel = getErrorChannel();
365 				if (errorChannel != null) {
366 					setAttributesIfNecessary(message, null);
367 					AmqpInboundGateway.this.messagingTemplate.send(errorChannel,
368 							buildErrorMessage(null,
369 									EndpointUtils.errorMessagePayload(message, channel, isManualAck, e)));
370 				}
371 				else {
372 					throw e;
373 				}
374 				return null;
375 			}
376 			return getMessageBuilderFactory()
377 					.withPayload(payload)
378 					.copyHeaders(headers)
379 					.build();
380 		}
381 
382 		private void process(Message message, org.springframework.messaging.Message<Object> messagingMessage) {
383 			setAttributesIfNecessary(message, messagingMessage);
384 			org.springframework.messaging.Message<?> reply = sendAndReceiveMessage(messagingMessage);
385 			if (reply != null) {
386 				Address replyTo;
387 				String replyToProperty = message.getMessageProperties().getReplyTo();
388 				if (replyToProperty != null) {
389 					replyTo = new Address(replyToProperty);
390 				}
391 				else {
392 					replyTo = AmqpInboundGateway.this.defaultReplyTo;
393 				}
394 
395 				org.springframework.amqp.core.Message amqpMessage =
396 						MappingUtils.mapReplyMessage(reply, AmqpInboundGateway.this.templateMessageConverter,
397 								AmqpInboundGateway.this.headerMapper,
398 								message.getMessageProperties().getReceivedDeliveryMode(),
399 								AmqpInboundGateway.this.replyHeadersMappedLast);
400 
401 				if (replyTo != null) {
402 					AmqpInboundGateway.this.amqpTemplate.send(replyTo.getExchangeName(), replyTo.getRoutingKey(),
403 							amqpMessage);
404 				}
405 				else {
406 					if (!AmqpInboundGateway.this.amqpTemplateExplicitlySet) {
407 						throw new IllegalStateException("There is no 'replyTo' message property " +
408 								"and the `defaultReplyTo` hasn't been configured.");
409 					}
410 					else {
411 						AmqpInboundGateway.this.amqpTemplate.send(amqpMessage);
412 					}
413 				}
414 			}
415 		}

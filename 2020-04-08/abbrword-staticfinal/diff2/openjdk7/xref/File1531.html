84      /**
85       * Write the given bean to the XML file.
86       * <p>
87       * Performs an atomic write, first writing in {@code <file>.new}, then
88       * renaming {@code <file>} to {@code <file>~}, then renaming
89       * renaming {@code <file>.new} to {@code <file>}.
90       * </p>
91       * @param bean The configuration to write in the XML file.
92       * @throws IOException if write to file failed.
93       **/
94      public synchronized void writeToFile(ScanManagerConfig bean)
95          throws IOException {
96  
97          // Creates a new file named <file>.new
98          final File f = newXmlTmpFile(file);
99          try {
100             final FileOutputStream out = new FileOutputStream(f);
101             boolean failed = true;
102             try {
103                 // writes to <file>.new
104                 write(bean,out,false);
105 
106                 // no exception: set failed=false for finaly {} block.
107                 failed = false;
108             } finally {
109                 out.close();
110                 // An exception was raised: delete temporary file.
111                 if (failed == true) f.delete();
112             }
113 
114             // rename <file> to <file>~ and <file>.new to <file>
115             commit(file,f);
116         } catch (JAXBException x) {
117             final IOException io =
118                     new IOException("Failed to write SessionConfigBean to " +
119                     file+": "+x,x);
120             throw io;
121         }
122     }

124     /**
125      * Creates an XML string representation of the given bean.
126      * @throws IllegalArgumentException if the bean class is not known by the
127      *         underlying XMLbinding context.
128      * @return An XML string representation of the given bean.
129      **/
130     public static String toString(Object bean) {
131         try {
132             final ByteArrayOutputStream baos = new ByteArrayOutputStream();
133             final Marshaller m = createMarshaller();
134             m.setProperty(m.JAXB_FRAGMENT,Boolean.TRUE);
135             m.marshal(bean, baos);
136             return baos.toString();
137         } catch (JAXBException x) {
138             final IllegalArgumentException iae =
139                     new IllegalArgumentException(
140                         "Failed to write SessionConfigBean: "+x,x);
141             throw iae;
142         }
143     }

161     /**
162      * Creates an XML clone of the given bean.
163      * <p>
164      * In other words, this method XML-serializes the given bean, and
165      * XML-deserializes a copy of that bean.
166      * </p>
167      * @throws IllegalArgumentException if the bean class is not known by the
168      *         underlying XML binding context.
169      * @return A deep-clone of the given bean.
170      **/
171     private static Object copy(Object bean) {
172         try {
173             final ByteArrayOutputStream baos = new ByteArrayOutputStream();
174             final Marshaller m = createMarshaller();
175             m.marshal(bean, baos);
176             final ByteArrayInputStream bais =
177                     new ByteArrayInputStream(baos.toByteArray());
178             return createUnmarshaller().unmarshal(bais);
179         } catch (JAXBException x) {
180             final IllegalArgumentException iae =
181                     new IllegalArgumentException("Failed to write SessionConfigBean: "+x,x);
182             throw iae;
183         }
184     }

202     /**
203      * Reads the configuration from the XML configuration file.
204      * @throws IOException if it fails to read the configuration.
205      * @return A {@code ScanManagerConfig} bean read from the
206      *         XML configuration file.
207      **/
208     public synchronized ScanManagerConfig readFromFile() throws IOException {
209         final File f = new File(file);
210         if (!f.exists())
211             throw new IOException("No such file: "+file);
212         if (!f.canRead())
213             throw new IOException("Can't read file: "+file);
214         try {
215             return read(f);
216         } catch (JAXBException x) {
217             final IOException io =
218                     new IOException("Failed to read SessionConfigBean from " +
219                     file+": "+x,x);
220             throw io;
221         }
222     }

230      */
231     public static ScanManagerConfig read(File f)

270      * @param os the output stream to write to.
271      * @param fragment whether the {@code <?xml ... ?>} header should be
272      *        included. The header is not included if the bean is just an
273      *        XML fragment encapsulated in a higher level XML element.
274      * @throws JAXBException An XML Binding exception occurred.

276     private static void writeXml(Object bean, OutputStream os, boolean fragment)
277         throws JAXBException {
278         final Marshaller m = createMarshaller();
279         if (fragment) m.setProperty(m.JAXB_FRAGMENT,Boolean.TRUE);
280         m.marshal(bean,os);
281     }

308     // Creates a new XML temporary file called <basename>.new
309     // This method is used to implement atomic writing to file.
310     // The usual sequence is:
311     //
312     // Final tmp = newXmlTmpFile(basename);
313     // boolean failed = true;
314     // try {
315     //      ... write to 'tmp' ...
316     //      // no exception: set failed=false for finaly {} block.
317     //      failed = false;
318     // } finally
319     //      // failed==true means there was an exception and
320     //      // commit won't be called...
321     //      if (failed==true) tmp.delete();
322     // }
323     // commit(tmp,basename)
324     //
325     private static File newXmlTmpFile(String basename) throws IOException {
326         final File f = new File(basename+".new");
327         if (!f.createNewFile())
328             throw new IOException("file "+f.getName()+" already exists");
329 
330         try {
331             final OutputStream newStream = new FileOutputStream(f);
332             try {
333                 final String decl =
334                     "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\"?>";
335                 newStream.write(decl.getBytes("UTF-8"));
336                 newStream.flush();
337             } finally {
338                 newStream.close();
339             }
340         } catch (IOException x) {
341             f.delete();
342             throw x;
343         }
344         return f;
345     }
346 
347     // Commit the temporary file by renaming <basename> to <baesname>~
348     // and tmpFile to <basename>.
349     private static File commit(String basename, File tmpFile)
350         throws IOException {
351         try {
352             final String backupName = basename+"~";
353             final File desired = new File(basename);
354             final File backup = new File(backupName);
355             backup.delete();
356             if (desired.exists()) {
357                 if (!desired.renameTo(new File(backupName)))
358                     throw new IOException("can't rename to "+backupName);
359             }
360             if (!tmpFile.renameTo(new File(basename)))
361                 throw new IOException("can't rename to "+basename);
362         } catch (IOException x) {
363             tmpFile.delete();
364             throw x;
365         }
366         return new File(basename);
367     }

369     /**
370      * Creates a new committed XML file for {@code <basename>}, containing only
371      * the {@code <?xml ...?>} header.
372      * <p>This method will rename {@code <basename>} to {@code <basename>~},
373      * if it exists.
374      * </p>
375      * @return A newly created XML file containing the regular
376      *         {@code <?xml ...?>} header.
377      * @param basename The name of the new file.
378      * @throws IOException if the new XML file couldn't be created.
379      */
380     public static File createNewXmlFile(String basename) throws IOException {
381         return commit(basename,newXmlTmpFile(basename));

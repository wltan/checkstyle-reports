49  /**
50   * The Code Point Input Method is a simple input method that allows Unicode
51   * characters to be entered using their code point or code unit values. See the
52   * accompanying file README.txt for more information.
53   *
54   * @author Brian Beck
55   */
56  public class CodePointInputMethod implements InputMethod {
57  
58      private static final int UNSET = 0;
59      private static final int ESCAPE = 1; // \u0000       - \uFFFF
60      private static final int SPECIAL_ESCAPE = 2; // \U000000     - \U10FFFF
61      private static final int SURROGATE_PAIR = 3; // \uD800\uDC00 - \uDBFF\uDFFF
62      private InputMethodContext context;
63      private Locale locale;
64      private StringBuffer buffer;
65      private int insertionPoint;
66      private int format = UNSET;
67  
68      public CodePointInputMethod() throws IOException {
69      }
70  
71      /**
72       * This is the input method's main routine.  The composed text is stored
73       * in buffer.
74       */
75      public void dispatchEvent(AWTEvent event) {
76          // This input method handles KeyEvent only.
77          if (!(event instanceof KeyEvent)) {
78              return;
79          }
80  
81          KeyEvent e = (KeyEvent) event;
82          int eventID = event.getID();
83          boolean notInCompositionMode = buffer.length() == 0;
84  
85          if (eventID == KeyEvent.KEY_PRESSED) {
86              // If we are not in composition mode, pass through
87              if (notInCompositionMode) {
88                  return;
89              }
90  
91              switch (e.getKeyCode()) {
92                  case KeyEvent.VK_LEFT:
93                      moveCaretLeft();
94                      break;
95                  case KeyEvent.VK_RIGHT:
96                      moveCaretRight();
97                      break;
98              }
99          } else if (eventID == KeyEvent.KEY_TYPED) {
100             char c = e.getKeyChar();
101 
102             // If we are not in composition mode, wait a back slash
103             if (notInCompositionMode) {
104                 // If the type character is not a back slash, pass through
105                 if (c != '\\') {
106                     return;
107                 }
108 
109                 startComposition();     // Enter to composition mode
110             } else {
111                 switch (c) {
112                     case ' ':       // Exit from composition mode
113                         finishComposition();
114                         break;
115                     case '\u007f':  // Delete
116                         deleteCharacter();
117                         break;
118                     case '\b':      // BackSpace
119                         deletePreviousCharacter();
120                         break;
121                     case '\u001b':  // Escape
122                         cancelComposition();
123                         break;
124                     case '\n':      // Return
125                     case '\t':      // Tab
126                         sendCommittedText();
127                         break;
128                     default:
129                         composeUnicodeEscape(c);
130                         break;
131                 }
132             }
133         } else {  // KeyEvent.KEY_RELEASED
134             // If we are not in composition mode, pass through
135             if (notInCompositionMode) {
136                 return;
137             }
138         }
139 
140         e.consume();
141     }

143     private void composeUnicodeEscape(char c) {
144         switch (buffer.length()) {
145             case 1:  // \\
146                 waitEscapeCharacter(c);
147                 break;
148             case 2:  // \\u or \\U
149             case 3:  // \\ux or \\Ux
150             case 4:  // \\uxx or \\Uxx
151                 waitDigit(c);
152                 break;
153             case 5:  // \\uxxx or \\Uxxx
154                 if (format == SPECIAL_ESCAPE) {
155                     waitDigit(c);
156                 } else {
157                     waitDigit2(c);
158                 }
159                 break;
160             case 6:  // \\uxxxx or \\Uxxxx
161                 if (format == SPECIAL_ESCAPE) {
162                     waitDigit(c);
163                 } else if (format == SURROGATE_PAIR) {
164                     waitBackSlashOrLowSurrogate(c);
165                 } else {
166                     beep();
167                 }
168                 break;
169             case 7:  // \\Uxxxxx
170                 // Only SPECIAL_ESCAPE format uses this state.
171                 // Since the second "\\u" of SURROGATE_PAIR format is inserted
172                 // automatically, users don't have to type these keys.
173                 waitDigit(c);
174                 break;
175             case 8:  // \\uxxxx\\u
176             case 9:  // \\uxxxx\\ux
177             case 10: // \\uxxxx\\uxx
178             case 11: // \\uxxxx\\uxxx
179                 if (format == SURROGATE_PAIR) {
180                     waitDigit(c);
181                 } else {
182                     beep();
183                 }
184                 break;
185             default:
186                 beep();
187                 break;
188         }
189     }
190 
191     private void waitEscapeCharacter(char c) {
192         if (c == 'u' || c == 'U') {
193             buffer.append(c);
194             insertionPoint++;
195             sendComposedText();
196             format = (c == 'u') ? ESCAPE : SPECIAL_ESCAPE;
197         } else {
198             if (c != '\\') {
199                 buffer.append(c);
200                 insertionPoint++;
201             }
202             sendCommittedText();
203         }
204     }
205 
206     private void waitDigit(char c) {
207         if (Character.digit(c, 16) != -1) {
208             buffer.insert(insertionPoint++, c);
209             sendComposedText();
210         } else {
211             beep();
212         }
213     }
214 
215     private void waitDigit2(char c) {
216         if (Character.digit(c, 16) != -1) {
217             buffer.insert(insertionPoint++, c);
218             char codePoint = (char) getCodePoint(buffer, 2, 5);
219             if (Character.isHighSurrogate(codePoint)) {
220                 format = SURROGATE_PAIR;
221                 buffer.append("\\u");
222                 insertionPoint = 8;
223             } else {
224                 format = ESCAPE;
225             }
226             sendComposedText();
227         } else {
228             beep();
229         }
230     }
231 
232     private void waitBackSlashOrLowSurrogate(char c) {
233         if (insertionPoint == 6) {
234             if (c == '\\') {
235                 buffer.append(c);
236                 buffer.append('u');
237                 insertionPoint = 8;
238                 sendComposedText();
239             } else if (Character.digit(c, 16) != -1) {
240                 buffer.append("\\u");
241                 buffer.append(c);
242                 insertionPoint = 9;
243                 sendComposedText();
244             } else {
245                 beep();
246             }
247         } else {
248             beep();
249         }
250     }
251 
252     /**
253      * Send the composed text to the client.
254      */
255     private void sendComposedText() {
256         AttributedString as = new AttributedString(buffer.toString());
257         as.addAttribute(TextAttribute.INPUT_METHOD_HIGHLIGHT,
258                 InputMethodHighlight.SELECTED_RAW_TEXT_HIGHLIGHT);
259         context.dispatchInputMethodEvent(
260                 InputMethodEvent.INPUT_METHOD_TEXT_CHANGED,
261                 as.getIterator(), 0,
262                 TextHitInfo.leading(insertionPoint), null);
263     }
264 
265     /**
266      * Send the committed text to the client.
267      */
268     private void sendCommittedText() {
269         AttributedString as = new AttributedString(buffer.toString());
270         context.dispatchInputMethodEvent(
271                 InputMethodEvent.INPUT_METHOD_TEXT_CHANGED,
272                 as.getIterator(), buffer.length(),
273                 TextHitInfo.leading(insertionPoint), null);
274 
275         buffer.setLength(0);
276         insertionPoint = 0;
277         format = UNSET;
278     }
279 
280     /**
281      * Move the insertion point one position to the left in the composed text.
282      * Do not let the caret move to the left of the "\\u" or "\\U".
283      */
284     private void moveCaretLeft() {
285         int len = buffer.length();
286         if (--insertionPoint < 2) {
287             insertionPoint++;
288             beep();
289         } else if (format == SURROGATE_PAIR && insertionPoint == 7) {
290             insertionPoint = 8;
291             beep();
292         }
293 
294         context.dispatchInputMethodEvent(
295                 InputMethodEvent.CARET_POSITION_CHANGED,
296                 null, 0,
297                 TextHitInfo.leading(insertionPoint), null);
298     }
299 
300     /**
301      * Move the insertion point one position to the right in the composed text.
302      */
303     private void moveCaretRight() {
304         int len = buffer.length();
305         if (++insertionPoint > len) {
306             insertionPoint = len;
307             beep();
308         }
309 
310         context.dispatchInputMethodEvent(
311                 InputMethodEvent.CARET_POSITION_CHANGED,
312                 null, 0,
313                 TextHitInfo.leading(insertionPoint), null);
314     }
315 
316     /**
317      * Delete the character preceding the insertion point in the composed text.
318      * If the insertion point is not at the end of the composed text and the
319      * preceding text is "\\u" or "\\U", ring the bell.
320      */
321     private void deletePreviousCharacter() {
322         if (insertionPoint == 2) {
323             if (buffer.length() == 2) {
324                 cancelComposition();
325             } else {
326                 // Do not allow deletion of the leading "\\u" or "\\U" if there
327                 // are other digits in the composed text.
328                 beep();
329             }
330         } else if (insertionPoint == 8) {
331             if (buffer.length() == 8) {
332                 if (format == SURROGATE_PAIR) {
333                     buffer.deleteCharAt(--insertionPoint);
334                 }
335                 buffer.deleteCharAt(--insertionPoint);
336                 sendComposedText();
337             } else {
338                 // Do not allow deletion of the second "\\u" if there are other
339                 // digits in the composed text.
340                 beep();
341             }
342         } else {
343             buffer.deleteCharAt(--insertionPoint);
344             if (buffer.length() == 0) {
345                 sendCommittedText();
346             } else {
347                 sendComposedText();
348             }
349         }
350     }
351 
352     /**
353      * Delete the character following the insertion point in the composed text.
354      * If the insertion point is at the end of the composed text, ring the bell.
355      */
356     private void deleteCharacter() {
357         if (insertionPoint < buffer.length()) {
358             buffer.deleteCharAt(insertionPoint);
359             sendComposedText();
360         } else {
361             beep();
362         }
363     }
364 
365     private void startComposition() {
366         buffer.append('\\');
367         insertionPoint = 1;
368         sendComposedText();
369     }
370 
371     private void cancelComposition() {
372         buffer.setLength(0);
373         insertionPoint = 0;
374         sendCommittedText();
375     }
376 
377     private void finishComposition() {
378         int len = buffer.length();
379         if (len == 6 && format != SPECIAL_ESCAPE) {
380             char codePoint = (char) getCodePoint(buffer, 2, 5);
381             if (Character.isValidCodePoint(codePoint) && codePoint != 0xFFFF) {
382                 buffer.setLength(0);
383                 buffer.append(codePoint);
384                 sendCommittedText();
385                 return;
386             }
387         } else if (len == 8 && format == SPECIAL_ESCAPE) {
388             int codePoint = getCodePoint(buffer, 2, 7);
389             if (Character.isValidCodePoint(codePoint) && codePoint != 0xFFFF) {
390                 buffer.setLength(0);
391                 buffer.appendCodePoint(codePoint);
392                 sendCommittedText();
393                 return;
394             }
395         } else if (len == 12 && format == SURROGATE_PAIR) {
396             char[] codePoint = {
397                 (char) getCodePoint(buffer, 2, 5),
398                 (char) getCodePoint(buffer, 8, 11)
399             };
400             if (Character.isHighSurrogate(codePoint[0]) && Character.
401                     isLowSurrogate(codePoint[1])) {
402                 buffer.setLength(0);
403                 buffer.append(codePoint);
404                 sendCommittedText();
405                 return;
406             }
407         }
408 
409         beep();
410     }
411 
412     private int getCodePoint(StringBuffer sb, int from, int to) {
413         int value = 0;
414         for (int i = from; i <= to; i++) {
415             value = (value << 4) + Character.digit(sb.charAt(i), 16);
416         }
417         return value;
418     }
419 
420     private static void beep() {
421         Toolkit.getDefaultToolkit().beep();
422     }
423 
424     public void activate() {
425         if (buffer == null) {
426             buffer = new StringBuffer(12);
427             insertionPoint = 0;
428         }
429     }
430 
431     public void deactivate(boolean isTemporary) {
432         if (!isTemporary) {
433             buffer = null;
434         }
435     }

444     public void endComposition() {
445         sendCommittedText();

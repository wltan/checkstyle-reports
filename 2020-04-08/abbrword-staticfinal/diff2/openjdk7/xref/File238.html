200    /**
201     * Sets the namespace prefix which will be used to identify elements in the
202     * XML Signature Namespace.
203     *
204     * <pre>
205     * Constants.setSignatureSpecNSprefix("dsig");
206     * </pre>
207     *
208     * @param newPrefix is the new namespace prefix.
209     * @throws XMLSecurityException
210     * @see com.sun.org.apache.xml.internal.security.utils.Constants#getSignatureSpecNSprefix
211     * $todo$ Add consistency checking for valid prefix
212     */
213    public static void setSignatureSpecNSprefix(String newPrefix) throws XMLSecurityException {
214       ElementProxy.setDefaultPrefix(Constants.SignatureSpecNS, newPrefix);
215    }
216 
217    /**
218     * Returns the XML namespace prefix which is used for elements in the XML
219     * Signature namespace.
220     *
221     * It is defaulted to <code>dsig</code>, but can be changed using the
222     * {@link #setSignatureSpecNSprefix} function.
223     *
224     * @return the current used namespace prefix
225     * @see #setSignatureSpecNSprefix
226     */
227    public static String getSignatureSpecNSprefix() {
228       return ElementProxy.getDefaultPrefix(Constants.SignatureSpecNS);
229    }

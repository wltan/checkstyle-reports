27  /** Factory class for bodies of the solar system.
28   * <p>The {@link #getSun() Sun}, the {@link #getMoon() Moon} and the planets
29   * (including the Pluto dwarf planet) are provided by this factory. In addition,
30   * two important points are provided for convenience: the {@link
31   * #getSolarSystemBarycenter() solar system barycenter} and the {@link
32   * #getEarthMoonBarycenter() Earth-Moon barycenter}.</p>
33   * <p>The underlying body-centered frames are either direct children of {@link
34   * org.orekit.frames.FramesFactory#getEME2000() EME2000} (for {@link #getMoon() Moon}
35   * and {@link #getEarthMoonBarycenter() Earth-Moon barycenter}) or children from other
36   * body-centered frames. For example, the path from EME2000 to
37   * Jupiter-centered frame is: EME2000, Earth-Moon barycenter centered,
38   * solar system barycenter centered, Jupiter-centered. The defining transforms
39   * of these frames are combinations of simple linear {@link
40   * org.orekit.frames.Transform#Transform(org.orekit.time.AbsoluteDate,
41   * org.hipparchus.geometry.euclidean.threed.Vector3D,
42   * org.hipparchus.geometry.euclidean.threed.Vector3D) translation/velocity} transforms
43   * without any rotation. The frame axes are therefore always parallel to
44   * {@link org.orekit.frames.FramesFactory#getEME2000() EME2000} frame axes.</p>
45   * <p>The position of the bodies provided by this class are interpolated using
46   * the JPL DE 405/DE 406 ephemerides.</p>
47   * @author Luc Maisonobe
48   */
49  public class CelestialBodyFactory {

116     /** Celestial body loaders map. */
117     private static final Map<String, List<CelestialBodyLoader>> LOADERS_MAP =

122         new HashMap<String, CelestialBody>();

132     /** Add a loader for celestial bodies.
133      * @param name name of the body (may be one of the predefined names or a user-defined name)
134      * @param loader custom loader to add for the body
135      * @see #addDefaultCelestialBodyLoader(String)
136      * @see #clearCelestialBodyLoaders(String)
137      * @see #clearCelestialBodyLoaders()
138      */
139     public static void addCelestialBodyLoader(final String name, final CelestialBodyLoader loader) {
140         synchronized (LOADERS_MAP) {
141             List<CelestialBodyLoader> loaders = LOADERS_MAP.get(name);
142             if (loaders == null) {
143                 loaders = new ArrayList<CelestialBodyLoader>();
144                 LOADERS_MAP.put(name, loaders);
145             }
146             loaders.add(loader);
147         }
148     }

150     /** Add the default loaders for all predefined celestial bodies.
151      * @param supportedNames regular expression for supported files names
152      * (may be null if the default JPL file names are used)
153      * <p>
154      * The default loaders look for DE405 or DE406 JPL ephemerides.
155      * </p>
156      * @see <a href="ftp://ssd.jpl.nasa.gov/pub/eph/planets/unix/de405">DE405 JPL ephemerides</a>
157      * @see <a href="ftp://ssd.jpl.nasa.gov/pub/eph/planets/unix/de406">DE406 JPL ephemerides</a>
158      * @see #addCelestialBodyLoader(String, CelestialBodyLoader)
159      * @see #addDefaultCelestialBodyLoader(String)
160      * @see #clearCelestialBodyLoaders(String)
161      * @see #clearCelestialBodyLoaders()
162      * @exception OrekitException if the header constants cannot be read
163      */

169         addDefaultCelestialBodyLoader(VENUS,                   supportedNames);
170         addDefaultCelestialBodyLoader(EARTH_MOON,              supportedNames);
171         addDefaultCelestialBodyLoader(EARTH,                   supportedNames);
172         addDefaultCelestialBodyLoader(MOON,                    supportedNames);
173         addDefaultCelestialBodyLoader(MARS,                    supportedNames);
174         addDefaultCelestialBodyLoader(JUPITER,                 supportedNames);
175         addDefaultCelestialBodyLoader(SATURN,                  supportedNames);
176         addDefaultCelestialBodyLoader(URANUS,                  supportedNames);
177         addDefaultCelestialBodyLoader(NEPTUNE,                 supportedNames);
178         addDefaultCelestialBodyLoader(PLUTO,                   supportedNames);
179     }

181     /** Add the default loaders for celestial bodies.
182      * @param name name of the body (if not one of the predefined names, the method does nothing)
183      * @param supportedNames regular expression for supported files names
184      * (may be null if the default JPL file names are used)
185      * <p>
186      * The default loaders look for DE405 or DE406 JPL ephemerides.
187      * </p>
188      * @see <a href="ftp://ssd.jpl.nasa.gov/pub/eph/planets/unix/de405">DE405 JPL ephemerides</a>
189      * @see <a href="ftp://ssd.jpl.nasa.gov/pub/eph/planets/unix/de406">DE406 JPL ephemerides</a>
190      * @see #addCelestialBodyLoader(String, CelestialBodyLoader)
191      * @see #addDefaultCelestialBodyLoader(String)
192      * @see #clearCelestialBodyLoaders(String)
193      * @see #clearCelestialBodyLoaders()
194      * @exception OrekitException if the header constants cannot be read
195      */
196     public static void addDefaultCelestialBodyLoader(final String name, final String supportedNames)
197         throws OrekitException {
198 
199         CelestialBodyLoader loader = null;
200         if (name.equalsIgnoreCase(SOLAR_SYSTEM_BARYCENTER)) {
201             loader =
202                 new JPLEphemeridesLoader(supportedNames, JPLEphemeridesLoader.EphemerisType.SOLAR_SYSTEM_BARYCENTER);
203         } else if (name.equalsIgnoreCase(SUN)) {
204             loader =
205                 new JPLEphemeridesLoader(supportedNames, JPLEphemeridesLoader.EphemerisType.SUN);
206         } else if (name.equalsIgnoreCase(MERCURY)) {
207             loader =
208                 new JPLEphemeridesLoader(supportedNames, JPLEphemeridesLoader.EphemerisType.MERCURY);
209         } else if (name.equalsIgnoreCase(VENUS)) {
210             loader =
211                 new JPLEphemeridesLoader(supportedNames, JPLEphemeridesLoader.EphemerisType.VENUS);
212         } else if (name.equalsIgnoreCase(EARTH_MOON)) {
213             loader =
214                 new JPLEphemeridesLoader(supportedNames, JPLEphemeridesLoader.EphemerisType.EARTH_MOON);
215         } else if (name.equalsIgnoreCase(EARTH)) {
216             loader =
217                 new JPLEphemeridesLoader(supportedNames, JPLEphemeridesLoader.EphemerisType.EARTH);
218         } else if (name.equalsIgnoreCase(MOON)) {
219             loader =
220                 new JPLEphemeridesLoader(supportedNames, JPLEphemeridesLoader.EphemerisType.MOON);
221         } else if (name.equalsIgnoreCase(MARS)) {
222             loader =
223                 new JPLEphemeridesLoader(supportedNames, JPLEphemeridesLoader.EphemerisType.MARS);
224         } else if (name.equalsIgnoreCase(JUPITER)) {
225             loader =
226                 new JPLEphemeridesLoader(supportedNames, JPLEphemeridesLoader.EphemerisType.JUPITER);
227         } else if (name.equalsIgnoreCase(SATURN)) {
228             loader =
229                 new JPLEphemeridesLoader(supportedNames, JPLEphemeridesLoader.EphemerisType.SATURN);
230         } else if (name.equalsIgnoreCase(URANUS)) {
231             loader =
232                 new JPLEphemeridesLoader(supportedNames, JPLEphemeridesLoader.EphemerisType.URANUS);
233         } else if (name.equalsIgnoreCase(NEPTUNE)) {
234             loader =
235                 new JPLEphemeridesLoader(supportedNames, JPLEphemeridesLoader.EphemerisType.NEPTUNE);
236         } else if (name.equalsIgnoreCase(PLUTO)) {
237             loader =
238                 new JPLEphemeridesLoader(supportedNames, JPLEphemeridesLoader.EphemerisType.PLUTO);
239         }
240 
241         if (loader != null) {
242             addCelestialBodyLoader(name, loader);
243         }
244 
245     }

247     /** Clear loaders for one celestial body.
248      * <p>
249      * Calling this method also clears the celestial body that
250      * has been loaded via this {@link CelestialBodyLoader}.
251      * </p>
252      * @param name name of the body
253      * @see #addCelestialBodyLoader(String, CelestialBodyLoader)
254      * @see #clearCelestialBodyLoaders()
255      * @see #clearCelestialBodyCache(String)
256      */
257     public static void clearCelestialBodyLoaders(final String name) {
258         // use same synchronization order as in getBody to prevent deadlocks
259         synchronized (CELESTIAL_BODIES_MAP) {
260             // take advantage of reentrent synchronization as
261             // clearCelestialBodyCache uses the same lock inside
262             clearCelestialBodyCache(name);
263 
264             synchronized (LOADERS_MAP) {
265                 LOADERS_MAP.remove(name);
266             }
267         }
268     }
269 
270     /** Clear loaders for all celestial bodies.
271      * <p>
272      * Calling this method also clears all loaded celestial bodies.
273      * </p>
274      * @see #addCelestialBodyLoader(String, CelestialBodyLoader)
275      * @see #clearCelestialBodyLoaders(String)
276      * @see #clearCelestialBodyCache()
277      */
278     public static void clearCelestialBodyLoaders() {
279         synchronized (CELESTIAL_BODIES_MAP) {
280             clearCelestialBodyCache();
281 
282             synchronized (LOADERS_MAP) {
283                 LOADERS_MAP.clear();
284             }
285         }
286     }
287 
288     /** Clear the specified celestial body from the internal cache.
289      * @param name name of the body
290      */
291     public static void clearCelestialBodyCache(final String name) {
292         synchronized (CELESTIAL_BODIES_MAP) {
293             CELESTIAL_BODIES_MAP.remove(name);
294         }
295     }
296 
297     /** Clear all loaded celestial bodies.
298      * <p>
299      * Calling this method will remove all loaded bodies from the internal
300      * cache. Subsequent calls to {@link #getBody(String)} or similar methods
301      * will result in a reload of the requested body from the configured loader(s).
302      * </p>
303      */
304     public static void clearCelestialBodyCache() {
305         synchronized (CELESTIAL_BODIES_MAP) {
306             CELESTIAL_BODIES_MAP.clear();
307         }
308     }

414     /** Get a celestial body.
415      * <p>
416      * If no {@link CelestialBodyLoader} has been added by calling {@link
417      * #addCelestialBodyLoader(String, CelestialBodyLoader)
418      * addCelestialBodyLoader} or if {@link #clearCelestialBodyLoaders(String)
419      * clearCelestialBodyLoaders} has been called afterwards,
420      * the {@link #addDefaultCelestialBodyLoader(String, String)
421      * addDefaultCelestialBodyLoader} method will be called automatically,
422      * once with the default name for JPL DE ephemerides and once with the
423      * default name for IMCCE INPOP files.
424      * </p>
425      * @param name name of the celestial body
426      * @return celestial body
427      * @exception OrekitException if the celestial body cannot be built
428      */
429     public static CelestialBody getBody(final String name)
430         throws OrekitException {
431         synchronized (CELESTIAL_BODIES_MAP) {
432             CelestialBody body = CELESTIAL_BODIES_MAP.get(name);
433             if (body == null) {
434                 synchronized (LOADERS_MAP) {
435                     List<CelestialBodyLoader> loaders = LOADERS_MAP.get(name);
436                     if ((loaders == null) || loaders.isEmpty()) {
437                         addDefaultCelestialBodyLoader(name, JPLEphemeridesLoader.DEFAULT_DE_SUPPORTED_NAMES);
438                         addDefaultCelestialBodyLoader(name, JPLEphemeridesLoader.DEFAULT_INPOP_SUPPORTED_NAMES);
439                         loaders = LOADERS_MAP.get(name);
440                     }
441                     OrekitException delayedException = null;
442                     for (CelestialBodyLoader loader : loaders) {
443                         try {
444                             body = loader.loadCelestialBody(name);
445                             if (body != null) {
446                                 break;
447                             }
448                         } catch (OrekitException oe) {
449                             delayedException = oe;
450                         }
451                     }
452                     if (body == null) {
453                         throw (delayedException != null) ?
454                               delayedException :
455                               new OrekitException(OrekitMessages.NO_DATA_LOADED_FOR_CELESTIAL_BODY, name);
456                     }
457 
458                 }
459 
460                 // save the body
461                 CELESTIAL_BODIES_MAP.put(name, body);
462 
463             }
464 
465             return body;
466 
467         }
468     }

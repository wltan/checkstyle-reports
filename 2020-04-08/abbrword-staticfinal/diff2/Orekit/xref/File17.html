32  /** This atmosphere model is the realization of the Modified Harris-Priester model.
33   * <p>
34   * This model is a static one that takes into account the diurnal density bulge.
35   * It doesn't need any space weather data but a density vs. altitude table, which
36   * depends on solar activity.
37   * </p>
38   * <p>
39   * The implementation relies on the book:<br>
40   * <b>Satellite Orbits</b><br>
41   * <i>Oliver Montenbruck, Eberhard Gill</i><br>
42   * Springer 2005
43   * </p>
44   * @author Pascal Parraud
45   */

224     /** Set a user define density table to deal with different solar activities.
225      *  @param tab density vs. altitude table
226      */
227     private void setTabDensity(final double[][] tab) {
228         this.tabAltRho = new double[tab.length][];
229         for (int i = 0; i < tab.length; i++) {
230             this.tabAltRho[i] = tab[i].clone();
231         }
232     }

234     /** Get the current density table.
235      *  <p>The density table is an array such as:
236      *  <ul>
237      *   <li>tabAltRho[][0] = altitude (m)</li>
238      *   <li>tabAltRho[][1] = min density (kg/m³)</li>
239      *   <li>tabAltRho[][2] = max density (kg/m³)</li>
240      *  </ul>
241      *  <p> The altitude must be increasing without limitation in range.
242      *
243      *  <p>
244      *  The returned density table is a copy of the current one.
245      *  </p>
246      *  @return density vs. altitude table
247      */
248     public double[][] getTabDensity() {
249         final double[][] copy = new double[tabAltRho.length][];
250         for (int i = 0; i < tabAltRho.length; i++) {
251             copy[i] = tabAltRho[i].clone();
252         }
253         return copy;
254     }

256     /** Get the minimal altitude for the model.
257      * <p>No computation is possible below this altitude.</p>
258      *  @return the minimal altitude (m)
259      */
260     public double getMinAlt() {
261         return tabAltRho[0][0];
262     }
263 
264     /** Get the maximal altitude for the model.
265      * <p>Above this altitude, density is assumed to be zero.</p>
266      *  @return the maximal altitude (m)
267      */
268     public double getMaxAlt() {
269         return tabAltRho[tabAltRho.length - 1][0];
270     }

272     /** Get the local density.
273      * @param sunInEarth position of the Sun in Earth frame (m)
274      * @param posInEarth target position in Earth frame (m)
275      * @return the local density (kg/m³)
276      * @exception OrekitException if altitude is below the model minimal altitude
277      */
278     public double getDensity(final Vector3D sunInEarth, final Vector3D posInEarth)
279         throws OrekitException {
280 
281         final double posAlt = getHeight(posInEarth);
282         // Check for height boundaries
283         if (posAlt < getMinAlt()) {
284             throw new OrekitException(OrekitMessages.ALTITUDE_BELOW_ALLOWED_THRESHOLD, posAlt, getMinAlt());
285         }
286         if (posAlt > getMaxAlt()) {
287             return 0.;
288         }
289 
290         // Diurnal bulge apex direction
291         final Vector3D sunDir = sunInEarth.normalize();
292         final Vector3D bulDir = new Vector3D(sunDir.getX() * COSLAG - sunDir.getY() * SINLAG,
293                                              sunDir.getX() * SINLAG + sunDir.getY() * COSLAG,
294                                              sunDir.getZ());
295 
296         // Cosine of angle Psi between the diurnal bulge apex and the satellite
297         final double cosPsi = bulDir.normalize().dotProduct(posInEarth.normalize());
298         // (1 + cos(Psi))/2 = cos²(Psi/2)
299         final double c2Psi2 = (1. + cosPsi) / 2.;
300         final double cPsi2  = FastMath.sqrt(c2Psi2);
301         final double cosPow = (cPsi2 > MIN_COS) ? c2Psi2 * FastMath.pow(cPsi2, n - 2) : 0.;
302 
303         // Search altitude index in density table
304         int ia = 0;
305         while (ia < tabAltRho.length - 2 && posAlt > tabAltRho[ia + 1][0]) {
306             ia++;
307         }
308 
309         // Fractional satellite height
310         final double dH = (tabAltRho[ia][0] - posAlt) / (tabAltRho[ia][0] - tabAltRho[ia + 1][0]);
311 
312         // Min exponential density interpolation
313         final double rhoMin = tabAltRho[ia][1] * FastMath.pow(tabAltRho[ia + 1][1] / tabAltRho[ia][1], dH);
314 
315         if (Precision.equals(cosPow, 0.)) {
316             return rhoMin;
317         } else {
318             // Max exponential density interpolation
319             final double rhoMax = tabAltRho[ia][2] * FastMath.pow(tabAltRho[ia + 1][2] / tabAltRho[ia][2], dH);
320             return rhoMin + (rhoMax - rhoMin) * cosPow;
321         }
322 
323     }

325     /** Get the local density at some position.
326      * @param date current date
327      * @param position current position
328      * @param frame the frame in which is defined the position
329      * @return local density (kg/m³)
330      * @exception OrekitException if some frame conversion cannot be performed
331      *            or if altitude is below the model minimal altitude
332      */
333     public double getDensity(final AbsoluteDate date, final Vector3D position, final Frame frame)
334         throws OrekitException {
335 
336         // Sun position in earth frame
337         final Vector3D sunInEarth = sun.getPVCoordinates(date, earth.getBodyFrame()).getPosition();
338 
339         // Target position in earth frame
340         final Vector3D posInEarth = frame.getTransformTo(earth.getBodyFrame(), date).transformPosition(position);
341 
342         return getDensity(sunInEarth, posInEarth);
343     }

345     /** Get the inertial velocity of atmosphere molecules.
346      * <p>
347      * Here the case is simplified : atmosphere is supposed to have a null velocity
348      * in earth frame.
349      * </p>
350      * @param date current date
351      * @param position current position
352      * @param frame the frame in which is defined the position
353      * @return velocity (m/s) (defined in the same frame as the position)
354      * @exception OrekitException if some frame conversion cannot be performed
355      */
356     public Vector3D getVelocity(final AbsoluteDate date, final Vector3D position, final Frame frame)
357         throws OrekitException {
358         final Transform bodyToFrame = earth.getBodyFrame().getTransformTo(frame, date);
359         final Vector3D posInBody    = bodyToFrame.getInverse().transformPosition(position);
360         final PVCoordinates pvBody  = new PVCoordinates(posInBody, Vector3D.ZERO);
361         final PVCoordinates pvFrame = bodyToFrame.transformPVCoordinates(pvBody);
362         return pvFrame.getVelocity();
363     }

365     /** Get the height above the Earth for the given position.
366      *  <p>
367      *  The height computation is an approximation valid for the considered atmosphere.
368      *  </p>
369      *  @param position current position in Earth frame
370      *  @return height (m)
371      */
372     private double getHeight(final Vector3D position) {
373         final double a    = earth.getEquatorialRadius();
374         final double f    = earth.getFlattening();
375         final double e2   = f * (2. - f);
376         final double r    = position.getNorm();
377         final double sl   = position.getZ() / r;
378         final double cl2  = 1. - sl * sl;
379         final double coef = FastMath.sqrt((1. - e2) / (1. - e2 * cl2));
380 
381         return r - a * coef;
382     }

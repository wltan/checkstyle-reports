186     /** Base name of the resource bundle in classpath. */
187     private static final String RESOURCE_BASE_NAME = "assets/org/orekit/localization/OrekitMessages";
188 
189     /** Source English format. */
190     private final String sourceFormat;
191 
192     /** Simple constructor.
193      * @param sourceFormat source English format to use when no
194      * localized version is available
195      */
196     OrekitMessages(final String sourceFormat) {
197         this.sourceFormat = sourceFormat;
198     }
199 
200     /** {@inheritDoc} */
201     public String getSourceString() {
202         return sourceFormat;
203     }
204 
205     /** {@inheritDoc} */
206     public String getLocalizedString(final Locale locale) {
207         try {
208             final ResourceBundle bundle =
209                     ResourceBundle.getBundle(RESOURCE_BASE_NAME, locale, new UTF8Control());
210             if (bundle.getLocale().getLanguage().equals(locale.getLanguage())) {
211                 final String translated = bundle.getString(name());
212                 if ((translated != null) &&
213                     (translated.length() > 0) &&
214                     (!translated.toLowerCase().contains("missing translation"))) {
215                     // the value of the resource is the translated format
216                     return translated;
217                 }
218             }
219 
220         } catch (MissingResourceException mre) {
221             // do nothing here
222         }
223 
224         // either the locale is not supported or the resource is not translated or
225         // it is unknown: don't translate and fall back to using the source format
226         return sourceFormat;
227 
228     }

230     /** Control class loading properties in UTF-8 encoding.
231      * <p>
232      * This class has been very slightly adapted from BalusC answer to question: <a
233      * href="http://stackoverflow.com/questions/4659929/how-to-use-utf-8-in-resource-properties-with-resourcebundle">
234      * How to use UTF-8 in resource properties with ResourceBundle</a>.
235      * </p>
236      * @since 6.0
237      */
238     public static class UTF8Control extends ResourceBundle.Control {
239 
240         /** {@inheritDoc} */
241         @Override
242         public ResourceBundle newBundle(final String baseName, final Locale locale, final String format,
243                                         final ClassLoader loader, final boolean reload)
244             throws IllegalAccessException, InstantiationException, IOException {
245             // The below is a copy of the default implementation.
246             final String bundleName = toBundleName(baseName, locale);
247             final String resourceName = toResourceName(bundleName, "utf8");
248             ResourceBundle bundle = null;
249             InputStream stream = null;
250             if (reload) {
251                 final URL url = loader.getResource(resourceName);
252                 if (url != null) {
253                     final URLConnection connection = url.openConnection();
254                     if (connection != null) {
255                         connection.setUseCaches(false);
256                         stream = connection.getInputStream();
257                     }
258                 }
259             } else {
260                 stream = loader.getResourceAsStream(resourceName);
261             }
262             if (stream != null) {
263                 try {
264                     // Only this line is changed to make it to read properties files as UTF-8.
265                     bundle = new PropertyResourceBundle(new InputStreamReader(stream, "UTF-8"));
266                 } finally {
267                     stream.close();
268                 }
269             }
270             return bundle;
271         }

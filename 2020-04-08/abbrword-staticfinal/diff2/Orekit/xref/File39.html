42  /** Loader for bulletin A files.
43   * <p>Bulletin A files contain {@link EOPEntry
44   * Earth Orientation Parameters} for a few days periods, they
45   * correspond to rapid data estimations, suitable for near-real time
46   * and prediction purposes. Prediction series are only available for
47   * pole motion xp, yp and UT1-UTC, they are not available for
48   * pole offsets (Δδψ/Δδε and x/y).</p>
49   * <p>A bulletin A published on Modified Julian Day mjd (nominally a
50   * Thursday) will generally contain:
51   * </p>
52   * <ul>
53   *   <li>rapid service xp, yp and UT1-UTC data from mjd-6 to mjd</li>
54   *   <li>prediction xp, yp and UT1-UTC data from mjd+1 to mjd+365</li>
55   *   <li>if it is first bulletin of month m, final values xp, yp and
56   *       UT1-UTC data from day 2 of month m-2 to day 1 of month m-1</li>
57   *   <li>rapid service pole offsets Δδψ/Δδε and x/y if available, for some
58   *       varying period somewhere from mjd-30 to mjd-10 (see below)</li>
59   *   <li>if it is first bulletin of month m, final values pole offsets
60   *       Δδψ/Δδε and x/y data from day 2 of month m-2 to day 1 of month
61   *       m-1</li>
62   * </ul>
63   * <p>
64   * There are some discrepancies in the rapid service time range above,
65   * mainly when the nominal publication Thursday corresponds to holidays.
66   * In this case a bulletin may be published the day before and have a 6
67   * days span only for rapid data, and a later bulletin will have an 8 days
68   * span to recover the normal schedule. This occurred for bulletin A Vol.
69   * XVIII No. 047, bulletin A Vol. XVIII No. 048, bulletin A Vol. XXI No.
70   * 052 and bulletin A Vol. XXII No. 001.
71   * </p>
72   * <p>Rapid service for pole offsets appears irregular. As extreme examples
73   * bulletin A Vol. XXVI No. 037 from 2013-09-12 contained 15 entries
74   * for pole offsets, from mjd-22 to mjd-8, bulletin A Vol. XXVI No. 039
75   * from 2013-09-26 contained only 3 entries for pole offsets, from mjd-15
76   * to mjd-13, and bulletin A Vol. XXVI No. 040 from 2013-10-03 contained no
77   * rapid service pole offsets at all, it contained only final values. Despite
78   * this irregularity, rapid service data is continuous over consecutive files,
79   * so the mean number of entries is 7 as the files are published on a weekly
80   * basis.
81   * </p>
82   * <p>
83   * There are no prediction data for pole offsets.
84   * </p>
85   * <p>
86   * This loader reads both the rapid service, the prediction and the final
87   * values parts. As successive files have overlaps between all these sections,
88   * values extracted from latest files (with respect to the covered dates)
89   * override values extracted from earlier files, regardless of the files
90   * reading order. If numerous bulletins A covering more than one year are read,
91   * one particular date will typically appear in the prediction section of
92   * 52 or 53 files, then in the rapid data section of one file, then it will
93   * be missing in a few files, and will finally appear a last time in the
94   * final values sections of a last file. In this case, the value retained
95   * will be the one extracted from the final values section in the more
96   * recent file.
97   * </p>
98   * <p>
99   * If only one bulletin A file is read and it correspond to the first bulletin
100  * of a month, it will have a roughly one month wide hole between the
101  * final data and the rapid data. This hole will trigger an error as EOP
102  * continuity is checked by default for at most 5 days holes. In this case,
103  * users should call something like {@link FramesFactory#setEOPContinuityThreshold(double)
104  * FramesFactory.setEOPContinuityThreshold(Constants.JULIAN_YEAR)} to prevent
105  * the error to be triggered.
106  * </p>
107  * <p>The bulletin A files are recognized thanks to their base names,
108  * which must match the pattern <code>bulletina-xxxx-###.txt</code>,
109  * (or the same ending with <code>.gz</code> for gzip-compressed files)
110  * where x stands for a roman numeral character and # stands for a digit
111  * character.</p>
112  * <p>
113  * This class is immutable and hence thread-safe
114  * </p>
115  * @author Luc Maisonobe
116  * @since 7.0
117  */
118 class BulletinAFilesLoader implements EOPHistoryLoader {

120     /** Conversion factor. */
121     private static final double MILLI_ARC_SECONDS_TO_RADIANS = Constants.ARC_SECONDS_TO_RADIANS / 1000;

123     /** Regular expression matching blanks at start of line. */
124     private static final String LINE_START_REGEXP     = "^\\p{Blank}+";
125 
126     /** Regular expression matching blanks at end of line. */
127     private static final String LINE_END_REGEXP       = "\\p{Blank}*$";
128 
129     /** Regular expression matching integers. */
130     private static final String INTEGER_REGEXP        = "[-+]?\\p{Digit}+";
131 
132     /** Regular expression matching real numbers. */
133     private static final String REAL_REGEXP           = "[-+]?(?:(?:\\p{Digit}+(?:\\.\\p{Digit}*)?)|(?:\\.\\p{Digit}+))(?:[eE][-+]?\\p{Digit}+)?";
134 
135     /** Regular expression matching an integer field to store. */
136     private static final String STORED_INTEGER_FIELD  = "\\p{Blank}*(" + INTEGER_REGEXP + ")";
137 
138     /** regular expression matching a Modified Julian Day field to store. */
139     private static final String STORED_MJD_FIELD      = "\\p{Blank}+(\\p{Digit}\\p{Digit}\\p{Digit}\\p{Digit}\\p{Digit})";
140 
141     /** Regular expression matching a real field to store. */
142     private static final String STORED_REAL_FIELD     = "\\p{Blank}+(" + REAL_REGEXP + ")";
143 
144     /** Regular expression matching a real field to ignore. */
145     private static final String IGNORED_REAL_FIELD    = "\\p{Blank}+" + REAL_REGEXP;

193                         STORED_INTEGER_FIELD + STORED_INTEGER_FIELD + STORED_INTEGER_FIELD +

195                         STORED_REAL_FIELD +
196                         STORED_REAL_FIELD +
197                         STORED_REAL_FIELD +
198                         LINE_END_REGEXP),

201         // section 3 always contain prediction data without error fields
202         //

204         //         The following formulas will not reproduce the predictions given below,

206         //
207         //         x =  0.0969 + 0.1110 cos A - 0.0103 sin A - 0.0435 cos C - 0.0171 sin C

307         /** Header pattern. */
308         private final Pattern header;
309 
310         /** Data pattern. */
311         private final Pattern data;
312 
313         /** Simple constructor.
314          * @param headerRegExp regular expression for header
315          * @param dataRegExp regular expression for data
316          */
317         Section(final String headerRegExp, final String dataRegExp) {
318             this.header = Pattern.compile(headerRegExp);
319             this.data   = Pattern.compile(dataRegExp);
320         }
321 
322         /** Check if a line matches the section header.
323          * @param line line to check
324          * @return true if the line matches the header
325          */
326         public boolean matchesHeader(final String line) {
327             return header.matcher(line).matches();
328         }
329 
330         /** Get the data fields from a line.
331          * @param line line to parse
332          * @return extracted fields, or null if line does not match data format
333          */
334         public String[] getFields(final String line) {
335             final Matcher matcher = data.matcher(line);
336             if (matcher.matches()) {
337                 final String[] fields = new String[matcher.groupCount()];
338                 for (int i = 0; i < fields.length; ++i) {
339                     fields[i] = matcher.group(i + 1);
340                 }
341                 return fields;
342             } else {
343                 return null;
344             }
345         }

368     /** Internal class performing the parsing. */
369     private static class Parser implements DataLoader {
370 
371         /** Map for xp, yp, dut1 fields read in different sections. */
372         private final Map<Integer, double[]> eopFieldsMap;
373 
374         /** Map for pole offsets fields read in different sections. */
375         private final Map<Integer, double[]> poleOffsetsFieldsMap;
376 
377         /** Current line number. */
378         private int lineNumber;
379 
380         /** Current line. */
381         private String line;
382 
383         /** Earliest parsed data. */
384         private int mjdMin;
385 
386         /** Latest parsed data. */
387         private int mjdMax;
388 
389         /** First MJD parsed in current file. */
390         private int firstMJD;
391 
392         /** Simple constructor.
393          */
394         Parser() {
395             this.eopFieldsMap         = new HashMap<Integer, double[]>();
396             this.poleOffsetsFieldsMap = new HashMap<Integer, double[]>();
397             this.lineNumber           = 0;
398             this.mjdMin               = Integer.MAX_VALUE;
399             this.mjdMax               = Integer.MIN_VALUE;
400             this.firstMJD             = -1;
401         }
402 
403         /** {@inheritDoc} */
404         public boolean stillAcceptsData() {
405             return true;
406         }

408         /** {@inheritDoc} */
409         public void loadData(final InputStream input, final String name)
410             throws OrekitException, IOException {
411 
412             // set up a reader for line-oriented bulletin A files
413             final BufferedReader reader = new BufferedReader(new InputStreamReader(input, "UTF-8"));
414             lineNumber =  0;
415             firstMJD   = -1;
416 
417             // loop over sections
418             final List<Section> remaining = new ArrayList<Section>();
419             remaining.addAll(Arrays.asList(Section.values()));
420             for (Section section = nextSection(remaining, reader, name);
421                  section != null;
422                  section = nextSection(remaining, reader, name)) {
423 
424                 switch (section) {
425                     case EOP_RAPID_SERVICE :
426                     case EOP_FINAL_VALUES  :
427                     case EOP_PREDICTION    :
428                         loadXYDT(section, reader, name);
429                         break;
430                     case POLE_OFFSETS_IAU_1980_RAPID_SERVICE :
431                     case POLE_OFFSETS_IAU_1980_FINAL_VALUES  :
432                         loadPoleOffsets(section, false, reader, name);
433                         break;
434                     case POLE_OFFSETS_IAU_2000_RAPID_SERVICE :
435                     case POLE_OFFSETS_IAU_2000_FINAL_VALUES  :
436                         loadPoleOffsets(section, true, reader, name);
437                         break;
438                     default :
439                         // this should never happen
440                         throw new OrekitInternalError(null);
441                 }
442 
443                 // remove the already parsed section from the list
444                 remaining.remove(section);
445 
446             }
447 
448             // check that the mandatory sections have been parsed
449             if (remaining.contains(Section.EOP_RAPID_SERVICE) ||
450                 remaining.contains(Section.EOP_PREDICTION) ||
451                 (remaining.contains(Section.POLE_OFFSETS_IAU_1980_RAPID_SERVICE) ^
452                  remaining.contains(Section.POLE_OFFSETS_IAU_2000_RAPID_SERVICE)) ||
453                 (remaining.contains(Section.POLE_OFFSETS_IAU_1980_FINAL_VALUES) ^
454                  remaining.contains(Section.POLE_OFFSETS_IAU_2000_FINAL_VALUES))) {
455                 throw new OrekitException(OrekitMessages.NOT_A_SUPPORTED_IERS_DATA_FILE, name);
456             }
457 
458         }

460         /** Fill EOP history obtained after reading several files.
461          * @param history history to fill up
462          * @exception OrekitException if UTC time scale cannot be retrieved
463          */
464         public void fill(final SortedSet<EOPEntry> history)
465             throws OrekitException {
466 
467             double[] currentEOP = null;
468             double[] nextEOP    = eopFieldsMap.get(mjdMin);
469             for (int mjd = mjdMin; mjd <= mjdMax; ++mjd) {
470 
471                 final double[] currentPole = poleOffsetsFieldsMap.get(mjd);
472 
473                 final double[] previousEOP = currentEOP;
474                 currentEOP = nextEOP;
475                 nextEOP    = eopFieldsMap.get(mjd + 1);
476 
477                 if (currentEOP == null) {
478                     if (currentPole != null) {
479                         // we have only pole offsets for this date
480                         history.add(new EOPEntry(mjd,
481                                                  0.0, 0.0, 0.0, 0.0,
482                                                  currentPole[1] * MILLI_ARC_SECONDS_TO_RADIANS,
483                                                  currentPole[2] * MILLI_ARC_SECONDS_TO_RADIANS,
484                                                  currentPole[3] * MILLI_ARC_SECONDS_TO_RADIANS,
485                                                  currentPole[4] * MILLI_ARC_SECONDS_TO_RADIANS));
486                     }
487                 } else {
488 
489                     // compute LOD as the opposite of the time derivative of UT1-UTC
490                     final double lod;
491                     if (previousEOP == null) {
492                         if (nextEOP == null) {
493                             // isolated point
494                             lod = 0;
495                         } else {
496                             // first entry, we use a forward difference
497                             lod = currentEOP[3] - nextEOP[3];
498                         }
499                     } else {
500                         if (nextEOP == null) {
501                             // last entry, we use a backward difference
502                             lod = previousEOP[3] - currentEOP[3];
503                         } else {
504                             // regular entry, we use a centered difference
505                             lod = 0.5 * (previousEOP[3] - nextEOP[3]);
506                         }
507                     }
508 
509                     if (currentPole == null) {
510                         // we have only EOP for this date
511                         history.add(new EOPEntry(mjd,
512                                                  currentEOP[3], lod,
513                                                  currentEOP[1] * Constants.ARC_SECONDS_TO_RADIANS,
514                                                  currentEOP[2] * Constants.ARC_SECONDS_TO_RADIANS,
515                                                  0.0, 0.0, 0.0, 0.0));
516                     } else {
517                         // we have complete data
518                         history.add(new EOPEntry(mjd,
519                                                  currentEOP[3], lod,
520                                                  currentEOP[1]  * Constants.ARC_SECONDS_TO_RADIANS,
521                                                  currentEOP[2]  * Constants.ARC_SECONDS_TO_RADIANS,
522                                                  currentPole[1] * MILLI_ARC_SECONDS_TO_RADIANS,
523                                                  currentPole[2] * MILLI_ARC_SECONDS_TO_RADIANS,
524                                                  currentPole[3] * MILLI_ARC_SECONDS_TO_RADIANS,

526                     }

533         /** Skip to next section header.
534          * @param sections sections to check for
535          * @param reader reader from where file content is obtained
536          * @param name name of the file (or zip entry)
537          * @return the next section or null if no section is found until end of file
538          * @exception IOException if data can't be read
539          */
540         private Section nextSection(final List<Section> sections,
541                                     final BufferedReader reader, final String name)
542             throws IOException {
543 
544             for (line = reader.readLine(); line != null; line = reader.readLine()) {
545                 ++lineNumber;
546                 for (Section section : sections) {
547                     if (section.matchesHeader(line)) {
548                         return section;
549                     }
550                 }
551             }
552 
553             // we have reached end of file and not found a matching section header
554             return null;
555 
556         }

558         /** Read X, Y, UT1-UTC.
559          * @param section section to parse
560          * @param reader reader from where file content is obtained
561          * @param name name of the file (or zip entry)
562          * @exception IOException if data can't be read
563          * @exception OrekitException if some data is missing or if some loader specific error occurs
564          */
565         private void loadXYDT(final Section section, final BufferedReader reader, final String name)
566             throws OrekitException, IOException {
567 
568             boolean inValuesPart = false;
569             for (line = reader.readLine(); line != null; line = reader.readLine()) {
570                 lineNumber++;
571                 final String[] fields = section.getFields(line);
572                 if (fields != null) {
573 
574                     // we are within the values part
575                     inValuesPart = true;
576 
577                     // this is a data line, build an entry from the extracted fields
578                     final int year  = Integer.parseInt(fields[0]);
579                     final int month = Integer.parseInt(fields[1]);
580                     final int day   = Integer.parseInt(fields[2]);
581                     final int mjd   = Integer.parseInt(fields[3]);
582                     final DateComponents dc = new DateComponents(DateComponents.MODIFIED_JULIAN_EPOCH, mjd);
583                     if ((dc.getYear() % 100) != (year % 100) ||
584                          dc.getMonth() != month ||
585                          dc.getDay() != day) {
586                         throw new OrekitException(OrekitMessages.INCONSISTENT_DATES_IN_IERS_FILE,
587                                                   name, year, month, day, mjd);
588                     }
589                     mjdMin = FastMath.min(mjdMin, mjd);
590                     mjdMax = FastMath.max(mjdMax, mjd);
591                     if (firstMJD < 0) {
592                         // store the first mjd parsed
593                         firstMJD = mjd;
594                     }
595 
596                     // get the entry at the same date if it was already parsed
597                     final double[] eop;
598                     if (eopFieldsMap.containsKey(mjd)) {
599                         eop = eopFieldsMap.get(mjd);
600                     } else {
601                         eop = new double[4];
602                         eopFieldsMap.put(mjd, eop);
603                     }
604 
605                     if (eop[0] <= firstMJD) {
606                         // either it is the first time we parse this date (eop[0] = 0),
607                         // or the new parsed data is from a more recent file
608                         // in both case, we should update the array
609                         eop[0] = firstMJD;
610                         eop[1] = Double.parseDouble(fields[4]);
611                         eop[2] = Double.parseDouble(fields[5]);
612                         eop[3] = Double.parseDouble(fields[6]);
613                     }
614 
615                 } else if (inValuesPart) {
616                     // we leave values part
617                     return;
618                 }
619             }
620 
621             throw new OrekitException(OrekitMessages.UNEXPECTED_END_OF_FILE_AFTER_LINE,
622                                       name, lineNumber);
623 
624         }

626         /** Read EOP data.
627          * @param section section to parse
628          * @param isNonRotatingOrigin if true, the file contain Non-Rotating Origin nutation corrections
629          * @param reader reader from where file content is obtained
630          * @param name name of the file (or zip entry)
631          * @exception IOException if data can't be read
632          * @exception OrekitException if some data is missing or if some loader specific error occurs
633          */
634         private void loadPoleOffsets(final Section section, final boolean isNonRotatingOrigin,
635                                      final BufferedReader reader, final String name)
636             throws OrekitException, IOException {
637 
638             boolean inValuesPart = false;
639             for (line = reader.readLine(); line != null; line = reader.readLine()) {
640                 lineNumber++;
641                 final String[] fields = section.getFields(line);
642                 if (fields != null) {
643 
644                     // we are within the values part
645                     inValuesPart = true;
646 
647                     // this is a data line, build an entry from the extracted fields
648                     final int mjd = Integer.parseInt(fields[0]);
649                     mjdMin = FastMath.min(mjdMin, mjd);
650                     mjdMax = FastMath.max(mjdMax, mjd);
651 
652                     // get the entry at the same date if it was already parsed
653                     final double[] pole;
654                     if (poleOffsetsFieldsMap.containsKey(mjd)) {
655                         pole = poleOffsetsFieldsMap.get(mjd);
656                     } else {
657                         pole = new double[5];
658                         poleOffsetsFieldsMap.put(mjd, pole);
659                     }
660

666                         if (isNonRotatingOrigin) {

668                             pole[2] = Double.parseDouble(fields[2]);
669                         } else {
670                             pole[3] = Double.parseDouble(fields[1]);
671                             pole[4] = Double.parseDouble(fields[2]);
672                         }
673                     }
674 
675                 } else if (inValuesPart) {
676                     // we leave values part
677                     return;
678                 }
679             }
680 
681             throw new OrekitException(OrekitMessages.UNEXPECTED_END_OF_FILE_AFTER_LINE,
682                                       name, lineNumber);
683 
684         }

118     /**
119      * Validates only 1 vertex is tangential (shared) between the interior and exterior of a polygon
120      */
121     protected void validateHole(BaseLineStringBuilder shell, BaseLineStringBuilder hole) {
122         HashSet exterior = Sets.newHashSet(shell.points);
123         HashSet interior = Sets.newHashSet(hole.points);
124         exterior.retainAll(interior);
125         if (exterior.size() >= 2) {
126             throw new InvalidShapeException("Invalid polygon, interior cannot share more than one point with the exterior");
127         }
128     }

130     /**
131      * The coordinates setup by the builder will be assembled to a polygon. The result will consist of
132      * a set of polygons. Each of these components holds a list of linestrings defining the polygon: the
133      * first set of coordinates will be used as the shell of the polygon. The others are defined to holes
134      * within the polygon.
135      * This Method also wraps the polygons at the dateline. In order to this fact the result may
136      * contains more polygons and less holes than defined in the builder it self.
137      * 
138      * @return coordinates of the polygon
139      */
140     public Coordinate[][][] coordinates() {
141         int numEdges = shell.points.size()-1; // Last point is repeated 
142         for (int i = 0; i < holes.size(); i++) {
143             numEdges += holes.get(i).points.size()-1;
144             validateHole(shell, this.holes.get(i));
145         }
146 
147         Edge[] edges = new Edge[numEdges];
148         Edge[] holeComponents = new Edge[holes.size()];
149         int offset = createEdges(0, orientation, shell, null, edges, 0);
150         for (int i = 0; i < holes.size(); i++) {
151             int length = createEdges(i+1, orientation, shell, this.holes.get(i), edges, offset);
152             holeComponents[i] = edges[offset];
153             offset += length;
154         }
155 
156         int numHoles = holeComponents.length;
157 
158         numHoles = merge(edges, 0, intersections(+DATELINE, edges), holeComponents, numHoles);
159         numHoles = merge(edges, 0, intersections(-DATELINE, edges), holeComponents, numHoles);
160 
161         return compose(edges, holeComponents, numHoles);
162     }

188     public Geometry buildGeometry(GeometryFactory factory, boolean fixDateline) {
189         if(fixDateline) {
190             Coordinate[][][] polygons = coordinates();
191             return polygons.length == 1
192                     ? polygon(factory, polygons[0])
193                     : multipolygon(factory, polygons);
194         } else {
195             return toPolygon(factory);
196         }
197     }

199     public Polygon toPolygon() {
200         return toPolygon(FACTORY);
201     }

203     protected Polygon toPolygon(GeometryFactory factory) {
204         final LinearRing shell = linearRing(factory, this.shell.points);
205         final LinearRing[] holes = new LinearRing[this.holes.size()];
206         Iterator<BaseLineStringBuilder<?>> iterator = this.holes.iterator();
207         for (int i = 0; iterator.hasNext(); i++) {
208             holes[i] = linearRing(factory, iterator.next().points);
209         }
210         return factory.createPolygon(shell, holes);
211     }

222     protected static Polygon polygon(GeometryFactory factory, Coordinate[][] polygon) {
223         LinearRing shell = factory.createLinearRing(polygon[0]);
224         LinearRing[] holes;
225         
226         if(polygon.length > 1) {
227             holes = new LinearRing[polygon.length-1];
228             for (int i = 0; i < holes.length; i++) {
229                 holes[i] = factory.createLinearRing(polygon[i+1]);
230             }
231         } else {
232             holes = null;
233         }
234         return factory.createPolygon(shell, holes);
235     }

237     /**
238      * Create a Multipolygon from a set of coordinates. Each primary array contains a polygon which
239      * in turn contains an array of linestrings. These line Strings are represented as an array of
240      * coordinates. The first linestring will be the shell of the polygon the others define holes
241      * within the polygon.
242      *      
243      * @param factory {@link GeometryFactory} to use
244      * @param polygons definition of polygons
245      * @return a new Multipolygon
246      */
247     protected static MultiPolygon multipolygon(GeometryFactory factory, Coordinate[][][] polygons) {
248         Polygon[] polygonSet = new Polygon[polygons.length];
249         for (int i = 0; i < polygonSet.length; i++) {
250             polygonSet[i] = polygon(factory, polygons[i]);
251         }
252         return factory.createMultiPolygon(polygonSet);
253     }

255     /**
256      * This method sets the component id of all edges in a ring to a given id and shifts the
257      * coordinates of this component according to the dateline 
258      * 
259      * @param edge An arbitrary edge of the component
260      * @param id id to apply to the component
261      * @param edges a list of edges to which all edges of the component will be added (could be <code>null</code>)
262      * @return number of edges that belong to this component
263      */
264     private static int component(final Edge edge, final int id, final ArrayList<Edge> edges) {
265         // find a coordinate that is not part of the dateline 
266         Edge any = edge;
267         while(any.coordinate.x == +DATELINE || any.coordinate.x == -DATELINE) {
268             if((any = any.next) == edge) {
269                 break;   
270             }
271         }
272 
273         double shiftOffset = any.coordinate.x > DATELINE ? DATELINE : (any.coordinate.x < -DATELINE ? -DATELINE : 0);
274         if (debugEnabled()) {
275             LOGGER.debug("shift: {[]}", shiftOffset);
276         }
277 
278         // run along the border of the component, collect the
279         // edges, shift them according to the dateline and
280         // update the component id
281         int length = 0, connectedComponents = 0;
282         // if there are two connected components, splitIndex keeps track of where to split the edge array
283         // start at 1 since the source coordinate is shared
284         int splitIndex = 1;
285         Edge current = edge;
286         Edge prev = edge;
287         // bookkeep the source and sink of each visited coordinate
288         HashMap<Coordinate, Pair<Edge, Edge>> visitedEdge = new HashMap<>();
289         do {
290             current.coordinate = shift(current.coordinate, shiftOffset);
291             current.component = id;
292 
293             if (edges != null) {
294                 // found a closed loop - we have two connected components so we need to slice into two distinct components
295                 if (visitedEdge.containsKey(current.coordinate)) {
296                     if (connectedComponents > 0 && current.next != edge) {
297                         throw new InvalidShapeException("Shape contains more than one shared point");
298                     }
299 
300                     // a negative id flags the edge as visited for the edges(...) method.
301                     // since we're splitting connected components, we want the edges method to visit
302                     // the newly separated component
303                     final int visitID = -id;
304                     Edge firstAppearance = visitedEdge.get(current.coordinate).getRight();
305                     // correct the graph pointers by correcting the 'next' pointer for both the
306                     // first appearance and this appearance of the edge
307                     Edge temp = firstAppearance.next;
308                     firstAppearance.next = current.next;
309                     current.next = temp;
310                     current.component = visitID;
311                     // backtrack until we get back to this coordinate, setting the visit id to
312                     // a non-visited value (anything positive)
313                     do {
314                         prev.component = visitID;
315                         prev = visitedEdge.get(prev.coordinate).getLeft();
316                         ++splitIndex;
317                     } while (!current.coordinate.equals(prev.coordinate));
318                     ++connectedComponents;
319                 } else {
320                     visitedEdge.put(current.coordinate, Pair.of(prev, current));
321                 }
322                 edges.add(current);
323                 prev = current;
324             }
325             length++;
326         } while(connectedComponents == 0 && (current = current.next) != edge);
327 
328         return (splitIndex != 1) ? length-splitIndex: length;
329     }

331     /**
332      * Compute all coordinates of a component
333      * @param component an arbitrary edge of the component
334      * @param coordinates Array of coordinates to write the result to
335      * @return the coordinates parameter
336      */
337     private static Coordinate[] coordinates(Edge component, Coordinate[] coordinates) {
338         for (int i = 0; i < coordinates.length; i++) {
339             coordinates[i] = (component = component.next).coordinate;
340         }
341         return coordinates;
342     }

344     private static Coordinate[][][] buildCoordinates(ArrayList<ArrayList<Coordinate[]>> components) {
345         Coordinate[][][] result = new Coordinate[components.size()][][];
346         for (int i = 0; i < result.length; i++) {
347             ArrayList<Coordinate[]> component = components.get(i);
348             result[i] = component.toArray(new Coordinate[component.size()][]);
349         }
350 
351         if(debugEnabled()) {
352             for (int i = 0; i < result.length; i++) {
353                 LOGGER.debug("Component {[]}:", i);
354                 for (int j = 0; j < result[i].length; j++) {
355                     LOGGER.debug("\t" + Arrays.toString(result[i][j]));
356                 }
357             }
358         }
359 
360         return result;
361     }

363     private static final Coordinate[][] EMPTY = new Coordinate[0][];
364 
365     private static Coordinate[][] holes(Edge[] holes, int numHoles) {
366         if (numHoles == 0) {
367             return EMPTY;
368         }
369         final Coordinate[][] points = new Coordinate[numHoles][];
370 
371         for (int i = 0; i < numHoles; i++) {
372             int length = component(holes[i], -(i+1), null); // mark as visited by inverting the sign
373             points[i] = coordinates(holes[i], new Coordinate[length+1]);
374         }
375 
376         return points;
377     }

379     private static Edge[] edges(Edge[] edges, int numHoles, ArrayList<ArrayList<Coordinate[]>> components) {
380         ArrayList<Edge> mainEdges = new ArrayList<>(edges.length);
381 
382         for (int i = 0; i < edges.length; i++) {
383             if (edges[i].component >= 0) {
384                 int length = component(edges[i], -(components.size()+numHoles+1), mainEdges);
385                 ArrayList<Coordinate[]> component = new ArrayList<>();
386                 component.add(coordinates(edges[i], new Coordinate[length+1]));
387                 components.add(component);
388             }
389         }
390 
391         return mainEdges.toArray(new Edge[mainEdges.size()]);
392     }

394     private static Coordinate[][][] compose(Edge[] edges, Edge[] holes, int numHoles) {
395         final ArrayList<ArrayList<Coordinate[]>> components = new ArrayList<>();
396         assign(holes, holes(holes, numHoles), numHoles, edges(edges, numHoles, components), components);
397         return buildCoordinates(components);
398     }

400     private static void assign(Edge[] holes, Coordinate[][] points, int numHoles, Edge[] edges, ArrayList<ArrayList<Coordinate[]>> components) {
401         // Assign Hole to related components
402         // To find the new component the hole belongs to all intersections of the
403         // polygon edges with a vertical line are calculated. This vertical line
404         // is an arbitrary point of the hole. The polygon edge next to this point
405         // is part of the polygon the hole belongs to.
406         if (debugEnabled()) {
407             LOGGER.debug("Holes: " + Arrays.toString(holes));
408         }
409         for (int i = 0; i < numHoles; i++) {
410             final Edge current = new Edge(holes[i].coordinate, holes[i].next);
411             // the edge intersects with itself at its own coordinate.  We need intersect to be set this way so the binary search 
412             // will get the correct position in the edge list and therefore the correct component to add the hole
413             current.intersect = current.coordinate;
414             final int intersections = intersections(current.coordinate.x, edges);
415             // if no intersection is found then the hole is not within the polygon, so
416             // don't waste time calling a binary search
417             final int pos;
418             boolean sharedVertex = false;
419             if (intersections == 0 || ((pos = Arrays.binarySearch(edges, 0, intersections, current, INTERSECTION_ORDER)) >= 0)
420                             && !(sharedVertex = (edges[pos].intersect.compareTo(current.coordinate) == 0)) ) {
421                 throw new InvalidShapeException("Invalid shape: Hole is not within polygon");
422             }
423             final int index = -((sharedVertex) ? 0 : pos+2);
424             final int component = -edges[index].component - numHoles - 1;
425 
426             if(debugEnabled()) {
427                 LOGGER.debug("\tposition ("+index+") of edge "+current+": " + edges[index]);
428                 LOGGER.debug("\tComponent: " + component);
429                 LOGGER.debug("\tHole intersections ("+current.coordinate.x+"): " + Arrays.toString(edges));
430             }
431 
432             components.get(component).add(points[i]);
433         }
434     }

436     private static int merge(Edge[] intersections, int offset, int length, Edge[] holes, int numHoles) {
437         // Intersections appear pairwise. On the first edge the inner of
438         // of the polygon is entered. On the second edge the outer face
439         // is entered. Other kinds of intersections are discard by the
440         // intersection function
441 
442         for (int i = 0; i < length; i += 2) {
443             Edge e1 = intersections[offset + i + 0];
444             Edge e2 = intersections[offset + i + 1];
445 
446             // If two segments are connected maybe a hole must be deleted
447             // Since Edges of components appear pairwise we need to check
448             // the second edge only (the first edge is either polygon or
449             // already handled)
450             if (e2.component > 0) {
451                 //TODO: Check if we could save the set null step
452                 numHoles--;
453                 holes[e2.component-1] = holes[numHoles];
454                 holes[numHoles] = null;
455             }
456             // only connect edges if intersections are pairwise 
457             // 1. per the comment above, the edge array is sorted by y-value of the intersection
458             // with the dateline.  Two edges have the same y intercept when they cross the 
459             // dateline thus they appear sequentially (pairwise) in the edge array. Two edges
460             // do not have the same y intercept when we're forming a multi-poly from a poly
461             // that wraps the dateline (but there are 2 ordered intercepts).  
462             // The connect method creates a new edge for these paired edges in the linked list. 
463             // For boundary conditions (e.g., intersect but not crossing) there is no sibling edge 
464             // to connect. Thus the first logic check enforces the pairwise rule
465             // 2. the second logic check ensures the two candidate edges aren't already connected by an
466             //    existing edge along the dateline - this is necessary due to a logic change in
467             //    ShapeBuilder.intersection that computes dateline edges as valid intersect points 
468             //    in support of OGC standards
469             if (e1.intersect != Edge.MAX_COORDINATE && e2.intersect != Edge.MAX_COORDINATE 
470                     && !(e1.next.next.coordinate.equals3D(e2.coordinate) && Math.abs(e1.next.coordinate.x) == DATELINE
471                     && Math.abs(e2.coordinate.x) == DATELINE) ) {
472                 connect(e1, e2);
473             }
474         }
475         return numHoles;
476     }

478     private static void connect(Edge in, Edge out) {
479         assert in != null && out != null;
480         assert in != out;
481         // Connecting two Edges by inserting the point at
482         // dateline intersection and connect these by adding
483         // two edges between this points. One per direction
484         if(in.intersect != in.next.coordinate) {
485             // NOTE: the order of the object creation is crucial here! Don't change it!
486             // first edge has no point on dateline
487             Edge e1 = new Edge(in.intersect, in.next);
488             
489             if(out.intersect != out.next.coordinate) {
490                 // second edge has no point on dateline
491                 Edge e2 = new Edge(out.intersect, out.next);
492                 in.next = new Edge(in.intersect, e2, in.intersect);
493             } else {
494                 // second edge intersects with dateline
495                 in.next = new Edge(in.intersect, out.next, in.intersect);
496             }
497             out.next = new Edge(out.intersect, e1, out.intersect);
498         } else if (in.next != out && in.coordinate != out.intersect) {
499             // first edge intersects with dateline
500             Edge e2 = new Edge(out.intersect, in.next, out.intersect);
501 
502             if(out.intersect != out.next.coordinate) {
503                 // second edge has no point on dateline
504                 Edge e1 = new Edge(out.intersect, out.next);
505                 in.next = new Edge(in.intersect, e1, in.intersect);
506                 
507             } else {
508                 // second edge intersects with dateline
509                 in.next = new Edge(in.intersect, out.next, in.intersect);
510             }
511             out.next = e2;
512         }
513     }

515     private static int createEdges(int component, Orientation orientation, BaseLineStringBuilder<?> shell,
516                                    BaseLineStringBuilder<?> hole,
517                                    Edge[] edges, int offset) {
518         // inner rings (holes) have an opposite direction than the outer rings
519         // XOR will invert the orientation for outer ring cases (Truth Table:, T/T = F, T/F = T, F/T = T, F/F = F)
520         boolean direction = (component == 0 ^ orientation == Orientation.RIGHT);
521         // set the points array accordingly (shell or hole)
522         Coordinate[] points = (hole != null) ? hole.coordinates(false) : shell.coordinates(false);
523         Edge.ring(component, direction, orientation == Orientation.LEFT, shell, points, 0, edges, offset, points.length-1);
524         return points.length-1;
525     }

527     public static class Ring<P extends ShapeBuilder> extends BaseLineStringBuilder<Ring<P>> {
528 
529         private final P parent;
530 
531         protected Ring(P parent) {
532             this(parent, new ArrayList<Coordinate>());
533         }
534 
535         protected Ring(P parent, ArrayList<Coordinate> points) {
536             super(points);
537             this.parent = parent;
538         }
539 
540         public P close() {
541             Coordinate start = points.get(0);
542             Coordinate end = points.get(points.size()-1);
543             if(start.x != end.x || start.y != end.y) {
544                 points.add(start);
545             }
546             return parent;
547         }
548 
549         @Override
550         public GeoShapeType type() {
551             return null;
552         }
553     }

111     private static class StringAppendState {
112         // remember the smallest position at which we saw something that
113         // concerns us

194     final Map<MethodDescriptor, int[]> preparedStatementMethods;
195     final Map<MethodDescriptor, int[]> executeMethods;
196     final Set<MethodDescriptor> allMethods = new HashSet<>();

198     public FindSqlInjection(BugReporter bugReporter) {
199         this.bugReporter = bugReporter;
200         this.bugAccumulator = new BugAccumulator(bugReporter);
201         Set<MethodParameter> baseExecuteMethods = new HashSet<>();
202         for(MethodDescriptor executeMethod : EXECUTE_METHODS) {
203             baseExecuteMethods.add(new MethodParameter(executeMethod, 0));
204         }
205         executeMethods = Global.getAnalysisCache().getDatabase(StringPassthruDatabase.class).findLinkedMethods(baseExecuteMethods);
206         Set<MethodParameter> basePrepareMethods = new HashSet<>();
207         for(String signature : PREPARE_STATEMENT_SIGNATURES) {
208             basePrepareMethods.add(new MethodParameter(new MethodDescriptor("java/sql/Connection", "prepareStatement", signature), 0));
209         }
210         preparedStatementMethods = Global.getAnalysisCache().getDatabase(StringPassthruDatabase.class).findLinkedMethods(basePrepareMethods);
211         allMethods.addAll(executeMethods.keySet());
212         allMethods.addAll(preparedStatementMethods.keySet());
213     }

215     @Override
216     public void visitClassContext(ClassContext classContext) {
217         JavaClass javaClass = classContext.getJavaClass();
218         if(!PreorderVisitor.hasInterestingMethod(javaClass.getConstantPool(), allMethods)) {
219             return;
220         }
221         Method[] methodList = javaClass.getMethods();
222 
223         for (Method method : methodList) {
224             MethodGen methodGen = classContext.getMethodGen(method);
225             if (methodGen == null) {
226                 continue;
227             }
228 
229             try {
230                 analyzeMethod(classContext, method);
231             } catch (DataflowAnalysisException e) {
232                 bugReporter.logError(
233                         "FindSqlInjection caught exception while analyzing " + classContext.getFullyQualifiedMethodName(method),
234                         e);
235             } catch (CFGBuilderException e) {
236                 bugReporter.logError(
237                         "FindSqlInjection caught exception while analyzing " + classContext.getFullyQualifiedMethodName(method),
238                         e);
239             } catch (RuntimeException e) {
240                 bugReporter.logError(
241                         "FindSqlInjection caught exception while analyzing " + classContext.getFullyQualifiedMethodName(method),
242                         e);
243             }
244         }
245     }

247     private boolean isStringAppend(Instruction ins, ConstantPoolGen cpg) {
248         if (ins instanceof INVOKEVIRTUAL) {
249             INVOKEVIRTUAL invoke = (INVOKEVIRTUAL) ins;
250 
251             if ("append".equals(invoke.getMethodName(cpg)) && invoke.getClassName(cpg).startsWith("java.lang.StringB")) {
252                 String sig = invoke.getSignature(cpg);
253                 char firstChar = sig.charAt(1);
254                 return firstChar == '[' || firstChar == 'L';
255             }
256         }
257 
258         return false;
259     }

261     private boolean isConstantStringLoad(Location location, ConstantPoolGen cpg)  {
262         Instruction ins = location.getHandle().getInstruction();
263         if (ins instanceof LDC) {
264             LDC load = (LDC) ins;
265             Object value = load.getValue(cpg);
266             if (value instanceof String) {
267                 return true;
268             }
269         }
270 
271         return false;
272     }

274     static final Pattern openQuotePattern = Pattern.compile("((^')|[^\\p{Alnum}]')$");
275 
276     public static boolean isOpenQuote(String s) {
277         return openQuotePattern.matcher(s).find();
278     }

280     static final Pattern closeQuotePattern = Pattern.compile("^'($|[^\\p{Alnum}])");
281 
282     public static boolean isCloseQuote(String s) {
283         return closeQuotePattern.matcher(s).find();
284     }

286     private StringAppendState updateStringAppendState(Location location, ConstantPoolGen cpg, StringAppendState stringAppendState)
287     {
288         InstructionHandle handle = location.getHandle();
289         Instruction ins = handle.getInstruction();
290         if (!isConstantStringLoad(location, cpg)) {
291             throw new IllegalArgumentException("instruction must be LDC");
292         }
293 
294         LDC load = (LDC) ins;
295         Object value = load.getValue(cpg);
296         String stringValue = ((String) value).trim();
297         if (stringValue.startsWith(",") || stringValue.endsWith(",")) {
298             stringAppendState.setSawComma(handle);
299         }
300         if (isCloseQuote(stringValue) && stringAppendState.getSawOpenQuote(handle)) {
301             stringAppendState.setSawCloseQuote(handle);
302         }
303         if (isOpenQuote(stringValue)) {
304             stringAppendState.setSawOpenQuote(handle);
305         }
306 
307         return stringAppendState;
308     }

310     private StringAppendState getStringAppendState(CFG cfg, ConstantPoolGen cpg) throws CFGBuilderException {
311         StringAppendState stringAppendState = new StringAppendState();
312         String sig = method.getSignature();
313         sig = sig.substring(0, sig.indexOf(')'));
314 
315         if (sig.indexOf("java/lang/String") >= 0) {
316             stringAppendState.setSawInitialTaint();
317         }
318         for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
319             Location location = i.next();
320             InstructionHandle handle = location.getHandle();
321             Instruction ins = handle.getInstruction();
322             if (isConstantStringLoad(location, cpg)) {
323                 stringAppendState = updateStringAppendState(location, cpg, stringAppendState);
324             } else if (isStringAppend(ins, cpg)) {
325                 stringAppendState.setSawAppend(handle);
326 
327                 Location prevLocation = getPreviousLocation(cfg, location, true);
328                 if (prevLocation != null && !isSafeValue(prevLocation, cpg)) {
329                     stringAppendState.setSawUnsafeAppend(handle);
330                 }
331 
332             } else if (ins instanceof InvokeInstruction) {
333                 InvokeInstruction inv = (InvokeInstruction) ins;
334                 String sig1 = inv.getSignature(cpg);
335                 String sig2 = sig1.substring(sig1.indexOf(')'));
336 
337                 if (sig2.indexOf("java/lang/String") >= 0) {
338                     String methodName = inv.getMethodName(cpg);
339                     String className = inv.getClassName(cpg);
340                     if ("valueOf".equals(methodName) && "java.lang.String".equals(className)
341                             && "(Ljava/lang/Object;)Ljava/lang/String;".equals(sig1)) {
342                         try {
343                             TypeDataflow typeDataflow = classContext.getTypeDataflow(method);
344                             TypeFrame frame = typeDataflow.getFactAtLocation(location);
345                             if (!frame.isValid()) {
346                                 // This basic block is probably dead
347                                 continue;
348                             }
349                             Type operandType = frame.getTopValue();
350                             if (operandType.equals(TopType.instance())) {
351                                 // unreachable
352                                 continue;
353                             }
354                             String sig3 = operandType.getSignature();
355                             if (!"Ljava/lang/String;".equals(sig3)) {
356                                 stringAppendState.setSawTaint(handle);
357                             }
358                         } catch (CheckedAnalysisException e) {
359                             stringAppendState.setSawTaint(handle);
360                         }
361                     } else if (className.startsWith("java.lang.String") || "java.lang.Long".equals(className)
362                             || "java.lang.Integer".equals(className) || "java.lang.Float".equals(className)
363                             || "java.lang.Double".equals(className) || "java.lang.Short".equals(className)
364                             || "java.lang.Byte".equals(className) || "java.lang.Character".equals(className)) {
365                         // ignore it
366                         assert true;
367                     } else if (methodName.startsWith("to") && methodName.endsWith("String") && methodName.length() > 8) {
368                         // ignore it
369                         assert true;
370                     } else if (className.startsWith("javax.servlet") && methodName.startsWith("get")) {
371                         stringAppendState.setSawTaint(handle);
372                         stringAppendState.setSawSeriousTaint(handle);
373                     } else {
374                         stringAppendState.setSawTaint(handle);
375                     }
376 
377                 }
378             } else if (ins instanceof GETFIELD) {
379                 GETFIELD getfield = (GETFIELD) ins;
380                 String sig2 = getfield.getSignature(cpg);
381                 if (sig2.indexOf("java/lang/String") >= 0) {
382                     stringAppendState.setSawTaint(handle);
383                 }
384             }
385         }
386 
387         return stringAppendState;
388     }

390     private boolean isSafeValue(Location location, ConstantPoolGen cpg) throws CFGBuilderException {
391         Instruction prevIns = location.getHandle().getInstruction();
392         if (prevIns instanceof LDC || prevIns instanceof GETSTATIC) {
393             return true;
394         }
395         if (prevIns instanceof InvokeInstruction) {
396             String methodName = ((InvokeInstruction) prevIns).getMethodName(cpg);
397             if (methodName.startsWith("to") && methodName.endsWith("String") && methodName.length() > 8) {
398                 return true;
399             }
400         }
401         if (prevIns instanceof AALOAD) {
402             CFG cfg = classContext.getCFG(method);
403 
404             Location prev = getPreviousLocation(cfg, location, true);
405             if (prev != null) {
406                 Location prev2 = getPreviousLocation(cfg, prev, true);
407                 if (prev2 != null && prev2.getHandle().getInstruction() instanceof GETSTATIC) {
408                     GETSTATIC getStatic = (GETSTATIC) prev2.getHandle().getInstruction();
409                     if ("[Ljava/lang/String;".equals(getStatic.getSignature(cpg))) {
410                         return true;
411                     }
412                 }
413             }
414         }
415         return false;
416     }

418     private @CheckForNull
419     InstructionHandle getPreviousInstruction(InstructionHandle handle, boolean skipNops) {
420         while (handle.getPrev() != null) {
421             handle = handle.getPrev();
422             Instruction prevIns = handle.getInstruction();
423             if (!(prevIns instanceof NOP && skipNops)) {
424                 return handle;
425             }
426         }
427         return null;
428     }

430     private @CheckForNull
431     Location getPreviousLocation(CFG cfg, Location startLocation, boolean skipNops) {
432         Location loc = startLocation;
433         InstructionHandle prev = getPreviousInstruction(loc.getHandle(), skipNops);
434         if (prev != null) {
435             return new Location(prev, loc.getBasicBlock());
436         }
437         BasicBlock block = loc.getBasicBlock();
438         while (true) {
439             block = cfg.getPredecessorWithEdgeType(block, EdgeTypes.FALL_THROUGH_EDGE);
440             if (block == null) {
441                 return null;
442             }
443             InstructionHandle lastInstruction = block.getLastInstruction();
444             if (lastInstruction != null) {
445                 return new Location(lastInstruction, block);
446             }
447         }
448     }

450     private BugInstance generateBugInstance(JavaClass javaClass, MethodGen methodGen, InstructionHandle handle,
451             StringAppendState stringAppendState, boolean isExecute) {
452         int priority = LOW_PRIORITY;
453         boolean sawSeriousTaint = false;
454         if (stringAppendState.getSawAppend(handle)) {
455             if (stringAppendState.getSawOpenQuote(handle) && stringAppendState.getSawCloseQuote(handle)) {
456                 priority = HIGH_PRIORITY;
457             } else if (stringAppendState.getSawComma(handle)) {
458                 priority = NORMAL_PRIORITY;
459             }
460 
461             if (!stringAppendState.getSawUnsafeAppend(handle)) {
462                 priority += 2;
463             } else if (stringAppendState.getSawSeriousTaint(handle)) {
464                 priority--;
465                 sawSeriousTaint = true;
466             } else if (!stringAppendState.getSawTaint(handle)) {
467                 priority++;
468             }
469         }
470 
471         String description;
472         if (isExecute) {
473             description = "SQL_NONCONSTANT_STRING_PASSED_TO_EXECUTE";
474         } else {
475             description = "SQL_PREPARED_STATEMENT_GENERATED_FROM_NONCONSTANT_STRING";
476         }
477 
478         BugInstance bug = new BugInstance(this, description, priority);
479         bug.addClassAndMethod(methodGen, javaClass.getSourceFileName());
480         if (sawSeriousTaint) {
481             bug.addString("non-constant SQL string involving HTTP taint");
482         }
483 
484         return bug;
485     }

491     private void analyzeMethod(ClassContext classContext, Method method) throws DataflowAnalysisException, CFGBuilderException {
492         JavaClass javaClass = classContext.getJavaClass();
493         ValueNumberDataflow vnd = classContext.getValueNumberDataflow(method);
494 
495         Set<ValueNumber> passthruParams = getPassthruParams(vnd, method, javaClass);
496 
497         this.method = method;
498         this.classContext = classContext;
499         MethodGen methodGen = classContext.getMethodGen(method);
500         if (methodGen == null) {
501             return;
502         }
503 
504         ConstantPoolGen cpg = methodGen.getConstantPool();
505         CFG cfg = classContext.getCFG(method);
506 
507         StringAppendState stringAppendState = getStringAppendState(cfg, cpg);
508 
509         ConstantDataflow dataflow = classContext.getConstantDataflow(method);
510         for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
511             Location location = i.next();
512             Instruction ins = location.getHandle().getInstruction();
513             if (!(ins instanceof InvokeInstruction)) {
514                 continue;
515             }
516             InvokeInstruction invoke = (InvokeInstruction) ins;
517             MethodDescriptor md = new MethodDescriptor(invoke, cpg);
518             boolean executeMethod;
519             int[] params = preparedStatementMethods.get(md);
520             int paramNumber;
521             // Currently only one method parameter is checked, though it's the most common case
522             // TODO: support methods which take several SQL statements
523             if(params != null) {
524                 executeMethod = false;
525                 paramNumber = params[0];
526             } else {
527                 params = executeMethods.get(md);
528                 if(params != null) {
529                     executeMethod = true;
530                     paramNumber = params[0];
531                 } else {
532                     continue;
533                 }
534             }
535             ConstantFrame frame = dataflow.getFactAtLocation(location);
536             SignatureParser parser = new SignatureParser(invoke.getSignature(cpg));
537             Constant value = frame.getArgument(invoke, cpg, paramNumber, parser);
538             ValueNumber vn = vnd.getFactAtLocation(location).getArgument(invoke, cpg, paramNumber, parser);
539 
540             if (!value.isConstantString() && !passthruParams.contains(vn)) {
541                 // TODO: verify it's the same string represented by
542                 // stringAppendState
543                 // FIXME: will false positive on const/static strings
544                 // returns by methods
545                 Location prev = getValueNumberCreationLocation(vnd, vn);
546                 if (prev == null || !isSafeValue(prev, cpg)) {
547                     BugInstance bug = generateBugInstance(javaClass, methodGen, location.getHandle(), stringAppendState,
548                             executeMethod);
549                     bugAccumulator.accumulateBug(
550                             bug,
551                             SourceLineAnnotation.fromVisitedInstruction(classContext, methodGen,
552                                     javaClass.getSourceFileName(), location.getHandle()));
553                 }
554             }
555         }
556         bugAccumulator.reportAccumulatedBugs();
557     }

559     private Location getValueNumberCreationLocation(ValueNumberDataflow vnd, ValueNumber vn) {
560         ConstantPoolGen cpg = vnd.getCFG().getMethodGen().getConstantPool();
561         for(Iterator<Location> it = vnd.getCFG().locationIterator(); it.hasNext(); ) {
562             Location loc = it.next();
563             if(loc.getHandle().getInstruction().produceStack(cpg) != 1) {
564                 continue;
565             }
566             try {
567                 ValueNumberFrame vnf = vnd.getFactAfterLocation(loc);
568                 if(vnf.getTopValue().equals(vn)) {
569                     return loc;
570                 }
571             } catch (DataflowAnalysisException e) {
572                 AnalysisContext.logError("While analyzing "+vnd.getCFG().getMethodGen()+" at "+loc, e);
573             }
574         }
575         return null;
576     }

578     private Set<ValueNumber> getPassthruParams(ValueNumberDataflow vnd, Method method, JavaClass javaClass) {
579         XMethod xMethod = XFactory.createXMethod(javaClass, method);
580         Set<ValueNumber> passthruParams = new HashSet<>();
581 
582         int[] p = preparedStatementMethods.get(xMethod);
583         if(p != null) {
584             for(int pNum : p) {
585                 passthruParams.add(vnd.getAnalysis().getEntryValueForParameter(pNum));
586             }
587         }
588         p = executeMethods.get(xMethod);
589         if(p != null) {
590             for(int pNum : p) {
591                 passthruParams.add(vnd.getAnalysis().getEntryValueForParameter(pNum));
592             }
593         }
594         return passthruParams;
595     }

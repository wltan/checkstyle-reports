19  package edu.umd.cs.findbugs.detect;
20  
21  import edu.umd.cs.findbugs.props.AbstractWarningProperty;
22  import edu.umd.cs.findbugs.props.PriorityAdjustment;
23  
24  /**
25   * Warning properties for FindRefComparison detector.
26   *
27   * @author David Hovemeyer
28   */
29  public class RefComparisonWarningProperty extends AbstractWarningProperty {
30      private RefComparisonWarningProperty(String name, PriorityAdjustment priorityAdjustment) {
31          super(name, priorityAdjustment);
32      }
33  
34      /** There is a call to equals() in the method. */
35      public static final RefComparisonWarningProperty SAW_CALL_TO_EQUALS = new RefComparisonWarningProperty("SAW_CALL_TO_EQUALS",
36              PriorityAdjustment.AT_MOST_LOW);
37  
38      /** Method is private (or package-protected). */
39      public static final RefComparisonWarningProperty PRIVATE_METHOD = new RefComparisonWarningProperty("PRIVATE_METHOD",
40              PriorityAdjustment.AT_MOST_MEDIUM);
41  
42      /** Compare inside test case */
43      public static final RefComparisonWarningProperty COMPARE_IN_TEST_CASE = new RefComparisonWarningProperty(
44              "COMPARE_IN_TEST_CASE", PriorityAdjustment.AT_MOST_LOW);
45  
46      /** Comparing static strings using equals operator. */
47      public static final RefComparisonWarningProperty COMPARE_STATIC_STRINGS = new RefComparisonWarningProperty(
48              "COMPARE_STATIC_STRINGS", PriorityAdjustment.FALSE_POSITIVE);
49  
50      /** Comparing a dynamic string using equals operator. */
51      public static final RefComparisonWarningProperty DYNAMIC_AND_UNKNOWN = new RefComparisonWarningProperty(
52              "DYNAMIC_AND_UNKNOWN", PriorityAdjustment.RAISE_PRIORITY);
53  
54      public static final RefComparisonWarningProperty STRING_PARAMETER_IN_PUBLIC_METHOD = new RefComparisonWarningProperty(
55              "STATIC_AND_PARAMETER_IN_PUBLIC_METHOD", PriorityAdjustment.RAISE_PRIORITY);
56  
57      public static final RefComparisonWarningProperty STRING_PARAMETER = new RefComparisonWarningProperty("STATIC_AND_PARAMETER",
58              PriorityAdjustment.NO_ADJUSTMENT);
59  
60      /** Comparing static string and an unknown string. */
61      public static final RefComparisonWarningProperty STATIC_AND_UNKNOWN = new RefComparisonWarningProperty("STATIC_AND_UNKNOWN",
62              PriorityAdjustment.LOWER_PRIORITY);
63      /** Comparing static string and an unknown string. */
64      public static final RefComparisonWarningProperty EMPTY_AND_UNKNOWN = new RefComparisonWarningProperty("EMPTY_AND_UNKNOWN",
65              PriorityAdjustment.NO_ADJUSTMENT);
66  
67      /** Saw a call to String.intern(). */
68      public static final RefComparisonWarningProperty SAW_INTERN = new RefComparisonWarningProperty("SAW_INTERN",
69              PriorityAdjustment.LOWER_PRIORITY);
70  }

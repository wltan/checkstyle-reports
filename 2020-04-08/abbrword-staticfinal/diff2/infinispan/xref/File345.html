31  @Test(groups = "functional", testName = "jmx.ComponentsJmxRegistrationTest")

33     public static final String JMX_DOMAIN = ComponentsJmxRegistrationTest.class.getSimpleName();
34     private MBeanServer mBeanServer;
35     private List<EmbeddedCacheManager> cacheContainers = new ArrayList<EmbeddedCacheManager>();
36  
37     @BeforeMethod
38     public void setUp() {
39        mBeanServer = MBeanServerFactory.createMBeanServer();
40        cacheContainers.clear();
41     }
42  
43     @AfterMethod
44     public void tearDown() {
45        MBeanServerFactory.releaseMBeanServer(mBeanServer);
46        TestingUtil.killCacheManagers(cacheContainers);
47        cacheContainers.clear();
48     }

50     public void testStopStartCM() throws Exception {
51        EmbeddedCacheManager cm = TestCacheManagerFactory.createCacheManager(false);
52        cacheContainers.add(cm);
53        cm.stop();
54        cm.start();
55     }
56  
57     public void testRegisterLocalCache() throws Exception {
58        EmbeddedCacheManager cm = TestCacheManagerFactory.createCacheManager(false);
59        cacheContainers.add(cm);
60        cm.start();
61        ConfigurationBuilder configuration = config();
62        configuration.clustering().cacheMode(CacheMode.LOCAL);
63        cm.defineConfiguration("first", configuration.build());
64        Cache first = cm.getCache("first");
65  
66        ComponentsJmxRegistration regComponents = buildRegistrator(first);
67        regComponents.registerMBeans();
68        String name = regComponents.getObjectName("Statistics").toString();
69        ObjectName name1 = new ObjectName(name);
70        assert mBeanServer.isRegistered(name1);
71        regComponents.unregisterMBeans();
72        assert !mBeanServer.isRegistered(name1);
73        assertCorrectJmxName(name1, first);
74     }
75  
76     private ComponentsJmxRegistration buildRegistrator(Cache cache) {
77        AdvancedCache ac = (AdvancedCache) cache;
78        Set<AbstractComponentRegistry.Component> components = ac.getComponentRegistry().getRegisteredComponents();
79        String groupName = "name=" + ObjectName.quote(cache.getName());
80        ComponentsJmxRegistration registrator = new ComponentsJmxRegistration(mBeanServer, components, groupName);
81        registrator.setJmxDomain(JMX_DOMAIN);
82        return registrator;
83     }
84  
85     public void testRegisterReplicatedCache() throws Exception {
86        GlobalConfigurationBuilder globalConfiguration = GlobalConfigurationBuilder.defaultClusteredBuilder();
87        globalConfiguration.globalJmxStatistics().enable().allowDuplicateDomains(true);
88        EmbeddedCacheManager cm = TestCacheManagerFactory.createClusteredCacheManager(globalConfiguration, new ConfigurationBuilder());
89        cacheContainers.add(cm);
90        cm.start();
91        ConfigurationBuilder configurationOverride = config();
92        configurationOverride.clustering().cacheMode(CacheMode.REPL_SYNC);
93        cm.defineConfiguration("first", configurationOverride.build());
94        Cache first = cm.getCache("first");
95  
96        ComponentsJmxRegistration regComponents = buildRegistrator(first);
97        regComponents.registerMBeans();
98        String name = regComponents.getObjectName("Statistics").toString();
99        ObjectName name1 = new ObjectName(name);
100       assertCorrectJmxName(name1, first);
101       assert mBeanServer.isRegistered(name1);
102       regComponents.unregisterMBeans();
103       assert !mBeanServer.isRegistered(name1);
104    }

106    public void testLocalAndReplicatedCache() throws Exception {
107       GlobalConfigurationBuilder globalConfiguration = GlobalConfigurationBuilder.defaultClusteredBuilder();
108       globalConfiguration.globalJmxStatistics().enable().allowDuplicateDomains(true);
109       EmbeddedCacheManager cm = TestCacheManagerFactory.createClusteredCacheManager(globalConfiguration, new ConfigurationBuilder());
110       cacheContainers.add(cm);
111       cm.start();
112       ConfigurationBuilder replicated = config();
113       ConfigurationBuilder local = config();
114       replicated.clustering().cacheMode(CacheMode.REPL_SYNC);
115       local.clustering().cacheMode(CacheMode.LOCAL);
116       cm.defineConfiguration("replicated", replicated.build());
117       cm.defineConfiguration("local", local.build());
118       Cache replicatedCache = cm.getCache("replicated");
119       Cache localCache = cm.getCache("local");
120 
121       ComponentsJmxRegistration replicatedRegComponents = buildRegistrator(replicatedCache);
122       ComponentsJmxRegistration localRegComponents = buildRegistrator(localCache);
123       replicatedRegComponents.registerMBeans();
124       localRegComponents.registerMBeans();
125 
126       String replicatedtCMgmtIntName = replicatedRegComponents.getObjectName("Statistics").toString();
127       String localCMgmtIntName = localRegComponents.getObjectName("Statistics").toString();
128       ObjectName replObjectName = new ObjectName(replicatedtCMgmtIntName);
129       ObjectName localObjName = new ObjectName(localCMgmtIntName);
130       assertCorrectJmxName(replObjectName, replicatedCache);
131 
132       assert mBeanServer.isRegistered(replObjectName);
133       assert mBeanServer.isRegistered(localObjName);
134       assert !localCMgmtIntName.equals(replicatedtCMgmtIntName);
135 
136       replicatedRegComponents.unregisterMBeans();
137       localRegComponents.unregisterMBeans();
138       assert !mBeanServer.isRegistered(new ObjectName(localCMgmtIntName));
139       assert !mBeanServer.isRegistered(new ObjectName(replicatedtCMgmtIntName));
140    }
141 
142    private void assertCorrectJmxName(ObjectName objectName, Cache cache) {
143       assert ObjectName.unquote(objectName.getKeyProperty(ComponentsJmxRegistration.NAME_KEY)).startsWith(cache.getName());
144       assert objectName.getKeyProperty(ComponentsJmxRegistration.COMPONENT_KEY) != null;
145    }
146 
147    private ConfigurationBuilder config() {
148       ConfigurationBuilder configuration = new ConfigurationBuilder();
149       configuration.clustering().stateTransfer().fetchInMemoryState(false).jmxStatistics().enable();
150       return configuration;
151    }

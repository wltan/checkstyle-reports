18  
19  import static org.infinispan.cdi.test.testutil.Deployments.baseDeployment;
20  import static org.infinispan.client.hotrod.test.HotRodClientTestingUtil.startHotRodServer;
21  import static org.infinispan.server.hotrod.test.HotRodTestingUtil.hotRodCacheConfiguration;
22  import static org.testng.Assert.assertEquals;

24  /**
25   * Tests that the use of a specific cache manager for one cache.
26   *
27   * @author Kevin Pollet <kevin.pollet@serli.com> (C) 2011 SERLI
28   */
29  @Test(groups = "functional", testName = "cdi.test.cache.remote.SpecificCacheManagerTest")
30  public class SpecificCacheManagerTest extends Arquillian {
31  
32     private static final String SERVER_LIST_KEY = "infinispan.client.hotrod.server_list";
33  
34     @Deployment
35     public static Archive<?> deployment() {
36        return baseDeployment()
37              .addClass(SpecificCacheManagerTest.class)
38              .addClass(Small.class);
39     }
40  
41     private static HotRodServer hotRodServer;
42     private static EmbeddedCacheManager embeddedCacheManager;
43  
44     @Inject
45     @Small
46     private RemoteCache<String, String> cache;
47  
48     @BeforeTest
49     public void beforeMethod() {
50        embeddedCacheManager = TestCacheManagerFactory.createCacheManager(
51              hotRodCacheConfiguration(TestCacheManagerFactory
52                    .getDefaultCacheConfiguration(false)));
53        embeddedCacheManager.defineConfiguration("small", embeddedCacheManager.getDefaultCacheConfiguration());
54        hotRodServer = startHotRodServer(embeddedCacheManager);
55     }
56  
57     @AfterTest(alwaysRun = true)
58     public void afterMethod() {
59        if (embeddedCacheManager != null) embeddedCacheManager.stop();
60        if (hotRodServer != null) hotRodServer.stop();
61     }
62  
63     public void testSpecificCacheManager() {
64        cache.put("pete", "British");
65        cache.put("manik", "Sri Lankan");
66  
67        assertEquals(cache.getName(), "small");
68        assertEquals(cache.get("pete"), "British");
69        assertEquals(cache.get("manik"), "Sri Lankan");
70     }
71  
72     /**
73      * Produces a specific cache manager for the small cache.
74      *
75      * @see Small
76      */
77     @Small
78     @Produces
79     @ApplicationScoped
80     public static RemoteCacheManager smallRemoteCacheManager() {
81        return new RemoteCacheManager(
82              new org.infinispan.client.hotrod.configuration.ConfigurationBuilder()
83                    .addServers("127.0.0.1:" + hotRodServer.getPort()).build());
84     }

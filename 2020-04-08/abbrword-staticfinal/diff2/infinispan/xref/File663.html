1   package org.jboss.as.clustering.infinispan.subsystem;
2   
3   import org.infinispan.configuration.cache.XSiteStateTransferConfigurationBuilder;
4   import org.jboss.as.clustering.infinispan.subsystem.CacheConfigOperationHandlers.CacheConfigAdd;
5   import org.jboss.as.controller.AttributeDefinition;
6   import org.jboss.as.controller.OperationStepHandler;
7   import org.jboss.as.controller.PathElement;
8   import org.jboss.as.controller.ReloadRequiredRemoveStepHandler;
9   import org.jboss.as.controller.ReloadRequiredWriteAttributeHandler;
10  import org.jboss.as.controller.SimpleAttributeDefinition;
11  import org.jboss.as.controller.SimpleAttributeDefinitionBuilder;
12  import org.jboss.as.controller.SimpleResourceDefinition;
13  import org.jboss.as.controller.registry.AttributeAccess;
14  import org.jboss.as.controller.registry.ManagementResourceRegistration;
15  import org.jboss.dmr.ModelNode;
16  import org.jboss.dmr.ModelType;
17  
18  /**
19   * Cross-Site state transfer resource.
20   *
21   * @author Pedro Ruivo
22   * @since 7.0
23   */
24  public class BackupSiteStateTransferResource extends SimpleResourceDefinition {
25  
26     static final SimpleAttributeDefinition STATE_TRANSFER_CHUNK_SIZE = new SimpleAttributeDefinitionBuilder(ModelKeys.CHUNK_SIZE, ModelType.INT, true)
27           .setXmlName(Attribute.CHUNK_SIZE.getLocalName())
28           .setAllowExpression(true)
29           .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
30           .setDefaultValue(new ModelNode().set(XSiteStateTransferConfigurationBuilder.DEFAULT_CHUNK_SIZE))
31           .build();
32     static final SimpleAttributeDefinition STATE_TRANSFER_TIMEOUT = new SimpleAttributeDefinitionBuilder(ModelKeys.TIMEOUT, ModelType.LONG, true)
33           .setXmlName(Attribute.TIMEOUT.getLocalName())
34           .setAllowExpression(true)
35           .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
36           .setDefaultValue(new ModelNode().set(XSiteStateTransferConfigurationBuilder.DEFAULT_TIMEOUT))
37           .build();
38     static final SimpleAttributeDefinition STATE_TRANSFER_WAIT_TIME = new SimpleAttributeDefinitionBuilder(ModelKeys.WAIT_TIME, ModelType.LONG, true)
39           .setXmlName(Attribute.WAIT_TIME.getLocalName())
40           .setAllowExpression(true)
41           .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
42           .setDefaultValue(new ModelNode().set(XSiteStateTransferConfigurationBuilder.DEFAULT_WAIT_TIME))
43           .build();
44     static final SimpleAttributeDefinition STATE_TRANSFER_MAX_RETRIES = new SimpleAttributeDefinitionBuilder(ModelKeys.MAX_RETRIES, ModelType.INT, true)
45           .setXmlName(Attribute.MAX_RETRIES.getLocalName())
46           .setAllowExpression(true)
47           .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
48           .setDefaultValue(new ModelNode().set(XSiteStateTransferConfigurationBuilder.DEFAULT_MAX_RETRIES))
49           .build();
50  
51     static final AttributeDefinition[] ATTRIBUTES = new AttributeDefinition[]{STATE_TRANSFER_CHUNK_SIZE, STATE_TRANSFER_MAX_RETRIES, STATE_TRANSFER_TIMEOUT, STATE_TRANSFER_WAIT_TIME};
52  
53     private final boolean runtimeRegistration;
54  
55     BackupSiteStateTransferResource(boolean runtimeRegistration) {
56        super(PathElement.pathElement(ModelKeys.STATE_TRANSFER, ModelKeys.STATE_TRANSFER_NAME), InfinispanExtension.getResourceDescriptionResolver(ModelKeys.BACKUP, ModelKeys.STATE_TRANSFER), new CacheConfigAdd(ATTRIBUTES), ReloadRequiredRemoveStepHandler.INSTANCE);
57        this.runtimeRegistration = runtimeRegistration;
58     }
59  
60     @Override
61     public void registerAttributes(ManagementResourceRegistration registration) {
62        final OperationStepHandler writeHandler = new ReloadRequiredWriteAttributeHandler(ATTRIBUTES);
63        for (AttributeDefinition attribute : ATTRIBUTES) {
64           registration.registerReadWriteAttribute(attribute, null, writeHandler);
65        }
66     }
67  
68     @Override
69     public void registerOperations(ManagementResourceRegistration resourceRegistration) {
70        super.registerOperations(resourceRegistration);
71     }
72  
73     public boolean isRuntimeRegistration() {
74        return runtimeRegistration;
75     }
76  
77     @Override
78     public void registerChildren(ManagementResourceRegistration resourceRegistration) {
79        super.registerChildren(resourceRegistration);
80     }
81  }

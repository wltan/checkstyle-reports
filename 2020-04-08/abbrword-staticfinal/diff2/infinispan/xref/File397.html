1   package org.infinispan.statetransfer;
2   
3   import org.infinispan.Cache;
4   import org.infinispan.persistence.spi.CacheLoader;
5   import org.infinispan.test.TestingUtil;
6   
7   /**
8    * StateTransferTestingUtil.
9    * 
10   * @author Galder Zamarre√±o
11   * @since 4.0
12   */
13  public class StateTransferTestingUtil {
14     public static final String A_B_NAME = "a_b_name";
15     public static final String A_C_NAME = "a_c_name";
16     public static final String A_D_NAME = "a_d_age";
17     public static final String A_B_AGE = "a_b_age";
18     public static final String A_C_AGE = "a_c_age";
19     public static final String A_D_AGE = "a_d_age";
20     public static final String JOE = "JOE";
21     public static final String BOB = "BOB";
22     public static final String JANE = "JANE";
23     public static final Integer TWENTY = 20;
24     public static final Integer FORTY = 40;
25  
26     public static void verifyNoDataOnLoader(Cache<Object, Object> c) throws Exception {
27        CacheLoader l = TestingUtil.getFirstLoader(c);
28        assert !l.contains(A_B_AGE);
29        assert !l.contains(A_B_NAME);
30        assert !l.contains(A_C_AGE);
31        assert !l.contains(A_C_NAME);
32        assert !l.contains(A_D_AGE);
33        assert !l.contains(A_D_NAME);
34     }
35  
36     public static void verifyNoData(Cache<Object, Object> c) {
37        assert c.isEmpty() : "Cache should be empty!";
38     }
39  
40     public static void writeInitialData(final Cache<Object, Object> c) {
41        c.put(A_B_NAME, JOE);
42        c.put(A_B_AGE, TWENTY);
43        c.put(A_C_NAME, BOB);
44        c.put(A_C_AGE, FORTY);
45        c.evict(A_B_NAME);
46        c.evict(A_B_AGE);
47        c.evict(A_C_NAME);
48        c.evict(A_C_AGE);
49        c.evict(A_D_NAME);
50        c.evict(A_D_AGE);
51     }
52  
53     public static void verifyInitialDataOnLoader(Cache<Object, Object> c) throws Exception {
54        CacheLoader l = TestingUtil.getFirstLoader(c);
55        assert l.contains(A_B_AGE);
56        assert l.contains(A_B_NAME);
57        assert l.contains(A_C_AGE);
58        assert l.contains(A_C_NAME);
59        assert l.load(A_B_AGE).getValue().equals(TWENTY);
60        assert l.load(A_B_NAME).getValue().equals(JOE);
61        assert l.load(A_C_AGE).getValue().equals(FORTY);
62        assert l.load(A_C_NAME).getValue().equals(BOB);
63     }
64  
65     public static void verifyInitialData(Cache<Object, Object> c) {
66        assert JOE.equals(c.get(A_B_NAME)) : "Incorrect value for key " + A_B_NAME;
67        assert TWENTY.equals(c.get(A_B_AGE)) : "Incorrect value for key " + A_B_AGE;
68        assert BOB.equals(c.get(A_C_NAME)) : "Incorrect value for key " + A_C_NAME;
69        assert FORTY.equals(c.get(A_C_AGE)) : "Incorrect value for key " + A_C_AGE;
70     }
71  }

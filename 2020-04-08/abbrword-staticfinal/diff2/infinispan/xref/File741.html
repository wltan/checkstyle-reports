22  package org.jboss.as.clustering.jgroups.subsystem;
23  
24  import static org.jboss.as.clustering.jgroups.JGroupsLogger.ROOT_LOGGER;
25  
26  import org.jboss.as.clustering.jgroups.ChannelFactory;
27  import org.jboss.as.clustering.jgroups.JGroupsMessages;
28  import org.jboss.msc.service.Service;
29  import org.jboss.msc.service.ServiceName;
30  import org.jboss.msc.service.StartContext;
31  import org.jboss.msc.service.StartException;
32  import org.jboss.msc.service.StopContext;
33  import org.jboss.msc.value.Value;
34  import org.jgroups.Address;
35  import org.jgroups.Channel;
36  import org.jgroups.ChannelListener;
37  
38  /**
39   * Provides a channel for use by dependent services.
40   * Channels produced by this service are unconnected.
41   * @author Paul Ferraro
42   */
43  public class ChannelService implements Service<Channel>, ChannelListener {
44  
45      private static final ServiceName SERVICE_NAME = ServiceName.JBOSS.append(JGroupsExtension.SUBSYSTEM_NAME).append("channel");
46  
47      public static ServiceName getServiceName(String id) {
48          return SERVICE_NAME.append(id);
49      }
50  
51      private final Value<ChannelFactory> factory;
52      private final String id;
53      private volatile Channel channel;
54  
55      public ChannelService(String id, Value<ChannelFactory> factory) {
56          this.id = id;
57          this.factory = factory;
58      }
59  
60      @Override
61      public Channel getValue() {
62          return this.channel;
63      }
64  
65      @Override
66      public void start(StartContext context) throws StartException {
67          ChannelFactory factory = this.factory.getValue();
68          try {
69              this.channel = factory.createChannel(this.id);
70              this.channel.addChannelListener(this);
71              // Don't connect the channel here
72              // This will be done by Infinispan (see AS7-5904)
73          } catch (Exception e) {
74              throw new StartException(e);
75          }
76  
77          if  (ROOT_LOGGER.isTraceEnabled())  {
78              String output = this.channel.getProtocolStack().printProtocolSpec(true);
79              ROOT_LOGGER.tracef("JGroups channel named %s created with configuration:\n %s", this.id, output);
80          }
81      }
82  
83      @Override
84      public void stop(StopContext context) {
85          if (this.channel != null) {
86              this.channel.removeChannelListener(this);
87          }
88          this.channel = null;
89      }
90  
91      @Override
92      public void channelClosed(Channel channel) {
93          // Do nothing
94      }
95  
96      @Override
97      public void channelConnected(Channel channel) {
98          // Validate view
99          String localName = channel.getName();
100         Address localAddress = channel.getAddress();
101         for (Address address: channel.getView()) {
102             String name = channel.getName(address);
103             if ((name != null) && name.equals(localName) && !address.equals(localAddress)) {
104                 channel.close();
105                 throw JGroupsMessages.MESSAGES.duplicateNodeName(this.factory.getValue().getProtocolStackConfiguration().getEnvironment().getNodeName());
106             }
107         }
108     }
109 
110     @Override
111     public void channelDisconnected(Channel channel) {
112         // Do nothing
113     }
114 }

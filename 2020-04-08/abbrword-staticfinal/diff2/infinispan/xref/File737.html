23  package org.jboss.as.clustering.jgroups;
24  
25  import static org.jboss.logging.Logger.Level.ERROR;
26  import static org.jboss.logging.Logger.Level.INFO;
27  import static org.jboss.logging.Logger.Level.TRACE;
28  import static org.jboss.logging.Logger.Level.WARN;
29  
30  import org.jboss.logging.BasicLogger;
31  import org.jboss.logging.annotations.Cause;
32  import org.jboss.logging.annotations.LogMessage;
33  import org.jboss.logging.Logger;
34  import org.jboss.logging.annotations.Message;
35  import org.jboss.logging.annotations.MessageLogger;
36  
37  /**
38   * Date: 29.08.2011
39   *
40   * @author <a href="mailto:jperkins@redhat.com">James R. Perkins</a>
41   */
42  @MessageLogger(projectCode = "JBAS")
43  public interface JGroupsLogger extends BasicLogger {
44      String ROOT_LOGGER_CATEGORY = JGroupsLogger.class.getPackage().getName();
45  
46      /**
47       * The root logger.
48       */
49      JGroupsLogger ROOT_LOGGER = Logger.getMessageLogger(JGroupsLogger.class, ROOT_LOGGER_CATEGORY);
50  
51      /**
52       * Logs an informational message indicating the JGroups subsystem is being activated.
53       */
54      @LogMessage(level = INFO)
55      @Message(id = 10260, value = "Activating JGroups subsystem.")
56      void activatingSubsystem();
57  
58      @LogMessage(level = TRACE)
59      @Message(id = 10261, value = "Setting %s.%s=%d")
60      void setProtocolPropertyValue(String protocol, String property, Object value);
61  
62      @LogMessage(level = TRACE)
63      @Message(id = 10262, value = "Failed to set non-existent %s.%s=%d")
64      void nonExistentProtocolPropertyValue(@Cause Throwable cause, String protocolName, String propertyName, Object propertyValue);
65  
66      @LogMessage(level = TRACE)
67      @Message(id = 10263, value = "Could not set %s.%s and %s.%s, %s socket binding does not specify a multicast socket")
68      void couldNotSetAddressAndPortNoMulticastSocket(@Cause Throwable cause, String protocolName, String addressProperty, String protocolNameAgain, String portProperty, String bindingName);
69  
70      @LogMessage(level = ERROR)
71      @Message(id = 10264, value = "Error accessing original value for property %s of protocol %s")
72      void unableToAccessProtocolPropertyValue(@Cause Throwable cause, String propertyName, String protocolName);
73  
74      @LogMessage(level = WARN)
75      @Message(id = 10265, value = "property %s for protocol %s attempting to override socket binding value %s : property value %s will be ignored")
76      void unableToOverrideSocketBindingValue(String propertyName, String protocolName, String bindingName, Object propertyValue);
77  
78  }

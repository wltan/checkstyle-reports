1   package org.infinispan.configuration.cache;
2   
3   import java.util.concurrent.TimeUnit;
4   
5   import org.infinispan.commons.configuration.BuiltBy;
6   import org.infinispan.commons.configuration.ConfigurationFor;
7   import org.infinispan.commons.configuration.attributes.Attribute;
8   import org.infinispan.commons.configuration.attributes.AttributeDefinition;
9   import org.infinispan.commons.configuration.attributes.AttributeSet;
10  import org.infinispan.persistence.cluster.ClusterLoader;
11  
12  /**
13   * ClusterLoaderConfiguration.
14   *
15   * @author Tristan Tarrant
16   * @since 5.2
17   */
18  @BuiltBy(ClusterLoaderConfigurationBuilder.class)
19  @ConfigurationFor(ClusterLoader.class)
20  public class ClusterLoaderConfiguration extends AbstractStoreConfiguration {
21     public static final AttributeDefinition<Long> REMOTE_CALL_TIMEOUT = AttributeDefinition.builder("remoteCallTimeout", TimeUnit.SECONDS.toMillis(15)).immutable().build();
22  
23     public static final AttributeSet attributeDefinitionSet() {
24        return new AttributeSet(ClusterLoaderConfiguration.class, AbstractStoreConfiguration.attributeDefinitionSet(), REMOTE_CALL_TIMEOUT);
25     }
26  
27     private final Attribute<Long> remoteCallTimeout;
28  
29     public ClusterLoaderConfiguration(AttributeSet attributes, AsyncStoreConfiguration async,
30           SingletonStoreConfiguration singletonStore) {
31        super(attributes, async, singletonStore);
32        remoteCallTimeout = attributes.attribute(REMOTE_CALL_TIMEOUT);
33     }
34  
35     public long remoteCallTimeout() {
36        return remoteCallTimeout.get();
37     }
38  
39     @Override
40     public String toString() {
41        return "ClusterLoaderConfiguration [attributes=" + attributes + "]";
42     }
43  
44  }

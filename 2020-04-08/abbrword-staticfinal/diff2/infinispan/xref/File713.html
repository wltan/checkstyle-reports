23  package org.jboss.as.clustering.infinispan.subsystem;
24  
25  import static org.jboss.as.controller.descriptions.ModelDescriptionConstants.ADD;
26  
27  import org.jboss.as.controller.AttributeDefinition;
28  import org.jboss.as.controller.OperationDefinition;
29  import org.jboss.as.controller.OperationStepHandler;
30  import org.jboss.as.controller.PathElement;
31  import org.jboss.as.controller.SimpleAttributeDefinition;
32  import org.jboss.as.controller.SimpleAttributeDefinitionBuilder;
33  import org.jboss.as.controller.SimpleOperationDefinitionBuilder;
34  import org.jboss.as.controller.registry.ManagementResourceRegistration;
35  import org.jboss.as.controller.registry.OperationEntry;
36  import org.jboss.dmr.ModelNode;
37  
38  /**
39   * Resource description for the addressable resource
40   *
41   * /subsystem=infinispan/cache-container=X/cache=Y/mixed-keyed-jdbc-store=MIXED_KEYED_JDBC_STORE
42   *
43   * @author Richard Achmatowicz (c) 2011 Red Hat Inc.
44   * @author Tristan Tarrant
45   */
46  public class MixedKeyedJDBCStoreResource extends BaseJDBCStoreResource {
47  
48      public static final PathElement MIXED_KEYED_JDBC_STORE_PATH = PathElement.pathElement(ModelKeys.MIXED_KEYED_JDBC_STORE);
49  
50      // attributes
51      static final AttributeDefinition[] MIXED_KEYED_JDBC_STORE_ATTRIBUTES = {STRING_KEYED_TABLE, BINARY_KEYED_TABLE};
52  
53     static final SimpleAttributeDefinition NAME =
54           new SimpleAttributeDefinitionBuilder(BaseStoreResource.NAME)
55                 .setDefaultValue(new ModelNode().set(ModelKeys.MIXED_KEYED_JDBC_STORE_NAME))
56                 .build();
57  
58      // operations
59      private static final OperationDefinition MIXED_KEYED_JDBC_STORE_ADD_DEFINITION = new SimpleOperationDefinitionBuilder(ADD, InfinispanExtension.getResourceDescriptionResolver(ModelKeys.MIXED_KEYED_JDBC_STORE))
60          .setParameters(COMMON_STORE_PARAMETERS)
61          .addParameter(DATA_SOURCE)
62          .addParameter(DIALECT)
63          .addParameter(STRING_KEYED_TABLE)
64          .addParameter(BINARY_KEYED_TABLE)
65          .setAttributeResolver(InfinispanExtension.getResourceDescriptionResolver(ModelKeys.JDBC_STORE))
66          .build();
67  
68  
69      public MixedKeyedJDBCStoreResource(CacheResource cacheResource) {
70          super(MIXED_KEYED_JDBC_STORE_PATH, ModelKeys.MIXED_KEYED_JDBC_STORE, cacheResource, MIXED_KEYED_JDBC_STORE_ATTRIBUTES);
71      }
72  
73      // override the add operation to provide a custom definition (for the optional PROPERTIES parameter to add())
74      @Override
75      protected void registerAddOperation(final ManagementResourceRegistration registration, final OperationStepHandler handler, OperationEntry.Flag... flags) {
76          registration.registerOperationHandler(MIXED_KEYED_JDBC_STORE_ADD_DEFINITION, handler);
77      }
78  
79  }

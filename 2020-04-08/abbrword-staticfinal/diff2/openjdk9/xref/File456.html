59  package jdk.internal.org.objectweb.asm;
60  
61  /**
62   * An edge in the control flow graph of a method body. See {@link Label Label}.
63   *
64   * @author Eric Bruneton
65   */
66  class Edge {
67  
68      /**
69       * Denotes a normal control flow graph edge.
70       */
71      static final int NORMAL = 0;
72  
73      /**
74       * Denotes a control flow graph edge corresponding to an exception handler.
75       * More precisely any {@link Edge} whose {@link #info} is strictly positive
76       * corresponds to an exception handler. The actual value of {@link #info} is
77       * the index, in the {@link ClassWriter} type table, of the exception that
78       * is catched.
79       */
80      static final int EXCEPTION = 0x7FFFFFFF;
81  
82      /**
83       * Information about this control flow graph edge. If
84       * {@link ClassWriter#COMPUTE_MAXS} is used this field is the (relative)
85       * stack size in the basic block from which this edge originates. This size
86       * is equal to the stack size at the "jump" instruction to which this edge
87       * corresponds, relatively to the stack size at the beginning of the
88       * originating basic block. If {@link ClassWriter#COMPUTE_FRAMES} is used,
89       * this field is the kind of this control flow graph edge (i.e. NORMAL or
90       * EXCEPTION).
91       */
92      int info;
93  
94      /**
95       * The successor block of the basic block from which this edge originates.
96       */
97      Label successor;
98  
99      /**
100      * The next edge in the list of successors of the originating basic block.
101      * See {@link Label#successors successors}.
102      */
103     Edge next;
104 }

24  package jdk.test;
25  
26  import java.net.URL;
27  
28  public class Main {
29      static final String JAVA_CLASS_PATH = "java.class.path";
30  
31      public static void main(String[] args) throws Exception {
32          boolean expected = args[0].equals("true");
33          String cpath = args.length > 1 ? args[1] : "";
34          String value = System.getProperty(JAVA_CLASS_PATH);
35          if (!value.equals(cpath)) {
36              throw new RuntimeException(JAVA_CLASS_PATH + "=" + value +
37                  " expected=" + cpath);
38          }
39  
40          ClassLoader loader = ClassLoader.getSystemClassLoader();
41          URL url = loader.getResource("jdk/test/res.properties");
42          if ((expected && url == null) || (!expected && url != null)) {
43              throw new RuntimeException("URL: " + url + " expected non-null: " + expected);
44          }
45      }
46  }

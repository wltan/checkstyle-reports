33  import java.security.*;
34  import javax.crypto.*;
35  import javax.crypto.spec.*;
36  import java.math.*;
37  
38  import java.util.*;
39  
40  public class TestKATForECB_IV
41  {
42      private static final String ALGO = "AES";
43      private static final String MODE = "ECB";
44      private static final String PADDING = "NoPadding";
45  
46      //ecb-iv.txt
47      private static String[] PTS = {
48          "000102030405060708090A0B0C0D0E0F",
49          "000102030405060708090A0B0C0D0E0F",
50          "000102030405060708090A0B0C0D0E0F"
51      };
52      private static String[] CTS = {
53          "0A940BB5416EF045F1C39458C653EA5A",
54          "0060BFFE46834BB8DA5CF9A61FF220AE",
55          "5A6E045708FB7196F02E553D02C3A692"
56      };
57      private static String[] KEYS = {
58          "000102030405060708090A0B0C0D0E0F",
59          "000102030405060708090A0B0C0D0E0F1011121314151617",
60          "000102030405060708090A0B0C0D0E0F101112131415161718191A1B1C1D1E1F"
61      };
62  
63      private static SecretKey constructAESKey(String s) throws Exception {
64          int len = s.length()/2;
65          if ((len != 16) && (len != 24) && (len != 32)) {
66              throw new IllegalArgumentException("Wrong Key Length: " + len);
67          }
68          byte[] rawKeyValue = constructByteArray(s);
69          SecretKeySpec key = new SecretKeySpec(rawKeyValue, "AES");
70          return key;
71      }
72  
73      private static byte[] constructByteArray(String s) {
74          int len = s.length()/2;
75          byte[] tempValue = new byte[len];
76          for (int i = 0; i < len; i++) {
77              tempValue[i] = Integer.valueOf(s.substring(2*i, 2*i+2),
78                                             16).byteValue();
79          }
80          return tempValue;
81  
82      }
83  
84      public boolean execute() throws Exception {
85          String transformation = ALGO+"/"+MODE+"/"+PADDING;
86          Cipher c = Cipher.getInstance(transformation, "SunJCE");
87          for (int i=0; i<KEYS.length; i++) {
88              SecretKey aesKey = constructAESKey(KEYS[i]);
89              if (aesKey.getEncoded().length*8 >
90                  Cipher.getMaxAllowedKeyLength(transformation)) {
91                  // skip if this key length is larger than what's
92                  // configured in the jce jurisdiction policy files
93                  continue;
94              }
95              try {
96              c.init(Cipher.ENCRYPT_MODE, aesKey);
97              byte[] plainText = constructByteArray(PTS[i]);
98              byte[] cipherText = c.doFinal(plainText);
99              byte[] answer = constructByteArray(CTS[i]);
100             if (!Arrays.equals(cipherText, answer)) {
101                 throw new Exception((i+1) + "th known answer test failed for encryption");
102             }
103 
104             c.init(Cipher.DECRYPT_MODE, aesKey);
105             byte[] recoveredText = c.doFinal(answer);
106             if (!Arrays.equals(recoveredText, plainText)) {
107                 throw new Exception("known answer test failed for decryption");
108             }
109             System.out.println("Finished KAT for " + aesKey.getEncoded().length + "-byte key");
110             } catch (SecurityException se) {
111                 TestUtil.handleSE(se);
112             }
113         }
114 
115         // passed all tests...hooray!
116         return true;
117     }
118 
119     public static void main (String[] args) throws Exception {
120         TestKATForECB_IV test = new TestKATForECB_IV();
121         String testName = test.getClass().getName() + "[" + ALGO +
122             "/" + MODE + "/" + PADDING + "]";
123         if (test.execute()) {
124             System.out.println(testName + ": Passed!");
125         }
126     }

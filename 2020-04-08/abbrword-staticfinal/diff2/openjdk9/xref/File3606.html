24  /*
25   * @test
26   * @bug 6355295
27   * @summary Certificate validation using OCSP fails for a particular class of certificates
28   * @modules java.base/sun.security.provider.certpath
29   *          java.base/sun.security.x509
30   */
31  
32  import java.io.*;
33  import java.security.MessageDigest;
34  import java.util.Arrays;
35  import sun.security.provider.certpath.CertId;
36  import sun.security.x509.X509CertImpl;
37  
38  /*
39   * Checks that the hash value for a certificate's issuer name is generated
40   * correctly. Requires any certificate that is not self-signed.
41   *
42   * NOTE: this test uses Sun private classes which are subject to change.
43   */
44  public class CheckCertId {
45  
46      private static final String CERT_FILENAME = "interCA.der";
47  
48      public static void main(String[] args) throws Exception {
49  
50          X509CertImpl cert = loadCert(CERT_FILENAME);
51  
52          /* Compute the hash in the same way as CertId constructor */
53          MessageDigest hash = MessageDigest.getInstance("SHA1");
54          hash.update(cert.getSubjectX500Principal().getEncoded());
55          byte[] expectedHash = hash.digest();
56  
57          CertId certId = new CertId(cert, null);
58          byte[] receivedHash = certId.getIssuerNameHash();
59  
60          if (! Arrays.equals(expectedHash, receivedHash)) {
61              throw new
62                  Exception("Bad hash value for issuer name in CertId object");
63          }
64      }
65  
66      /*
67       * Load an X.509 certificate from a file.
68       * Return it as a Sun private class.
69       */
70      private static X509CertImpl loadCert(String filename) throws Exception {
71  
72          BufferedInputStream bis =
73              new BufferedInputStream(
74                  new FileInputStream(
75                      new File(System.getProperty("test.src", "."), filename)));
76  
77          return new X509CertImpl(bis);
78      }
79  }

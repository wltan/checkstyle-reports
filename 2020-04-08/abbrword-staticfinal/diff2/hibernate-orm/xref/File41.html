431 	@SuppressWarnings({"unchecked", "ConstantConditions"})
432 	private Class<SessionEventListener> resolveAutoSessionEventListenerClass(
433 			Properties properties,
434 			ClassLoaderService classLoaderService) {
435 		final Object setting = properties.getProperty( AvailableSettings.AUTO_SESSION_EVENTS_LISTENER );
436 		if ( setting == null ) {
437 			return null;
438 		}
439 
440 		if ( Class.class.isInstance( setting ) ) {
441 			return (Class<SessionEventListener>) setting;
442 		}
443 		else {
444 			final String settingStr = setting.toString();
445 			return classLoaderService.classForName( settingStr );
446 		}
447 	}

449 	private MultiTableBulkIdStrategy getMultiTableBulkIdStrategy(
450 			Properties properties,
451 			Dialect dialect,
452 			ClassLoaderService classLoaderService) {
453 		final Object setting = properties.get( AvailableSettings.HQL_BULK_ID_STRATEGY );
454 		if ( setting != null ) {
455 			if ( MultiTableBulkIdStrategy.class.isInstance( setting ) ) {
456 				return (MultiTableBulkIdStrategy) setting;
457 			}
458 			final Class strategyClass;
459 			if ( Class.class.isInstance( setting ) ) {
460 				strategyClass = (Class) setting;
461 			}
462 			else {
463 				final String settingStr = setting.toString();
464 				if ( PersistentTableBulkIdStrategy.SHORT_NAME.equals( settingStr ) ) {
465 					return new PersistentTableBulkIdStrategy();
466 				}
467 				else if ( TemporaryTableBulkIdStrategy.SHORT_NAME.equals( settingStr ) ) {
468 					return TemporaryTableBulkIdStrategy.INSTANCE;
469 				}
470 				strategyClass = classLoaderService.classForName( settingStr );
471 			}
472 			try {
473 				return (MultiTableBulkIdStrategy) strategyClass.newInstance();
474 			}
475 			catch (Exception e) {
476 				throw new HibernateException( "Unable to interpret MultiTableBulkIdStrategy setting [" + setting + "]", e );
477 			}
478 		}
479 		else {
480 			return dialect.supportsTemporaryTables()
481 					? TemporaryTableBulkIdStrategy.INSTANCE
482 					: new PersistentTableBulkIdStrategy();
483 		}
484 	}

500 	protected QueryCacheFactory createQueryCacheFactory(Properties properties, ServiceRegistry serviceRegistry) {
501 		String queryCacheFactoryClassName = ConfigurationHelper.getString(
502 				AvailableSettings.QUERY_CACHE_FACTORY, properties, StandardQueryCacheFactory.class.getName()
503 		);
504 		LOG.debugf( "Query cache factory: %s", queryCacheFactoryClassName );
505 		try {
506 			return (QueryCacheFactory) serviceRegistry.getService( ClassLoaderService.class )
507 					.classForName( queryCacheFactoryClassName )
508 					.newInstance();
509 		}
510 		catch (Exception e) {
511 			throw new HibernateException( "could not instantiate QueryCacheFactory: " + queryCacheFactoryClassName, e );
512 		}
513 	}

515 	private static RegionFactory createRegionFactory(Properties properties, boolean cachingEnabled, ServiceRegistry serviceRegistry) {
516 		String regionFactoryClassName = RegionFactoryInitiator.mapLegacyNames(
517 				ConfigurationHelper.getString(
518 						AvailableSettings.CACHE_REGION_FACTORY, properties, null
519 				)
520 		);
521 		if ( regionFactoryClassName == null || !cachingEnabled) {
522 			regionFactoryClassName = DEF_CACHE_REG_FACTORY;
523 		}
524 		LOG.debugf( "Cache region factory : %s", regionFactoryClassName );
525 		try {
526 			try {
527 				return (RegionFactory) serviceRegistry.getService( ClassLoaderService.class )
528 						.classForName( regionFactoryClassName )
529 						.getConstructor( Properties.class )
530 						.newInstance( properties );
531 			}
532 			catch ( NoSuchMethodException e ) {
533 				// no constructor accepting Properties found, try no arg constructor
534 				LOG.debugf(
535 						"%s did not provide constructor accepting java.util.Properties; attempting no-arg constructor.",
536 						regionFactoryClassName
537 				);
538 				return (RegionFactory) serviceRegistry.getService( ClassLoaderService.class )
539 						.classForName( regionFactoryClassName )
540 						.newInstance();
541 			}
542 		}
543 		catch ( Exception e ) {
544 			throw new HibernateException( "could not instantiate RegionFactory [" + regionFactoryClassName + "]", e );
545 		}
546 	}

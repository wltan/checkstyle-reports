136     /**
137      * @return Returns the portNumbers.
138      */
139     public String getPortNumbers() {
140         if (portNumbers == null || portNumbers.length < 1)
141             return "NONE";
142         int len = portNumbers.length;
143         String first = String.valueOf(portNumbers[0]);
144         if (len == 1)
145             return first;
146         StringBuffer sb = new StringBuffer(first);
147         for (int i = 1; i < len; i++)
148             sb.append(",").append(portNumbers[i]);
149         return sb.toString();
150     }

152     /**
153      * @param portNumbers
154      *            The portNumbers to set.
155      */
156     public void setPortNumbers(String ports) {
157         if (ports == null || "NONE".equalsIgnoreCase(ports)) {
158             portNumbers = null;
159         } else {
160             StringTokenizer st = new StringTokenizer(ports, ",");
161             portNumbers = new int[st.countTokens()];
162             for (int i = 0; st.hasMoreTokens(); i++) {
163                 portNumbers[i] = Integer.parseInt(st.nextToken());
164             }
165         }
166     }

197     public final String getOtherServiceAETAttrs() {
198         StringBuffer sb = new StringBuffer();
199         for (int i = 0; i < otherServiceNames.length; i++) {
200             sb.append(otherServiceNames[i].toString()).append('#').append(
201                     otherServiceAETAttrs[i]).append("\r\n");
202         }
203         return sb.toString();
204     }

206     public final void setOtherServiceAETAttrs(String s) {
207         StringTokenizer stk = new StringTokenizer(s, "\r\n\t; ");
208         int count = stk.countTokens();
209         ObjectName[] names = new ObjectName[count];
210         String[] attrs = new String[count];
211         String tk = null;
212         try {
213             int endName;
214             for (int i = 0; i < names.length; i++) {
215                 tk = stk.nextToken();
216                 endName = tk.indexOf('#');
217                 names[i] = new ObjectName(tk.substring(0, endName));
218                 attrs[i] = tk.substring(endName+1);
219             }
220         } catch (Exception e) {
221             throw new IllegalArgumentException(tk);
222         }
223         otherServiceNames = names;
224         otherServiceAETAttrs = attrs;
225     }

258     public void updateAETitle(String prevAET, String newAET) throws Exception {
259     	prevAET = prevAET.trim();
260     	newAET = newAET.trim();
261         if (prevAET.equals(newAET)) {
262             return;
263         }
264         checkAETitle(newAET);
265         
266         AEManager aeManager = aeMgr();
267         AEDTO ae = aeManager.findByAET(prevAET);
268         ae.setTitle(newAET);
269         aeManager.updateAE(ae);
270         
271         updateFileSystemRetrieveAET(prevAET, newAET);
272         updateOtherServicesForAETChange(prevAET, newAET);
273         
274         log.info("AE title changed from " + prevAET + " to " + newAET);
275         notifyAETchange(prevAET, newAET, "");
276     }

278 	protected void updateOtherServicesForAETChange(String prevAET, String newAET) throws Exception {
279 		for (int i = 0; i < otherServiceNames.length; i++) {
280             if (server.isRegistered(otherServiceNames[i])) {
281                 updateAETitle(otherServiceNames[i], otherServiceAETAttrs[i],
282                         prevAET, newAET);
283             } else {
284                 if (log.isDebugEnabled()) {
285                     log.debug("Service: " + otherServiceNames[i]
286                       + " not registered -> cannot update AETitle in attribute: "
287                       + otherServiceNames[i] + "#" + otherServiceAETAttrs[i]);
288                 }
289             }
290         }
291 	}

301     private boolean updateAETitle(ObjectName name, String attr,
302             String prevAET, String newAET) throws Exception {
303     	checkAETitle(newAET);
304         try {
305             String val = (String) server.getAttribute(name, attr);
306             String[] aets = StringUtils.split(val, '\\');
307             boolean modified = false;
308             for (int i = 0; i < aets.length; i++) {
309                 if (aets[i].equals(prevAET)) {
310                     aets[i] = newAET;
311                     modified = true;
312                 }
313             }
314             if (modified) {
315                 server.setAttribute(name, 
316                         new Attribute(attr, StringUtils.toString(aets, '\\')));
317                 log.info("Update AETitle in attribute: " + name + "#" + attr);
318             }
319             return modified;
320         } catch (AttributeNotFoundException e) {
321             log.info("No such attribute: " + name + "#" + attr);
322             return false;
323         }
324     }

331     public AEDTO getAE(String aet, InetAddress addr) throws Exception {
332         AEManager aetMgr = aeMgr();
333         try {
334             return updateIP(aetMgr.findByAET(aet), addr, aetMgr);
335         } catch (UnknownAETException e) {
336             return autoConfigAE(aet, addr, aetMgr);
337         }
338     }

340     private AEDTO updateIP(AEDTO ae, InetAddress addr, AEManager aetMgr)
341             throws Exception {
342         String oldIP, newIP;
343         if (autoUpdateIP && addr != null
344                 && ActiveParticipant.isIP(oldIP = ae.getHostName())
345                 && !oldIP.equals(newIP =  addr.getHostAddress())) {
346             ae.setHostName(newIP);
347             aetMgr.updateAE(ae);
348             log.info("Update IP of " + ae );
349         }
350         return ae;
351     }

353     private AEDTO autoConfigAE(String aet, InetAddress addr, AEManager aetMgr) 
354     throws Exception {
355         if (portNumbers == null || addr == null) {
356             return null;
357         }
358         String aeHost = addr.getHostName();
359         for (int i = 0; i < portNumbers.length; i++) {
360             AEDTO ae = new AEDTO();
361             ae.setTitle(aet);
362             ae.setHostName(aeHost);
363             ae.setPort(portNumbers[i]);
364             if (echo(ae)) {
365                 if (dontSaveIP) {
366                     if (!aeHost.equals(addr.getHostAddress()))
367                         aetMgr.newAE(ae);
368                 } else {
369                     aetMgr.newAE(ae);
370                 }
371                 logActorConfig("Add new auto-configured AE " + ae, 
372                         SecurityAlertMessage.NETWORK_CONFIGURATION);
373                 notifyAETchange(null,aet,"");
374                 return ae;
375             }
376         }
377         return null;
378     }

380     /**
381      * Adds (replace) a new AE Title.
382      * 
383      * @param aet
384      *            Application Entity Title
385      * @param host
386      *            Hostname or IP addr.
387      * @param port
388      *            port number
389      * @param cipher
390      *            String with cypher(s) to create a secure connection (seperated
391      *            with ',') or null
392      * @param checkHost
393      *            Enable/disable checking if the host can be resolved.
394      * 
395      * @throws Exception
396      * @throws RemoteException
397      */
398     public void updateAE(long pk, String title, String host, int port,
399             String cipher, String issuerOfPatientID,
400             String issuerOfAccessionNumber, String user, String passwd,
401             String fsGroupID, String group, String desc, String wadoUrl,
402             String stationName, String institution, String department,
403             boolean installed, boolean checkHost)
404     throws Exception {
405         if (checkHost) {
406             try {
407                 host = InetAddress.getByName(host).getCanonicalHostName();
408             } catch (UnknownHostException x) {
409                 throw new IllegalArgumentException("Host " + host
410                         + " cant be resolved! Disable hostname check to add new AE anyway!");
411             }
412         }
413         checkAETitle(title);
414 
415         AEDTO newAE = new AEDTO();
416         newAE.setPk(pk);
417         newAE.setTitle(title);
418         newAE.setHostName(host);
419         newAE.setPort(port);
420         newAE.setCipherSuitesAsString(cipher);
421         newAE.setIssuerOfPatientID(issuerOfPatientID);
422         newAE.setIssuerOfAccessionNumberAsString(issuerOfAccessionNumber);
423         newAE.setUserID(user);
424         newAE.setPassword(passwd);
425         newAE.setFileSystemGroupID(fsGroupID);
426         newAE.setGroup(group);
427         newAE.setDescription(desc);
428         newAE.setWadoURL(wadoUrl);
429         newAE.setStationName(stationName);
430         newAE.setInstitution(institution);
431         newAE.setDepartment(department);
432         newAE.setInstalled(installed);
433         AEManager aeManager = aeMgr();
434         if (pk == -1) {
435             aeManager.newAE(newAE);
436             logActorConfig("Add AE " + newAE + " cipher:"
437                     + newAE.getCipherSuitesAsString(),
438                     SecurityAlertMessage.NETWORK_CONFIGURATION);
439             notifyAETchange(null, title,"");
440 
441         } else {
442             AEDTO oldAE = aeManager.findByPrimaryKey(pk);
443             String oldAET = null;
444             if (!oldAE.getTitle().equals(title)) {
445                 try {
446                     AEDTO aeOldByTitle = aeManager.findByAET(title);
447                     throw new IllegalArgumentException("AE Title " + title
448                             + " already exists!:" + aeOldByTitle);
449                 } catch (UnknownAETException e) {}
450                 oldAET = oldAE.getTitle();
451             }
452             aeManager.updateAE(newAE);
453             logActorConfig("Modify AE " + oldAE + " -> " + newAE,
454                     SecurityAlertMessage.NETWORK_CONFIGURATION);
455             if ( oldAET != null )
456                 notifyAETchange(oldAET, title,"");
457         }
458     }

462             String user, String passwd, String fsGroupID,

472     public void removeAE(String titles) throws Exception {
473         StringTokenizer st = new StringTokenizer(titles, " ,;\t\r\n");
474         AEDTO ae;
475         AEManager aeManager = aeMgr();
476         while (st.hasMoreTokens()) {
477             ae = aeManager.findByAET(st.nextToken());
478             aeManager.removeAE(ae.getPk());
479             logActorConfig("Remove AE " + ae,
480                     SecurityAlertMessage.NETWORK_CONFIGURATION);
481             notifyAETchange(ae.getTitle(),null,"");
482         }
483     }

485     protected void logActorConfig(String desc, AuditEvent.TypeCode eventTypeCode) {
486         log.info(desc);
487         try {
488             HttpUserInfo userInfo = new HttpUserInfo(AuditMessage.isEnableDNSLookups());
489             SecurityAlertMessage msg = new SecurityAlertMessage(eventTypeCode);
490             msg.addReportingProcess(AuditMessage.getProcessID(),
491                     AuditMessage.getLocalAETitles(),
492                     AuditMessage.getProcessName(),
493                     AuditMessage.getLocalHostName());
494             msg.addPerformingPerson(userInfo.getUserId(), null, null, userInfo.getHostName());
495             msg.addAlertSubjectWithNodeID(AuditMessage.getLocalNodeID(), desc);
496             msg.validate();
497             Logger.getLogger("auditlog").info(msg);
498         } catch (Exception e) {
499             log.warn("Failed to log ActorConfig:", e);
500         }
501     }

503     public void notifyAETchange(String oldTitle, String newTitle, String message) {
504         long eventID = this.getNextNotificationSequenceNumber();
505         Notification notif = new Notification(AetChanged.class.getName(), this, eventID, message );
506         notif.setUserData(new AetChanged(oldTitle, newTitle) );
507         log.debug("send AE Title changed notif:"+notif);
508         this.sendNotification( notif );
509     }

511     private boolean echo(AEDTO ae) {
512         try {
513             Boolean result = (Boolean) server.invoke(this.echoServiceName,
514                     "checkEcho", new Object[] { ae },
515                     new String[] { AEDTO.class.getName() });
516             return result.booleanValue();
517         } catch (Exception e) {
518             log.warn("Failed to use echo service:", e);
519             return false;
520         }

1   package org.dcm4chex.archive.perf;
2   
3   import java.util.Arrays;
4   
5   /**
6    * Performance property enums for counters. It is couter's responsibility
7    * to retrieve appropriate data for performance measurement purpose.
8    * 
9    * @author Fang Yang (fang.yang@agfa.com)
10   * @version $Id: PerfPropertyEnum.java 2956 2006-11-15 22:03:47Z fangatagfa $
11   * @since Nov 13, 2006
12   */
13  public class PerfPropertyEnum {
14  	private static final String[] ENUM = { 
15  		"REQ_DIMSE",
16      	"REQ_DATASET",
17      	"DICOM_FILE",
18      	"RSP_DATASET",
19      	"NUM_OF_RESULTS",
20      	"STUDY_IUID"
21      };
22  
23      public static final int REQ_DIMSE = 1;
24      public static final int REQ_DATASET = 2;
25      public static final int DICOM_FILE = 3;
26      public static final int RSP_DATASET = 4;
27      public static final int NUM_OF_RESULTS = 5;
28      public static final int STUDY_IUID = 6;
29      
30      public static final String toString(int value) {
31         return ENUM[++value];
32      }
33  
34      public static final int toInt(String s) {
35      	final int index = Arrays.asList(ENUM).indexOf(s);
36          if (index == -1)
37              throw new IllegalArgumentException(s);
38          return index;
39      }
40  }

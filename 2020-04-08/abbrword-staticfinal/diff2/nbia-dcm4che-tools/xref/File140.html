321     public Dataset createPatient(Dataset ds) throws RemoteException,
322     CreateException, HomeFactoryException, FinderException {
323         if (log.isDebugEnabled())
324             log.debug("create Partient");
325         String pid = ds.getString(Tags.PatientID);
326         String issuer = ds.getString(Tags.IssuerOfPatientID);
327         Dataset ds1 = lookupContentManager().getPatientByID(pid, issuer);
328         if (ds1 != null) {
329             throw new CreateException("Patient with ID(pid:" + pid + " issuer:"
330                     + issuer + ") already exists!");
331         }
332         ds1 = lookupContentEdit().createPatient(ds);
333         sendHL7PatientXXX(ds, "ADT^A04");// use update to create patient, msg
334         // type is 'Register a patient'
335         logPatientRecord(ds, PatientRecordMessage.CREATE);
336         return ds1;
337     }

339     public Map mergePatients(Long patPk, long[] mergedPks)
340     throws RemoteException, HomeFactoryException, CreateException, RemoveException, EJBException {
341         if (log.isDebugEnabled())
342             log.debug("merge Partient");
343         Map map = lookupContentEdit().mergePatients(patPk.longValue(),
344                 mergedPks, keepPriorPatientAfterMerge);
345         if (!map.containsKey("ERROR")) {
346             sendHL7PatientMerge((Dataset) map.get("DOMINANT"), (Dataset[]) map
347                     .get("MERGED"));
348             String patID = ((Dataset) map.get("DOMINANT"))
349             .getString(Tags.PatientID);
350             sendJMXNotification(new PatientUpdated(patID, "Patient merge"));
351         }
352         return map;
353     }

408     private Map doLinkMppsToMwl(Object o, String[] mppsIUIDs) {
409         try {
410             Map map = (Map) server.invoke(mppsScpServiceName, "linkMppsToMwl",
411                     new Object[] { o, mppsIUIDs }, new String[] {
412                     o.getClass().getName(), String[].class.getName() });
413             List<Dataset> studyDsN = (List<Dataset>) map.get("StudyMgtDS");
414             Dataset dsN;
415             for (Iterator iter = studyDsN.iterator(); iter.hasNext();) {
416                 dsN = (Dataset) iter.next();
417                 sendStudyMgt(dsN.getString(Tags.StudyInstanceUID),
418                         Command.N_SET_RQ, 0, dsN);
419                 sendSeriesUpdatedNotifications(dsN, "Series update");
420             }
421             return map;
422         } catch (Exception x) {
423             log.error("Exception occured in linkMppsToMwl: " + x.getMessage(),
424                     x);
425             return null;
426         }

430     private void sendStudyMgt(String iuid, int commandField, int actionTypeID,
431             Dataset dataset) {
432         String infoStr = "iuid: " + iuid + ", cmd:" + commandField
433         + ", action: " + actionTypeID + ", ds:";
434         if (log.isDebugEnabled()) {
435             log.debug("send StudyMgt command: " + infoStr);
436             log.debug(dataset);
437         }
438 
439         try {
440             server.invoke(this.studyMgtScuServiceName, "forward", new Object[] {
441                     this.callingAET, this.calledAET, iuid,
442                     new Integer(commandField), new Integer(actionTypeID),
443                     dataset }, new String[] { String.class.getName(),
444                     String.class.getName(), String.class.getName(),
445                     int.class.getName(), int.class.getName(),
446                     Dataset.class.getName() });
447         } catch (InstanceNotFoundException infe) {
448             log.warn("The MBean service [" + studyMgtScuServiceName
449                     + "] is not registered. Ignore sending StudyMgt command: "
450                     + infoStr);
451         } catch (Exception e) {
452             log.error("Failed to send StudyMgt command:" + infoStr, e);
453         }
454     }

548     private void logMovePatientToTrash(Collection<Dataset> ians,
549             boolean sendHL7_ADT_A23) {
550         Dataset ds = null;
551         for (Iterator iter = ians.iterator(); iter.hasNext();) {
552             ds = (Dataset) iter.next();
553             if (ds.containsValue(Tags.StudyInstanceUID)) { // patient without  study?
554                 if (createIANonMoveToTrash) {
555                     sendJMXNotification(new StudyDeleted(ds));
556                 }
557                 logStudyDeleted(ds);
558             }
559         }
560         if (sendHL7_ADT_A23) {
561             sendHL7PatientXXX(ds, "ADT^A23");// Send Patient delete message
562         }
563         logPatientRecord(ds, PatientRecordMessage.DELETE);
564         if (log.isDebugEnabled()) {
565             log.debug("Patient moved to trash. ds:");
566             log.debug(ds);
567         }
568     }

638     public Dataset moveSeriesToTrash(String[] iuids) throws Exception {
639         if (log.isDebugEnabled())
640             log.debug("Move "+iuids != null ? iuids.length : null +" Series to trash: " + iuids);
641         Collection<Dataset> dss = (Collection<Dataset>)lookupPrivateManager().moveSeriesToTrash(iuids);
642         if (dss.size() != 1)
643             throw new Exception("moveSeriesToTrash failed");
644         for ( Dataset ds : dss ) {
645             if (createIANonMoveToTrash) {

726     /**
727      * @param files
728      */
729     private List recoverFiles(List[] files) {
730         if (files == null || files.length != 2 || files[0] == null
731                 || files[1] == null || files[0].size() != files[1].size()) {
732             throw new IllegalArgumentException(
733                     "List array for files to recover is illegal:" + files);
734         }
735         List failed = new ArrayList();
736         Iterator iterFileDTO = files[0].iterator();
737         Iterator iterDS = files[1].iterator();
738         String prevseriuid = null;
739         while (iterFileDTO.hasNext()) {
740             FileDTO fileDTO = (FileDTO) iterFileDTO.next();
741             Dataset ds = (Dataset) iterDS.next();
742             try {
743                 importFile(fileDTO, ds, prevseriuid, !iterFileDTO.hasNext());
744                 prevseriuid = ds.getString(Tags.SeriesInstanceUID);
745             } catch (Exception e) {
746                 failed.add(fileDTO);
747                 log.warn("Undelete failed for file " + fileDTO + " ds:", e);
748                 log.warn(ds);
749             }
750         }
751         return failed;
752     }

805     RemoteException {
806         if (log.isDebugEnabled())

811             Dataset[] ds = lookupContentEdit().getStudyMgtDatasetForStudies(study_pks);

814              }

825             sendStudyMgt(dsStdyMgt.getString(Tags.StudyInstanceUID), Command.N_SET_RQ, 0, dsStdyMgt);

827         }

964     protected void logInstancesAccessed(Collection<Dataset> studies,
965             AuditEvent.ActionCode actionCode, boolean addIUID, String detailMessage) {
966         if (!auditEnabled)
967             return;
968         HttpUserInfo userInfo = new HttpUserInfo(AuditMessage
969                 .isEnableDNSLookups());
970         log.debug("log instances Accessed! actionCode:" + actionCode);
971         try {
972             InstancesAccessedMessage msg = new InstancesAccessedMessage(
973                     actionCode);
974             msg.addUserPerson(userInfo.getUserId(), null, null, userInfo
975                     .getHostName(), true);
976             Iterator<Dataset> iter = studies.iterator();
977             Dataset studyMgtDs = (Dataset) iter.next();
978             PersonName pn = studyMgtDs.getPersonName(Tags.PatientName);
979             String pname = pn != null ? pn.format() : null;
980             msg.addPatient(studyMgtDs.getString(Tags.PatientID), pname);
981             ParticipantObject study;
982             while (studyMgtDs != null) {
983                 study = msg.addStudy(studyMgtDs.getString(Tags.StudyInstanceUID),
984                         getStudyDescription(studyMgtDs, addIUID));
985                 if ( detailMessage != null )
986                     study.addParticipantObjectDetail("Description", getStudySeriesDetail(detailMessage, studyMgtDs));
987                 studyMgtDs = iter.hasNext() ? iter.next() : null;
988             }
989             msg.validate();
990             Logger.getLogger("auditlog").info(msg);
991         } catch (Exception x) {
992             log.warn("Audit Log 'Instances Accessed' (actionCode:" + actionCode
993                     + ") failed:", x);
994         }
995     }

1041     protected void addSOPClassInfo(ParticipantObjectDescription desc,
1042             Dataset studyMgtDs, boolean addIUID) {
1043         DcmElement refSeries = studyMgtDs.get(Tags.RefSeriesSeq);
1044         if (refSeries == null)
1045             return;
1046         String suid = studyMgtDs.getString(Tags.StudyInstanceUID);
1047         InstanceSorter sorter = new InstanceSorter();
1048         Dataset ds;
1049         DcmElement refSopSeq;
1050         for (int i = 0, len = refSeries.countItems(); i < len; i++) {
1051             refSopSeq = refSeries.getItem(i).get(Tags.RefSOPSeq);
1052             if (refSopSeq != null) {
1053                 for (int j = 0, jlen = refSopSeq.countItems(); j < jlen; j++) {
1054                     ds = refSopSeq.getItem(j);
1055                     sorter.addInstance(suid,
1056                             ds.getString(Tags.RefSOPClassUID),
1057                             ds.getString(Tags.RefSOPInstanceUID), null);
1058                 }
1059             }
1060         }
1061         
1062         for (String cuid : sorter.getCUIDs(suid)) {
1063             ParticipantObjectDescription.SOPClass sopClass = new ParticipantObjectDescription.SOPClass(
1064                     cuid);
1065             sopClass.setNumberOfInstances(sorter.countInstances(suid,
1066                     cuid));
1067             if ( addIUID ) {
1068                 for ( String iuid : sorter.getIUIDs(suid, cuid) ) {
1069                     sopClass.addInstance(iuid);
1070                 }
1071             }
1072             desc.addSOPClass(sopClass);
1073         }
1074     }

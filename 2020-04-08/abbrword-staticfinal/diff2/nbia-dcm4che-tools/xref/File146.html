153     public final void setConcurrency(int concurrency) throws Exception {
154         if (concurrency <= 0)
155             throw new IllegalArgumentException("Concurrency: " + concurrency);
156         if (this.concurrency != concurrency) {
157             final boolean restart = getState() == STARTED;
158             if (restart)
159                 stop();
160             this.concurrency = concurrency;
161             if (restart)
162                 start();
163         }
164     }

224     /**
225      * JMS Message listener anonymous inner class. Upon a message reception,
226      * calls the method <i>process</i>

231             ObjectMessage objmsg = (ObjectMessage) msg;
232             try {

237             }
238         }

251     /**
252      * Core method of the service. Sends a new mail. A huge number of Exception
253      * types may be arised during the process of the method, but every Exception
254      * is thrown as a MessagingException eventually including the Exception
255      * source that originated it.
256      * 
257      * @param mailProps
258      *            This Map holds the properties (subject, toAddress, body,..)
259      *            for the mail to be sent.
260      * @throws MessagingException
261      *             if an Exception of any kind is thrown during
262      * @throws JMSException
263      * 
264      */
265     private void process(Map mailProps) throws Exception {
266         log.info("process " + mailProps);
267 
268         // mail session and mime message creation
269         Session session = Session.getInstance(createSessionProperties());
270         // session.setDebugOut( System.out );
271         // session.setDebug( true );
272         log.info("Mail session props:" + session.getProperties());
273         MimeMessage message = new MimeMessage(session);
274 
275         // set subject, sent date, from, to and reply-to addresses to the
276         // message
277         initSimpleMail(mailProps, message);
278 
279         File[] files = (File[]) mailProps.get(MAIL_FILES);
280         if (files != null && files.length > 0) // there are files to be sent
281         {
282             Multipart multipart = new MimeMultipart("mixed");
283 
284             // body of the message.
285             BodyPart textbodypart = new MimeBodyPart();
286             textbodypart.setText((String) mailProps.get(MAIL_BODY));
287             multipart.addBodyPart(textbodypart);
288 
289             // add each file as an attachment
290             for (int i = 0; i < files.length; i++) {
291                 if (files[i].isFile()) {
292                     MimeBodyPart filebodypart = new MimeBodyPart();
293                     FileDataSource source = new FileDataSource(files[i]);
294                     filebodypart.setDataHandler(new DataHandler(source));
295                     filebodypart.setFileName(files[i].getName());
296                     // TODO filebodypart.setHeader("Content-Type",
297                     // "Application/dicom");
298                     multipart.addBodyPart(filebodypart);
299                 }
300             }
301             message.setContent(multipart);
302             log.info(new StringBuffer().append("sending message: ").append(
303                     files.length).append(" files attached"));
304         }
305         else {
306             message.setText((String) mailProps.get(MAIL_BODY));
307         }
308 
309         try {
310             message.saveChanges();
311             Transport.send(message);
312             log.info("Message sent to " + mailProps.get(MAIL_TO_ADDR));
313             sendJMXNotification(new SendMailMessageStatus(mailProps));
314         }
315         catch (Exception e) {
316             log.warn("Failed to send " + mailProps, e);
317             int failureCount = 0;
318             Integer failures = (Integer) mailProps.get(MAIL_FAILURE_COUNT);
319             if (failures != null) {
320                 failureCount = failures.intValue();
321             }
322             failureCount++;
323             mailProps.put(MAIL_FAILURE_COUNT, new Integer(failureCount));
324             final long delay = retryIntervalls.getIntervall(failureCount);
325             if (delay == -1L) {
326                 log.error("Give up to send " + mailProps);
327                 sendJMXNotification(new SendMailMessageStatus(false, mailProps));
328             }
329             else {
330                 log.warn("Failed to send " + mailProps + ". Scheduling retry.");
331                 jmsDelegate.queue(queueName, (Serializable) mailProps, 0,
332                         System.currentTimeMillis() + delay);
333             }
334         }
335 
336     }

338     /**
339      * public method used (principally for debug reasons) to send a dcmmail
340      * message without the correspondent reception of a JMS message.
341      */
342     public String send(String subject, String fromAddress, String toAddress,
343             String body) {
344         HashMap mail = new HashMap();
345         mail.put(MAIL_BODY, body);
346         mail.put(MAIL_FROM_ADDR, fromAddress);
347         mail.put(MAIL_SUBJECT, subject);
348         mail.put(MAIL_TO_ADDR, toAddress);
349 
350         try {
351             jmsDelegate.queue(queueName, mail, Message.DEFAULT_PRIORITY, 0L);
352             return "Mail '" + subject + "' succesfully sent to " + toAddress;
353         }
354         catch (Exception e) {
355             log.error(e.getMessage(), e);
356             return "Failed to send mail '" + subject + "' to " + toAddress
357                     + " ! Reason:" + e.getMessage();
358         }
359     }

361     /**
362      * Creates and initialize a mail session Properties object.
363      * 
364      * @return an initialized a mail session Properties object.
365      */
366     private Properties createSessionProperties() {
367         Properties properties = new Properties();
368 
369         if (getSmtpHost().equals(null)) {
370             properties.put("mail.smtp.host", "127.0.0.1");
371         }
372         else {
373             properties.put("mail.smtp.host", smtpHost);
374         }
375 
376         properties.put("mail.smtp.port", String.valueOf(this.smtpPort));
377 
378         if (smtpUser != null && smtpUser.length() > 0 && smtpPassword != null
379                 && smtpPassword.length() > 0) {
380             properties.put("mail.smtp.auth", String.valueOf(smtpAuth));
381             properties.put("mail.smtp.user", smtpUser);
382             properties.put("mail.smtp.password", smtpPassword);
383         }
384         else {
385             properties.put("mail.smtp.auth", "false");
386         }
387 
388         return properties;
389     }

391     /**
392      * Sets from, to and reply-to addresses for this message. Reply-to address
393      * is set equal to the From address.
394      * 
395      * @param dcmMail
396      *            the object representig the dicom mail
397      * @param message
398      *            the MimeMessage mail messagge
399      * @throws AddressException
400      *             if addresses are not present or not valid
401      * @throws MessagingException -
402      */
403     private void initSimpleMail(Map map, MimeMessage message)
404             throws AddressException, MessagingException {
405 
406         // set subject and date to the message
407         if (map.get(MAIL_SUBJECT) != null)
408             message.setSubject((String) map.get(MAIL_SUBJECT));
409         message.setSentDate(new Date());
410 
411         String fromAddr = (String) map.get(MAIL_FROM_ADDR);
412         if (fromAddr == null) {
413             // throw new AddressException("no from address");
414         }
415         else {
416             try {
417                 message.setFrom(new InternetAddress(fromAddr));
418                 String replyTo = (String) map.get(MAIL_REPLY_TO);
419                 if (replyTo == null)
420                     replyTo = fromAddr;
421                 try {
422                     message.setReplyTo(new Address[] { new InternetAddress(
423                             replyTo) });
424                 }
425                 catch (Exception x) {
426                     // ignore; some implementation doesnt support this method!
427                 }
428             }
429             catch (AddressException e) {
430                 throw new AddressException("invalid from address: " + fromAddr
431                         + ": " + e.getMessage());
432             }
433         }
434 
435         String toAddr = (String) map.get(MAIL_TO_ADDR);
436         if (toAddr == null) {
437             throw new AddressException("no to address");
438         }
439         else {
440             try {
441                 message.setRecipients(javax.mail.Message.RecipientType.TO,
442                         InternetAddress.parse(toAddr));
443             }
444             catch (AddressException e) {
445                 throw new AddressException("invalid to address: " + toAddr
446                         + ": " + e.getMessage());
447             }
448         }
449         String ccAddr = (String) map.get(MAIL_CC_ADDR);
450         if (ccAddr != null) {
451             try {
452                 message.setRecipients(javax.mail.Message.RecipientType.CC,
453                         InternetAddress.parse(ccAddr));
454             }
455             catch (AddressException e) {
456                 throw new AddressException("invalid cc address: " + ccAddr
457                         + ": " + e.getMessage());
458             }
459         }
460     }

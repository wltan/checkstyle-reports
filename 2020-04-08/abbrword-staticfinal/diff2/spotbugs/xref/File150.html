228     /**
229      * Factory method for creating a source line annotation describing an entire
230      * method.
231      *
232      * @param methodGen
233      *            the method being visited
234      * @return the SourceLineAnnotation, or null if we do not have line number
235      *         information for the method
236      */
237     public static SourceLineAnnotation fromVisitedMethod(MethodGen methodGen, String sourceFile) {
238         LineNumberTable lineNumberTable = methodGen.getLineNumberTable(methodGen.getConstantPool());
239         String className = methodGen.getClassName();
240         int codeSize = methodGen.getInstructionList().getLength();
241         if (lineNumberTable == null) {
242             return createUnknown(className, sourceFile, 0, codeSize - 1);
243         }
244         return forEntireMethod(className, sourceFile, lineNumberTable, codeSize);
245     }

247     /**
248      * Create a SourceLineAnnotation covering an entire method.
249      *
250      * @param className
251      *            name of the class the method is in
252      * @param sourceFile
253      *            source file containing the method
254      * @param lineNumberTable
255      *            the method's LineNumberTable
256      * @param codeSize
257      *            size in bytes of the method's code
258      * @return a SourceLineAnnotation covering the entire method
259      */
260     public static SourceLineAnnotation forEntireMethod(@DottedClassName String className, String sourceFile, LineNumberTable lineNumberTable,
261             int codeSize) {
262         LineNumber[] table = lineNumberTable.getLineNumberTable();
263         if (table != null && table.length > 0) {
264             LineNumber first = table[0];
265             LineNumber last = table[table.length - 1];
266             return new SourceLineAnnotation(className, sourceFile, first.getLineNumber(), last.getLineNumber(), 0, codeSize - 1);
267         } else {
268             return createUnknown(className, sourceFile, 0, codeSize - 1);
269         }
270     }

272     /**
273      * Create a SourceLineAnnotation covering an entire method.
274      *
275      * @param javaClass
276      *            JavaClass containing the method
277      * @param method
278      *            the method
279      * @return a SourceLineAnnotation for the entire method
280      */
281     public static SourceLineAnnotation forEntireMethod(JavaClass javaClass, @CheckForNull Method method) {
282         String sourceFile = javaClass.getSourceFileName();
283         if (method == null) {
284             return createUnknown(javaClass.getClassName(), sourceFile);
285         }
286         Code code = method.getCode();
287         LineNumberTable lineNumberTable = method.getLineNumberTable();
288         if (code == null || lineNumberTable == null) {
289             return createUnknown(javaClass.getClassName(), sourceFile);
290         }
291 
292         return forEntireMethod(javaClass.getClassName(), sourceFile, lineNumberTable, code.getLength());
293     }

295     /**
296      * Create a SourceLineAnnotation covering an entire method.
297      *
298      * @param javaClass
299      *            JavaClass containing the method
300      * @param xmethod
301      *            the method
302      * @return a SourceLineAnnotation for the entire method
303      */
304     public static SourceLineAnnotation forEntireMethod(JavaClass javaClass, XMethod xmethod) {
305         JavaClassAndMethod m = Hierarchy.findMethod(javaClass, xmethod.getName(), xmethod.getSignature());
306         if (m == null) {
307             return createUnknown(javaClass.getClassName(), javaClass.getSourceFileName());
308         } else {
309             return forEntireMethod(javaClass, m.getMethod());
310         }
311     }

313     /**
314      * Make a best-effort attempt to create a SourceLineAnnotation for the first
315      * line of a method.
316      *
317      * @param methodDescriptor
318      *            a method
319      * @return SourceLineAnnotation describing the first line of the method
320      *         (insofar as we can actually figure that out from the bytecode)
321      */
322     public static SourceLineAnnotation forFirstLineOfMethod(MethodDescriptor methodDescriptor) {
323         SourceLineAnnotation result = null;
324 
325         try {
326             Method m = Global.getAnalysisCache().getMethodAnalysis(Method.class, methodDescriptor);
327             XClass xclass = Global.getAnalysisCache().getClassAnalysis(XClass.class, methodDescriptor.getClassDescriptor());
328             LineNumberTable lnt = m.getLineNumberTable();
329             String sourceFile = xclass.getSource();
330             if (sourceFile != null && lnt != null) {
331                 int firstLine = Integer.MAX_VALUE;
332                 int bytecode = 0;
333                 LineNumber[] entries = lnt.getLineNumberTable();
334                 for (LineNumber entry : entries) {
335                     if (entry.getLineNumber() < firstLine) {
336                         firstLine = entry.getLineNumber();
337                         bytecode = entry.getStartPC();
338                     }
339                 }
340                 if (firstLine < Integer.MAX_VALUE) {
341 
342                     result = new SourceLineAnnotation(methodDescriptor.getClassDescriptor().toDottedClassName(), sourceFile,
343                             firstLine, firstLine, bytecode, bytecode);
344                 }
345             }
346         } catch (CheckedAnalysisException e) {
347             // ignore
348         }
349 
350         if (result == null) {
351             result = createUnknown(methodDescriptor.getClassDescriptor().toDottedClassName());
352         }
353         return result;
354     }

621     private static LineNumberTable getLineNumberTable(PreorderVisitor visitor) {
622         Code code = visitor.getMethod().getCode();
623         if (code == null) {
624             return null;
625         }
626         return code.getLineNumberTable();
627     }

717     @Override
718     public String format(String key, ClassAnnotation primaryClass) {
719         if ("hash".equals(key)) {
720             return "";
721         }
722         if ("".equals(key)) {
723             StringBuilder buf = new StringBuilder();
724             buf.append(sourceFile);
725             appendLines(buf);
726             return buf.toString();
727         } else if ("lineNumber".equals(key)) {
728             StringBuilder buf = new StringBuilder();
729             appendLinesRaw(buf);
730             return buf.toString();
731         } else if ("full".equals(key)) {
732             StringBuilder buf = new StringBuilder();
733             String pkgName = getPackageName();
734             if (!"".equals(pkgName)) {
735                 buf.append(pkgName.replace('.', CANONICAL_PACKAGE_SEPARATOR));
736                 buf.append(CANONICAL_PACKAGE_SEPARATOR);
737             }
738             buf.append(sourceFile);
739             appendLines(buf);
740             return buf.toString();
741         } else {
742             throw new IllegalArgumentException("Unknown format key " + key);
743         }
744     }

746     private void appendLines(StringBuilder buf) {
747         if (isUnknown()) {
748             return;
749         }
750         buf.append(":[");
751         appendLinesRaw(buf);
752         buf.append(']');
753     }
754 
755     private void appendLinesRaw(StringBuilder buf) {
756         if (isUnknown()) {
757             return;
758         }
759         if (startLine == endLine) {
760             buf.append("line ");
761             buf.append(startLine);
762         } else {
763             buf.append("lines ");
764             buf.append(startLine);
765             buf.append('-');
766             buf.append(endLine);
767         }
768 
769     }

792     @Override
793     public int compareTo(BugAnnotation o) {
794         if (!(o instanceof SourceLineAnnotation)) {
795             // comparable
796             return this.getClass().getName().compareTo(o.getClass().getName());
797         }
798 
799         SourceLineAnnotation other = (SourceLineAnnotation) o;
800 
801         int cmp = className.compareTo(other.className);
802         if (cmp != 0) {
803             return cmp;
804         }
805         cmp = startLine - other.startLine;
806         if (cmp != 0) {
807             return cmp;
808         }
809         cmp = endLine - other.endLine;
810         if (startLine != -1) {
811             return 0;
812         }
813         if (cmp != 0) {
814             return cmp;
815         }
816         cmp = startBytecode - other.startBytecode;
817         if (cmp != 0) {
818             return cmp;
819         }
820         return endBytecode - other.endBytecode;
821     }

827         }

831     @Override
832     public boolean equals(Object o) {
833         if (!(o instanceof SourceLineAnnotation)) {
834             return false;
835         }
836         SourceLineAnnotation other = (SourceLineAnnotation) o;
837         if (!getDescription().equals(other.getDescription())) {
838             return false;
839         }
840         if (startLine != -1) {
841             return className.equals(other.className) && startLine == other.startLine && endLine == other.endLine;
842         }
843         return className.equals(other.className) && startBytecode == other.startBytecode && endBytecode == other.endBytecode;

872     @Override
873     public void writeXML(XMLOutput xmlOutput, boolean addMessages, boolean isPrimary) throws IOException {
874         String classname = getClassName();
875         String sourcePath = getSourcePath();
876 
877         XMLAttributeList attributeList = new XMLAttributeList().addAttribute("classname", classname);
878         if (isPrimary) {
879             attributeList.addAttribute("primary", "true");
880         }
881 
882         int n = getStartLine(); // start/end are now optional (were too many
883         // "-1"s in the xml)
884         if (n >= 0) {
885             attributeList.addAttribute("start", String.valueOf(n));
886         }
887         n = getEndLine();
888         if (n >= 0) {
889             attributeList.addAttribute("end", String.valueOf(n));
890         }
891         n = getStartBytecode(); // startBytecode/endBytecode haven't been set
892         // for a while now
893         if (n >= 0) {
894             attributeList.addAttribute("startBytecode", String.valueOf(n));
895         }
896         n = getEndBytecode();
897         if (n >= 0) {
898             attributeList.addAttribute("endBytecode", String.valueOf(n));
899         }
900 
901         if (isSourceFileKnown()) {
902             attributeList.addAttribute("sourcefile", sourceFile);
903             attributeList.addAttribute("sourcepath", sourcePath);
904             Project project = myProject.get();
905             if (project != null) {
906                 try {
907                     SourceFinder mySourceFinder = project.getSourceFinder();
908                     String fullPath = new File(mySourceFinder.findSourceFile(this).getFullFileName()).getCanonicalPath();
909                     String myRelativeSourceBase = relativeSourceBase.get();
910                     if (fullPath.startsWith(myRelativeSourceBase) && fullPath.length() > myRelativeSourceBase.length()) {
911                         attributeList.addAttribute("relSourcepath", fullPath.substring(myRelativeSourceBase.length() + 1));
912                     }
913                 } catch (IOException e) {
914                     assert true;
915                 }
916             }
917         }
918 
919         String role = getDescription();
920         if (!DEFAULT_ROLE.equals(role)) {
921             attributeList.addAttribute("role", getDescription());
922         }
923         if (synthetic) {
924             attributeList.addAttribute("synthetic", "true");
925         }
926         if (addMessages) {
927             xmlOutput.openTag(ELEMENT_NAME, attributeList);
928             xmlOutput.openTag("Message");
929             xmlOutput.writeText(this.toString());
930             xmlOutput.closeTag("Message");
931             xmlOutput.closeTag(ELEMENT_NAME);
932         } else {
933             xmlOutput.openCloseTag(ELEMENT_NAME, attributeList);
934         }
935     }

960     static SourceLineAnnotation getSourceAnnotationForMethod(String className, String methodName, String methodSig) {
961         JavaClassAndMethod targetMethod = null;
962         Code code = null;
963 
964         try {
965             JavaClass targetClass = AnalysisContext.currentAnalysisContext().lookupClass(className);
966             targetMethod = Hierarchy.findMethod(targetClass, methodName, methodSig);
967             if (targetMethod != null) {
968                 Method method = targetMethod.getMethod();
969                 if (method != null) {
970                     code = method.getCode();
971                 }
972             }
973 
974         } catch (ClassNotFoundException e) {
975             AnalysisContext.reportMissingClass(e);
976         }
977         SourceInfoMap sourceInfoMap = AnalysisContext.currentAnalysisContext().getSourceInfoMap();
978         SourceInfoMap.SourceLineRange range = sourceInfoMap.getMethodLine(className, methodName, methodSig);
979 
980         if (range != null) {
981             return new SourceLineAnnotation(className, AnalysisContext.currentAnalysisContext().lookupSourceFile(className),
982                     range.getStart(), range.getEnd(), 0, code == null ? -1 : code.getLength());
983         }
984 
985         if (sourceInfoMap.fallBackToClassfile() && targetMethod != null) {
986             return forEntireMethod(targetMethod.getJavaClass(), targetMethod.getMethod());
987         }
988 
989         // If we couldn't find the source lines,
990         // create an unknown source line annotation referencing
991         // the class and source file.
992 
993         return createUnknown(className);
994     }

996     static SourceLineAnnotation getSourceAnnotationForClass(String className, String sourceFileName) {
997 
998         int lastLine = -1;
999         int firstLine = Integer.MAX_VALUE;
1000 
1001         try {
1002             JavaClass targetClass = AnalysisContext.currentAnalysisContext().lookupClass(className);
1003             for (Method m : targetClass.getMethods()) {
1004                 Code c = m.getCode();
1005                 if (c != null) {
1006                     LineNumberTable table = c.getLineNumberTable();
1007                     if (table != null) {
1008                         for (LineNumber line : table.getLineNumberTable()) {
1009                             lastLine = Math.max(lastLine, line.getLineNumber());
1010                             firstLine = Math.min(firstLine, line.getLineNumber());
1011                         }
1012                     }
1013                 }
1014             }
1015         } catch (ClassNotFoundException e) {
1016             AnalysisContext.reportMissingClass(e);
1017         }
1018         if (firstLine < Integer.MAX_VALUE) {
1019             return new SourceLineAnnotation(className, sourceFileName, firstLine, lastLine, -1, -1);
1020         }
1021         return SourceLineAnnotation.createUnknown(className, sourceFileName);
1022     }

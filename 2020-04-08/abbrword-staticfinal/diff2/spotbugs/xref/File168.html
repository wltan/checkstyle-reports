20  package edu.umd.cs.findbugs.ba;
21  
22  import java.util.BitSet;
23  
24  /**
25   * Dataflow value representing the current nesting of catch and finally blocks.
26   * We assume that any catch block with a non-empty catch type is a user catch
27   * block, and any catch block with an empty catch type (i.e., catch all
28   * exceptions) is a finally block. This assumption isn't quite accurate, but it
29   * seems to be a reasonable first approximation.
30   *
31   * <p>
32   * If valid (isValid() returns true), a BlockType value is a stack of elements,
33   * which are either CATCH or FINALLY values. Call getDepth() to get the current
34   * nesting depth. Call get(int <i>n</i>) to get the <i>n</i>th stack item. Call
35   * getTopValue() to get the current top of the stack.
36   * </p>
37   *
38   * <p>
39   * If invalid (isValid() returns false), a BlockType value is either <i>top</i>
40   * or <i>bottom</i>. These are the special values at the top and bottom of the
41   * dataflow lattice.
42   * </p>
43   *
44   * <p>
45   * The dataflow lattice is effectively finite-height because real Java methods
46   * are guaranteed to have a finite catch and finally block nesting level.
47   * </p>
48   *
49   * @see BlockTypeAnalysis
50   * @author David Hovemeyer
51   */
52  public class BlockType extends BitSet {
53      /**
54       *
55       */
56      private static final long serialVersionUID = 1L;
57  
58      private static final int PRIME = 31;
59  
60      public static final boolean CATCH = false;
61  
62      public static final boolean FINALLY = true;
63  
64      private boolean isValid;
65  
66      private boolean isTop;
67  
68      private int depth;
69  
70      /**
71       * Constructor. Should only be called by BlockTypeAnalysis.
72       */
73      BlockType() {
74      }
75  
76      @Override
77      public int hashCode() {
78          int result = super.hashCode();
79          result = PRIME * result + depth;
80          result = PRIME * result + (isTop ? 1231 : 1237);
81          result = PRIME * result + (isValid ? 1231 : 1237);
82          return result;
83      }
84  
85      @Override
86      public boolean equals(Object obj) {
87          if (this == obj) {
88              return true;
89          }
90          if (!super.equals(obj)) {
91              return false;
92          }
93          if (!(obj instanceof BlockType)) {
94              return false;
95          }
96          final BlockType other = (BlockType) obj;
97          if (depth != other.depth) {
98              return false;
99          }
100         if (isTop != other.isTop) {
101             return false;
102         }
103         if (isValid != other.isValid) {
104             return false;
105         }
106         return true;
107     }
108 
109     /**
110      * Return whether or not this value is valid, meaning it contains a valid
111      * representation of the nesting of catch and finally blocks.
112      */
113     public boolean isValid() {
114         return isValid;
115     }
116 
117     /**
118      * Get the current nesting depth. The value must be valid.
119      */
120     public int getDepth() {
121         if (!isValid) {
122             throw new IllegalStateException();
123         }
124         return depth;
125     }
126 
127     /**
128      * Get the top value on the catch and finally block nesting stack.
129      */
130     public boolean getTopValue() {
131         if (depth == 0) {
132             throw new IllegalStateException();
133         }
134         return get(depth - 1);
135     }
136 
137     /**
138      * Return whether or not this value represents "normal" control-flow. Normal
139      * control flow are all blocks outside any catch or finally block.
140      */
141     public boolean isNormal() {
142         if (!isValid) {
143             throw new IllegalStateException();
144         }
145         return getDepth() == 0;
146     }
147 
148     /**
149      * Make this value represent "normal" control flow.
150      */
151     public void setNormal() {
152         this.isValid = true;
153         this.depth = 0;
154     }
155 
156     /**
157      * Return whether or not this is the special "top" dataflow value.
158      */
159     public boolean isTop() {
160         return !isValid && isTop;
161     }
162 
163     /**
164      * Make this the special "top" dataflow value.
165      */
166     public void setTop() {
167         this.isValid = false;
168         this.isTop = true;
169     }
170 
171     /**
172      * Return whether or not this is the special "bottom" dataflow value.
173      */
174     public boolean isBottom() {
175         return !isValid && !isTop;
176     }
177 
178     /**
179      * Make this the special "bottom" dataflow value.
180      */
181     public void setBottom() {
182         this.isValid = false;
183         this.isTop = false;
184     }
185 
186     /**
187      * Make this object an exact duplicate of given object.
188      *
189      * @param other
190      *            the other BlockType object
191      */
192     public void copyFrom(BlockType other) {
193         this.isValid = other.isValid;
194         this.isTop = other.isTop;
195         if (isValid) {
196             this.depth = other.depth;
197             this.clear();
198             this.or(other);
199         }
200     }
201 
202     /**
203      * Return whether or not this object is identical to the one given.
204      *
205      * @param other
206      *            the other BlockType object
207      * @return true if this object is identical to the one given, false
208      *         otherwise
209      */
210     public boolean sameAs(BlockType other) {
211         if (!this.isValid) {
212             return !other.isValid && (this.isTop == other.isTop);
213         } else {
214             if (!other.isValid) {
215                 return false;
216             } else {
217                 // Both facts are valid
218                 if (this.depth != other.depth) {
219                     return false;
220                 }
221 
222                 // Compare bits
223                 for (int i = 0; i < this.depth; ++i) {
224                     if (this.get(i) != other.get(i)) {
225                         return false;
226                     }
227                 }
228 
229                 return true;
230             }
231         }
232     }
233 
234     /**
235      * Merge other dataflow value into this value.
236      *
237      * @param other
238      *            the other BlockType value
239      */
240     public void mergeWith(BlockType other) {
241         if (this.isTop() || other.isBottom()) {
242             copyFrom(other);
243         } else if (isValid()) {
244             // To merge, we take the common prefix
245             int pfxLen = Math.min(this.depth, other.depth);
246             int commonLen;
247             for (commonLen = 0; commonLen < pfxLen; ++commonLen) {
248                 if (this.get(commonLen) != other.get(commonLen)) {
249                     break;
250                 }
251             }
252             this.depth = commonLen;
253         }
254     }
255 
256     /**
257      * Enter a catch block.
258      */
259     public void pushCatch() {
260         push(CATCH);
261     }
262 
263     /**
264      * Enter a finally block.
265      */
266     public void pushFinally() {
267         push(FINALLY);
268     }
269 
270     @Override
271     public String toString() {
272         if (isTop()) {
273             return "<top>";
274         } else if (isBottom()) {
275             return "<bottom>";
276         } else {
277             StringBuilder buf = new StringBuilder();
278             buf.append("N");
279             for (int i = 0; i < depth; ++i) {
280                 buf.append(get(i) == CATCH ? "C" : "F");
281             }
282             return buf.toString();
283         }
284     }
285 
286     private void push(boolean value) {
287         set(depth, value);
288         ++depth;
289     }
290 }

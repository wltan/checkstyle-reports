24  import java.nio.file.Path;
25  import java.nio.file.Paths;
26  import java.util.Arrays;
27  import java.util.List;
28  
29  import jdk.test.lib.compiler.CompilerUtils;
30  import static jdk.testlibrary.ProcessTools.executeTestJava;
31  
32  import org.testng.annotations.BeforeTest;
33  import org.testng.annotations.Test;
34  import static org.testng.Assert.*;
35  
36  /**
37   * @test
38   * @library /lib/testlibrary /test/lib
39   * @modules jdk.compiler
40   * @build ProxyTest q.U jdk.testlibrary.*
41   *        jdk.test.lib.compiler.CompilerUtils
42   * @run testng ProxyTest
43   * @summary Driver for testing proxies accessing interfaces in named modules
44   */
45  
46  public class ProxyTest {
47  
48      private static final String TEST_SRC = System.getProperty("test.src");
49      private static final String TEST_CLASSES = System.getProperty("test.classes");
50  
51      private static final Path SRC_DIR = Paths.get(TEST_SRC, "src");
52      private static final Path MODS_DIR = Paths.get("mods");
53      private static final Path CPATH_DIR = Paths.get(TEST_CLASSES);
54  
55      // the names of the modules in this test
56      private static List<String> modules = Arrays.asList("test", "m1", "m2", "m3");
57  
58  
59      /**
60       * Compiles all modules used by the test
61       */
62      @BeforeTest
63      public void compileAll() throws Exception {
64          for (String mn : modules) {
65              Path msrc = SRC_DIR.resolve(mn);
66              assertTrue(CompilerUtils.compile(msrc, MODS_DIR, "--module-source-path", SRC_DIR.toString()));
67          }
68      }
69  
70      /**
71       * Run the modular test
72       */
73      @Test
74      public void runTest() throws Exception {
75          int exitValue = executeTestJava("-cp", CPATH_DIR.toString(),
76                                          "--module-path", MODS_DIR.toString(),
77                                          "-m", "test/jdk.test.Main")
78                              .outputTo(System.out)
79                              .errorTo(System.out)
80                              .getExitValue();
81  
82          assertTrue(exitValue == 0);
83      }
84  }

24  /*
25   * @test
26   * @bug 6824477
27   * @summary Selector.select can fail with IOException "Invalid argument" on
28   *     Solaris if maximum number of file descriptors is less than 10000
29   * @requires (os.family != "windows")
30   * @library /test/lib
31   * @build jdk.test.lib.Utils
32   *        jdk.test.lib.Asserts
33   *        jdk.test.lib.JDKToolFinder
34   *        jdk.test.lib.JDKToolLauncher
35   *        jdk.test.lib.Platform
36   *        jdk.test.lib.process.*
37   *        LotsOfUpdates
38   * @run main LotsOfUpdatesTest
39   */
40  
41  import jdk.test.lib.process.ProcessTools;
42  
43  public class LotsOfUpdatesTest {
44  
45      //hard limit needs to be less than 10000 for this bug
46      private static final String ULIMIT_SET_CMD = "ulimit -n 2048";
47  
48      private static final String JAVA_CMD = ProcessTools.getCommandLine(
49              ProcessTools.createJavaProcessBuilder(LotsOfUpdates.class.getName()));
50  
51      public static void main(String[] args) throws Throwable {
52          ProcessTools.executeCommand("sh", "-c", ULIMIT_SET_CMD + " && " + JAVA_CMD)
53                      .shouldHaveExitValue(0);
54      }
55  }

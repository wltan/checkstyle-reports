57  	/**
58  	 * If {@code value} is an enum constant of type {@code enumClass}, then the enum value will be returned.
59  	 * {@code null} will be returned if {@code value} is an actual null literal.
60  	 * 
61  	 * @throws ResolutionException If {@code value} cannot be converted.
62  	 */
63  	public <E extends Enum<E>> E resolveEnum(Class<E> enumClass, Node value) {
64  		// case1: null
65  		if (value instanceof NullLiteral) return null;
66  		
67  		// case2: Identifier
68  		String enumName = null;
69  		
70  		if (value instanceof Identifier) {
71  			enumName = ((Identifier)value).astValue();
72  		}
73  		
74  		// case3: EnumSimpleName.Identifier or EnumFQN.Identifier
75  		String typeName = null;
76  		String packageName = null;
77  		if (value instanceof Select) {
78  			List<String> chain = unwrapSelectChain((Select) value);
79  			switch (chain.size()) {
80  			case 0:
81  				throw new ResolutionException(value, "empty");
82  			default:
83  				packageName = Joiner.on('.').join(chain.subList(0, chain.size() - 2));
84  			case 2:
85  				typeName = chain.get(chain.size() - 2);
86  			case 1:
87  				enumName = chain.get(chain.size() - 1);
88  			}
89  			
90  			boolean unexpectedType = false;
91  			if (packageName != null) {
92  				Package p = enumClass.getPackage();
93  				unexpectedType = p != null && !p.getName().equals(packageName);
94  			}
95  			unexpectedType |= (typeName != null && !enumClass.getSimpleName().equals(typeName));
96  			
97  			if (unexpectedType) throw new ResolutionException(value, "Expected " + enumClass.getName() + " and not " + packageName + "." + typeName);
98  		}
99  		
100 		for (E enumConstant : enumClass.getEnumConstants()) {
101 			String target = enumConstant.name();
102 			if (target.equals(enumName)) return enumConstant;
103 		}
104 		
105 		throw new ResolutionException(value, "Not a valid value for enum " + enumClass.getSimpleName() + ": " + enumName);
106 	}

121 	private static class ImportList {
122 		final List<String> explicits = new ArrayList<String>();
123 		final List<String> stars = new ArrayList<String>();
124 		
125 		ImportList() {
126 			stars.add("java.lang");
127 		}
128 	}
129 	
130 	private ImportList getImportList(Node n) {
131 		ImportList il = new ImportList();
132 		
133 		while (n != null) {
134 			if (n instanceof CompilationUnit) {
135 				CompilationUnit cu = (CompilationUnit) n;
136 				PackageDeclaration pkg = cu.astPackageDeclaration();
137 				if (pkg != null) il.stars.add(pkg.getPackageName());
138 				for (ImportDeclaration imp : cu.astImportDeclarations()) {
139 					if (imp.astStaticImport()) continue;
140 					if (imp.astStarImport()) {
141 						String i = imp.asFullyQualifiedName();
142 						i = i.substring(0, i.length() - 2);
143 						il.stars.add(i);
144 					} else {
145 						il.explicits.add(imp.asFullyQualifiedName());
146 					}
147 				}
148 			}
149 			
150 			n = n.getParent();
151 		}
152 		
153 		return il;
154 	}

156 	/**
157 	 * Checks if the given {@code typeReference} could legally be referring to the listed fully qualified {@code typeName}.
158 	 * Will check import statements, and checks for any shadowing by types in this file with the same name.
159 	 * <p>
160 	 * <em>WARNING:</em> This method will return {@code true} if there's a star import of i.e. {@code java.util.*},
161 	 * you ask if {@code ArrayList} could be referring to {@code java.util.ArrayList}, and there is another class in the same package also named
162 	 * {@code ArrayList}. The right answer is of course {@code false}, but without a classpath/sourcepath this cannot be determined. Therefore,
163 	 * this method is 99% but not 100% accurate. Also, god kills a puppy everytime you write a star import.
164 	 * <p>
165 	 * <em>NB:</em> Any type arguments (generics) on either the {@code typeReference} or {@code wanted} are stripped off before comparing the two.
166 	 * <p>
167 	 * <em>NB:</em> If you want array types, just put array brackets at the end of your {@code wanted} string, for example: "{@code java.lang.String[]}".
168 	 * 
169 	 * @param wanted A fully qualified type name, such as {@code java.lang.String}.
170 	 * @param typeReference A type reference node.
171 	 * @return {@code true} if the typeReference is most likely a reference to {@code wanted}.
172 	 */
173 	public boolean typesMatch(String wanted, TypeReference typeReference) {
174 		String name = stripGenerics(typeReference.getTypeName());
175 		wanted = stripGenerics(wanted);
176 		
177 		if (name.equals(wanted)) return true;
178 		
179 		/* check array dimensions */ {
180 			int dims1 = typeReference.astArrayDimensions();
181 			int dims2 = 0;
182 			while (wanted.endsWith("[]")) {
183 				dims2++;
184 				wanted = wanted.substring(0, wanted.length() - 2);
185 			}
186 			if (dims1 != dims2) return false;
187 			if (dims1 > 0) name = name.substring(0, name.length() - dims1 * 2);
188 		}
189 		
190 		int dot = wanted.lastIndexOf('.');
191 		String wantedPkg = dot == -1 ? "" : wanted.substring(0, dot);
192 		String wantedName = dot == -1 ? wanted : wanted.substring(dot + 1);
193 		
194 		// If type ref appears fully qualified and it hasn't matched by now, we're done.
195 		if (name.indexOf('.') > -1) return false;
196 		
197 		// 'Baz' will never be a match for foo.bar.NotBaz.
198 		if (!wantedName.equals(name)) return false;
199 		
200 		ImportList imports = getImportList(typeReference);
201 		
202 		/* If matching List to java.util.List and java.awt.List is explicitly imported, no match. */ {
203 			String ending = "." + wantedName;
204 			for (String explicit : imports.explicits) {
205 				if (explicit.endsWith(ending) && !explicit.equals(wanted)) return false;
206 			}
207 		}
208 		
209 		potentialMatchFound:
210 		/* To match List to java.util.List either there has to be a java.util.* or java.util.List import. */ {
211 			if (wantedPkg.length() > 0 && imports.stars.contains(wantedPkg)) break potentialMatchFound;
212 			if (imports.explicits.contains(wanted)) break potentialMatchFound;
213 			return false;
214 		}
215 		
216 		//name is definitely a simple name, and it might match. Walk up type tree and if it doesn't match any of those, delve into import statements.
217 		Node n = typeReference.getParent();
218 		Node prevN = null;
219 		CompilationUnit cu = null;
220 		while (n != null) {
221 			RawListAccessor<?, ?> list;
222 			boolean stopAtSelf;
223 			
224 			if (n instanceof Block) {
225 				list = ((Block) n).rawContents();
226 				stopAtSelf = true;
227 			} else if (n instanceof TypeBody) {
228 				list = ((TypeBody) n).rawMembers();
229 				stopAtSelf = false;
230 			} else if (n instanceof CompilationUnit) {
231 				list = ((CompilationUnit) n).rawTypeDeclarations();
232 				cu = (CompilationUnit) n;
233 				stopAtSelf = false;
234 			} else {
235 				list = null;
236 				stopAtSelf = false;
237 			}
238 			
239 			if (list != null) {
240 				for (Node c : list) {
241 					if (c instanceof TypeDeclaration && namesMatch(name, ((TypeDeclaration) c).astName())) return false;
242 					if (stopAtSelf && c == prevN) break;
243 				}
244 			}
245 			
246 			prevN = n;
247 			n = n.getParent();
248 		}
249 		
250 		//A locally defined type is definitely not what's targeted so it could still be our wanted type reference. Let's check imports.
251 		if (wantedPkg.isEmpty()) return cu == null || cu.rawPackageDeclaration() == null;
252 		
253 		return true;
254 	}

256 	private String stripGenerics(String name) {
257 		int genericsStart = name.indexOf('<');
258 		int genericsEnd = name.lastIndexOf('>');
259 		if (genericsStart != -1 && genericsEnd == name.length() -1) name = name.substring(0, genericsStart);
260 		return name;
261 	}
262 	
263 	private boolean namesMatch(String name, Identifier astName) {
264 		return name == null ? astName.astValue() == null : name.equals(astName.astValue());
265 	}

267 	/**
268 	 * Turns an annotation AST node into an actual instance of an annotation class provided you already know its type.
269 	 * <strong>NB: non-literal compile-time constants cannot be converted, and you should avoid querying classes;
270 	 * instead call {@code resolver.getAnnotationClassAsString(objectReturnedByThisMethod, "annotation method name")}.
271 	 * 
272 	 * @see #getAnnotationClassesAsStrings(java.lang.annotation.Annotation, String)
273 	 * @see #getAnnotationClassAsString(java.lang.annotation.Annotation, String)
274 	 */
275 	public <A extends java.lang.annotation.Annotation> A toAnnotationInstance(Class<A> type, Annotation node) {
276 		return type.cast(Proxy.newProxyInstance(type.getClassLoader(), new Class[] { type }, new AnnotationProxy(this, node)));
277 	}

279 	/**
280 	 * Use together with {@link #toAnnotationInstance(Class, Annotation)}.
281 	 * 
282 	 * @see #getAnnotationClassesAsStrings(java.lang.annotation.Annotation, String)
283 	 */
284 	public String getAnnotationClassAsString(java.lang.annotation.Annotation annotation, String methodName) {
285 		try {
286 			Method m = annotation.getClass().getMethod(methodName);
287 			if (m.getReturnType() != Class.class) throw new IllegalArgumentException("Method " + methodName + " does not have 'Class' as return type");
288 			try {
289 				return Class.class.cast(m.invoke(annotation)).toString();
290 			} catch (AnnotationClassNotAvailableException e) {
291 				return e.getClassName();
292 			}
293 		} catch (IllegalAccessException e) {
294 			throw new IllegalArgumentException("Method " + methodName + " isn't accessible", e);
295 		} catch (InvocationTargetException e) {
296 			throw new IllegalArgumentException("Method " + methodName + " cannot be invoked", e);
297 		} catch (NoSuchMethodException e) {
298 			throw new IllegalArgumentException("Method " + methodName + " does not exist");
299 		}
300 	}

302 	/**
303 	 * Use together with {@link #toAnnotationInstance(Class, Annotation)}.
304 	 * 
305 	 * @see #getAnnotationClassAsString(java.lang.annotation.Annotation, String)
306 	 */
307 	public List<String> getAnnotationClassesAsStrings(java.lang.annotation.Annotation annotation, String methodName) {
308 		try {
309 			Method m = annotation.getClass().getMethod(methodName);
310 			boolean array;
311 			if (m.getReturnType() == Class.class) array = false;
312 			else if (m.getReturnType() == Class[].class) array = true;
313 			else throw new IllegalArgumentException("Method " + methodName + " does not have 'Class' or 'Class[]' as return type");
314 			
315 			try {
316 				Class<?>[] cs;
317 				if (array) {
318 					cs = Class[].class.cast(m.invoke(annotation));
319 				} else {
320 					cs = new Class[1];
321 					cs[0] = Class.class.cast(m.invoke(annotation));
322 				}
323 				
324 				List<String> result = Lists.newArrayList();
325 				for (Class<?> c : cs) result.add(c.getName());
326 				return result;
327 			} catch (AnnotationClassNotAvailableException e) {
328 				return e.getClassNames();
329 			}
330 		} catch (IllegalAccessException e) {
331 			throw new IllegalArgumentException("Method " + methodName + " isn't accessible", e);
332 		} catch (InvocationTargetException e) {
333 			throw new IllegalArgumentException("Method " + methodName + " cannot be invoked", e);
334 		} catch (NoSuchMethodException e) {
335 			throw new IllegalArgumentException("Method " + methodName + " does not exist");
336 		}
337 	}
338 	
339 	private List<String> unwrapSelectChain(Select s) {
340 		List<String> list = Lists.newArrayList();
341 		while (s != null) {
342 			list.add(s.astIdentifier().astValue());
343 			Expression parent = s.astOperand();
344 			if (parent instanceof Select) {
345 				s = (Select) parent;
346 			} else if (parent instanceof Identifier) {
347 				s = null;
348 				list.add(((Identifier)parent).astValue());
349 			} else if (parent == null) {
350 				break;
351 			} else {
352 				throw new ResolutionException(parent, "Identifies expected here, not a " + parent.getClass().getSimpleName());
353 			}
354 		}
355 		
356 		Collections.reverse(list);
357 		return list;
358 	}

111 	private void doResultTransformerTest(ResultTransformer transformer, boolean isSingleton) {
112 		Map transformerMap = new HashMap();
113 
114 		transformerMap.put( transformer, "" );
115 		assert transformerMap.size() == 1 : "really messed up";
116 		Object old = transformerMap.put( transformer, "value" );
117 		assert old != null && transformerMap.size() == 1 : "apparent QueryKey equals/hashCode issue";
118 
119 		// finally, lets serialize it and see what happens
120 		ResultTransformer transformer2 = ( ResultTransformer ) SerializationHelper.clone( transformer );
121 		old = transformerMap.put( transformer2, "new value" );
122 		assert old != null && transformerMap.size() == 1 : "deserialization did not set hashCode or equals properly";
123 		if ( isSingleton ) {
124 			assert transformer == transformer2: "deserialization issue for singleton transformer";
125 		}
126 		else {
127 			assert transformer != transformer2: "deserialization issue for non-singleton transformer";
128 		}
129 		assert transformer.equals( transformer2 ): "deep copy issue";
130 	}

228 	private QueryKey buildBasicKey(CacheableResultTransformer resultTransformer) {
229 		return new QueryKey(
230 				QUERY_STRING,
231 				ArrayHelper.EMPTY_TYPE_ARRAY, 		// positional param types
232 				ArrayHelper.EMPTY_OBJECT_ARRAY,		// positional param values
233 				Collections.EMPTY_MAP,				// named params
234 				null,								// firstRow selection
235 				null,								// maxRows selection
236 				Collections.EMPTY_SET, 				// filter keys
237 				null,								// tenantIdentifier
238 				resultTransformer					// the result transformer
239 		);
240 	}

242 	private void doTest(QueryKey key) {
243 		Map keyMap = new HashMap();
244 		Map transformerMap = new HashMap();
245 
246 		keyMap.put( key, "" );
247 		assert keyMap.size() == 1 : "really messed up";
248 		Object old = keyMap.put( key, "value" );
249 		assert old != null && keyMap.size() == 1 : "apparent QueryKey equals/hashCode issue";
250 
251 		if ( key.getResultTransformer() != null ) {
252 			transformerMap.put( key.getResultTransformer(), "" );
253 			assert transformerMap.size() == 1 : "really messed up";
254 			old = transformerMap.put( key.getResultTransformer(), "value" );
255 			assert old != null && transformerMap.size() == 1 : "apparent QueryKey equals/hashCode issue";
256 		}
257 
258 		// finally, lets serialize it and see what happens
259 		QueryKey key2 = ( QueryKey ) SerializationHelper.clone( key );
260 		assert key != key2 : "deep copy issue";
261 		old = keyMap.put( key2, "new value" );
262 		assert old != null && keyMap.size() == 1 : "deserialization did not set hashCode or equals properly";
263 		if ( key.getResultTransformer() == null ) {
264 			assert key2.getResultTransformer() == null;
265 		}
266 		else {
267 			old = transformerMap.put( key2.getResultTransformer(), "new value" );
268 			assert old != null && transformerMap.size() == 1 : "deserialization did not set hashCode or equals properly";
269 				assert key.getResultTransformer() != key2.getResultTransformer(): "deserialization issue for non-singleton transformer";
270 				assert key.getResultTransformer().equals( key2.getResultTransformer() ): "deep copy issue";
271 		}
272 	}

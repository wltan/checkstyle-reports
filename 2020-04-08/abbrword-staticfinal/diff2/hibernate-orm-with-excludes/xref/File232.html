24  package org.hibernate.proxy.pojo.javassist;
25  import java.io.Serializable;
26  import java.lang.reflect.Method;
27  import java.util.Set;
28  
29  import org.hibernate.HibernateException;
30  import org.hibernate.engine.spi.SessionImplementor;
31  import org.hibernate.internal.util.ReflectHelper;
32  import org.hibernate.proxy.HibernateProxy;
33  import org.hibernate.proxy.ProxyFactory;
34  import org.hibernate.type.CompositeType;
35  
36  /**
37   * A {@link ProxyFactory} implementation for producing Javassist-based proxies.
38   *
39   * @author Muga Nishizawa
40   */
41  public class JavassistProxyFactory implements ProxyFactory, Serializable {
42  
43  	protected static final Class[] NO_CLASSES = new Class[0];
44  	private Class persistentClass;
45  	private String entityName;
46  	private Class[] interfaces;
47  	private Method getIdentifierMethod;
48  	private Method setIdentifierMethod;
49  	private CompositeType componentIdType;
50  	private Class factory;
51  	private boolean overridesEquals;
52  
53  	public void postInstantiate(
54  			final String entityName,
55  			final Class persistentClass,
56  			final Set interfaces,
57  			final Method getIdentifierMethod,
58  			final Method setIdentifierMethod,
59  			CompositeType componentIdType) throws HibernateException {
60  		this.entityName = entityName;
61  		this.persistentClass = persistentClass;
62  		this.interfaces = (Class[]) interfaces.toArray(NO_CLASSES);
63  		this.getIdentifierMethod = getIdentifierMethod;
64  		this.setIdentifierMethod = setIdentifierMethod;
65  		this.componentIdType = componentIdType;
66  		factory = JavassistLazyInitializer.getProxyFactory( persistentClass, this.interfaces );
67  		overridesEquals = ReflectHelper.overridesEquals(persistentClass);
68  	}
69  
70  	public HibernateProxy getProxy(
71  			Serializable id,
72  			SessionImplementor session) throws HibernateException {
73  		return JavassistLazyInitializer.getProxy(
74  				factory,
75  				entityName,
76  				persistentClass,
77  				interfaces,
78  				getIdentifierMethod,
79  				setIdentifierMethod,
80  				componentIdType,
81  				id,
82  				session,
83  				overridesEquals
84  		);
85  	}
86  
87  }

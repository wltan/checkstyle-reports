89      @Test
90      public void testZipFileReadsUnicodeFields() throws IOException {
91          File file = File.createTempFile("unicode-test", ".zip");
92          ZipFile zf = null;
93          try {
94              createTestFile(file, US_ASCII, false, true);
95              zf = new ZipFile(file, US_ASCII, true);
96              assertCanRead(zf, ASCII_TXT);
97              assertCanRead(zf, EURO_FOR_DOLLAR_TXT);
98              assertCanRead(zf, OIL_BARREL_TXT);
99          } finally {
100             ZipFile.closeQuietly(zf);
101             if (file.exists()) {
102                 file.delete();
103             }
104         }
105     }

107     private static void testFileRoundtrip(String encoding, boolean withEFS,
108                                           boolean withExplicitUnicodeExtra)
109         throws IOException {
110 
111         File file = File.createTempFile(encoding + "-test", ".zip");
112         try {
113             createTestFile(file, encoding, withEFS, withExplicitUnicodeExtra);
114             testFile(file, encoding);
115         } finally {
116             if (file.exists()) {
117                 file.delete();
118             }
119         }
120     }
121 
122     private static void createTestFile(File file, String encoding,
123                                        boolean withEFS,
124                                        boolean withExplicitUnicodeExtra)
125         throws IOException {
126 
127         ZipEncoding zipEncoding = ZipEncodingHelper.getZipEncoding(encoding);
128 
129         ZipOutputStream zos = null;
130         try {
131             zos = new ZipOutputStream(file);
132             zos.setEncoding(encoding);
133             zos.setUseLanguageEncodingFlag(withEFS);
134             zos.setCreateUnicodeExtraFields(withExplicitUnicodeExtra ? 
135                                             ZipOutputStream
136                                             .UnicodeExtraFieldPolicy.NEVER
137                                             : ZipOutputStream
138                                             .UnicodeExtraFieldPolicy.ALWAYS);
139 
140             ZipEntry ze = new ZipEntry(OIL_BARREL_TXT);
141             if (withExplicitUnicodeExtra
142                 && !zipEncoding.canEncode(ze.getName())) {
143 
144                 ByteBuffer en = zipEncoding.encode(ze.getName());
145 
146                 ze.addExtraField(new UnicodePathExtraField(ze.getName(),
147                                                            en.array(),
148                                                            en.arrayOffset(),
149                                                            en.limit()));
150             }
151 
152             zos.putNextEntry(ze);
153             zos.write("Hello, world!".getBytes("US-ASCII"));
154             zos.closeEntry();
155 
156             ze = new ZipEntry(EURO_FOR_DOLLAR_TXT);
157             if (withExplicitUnicodeExtra
158                 && !zipEncoding.canEncode(ze.getName())) {
159 
160                 ByteBuffer en = zipEncoding.encode(ze.getName());
161 
162                 ze.addExtraField(new UnicodePathExtraField(ze.getName(),
163                                                            en.array(),
164                                                            en.arrayOffset(),
165                                                            en.limit()));
166             }
167 
168             zos.putNextEntry(ze);
169             zos.write("Give me your money!".getBytes("US-ASCII"));
170             zos.closeEntry();
171 
172             ze = new ZipEntry(ASCII_TXT);
173 
174             if (withExplicitUnicodeExtra
175                 && !zipEncoding.canEncode(ze.getName())) {
176 
177                 ByteBuffer en = zipEncoding.encode(ze.getName());
178 
179                 ze.addExtraField(new UnicodePathExtraField(ze.getName(),
180                                                            en.array(),
181                                                            en.arrayOffset(),
182                                                            en.limit()));
183             }
184 
185             zos.putNextEntry(ze);
186             zos.write("ascii".getBytes("US-ASCII"));
187             zos.closeEntry();
188         } finally {
189             if (zos != null) {
190                 try {
191                     zos.close();
192                 } catch (IOException e) { /* swallow */ }
193             }
194         }
195     }

197     private static void testFile(File file, String encoding)
198         throws IOException {
199         ZipFile zf = null;
200         try {
201             zf = new ZipFile(file, encoding, false);
202 
203             Enumeration e = zf.getEntries();
204             while (e.hasMoreElements()) {
205                 ZipEntry ze = (ZipEntry) e.nextElement();
206 
207                 if (ze.getName().endsWith("sser.txt")) {
208                     assertUnicodeName(ze, OIL_BARREL_TXT, encoding);
209 
210                 } else if (ze.getName().endsWith("_for_Dollar.txt")) {
211                     assertUnicodeName(ze, EURO_FOR_DOLLAR_TXT, encoding);
212                 } else if (!ze.getName().equals(ASCII_TXT)) {
213                     throw new AssertionError("Urecognized ZIP entry with name ["
214                                              + ze.getName() + "] found.");
215                 }
216             }
217         } finally {
218             ZipFile.closeQuietly(zf);
219         }
220     }
221 
222     private static UnicodePathExtraField findUniCodePath(ZipEntry ze) {
223         return (UnicodePathExtraField)
224             ze.getExtraField(UnicodePathExtraField.UPATH_ID);
225     }
226 
227     private static void assertUnicodeName(ZipEntry ze,
228                                           String expectedName,
229                                           String encoding)
230         throws IOException {
231         if (!expectedName.equals(ze.getName())) {
232             UnicodePathExtraField ucpf = findUniCodePath(ze);
233             assertNotNull(ucpf);
234 
235             ZipEncoding enc = ZipEncodingHelper.getZipEncoding(encoding);
236             ByteBuffer ne = enc.encode(ze.getName());
237 
238             CRC32 crc = new CRC32();
239             crc.update(ne.array(),ne.arrayOffset(),ne.limit());
240 
241             assertEquals(crc.getValue(), ucpf.getNameCRC32());
242             assertEquals(expectedName, new String(ucpf.getUnicodeName(),
243                                                   UTF_8));
244         }
245     }
246 
247     private static void assertCanRead(ZipFile zf, String fileName) throws IOException {
248         ZipEntry entry = zf.getEntry(fileName);
249         assertNotNull("Entry " + fileName + " doesn't exist", entry);
250         InputStream is = zf.getInputStream(entry);
251         assertNotNull("InputStream is null", is);
252         try {
253             is.read();
254         } finally {
255             is.close();
256         }
257     }

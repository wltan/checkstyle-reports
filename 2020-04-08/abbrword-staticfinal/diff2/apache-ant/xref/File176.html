158     /**
159      * Return an array of <code>Package Specification</code> objects.
160      * If there are no such optional packages, a zero-length array is returned.
161      *
162      * @param manifest Manifest to be parsed
163      * @return the Package Specifications extensions in specified manifest
164      * @throws ParseException if the attributes of the specifications cannot
165      * be parsed according to their expected formats.
166      */
167     public static Specification[] getSpecifications(final Manifest manifest)
168         throws ParseException {
169         if (null == manifest) {
170             return new Specification[ 0 ];
171         }
172 
173         final ArrayList results = new ArrayList();
174 
175         final Map entries = manifest.getEntries();
176         final Iterator keys = entries.keySet().iterator();
177         while (keys.hasNext()) {
178             final String key = (String) keys.next();
179             final Attributes attributes = (Attributes) entries.get(key);
180             final Specification specification
181                 = getSpecification(key, attributes);
182             if (null != specification) {
183                 results.add(specification);
184             }
185         }
186 
187         final ArrayList trimmedResults = removeDuplicates(results);
188         return (Specification[]) trimmedResults.toArray(new Specification[trimmedResults.size()]);
189     }

214     /**
215      * The constructor to create Package Specification object.
216      * Note that every component is allowed to be specified
217      * but only the specificationTitle is mandatory.
218      *
219      * @param specificationTitle the name of specification.
220      * @param specificationVersion the specification Version.
221      * @param specificationVendor the specification Vendor.
222      * @param implementationTitle the title of implementation.
223      * @param implementationVersion the implementation Version.
224      * @param implementationVendor the implementation Vendor.
225      * @param sections the sections/packages that Specification applies to.
226      */
227     public Specification(final String specificationTitle,
228                           final String specificationVersion,
229                           final String specificationVendor,
230                           final String implementationTitle,
231                           final String implementationVersion,
232                           final String implementationVendor,
233                           final String[] sections) {
234         this.specificationTitle = specificationTitle;
235         this.specificationVendor = specificationVendor;
236 
237         if (null != specificationVersion) {
238             try {
239                 this.specificationVersion
240                     = new DeweyDecimal(specificationVersion);
241             } catch (final NumberFormatException nfe) {
242                 final String error = "Bad specification version format '"
243                     + specificationVersion + "' in '" + specificationTitle
244                     + "'. (Reason: " + nfe + ")";
245                 throw new IllegalArgumentException(error);
246             }
247         }
248 
249         this.implementationTitle = implementationTitle;
250         this.implementationVendor = implementationVendor;
251         this.implementationVersion = implementationVersion;
252 
253         if (null == this.specificationTitle) {
254             throw new NullPointerException("specificationTitle");
255         }
256 
257         String[] copy = null;
258         if (null != sections) {
259             copy = new String[ sections.length ];
260             System.arraycopy(sections, 0, copy, 0, sections.length);
261         }
262         this.sections = copy;
263     }

335     /**
336      * Return a Compatibility enum indicating the relationship of this
337      * <code>Package Specification</code> with the specified
338      * <code>Extension</code>.
339      *
340      * @param other the other specification
341      * @return the enum indicating the compatibility (or lack thereof)
342      *         of specified Package Specification
343      */
344     public Compatibility getCompatibilityWith(final Specification other) {
345         // Specification Name must match
346         if (!specificationTitle.equals(other.getSpecificationTitle())) {
347             return INCOMPATIBLE;
348         }
349 
350         // Available specification version must be >= required
351         final DeweyDecimal otherSpecificationVersion
352             = other.getSpecificationVersion();
353         if (null != specificationVersion) {
354             if (null == otherSpecificationVersion
355                 || !isCompatible(specificationVersion, otherSpecificationVersion)) {
356                 return REQUIRE_SPECIFICATION_UPGRADE;
357             }
358         }
359 
360         // Implementation Vendor ID must match
361         final String otherImplementationVendor
362             = other.getImplementationVendor();
363         if (null != implementationVendor) {
364             if (null == otherImplementationVendor
365                 || !implementationVendor.equals(otherImplementationVendor)) {
366                 return REQUIRE_VENDOR_SWITCH;
367             }
368         }
369 
370         // Implementation version must be >= required
371         final String otherImplementationVersion
372             = other.getImplementationVersion();
373         if (null != implementationVersion) {
374             if (null == otherImplementationVersion
375                 || !implementationVersion.equals(otherImplementationVersion)) {
376                 return REQUIRE_IMPLEMENTATION_CHANGE;
377             }
378         }
379 
380         // This available optional package satisfies the requirements
381         return COMPATIBLE;
382     }

384     /**
385      * Return <code>true</code> if the specified <code>package</code>
386      * is satisfied by this <code>Specification</code>. Otherwise, return
387      * <code>false</code>.
388      *
389      * @param other the specification
390      * @return true if the specification is compatible with this specification
391      */
392     public boolean isCompatibleWith(final Specification other) {
393         return (COMPATIBLE == getCompatibilityWith(other));
394     }

396     /**
397      * Return a String representation of this object.
398      *
399      * @return string representation of object.
400      */
401     public String toString() {
402         final String brace = ": ";
403 
404         final StringBuffer sb
405             = new StringBuffer(SPECIFICATION_TITLE.toString());
406         sb.append(brace);
407         sb.append(specificationTitle);
408         sb.append(StringUtils.LINE_SEP);
409 
410         if (null != specificationVersion) {
411             sb.append(SPECIFICATION_VERSION);
412             sb.append(brace);
413             sb.append(specificationVersion);
414             sb.append(StringUtils.LINE_SEP);
415         }
416 
417         if (null != specificationVendor) {
418             sb.append(SPECIFICATION_VENDOR);
419             sb.append(brace);
420             sb.append(specificationVendor);
421             sb.append(StringUtils.LINE_SEP);
422         }
423 
424         if (null != implementationTitle) {
425             sb.append(IMPLEMENTATION_TITLE);
426             sb.append(brace);
427             sb.append(implementationTitle);
428             sb.append(StringUtils.LINE_SEP);
429         }
430 
431         if (null != implementationVersion) {
432             sb.append(IMPLEMENTATION_VERSION);
433             sb.append(brace);
434             sb.append(implementationVersion);
435             sb.append(StringUtils.LINE_SEP);
436         }
437 
438         if (null != implementationVendor) {
439             sb.append(IMPLEMENTATION_VENDOR);
440             sb.append(brace);
441             sb.append(implementationVendor);
442             sb.append(StringUtils.LINE_SEP);
443         }
444 
445         return sb.toString();
446     }

448     /**
449      * Return <code>true</code> if the first version number is greater than
450      * or equal to the second; otherwise return <code>false</code>.
451      *
452      * @param first First version number (dotted decimal)
453      * @param second Second version number (dotted decimal)
454      */
455     private boolean isCompatible(final DeweyDecimal first,
456                                  final DeweyDecimal second) {
457         return first.isGreaterThanOrEqual(second);
458     }

460     /**
461      * Combine all specifications objects that are identical except
462      * for the sections.
463      *
464      * <p>Note this is very inefficent and should probably be fixed
465      * in the future.</p>
466      *
467      * @param list the array of results to trim
468      * @return an array list with all duplicates removed
469      */
470     private static ArrayList removeDuplicates(final ArrayList list) {
471         final ArrayList results = new ArrayList();
472         final ArrayList sections = new ArrayList();
473         while (list.size() > 0) {
474             final Specification specification = (Specification) list.remove(0);
475             final Iterator iterator = list.iterator();
476             while (iterator.hasNext()) {
477                 final Specification other = (Specification) iterator.next();
478                 if (isEqual(specification, other)) {
479                     final String[] otherSections = other.getSections();
480                     if (null != otherSections) {
481                         sections.addAll(Arrays.asList(otherSections));
482                     }
483                     iterator.remove();
484                 }
485             }
486 
487             final Specification merged =
488                 mergeInSections(specification, sections);
489             results.add(merged);
490             //Reset list of sections
491             sections.clear();
492         }
493 
494         return results;
495     }

497     /**
498      * Test if two specifications are equal except for their sections.
499      *
500      * @param specification one specificaiton
501      * @param other the ohter specification
502      * @return true if two specifications are equal except for their
503      *         sections, else false
504      */
505     private static boolean isEqual(final Specification specification,
506                                     final Specification other) {
507         return
508             specification.getSpecificationTitle().equals(other.getSpecificationTitle())
509             && specification.getSpecificationVersion().isEqual(other.getSpecificationVersion())
510             && specification.getSpecificationVendor().equals(other.getSpecificationVendor())
511             && specification.getImplementationTitle().equals(other.getImplementationTitle())
512             && specification.getImplementationVersion().equals(other.getImplementationVersion())
513             && specification.getImplementationVendor().equals(other.getImplementationVendor());
514     }

516     /**
517      * Merge the specified sections into specified section and return result.
518      * If no sections to be added then just return original specification.
519      *
520      * @param specification the specification
521      * @param sectionsToAdd the list of sections to merge
522      * @return the merged specification
523      */
524     private static Specification mergeInSections(final Specification specification,
525                                               final ArrayList sectionsToAdd) {
526         if (0 == sectionsToAdd.size()) {
527             return specification;
528         }
529         sectionsToAdd.addAll(Arrays.asList(specification.getSections()));
530 
531         final String[] sections =
532             (String[]) sectionsToAdd.toArray(new String[sectionsToAdd.size()]);
533 
534         return new Specification(specification.getSpecificationTitle(),
535                 specification.getSpecificationVersion().toString(),
536                 specification.getSpecificationVendor(),
537                 specification.getImplementationTitle(),
538                 specification.getImplementationVersion(),
539                 specification.getImplementationVendor(),
540                 sections);
541     }

544      * Trim the supplied string if the string is non-null
545      *
546      * @param value the string to trim or null
547      * @return the trimmed string or null
548      */
549     private static String getTrimmedString(final String value) {
550         return value == null ? null : value.trim();
551     }

553     /**
554      * Extract an Package Specification from Attributes.
555      *
556      * @param attributes Attributes to searched
557      * @return the new Specification object, or null
558      */
559     private static Specification getSpecification(final String section,
560                                                    final Attributes attributes)
561         throws ParseException {
562         //WARNING: We trim the values of all the attributes because
563         //Some extension declarations are badly defined (ie have spaces
564         //after version or vendor)
565         final String name
566             = getTrimmedString(attributes.getValue(SPECIFICATION_TITLE));
567         if (null == name) {
568             return null;
569         }
570 
571         final String specVendor
572             = getTrimmedString(attributes.getValue(SPECIFICATION_VENDOR));
573         if (null == specVendor) {
574             throw new ParseException(MISSING + SPECIFICATION_VENDOR, 0);
575         }
576 
577         final String specVersion
578             = getTrimmedString(attributes.getValue(SPECIFICATION_VERSION));
579         if (null == specVersion) {
580             throw new ParseException(MISSING + SPECIFICATION_VERSION, 0);
581         }
582 
583         final String impTitle
584             = getTrimmedString(attributes.getValue(IMPLEMENTATION_TITLE));
585         if (null == impTitle) {
586             throw new ParseException(MISSING + IMPLEMENTATION_TITLE, 0);
587         }
588 
589         final String impVersion
590             = getTrimmedString(attributes.getValue(IMPLEMENTATION_VERSION));
591         if (null == impVersion) {
592             throw new ParseException(MISSING + IMPLEMENTATION_VERSION, 0);
593         }
594 
595         final String impVendor
596             = getTrimmedString(attributes.getValue(IMPLEMENTATION_VENDOR));
597         if (null == impVendor) {
598             throw new ParseException(MISSING + IMPLEMENTATION_VENDOR, 0);
599         }
600 
601         return new Specification(name, specVersion, specVendor,
602                                   impTitle, impVersion, impVendor,
603                                   new String[]{section});
604     }

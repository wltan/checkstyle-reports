21  public class Euler17Test {
22      /**
23       * <strong>Problem 17: Number letter counts</strong>
24       * <p>
25       * If the numbers 1 to 5 are written out in words: one, two, three, four,
26       * five, then there are 3 + 3 + 5 + 4 + 4 = 19 letters used in total.
27       * <p>
28       * If all the numbers from 1 to 1000 (one thousand) inclusive were written
29       * out in words, how many letters would be used?
30       * <p>
31       * NOTE: Do not count spaces or hyphens. For example, 342 (three hundred and
32       * forty-two) contains 23 letters and 115 (one hundred and fifteen) contains
33       * 20 letters. The use of "and" when writing out numbers is in compliance
34       * with British usage.
35       */
36      @Test
37      public void shouldSolveProblem17() {
38          runTestsFor(new SolutionA());
39          runTestsFor(new SolutionB());
40  
41          // Additional capability, only for more general solution B
42          assertThat(new SolutionB().letterCount(Integer.MAX_VALUE)).isEqualTo("twobilliononehundredandfortysevenmillionfourhundredandeightythreethousandsixhundredandfortyseven".length());
43      }
44  
45      private static void runTestsFor(SolutionProblem17 solution) {
46          List.of("one", "two", "three", "four", "five", "six", "seven", "eight", "nine", "ten", "eleven", "twelve", "thirteen", "fourteen", "fifteen", "sixteen", "seventeen", "eighteen", "nineteen", "twenty", "twentyone")
47                  .zipWithIndex()
48                  .forEach(t -> {
49                      final int number = t._2 + 1;
50                      final String numberAsString = t._1;
51                      assertThat(numberAsString).hasSize(solution.letterCount(number));
52                  });
53  
54          assertThat(solution.letterCount(200)).isEqualTo("twohundred".length());
55          assertThat(solution.letterCount(201)).isEqualTo("twohundredandone".length());
56  
57          assertThat(solution.letterCount(342)).isEqualTo(23);
58          assertThat(solution.letterCount(115)).isEqualTo(20);
59  
60          assertThat(solution.letterCount(rangeClosed(1, 5))).isEqualTo(19);
61          assertThat(solution.letterCount(rangeClosed(1, 1000))).isEqualTo(21124);
62      }
63  
64      private interface SolutionProblem17 {
65          int letterCount(int num);
66  
67          default int letterCount(Seq<Integer> range) {
68              return range.map(this::letterCount)
69                      .sum().intValue();
70          }
71      }

108     /**
109      * Solution using Vavr Pattern Matching.
110      */
111     private static class SolutionA implements SolutionProblem17 {
112         @Override
113         public int letterCount(int num) {
114             return Match(num).of( /*@formatter:off*/
115                     Case($(n -> n >= 1000),           n -> length(n / 1000) + length(1000) + letterCount(n % 1000)),
116                     Case($(n -> n >= 100),            n -> Match(n).of(
117                         Case($(n1 -> (n1 % 100) > 0), n1 -> length(n1 / 100) + length(100) + CONJUNCTION.length() + letterCount(n1 % 100)),
118                         Case($(),                  length(n / 100) + length(100)))),
119                     Case($(n -> n >= 20),             n -> length(n - (n % 10)) + letterCount(n % 10)),
120                     Case($(0),                        0),
121                     Case($(),                         n -> length(n))
122             ); /*@formatter:on*/
123         }
124 
125         private static int length(int number) {
126             return LENGTHS.get(number).map(String::length).get();
127         }
128     }
129 
130     /**
131      * A more general solution using functionality of the Vavr Collections.
132      */
133     private static class SolutionB implements SolutionProblem17 {
134         @Override
135         public int letterCount(int number) {
136             return asText(number).length();
137         }
138 
139         private static String asText(int number) {
140             return LENGTHS.foldRight(Tuple.of(Vector.<String> empty(), number), (magnitudeAndText, lengthsAndRemainder) -> {
141                 final int magnitude = magnitudeAndText._1;
142                 final int remainder = lengthsAndRemainder._2;
143 
144                 return ((remainder >= magnitude) && (remainder > 0)) ? asText(magnitude, magnitudeAndText._2, lengthsAndRemainder._1, remainder)
145                                                                      : lengthsAndRemainder;
146             })._1.mkString();
147         }
148 
149         private static Tuple2<Vector<String>, Integer> asText(int magnitude, String text, Vector<String> chunks, int remainder) {
150             if (remainder >= 100) {
151                 text = asText(remainder / magnitude) + text;
152                 if ((remainder < 1000) && ((remainder % magnitude) != 0)) {
153                     text += CONJUNCTION;
154                 }
155             }
156             return Tuple.of(chunks.append(text), remainder % magnitude);
157         }
158     }

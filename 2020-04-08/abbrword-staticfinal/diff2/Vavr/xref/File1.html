39          return of(includeNames, benchmarkClasses, runResults, "vavr", 0.3, 0.05);

43          return new BenchmarkPerformanceReporter(includeNames, benchmarkClasses, runResults, targetImplementation, outlierLowPct, outlierHighPct);
44      }

46      /**
47       * This class prints performance reports about the execution of individual tests, comparing their performance
48       * against other implementations as required.
49       *
50       * @param benchmarkClasses     The benchmarked source class names
51       * @param runResults           The results
52       * @param targetImplementation The target implementation we want to focus on in the Ratio report.
53       *                             It is case insensitive. If we enter "vavr", it will match "VaVr" and "va_vr".
54       * @param outlierLowPct        The percentage of samples on the lower end that will be ignored from the statistics
55       * @param outlierHighPct       The percentage of samples on the higher end that will be ignored from the statistics
56       */
57      private BenchmarkPerformanceReporter(Array<String> includeNames, Array<String> benchmarkClasses, Array<RunResult> runResults, String targetImplementation, double outlierLowPct, double outlierHighPct) {
58          this.includeNames = includeNames;
59          this.benchmarkClasses = benchmarkClasses;
60          this.runResults = runResults;
61          this.targetImplementation = targetImplementation;
62          this.outlierLowPct = outlierLowPct;
63          this.outlierHighPct = outlierHighPct;
64      }
65  
66      /**
67       * Prints all performance reports
68       */
69      public void print() {
70          printDetailedPerformanceReport();
71          printRatioPerformanceReport();
72      }

74      /**
75       * Prints the detail performance report for each individual test.
76       * <br>
77       * For each test it prints out:
78       * <ul>
79       * <li>Group</li>
80       * <li>Test Name</li>
81       * <li>Implementation - tests can have different implementations, e.g. Scala, Java, Vavr</li>
82       * <li>Parameters</li>
83       * <li>Score</li>
84       * <li>Error - 99% confidence interval expressed in % of the Score</li>
85       * <li>Unit - units for the Score</li>
86       * <li>Alternative implementations - compares performance of this test against alternative implementations</li>
87       * </ul>
88       */
89      public void printDetailedPerformanceReport() {
90          final Array<TestExecution> results = mapToTestExecutions();
91          if (results.isEmpty()) {
92              return;
93          }
94          new DetailedPerformanceReport(results).print();
95      }
96  
97      /**
98       * Prints the performance ratio report for each test, and compares the performance against different implementations
99       * of the same operation.
100      * <br>
101      * For each test it prints out:
102      * <ul>
103      * <li>Group</li>
104      * <li>Test Name</li>
105      * <li>Ratio - A/B means implementation A is compared against base implementation B</li>
106      * <li>Results - How many times faster implementation A is compared with B</li>
107      * </ul>
108      */
109     public void printRatioPerformanceReport() {
110         final Array<TestExecution> results = mapToTestExecutions();
111         if (results.isEmpty()) {
112             return;
113         }
114         new RatioPerformanceReport(results, targetImplementation).print();
115     }
116 
117     private Array<TestExecution> mapToTestExecutions() {
118         Array<TestExecution> executions = Array.empty();
119         for (RunResult runResult : runResults) {
120             executions = executions.append(TestExecution.of(runResult.getAggregatedResult(), outlierLowPct, outlierHighPct));
121         }
122         return sort(executions, includeNames);
123     }
124 
125     private Array<TestExecution> sort(Array<TestExecution> results, Array<String> includeNames) {
126         final Comparator<TestExecution> comparator = Comparator
127                 .<TestExecution, Integer> comparing(t -> benchmarkClasses.indexWhere(c -> c.endsWith(t.getOperation())))
128                 .thenComparing(t -> includeNames.indexWhere(i -> t.getImplementation().startsWith(i)));
129         return results.sorted(comparator);
130     }
131 
132     private String padLeft(String str, int size) {
133         return str + CharSeq.repeat(' ', size - str.length());
134     }
135 
136     private String padRight(String str, int size) {
137         return CharSeq.repeat(' ', size - str.length()) + str;
138     }

140     private class DetailedPerformanceReport {
141         private final Array<TestExecution> results;
142         private final Map<String, Array<TestExecution>> resultsByKey;
143         private final int paramKeySize;
144         private final int groupSize;
145         private final int nameSize;
146         private final int implSize;
147         private final int countSize;
148         private final int scoreSize;
149         private final int errorSize;
150         private final int unitSize;
151         private final Array<String> alternativeImplementations;
152 
153         public DetailedPerformanceReport(Array<TestExecution> results) {
154             this.results = results;
155             resultsByKey = results.groupBy(TestExecution::getTestNameParamKey);
156             paramKeySize = Math.max(results.map(r -> r.getParamKey().length()).max().get(), 10);
157             groupSize = Math.max(results.map(r -> r.getTarget().length()).max().get(), 10);
158             nameSize = Math.max(results.map(r -> r.getOperation().length()).max().get(), 10);
159             implSize = Math.max(results.map(r -> r.getImplementation().length()).max().get(), 10);
160             countSize = Math.max(results.map(r -> Long.toString(r.getSampleCount()).length()).max().get(), 5);
161             scoreSize = Math.max(results.map(r -> r.getScoreFormatted().length()).max().get(), 15);
162             errorSize = Math.max(results.map(r -> r.getScoreErrorPct().length()).max().get(), 10);
163             unitSize = Math.max(results.map(r -> r.getUnit().length()).max().get(), 7);
164 
165             alternativeImplementations = results.map(TestExecution::getImplementation).distinct();
166         }
167 
168         public void print() {
169             printHeader();
170             printDetails();
171         }
172 
173         private void printHeader() {
174             final String alternativeImplHeader = alternativeImplementations.map(altImpl -> padRight(altImpl, altImplColSize(altImpl))).mkString("  ");
175             final String header = String.format("%s  %s  %s  %s  %s  %s  ±%s %s  %s",
176                     padLeft("Target", groupSize),
177                     padLeft("Operation", nameSize),
178                     padLeft("Impl", implSize),
179                     padRight("Params", paramKeySize),
180                     padRight("Count", countSize),
181                     padRight("Score", scoreSize),
182                     padRight("Error", errorSize),
183                     padRight("Unit", unitSize),
184                     alternativeImplHeader
185             );
186 
187             System.out.println("\n\n\n");
188             System.out.println("Detailed Performance Execution Report");
189             System.out.println(CharSeq.of("=").repeat(header.length()));
190             System.out.println("  (Error: ±99% confidence interval, expressed as % of Score)");
191             if (outlierLowPct > 0.0 && outlierHighPct > 0.0) {
192                 System.out.println(String.format("  (Outliers removed: %s low end, %s high end)", PCT_FORMAT.format(outlierLowPct), PCT_FORMAT.format(outlierHighPct)));
193             }
194             if (!alternativeImplementations.isEmpty()) {
195                 System.out.println(String.format("  (%s: read as current row implementation is x times faster than alternative implementation)", alternativeImplementations.mkString(", ")));
196             }
197             System.out.println();
198             System.out.println(header);
199         }
200 
201         private void printDetails() {
202             for (TestExecution result : results) {
203                 System.out.println(String.format("%s  %s  %s  %s  %s  %s  ±%s %s %s",
204                         padLeft(result.getTarget(), groupSize),
205                         padLeft(result.getOperation(), nameSize),
206                         padLeft(result.getImplementation(), implSize),
207                         padRight(result.getParamKey(), paramKeySize),
208                         padRight(Long.toString(result.getSampleCount()), countSize),
209                         padRight(result.getScoreFormatted(), scoreSize),
210                         padRight(result.getScoreErrorPct(), errorSize),
211                         padRight(result.getUnit(), unitSize),
212                         calculatePerformanceStr(result, alternativeImplementations, resultsByKey)
213                 ));
214             }
215             System.out.println("\n");
216         }
217 
218         private int altImplColSize(String name) {
219             return Math.max(5, name.length());
220         }
221 
222         private String calculatePerformanceStr(TestExecution result, Array<String> alternativeImplementations, Map<String, Array<TestExecution>> resultsByKey) {
223             final String aggregateKey = result.getTestNameParamKey();
224             final Array<TestExecution> alternativeResults = resultsByKey.get(aggregateKey).getOrElse(Array::empty);
225 
226             return alternativeImplementations.map(altImpl -> Tuple.of(altImpl, alternativeResults.find(r -> altImpl.equals(r.getImplementation()))))
227                     .map(alt -> Tuple.of(alt._1, calculateRatioStr(result, alt._2)))
228                     .map(alt -> padRight(alt._2, altImplColSize(alt._1)))
229                     .mkString("  ");
230         }
231 
232         private String calculateRatioStr(TestExecution baseResult, Option<TestExecution> alternativeResult) {
233             if (!alternativeResult.isDefined()) {
234                 return "";
235             }
236             final double alternativeScore = alternativeResult.get().getScore();
237             if (alternativeScore == 0.0) {
238                 return "";
239             }
240             final double ratio = baseResult.getScore() / alternativeScore;
241             return ratio == 1.0 ? "" : PERFORMANCE_FORMAT.format(ratio) + "×";
242         }
243     }

245     private class RatioPerformanceReport {
246         private final Map<String, Array<TestExecution>> resultsByKey;
247         private final int groupSize;
248         private final int nameSize;
249         private final Array<String> paramKeys;
250         private final int paramKeySize;
251         private final Array<String> alternativeImplementations;
252         private final int alternativeImplSize;
253         private final int ratioSize;
254         private final Array<String> targetImplementations;
255         private final String targetImplementation;
256 
257         public RatioPerformanceReport(Array<TestExecution> results, String targetImplementation) {
258             this.targetImplementation = targetImplementation;
259 
260             resultsByKey = results.groupBy(TestExecution::getTestNameKey);
261             groupSize = Math.max(results.map(r -> r.getTarget().length()).max().get(), 10);
262             nameSize = Math.max(results.map(r -> r.getOperation().length()).max().get(), 9);
263 
264             paramKeys = results.map(TestExecution::getParamKey).distinct().sorted(TO_STRING_COMPARATOR);
265             paramKeySize = Math.max(results.map(r -> r.getParamKey().length()).max().get(), 8);
266 
267             alternativeImplementations = results.map(TestExecution::getImplementation).distinct();
268             targetImplementations = alternativeImplementations.filter(i -> i.toLowerCase().contains(targetImplementation.toLowerCase()));
269             alternativeImplSize = Math.max(alternativeImplementations.map(String::length).max().getOrElse(0), 10);
270 
271             final int targetImplSize = Math.max(targetImplementations.map(String::length).max().getOrElse(0), 10);
272             ratioSize = Math.max(targetImplSize + 1 + alternativeImplSize, 10);
273         }
274 
275         public void print() {
276             printHeader();
277             printReport();
278         }

280         private void printHeader() {
281             System.out.println("\n\n");
282             System.out.println("Performance Ratios");
283             System.out.println(CharSeq.of("=").repeat(ratioHeaderNumerator().length()));
284             if (outlierLowPct > 0.0 && outlierHighPct > 0.0) {
285                 System.out.println(String.format("  (Outliers removed: %s low end, %s high end)", PCT_FORMAT.format(outlierLowPct), PCT_FORMAT.format(outlierHighPct)));
286             }
287         }
288 
289         private String ratioHeaderNumerator() {
290             final String paramKeyHeader = paramKeys.map(type -> padRight(type, paramKeySize)).mkString(" ");
291             return String.format("%s  %s  %s  %s ",
292                     padLeft("Target", groupSize),
293                     padLeft("Operation", nameSize),
294                     padLeft("Ratio", ratioSize),
295                     paramKeyHeader
296             );
297         }
298 
299         private String ratioHeaderDenominator() {
300             final String paramKeyHeader = paramKeys.map(type -> padRight(type, paramKeySize)).mkString(" ");
301             return String.format("%s  %s  %s  %s ",
302                     padLeft("Target", groupSize),
303                     padLeft("Operation", nameSize),
304                     padRight("Ratio", ratioSize),
305                     paramKeyHeader
306             );
307         }
308 
309         private void printReport() {
310             if (alternativeImplementations.size() < 2) {
311                 System.out.println("(nothing to report, you need at least two different implementation)");
312                 return;
313             }
314 
315             printTargetInDenominator();
316             printTargetInNumerator();
317 
318             System.out.println("\n");
319         }
320 
321         @SuppressWarnings("Convert2MethodRef")
322         private void printTargetInNumerator() {
323             System.out.println(String.format("\nRatios %s / <alternative_impl>", targetImplementation));
324             System.out.println(ratioHeaderNumerator());
325             for (String targetImpl : targetImplementations) {
326                 for (Tuple2<String, Array<TestExecution>> execution : resultsByKey) {
327                     printRatioForBaseType(targetImpl, execution._2,
328                             (baseImpl, alternativeImpl) -> padLeft(String.format("%s/%s", baseImpl, alternativeImpl), ratioSize),
329                             (baseExec, alternativeExec) -> calculateRatios(baseExec, alternativeExec));
330                 }
331             }
332         }
333 
334         private void printTargetInDenominator() {
335             System.out.println(String.format("\nRatios <alternative_impl> / %s", targetImplementation));
336             System.out.println(ratioHeaderDenominator());
337             for (String targetImpl : targetImplementations) {
338                 for (Tuple2<String, Array<TestExecution>> execution : resultsByKey) {
339                     printRatioForBaseType(targetImpl, execution._2,
340                             (baseImpl, alternativeImpl) -> padRight(String.format("%s/%s", alternativeImpl, baseImpl), ratioSize),
341                             (baseExec, alternativeExec) -> calculateRatios(alternativeExec, baseExec));
342                 }
343             }
344         }
345 
346         private void printRatioForBaseType(String baseType, Array<TestExecution> testExecutions,
347                 BiFunction<String, String, String> ratioNamePrinter,
348                 BiFunction<Array<TestExecution>, Array<TestExecution>, String> ratioCalculator) {
349             final Array<TestExecution> baseImplExecutions = testExecutions.filter(e -> e.getImplementation().equals(baseType));
350             if (baseImplExecutions.isEmpty()) {
351                 return;
352             }
353             final TestExecution baseTypeExecution = baseImplExecutions.head();
354 
355             for (String alternativeImpl : alternativeImplementations) {
356                 if (alternativeImpl.equals(baseType)) {
357                     continue;
358                 }
359                 final Array<TestExecution> alternativeExecutions = testExecutions.filter(e -> e.getImplementation().equals(alternativeImpl));
360                 if (alternativeExecutions.isEmpty()) {
361                     continue;
362                 }
363                 System.out.println(String.format("%s  %s  %s  %s",
364                         padLeft(baseTypeExecution.getTarget(), groupSize),
365                         padLeft(baseTypeExecution.getOperation(), nameSize),
366                         ratioNamePrinter.apply(baseType, alternativeImpl),
367                         ratioCalculator.apply(baseImplExecutions, alternativeExecutions)));
368             }
369             System.out.println();
370         }
371 
372         private String calculateRatios(Array<TestExecution> alternativeExecutions, Array<TestExecution> baseImplExecutions) {
373             Array<String> ratioStings = Array.empty();
374             for (String paramKey : paramKeys) {
375                 final Option<TestExecution> alternativeExecution = alternativeExecutions.find(e -> e.getParamKey().equals(paramKey));
376                 final Option<TestExecution> baseExecution = baseImplExecutions.find(e -> e.getParamKey().equals(paramKey));
377                 final String paramRatio = alternativeExecution.isEmpty() || baseExecution.isEmpty() || baseExecution.get().getScore() == 0.0
378                                           ? ""
379                                           : PERFORMANCE_FORMAT.format(alternativeExecution.get().getScore() / baseExecution.get().getScore()) + "×";
380                 ratioStings = ratioStings.append(padRight(paramRatio, paramKeySize));
381             }
382             return ratioStings.mkString(" ");
383         }

386     public static class TestExecution implements Comparable<TestExecution> {
387         private static double outlierLowPct;
388         private static double outlierHighPct;
389         private final String paramKey;
390         private final String fullName;
391         private final String target;
392         private final String operation;
393         private final String implementation;
394         private final long sampleCount;
395         private final double score;
396         private final double scoreError;
397         private final String unit;
398 
399         public static TestExecution of(BenchmarkResult benchmarkResult, double outlierLowPct, double outlierHighPct) {
400             TestExecution.outlierLowPct = outlierLowPct;
401             TestExecution.outlierHighPct = outlierHighPct;
402             return new TestExecution(benchmarkResult);
403         }
404 
405         public TestExecution(BenchmarkResult benchmark) {
406             final Result<?> primaryResult = benchmark.getPrimaryResult();
407             fullName = benchmark.getParams().getBenchmark();
408             target = extractPart(fullName, 2);
409             operation = extractPart(fullName, 1);
410             implementation = extractPart(fullName, 0);
411             paramKey = getParameterKey(benchmark);
412 
413             final ListStatistics statistics = createStatisticsWithoutOutliers(benchmark, outlierLowPct, outlierHighPct);
414             sampleCount = statistics.getN();
415             score = statistics.getMean();
416             scoreError = statistics.getMeanErrorAt(0.999);
417             unit = primaryResult.getScoreUnit();
418         }
419 
420         private ListStatistics createStatisticsWithoutOutliers(BenchmarkResult benchmark, double outlierLowPct, double outlierHighPct) {
421             Array<Double> results = benchmark.getIterationResults().stream()
422                     .map(r -> r.getPrimaryResult().getScore())
423                     .collect(Array.collector());
424             final int size = results.size();
425             final int outliersLow = (int) (size * outlierLowPct);
426             final int outliersHigh = (int) (size * outlierHighPct);
427             results = results.drop(outliersLow).dropRight(outliersHigh);
428             return new ListStatistics(results.toJavaList().stream().mapToDouble(r -> r).toArray());
429         }
430 
431         private String getParameterKey(BenchmarkResult benchmarkResult) {
432             final BenchmarkParams params = benchmarkResult.getParams();
433             return params.getParamsKeys().stream().map(params::getParam).collect(Collectors.joining(";"));
434         }
435 
436         public String getTestNameParamKey() {
437             return target + ":" + operation + ":" + unit + ":" + paramKey;
438         }
439 
440         public String getTestNameKey() {
441             return target + ":" + operation + ":" + unit;
442         }

444         private String extractPart(String fullyQualifiedName, int indexFromLast) {
445             final String[] parts = fullyQualifiedName.split("\\.");
446             return parts.length > indexFromLast ? parts[parts.length - indexFromLast - 1] : "";
447         }

449         public String getParamKey() {
450             return paramKey;
451         }
452 
453         public String getTarget() {
454             return target;
455         }
456 
457         public String getOperation() {
458             return operation;

461         public String getImplementation() {

466             return sampleCount;

473         public String getScoreFormatted() {
474             return DECIMAL_FORMAT.format(score);
475         }
476 
477         public String getScoreErrorPct() {
478             return PCT_FORMAT.format(score == 0 ? 0 : scoreError / score);
479         }
480 
481         public String getUnit() {
482             return unit;
483         }
484 
485         @Override
486         public String toString() {
487             return String.format("%s %s %s %s -> %s (± %s)", paramKey, target, operation, implementation, getScoreFormatted(), getScoreErrorPct());
488         }

490         Comparator<TestExecution> comparator = Comparator
491                 .comparing(TestExecution::getUnit)
492                 .thenComparing(TestExecution::getTarget)
493                 .thenComparing(TestExecution::getParamKey)
494                 .thenComparing(TestExecution::getOperation)
495                 .thenComparing(TestExecution::getImplementation);
496 
497         @Override
498         public int compareTo(TestExecution o) {
499             return comparator.compare(this, o);
500         }

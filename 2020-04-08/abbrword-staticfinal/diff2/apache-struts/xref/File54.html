89      public List<Validator> getValidators(Class clazz, String context, String method) {
90          final String validatorKey = buildValidatorKey(clazz, context);
91          final List<ValidatorConfig> cfgs;
92  
93          if (validatorCache.containsKey(validatorKey)) {
94              if (reloadingConfigs) {
95                  validatorCache.put(validatorKey, buildValidatorConfigs(clazz, context, true, null));
96              }
97          } else {
98              validatorCache.put(validatorKey, buildValidatorConfigs(clazz, context, false, null));
99          }
100 
101         // get the set of validator configs
102         cfgs = new ArrayList<ValidatorConfig>(validatorCache.get(validatorKey));
103 
104         ValueStack stack = ActionContext.getContext().getValueStack();
105 
106         // create clean instances of the validators for the caller's use
107         ArrayList<Validator> validators = new ArrayList<>(cfgs.size());
108         for (ValidatorConfig cfg : cfgs) {
109             if (method == null || method.equals(cfg.getParams().get("methodName"))) {
110                 Validator validator = validatorFactory.getValidator(
111                         new ValidatorConfig.Builder(cfg)
112                                 .removeParam("methodName")
113                                 .build());
114                 validator.setValidatorType(cfg.getType());
115                 validator.setValueStack(stack);
116                 validators.add(validator);
117             }
118         }
119 
120         return validators;
121     }

136     public void validate(Object object, String context, ValidatorContext validatorContext, String method) throws ValidationException {
137         List<Validator> validators = getValidators(object.getClass(), context, method);
138         Set<String> shortcircuitedFields = null;
139 
140         for (final Validator validator : validators) {
141             try {
142                 validator.setValidatorContext(validatorContext);
143 
144                 LOG.debug("Running validator: {} for object {} and method {}", validator, object, method);
145 
146                 FieldValidator fValidator = null;
147                 String fullFieldName = null;
148 
149                 if (validator instanceof FieldValidator) {
150                     fValidator = (FieldValidator) validator;
151                     fullFieldName = fValidator.getValidatorContext().getFullFieldName(fValidator.getFieldName());
152 
153                     if ((shortcircuitedFields != null) && shortcircuitedFields.contains(fullFieldName)) {
154                         LOG.debug("Short-circuited, skipping");
155                         continue;
156                     }
157                 }
158 
159                 if (validator instanceof ShortCircuitableValidator && ((ShortCircuitableValidator) validator).isShortCircuit()) {
160                     // get number of existing errors
161                     List<String> errs = null;
162 
163                     if (fValidator != null) {
164                         if (validatorContext.hasFieldErrors()) {
165                             Collection<String> fieldErrors = validatorContext.getFieldErrors().get(fullFieldName);
166 
167                             if (fieldErrors != null) {
168                                 errs = new ArrayList<>(fieldErrors);
169                             }
170                         }
171                     } else if (validatorContext.hasActionErrors()) {
172                         Collection<String> actionErrors = validatorContext.getActionErrors();
173 
174                         if (actionErrors != null) {
175                             errs = new ArrayList<>(actionErrors);
176                         }
177                     }
178 
179                     validator.validate(object);
180 
181                     if (fValidator != null) {
182                         if (validatorContext.hasFieldErrors()) {
183                             Collection<String> errCol = validatorContext.getFieldErrors().get(fullFieldName);
184 
185                             if ((errCol != null) && !errCol.equals(errs)) {
186                                 LOG.debug("Short-circuiting on field validation");
187 
188                                 if (shortcircuitedFields == null) {
189                                     shortcircuitedFields = new TreeSet<String>();
190                                 }
191 
192                                 shortcircuitedFields.add(fullFieldName);
193                             }
194                         }
195                     } else if (validatorContext.hasActionErrors()) {
196                         Collection<String> errCol = validatorContext.getActionErrors();
197 
198                         if ((errCol != null) && !errCol.equals(errs)) {
199                             LOG.debug("Short-circuiting");
200                             break;
201                         }
202                     }
203 
204                     continue;
205                 }
206 
207                 validator.validate(object);
208             } finally {
209                 validator.setValidatorContext(null);
210             }
211 
212         }
213     }

215     /**
216      * Builds a key for validators - used when caching validators.
217      *
218      * @param clazz the action.
219      * @param context context
220      * @return a validator key which is the class name plus context.
221      */
222     protected String buildValidatorKey(Class clazz, String context) {
223         ActionInvocation invocation = ActionContext.getContext().getActionInvocation();
224         ActionProxy proxy = invocation.getProxy();
225         ActionConfig config = proxy.getConfig();
226 
227         StringBuilder sb = new StringBuilder(clazz.getName());
228         sb.append("/");
229         if (StringUtils.isNotBlank(config.getPackageName())) {
230             sb.append(config.getPackageName());
231             sb.append("/");
232         }
233 
234         // the key needs to use the name of the action from the config file,
235         // instead of the url, so wild card actions will have the same validator
236         // see WW-2996
237 
238         // UPDATE:
239         // WW-3753 Using the config name instead of the context only for
240         // wild card actions to keep the flexibility provided
241         // by the original design (such as mapping different contexts
242         // to the same action and method if desired)
243 
244         // UPDATE:
245         // WW-4536 Using NameVariablePatternMatcher allows defines actions
246         // with patterns enclosed with '{}', it's similar case to WW-3753
247         String configName = config.getName();
248         if (configName.contains(ActionConfig.WILDCARD) || (configName.contains("{") && configName.contains("}"))) {
249             sb.append(configName);
250             sb.append("|");
251             sb.append(proxy.getMethod());
252         } else {
253             sb.append(context);
254         }
255         
256         return sb.toString();
257     }

259     private List<ValidatorConfig> buildAliasValidatorConfigs(Class aClass, String context, boolean checkFile) {
260         String fileName = aClass.getName().replace('.', '/') + "-" + context.replace('/', '-') + VALIDATION_CONFIG_SUFFIX;
261 
262         return loadFile(fileName, aClass, checkFile);
263     }

266     protected List<ValidatorConfig> buildClassValidatorConfigs(Class aClass, boolean checkFile) {
267 
268         String fileName = aClass.getName().replace('.', '/') + VALIDATION_CONFIG_SUFFIX;
269 
270         List<ValidatorConfig> result = new ArrayList<>(loadFile(fileName, aClass, checkFile));
271 
272         AnnotationValidationConfigurationBuilder builder = new AnnotationValidationConfigurationBuilder(validatorFactory);
273 
274         List<ValidatorConfig> annotationResult = new ArrayList<>(builder.buildAnnotationClassValidatorConfigs(aClass));
275 
276         result.addAll(annotationResult);
277 
278         return result;
279 
280     }

282     /**
283      * <p>This method 'collects' all the validator configurations for a given
284      * action invocation.</p>
285      *
286      * <p>It will traverse up the class hierarchy looking for validators for every super class
287      * and directly implemented interface of the current action, as well as adding validators for
288      * any alias of this invocation. Nifty!</p>
289      *
290      * <p>Given the following class structure:</p>
291      * <pre>
292      *   interface Thing;
293      *   interface Animal extends Thing;
294      *   interface Quadraped extends Animal;
295      *   class AnimalImpl implements Animal;
296      *   class QuadrapedImpl extends AnimalImpl implements Quadraped;
297      *   class Dog extends QuadrapedImpl;
298      * </pre>
299      *
300      * <p>This method will look for the following config files for Dog:</p>
301      * <pre>
302      *   Animal
303      *   Animal-context
304      *   AnimalImpl
305      *   AnimalImpl-context
306      *   Quadraped
307      *   Quadraped-context
308      *   QuadrapedImpl
309      *   QuadrapedImpl-context
310      *   Dog
311      *   Dog-context
312      * </pre>
313      *
314      * <p>Note that the validation rules for Thing is never looked for because no class in the
315      * hierarchy directly implements Thing.</p>
316      *
317      * @param clazz     the Class to look up validators for.
318      * @param context   the context to use when looking up validators.
319      * @param checkFile true if the validation config file should be checked to see if it has been
320      *                  updated.
321      * @param checked   the set of previously checked class-contexts, null if none have been checked
322      * @return a list of validator configs for the given class and context.
323      */
324     private List<ValidatorConfig> buildValidatorConfigs(Class clazz, String context, boolean checkFile, Set<String> checked) {
325         List<ValidatorConfig> validatorConfigs = new ArrayList<>();
326 
327         if (checked == null) {
328             checked = new TreeSet<>();
329         } else if (checked.contains(clazz.getName())) {
330             return validatorConfigs;
331         }
332 
333         if (clazz.isInterface()) {
334             Class[] interfaces = clazz.getInterfaces();
335 
336             for (Class anInterface : interfaces) {
337                 validatorConfigs.addAll(buildValidatorConfigs(anInterface, context, checkFile, checked));
338             }
339         } else {
340             if (!clazz.equals(Object.class)) {
341                 validatorConfigs.addAll(buildValidatorConfigs(clazz.getSuperclass(), context, checkFile, checked));
342             }
343         }
344 
345         // look for validators for implemented interfaces
346         Class[] interfaces = clazz.getInterfaces();
347 
348         for (Class anInterface1 : interfaces) {
349             if (checked.contains(anInterface1.getName())) {
350                 continue;
351             }
352 
353             validatorConfigs.addAll(buildClassValidatorConfigs(anInterface1, checkFile));
354 
355             if (context != null) {
356                 validatorConfigs.addAll(buildAliasValidatorConfigs(anInterface1, context, checkFile));
357             }
358 
359             checked.add(anInterface1.getName());
360         }
361 
362         validatorConfigs.addAll(buildClassValidatorConfigs(clazz, checkFile));
363 
364         if (context != null) {
365             validatorConfigs.addAll(buildAliasValidatorConfigs(clazz, context, checkFile));
366         }
367 
368         checked.add(clazz.getName());
369 
370         return validatorConfigs;
371     }

373     private List<ValidatorConfig> loadFile(String fileName, Class clazz, boolean checkFile) {
374         List<ValidatorConfig> retList = Collections.emptyList();
375 
376         URL fileUrl = ClassLoaderUtil.getResource(fileName, clazz);
377 
378         if ((checkFile && fileManager.fileNeedsReloading(fileUrl)) || !validatorFileCache.containsKey(fileName)) {
379             try (InputStream is = fileManager.loadFile(fileUrl)) {
380                 if (is != null) {
381                     retList = new ArrayList<>(validatorFileParser.parseActionValidatorConfigs(validatorFactory, is, fileName));
382                 }
383             } catch (IOException e) {
384                 LOG.error("Caught exception while loading file {}", fileName, e);
385             }
386 
387             validatorFileCache.put(fileName, retList);
388         } else {
389             retList = validatorFileCache.get(fileName);
390         }
391 
392         return retList;
393     }

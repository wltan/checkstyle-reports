38  /**
39   * Value Stack's Context related Utilities.
40   *
41   */
42  public class ContextUtil {
43      public static final String REQUEST = "request";
44      public static final String RESPONSE = "response";
45      public static final String SESSION = "session";
46      public static final String BASE = "base";
47      public static final String STACK = "stack";
48      public static final String OGNL = "ognl";
49      public static final String STRUTS = "struts";
50      public static final String ACTION = "action";
51      
52      public static Map<String, Object> getStandardContext(ValueStack stack, HttpServletRequest req, HttpServletResponse res) {
53          HashMap<String, Object> map = new HashMap<>();
54          map.put(REQUEST, req);
55          map.put(RESPONSE, res);
56          map.put(SESSION, req.getSession(false));
57          map.put(BASE, req.getContextPath());
58          map.put(STACK, stack);
59          map.put(OGNL, ((Container)stack.getContext().get(ActionContext.CONTAINER)).getInstance(OgnlTool.class));
60          map.put(STRUTS, new StrutsUtil(stack, req, res));
61  
62          ActionInvocation invocation = (ActionInvocation) stack.getContext().get(ActionContext.ACTION_INVOCATION);
63          if (invocation != null) {
64              map.put(ACTION, invocation.getAction());
65          }
66          return map;
67      }
68  
69      /**
70       * Return true if either Configuration's altSyntax is on or the stack context's useAltSyntax is on
71       * @param context stack's context
72       * @return boolean
73       */
74      public static boolean isUseAltSyntax(Map context) {
75          // We didn't make altSyntax static cause, if so, struts.configuration.xml.reload will not work
76          // plus the Configuration implementation should cache the properties, which the framework's
77          // configuration implementation does
78          return "true".equals(((Container)context.get(ActionContext.CONTAINER)).getInstance(String.class, StrutsConstants.STRUTS_TAG_ALTSYNTAX)) ||(
79                  (context.containsKey("useAltSyntax") &&
80                          context.get("useAltSyntax") != null &&
81                          "true".equals(context.get("useAltSyntax").toString())));
82      }
83      
84      /**
85       * Returns a String for overriding the default templateSuffix if templateSuffix is on the stack
86       * @param context stack's context
87       * @return String
88       */
89      public static String getTemplateSuffix(Map context) {
90          return context.containsKey("templateSuffix") ? (String) context.get("templateSuffix") : null;
91      }
92  }

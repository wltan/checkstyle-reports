43  /**
44   * Main ServletFilter that configures and enables all JSecurity functions within a web application.
45   *
46   * The following is a fully commented example that documents how to configure it:
47   *
48   * <pre>&lt;filter&gt;
49   * &lt;filter-name&gt;JSecurityFilter&lt;/filter-name&gt;
50   * &lt;filter-class&gt;org.jsecurity.web.servlet.JSecurityFilter&lt;/filter-class&gt;
51   * &lt;init-param&gt;&lt;param-name&gt;config&lt;/param-name&gt;&lt;param-value&gt;
52   *
53   * #NOTE:  This config looks pretty long - but its not - its only 5 lines of actual config.
54   * #       Everything else is just heavily commented to explain things in-depth. Feel free to delete any
55   * #       comments that you don't want to read from your own configuration ;)
56   * #
57   * # Any commented values below are JSecurity's defaults.  If you want to change any values, you only
58   * # need to uncomment the lines you want to change.
59   *
60   * [main]
61   * # The 'main' section defines JSecurity-wide configuration.
62   * #
63   * # Session Mode: By default, JSecurity's Session infrastructure in a web environment will use the
64   * # Servlet container's HttpSession.  However, if you need to share session state across client types
65   * # (e.g. Web MVC plus Java Web Start or Flash), or are doing distributed/shared Sessions for
66   * # Single Sign On, HttpSessions aren't good enough.  You'll need to use JSecurity's more powerful
67   * # (and client-agnostic) session management.  You can enable this by uncommenting the following line
68   * # and changing 'http' to 'jsecurity'
69   * #
70   * #securityManager = {@link org.jsecurity.web.DefaultWebSecurityManager org.jsecurity.web.DefaultWebSecurityManager}
71   * #securityManager.{@link org.jsecurity.web.DefaultWebSecurityManager#setSessionMode(String) sessionMode} = http
72   *
73   * [filters]
74   * # This section defines the 'pool' of all Filters available to the url path definitions in the [urls] section below.
75   * #
76   * # The following commented values are already provided by JSecurity by default and are immediately usable
77   * # in the [urls] definitions below.  If you like, you may override any values by uncommenting only the lines
78   * # you need to change.
79   * #
80   * # Each Filter is configured based on its functionality and/or protocol.  You should read each
81   * # Filter's JavaDoc to fully understand what each does and how it works as well as how it would
82   * # affect the user experience.
83   * #
84   * # Form-based Authentication filter:
85   * #<a name="authc"></a>authc = {@link org.jsecurity.web.filter.authc.FormAuthenticationFilter}
86   * #authc.{@link org.jsecurity.web.filter.authc.FormAuthenticationFilter#setLoginUrl(String) url} = /login.jsp
87   * #authc.{@link org.jsecurity.web.filter.authc.FormAuthenticationFilter#setUsernameParam(String) usernameParam} = username
88   * #authc.{@link org.jsecurity.web.filter.authc.FormAuthenticationFilter#setPasswordParam(String) passwordParam} = password
89   * #authc.{@link org.jsecurity.web.filter.authc.FormAuthenticationFilter#setRememberMeParam(String) rememberMeParam} = rememberMe
90   * #authc.{@link org.jsecurity.web.filter.authc.FormAuthenticationFilter#setSuccessUrl(String) successUrl}  = /login.jsp
91   * #authc.{@link org.jsecurity.web.filter.authc.FormAuthenticationFilter#setFailureKeyAttribute(String) failureKeyAttribute} = {@link org.jsecurity.web.filter.authc.FormAuthenticationFilter#DEFAULT_ERROR_KEY_ATTRIBUTE_NAME}
92   * #
93   * # Http BASIC Authentication filter:
94   * #<a name="authcBasic"></a>authcBasic = {@link org.jsecurity.web.filter.authc.BasicHttpAuthenticationFilter}
95   * #authcBasic.{@link org.jsecurity.web.filter.authc.BasicHttpAuthenticationFilter#setApplicationName(String) applicationName} = application
96   * #
97   * # Roles filter: requires the requesting user to have one or more roles for the request to continue.
98   * # If they do not have the specified roles, they are redirected to the specified URL.
99   * #<a name="roles"></a>roles = {@link org.jsecurity.web.filter.authz.RolesAuthorizationFilter}
100  * #roles.{@link org.jsecurity.web.filter.authz.RolesAuthorizationFilter#setUnauthorizedUrl(String) url} =
101  * # (note the above url is null by default, which will cause an HTTP 403 (Access Denied) response instead
102  * # of redirecting to a page.  If you want to show a 'nice page' instead, you should specify that url.
103  * #
104  * # Permissions filter: requires the requesting user to have one or more permissions for the request to
105  * # continue, and if they do not, redirects them to the specified URL.
106  * #<a name="perms"></a>perms = {@link org.jsecurity.web.filter.authz.PermissionsAuthorizationFilter}
107  * #perms.{@link org.jsecurity.web.filter.authz.PermissionsAuthorizationFilter#setUnauthorizedUrl(String) url} =
108  * # (note the above url is null by default, which will cause an HTTP 403 (Access Denied) response instead
109  * # of redirecting to a page.  If you want to show a 'nice page' instead, you should specify that url.  Many
110  * # applications like to use the same url specified in roles.url above.
111  * #
112  * #
113  * # Define your own filters here.  To properly handle url path matching (see the [urls] section below), your
114  * # filter should extend the {@link org.jsecurity.web.filter.PathMatchingFilter PathMatchingFilter} abstract class.
115  *
116  * [urls]
117  * # This section defines url path mappings.  Each mapping entry must be on a single line and conform to the
118  * # following representation:
119  * #
120  * # ant_path_expression = path_specific_filter_chain_definition
121  * #
122  * # For any request that matches a specified path, the corresponding value defines a comma-delimited chain of
123  * # filters to execute for that request.
124  * #
125  * # This is incredibly powerful in that you can define arbitrary filter chains for any given request pattern
126  * # to greatly customize the security experience.
127  * #
128  * # The path_specific_filter_chain_definition must match the following format:
129  * #
130  * # filter1[optional_config1], filter2[optional_config2], ..., filterN[optional_configN]
131  * #
132  * # where 'filterN' is the name of an filter defined above in the [filters] section and
133  * # '[optional_configN]' is an optional bracketed string that has meaning for that particular filter for
134  * # _that particular path_.  If the filter does not need specific config for that url path, you may
135  * # discard the brackets - that is, filterN[] just becomes filterN.
136  * #
137  * # And because filter tokens define chains, order matters!  Define the tokens for each path pattern
138  * # in the order you want them to filter (comma-delimited).
139  * #
140  * # Finally, each filter is free to handle the response however it wants if its necessary
141  * # conditions are not met (redirect, HTTP error code, direct rendering, etc).  Otherwise, it is expected to allow
142  * # the request to continue through the chain on to the final destination view.
143  * #
144  * # Examples:
145  * #
146  * # To illustrate chain configuration, look at the /account/** mapping below.  This says
147  * # &quot;apply the above 'authcBasic' filter to any request matching the '/account/**' pattern&quot;.  Since the
148  * # 'authcBasic' filter does not need any path-specific config, it doesn't have any config brackets [].
149  * #
150  * # The /remoting/** definition on the other hand uses the 'roles' and 'perms' filters which do use
151  * # bracket notation.  That definition says:
152  * #
153  * # &quot;To access /remoting/** urls, ensure that the user is first authenticated ('authcBasic'), then ensure that user
154  * # has the 'b2bClient' role, and then finally ensure that they have the 'remote:invoke:lan,wan' permission.&quot;
155  * #
156  * # (Note that because elements within brackets [ ] are comma-delimited themselves, we needed to escape the permission
157  * # actions of 'lan,wan' with quotes.  If we didn't do that, the permission filter would interpret
158  * # the text between the brackets as two permissions: 'remote:invoke:lan' and 'wan' instead of the
159  * # single desired 'remote:invoke:lan,wan' token.  So, you can use quotes wherever you need to escape internal
160  * # commas.)
161  *
162  * /account/** = <a href="#authcBasic">authcBasic</a>
163  * /remoting/** = <a href="#authcBasic">authcBasic</a>, <a href="#roles">roles</a>[b2bClient], <a href="#perms">perms</a>[remote:invoke:"lan,wan"]
164  *
165  * &lt;/param-value&gt;&lt;/init-param&gt;
166  * &lt;/filter&gt;
167  *
168  *
169  * &lt;filter-mapping&gt;
170  * &lt;filter-name&gt;JSecurityFilter&lt;/filter-name&gt;
171  * &lt;url-pattern&gt;/*&lt;/url-pattern&gt;
172  * &lt;/filter-mapping&gt;</pre>
173  *
174  * @author Les Hazlewood
175  * @author Jeremy Haile
176  * @since 0.1
177  */
178 public class JSecurityFilter extends OncePerRequestFilter {
179 
180     public static final String SECURITY_MANAGER_CONTEXT_KEY = SecurityManager.class.getName() + "_SERVLET_CONTEXT_KEY";
181 
182     public static final String CONFIG_CLASS_NAME_INIT_PARAM_NAME = "configClassName";
183     public static final String CONFIG_INIT_PARAM_NAME = "config";
184     public static final String CONFIG_URL_INIT_PARAM_NAME = "configUrl";
185 
186     private static final Log log = LogFactory.getLog(JSecurityFilter.class);    
187 
188     protected String config;
189     protected String configUrl;
190     protected String configClassName;
191     protected WebConfiguration configuration;
192 
193     // Reference to the security manager used by this filter
194     protected SecurityManager securityManager;
195 
196     public JSecurityFilter() {
197         this.configClassName = IniWebConfiguration.class.getName();
198     }
199 
200     public WebConfiguration getConfiguration() {
201         return configuration;
202     }
203 
204     public void setConfiguration(WebConfiguration configuration) {
205         this.configuration = configuration;
206     }
207 
208     public SecurityManager getSecurityManager() {
209         return securityManager;
210     }
211 
212     protected void setSecurityManager(SecurityManager sm) {
213         this.securityManager = sm;
214     }
215 
216     protected void onFilterConfigSet() throws Exception {
217         applyInitParams();
218         WebConfiguration config = configure();
219         setConfiguration(config);
220 
221         // Retrieve and store a reference to the security manager
222         SecurityManager sm = ensureSecurityManager(config);
223         setSecurityManager(sm);
224     }
225 
226     /**
227      * Retrieves the security manager for the given configuration.
228      *
229      * @param config the configuration for this filter.
230      * @return the security manager that this filter should use.
231      */
232     protected SecurityManager ensureSecurityManager(Configuration config) {
233         SecurityManager sm = config.getSecurityManager();
234 
235         // If the config doesn't return a security manager, build one by default.
236         if (sm == null) {
237             if (log.isInfoEnabled()) {
238                 log.info("Configuration instance [" + config + "] did not provide a SecurityManager.  No config " +
239                         "specified?  Defaulting to a " + DefaultWebSecurityManager.class.getName() + " instance...");
240             }
241             sm = new DefaultWebSecurityManager();
242         }
243 
244         return sm;
245     }
246 
247     protected void applyInitParams() {
248         FilterConfig config = getFilterConfig();
249 
250         String configCN = clean(config.getInitParameter(CONFIG_CLASS_NAME_INIT_PARAM_NAME));
251         if (configCN != null) {
252             if (ClassUtils.isAvailable(configCN)) {
253                 this.configClassName = configCN;
254             } else {
255                 String msg = "configClassName fully qualified class name value [" + configCN + "] is not " +
256                         "available in the classpath.  Please ensure you have typed it correctly and the " +
257                         "corresponding class or jar is in the classpath.";
258                 throw new ConfigurationException(msg);
259             }
260         }
261 
262         this.config = clean(config.getInitParameter(CONFIG_INIT_PARAM_NAME));
263         this.configUrl = clean(config.getInitParameter(CONFIG_URL_INIT_PARAM_NAME));
264     }
265 
266     protected WebConfiguration configure() {
267         WebConfiguration conf = (WebConfiguration) ClassUtils.newInstance(this.configClassName);
268         applyFilterConfig(conf);
269         applyUrlConfig(conf);
270         applyEmbeddedConfig(conf);
271         LifecycleUtils.init(conf);
272         return conf;
273     }
274 
275     protected void applyFilterConfig(WebConfiguration conf) {
276         if (log.isDebugEnabled()) {
277             String msg = "Attempting to inject the FilterConfig (using 'setFilterConfig' method) into the " +
278                     "instantiated WebConfiguration for any wrapped Filter initialization...";
279             log.debug(msg);
280         }
281         try {
282             PropertyDescriptor pd = PropertyUtils.getPropertyDescriptor(conf, "filterConfig");
283             if (pd != null) {
284                 PropertyUtils.setProperty(conf, "filterConfig", getFilterConfig());
285             }
286         } catch (Exception e) {
287             if (log.isDebugEnabled()) {
288                 log.debug("Error setting FilterConfig on WebConfiguration instance.", e);
289             }
290         }
291     }
292 
293     protected void applyEmbeddedConfig(WebConfiguration conf) {
294         if (this.config != null) {
295             try {
296                 PropertyDescriptor pd = PropertyUtils.getPropertyDescriptor(conf, "config");
297 
298                 if (pd != null) {
299                     PropertyUtils.setProperty(conf, "config", this.config);
300                 } else {
301                     String msg = "The 'config' filter param was specified, but there is no " +
302                             "'setConfig(String)' method on the Configuration instance [" + conf + "].  If you do " +
303                             "not require the 'config' filter param, please comment it out, or if you do need it, " +
304                             "please ensure your Configuration instance has a 'setConfig(String)' method to receive it.";
305                     throw new ConfigurationException(msg);
306                 }
307             } catch (Exception e) {
308                 String msg = "There was an error setting the 'config' property of the Configuration object.";
309                 throw new ConfigurationException(msg, e);
310             }
311         }
312     }
313 
314     protected void applyUrlConfig(WebConfiguration conf) {
315         if (this.configUrl != null) {
316             try {
317                 PropertyDescriptor pd = PropertyUtils.getPropertyDescriptor(conf, "configUrl");
318 
319                 if (pd != null) {
320                     PropertyUtils.setProperty(conf, "configUrl", this.configUrl);
321                 } else {
322                     String msg = "The 'configUrl' filter param was specified, but there is no " +
323                             "'setConfigUrl(String)' method on the Configuration instance [" + conf + "].  If you do " +
324                             "not require the 'configUrl' filter param, please comment it out, or if you do need it, " +
325                             "please ensure your Configuration instance has a 'setConfigUrl(String)' method to receive it.";
326                     throw new ConfigurationException(msg);
327                 }
328             } catch (Exception e) {
329                 String msg = "There was an error setting the 'configUrl' property of the Configuration object.";
330                 throw new ConfigurationException(msg, e);
331             }
332         }
333     }
334 
335     protected boolean isHttpSessions() {
336         SecurityManager secMgr = getSecurityManager();
337         if (secMgr instanceof DefaultWebSecurityManager) {
338             return ((DefaultWebSecurityManager) secMgr).isHttpSessionMode();
339         } else {
340             return true;
341         }
342     }
343 
344     protected InetAddress getInetAddress(ServletRequest request) {
345         return WebUtils.getInetAddress(request);
346     }
347 
348     protected void doFilterInternal(ServletRequest servletRequest, ServletResponse servletResponse,
349                                     FilterChain origChain) throws ServletException, IOException {
350 
351         HttpServletRequest request = (HttpServletRequest) servletRequest;
352         HttpServletResponse response = (HttpServletResponse) servletResponse;
353 
354         ThreadContext.bind(getInetAddress(request));
355 
356         boolean httpSessions = isHttpSessions();
357         request = new JSecurityHttpServletRequest(request, getServletContext(), httpSessions);
358         if (!httpSessions) {
359             //the JSecurityHttpServletResponse exists to support URL rewriting for session ids.  This is only needed if
360             //using JSecurity sessions (i.e. not simple HttpSession based sessions):
361             response = new JSecurityHttpServletResponse(response, getServletContext(), (JSecurityHttpServletRequest) request);
362         }
363 
364         WebUtils.bind(request);
365         WebUtils.bind(response);
366         ThreadContext.bind(getSecurityManager());
367         ThreadContext.bind(getSecurityManager().getSubject());
368 
369         FilterChain chain = getConfiguration().getChain(request, response, origChain);
370         if (chain == null) {
371             chain = origChain;
372             if (log.isTraceEnabled()) {
373                 log.trace("No security filter chain configured for the current request.  Using default.");
374             }
375         } else {
376             if (log.isTraceEnabled()) {
377                 log.trace(" Using configured filter chain for the current request.");
378             }
379         }
380 
381         try {
382             chain.doFilter(request, response);
383         } finally {
384             ThreadContext.unbindSubject();
385             ThreadContext.unbindSecurityManager();
386             WebUtils.unbindServletResponse();
387             WebUtils.unbindServletRequest();
388             ThreadContext.unbindInetAddress();
389         }
390     }
391 
392     public void destroy() {
393         LifecycleUtils.destroy(getConfiguration());
394     }

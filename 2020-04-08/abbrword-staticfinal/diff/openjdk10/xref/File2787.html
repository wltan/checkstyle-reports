264     private static void runTest(CertificateFactory cf,
265             List<X509Certificate> certList, TrustAnchor anchor)
266             throws Exception {
267         CertPath path = cf.generateCertPath(certList);
268         CertPathValidator validator = CertPathValidator.getInstance("PKIX");
269 
270         System.out.println(anchor);
271 
272         // Attach the OCSP responses to a PKIXParameters object
273         PKIXRevocationChecker pkrev =
274                 (PKIXRevocationChecker)validator.getRevocationChecker();
275         Map<X509Certificate, byte[]> responseMap = new HashMap<>();
276         responseMap.put(certList.get(0), DECODER.decode(EE_OCSP_RESP));
277         responseMap.put(certList.get(1), DECODER.decode(INT_CA_OCSP_RESP));
278         pkrev.setOcspResponses(responseMap);
279         PKIXParameters params =
280                 new PKIXParameters(Collections.singleton(anchor));
281         params.addCertPathChecker(pkrev);
282         params.setDate(EVAL_DATE);
283 
284         validator.validate(path, params);
285     }

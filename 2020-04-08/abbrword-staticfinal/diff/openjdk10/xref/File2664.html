24  /* @test
25   * @bug 8054029
26   * @summary Block devices should not report size=0 on Linux
27   */
28  
29  import java.io.RandomAccessFile;
30  import java.nio.file.Path;
31  import java.nio.file.Paths;
32  import java.nio.channels.FileChannel;
33  import java.nio.file.AccessDeniedException;
34  import java.nio.file.NoSuchFileException;
35  import static java.nio.file.StandardOpenOption.*;
36  
37  
38  public class BlockDeviceSize {
39      private static final String BLK_FNAME = "/dev/sda1";
40      private static final Path BLK_PATH = Paths.get(BLK_FNAME);
41  
42      public static void main(String[] args) throws Throwable {
43          try (FileChannel ch = FileChannel.open(BLK_PATH, READ);
44               RandomAccessFile file = new RandomAccessFile(BLK_FNAME, "r")) {
45  
46              long size1 = ch.size();
47              long size2 = file.length();
48              if (size1 != size2) {
49                  throw new RuntimeException("size differs when retrieved" +
50                          " in different ways: " + size1 + " != " + size2);
51              }
52              System.out.println("OK");
53  
54          } catch (NoSuchFileException nsfe) {
55              System.err.println("File " + BLK_FNAME + " not found." +
56                      " Skipping test");
57          } catch (AccessDeniedException ade) {
58              System.err.println("Access to " + BLK_FNAME + " is denied." +
59                      " Run test as root.");
60          }
61      }
62  }

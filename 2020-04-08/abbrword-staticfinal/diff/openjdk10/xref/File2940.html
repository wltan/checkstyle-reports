27  package com.bar;
28  
29  import java.util.*;
30  import java.util.spi.*;
31  
32  import com.foobar.Utils;
33  
34  /**
35   * Implementation class for getGenericTimeZoneName which returns "Generic "+<standard name in OSAKA>.
36   */
37  public class GenericTimeZoneNameProviderImpl extends TimeZoneNameProviderImpl {
38      static final Locale jaJPGeneric = new Locale("ja", "JP", "generic");
39      static final Locale OSAKA = new Locale("ja", "JP", "osaka");
40  
41      static Locale[] avail = {
42          jaJPGeneric
43      };
44  
45      @Override
46      public Locale[] getAvailableLocales() {
47          return avail;
48      }
49  
50      @Override
51      public String getGenericDisplayName(String id, int style, Locale locale) {
52          if (!jaJPGeneric.equals(locale)) {
53              return null;
54          }
55          String std = super.getDisplayName(id, false, style, OSAKA);
56          return (std != null) ? "Generic " + std : null;
57      }
58  }

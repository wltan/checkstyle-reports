240             throws CacheException {
241       if (log.isDebugEnabled()) log.debug("Building query results cache region [" + regionName + "]");
242       String cacheName = typeOverrides.get(QUERY_KEY).getCacheName();
243       // If region name is not default one, lookup a cache for that region name

245          cacheName = regionName;
246 
247       AdvancedCache cache = getCache(cacheName, QUERY_KEY, properties);
248       QueryResultsRegionImpl region = new QueryResultsRegionImpl(
249             cache, regionName, this);
250       startRegion(region, regionName);
251       return region;

301    /**
302     * {@inheritDoc}
303     */
304    public void start(Settings settings, Properties properties) throws CacheException {
305       log.debug("Starting Infinispan region factory");
306       try {
307          transactionManagerlookup = createTransactionManagerLookup(settings, properties);
308          manager = createCacheManager(properties);
309          initGenericDataTypeOverrides();
310          Enumeration keys = properties.propertyNames();
311          while (keys.hasMoreElements()) {
312             String key = (String) keys.nextElement();
313             int prefixLoc;
314             if ((prefixLoc = key.indexOf(PREFIX)) != -1) {
315                dissectProperty(prefixLoc, key, properties);
316             }
317          }
318          defineGenericDataTypeCacheConfigurations(settings, properties);
319          definePendingPutsCache();
320       } catch (CacheException ce) {
321          throw ce;
322       } catch (Throwable t) {
323           throw new CacheException("Unable to start region factory", t);
324       }
325    }

327    private void definePendingPutsCache() {
328       ConfigurationBuilder builder = new ConfigurationBuilder();
329       // A local, lightweight cache for pending puts, which is
330       // non-transactional and has aggressive expiration settings.
331       // Locking is still required since the putFromLoad validator
332       // code uses conditional operations (i.e. putIfAbsent).
333       builder.clustering().cacheMode(CacheMode.LOCAL)
334          .transaction().transactionMode(TransactionMode.NON_TRANSACTIONAL)
335          .expiration().maxIdle(TimeUnit.SECONDS.toMillis(60))
336          .storeAsBinary().enabled(false)
337          .locking().isolationLevel(IsolationLevel.READ_COMMITTED)
338          .jmxStatistics().disable();
339 
340       manager.defineConfiguration(PENDING_PUTS_CACHE_NAME, builder.build());
341    }

383    protected EmbeddedCacheManager createCacheManager(Properties properties) throws CacheException {
384       try {
385          String configLoc = ConfigurationHelper.getString(INFINISPAN_CONFIG_RESOURCE_PROP, properties, DEF_INFINISPAN_CONFIG_RESOURCE);
386 
387          EmbeddedCacheManager manager = new DefaultCacheManager(configLoc, false);
388          String globalStats = extractProperty(INFINISPAN_GLOBAL_STATISTICS_PROP, properties);
389          if (globalStats != null) {
390             // Hack to enable global JMX stats being enabled in both 5.1 and 5.2
391 
392             // 1. Create a configuration builder holder
393             ConfigurationBuilderHolder holder = new ConfigurationBuilderHolder();
394 
395             // 2. Build global configuration with custom settings
396             GlobalConfigurationBuilder globalBuilder = holder.getGlobalConfigurationBuilder();
397             globalBuilder.read(manager.getCacheManagerConfiguration());
398             globalBuilder.globalJmxStatistics().enabled(Boolean.parseBoolean(globalStats));
399 
400             // 3. Build default configuration
401             holder.getDefaultConfigurationBuilder().read(manager.getDefaultCacheConfiguration());
402 
403             // 4. Build all defined caches
404             for (String cacheName : manager.getCacheNames()){
405                ConfigurationBuilder builder = holder.newConfigurationBuilder(cacheName);
406                builder.read(manager.getCacheConfiguration(cacheName));
407             }
408 
409             // 5. Discard existing cache manager and create a brand new one
410             manager.stop();
411             manager = new DefaultCacheManager(holder, false);
412          }
413 
414          manager.start();
415          return manager;
416       } catch (IOException e) {
417          throw new CacheException("Unable to create default cache manager", e);
418       }
419    }

426    private Map<String, TypeOverrides> initGenericDataTypeOverrides() {
427       TypeOverrides entityOverrides = new TypeOverrides();
428       entityOverrides.setCacheName(DEF_ENTITY_RESOURCE);
429       typeOverrides.put(ENTITY_KEY, entityOverrides);
430       TypeOverrides collectionOverrides = new TypeOverrides();
431       collectionOverrides.setCacheName(DEF_ENTITY_RESOURCE);
432       typeOverrides.put(COLLECTION_KEY, collectionOverrides);
433       TypeOverrides naturalIdOverrides = new TypeOverrides();
434       naturalIdOverrides.setCacheName(DEF_ENTITY_RESOURCE);
435       typeOverrides.put(NATURAL_ID_KEY, naturalIdOverrides);
436       TypeOverrides timestampOverrides = new TimestampTypeOverrides();
437       timestampOverrides.setCacheName(DEF_TIMESTAMPS_RESOURCE);
438       typeOverrides.put(TIMESTAMPS_KEY, timestampOverrides);
439       TypeOverrides queryOverrides = new TypeOverrides();
440       queryOverrides.setCacheName(DEF_QUERY_RESOURCE);
441       typeOverrides.put(QUERY_KEY, queryOverrides);
442       return typeOverrides;
443    }

445    private void dissectProperty(int prefixLoc, String key, Properties properties) {
446       TypeOverrides cfgOverride;
447       int suffixLoc;
448       if (!key.equals(INFINISPAN_CONFIG_RESOURCE_PROP) && (suffixLoc = key.indexOf(CONFIG_SUFFIX)) != -1) {
449          cfgOverride = getOrCreateConfig(prefixLoc, key, suffixLoc);
450          cfgOverride.setCacheName(extractProperty(key, properties));
451       } else if ((suffixLoc = key.indexOf(STRATEGY_SUFFIX)) != -1) {
452          cfgOverride = getOrCreateConfig(prefixLoc, key, suffixLoc);
453          cfgOverride.setEvictionStrategy(extractProperty(key, properties));
454       } else if ((suffixLoc = key.indexOf(WAKE_UP_INTERVAL_SUFFIX)) != -1) {
455          cfgOverride = getOrCreateConfig(prefixLoc, key, suffixLoc);
456          cfgOverride.setEvictionWakeUpInterval(Long.parseLong(extractProperty(key, properties)));
457       } else if ((suffixLoc = key.indexOf(MAX_ENTRIES_SUFFIX)) != -1) {
458          cfgOverride = getOrCreateConfig(prefixLoc, key, suffixLoc);
459          cfgOverride.setEvictionMaxEntries(Integer.parseInt(extractProperty(key, properties)));
460       } else if ((suffixLoc = key.indexOf(LIFESPAN_SUFFIX)) != -1) {
461          cfgOverride = getOrCreateConfig(prefixLoc, key, suffixLoc);
462          cfgOverride.setExpirationLifespan(Long.parseLong(extractProperty(key, properties)));
463       } else if ((suffixLoc = key.indexOf(MAX_IDLE_SUFFIX)) != -1) {
464          cfgOverride = getOrCreateConfig(prefixLoc, key, suffixLoc);
465          cfgOverride.setExpirationMaxIdle(Long.parseLong(extractProperty(key, properties)));
466       }
467    }

471       log.debugf("Configuration override via property %s: %s", key, value);
472       return value;

475    private TypeOverrides getOrCreateConfig(int prefixLoc, String key, int suffixLoc) {
476       String name = key.substring(prefixLoc + PREFIX.length(), suffixLoc);
477       TypeOverrides cfgOverride = typeOverrides.get(name);
478       if (cfgOverride == null) {
479          cfgOverride = new TypeOverrides();
480          typeOverrides.put(name, cfgOverride);
481       }
482       return cfgOverride;
483    }

485    private void defineGenericDataTypeCacheConfigurations(Settings settings, Properties properties) throws CacheException {
486       String[] defaultGenericDataTypes = new String[]{ENTITY_KEY, COLLECTION_KEY, TIMESTAMPS_KEY, QUERY_KEY};
487       for (String type : defaultGenericDataTypes) {
488          TypeOverrides override = overrideStatisticsIfPresent(typeOverrides.get(type), properties);
489          String cacheName = override.getCacheName();
490          Configuration newCacheCfg = override.createInfinispanConfiguration();
491          // Apply overrides
492          Configuration cacheConfig = manager.defineConfiguration(cacheName, cacheName, newCacheCfg);
493          // Configure transaction manager
494          cacheConfig = configureTransactionManager(cacheConfig, cacheName, properties);
495          manager.defineConfiguration(cacheName, cacheName, cacheConfig);
496          definedConfigurations.add(cacheName);
497          override.validateInfinispanConfiguration(cacheConfig);
498       }
499    }

501    private AdvancedCache getCache(String regionName, String typeKey, Properties properties) {
502       TypeOverrides regionOverride = typeOverrides.get(regionName);
503       if (!definedConfigurations.contains(regionName)) {
504          String templateCacheName = null;
505          Configuration regionCacheCfg = null;
506          if (regionOverride != null) {
507             if (log.isDebugEnabled()) log.debug("Cache region specific configuration exists: " + regionOverride);
508             regionOverride = overrideStatisticsIfPresent(regionOverride, properties);
509             regionCacheCfg = regionOverride.createInfinispanConfiguration();
510             String cacheName = regionOverride.getCacheName();
511             if (cacheName != null) // Region specific override with a given cache name
512                templateCacheName = cacheName; 
513             else // Region specific override without cache name, so template cache name is generic for data type.
514                templateCacheName = typeOverrides.get(typeKey).getCacheName(); 
515          } else {
516             // No region specific overrides, template cache name is generic for data type.
517             templateCacheName = typeOverrides.get(typeKey).getCacheName();
518             regionCacheCfg = typeOverrides.get(typeKey).createInfinispanConfiguration();
519          }
520          // Configure transaction manager
521          regionCacheCfg = configureTransactionManager(regionCacheCfg, templateCacheName, properties);
522          // Apply overrides
523          manager.defineConfiguration(regionName, templateCacheName, regionCacheCfg);
524          definedConfigurations.add(regionName);
525       }
526       AdvancedCache cache = manager.getCache(regionName).getAdvancedCache();
527       if (!cache.getStatus().allowInvocations()) {
528          cache.start();
529       }
530       return createCacheWrapper(cache);
531    }

533    private CacheCommandFactory getCacheCommandFactory(AdvancedCache cache) {
534       GlobalComponentRegistry globalCr = cache.getComponentRegistry()
535             .getGlobalComponentRegistry();
536 
537       Map<Byte, ModuleCommandFactory> factories =
538          (Map<Byte, ModuleCommandFactory>) globalCr
539                .getComponent("org.infinispan.modules.command.factories");
540 
541       for (ModuleCommandFactory factory : factories.values()) {
542          if (factory instanceof CacheCommandFactory)
543             return (CacheCommandFactory) factory;
544       }
545 
546       throw new CacheException("Infinispan custom cache command factory not " +
547             "installed (possibly because the classloader where Infinispan " +
548             "lives couldn't find the Hibernate Infinispan cache provider)");
549    }

555    private Configuration configureTransactionManager(Configuration regionOverrides, String templateCacheName, Properties properties) {
556       // Get existing configuration to verify whether a tm was configured or not.
557       Configuration templateConfig = manager.defineConfiguration(templateCacheName, new Configuration());
558       if (templateConfig.isTransactionalCache()) {
559          String ispnTmLookupClassName = templateConfig.getTransactionManagerLookupClass();
560          String hbTmLookupClassName = org.hibernate.cache.infinispan.tm.HibernateTransactionManagerLookup.class.getName();
561          if (ispnTmLookupClassName != null && !ispnTmLookupClassName.equals(hbTmLookupClassName)) {
562             log.debug("Infinispan is configured [" + ispnTmLookupClassName + "] with a different transaction manager lookup " +
563                             "class than Hibernate [" + hbTmLookupClassName + "]");
564          } else {
565             regionOverrides.setTransactionManagerLookup(transactionManagerlookup);
566          }
567 
568          String useSyncProp = extractProperty(INFINISPAN_USE_SYNCHRONIZATION_PROP, properties);
569          boolean useSync = useSyncProp == null ? DEF_USE_SYNCHRONIZATION : Boolean.parseBoolean(useSyncProp);
570          regionOverrides.fluent().transaction().useSynchronization(useSync);
571       }
572 
573       return regionOverrides;
574    }

576    private TypeOverrides overrideStatisticsIfPresent(TypeOverrides override, Properties properties) {
577       String globalStats = extractProperty(INFINISPAN_GLOBAL_STATISTICS_PROP, properties);
578       if (globalStats != null) {
579          override.setExposeStatistics(Boolean.parseBoolean(globalStats));
580       }
581       return override;

39  /**
40   * Represents the state of "stuff" Hibernate is tracking, including (not exhaustive):
41   * <ul>
42   *     <li>entities</li>
43   *     <li>collections</li>
44   *     <li>snapshots</li>
45   *     <li>proxies</li>
46   * </ul>
47   * <p/>
48   * Often referred to as the "first level cache".
49   * 
50   * @author Gavin King
51   * @author Steve Ebersole
52   */

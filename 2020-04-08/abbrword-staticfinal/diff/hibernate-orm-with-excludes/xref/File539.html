44  /**
45   * Encapsulates logic to allow a {@link TransactionalAccessDelegate} to determine
46   * whether a {@link TransactionalAccessDelegate#putFromLoad(Object, Object, long, Object, boolean)}
47   * call should be allowed to update the cache. A <code>putFromLoad</code> has
48   * the potential to store stale data, since the data may have been removed from the
49   * database and the cache between the time when the data was read from the database 
50   * and the actual call to <code>putFromLoad</code>.
51   * <p>
52   * The expected usage of this class by a thread that read the cache and did
53   * not find data is:
54   *
55   * <ol>
56   * <li> Call {@link #registerPendingPut(Object)}</li>
57   * <li> Read the database</li>
58   * <li> Call {@link #acquirePutFromLoadLock(Object)}
59   * <li> if above returns <code>false</code>, the thread should not cache the data;
60   *      only if above returns <code>true</code>, put data in the cache and...</li>
61   * <li> then call {@link #releasePutFromLoadLock(Object)}</li>
62   * </ol>
63   * </p>
64   *
65   * <p>
66   * The expected usage by a thread that is taking an action such that any pending
67   * <code>putFromLoad</code> may have stale data and should not cache it is to either
68   * call
69   *
70   * <ul>
71   * <li> {@link #invalidateKey(Object)} (for a single key invalidation)</li>
72   * <li>or {@link #invalidateRegion()} (for a general invalidation all pending puts)</li>
73   * </ul>
74   * </p>
75   *
76   * <p>
77   * This class also supports the concept of "naked puts", which are calls to
78   * {@link #acquirePutFromLoadLock(Object)} without a preceding {@link #registerPendingPut(Object)}
79   * call.
80   * </p>
81   *
82   * @author Brian Stansberry
83   * 
84   * @version $Revision: $
85   */
86  public class PutFromLoadValidator {
87     /**
88      * Period (in ms) after a removal during which a call to
89      * {@link #acquirePutFromLoadLock(Object)} that hasn't been
90      * {@link #registerPendingPut(Object) pre-registered} (aka a "naked put")
91      * will return false.
92      * will return false.
93      */
94     public static final long NAKED_PUT_INVALIDATION_PERIOD = TimeUnit.SECONDS.toMillis(20); 
95  
96     /**
97      * Used to determine whether the owner of a pending put is a thread or a transaction
98      */
99     private final TransactionManager transactionManager;
100 
101    private final long nakedPutInvalidationPeriod;
102 
103    /**
104     * Registry of expected, future, isPutValid calls. If a key+owner is registered in this map, it
105     * is not a "naked put" and is allowed to proceed.
106     */
107    private final ConcurrentMap<Object, PendingPutMap> pendingPuts;
108 
109    private final ConcurrentMap<Object, Long> recentRemovals = new ConcurrentHashMap<Object, Long>();
110    /**
111     * List of recent removals. Used to ensure we don't leak memory via the recentRemovals map
112     */
113    private final List<RecentRemoval> removalsQueue = new LinkedList<RecentRemoval>();
114    /**
115     * The time when the first element in removalsQueue will expire. No reason to do housekeeping on
116     * the queue before this time.
117     */
118    private volatile long earliestRemovalTimestamp;
119    /** Lock controlling access to removalsQueue */
120    private final Lock removalsLock = new ReentrantLock();
121 
122    /**
123     * The time of the last call to regionRemoved(), plus NAKED_PUT_INVALIDATION_PERIOD. All naked
124     * puts will be rejected until the current time is greater than this value.
125     */
126    private volatile long invalidationTimestamp;
127 
128    /**
129     * Creates a new PutFromLoadValidator.
130     */
131    public PutFromLoadValidator(AdvancedCache cache) {
132       this(cache, NAKED_PUT_INVALIDATION_PERIOD);
133    }

144    public PutFromLoadValidator(EmbeddedCacheManager cacheManager,
145          TransactionManager tm, long nakedPutInvalidationPeriod) {
146       this.pendingPuts = cacheManager
147             .getCache(InfinispanRegionFactory.PENDING_PUTS_CACHE_NAME);
148       this.transactionManager = tm;
149       this.nakedPutInvalidationPeriod = nakedPutInvalidationPeriod;
150    }

154    /**
155     * Acquire a lock giving the calling thread the right to put data in the
156     * cache for the given key.
157     * <p>
158     * <strong>NOTE:</strong> A call to this method that returns <code>true</code>
159     * should always be matched with a call to {@link #releasePutFromLoadLock(Object)}.
160     * </p>
161     *
162     * @param key the key
163     *
164     * @return <code>true</code> if the lock is acquired and the cache put
165     *         can proceed; <code>false</code> if the data should not be cached
166     */
167 	public boolean acquirePutFromLoadLock(Object key) {
168       boolean valid = false;
169       boolean locked = false;
170       long now = System.currentTimeMillis();
171 
172       try {
173          PendingPutMap pending = pendingPuts.get(key);
174          if (pending != null) {
175             locked = pending.acquireLock(100, TimeUnit.MILLISECONDS);
176             if (locked) {
177                try {
178                   PendingPut toCancel = pending.remove(getOwnerForPut());
179                   if (toCancel != null) {
180                      valid = !toCancel.completed;
181                      toCancel.completed = true;
182                   }
183                }
184                finally {
185                   if (!valid) {
186                      pending.releaseLock();
187                      locked = false;
188                   }
189                }
190             }
191          }
192          else {
193             // Key wasn't in pendingPuts, so either this is a "naked put"
194             // or regionRemoved has been called. Check if we can proceed
195             if (now > invalidationTimestamp) {
196                Long removedTime = recentRemovals.get(key);
197                if (removedTime == null || now > removedTime ) {
198                   // It's legal to proceed. But we have to record this key
199                   // in pendingPuts so releasePutFromLoadLock can find it.
200                   // To do this we basically simulate a normal "register
201                   // then acquire lock" pattern
202                   registerPendingPut(key);
203                   locked = acquirePutFromLoadLock(key);
204                   valid = locked;
205                }
206             }
207          }
208       }
209       catch (Throwable t) {
210          if (locked) {
211             PendingPutMap toRelease = pendingPuts.get(key);
212             if (toRelease != null) {
213                toRelease.releaseLock();
214             }
215          }
216 
217          if (t instanceof RuntimeException) {
218             throw (RuntimeException) t;
219          } else if (t instanceof Error) {
220             throw (Error) t;
221          } else {
222             throw new RuntimeException(t);
223          }
224       }
225 
226       return valid;
227    }

229    /**
230     * Releases the lock previously obtained by a call to
231     * {@link #acquirePutFromLoadLock(Object)} that returned <code>true</code>.
232     *
233     * @param key the key
234     */
235    public void releasePutFromLoadLock(Object key) {
236       PendingPutMap pending = pendingPuts.get(key);
237       if (pending != null) {
238          if (pending.size() == 0) {
239             pendingPuts.remove(key, pending);
240          }
241          pending.releaseLock();
242       }
243    }

245    /**
246     * Invalidates any {@link #registerPendingPut(Object) previously registered pending puts} ensuring a subsequent call to
247     * {@link #acquirePutFromLoadLock(Object)} will return <code>false</code>. <p> This method will block until any
248     * concurrent thread that has {@link #acquirePutFromLoadLock(Object) acquired the putFromLoad lock} for the given key
249     * has released the lock. This allows the caller to be certain the putFromLoad will not execute after this method
250     * returns, possibly caching stale data. </p>
251     *
252     * @param key key identifying data whose pending puts should be invalidated
253     * @return <code>true</code> if the invalidation was successful; <code>false</code> if a problem occured (which the
254     *         caller should treat as an exception condition)
255     */
256    public boolean invalidateKey(Object key) {
257       boolean success = true;
258 
259       // Invalidate any pending puts
260       PendingPutMap pending = pendingPuts.get(key);
261       if (pending != null) {
262          // This lock should be available very quickly, but we'll be
263          // very patient waiting for it as callers should treat not
264          // acquiring it as an exception condition
265          if (pending.acquireLock(60, TimeUnit.SECONDS)) {
266             try {
267                pending.invalidate();
268             }
269             finally {
270                pending.releaseLock();
271             }
272          } else {
273             success = false;
274          }
275       }
276 
277       // Record when this occurred to invalidate later naked puts
278       RecentRemoval removal = new RecentRemoval(key, this.nakedPutInvalidationPeriod);
279       recentRemovals.put(key, removal.timestamp);
280 
281       // Don't let recentRemovals map become a memory leak
282       RecentRemoval toClean = null;
283       boolean attemptClean = removal.timestamp > earliestRemovalTimestamp;
284       removalsLock.lock();
285       try {
286          removalsQueue.add(removal);
287 
288          if (attemptClean) {
289             if (removalsQueue.size() > 1) { // we have at least one as we
290                // just added it
291                toClean = removalsQueue.remove(0);
292             }
293             earliestRemovalTimestamp = removalsQueue.get( 0 ).timestamp;
294          }
295       } finally {
296          removalsLock.unlock();
297       }
298 
299       if (toClean != null) {
300          Long cleaned = recentRemovals.get(toClean.key);
301          if (cleaned != null && cleaned.equals(toClean.timestamp)) {
302             cleaned = recentRemovals.remove(toClean.key);
303             if (cleaned != null && !cleaned.equals(toClean.timestamp)) {
304                // Oops; removed the wrong timestamp; restore it
305                recentRemovals.putIfAbsent(toClean.key, cleaned);
306             }
307          }
308       }
309 
310       return success;
311    }

313    /**
314     * Invalidates all {@link #registerPendingPut(Object) previously registered pending puts} ensuring a subsequent call to
315     * {@link #acquirePutFromLoadLock(Object)} will return <code>false</code>. <p> This method will block until any
316     * concurrent thread that has {@link #acquirePutFromLoadLock(Object) acquired the putFromLoad lock} for the any key has
317     * released the lock. This allows the caller to be certain the putFromLoad will not execute after this method returns,
318     * possibly caching stale data. </p>
319     *
320     * @return <code>true</code> if the invalidation was successful; <code>false</code> if a problem occured (which the
321     *         caller should treat as an exception condition)
322     */
323    public boolean invalidateRegion() {
324 
325       boolean ok = false;
326       invalidationTimestamp = System.currentTimeMillis() + this.nakedPutInvalidationPeriod;
327 
328       try {
329 
330          // Acquire the lock for each entry to ensure any ongoing
331          // work associated with it is completed before we return
332          for (PendingPutMap entry : pendingPuts.values()) {
333             if (entry.acquireLock(60, TimeUnit.SECONDS)) {
334                try {
335                   entry.invalidate();
336                }
337                finally {
338                   entry.releaseLock();
339                }
340             } else {
341                ok = false;
342             }
343          }
344 
345          removalsLock.lock();
346          try {
347             recentRemovals.clear();
348             removalsQueue.clear();
349 
350             ok = true;
351 
352          } finally {
353             removalsLock.unlock();
354          }
355       }
356       catch (Exception e) {
357          ok = false;
358       }
359       finally {
360          earliestRemovalTimestamp = invalidationTimestamp;
361       }
362 
363       return ok;
364 	}

366    /**
367     * Notifies this validator that it is expected that a database read followed by a subsequent {@link
368     * #acquirePutFromLoadLock(Object)} call will occur. The intent is this method would be called following a cache miss
369     * wherein it is expected that a database read plus cache put will occur. Calling this method allows the validator to
370     * treat the subsequent <code>acquirePutFromLoadLock</code> as if the database read occurred when this method was
371     * invoked. This allows the validator to compare the timestamp of this call against the timestamp of subsequent removal
372     * notifications. A put that occurs without this call preceding it is "naked"; i.e the validator must assume the put is
373     * not valid if any relevant removal has occurred within {@link #NAKED_PUT_INVALIDATION_PERIOD} milliseconds.
374     *
375     * @param key key that will be used for subsequent cache put
376     */
377    public void registerPendingPut(Object key) {
378       PendingPut pendingPut = new PendingPut(getOwnerForPut());
379       PendingPutMap pendingForKey = new PendingPutMap(pendingPut);
380 
381       for (;;) {
382          PendingPutMap existing = pendingPuts.putIfAbsent(key, pendingForKey);
383          if (existing != null) {
384             if (existing.acquireLock(10, TimeUnit.SECONDS)) {
385 
386                try {
387                   existing.put(pendingPut);
388                   PendingPutMap doublecheck = pendingPuts.putIfAbsent(key, existing);
389                   if (doublecheck == null || doublecheck == existing) {
390                      break;
391                   }
392                   // else we hit a race and need to loop to try again
393                }
394                finally {
395                   existing.releaseLock();
396                }
397             } else {
398                // Can't get the lock; when we come back we'll be a "naked put"
399                break;
400             }
401          } else {
402             // normal case
403             break;
404          }
405       }
406    }

410    /** Only for use by unit tests; may be removed at any time */
411    protected int getRemovalQueueLength() {
412       removalsLock.lock();
413       try {
414          return removalsQueue.size();
415       } finally {
416          removalsLock.unlock();
417       }
418    }

422    private Object getOwnerForPut() {
423       Transaction tx = null;
424       try {
425          if (transactionManager != null) {
426             tx = transactionManager.getTransaction();
427          }
428       } catch (SystemException se) {
429          throw new CacheException("Could not obtain transaction", se);
430       }
431       return tx == null ? Thread.currentThread() : tx;
432 
433    }

435    /**
436     * Lazy-initialization map for PendingPut. Optimized for the expected usual case where only a
437     * single put is pending for a given key.
438     * 
439     * This class is NOT THREAD SAFE. All operations on it must be performed with the lock held.
440     */
441    private static class PendingPutMap {
442       private PendingPut singlePendingPut;
443       private Map<Object, PendingPut> fullMap;
444       private final Lock lock = new ReentrantLock();
445 
446       PendingPutMap(PendingPut singleItem) {
447          this.singlePendingPut = singleItem;
448       }
449 
450       public void put(PendingPut pendingPut) {
451          if (singlePendingPut == null) {
452             if (fullMap == null) {
453                // initial put
454                singlePendingPut = pendingPut;
455             } else {
456                fullMap.put(pendingPut.owner, pendingPut);
457             }
458          } else {
459             // 2nd put; need a map
460             fullMap = new HashMap<Object, PendingPut>(4);
461             fullMap.put(singlePendingPut.owner, singlePendingPut);
462             singlePendingPut = null;
463             fullMap.put(pendingPut.owner, pendingPut);
464          }
465       }
466 
467       public PendingPut remove(Object ownerForPut) {
468          PendingPut removed = null;
469          if (fullMap == null) {
470             if (singlePendingPut != null
471                   && singlePendingPut.owner.equals(ownerForPut)) {
472                removed = singlePendingPut;
473                singlePendingPut = null;
474             }
475          } else {
476             removed = fullMap.remove(ownerForPut);
477          }
478          return removed;
479       }
480 
481       public int size() {
482          return fullMap == null ? (singlePendingPut == null ? 0 : 1)
483                : fullMap.size();
484       }
485 
486       public boolean acquireLock(long time, TimeUnit unit) {
487          try {
488             return lock.tryLock(time, unit);
489          } catch (InterruptedException e) {
490             Thread.currentThread().interrupt();
491             return false;
492          }
493       }
494 
495       public void releaseLock() {
496          lock.unlock();
497       }
498 
499       public void invalidate() {
500          if (singlePendingPut != null) {
501             singlePendingPut.completed = true;
502             // Nullify to avoid leaking completed pending puts
503             singlePendingPut = null;
504          } else if (fullMap != null) {
505             for (PendingPut pp : fullMap.values()) {
506                pp.completed = true;
507             }
508             // Nullify to avoid leaking completed pending puts
509             fullMap = null;
510          }
511       }
512    }

514    private static class PendingPut {
515       private final Object owner;
516       private volatile boolean completed;
517 
518       private PendingPut(Object owner) {
519          this.owner = owner;
520       }
521    }

523    private static class RecentRemoval {
524       private final Object key;
525       private final Long timestamp;
526 
527       private RecentRemoval(Object key, long nakedPutInvalidationPeriod) {
528          this.key = key;
529          timestamp = System.currentTimeMillis() + nakedPutInvalidationPeriod;
530       }

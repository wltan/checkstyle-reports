24  package org.hibernate.metamodel.relational;
25  
26  import java.util.HashSet;
27  import java.util.Set;
28  import java.util.UUID;
29  import java.util.concurrent.atomic.AtomicInteger;
30  
31  import org.hibernate.dialect.Dialect;
32  
33  /**
34   * Convenience base class for {@link org.hibernate.mapping.AuxiliaryDatabaseObject}s.
35   * <p/>
36   * This implementation performs dialect scoping checks strictly based on
37   * dialect name comparisons.  Custom implementations might want to do
38   * instanceof-type checks.
39   *
40   * @author Steve Ebersole
41   */
42  public abstract class AbstractAuxiliaryDatabaseObject implements AuxiliaryDatabaseObject {
43  	// Use a UUID in identifier prefix because this object is not qualified by a schema/catalog
44  	// (not sure this matters...)
45  	private static final String EXPORT_IDENTIFIER_PREFIX = "auxiliary-object-" + UUID.randomUUID();
46  	private static final AtomicInteger counter = new AtomicInteger( 0 );
47  	private final String exportIdentifier;
48  	private final Set<String> dialectScopes;
49  
50  	protected AbstractAuxiliaryDatabaseObject(Set<String> dialectScopes) {
51  		this.dialectScopes =  dialectScopes == null ? new HashSet<String>() : dialectScopes;
52  		this.exportIdentifier =
53  				new StringBuilder( EXPORT_IDENTIFIER_PREFIX )
54  						.append( '.' )
55  						.append( counter.getAndIncrement() )
56  						.toString();
57  	}
58  
59  	public void addDialectScope(String dialectName) {
60  		dialectScopes.add( dialectName );
61  	}
62  
63  	public Iterable<String> getDialectScopes() {
64  		return dialectScopes;
65  	}
66  
67  	public boolean appliesToDialect(Dialect dialect) {
68  		// empty means no scoping
69  		return dialectScopes.isEmpty() || dialectScopes.contains( dialect.getClass().getName() );
70  	}
71  
72  	@Override
73  	public String getExportIdentifier() {
74  		return exportIdentifier;
75  	}
76  }

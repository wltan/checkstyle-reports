36  /** This class is a container for a single set of TLE data.
37   *
38   * <p>TLE sets can be built either by providing directly the two lines, in
39   * which case parsing is performed internally or by providing the already
40   * parsed elements.</p>
41   * <p>TLE are not transparently convertible to {@link org.orekit.orbits.Orbit Orbit}
42   * instances. They are significant only with respect to their dedicated {@link
43   * TLEPropagator propagator}, which also computes position and velocity coordinates.
44   * Any attempt to directly use orbital parameters like {@link #getE() eccentricity},
45   * {@link #getI() inclination}, etc. without any reference to the {@link TLEPropagator
46   * TLE propagator} is prone to errors.</p>
47   * <p>More information on the TLE format can be found on the
48   * <a href="http://www.celestrak.com/">CelesTrak website.</a></p>
49   * @author Fabien Maussion
50   * @author Luc Maisonobe
51   */
52  public class TLE implements TimeStamped, Serializable {

149     /** Simple constructor from unparsed two lines.
150      * <p>The static method {@link #isFormatOK(String, String)} should be called
151      * before trying to build this object.<p>
152      * @param line1 the first element (69 char String)
153      * @param line2 the second element (69 char String)
154      * @exception OrekitException if some format error occurs or lines are inconsistent
155      */
156     public TLE(final String line1, final String line2) throws OrekitException {
157 
158         // identification
159         satelliteNumber = parseInteger(line1, 2, 5);
160         final int satNum2 = parseInteger(line2, 2, 5);
161         if (satelliteNumber != satNum2) {
162             throw new OrekitException(OrekitMessages.TLE_LINES_DO_NOT_REFER_TO_SAME_OBJECT,
163                                       line1, line2);
164         }
165         classification  = line1.charAt(7);
166         launchYear      = parseYear(line1, 9);
167         launchNumber    = parseInteger(line1, 11, 3);
168         launchPiece     = line1.substring(14, 17).trim();
169         ephemerisType   = parseInteger(line1, 62, 1);
170         elementNumber   = parseInteger(line1, 64, 4);
171 
172         // Date format transform (nota: 27/31250 == 86400/100000000)
173         final int    year      = parseYear(line1, 18);
174         final int    dayInYear = parseInteger(line1, 20, 3);
175         final long   df        = 27l * parseInteger(line1, 24, 8);
176         final int    secondsA  = (int) (df / 31250l);
177         final double secondsB  = (df % 31250l) / 31250.0;
178         epoch = new AbsoluteDate(new DateComponents(year, dayInYear),
179                                  new TimeComponents(secondsA, secondsB),
180                                  TimeScalesFactory.getUTC());
181 
182         // mean motion development
183         // converted from rev/day, 2 * rev/day^2 and 6 * rev/day^3 to rad/s, rad/s^2 and rad/s^3
184         meanMotion                 = parseDouble(line2, 52, 11) * FastMath.PI / 43200.0;
185         meanMotionFirstDerivative  = parseDouble(line1, 33, 10) * FastMath.PI / 1.86624e9;
186         meanMotionSecondDerivative = Double.parseDouble((line1.substring(44, 45) + '.' +
187                                                          line1.substring(45, 50) + 'e' +
188                                                          line1.substring(50, 52)).replace(' ', '0')) *
189                                      FastMath.PI / 5.3747712e13;
190 
191         eccentricity = Double.parseDouble("." + line2.substring(26, 33).replace(' ', '0'));
192         inclination  = FastMath.toRadians(parseDouble(line2, 8, 8));
193         pa           = FastMath.toRadians(parseDouble(line2, 34, 8));
194         raan         = FastMath.toRadians(Double.parseDouble(line2.substring(17, 25).replace(' ', '0')));
195         meanAnomaly  = FastMath.toRadians(parseDouble(line2, 43, 8));
196 
197         revolutionNumberAtEpoch = parseInteger(line2, 63, 5);
198         bStar = Double.parseDouble((line1.substring(53, 54) + '.' +
199                                     line1.substring(54, 59) + 'e' +
200                                     line1.substring(59, 61)).replace(' ', '0'));
201 
202         // save the lines
203         this.line1 = line1;
204         this.line2 = line2;
205 
206     }

290     /** Build the line 1 from the parsed elements.
291      * @exception OrekitException if UTC conversion cannot be done or
292      * some parameter is too large to fit format
293      */
294     private void buildLine1()
295         throws OrekitException {
296 
297         final StringBuffer buffer = new StringBuffer();
298 
299         buffer.append('1');
300 
301         buffer.append(' ');
302         buffer.append(addPadding("satelliteNumber-1", satelliteNumber, '0', 5, true));
303         buffer.append(classification);
304 
305         buffer.append(' ');
306         buffer.append(addPadding("launchYear",   launchYear % 100, '0', 2, true));
307         buffer.append(addPadding("launchNumber", launchNumber, '0', 3, true));
308         buffer.append(addPadding("launchPiece",  launchPiece, ' ', 3, false));
309 
310         buffer.append(' ');
311         final DateTimeComponents dtc = epoch.getComponents(TimeScalesFactory.getUTC());
312         buffer.append(addPadding("year", dtc.getDate().getYear() % 100, '0', 2, true));
313         buffer.append(addPadding("day",  dtc.getDate().getDayOfYear(),  '0', 3, true));
314         buffer.append('.');
315         // nota: 31250/27 == 100000000/86400
316         final int fraction = (int) FastMath.rint(31250 * dtc.getTime().getSecondsInUTCDay() / 27.0);
317         buffer.append(addPadding("fraction", fraction,  '0', 8, true));
318 
319         buffer.append(' ');
320         final double n1 = meanMotionFirstDerivative * 1.86624e9 / FastMath.PI;
321         final String sn1 = addPadding("meanMotionFirstDerivative",
322                                       new DecimalFormat(".00000000", SYMBOLS).format(n1), ' ', 10, true);
323         buffer.append(sn1);
324 
325         buffer.append(' ');
326         final double n2 = meanMotionSecondDerivative * 5.3747712e13 / FastMath.PI;
327         buffer.append(formatExponentMarkerFree("meanMotionSecondDerivative", n2, 5, ' ', 8, true));
328 
329         buffer.append(' ');
330         buffer.append(formatExponentMarkerFree("B*", bStar, 5, ' ', 8, true));
331 
332         buffer.append(' ');
333         buffer.append(ephemerisType);
334 
335         buffer.append(' ');
336         buffer.append(addPadding("elementNumber", elementNumber, ' ', 4, true));
337 
338         buffer.append(Integer.toString(checksum(buffer)));
339 
340         line1 = buffer.toString();
341 
342     }

344     /** Format a real number without 'e' exponent marker.
345      * @param name parameter name
346      * @param d number to format
347      * @param mantissaSize size of the mantissa (not counting initial '-' or ' ' for sign)
348      * @param c padding character
349      * @param size desired size
350      * @param rightJustified if true, the resulting string is
351      * right justified (i.e. space are added to the left)
352      * @return formatted and padded number
353      * @exception OrekitException if parameter is too large to fit format
354      */
355     private String formatExponentMarkerFree(final String name, final double d, final int mantissaSize,
356                                             final char c, final int size, final boolean rightJustified)
357         throws OrekitException {
358         final double dAbs = FastMath.abs(d);
359         int exponent = (dAbs < 1.0e-9) ? -9 : (int) FastMath.ceil(FastMath.log10(dAbs));
360         final long mantissa = FastMath.round(dAbs * FastMath.pow(10.0, mantissaSize - exponent));
361         if (mantissa == 0) {
362             exponent = 0;
363         }
364         final String sMantissa = addPadding(name, (int) mantissa, '0', mantissaSize, true);
365         final String sExponent = Integer.toString(FastMath.abs(exponent));
366         final String formatted = (d <  0 ? '-' : ' ') + sMantissa + (exponent <= 0 ? '-' : '+') + sExponent;
367 
368         return addPadding(name, formatted, c, size, rightJustified);
369 
370     }

372     /** Build the line 2 from the parsed elements.
373      * @exception OrekitException if some parameter is too large to fit format
374      */
375     private void buildLine2() throws OrekitException {
376 
377         final StringBuffer buffer = new StringBuffer();
378         final DecimalFormat f34   = new DecimalFormat("##0.0000", SYMBOLS);
379         final DecimalFormat f211  = new DecimalFormat("#0.00000000", SYMBOLS);
380 
381         buffer.append('2');
382 
383         buffer.append(' ');
384         buffer.append(addPadding("satelliteNumber-2", satelliteNumber, '0', 5, true));
385 
386         buffer.append(' ');
387         buffer.append(addPadding("inclination", f34.format(FastMath.toDegrees(inclination)), ' ', 8, true));
388         buffer.append(' ');
389         buffer.append(addPadding("raan", f34.format(FastMath.toDegrees(raan)), ' ', 8, true));
390         buffer.append(' ');
391         buffer.append(addPadding("eccentricity", (int) FastMath.rint(eccentricity * 1.0e7), '0', 7, true));
392         buffer.append(' ');
393         buffer.append(addPadding("pa", f34.format(FastMath.toDegrees(pa)), ' ', 8, true));
394         buffer.append(' ');
395         buffer.append(addPadding("meanAnomaly", f34.format(FastMath.toDegrees(meanAnomaly)), ' ', 8, true));
396 
397         buffer.append(' ');
398         buffer.append(addPadding("meanMotion", f211.format(meanMotion * 43200.0 / FastMath.PI), ' ', 11, true));
399         buffer.append(addPadding("revolutionNumberAtEpoch", revolutionNumberAtEpoch, ' ', 5, true));
400 
401         buffer.append(Integer.toString(checksum(buffer)));
402 
403         line2 = buffer.toString();
404 
405     }

407     /** Add padding characters before an integer.
408      * @param name parameter name
409      * @param k integer to pad
410      * @param c padding character
411      * @param size desired size
412      * @param rightJustified if true, the resulting string is
413      * right justified (i.e. space are added to the left)
414      * @return padded string
415      * @exception OrekitException if parameter is too large to fit format
416      */
417     private String addPadding(final String name, final int k, final char c,
418                               final int size, final boolean rightJustified)
419         throws OrekitException {
420         return addPadding(name, Integer.toString(k), c, size, rightJustified);
421     }
422 
423     /** Add padding characters to a string.
424      * @param name parameter name
425      * @param string string to pad
426      * @param c padding character
427      * @param size desired size
428      * @param rightJustified if true, the resulting string is
429      * right justified (i.e. space are added to the left)
430      * @return padded string
431      * @exception OrekitException if parameter is too large to fit format
432      */
433     private String addPadding(final String name, final String string, final char c,
434                               final int size, final boolean rightJustified)
435         throws OrekitException {
436 
437         if (string.length() > size) {
438             throw new OrekitException(OrekitMessages.TLE_INVALID_PARAMETER,
439                                       satelliteNumber, name, string);
440         }
441 
442         final StringBuffer padding = new StringBuffer();
443         for (int i = 0; i < size; ++i) {
444             padding.append(c);
445         }
446 
447         if (rightJustified) {
448             final String concatenated = padding + string;
449             final int l = concatenated.length();
450             return concatenated.substring(l - size, l);
451         }
452 
453         return (string + padding).substring(0, size);
454 
455     }

457     /** Parse a double.
458      * @param line line to parse
459      * @param start start index of the first character
460      * @param length length of the string
461      * @return value of the double
462      */
463     private double parseDouble(final String line, final int start, final int length) {
464         final String field = line.substring(start, start + length).trim();
465         return field.length() > 0 ? Double.parseDouble(field.replace(' ', '0')) : 0;
466     }

476         return field.length() > 0 ? Integer.parseInt(field.replace(' ', '0')) : 0;

479     /** Parse a year written on 2 digits.
480      * @param line line to parse
481      * @param start start index of the first character
482      * @return value of the year
483      */
484     private int parseYear(final String line, final int start) {
485         final int year = 2000 + parseInteger(line, start, 2);
486         return (year > 2056) ? (year - 100) : year;

629     /** Check the lines format validity.
630      * @param line1 the first element
631      * @param line2 the second element
632      * @return true if format is recognized (non null lines, 69 characters length,
633      * line content), false if not
634      * @exception OrekitException if checksum is not valid
635      */
636     public static boolean isFormatOK(final String line1, final String line2)
637         throws OrekitException {
638 
639         if (line1 == null || line1.length() != 69 ||
640             line2 == null || line2.length() != 69) {
641             return false;
642         }
643 
644         if (!(LINE_1_PATTERN.matcher(line1).matches() &&
645               LINE_2_PATTERN.matcher(line2).matches())) {
646             return false;
647         }
648 
649         // check sums
650         final int checksum1 = checksum(line1);
651         if (Integer.parseInt(line1.substring(68)) != (checksum1 % 10)) {
652             throw new OrekitException(OrekitMessages.TLE_CHECKSUM_ERROR,
653                                       1, line1.substring(68), checksum1 % 10, line1);
654         }
655 
656         final int checksum2 = checksum(line2);
657         if (Integer.parseInt(line2.substring(68)) != (checksum2 % 10)) {
658             throw new OrekitException(OrekitMessages.TLE_CHECKSUM_ERROR,
659                                       2, line2.substring(68), checksum2 % 10, line2);
660         }
661 
662         return true;
663 
664     }

666     /** Compute the checksum of the first 68 characters of a line.
667      * @param line line to check
668      * @return checksum
669      */
670     private static int checksum(final CharSequence line) {
671         int sum = 0;
672         for (int j = 0; j < 68; j++) {
673             final char c = line.charAt(j);
674             if (Character.isDigit(c)) {
675                 sum += Character.digit(c, 10);
676             } else if (c == '-') {
677                 ++sum;
678             }
679         }
680         return sum % 10;
681     }

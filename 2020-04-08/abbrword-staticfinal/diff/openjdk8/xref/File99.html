25  package com.apple.internal.jobjc.generator.classes;
26  
27  import java.io.PrintStream;
28  import java.util.List;
29  
30  import com.apple.internal.jobjc.generator.ClassGenerator;
31  import com.apple.internal.jobjc.generator.model.Framework;
32  import com.apple.internal.jobjc.generator.utils.JavaLang;
33  import com.apple.internal.jobjc.generator.utils.JavaLang.JLCtor;
34  import com.apple.internal.jobjc.generator.utils.JavaLang.JLField;
35  import com.apple.internal.jobjc.generator.utils.JavaLang.JLMethod;
36  import com.apple.jobjc.JObjCRuntime;
37  
38  public class RootJObjCClass extends GeneratedClassFile {
39      private static final String JOBJC_CLASSNAME = "JObjC";
40      public static final String FULL_JOBJC_CLASSNAME = ClassGenerator.JOBJC_PACKAGE + "." + JOBJC_CLASSNAME;
41      public static final String JOBJC_RUNTIME_INST = FULL_JOBJC_CLASSNAME + ".getInstance()";
42      public static final String JOBJC_RUNTIME_INST_R = FULL_JOBJC_CLASSNAME + ".getInstance(getRuntime())";
43  
44      public static final String runtimeFrameworkInst(String fwname){
45          return JOBJC_RUNTIME_INST + "." + fwname + "()";
46      }
47  
48      public static final String runtimeFrameworkInstR(String fwname){
49          return JOBJC_RUNTIME_INST_R + "." + fwname + "()";
50      }
51  
52      final List<Framework> frameworks;
53  
54      public RootJObjCClass(final List<Framework> frameworks) {
55          super(ClassGenerator.JOBJC_PACKAGE, JOBJC_CLASSNAME, "java.lang.Object");
56          this.frameworks = frameworks;
57      }
58  
59      @Override public void writeBeginning(final PrintStream out) {
60          out.print(new JLField("private static", JOBJC_CLASSNAME, "instance"));
61          out.print(new JLField("private final", JObjCRuntime.class.getName(), "runtime"));
62  
63          JLMethod getInstR = new JLMethod("public static", JOBJC_CLASSNAME, "getInstance", "final JObjCRuntime runtime");
64          getInstR.body.add("if(runtime == null) throw new NullPointerException(\"runtime\");");
65          getInstR.body.add("if(instance == null) instance = new JObjC(runtime);");
66          getInstR.body.add("return instance;");
67          out.print(getInstR);
68  
69          JLMethod getInst = new JLMethod("public static", JOBJC_CLASSNAME, "getInstance");
70          getInst.body.add("return getInstance(JObjCRuntime.getInstance());");
71          out.print(getInst);
72  
73          JLCtor ctor = new JLCtor("private", JOBJC_CLASSNAME, "final JObjCRuntime runtime");
74          ctor.body.add("this.runtime = runtime;");
75          for (final Framework f : frameworks)
76              ctor.body.add("runtime.registerPackage(\"" + f.pkg + "\");");
77          out.print(ctor);
78      }
79  
80      @Override public void writeBody(final PrintStream out) {
81          for (final Framework f : frameworks)
82              out.println(JavaLang.makeSingleton("_" + f.name, f.name, f.pkg + "." + f.name + "Framework", "runtime"));
83      }
84  }

79      private WorkItem(IResource resource, IJavaElement javaElt, IProject project) {

81          this.javaElt = javaElt;
82          this.project = project;
83          Assert.isLegal(resource != null || javaElt != null);

86      public void addFilesToProject(Project fbProject, Map<IPath, IPath> outputLocations) {
87          IResource res = getCorespondingResource();
88          if (res instanceof IProject) {
89              for (IPath outDir : outputLocations.values()) {
90                  fbProject.addFile(outDir.toOSString());
91              }
92          } else if (res instanceof IFolder) {
93              // assumption: this is a source folder.
94              boolean added = addClassesForFolder((IFolder) res, outputLocations, fbProject);
95              if (!added) {
96                  // What if this is a class folder???
97                  addJavaElementPath(fbProject);
98              }
99          } else if (res instanceof IFile) {
100             // ID: 2734173: allow to analyze classes inside archives
101             if (Util.isClassFile(res) || Util.isJavaArchive(res)) {
102                 fbProject.addFile(res.getLocation().toOSString());
103             } else if (Util.isJavaFile(res)) {
104                 addClassesForFile((IFile) res, outputLocations, fbProject);
105             }
106         } else {
107             addJavaElementPath(fbProject);
108         }
109     }
110 
111     private void addJavaElementPath(Project fbProject) {
112         if (javaElt != null) {
113             IPath path = getPath();
114             if (path != null) {
115                 fbProject.addFile(path.toOSString());
116             }
117         }
118     }

120     public void clearMarkers() throws CoreException {
121         IResource res = getMarkerTarget();
122         if (javaElt == null || !(res instanceof IProject)) {
123             MarkerUtil.removeMarkers(res);
124         } else {
125             // this is the case of external class folders/libraries: if we would
126             // cleanup ALL project markers, it would also remove markers from
127             // ALL
128             // source/class files, not only for the selected one.
129             IMarker[] allMarkers = MarkerUtil.getAllMarkers(res);
130             Set<IMarker> set = MarkerUtil.findMarkerForJavaElement(javaElt, allMarkers, true);
131             // TODO can be very slow. May think about batch operation w/o
132             // resource notifications
133             // P.S. if executing "clean+build", package explorer first doesn't
134             // notice
135             // any change because the external class file labels are not
136             // refreshed after clean,
137             // but after build we trigger an explicit view refresh, see
138             // FindBugsAction
139             for (IMarker marker : set) {
140                 marker.delete();
141             }
142         }
143     }

153     /**
154      * @return false if no classes was added
155      */
156     private static boolean addClassesForFolder(IFolder folder, Map<IPath, IPath> outLocations, Project fbProject) {
157         IPath path = folder.getLocation();
158         IPath srcRoot = getMatchingSourceRoot(path, outLocations);
159         if (srcRoot == null) {
160             return false;
161         }
162         IPath outputRoot = outLocations.get(srcRoot);
163         int firstSegments = path.matchingFirstSegments(srcRoot);
164         // add relative path to the output path
165         IPath out = outputRoot.append(path.removeFirstSegments(firstSegments));
166         File directory = out.toFile();
167         return fbProject.addFile(directory.getAbsolutePath());
168         // TODO child directories too. Should add preference???
169     }
170 
171     private static void addClassesForFile(IFile file, Map<IPath, IPath> outLocations, Project fbProject) {
172         IPath path = file.getLocation();
173         IPath srcRoot = getMatchingSourceRoot(path, outLocations);
174         if (srcRoot == null) {
175             return;
176         }
177         IPath outputRoot = outLocations.get(srcRoot);
178         int firstSegments = path.matchingFirstSegments(srcRoot);
179         // add relative path to the output path
180         IPath out = outputRoot.append(path.removeFirstSegments(firstSegments));
181         String fileName = path.removeFileExtension().lastSegment();
182         String namePattern = fileName + "\\.class|" + fileName + "\\$.*\\.class";
183         namePattern = addSecondaryTypesToPattern(file, fileName, namePattern);
184         File directory = out.removeLastSegments(1).toFile();
185 
186         // add parent folder and regexp for file names
187         Pattern classNamesPattern = Pattern.compile(namePattern);
188         ResourceUtils.addFiles(fbProject, directory, classNamesPattern);
189     }

191     /**
192      * Add secondary types patterns (not nested in the type itself but contained
193      * in the java file)
194      *
195      * @param fileName
196      *            java file name (not path!) without .java suffix
197      * @param classNamePattern
198      *            non null pattern for all matching .class file names
199      * @return modified classNamePattern, if there are more then one type
200      *         defined in the java file
201      */
202     private static String addSecondaryTypesToPattern(IFile file, String fileName, String classNamePattern) {
203         ICompilationUnit cu = JavaCore.createCompilationUnitFrom(file);
204         if (cu == null) {
205             FindbugsPlugin.getDefault().logError(
206                     "NULL compilation unit for " + file + ", FB analysis might  be incomplete for included types");
207             return classNamePattern;
208         }
209         try {
210             IType[] types = cu.getTypes();
211             if (types.length > 1) {
212                 StringBuilder sb = new StringBuilder(classNamePattern);
213                 for (IType type : types) {
214                     if (fileName.equals(type.getElementName())) {
215                         // "usual" type with the same name: we have it already
216                         continue;
217                     }
218                     sb.append("|").append(type.getElementName());
219                     sb.append("\\.class|").append(type.getElementName());
220                     sb.append("\\$.*\\.class");
221                 }
222                 classNamePattern = sb.toString();
223             }
224         } catch (JavaModelException e) {
225             FindbugsPlugin.getDefault().logException(e, "Cannot get types from compilation unit: " + cu);
226         }
227         return classNamePattern;
228     }

230     public @CheckForNull
231     IResource getCorespondingResource() {
232         if (resource != null) {
233             return resource;
234         }
235         try {
236             IResource resource1 = javaElt.getCorrespondingResource();
237             if(resource1 != null) {
238                 return resource1;
239             }
240             IJavaElement ancestor = javaElt.getAncestor(IJavaElement.COMPILATION_UNIT);
241             if(ancestor != null){
242                 return ancestor.getCorrespondingResource();
243             }
244         } catch (JavaModelException e) {
245             // ignore, just return nothing
246         }
247         return null;
248     }

258     /**
259      * @return the resource which can be used to attach markers found for this
260      *         item. This resource must exist, and the return value can not be
261      *         null. The return value can be absolutely unrelated to the
262      *         {@link #getCorespondingResource()}.
263      */
264     public @Nonnull
265     IResource getMarkerTarget() {
266         IResource res = getCorespondingResource();
267         if (res != null) {
268             return res;
269         }
270         if (javaElt != null) {
271             IResource resource2 = javaElt.getResource();
272             if (resource2 != null) {
273                 return resource2;
274             }
275         }
276         // probably not the best solution, but this should always work
277         return project;
278     }

288     /**
289      * @return markers which are <b>already</b> reported for given work item
290      */
291     public Set<IMarker> getMarkers(boolean recursive) {
292         IResource res = getCorespondingResource();
293         if (res != null) {
294             if (res.getType() == IResource.PROJECT || javaElt instanceof IPackageFragmentRoot) {
295                 // for project, depth_one does not make any sense here
296                 recursive = true;
297             }
298             IMarker[] markers = MarkerUtil.getMarkers(res, recursive ? IResource.DEPTH_INFINITE : IResource.DEPTH_ONE);
299             return new HashSet<>(Arrays.asList(markers));
300         }
301         IResource markerTarget = getMarkerTarget();
302         if(!markerTarget.isAccessible()) {
303             return Collections.emptySet();
304         }
305         if (!recursive
306                 && ((markerTarget.getType() == IResource.PROJECT && (javaElt instanceof IPackageFragmentRoot) || Util
307                         .isClassFile(javaElt)) || (Util.isJavaArchive(markerTarget) && Util.isClassFile(javaElt)))) {
308             recursive = true;
309         }
310         IMarker[] markers = MarkerUtil.getMarkers(markerTarget, recursive ? IResource.DEPTH_INFINITE : IResource.DEPTH_ONE);
311         Set<IMarker> forJavaElement = MarkerUtil.findMarkerForJavaElement(javaElt, markers, recursive);
312         return forJavaElement;
313     }

315     /**
316      * @param srcPath may be null
317      * @param outLocations
318      *            key is the source root, value is output folder
319      * @return source root folder matching (parent of) given path, or null
320      */
321     private static @Nullable IPath getMatchingSourceRoot(@Nullable IPath srcPath, Map<IPath, IPath> outLocations) {
322         if(srcPath == null) {
323             return null;
324         }
325         Set<Entry<IPath, IPath>> outEntries = outLocations.entrySet();
326         IPath result = null;
327         int maxSegments = 0;
328         for (Entry<IPath, IPath> entry : outEntries) {
329             IPath srcRoot = entry.getKey();
330             int firstSegments = srcPath.matchingFirstSegments(srcRoot);
331             if (firstSegments > maxSegments && firstSegments == srcRoot.segmentCount()) {
332                 maxSegments = firstSegments;
333                 result = srcRoot;
334             }
335         }
336         return result;
337     }

339     public String getName() {
340         return resource != null ? resource.getName() : javaElt.getElementName();
341     }

343     /**
344      *
345      * @return full absolute path corresponding to the work item (file or
346      *         directory). If the work item is a part of an archive, it's the
347      *         path to the archive file. If the work item is a project, it's the
348      *         path to the project root. TODO If the work item is an internal
349      *         java element (method, inner class etc), results are undefined
350      *         yet.
351      */
352     public @CheckForNull
353     IPath getPath() {
354         IResource corespondingResource = getCorespondingResource();
355         if (corespondingResource != null) {
356             return corespondingResource.getLocation();
357         }
358         if (javaElt != null) {
359             return javaElt.getPath();
360         }
361         return null;
362     }
363 
364     public boolean isDirectory() {
365         IResource corespondingResource = getCorespondingResource();
366         if (corespondingResource != null) {
367             return corespondingResource.getType() == IResource.FOLDER || corespondingResource.getType() == IResource.PROJECT;
368         }
369         return false;
370     }
371 
372     /**
373      *
374      * @return true if the given element is contained inside archive
375      */
376     public boolean isFromArchive() {
377         IPath path = getPath();
378         if (path == null) {
379             return false;
380         }
381         File file = path.toFile();
382         if (file.isDirectory()) {
383             return false;
384         }
385         return Archive.isArchiveFileName(file.getName());
386     }

393     @Override
394     public int hashCode() {
395         int result = 1;
396         result = PRIME * result + ((javaElt == null) ? 0 : javaElt.hashCode());
397         result = PRIME * result + ((resource == null) ? 0 : resource.hashCode());
398         return result;
399     }

401     @Override
402     public boolean equals(Object obj) {
403         if (this == obj) {
404             return true;
405         }
406         if (!(obj instanceof WorkItem)) {
407             return false;
408         }
409         WorkItem other = (WorkItem) obj;
410         if (javaElt == null) {
411             if (other.javaElt != null) {
412                 return false;
413             }
414         } else if (!javaElt.equals(other.javaElt)) {
415             return false;
416         }
417         if (resource == null) {
418             if (other.resource != null) {
419                 return false;
420             }
421         } else if (!resource.equals(other.resource)) {
422             return false;
423         }
424         return true;
425     }

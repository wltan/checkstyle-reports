41  /**
42   * Factory for creating ClassDescriptors, MethodDescriptors, and
43   * FieldDescriptors.
44   *
45   * @author David Hovemeyer
46   */
47  public class DescriptorFactory {
48      private static ThreadLocal<DescriptorFactory> instanceThreadLocal = new ThreadLocal<DescriptorFactory>() {
49          @Override
50          protected DescriptorFactory initialValue() {
51              return new DescriptorFactory();
52          }
53      };
54  
55      private final Map<String, ClassDescriptor> classDescriptorMap;
56  
57      private final Map<String, ClassDescriptor> dottedClassDescriptorMap;
58  
59      private final Map<MethodDescriptor, MethodDescriptor> methodDescriptorMap;
60  
61      private final Map<FieldDescriptor, FieldDescriptor> fieldDescriptorMap;
62  
63      private static final ClassDescriptor MODULE_INFO = new ClassDescriptor("module-info");
64  
65      private DescriptorFactory() {
66          this.classDescriptorMap = new HashMap<>();
67          this.dottedClassDescriptorMap = new HashMap<>();
68          this.methodDescriptorMap = new HashMap<>();
69          this.fieldDescriptorMap = new HashMap<>();
70      }

77       * @deprecated this hack is needless for modern JVM, at least Java8

101     public void purge(Collection<ClassDescriptor> unusable) {
102         for (ClassDescriptor c : unusable) {
103             classDescriptorMap.remove(c.getClassName());
104             dottedClassDescriptorMap.remove(c.getClassName().replace('/', '.'));
105         }
106     }

110         return getClassDescriptorForDottedClassName(actualClass.getName());
111     }

113     /**
114      * Get a ClassDescriptor for a class name in VM (slashed) format.
115      *
116      * @param className
117      *            a class name in VM (slashed) format
118      * @return ClassDescriptor for that class
119      */
120     public @Nonnull
121     ClassDescriptor getClassDescriptor(@SlashedClassName String className) {
122         assert className.indexOf('.') == -1;
123         ClassDescriptor classDescriptor = classDescriptorMap.get(className);
124         if (classDescriptor == null) {
125             if (MODULE_INFO.getClassName().equals(className)) {
126                 // don't allow module info to be added to the map,
127                 // which could be used to check referenced classes
128                 return MODULE_INFO;
129             }
130             classDescriptor = new ClassDescriptor(className);
131             classDescriptorMap.put(className, classDescriptor);
132         }
133         return classDescriptor;
134     }
135 
136     /**
137      * Get a ClassDescriptor for a class name in dotted format.
138      *
139      * @param dottedClassName
140      *            a class name in dotted format
141      * @return ClassDescriptor for that class
142      */
143     public ClassDescriptor getClassDescriptorForDottedClassName(@DottedClassName String dottedClassName) {
144         assert dottedClassName != null;
145         ClassDescriptor classDescriptor = dottedClassDescriptorMap.get(dottedClassName);
146         if (classDescriptor == null) {
147             classDescriptor = getClassDescriptor(dottedClassName.replace('.', '/'));
148             dottedClassDescriptorMap.put(dottedClassName, classDescriptor);
149         }
150         return classDescriptor;
151     }

153     public MethodDescriptor getMethodDescriptor(JavaClass jClass, Method method) {
154         return getMethodDescriptor(ClassName.toSlashedClassName(jClass.getClassName()), method.getName(), method.getSignature(),
155                 method.isStatic());
156     }

158     /**
159      * Get a MethodDescriptor.
160      *
161      * @param className
162      *            name of the class containing the method, in VM format (e.g.,
163      *            "java/lang/String")
164      * @param name
165      *            name of the method
166      * @param signature
167      *            signature of the method
168      * @param isStatic
169      *            true if method is static, false otherwise
170      * @return MethodDescriptor
171      */
172     public MethodDescriptor getMethodDescriptor(@SlashedClassName String className, String name, String signature,
173             boolean isStatic) {
174         if (className == null) {
175             throw new NullPointerException("className must be nonnull");
176         }
177         MethodDescriptor methodDescriptor = new MethodDescriptor(className, name, signature, isStatic);
178         MethodDescriptor existing = methodDescriptorMap.get(methodDescriptor);
179         if (existing == null) {
180             methodDescriptorMap.put(methodDescriptor, methodDescriptor);
181             existing = methodDescriptor;
182         }
183         return existing;
184     }

186     public void profile() {
187         int total = 0;
188         int keys = 0;
189         int values = 0;
190         int bad = 0;
191         for (Map.Entry<MethodDescriptor, MethodDescriptor> e : methodDescriptorMap.entrySet()) {
192             total++;
193             if (e.getKey() instanceof MethodInfo) {
194                 keys++;
195             }
196             if (e.getValue() instanceof MethodInfo) {
197                 values++;
198             }
199         }
200         System.out.printf("Descriptor factory: %d/%d/%d%n", keys, values, total);
201 
202     }

204     public void canonicalize(MethodDescriptor m) {
205         MethodDescriptor existing = methodDescriptorMap.get(m);
206         if (m != existing) {
207             methodDescriptorMap.put(m, m);
208         }
209 
210     }

212     public void canonicalize(FieldDescriptor m) {
213         FieldDescriptor existing = fieldDescriptorMap.get(m);
214         if (m != existing) {
215             fieldDescriptorMap.put(m, m);
216         }
217 
218     }

227      *
228      * @param className

231      * @param name
232      *            the name of the field

234      *            the field signature (type)
235      * @param isStatic
236      *            true if field is static, false if not
237      * @return FieldDescriptor

239     public FieldDescriptor getFieldDescriptor(@SlashedClassName String className, String name, String signature, boolean isStatic) {
240         FieldDescriptor fieldDescriptor = new FieldDescriptor(className, name, signature, isStatic);
241         FieldDescriptor existing = fieldDescriptorMap.get(fieldDescriptor);
242         if (existing == null) {
243             fieldDescriptorMap.put(fieldDescriptor, fieldDescriptor);
244             existing = fieldDescriptor;
245         }
246         return existing;
247     }

274     /**
275      * Create a class descriptor from a resource name.
276      *
277      * @param resourceName
278      *            the resource name
279      * @return the class descriptor
280      */
281     public static ClassDescriptor createClassDescriptorFromResourceName(String resourceName) {
282         if (!isClassResource(resourceName)) {
283             throw new IllegalArgumentException("Resource " + resourceName + " is not a class");
284         }
285         return createClassDescriptor(resourceName.substring(0, resourceName.length() - 6));
286     }
287 
288     /**
289      * Create a class descriptor from a field signature
290      *
291      */
292     public static @CheckForNull
293     ClassDescriptor createClassDescriptorFromFieldSignature(String signature) {
294         int start = signature.indexOf('L');
295         if (start < 0) {
296             return null;
297         }
298         int end = signature.indexOf(';', start);
299         if (end < 0) {
300             return null;
301         }
302         return createClassDescriptor(signature.substring(start + 1, end));
303     }

305     /**
306      * Determine whether or not the given resource name refers to a class.
307      *
308      * @param resourceName
309      *            the resource name
310      * @return true if the resource is a class, false otherwise
311      */
312     public static boolean isClassResource(String resourceName) {
313         // This could be more sophisticated.
314         return resourceName.endsWith(".class") && !isModuleInfo(resourceName);
315     }
316 
317     /**
318      * Determine whether or not the given resource name refers to a module-info class.
319      *
320      * @param resourceName
321      *            the resource name
322      * @return true if the resource is a module-info class, false otherwise
323      */
324     public static boolean isModuleInfo(String resourceName) {
325         return resourceName.equals("module-info.class");
326     }
327 
328     public static ClassDescriptor createClassDescriptorFromSignature(String signature) {
329         int length = signature.length();
330         if (length == 0) {
331             throw new IllegalArgumentException("Empty signature");
332         }
333         if (signature.charAt(0) == 'L' && signature.endsWith(";")) {
334             signature = signature.substring(1, signature.length() - 1);
335         }
336         return createClassDescriptor(signature);
337     }
338 
339     public static ClassDescriptor createClassOrObjectDescriptorFromSignature(String signature) {
340         if (signature.charAt(0) == '[') {
341             return createClassDescriptor("java/lang/Object");
342         }
343         return createClassDescriptorFromSignature(signature);
344     }

354     public static ClassDescriptor[] createClassDescriptor(String[] classNames) {
355         if (classNames.length == 0) {
356             return ClassDescriptor.EMPTY_ARRAY;
357         }
358         ClassDescriptor[] result = new ClassDescriptor[classNames.length];
359         for (int i = 0; i < classNames.length; i++) {
360             result[i] = createClassDescriptor(classNames[i]);
361         }
362         return result;
363     }
364 
365     public static ClassDescriptor createClassDescriptorFromDottedClassName(String dottedClassName) {
366         return createClassDescriptor(dottedClassName.replace('.', '/'));
367     }

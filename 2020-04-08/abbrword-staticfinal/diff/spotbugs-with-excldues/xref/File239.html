20  package edu.umd.cs.findbugs.ba.type;
21  
22  /**
23   * Extended type codes used by StackAndLocalTypes and StackAndLocalTypeAnalysis
24   * for typing locals and stack values used in Java bytecode.
25   *
26   * @author David Hovemeyer
27   * @see TypeFrame
28   * @see TypeAnalysis
29   */
30  public interface ExtendedTypes {
31      // FIXME: these values depend on those in org.apache.bcel.Const.
32      // They need to be distinct from all type codes defined there.
33      // It would be nice if BCEL provided built-in functionality for
34      // some or all of these.
35  
36      /**
37       * Special type code for the "Top" type in the lattice.
38       */
39      public static final byte T_TOP = 17;
40  
41      /**
42       * Special type that represents the value store in local <i>n+1</i> when a
43       * long value is stored in local <i>n</i>.
44       */
45      public static final byte T_LONG_EXTRA = 18;
46  
47      /**
48       * Special type that represents the value store in local <i>n+1</i> when a
49       * double value is stored in local <i>n</i>.
50       */
51      public static final byte T_DOUBLE_EXTRA = 19;
52  
53      /**
54       * Special type code for the "Bottom" type in the lattice.
55       */
56      public static final byte T_BOTTOM = 20;
57  
58      /**
59       * Special type code for the "Null" type. This is a type which is higher in
60       * the lattice than any object type, but lower than the overall Top type. It
61       * represents the type of the null value, which may logically be merged with
62       * any object type without loss of information.
63       */
64      public static final byte T_NULL = 21;
65  
66      /**
67       * Special type code for "Exception" object type, which is an instance of
68       * the ExceptionObjectType class. These instances maintain an ExceptionSet
69       * to keep track of possible exceptions that are feasible, and whether those
70       * exceptions are explicit or implicit.
71       */
72      public static final byte T_EXCEPTION = 22;
73  
74      /**
75       * A Type code that is available for "user-defined" types. Any type code
76       * equal or greated than this one is guaranteed to be distinct from both
77       * standard and extended types.
78       */
79      public static final byte T_AVAIL_TYPE = 100;
80  }

45      public static String toString(byte[] b) {
46          if (b == null) {
47              return "(null)";
48          }
49          StringBuffer sb = new StringBuffer(b.length * 3);
50          for (int i = 0; i < b.length; i++) {
51              int k = b[i] & 0xff;
52              if (i != 0) {
53                  sb.append(':');
54              }
55              sb.append(hexDigits[k >>> 4]);
56              sb.append(hexDigits[k & 0xf]);
57          }
58          return sb.toString();
59      }

61      public static byte[] parse(String s) {
62          try {
63              int n = s.length();
64              ByteArrayOutputStream out = new ByteArrayOutputStream(n / 3);
65              StringReader r = new StringReader(s);
66              while (true) {
67                  int b1 = nextNibble(r);
68                  if (b1 < 0) {
69                      break;
70                  }
71                  int b2 = nextNibble(r);
72                  if (b2 < 0) {
73                      throw new RuntimeException("Invalid string " + s);
74                  }
75                  int b = (b1 << 4) | b2;
76                  out.write(b);
77              }
78              return out.toByteArray();
79          } catch (IOException e) {
80              throw new RuntimeException(e);
81          }
82      }
83  
84      public static byte[] b(String s) {
85          return parse(s);
86      }

88      private static int nextNibble(StringReader r) throws IOException {
89          while (true) {
90              int ch = r.read();
91              if (ch == -1) {
92                  return -1;
93              } else if ((ch >= '0') && (ch <= '9')) {
94                  return ch - '0';
95              } else if ((ch >= 'a') && (ch <= 'f')) {
96                  return ch - 'a' + 10;
97              } else if ((ch >= 'A') && (ch <= 'F')) {
98                  return ch - 'A' + 10;
99              }
100         }
101     }

123     private static void testEncDec(String alg, int len, Key encKey, Key decKey) throws Exception {
124         System.out.println("Testing en/decryption using " + alg + "...");
125         Cipher c = Cipher.getInstance(alg, p);
126 
127         byte[] b = new byte[len];
128         RANDOM.nextBytes(b);
129         b[0] &= 0x3f;
130         b[0] |= 1;
131 
132         c.init(Cipher.ENCRYPT_MODE, encKey);
133         byte[] enc = c.doFinal(b);
134 
135         c.init(Cipher.DECRYPT_MODE, decKey);
136         byte[] dec = c.doFinal(enc);
137 
138         if (Arrays.equals(b, dec) == false) {
139             System.out.println("in:  " + toString(b));
140             System.out.println("dec: " + toString(dec));
141             throw new Exception("Failure");
142         }
143     }

152             throw new Exception("Unexpected test result");

168     public static void main(String[] args) throws Exception {
169         long start = System.currentTimeMillis();
170 
171         p = Security.getProvider("SunJCE");
172         System.out.println("Testing provider " + p.getName() + "...");
173 
174         KeyFactory kf;
175         try {
176             kf = KeyFactory.getInstance("RSA", p);
177         } catch (NoSuchAlgorithmException e) {
178             kf = KeyFactory.getInstance("RSA");
179         }
180 
181         RSAPublicKeySpec pubSpec = new RSAPublicKeySpec(N, E);
182         PublicKey publicKey = kf.generatePublic(pubSpec);
183 
184         RSAPrivateKeySpec privSpec = new RSAPrivateKeySpec(N, D);
185         PrivateKey privateKey = kf.generatePrivate(privSpec);
186 
187         // blocktype 2
188         testEncDec("RSA/ECB/PKCS1Padding", 96, publicKey, privateKey);
189         // blocktype 1
190         testEncDec("RSA/ECB/PKCS1Padding", 96, privateKey, publicKey);
191 
192         testEncDec("RSA/ECB/NoPadding", 128, publicKey, privateKey);
193         testEncDec("RSA/ECB/NoPadding", 128, privateKey, publicKey);
194 
195         // expected failure, blocktype 2 random padding bytes are different
196         testKat("RSA/ECB/PKCS1Padding", Cipher.ENCRYPT_MODE, publicKey, in2, out2, false);
197         testKat("RSA/ECB/PKCS1Padding", Cipher.DECRYPT_MODE, privateKey, out2, in2, true);
198 
199         testKat("RSA/ECB/PKCS1Padding", Cipher.ENCRYPT_MODE, privateKey, in1, out1, true);
200         testKat("RSA/ECB/PKCS1Padding", Cipher.DECRYPT_MODE, publicKey, out1, in1, true);
201 
202         testKat("RSA/ECB/NoPadding", Cipher.ENCRYPT_MODE, publicKey, rin1, rout1, true);
203         testKat("RSA/ECB/NoPadding", Cipher.DECRYPT_MODE, privateKey, rout1, rin1, true);
204 
205         testKat("RSA/ECB/NoPadding", Cipher.ENCRYPT_MODE, privateKey, rin2, rout2, true);
206         testKat("RSA/ECB/NoPadding", Cipher.DECRYPT_MODE, publicKey, rout2, rin2, true);
207 
208         System.out.println("Testing error cases...");
209         try {
210             // decrypt something not PKCS#1 formatted
211             testKat("RSA/ECB/PKCS1Padding", Cipher.DECRYPT_MODE, privateKey, rout1, rin1, true);
212             throw new Exception("Unexpected success");
213         } catch (BadPaddingException e) {
214             // ok
215         }
216 
217         try {
218             // decrypt with wrong key
219             testKat("RSA/ECB/PKCS1Padding", Cipher.DECRYPT_MODE, privateKey, out1, in1, true);
220             throw new Exception("Unexpected success");
221         } catch (BadPaddingException e) {
222             // ok
223         }
224 
225         try {
226             // encrypt data that is too long
227             testKat("RSA/ECB/PKCS1Padding", Cipher.ENCRYPT_MODE, privateKey, out1, in1, true);
228             throw new Exception("Unexpected success");
229         } catch (IllegalBlockSizeException e) {
230             // ok
231         }
232 
233         long stop = System.currentTimeMillis();
234         System.out.println("Done (" + (stop - start) + " ms).");
235     }

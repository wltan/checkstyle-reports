136     public static void main(String[] args) throws Exception {
137         boolean result = true;
138         Provider p = Security.getProvider("SunJCE");
139         System.out.println("Testing provider " + p.getName() + "...");
140         // Test#1: make sure Cipher.getMaxAllowedKeyLength returns the
141         // correct value
142         if (Cipher.getMaxAllowedKeyLength("RSA") != Integer.MAX_VALUE) {
143             result = false;
144             System.out.println("Test#1 failed");
145         }
146         // Test#2: try initializing RSA cipher with 4096 key
147         String algo = "RSA";
148         RSAPublicKeySpec pubKeySpec = new RSAPublicKeySpec
149             (new BigInteger(MODULUS4096), new BigInteger(PUB4096));
150         KeyFactory kf = KeyFactory.getInstance(algo);
151         RSAPublicKey pubKey = (RSAPublicKey) kf.generatePublic(pubKeySpec);
152 
153         Cipher c = Cipher.getInstance(algo + "/ECB/NoPadding", p);
154         try {
155             c.init(Cipher.ENCRYPT_MODE, pubKey);
156         } catch (InvalidKeyException ike) {
157             result = false;
158             System.out.println("Test#2 failed");
159             ike.printStackTrace();
160         }
161 
162         if (result) {
163             System.out.println("All tests passed!");
164         } else {
165             throw new Exception("One or more test failed!");
166         }
167     }

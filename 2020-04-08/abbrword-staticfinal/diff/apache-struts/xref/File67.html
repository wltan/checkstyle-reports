19  package org.apache.struts2.components;
20  
21  import com.opensymphony.xwork2.util.ValueStack;
22  import org.apache.commons.lang3.StringUtils;
23  import org.apache.struts2.views.annotations.StrutsTag;
24  import org.apache.struts2.views.annotations.StrutsTagAttribute;
25  
26  import javax.servlet.http.HttpServletRequest;
27  import javax.servlet.http.HttpServletResponse;
28  import java.util.Collection;
29  import java.util.List;
30  
31  /**
32   * <!-- START SNIPPET: javadoc -->
33   *
34   * Render action messages if they exists, specific rendering layout depends on the
35   * theme itself. Empty (null or blank string) messages will not be printed. The action message
36   * strings will be html escaped by default.
37   *
38   * <!-- END SNIPPET: javadoc -->
39   *
40   * <p><b>Examples</b></p>
41   *
42   * <pre>
43   * <!-- START SNIPPET: example -->
44   *    &lt;s:actionmessage /&gt;
45   *    &lt;s:form .... &gt;
46   *       ....
47   *    &lt;/s:form&gt;
48   * <!-- END SNIPPET: example -->
49   * </pre>
50   *
51   */
52  @StrutsTag(name="actionmessage", tldBodyContent="empty", tldTagClass="org.apache.struts2.views.jsp.ui.ActionMessageTag", description="Render action messages if they exists")
53  public class ActionMessage extends UIBean {
54  
55      private static final String TEMPLATE = "actionmessage";
56      protected boolean escape = true;
57  
58      public ActionMessage(ValueStack stack, HttpServletRequest request, HttpServletResponse response) {
59          super(stack, request, response);
60      }
61  
62      protected String getDefaultTemplate() {
63          return TEMPLATE;
64      }
65  
66      protected void evaluateExtraParams() {
67          boolean isEmptyList = true;
68          Collection<String> actionMessages = (List) findValue("actionMessages");
69          if (actionMessages != null) {
70              for (String message : actionMessages) {
71                  if (StringUtils.isNotBlank(message)) {
72                      isEmptyList = false;
73                      break;
74                  }
75              }
76          }
77  
78          addParameter("isEmptyList", isEmptyList);
79          addParameter("escape", escape);
80      }
81  
82      @StrutsTagAttribute(description = "Whether to escape HTML", type = "Boolean", defaultValue = "true")
83      public void setEscape(boolean escape) {
84          this.escape = escape;
85      }
86  }

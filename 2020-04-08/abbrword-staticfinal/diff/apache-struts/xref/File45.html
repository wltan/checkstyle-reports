23  /**
24   * This class is an utility class that perform wilcard-patterns matching and
25   * isolation taken from Apache Cocoon.
26   *
27   * @version $Rev$ $Date: 2005-05-07 12:11:38 -0400 (Sat, 07 May 2005)
28   *          $
29   */
30  public class WildcardHelper implements PatternMatcher<int[]> {
31      /**
32       * The int representing '*' in the pattern <code>int []</code>.
33       */
34      protected static final int MATCH_FILE = -1;
35  
36      /**
37       * The int representing '**' in the pattern <code>int []</code>.
38       */
39      protected static final int MATCH_PATH = -2;
40  
41      /**
42       * The int representing begin in the pattern <code>int []</code>.
43       */
44      protected static final int MATCH_BEGIN = -4;
45  
46      /**
47       * The int representing end in pattern <code>int []</code>.
48       */
49      protected static final int MATCH_THEEND = -5;
50  
51      /**
52       * The int value that terminates the pattern <code>int []</code>.
53       */
54      protected static final int MATCH_END = -3;
55  
56      /**
57       * Determines if the pattern contains any * characters
58       *
59       * @param pattern The pattern
60       * @return True if no wildcards are found
61       */
62      public boolean isLiteral(String pattern) {
63          return (pattern == null || pattern.indexOf('*') == -1);
64      }

66      /**
67       * <p> Translate the given <code>String</code> into a <code>int []</code>
68       * representing the pattern matchable by this class. <br> This function
69       * translates a <code>String</code> into an int array converting the
70       * special '*' and '\' characters. <br> Here is how the conversion
71       * algorithm works:</p>
72       *
73       * <ul>
74       *
75       * <li>The '*' character is converted to MATCH_FILE, meaning that zero or
76       * more characters (excluding the path separator '/') are to be
77       * matched.</li>
78       *
79       * <li>The '**' sequence is converted to MATCH_PATH, meaning that zero or
80       * more characters (including the path separator '/') are to be
81       * matched.</li>
82       *
83       * <li>The '\' character is used as an escape sequence ('\*' is translated
84       * in '*', not in MATCH_FILE). If an exact '\' character is to be matched
85       * the source string must contain a '\\'. sequence.</li>
86       *
87       * </ul>
88       *
89       * <p>When more than two '*' characters, not separated by another
90       * character, are found their value is considered as '**' (MATCH_PATH).
91       * <br> The array is always terminated by a special value (MATCH_END).
92       * <br> All MATCH* values are less than zero, while normal characters are
93       * equal or greater.</p>
94       *
95       * @param data The string to translate.
96       * @return The encoded string as an int array, terminated by the MATCH_END
97       *         value (don't consider the array length).
98       * @throws NullPointerException If data is null.
99       */
100     public int[] compilePattern(String data) {
101         // Prepare the arrays
102         int[] expr = new int[data.length() + 2];
103         char[] buff = data.toCharArray();
104 
105         // Prepare variables for the translation loop
106         int y = 0;
107         boolean slash = false;
108 
109         // Must start from beginning
110         expr[y++] = MATCH_BEGIN;
111 
112         if (buff.length > 0) {
113             if (buff[0] == '\\') {
114                 slash = true;
115             } else if (buff[0] == '*') {
116                 expr[y++] = MATCH_FILE;
117             } else {
118                 expr[y++] = buff[0];
119             }
120 
121             // Main translation loop
122             for (int x = 1; x < buff.length; x++) {
123                 // If the previous char was '\' simply copy this char.
124                 if (slash) {
125                     expr[y++] = buff[x];
126                     slash = false;
127 
128                     // If the previous char was not '\' we have to do a bunch of
129                     // checks
130                 } else {
131                     // If this char is '\' declare that and continue
132                     if (buff[x] == '\\') {
133                         slash = true;
134 
135                         // If this char is '*' check the previous one
136                     } else if (buff[x] == '*') {
137                         // If the previous character als was '*' match a path
138                         if (expr[y - 1] <= MATCH_FILE) {
139                             expr[y - 1] = MATCH_PATH;
140                         } else {
141                             expr[y++] = MATCH_FILE;
142                         }
143                     } else {
144                         expr[y++] = buff[x];
145                     }
146                 }
147             }
148         }
149 
150         // Must match end at the end
151         expr[y] = MATCH_THEEND;
152 
153         return expr;
154     }

156     /**
157      * Match a pattern agains a string and isolates wildcard replacement into
158      * a <code>Stack</code>.
159      *
160      * @param map  The map to store matched values
161      * @param data The string to match
162      * @param expr The compiled wildcard expression
163      * @return True if a match
164      * @throws NullPointerException If any parameters are null
165      */
166     public boolean match(Map<String, String> map, String data, int[] expr) {
167         if (map == null) {
168             throw new NullPointerException("No map provided");
169         }
170 
171         if (data == null) {
172             throw new NullPointerException("No data provided");
173         }
174 
175         if (expr == null) {
176             throw new NullPointerException("No pattern expression provided");
177         }
178 
179         char[] buff = data.toCharArray();
180 
181         // Allocate the result buffer
182         char[] rslt = new char[expr.length + buff.length];
183 
184         // The previous and current position of the expression character
185         // (MATCH_*)
186         int charpos = 0;
187 
188         // The position in the expression, input, translation and result arrays
189         int exprpos = 0;
190         int buffpos = 0;
191         int rsltpos = 0;
192         int offset = -1;
193 
194         // The matching count
195         int mcount = 0;
196 
197         // We want the complete data be in {0}
198         map.put(Integer.toString(mcount), data);
199 
200         // First check for MATCH_BEGIN
201         boolean matchBegin = false;
202 
203         if (expr[charpos] == MATCH_BEGIN) {
204             matchBegin = true;
205             exprpos = ++charpos;
206         }
207 
208         // Search the fist expression character (except MATCH_BEGIN - already
209         // skipped)
210         while (expr[charpos] >= 0) {
211             charpos++;
212         }
213 
214         // The expression charater (MATCH_*)
215         int exprchr = expr[charpos];
216 
217         while (true) {
218             // Check if the data in the expression array before the current
219             // expression character matches the data in the input buffer
220             if (matchBegin) {
221                 if (!matchArray(expr, exprpos, charpos, buff, buffpos)) {
222                     return (false);
223                 }
224 
225                 matchBegin = false;
226             } else {
227                 offset = indexOfArray(expr, exprpos, charpos, buff, buffpos);
228 
229                 if (offset < 0) {
230                     return (false);
231                 }
232             }
233 
234             // Check for MATCH_BEGIN
235             if (matchBegin) {
236                 if (offset != 0) {
237                     return (false);
238                 }
239 
240                 matchBegin = false;
241             }
242 
243             // Advance buffpos
244             buffpos += (charpos - exprpos);
245 
246             // Check for END's
247             if (exprchr == MATCH_END) {
248                 if (rsltpos > 0) {
249                     map.put(Integer.toString(++mcount),
250                         new String(rslt, 0, rsltpos));
251                 }
252 
253                 // Don't care about rest of input buffer
254                 return (true);
255             } else if (exprchr == MATCH_THEEND) {
256                 if (rsltpos > 0) {
257                     map.put(Integer.toString(++mcount),
258                         new String(rslt, 0, rsltpos));
259                 }
260 
261                 // Check that we reach buffer's end
262                 return (buffpos == buff.length);
263             }
264 
265             // Search the next expression character
266             exprpos = ++charpos;
267 
268             while (expr[charpos] >= 0) {
269                 charpos++;
270             }
271 
272             int prevchr = exprchr;
273 
274             exprchr = expr[charpos];
275 
276             // We have here prevchr == * or **.
277             offset =
278                 (prevchr == MATCH_FILE)
279                 ? indexOfArray(expr, exprpos, charpos, buff, buffpos)
280                 : lastIndexOfArray(expr, exprpos, charpos, buff, buffpos);
281 
282             if (offset < 0) {
283                 return (false);
284             }
285 
286             // Copy the data from the source buffer into the result buffer
287             // to substitute the expression character
288             if (prevchr == MATCH_PATH) {
289                 while (buffpos < offset) {
290                     rslt[rsltpos++] = buff[buffpos++];
291                 }
292             } else {
293                 // Matching file, don't copy '/'
294                 while (buffpos < offset) {
295                     if (buff[buffpos] == '/') {
296                         return (false);
297                     }
298 
299                     rslt[rsltpos++] = buff[buffpos++];
300                 }
301             }
302 
303             map.put(Integer.toString(++mcount), new String(rslt, 0, rsltpos));
304             rsltpos = 0;
305         }
306     }

308     /**
309      * Get the offset of a part of an int array within a char array. <br> This
310      * method return the index in d of the first occurrence after dpos of that
311      * part of array specified by r, starting at rpos and terminating at
312      * rend.
313      *
314      * @param r    The array containing the data that need to be matched in
315      *             d.
316      * @param rpos The index of the first character in r to look for.
317      * @param rend The index of the last character in r to look for plus 1.
318      * @param d    The array of char that should contain a part of r.
319      * @param dpos The starting offset in d for the matching.
320      * @return The offset in d of the part of r matched in d or -1 if that was
321      *         not found.
322      */
323     protected int indexOfArray(int[] r, int rpos, int rend, char[] d, int dpos) {
324         // Check if pos and len are legal
325         if (rend < rpos) {
326             throw new IllegalArgumentException("rend < rpos");
327         }
328 
329         // If we need to match a zero length string return current dpos
330         if (rend == rpos) {
331             return (d.length); //?? dpos?
332         }
333 
334         // If we need to match a 1 char length string do it simply
335         if ((rend - rpos) == 1) {
336             // Search for the specified character
337             for (int x = dpos; x < d.length; x++) {
338                 if (r[rpos] == d[x]) {
339                     return (x);
340                 }
341             }
342         }
343 
344         // Main string matching loop. It gets executed if the characters to
345         // match are less then the characters left in the d buffer
346         while (((dpos + rend) - rpos) <= d.length) {
347             // Set current startpoint in d
348             int y = dpos;
349 
350             // Check every character in d for equity. If the string is matched
351             // return dpos
352             for (int x = rpos; x <= rend; x++) {
353                 if (x == rend) {
354                     return (dpos);
355                 }
356 
357                 if (r[x] != d[y++]) {
358                     break;
359                 }
360             }
361 
362             // Increase dpos to search for the same string at next offset
363             dpos++;
364         }
365 
366         // The remaining chars in d buffer were not enough or the string
367         // wasn't matched
368         return (-1);
369     }

371     /**
372      * Get the offset of a last occurance of an int array within a char array.
373      * <br> This method return the index in d of the last occurrence after
374      * dpos of that part of array specified by r, starting at rpos and
375      * terminating at rend.
376      *
377      * @param r    The array containing the data that need to be matched in
378      *             d.
379      * @param rpos The index of the first character in r to look for.
380      * @param rend The index of the last character in r to look for plus 1.
381      * @param d    The array of char that should contain a part of r.
382      * @param dpos The starting offset in d for the matching.
383      * @return The offset in d of the last part of r matched in d or -1 if
384      *         that was not found.
385      */
386     protected int lastIndexOfArray(int[] r, int rpos, int rend, char[] d,
387         int dpos) {
388         // Check if pos and len are legal
389         if (rend < rpos) {
390             throw new IllegalArgumentException("rend < rpos");
391         }
392 
393         // If we need to match a zero length string return current dpos
394         if (rend == rpos) {
395             return (d.length); //?? dpos?
396         }
397 
398         // If we need to match a 1 char length string do it simply
399         if ((rend - rpos) == 1) {
400             // Search for the specified character
401             for (int x = d.length - 1; x > dpos; x--) {
402                 if (r[rpos] == d[x]) {
403                     return (x);
404                 }
405             }
406         }
407 
408         // Main string matching loop. It gets executed if the characters to
409         // match are less then the characters left in the d buffer
410         int l = d.length - (rend - rpos);
411 
412         while (l >= dpos) {
413             // Set current startpoint in d
414             int y = l;
415 
416             // Check every character in d for equity. If the string is matched
417             // return dpos
418             for (int x = rpos; x <= rend; x++) {
419                 if (x == rend) {
420                     return (l);
421                 }
422 
423                 if (r[x] != d[y++]) {
424                     break;
425                 }
426             }
427 
428             // Decrease l to search for the same string at next offset
429             l--;
430         }
431 
432         // The remaining chars in d buffer were not enough or the string
433         // wasn't matched
434         return (-1);
435     }

437     /**
438      * Matches elements of array r from rpos to rend with array d, starting
439      * from dpos. <br> This method return true if elements of array r from
440      * rpos to rend equals elements of array d starting from dpos to
441      * dpos+(rend-rpos).
442      *
443      * @param r    The array containing the data that need to be matched in
444      *             d.
445      * @param rpos The index of the first character in r to look for.
446      * @param rend The index of the last character in r to look for.
447      * @param d    The array of char that should start from a part of r.
448      * @param dpos The starting offset in d for the matching.
449      * @return true if array d starts from portion of array r.
450      */
451     protected boolean matchArray(int[] r, int rpos, int rend, char[] d, int dpos) {
452         if ((d.length - dpos) < (rend - rpos)) {
453             return (false);
454         }
455 
456         for (int i = rpos; i < rend; i++) {
457             if (r[i] != d[dpos++]) {
458                 return (false);
459             }
460         }
461 
462         return (true);
463     }

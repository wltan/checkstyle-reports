108 	@Test
109 	public void getRequestRemoveSemicolonContent() throws UnsupportedEncodingException {
110 		helper.setRemoveSemicolonContent(true);
111 
112 		request.setRequestURI("/foo;f=F;o=O;o=O/bar;b=B;a=A;r=R");
113 		assertEquals("/foo/bar", helper.getRequestUri(request));
114 	}
115 
116 	@Test
117 	public void getRequestKeepSemicolonContent() throws UnsupportedEncodingException {
118 		helper.setRemoveSemicolonContent(false);
119 
120 		request.setRequestURI("/foo;a=b;c=d");
121 		assertEquals("/foo;a=b;c=d", helper.getRequestUri(request));
122 
123 		request.setRequestURI("/foo;jsessionid=c0o7fszeb1");
124 		assertEquals("jsessionid should always be removed", "/foo", helper.getRequestUri(request));
125 
126 		request.setRequestURI("/foo;a=b;jsessionid=c0o7fszeb1;c=d");
127 		assertEquals("jsessionid should always be removed", "/foo;a=b;c=d", helper.getRequestUri(request));
128 
129 		// SPR-10398
130 
131 		request.setRequestURI("/foo;a=b;JSESSIONID=c0o7fszeb1;c=d");
132 		assertEquals("JSESSIONID should always be removed", "/foo;a=b;c=d", helper.getRequestUri(request));
133 	}

17  public class Euler32Test {
18  
19      /**
20       * <strong>Problem 23 Pandigital products</strong>
21       * <p>
22       * We shall say that an n-digit number is pandigital if it makes use of all
23       * the digits 1 to n exactly once; for example, the 5-digit number, 15234,
24       * is 1 through 5 pandigital.
25       * <p>
26       * The product 7254 is unusual, as the identity, 39 × 186 = 7254, containing
27       * multiplicand, multiplier, and product is 1 through 9 pandigital.
28       * <p>
29       * Find the sum of all products whose multiplicand/multiplier/product
30       * identity can be written as a 1 through 9 pandigital.
31       * <p>
32       * HINT: Some products can be obtained in more than one way so be sure to
33       * only include it once in your sum.
34       * <p>
35       * See also <a href="https://projecteuler.net/problem=32">projecteuler.net
36       * problem 32</a>.
37       */
38      @Test
39      public void shouldSolveProblem32() {
40          assertThat(isPandigital(1, 5, "15234")).isTrue();
41          assertThat(isPandigital(1, 9, "39" + "186" + "7254")).isTrue();
42          assertThat(isPandigital(1, 5, "55555")).isFalse();
43          assertThat(isPandigital(1, 5, "12340")).isFalse();
44          assertThat(sumOfAllProductsPandigital1Through9()).isEqualTo(45228);
45      }
46  
47      private static boolean isPandigital(int from, int to, String num) {
48          return num.length() == to - from + 1 && List.rangeClosed(from, to).forAll(i -> num.contains(Integer.toString(i)));
49      }
50  
51      private static final CharSeq DIGITS_1_9 = CharSeq.of("123456789");
52  
53      private static long sumOfAllProductsPandigital1Through9() {
54          return List.of(1, 2)
55                  .flatMap(i -> DIGITS_1_9.crossProduct(i)
56                          .flatMap(multiplicand -> DIGITS_1_9.removeAll(multiplicand).crossProduct(5 - i)
57                                  .map(multiplier -> Tuple.of(multiplicand.mkString(), multiplier.mkString()))
58                          )
59                  )
60                  .map(t -> Tuple.of(t._1, t._2, Long.valueOf(t._1) * Long.valueOf(t._2)))
61                  .filter(t -> isPandigital(1, 9, t._1 + t._2 + Long.toString(t._3)))
62                  .map(Tuple3::_3)
63                  .distinct()
64                  .sum().longValue();
65      }

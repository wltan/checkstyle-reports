67    /** Returns the array type of {@code componentType}. */
68    static Type newArrayType(Type componentType) {
69      if (componentType instanceof WildcardType) {
70        WildcardType wildcard = (WildcardType) componentType;
71        Type[] lowerBounds = wildcard.getLowerBounds();
72        checkArgument(lowerBounds.length <= 1, "Wildcard cannot have more than one lower bounds.");
73        if (lowerBounds.length == 1) {
74          return supertypeOf(newArrayType(lowerBounds[0]));
75        } else {
76          Type[] upperBounds = wildcard.getUpperBounds();
77          checkArgument(upperBounds.length == 1, "Wildcard should have only one upper bound.");
78          return subtypeOf(newArrayType(upperBounds[0]));
79        }
80      }
81      return JavaVersion.CURRENT.newArrayType(componentType);
82    }

85     * Returns a type where {@code rawType} is parameterized by {@code arguments} and is owned by
86     * {@code ownerType}.
87     */
88    static ParameterizedType newParameterizedTypeWithOwner(
89        @Nullable Type ownerType, Class<?> rawType, Type... arguments) {
90      if (ownerType == null) {
91        return newParameterizedType(rawType, arguments);
92      }
93      // ParameterizedTypeImpl constructor already checks, but we want to throw NPE before IAE
94      checkNotNull(arguments);
95      checkArgument(rawType.getEnclosingClass() != null, "Owner type for unenclosed %s", rawType);
96      return new ParameterizedTypeImpl(ownerType, rawType, arguments);
97    }

105   /** Decides what owner type to use for constructing {@link ParameterizedType} from a raw class. */
106   private enum ClassOwnership {
107     OWNED_BY_ENCLOSING_CLASS {
108       @Override
109       @Nullable
110       Class<?> getOwnerType(Class<?> rawType) {
111         return rawType.getEnclosingClass();
112       }
113     },
114     LOCAL_CLASS_HAS_NO_OWNER {
115       @Override
116       @Nullable
117       Class<?> getOwnerType(Class<?> rawType) {
118         if (rawType.isLocalClass()) {
119           return null;
120         } else {
121           return rawType.getEnclosingClass();
122         }
123       }
124     };
125 
126     abstract @Nullable Class<?> getOwnerType(Class<?> rawType);
127 
128     static final ClassOwnership JVM_BEHAVIOR = detectJvmBehavior();
129 
130     private static ClassOwnership detectJvmBehavior() {
131       class LocalClass<T> {}
132       Class<?> subclass = new LocalClass<String>() {}.getClass();
133       ParameterizedType parameterizedType = (ParameterizedType) subclass.getGenericSuperclass();
134       for (ClassOwnership behavior : ClassOwnership.values()) {
135         if (behavior.getOwnerType(LocalClass.class) == parameterizedType.getOwnerType()) {
136           return behavior;
137         }
138       }
139       throw new AssertionError();
140     }
141   }

174   static @Nullable Type getComponentType(Type type) {
175     checkNotNull(type);
176     final AtomicReference<Type> result = new AtomicReference<>();
177     new TypeVisitor() {
178       @Override
179       void visitTypeVariable(TypeVariable<?> t) {
180         result.set(subtypeOfComponentType(t.getBounds()));
181       }
182 
183       @Override
184       void visitWildcardType(WildcardType t) {
185         result.set(subtypeOfComponentType(t.getUpperBounds()));
186       }
187 
188       @Override
189       void visitGenericArrayType(GenericArrayType t) {
190         result.set(t.getGenericComponentType());
191       }
192 
193       @Override
194       void visitClass(Class<?> t) {
195         result.set(t.getComponentType());
196       }
197     }.visit(type);
198     return result.get();
199   }

201   /**
202    * Returns {@code ? extends X} if any of {@code bounds} is a subtype of {@code X[]}; or null
203    * otherwise.
204    */
205   private static @Nullable Type subtypeOfComponentType(Type[] bounds) {
206     for (Type bound : bounds) {
207       Type componentType = getComponentType(bound);
208       if (componentType != null) {
209         // Only the first bound can be a class or array.
210         // Bounds after the first can only be interfaces.
211         if (componentType instanceof Class) {
212           Class<?> componentClass = (Class<?>) componentType;
213           if (componentClass.isPrimitive()) {
214             return componentClass;
215           }
216         }
217         return subtypeOf(componentType);
218       }
219     }
220     return null;
221   }

258   private static final class ParameterizedTypeImpl implements ParameterizedType, Serializable {
259 
260     private final @Nullable Type ownerType;
261     private final ImmutableList<Type> argumentsList;
262     private final Class<?> rawType;
263 
264     ParameterizedTypeImpl(@Nullable Type ownerType, Class<?> rawType, Type[] typeArguments) {
265       checkNotNull(rawType);
266       checkArgument(typeArguments.length == rawType.getTypeParameters().length);
267       disallowPrimitiveType(typeArguments, "type parameter");
268       this.ownerType = ownerType;
269       this.rawType = rawType;
270       this.argumentsList = JavaVersion.CURRENT.usedInGenericType(typeArguments);
271     }
272 
273     @Override
274     public Type[] getActualTypeArguments() {
275       return toArray(argumentsList);
276     }
277 
278     @Override
279     public Type getRawType() {
280       return rawType;
281     }
282 
283     @Override
284     public Type getOwnerType() {
285       return ownerType;
286     }
287 
288     @Override
289     public String toString() {
290       StringBuilder builder = new StringBuilder();
291       if (ownerType != null && JavaVersion.CURRENT.jdkTypeDuplicatesOwnerName()) {
292         builder.append(JavaVersion.CURRENT.typeName(ownerType)).append('.');
293       }
294       return builder
295           .append(rawType.getName())
296           .append('<')
297           .append(COMMA_JOINER.join(transform(argumentsList, TYPE_NAME)))
298           .append('>')
299           .toString();
300     }
301 
302     @Override
303     public int hashCode() {
304       return (ownerType == null ? 0 : ownerType.hashCode())
305           ^ argumentsList.hashCode()
306           ^ rawType.hashCode();
307     }
308 
309     @Override
310     public boolean equals(Object other) {
311       if (!(other instanceof ParameterizedType)) {
312         return false;
313       }
314       ParameterizedType that = (ParameterizedType) other;
315       return getRawType().equals(that.getRawType())
316           && Objects.equal(getOwnerType(), that.getOwnerType())
317           && Arrays.equals(getActualTypeArguments(), that.getActualTypeArguments());
318     }
319 
320     private static final long serialVersionUID = 0;
321   }

323   private static <D extends GenericDeclaration> TypeVariable<D> newTypeVariableImpl(
324       D genericDeclaration, String name, Type[] bounds) {
325     TypeVariableImpl<D> typeVariableImpl =
326         new TypeVariableImpl<D>(genericDeclaration, name, bounds);
327     @SuppressWarnings("unchecked")
328     TypeVariable<D> typeVariable =
329         Reflection.newProxy(
330             TypeVariable.class, new TypeVariableInvocationHandler(typeVariableImpl));
331     return typeVariable;
332   }

334   /**
335    * Invocation handler to work around a compatibility problem between Java 7 and Java 8.
336    *
337    * <p>Java 8 introduced a new method {@code getAnnotatedBounds()} in the {@link TypeVariable}
338    * interface, whose return type {@code AnnotatedType[]} is also new in Java 8. That means that we
339    * cannot implement that interface in source code in a way that will compile on both Java 7 and
340    * Java 8. If we include the {@code getAnnotatedBounds()} method then its return type means it
341    * won't compile on Java 7, while if we don't include the method then the compiler will complain
342    * that an abstract method is unimplemented. So instead we use a dynamic proxy to get an
343    * implementation. If the method being called on the {@code TypeVariable} instance has the same
344    * name as one of the public methods of {@link TypeVariableImpl}, the proxy calls the same method
345    * on its instance of {@code TypeVariableImpl}. Otherwise it throws {@link
346    * UnsupportedOperationException}; this should only apply to {@code getAnnotatedBounds()}. This
347    * does mean that users on Java 8 who obtain an instance of {@code TypeVariable} from {@link
348    * TypeResolver#resolveType} will not be able to call {@code getAnnotatedBounds()} on it, but that
349    * should hopefully be rare.
350    *
351    * <p>This workaround should be removed at a distant future time when we no longer support Java
352    * versions earlier than 8.
353    */
354   private static final class TypeVariableInvocationHandler implements InvocationHandler {
355     private static final ImmutableMap<String, Method> typeVariableMethods;
356 
357     static {
358       ImmutableMap.Builder<String, Method> builder = ImmutableMap.builder();
359       for (Method method : TypeVariableImpl.class.getMethods()) {
360         if (method.getDeclaringClass().equals(TypeVariableImpl.class)) {
361           try {
362             method.setAccessible(true);
363           } catch (AccessControlException e) {
364             // OK: the method is accessible to us anyway. The setAccessible call is only for
365             // unusual execution environments where that might not be true.
366           }
367           builder.put(method.getName(), method);
368         }
369       }
370       typeVariableMethods = builder.build();
371     }
372 
373     private final TypeVariableImpl<?> typeVariableImpl;
374 
375     TypeVariableInvocationHandler(TypeVariableImpl<?> typeVariableImpl) {
376       this.typeVariableImpl = typeVariableImpl;
377     }
378 
379     @Override
380     public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
381       String methodName = method.getName();
382       Method typeVariableMethod = typeVariableMethods.get(methodName);
383       if (typeVariableMethod == null) {
384         throw new UnsupportedOperationException(methodName);
385       } else {
386         try {
387           return typeVariableMethod.invoke(typeVariableImpl, args);
388         } catch (InvocationTargetException e) {
389           throw e.getCause();
390         }
391       }
392     }
393   }

395   private static final class TypeVariableImpl<D extends GenericDeclaration> {
396 
397     private final D genericDeclaration;
398     private final String name;
399     private final ImmutableList<Type> bounds;
400 
401     TypeVariableImpl(D genericDeclaration, String name, Type[] bounds) {
402       disallowPrimitiveType(bounds, "bound for type variable");
403       this.genericDeclaration = checkNotNull(genericDeclaration);
404       this.name = checkNotNull(name);
405       this.bounds = ImmutableList.copyOf(bounds);
406     }
407 
408     public Type[] getBounds() {
409       return toArray(bounds);
410     }
411 
412     public D getGenericDeclaration() {
413       return genericDeclaration;
414     }
415 
416     public String getName() {
417       return name;
418     }
419 
420     public String getTypeName() {
421       return name;
422     }
423 
424     @Override
425     public String toString() {
426       return name;
427     }
428 
429     @Override
430     public int hashCode() {
431       return genericDeclaration.hashCode() ^ name.hashCode();
432     }
433 
434     @Override
435     public boolean equals(Object obj) {
436       if (NativeTypeVariableEquals.NATIVE_TYPE_VARIABLE_ONLY) {
437         // equal only to our TypeVariable implementation with identical bounds
438         if (obj != null
439             && Proxy.isProxyClass(obj.getClass())
440             && Proxy.getInvocationHandler(obj) instanceof TypeVariableInvocationHandler) {
441           TypeVariableInvocationHandler typeVariableInvocationHandler =
442               (TypeVariableInvocationHandler) Proxy.getInvocationHandler(obj);
443           TypeVariableImpl<?> that = typeVariableInvocationHandler.typeVariableImpl;
444           return name.equals(that.getName())
445               && genericDeclaration.equals(that.getGenericDeclaration())
446               && bounds.equals(that.bounds);
447         }
448         return false;
449       } else {
450         // equal to any TypeVariable implementation regardless of bounds
451         if (obj instanceof TypeVariable) {
452           TypeVariable<?> that = (TypeVariable<?>) obj;
453           return name.equals(that.getName())
454               && genericDeclaration.equals(that.getGenericDeclaration());
455         }
456         return false;
457       }
458     }
459   }

461   static final class WildcardTypeImpl implements WildcardType, Serializable {
462 
463     private final ImmutableList<Type> lowerBounds;
464     private final ImmutableList<Type> upperBounds;
465 
466     WildcardTypeImpl(Type[] lowerBounds, Type[] upperBounds) {
467       disallowPrimitiveType(lowerBounds, "lower bound for wildcard");
468       disallowPrimitiveType(upperBounds, "upper bound for wildcard");
469       this.lowerBounds = JavaVersion.CURRENT.usedInGenericType(lowerBounds);
470       this.upperBounds = JavaVersion.CURRENT.usedInGenericType(upperBounds);
471     }
472 
473     @Override
474     public Type[] getLowerBounds() {
475       return toArray(lowerBounds);
476     }
477 
478     @Override
479     public Type[] getUpperBounds() {
480       return toArray(upperBounds);
481     }
482 
483     @Override
484     public boolean equals(Object obj) {
485       if (obj instanceof WildcardType) {
486         WildcardType that = (WildcardType) obj;
487         return lowerBounds.equals(Arrays.asList(that.getLowerBounds()))
488             && upperBounds.equals(Arrays.asList(that.getUpperBounds()));
489       }
490       return false;
491     }
492 
493     @Override
494     public int hashCode() {
495       return lowerBounds.hashCode() ^ upperBounds.hashCode();
496     }
497 
498     @Override
499     public String toString() {
500       StringBuilder builder = new StringBuilder("?");
501       for (Type lowerBound : lowerBounds) {
502         builder.append(" super ").append(JavaVersion.CURRENT.typeName(lowerBound));
503       }
504       for (Type upperBound : filterUpperBounds(upperBounds)) {
505         builder.append(" extends ").append(JavaVersion.CURRENT.typeName(upperBound));
506       }
507       return builder.toString();
508     }
509 
510     private static final long serialVersionUID = 0;
511   }

513   private static Type[] toArray(Collection<Type> types) {
514     return types.toArray(new Type[0]);
515   }

517   private static Iterable<Type> filterUpperBounds(Iterable<Type> bounds) {
518     return Iterables.filter(bounds, Predicates.not(Predicates.<Type>equalTo(Object.class)));
519   }

521   private static void disallowPrimitiveType(Type[] types, String usedAs) {
522     for (Type type : types) {
523       if (type instanceof Class) {
524         Class<?> cls = (Class<?>) type;
525         checkArgument(!cls.isPrimitive(), "Primitive type '%s' used as %s", cls, usedAs);
526       }
527     }

538   // TODO(benyu): Once behavior is the same for all Java versions we support, delete this.
539   enum JavaVersion {
540     JAVA6 {
541       @Override
542       GenericArrayType newArrayType(Type componentType) {
543         return new GenericArrayTypeImpl(componentType);
544       }
545 
546       @Override
547       Type usedInGenericType(Type type) {
548         checkNotNull(type);
549         if (type instanceof Class) {
550           Class<?> cls = (Class<?>) type;
551           if (cls.isArray()) {
552             return new GenericArrayTypeImpl(cls.getComponentType());
553           }
554         }
555         return type;
556       }
557     },
558     JAVA7 {
559       @Override
560       Type newArrayType(Type componentType) {
561         if (componentType instanceof Class) {
562           return getArrayClass((Class<?>) componentType);
563         } else {
564           return new GenericArrayTypeImpl(componentType);
565         }
566       }
567 
568       @Override
569       Type usedInGenericType(Type type) {
570         return checkNotNull(type);
571       }
572     },
573     JAVA8 {
574       @Override
575       Type newArrayType(Type componentType) {
576         return JAVA7.newArrayType(componentType);
577       }
578 
579       @Override
580       Type usedInGenericType(Type type) {
581         return JAVA7.usedInGenericType(type);
582       }
583 
584       @Override
585       String typeName(Type type) {
586         try {
587           Method getTypeName = Type.class.getMethod("getTypeName");
588           return (String) getTypeName.invoke(type);
589         } catch (NoSuchMethodException e) {
590           throw new AssertionError("Type.getTypeName should be available in Java 8");
591           /*
592            * Do not merge the 2 catch blocks below. javac would infer a type of
593            * ReflectiveOperationException, which Animal Sniffer would reject. (Old versions of
594            * Android don't *seem* to mind, but there might be edge cases of which we're unaware.)
595            */
596         } catch (InvocationTargetException e) {
597           throw new RuntimeException(e);
598         } catch (IllegalAccessException e) {
599           throw new RuntimeException(e);
600         }
601       }
602     },
603     JAVA9 {
604       @Override
605       Type newArrayType(Type componentType) {
606         return JAVA8.newArrayType(componentType);
607       }
608 
609       @Override
610       Type usedInGenericType(Type type) {
611         return JAVA8.usedInGenericType(type);
612       }
613 
614       @Override
615       String typeName(Type type) {
616         return JAVA8.typeName(type);
617       }
618 
619       @Override
620       boolean jdkTypeDuplicatesOwnerName() {
621         return false;
622       }
623     };

625     static final JavaVersion CURRENT;
626 
627     static {
628       if (AnnotatedElement.class.isAssignableFrom(TypeVariable.class)) {
629         if (new TypeCapture<Entry<String, int[][]>>() {}.capture()
630             .toString()
631             .contains("java.util.Map.java.util.Map")) {
632           CURRENT = JAVA8;
633         } else {
634           CURRENT = JAVA9;
635         }
636       } else if (new TypeCapture<int[]>() {}.capture() instanceof Class) {
637         CURRENT = JAVA7;
638       } else {
639         CURRENT = JAVA6;
640       }
641     }

647     final ImmutableList<Type> usedInGenericType(Type[] types) {
648       ImmutableList.Builder<Type> builder = ImmutableList.builder();
649       for (Type type : types) {
650         builder.add(usedInGenericType(type));
651       }
652       return builder.build();

664   /**
665    * Per <a href="https://code.google.com/p/guava-libraries/issues/detail?id=1635">issue 1635</a>,
666    * In JDK 1.7.0_51-b13, {@link TypeVariableImpl#equals(Object)} is changed to no longer be equal
667    * to custom TypeVariable implementations. As a result, we need to make sure our TypeVariable
668    * implementation respects symmetry. Moreover, we don't want to reconstruct a native type variable
669    * {@code <A>} using our implementation unless some of its bounds have changed in resolution. This
670    * avoids creating unequal TypeVariable implementation unnecessarily. When the bounds do change,
671    * however, it's fine for the synthetic TypeVariable to be unequal to any native TypeVariable
672    * anyway.
673    */
674   static final class NativeTypeVariableEquals<X> {
675     static final boolean NATIVE_TYPE_VARIABLE_ONLY =
676         !NativeTypeVariableEquals.class.getTypeParameters()[0].equals(
677             newArtificialTypeVariable(NativeTypeVariableEquals.class, "X"));
678   }

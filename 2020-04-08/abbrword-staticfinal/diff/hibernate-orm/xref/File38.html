44  /**
45   * Provides access to configuration info passed in <tt>Properties</tt> objects.
46   * <br><br>
47   * Hibernate has two property scopes:
48   * <ul>
49   * <li><b>Factory-level</b> properties may be passed to the <tt>SessionFactory</tt> when it
50   * instantiated. Each instance might have different property values. If no
51   * properties are specified, the factory calls <tt>Environment.getProperties()</tt>.
52   * <li><b>System-level</b> properties are shared by all factory instances and are always
53   * determined by the <tt>Environment</tt> properties.
54   * </ul>
55   * The only system-level properties are
56   * <ul>
57   * <li><tt>hibernate.jdbc.use_streams_for_binary</tt>
58   * <li><tt>hibernate.cglib.use_reflection_optimizer</tt>
59   * </ul>
60   * <tt>Environment</tt> properties are populated by calling <tt>System.getProperties()</tt>
61   * and then from a resource named <tt>/hibernate.properties</tt> if it exists. System
62   * properties override properties specified in <tt>hibernate.properties</tt>.<br>
63   * <br>
64   * The <tt>SessionFactory</tt> is controlled by the following properties.
65   * Properties may be either be <tt>System</tt> properties, properties
66   * defined in a resource named <tt>/hibernate.properties</tt> or an instance of
67   * <tt>java.util.Properties</tt> passed to
68   * <tt>Configuration.buildSessionFactory()</tt><br>
69   * <br>
70   * <table>
71   * <tr><td><b>property</b></td><td><b>meaning</b></td></tr>
72   * <tr>
73   *   <td><tt>hibernate.dialect</tt></td>
74   *   <td>classname of <tt>org.hibernate.dialect.Dialect</tt> subclass</td>
75   * </tr>
76   * <tr>
77   *   <td><tt>hibernate.connection.provider_class</tt></td>
78   *   <td>classname of <tt>org.hibernate.service.jdbc.connections.spi.ConnectionProvider</tt>
79   *   subclass (if not specified hueristics are used)</td>
80   * </tr>
81   * <tr><td><tt>hibernate.connection.username</tt></td><td>database username</td></tr>
82   * <tr><td><tt>hibernate.connection.password</tt></td><td>database password</td></tr>
83   * <tr>
84   *   <td><tt>hibernate.connection.url</tt></td>
85   *   <td>JDBC URL (when using <tt>java.sql.DriverManager</tt>)</td>
86   * </tr>
87   * <tr>
88   *   <td><tt>hibernate.connection.driver_class</tt></td>
89   *   <td>classname of JDBC driver</td>
90   * </tr>
91   * <tr>
92   *   <td><tt>hibernate.connection.isolation</tt></td>
93   *   <td>JDBC transaction isolation level (only when using
94   *     <tt>java.sql.DriverManager</tt>)
95   *   </td>
96   * </tr>
97   *   <td><tt>hibernate.connection.pool_size</tt></td>
98   *   <td>the maximum size of the connection pool (only when using
99   *     <tt>java.sql.DriverManager</tt>)
100  *   </td>
101  * </tr>
102  * <tr>
103  *   <td><tt>hibernate.connection.datasource</tt></td>
104  *   <td>databasource JNDI name (when using <tt>javax.sql.Datasource</tt>)</td>
105  * </tr>
106  * <tr>
107  *   <td><tt>hibernate.jndi.url</tt></td><td>JNDI <tt>InitialContext</tt> URL</td>
108  * </tr>
109  * <tr>
110  *   <td><tt>hibernate.jndi.class</tt></td><td>JNDI <tt>InitialContext</tt> classname</td>
111  * </tr>
112  * <tr>
113  *   <td><tt>hibernate.max_fetch_depth</tt></td>
114  *   <td>maximum depth of outer join fetching</td>
115  * </tr>
116  * <tr>
117  *   <td><tt>hibernate.jdbc.batch_size</tt></td>
118  *   <td>enable use of JDBC2 batch API for drivers which support it</td>
119  * </tr>
120  * <tr>
121  *   <td><tt>hibernate.jdbc.fetch_size</tt></td>
122  *   <td>set the JDBC fetch size</td>
123  * </tr>
124  * <tr>
125  *   <td><tt>hibernate.jdbc.use_scrollable_resultset</tt></td>
126  *   <td>enable use of JDBC2 scrollable resultsets (you only need this specify
127  *   this property when using user supplied connections)</td>
128  * </tr>
129  * <tr>
130  *   <td><tt>hibernate.jdbc.use_getGeneratedKeys</tt></td>
131  *   <td>enable use of JDBC3 PreparedStatement.getGeneratedKeys() to retrieve
132  *   natively generated keys after insert. Requires JDBC3+ driver and JRE1.4+</td>
133  * </tr>
134  * <tr>
135  *   <td><tt>hibernate.hbm2ddl.auto</tt></td>
136  *   <td>enable auto DDL export</td>
137  * </tr>
138  * <tr>
139  *   <td><tt>hibernate.default_schema</tt></td>
140  *   <td>use given schema name for unqualified tables (always optional)</td>
141  * </tr>
142  * <tr>
143  *   <td><tt>hibernate.default_catalog</tt></td>
144  *   <td>use given catalog name for unqualified tables (always optional)</td>
145  * </tr>
146  * <tr>
147  *   <td><tt>hibernate.session_factory_name</tt></td>
148  *   <td>If set, the factory attempts to bind this name to itself in the
149  *   JNDI context. This name is also used to support cross JVM <tt>
150  *   Session</tt> (de)serialization.</td>
151  * </tr>
152  * <tr>
153  *   <td><tt>hibernate.transaction.jta.platform</tt></td>
154  *   <td>classname of <tt>org.hibernate.engine.transaction.jta.platform.spi.JtaPlatform</tt>
155  *   implementor</td>
156  * </tr>
157  * <tr>
158  *   <td><tt>hibernate.transaction.factory_class</tt></td>
159  *   <td>the factory to use for instantiating <tt>Transaction</tt>s.
160  *   (Defaults to <tt>JdbcTransactionFactory</tt>.)</td>
161  * </tr>
162  * <tr>
163  *   <td><tt>hibernate.query.substitutions</tt></td><td>query language token substitutions</td>
164  * </tr>
165  * </table>
166  *
167  * @see org.hibernate.SessionFactory
168  * @author Gavin King
169  */
170 public final class Environment implements AvailableSettings {
171 	private static final CoreMessageLogger LOG = Logger.getMessageLogger(CoreMessageLogger.class, Environment.class.getName());
172 
173 	private static final BytecodeProvider BYTECODE_PROVIDER_INSTANCE;
174 	private static final boolean ENABLE_BINARY_STREAMS;
175 	private static final boolean ENABLE_REFLECTION_OPTIMIZER;
176 	private static final boolean JVM_HAS_TIMESTAMP_BUG;
177 
178 	private static final Properties GLOBAL_PROPERTIES;
179 	private static final Map<Integer,String> ISOLATION_LEVELS;
180 
181 	private static final Map OBSOLETE_PROPERTIES = new HashMap();
182 	private static final Map RENAMED_PROPERTIES = new HashMap();
183 
184 	/**
185 	 * Issues warnings to the user when any obsolete or renamed property names are used.
186 	 *
187 	 * @param configurationValues The specified properties.
188 	 */
189 	public static void verifyProperties(Map<?,?> configurationValues) {
190 		final Map propertiesToAdd = new HashMap();
191 		for ( Map.Entry entry : configurationValues.entrySet() ) {
192 			final Object replacementKey = OBSOLETE_PROPERTIES.get( entry.getKey() );
193 			if ( replacementKey != null ) {
194 				LOG.unsupportedProperty( entry.getKey(), replacementKey );
195 			}
196 			final Object renamedKey = RENAMED_PROPERTIES.get( entry.getKey() );
197 			if ( renamedKey != null ) {
198 				LOG.renamedProperty( entry.getKey(), renamedKey );
199 				propertiesToAdd.put( renamedKey, entry.getValue() );
200 			}
201 		}
202 		configurationValues.putAll( propertiesToAdd );
203 	}

205 	static {
206 		Version.logVersion();
207 
208 		Map<Integer,String> temp = new HashMap<Integer,String>();
209 		temp.put( Connection.TRANSACTION_NONE, "NONE" );
210 		temp.put( Connection.TRANSACTION_READ_UNCOMMITTED, "READ_UNCOMMITTED" );
211 		temp.put( Connection.TRANSACTION_READ_COMMITTED, "READ_COMMITTED" );
212 		temp.put( Connection.TRANSACTION_REPEATABLE_READ, "REPEATABLE_READ" );
213 		temp.put( Connection.TRANSACTION_SERIALIZABLE, "SERIALIZABLE" );
214 		ISOLATION_LEVELS = Collections.unmodifiableMap( temp );
215 		GLOBAL_PROPERTIES = new Properties();
216 		//Set USE_REFLECTION_OPTIMIZER to false to fix HHH-227
217 		GLOBAL_PROPERTIES.setProperty( USE_REFLECTION_OPTIMIZER, Boolean.FALSE.toString() );
218 
219 		try {
220 			InputStream stream = ConfigHelper.getResourceAsStream( "/hibernate.properties" );
221 			try {
222 				GLOBAL_PROPERTIES.load(stream);
223 				LOG.propertiesLoaded( ConfigurationHelper.maskOut( GLOBAL_PROPERTIES, PASS ) );
224 			}
225 			catch (Exception e) {
226 				LOG.unableToLoadProperties();
227 			}
228 			finally {
229 				try{
230 					stream.close();
231 				}
232 				catch (IOException ioe){
233 					LOG.unableToCloseStreamError( ioe );
234 				}
235 			}
236 		}
237 		catch (HibernateException he) {
238 			LOG.propertiesNotFound();
239 		}
240 
241 		try {
242 		    Properties systemProperties = System.getProperties();
243 		    // Must be thread-safe in case an application changes System properties during Hibernate initialization.
244 		    // See HHH-8383.
245 		    synchronized (systemProperties) {
246 		    	GLOBAL_PROPERTIES.putAll(systemProperties);
247 		    }
248 		} catch (SecurityException se) {
249 		    LOG.unableToCopySystemProperties();
250 		}
251 
252 		verifyProperties(GLOBAL_PROPERTIES);
253 
254 		ENABLE_BINARY_STREAMS = ConfigurationHelper.getBoolean(USE_STREAMS_FOR_BINARY, GLOBAL_PROPERTIES);
255 		if ( ENABLE_BINARY_STREAMS ) {
256 			LOG.usingStreams();
257 		}
258 
259 		ENABLE_REFLECTION_OPTIMIZER = ConfigurationHelper.getBoolean(USE_REFLECTION_OPTIMIZER, GLOBAL_PROPERTIES);
260 		if ( ENABLE_REFLECTION_OPTIMIZER ) {
261 			LOG.usingReflectionOptimizer();
262 		}
263 
264 		BYTECODE_PROVIDER_INSTANCE = buildBytecodeProvider( GLOBAL_PROPERTIES );
265 
266 		long x = 123456789;
267 		JVM_HAS_TIMESTAMP_BUG = new Timestamp(x).getTime() != x;
268 		if ( JVM_HAS_TIMESTAMP_BUG ) {
269 			LOG.usingTimestampWorkaround();
270 		}
271 	}
272 
273 	public static BytecodeProvider getBytecodeProvider() {
274 		return BYTECODE_PROVIDER_INSTANCE;
275 	}

277 	/**
278 	 * Does this JVM's implementation of {@link java.sql.Timestamp} have a bug in which the following is true:<code>
279 	 * new java.sql.Timestamp( x ).getTime() != x
280 	 * </code>
281 	 * <p/>
282 	 * NOTE : IBM JDK 1.3.1 the only known JVM to exhibit this behavior.
283 	 *
284 	 * @return True if the JVM's {@link Timestamp} implementa
285 	 */
286 	public static boolean jvmHasTimestampBug() {
287 		return JVM_HAS_TIMESTAMP_BUG;
288 	}

298 		return ENABLE_BINARY_STREAMS;
299 	}

301 	/**

303 	 *
304 	 * @return True if reflection optimization should be used; false otherwise.

321 	/**
322 	 * Return <tt>System</tt> properties, extended by any properties specified
323 	 * in <tt>hibernate.properties</tt>.
324 	 * @return Properties
325 	 */
326 	public static Properties getProperties() {
327 		Properties copy = new Properties();
328 		copy.putAll(GLOBAL_PROPERTIES);
329 		return copy;
330 	}

332 	/**
333 	 * Get the name of a JDBC transaction isolation level
334 	 *
335 	 * @see java.sql.Connection
336 	 * @param isolation as defined by <tt>java.sql.Connection</tt>
337 	 * @return a human-readable name
338 	 */
339 	public static String isolationLevelToString(int isolation) {
340 		return ISOLATION_LEVELS.get( isolation );
341 	}

343 	public static BytecodeProvider buildBytecodeProvider(Properties properties) {
344 		String provider = ConfigurationHelper.getString( BYTECODE_PROVIDER, properties, "javassist" );
345 		LOG.bytecodeProvider( provider );
346 		return buildBytecodeProvider( provider );
347 	}
348 
349 	private static BytecodeProvider buildBytecodeProvider(String providerName) {
350 		if ( "javassist".equals( providerName ) ) {
351 			return new org.hibernate.bytecode.internal.javassist.BytecodeProviderImpl();
352 		}
353 
354 		LOG.unknownBytecodeProvider( providerName );
355 		return new org.hibernate.bytecode.internal.javassist.BytecodeProviderImpl();
356 	}

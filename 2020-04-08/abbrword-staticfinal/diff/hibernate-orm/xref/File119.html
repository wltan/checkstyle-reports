24  package org.hibernate.exception.internal;
25  
26  import java.sql.SQLException;
27  import java.util.HashSet;
28  import java.util.Set;
29  
30  import org.hibernate.JDBCException;
31  import org.hibernate.exception.ConstraintViolationException;
32  import org.hibernate.exception.DataException;
33  import org.hibernate.exception.spi.AbstractSQLExceptionConversionDelegate;
34  import org.hibernate.exception.spi.ConversionContext;
35  import org.hibernate.internal.util.JdbcExceptionHelper;
36  
37  /**
38   * A {@link org.hibernate.exception.spi.SQLExceptionConversionDelegate}
39   * implementation specific to Cach&eacute; SQL, accounting for its custom
40   * integrity constraint violation error codes.
41   *
42   * @author Jonathan Levinson
43   */
44  public class CacheSQLExceptionConversionDelegate extends AbstractSQLExceptionConversionDelegate {
45  
46  	private static final Set<String> DATA_CATEGORIES = new HashSet<String>();
47  	private static final Set<Integer> INTEGRITY_VIOLATION_CATEGORIES = new HashSet<Integer>();
48  
49  	static {
50  		DATA_CATEGORIES.add( "22" );
51  		DATA_CATEGORIES.add( "21" );
52  		DATA_CATEGORIES.add( "02" );
53  
54  		INTEGRITY_VIOLATION_CATEGORIES.add( 119 );
55  		INTEGRITY_VIOLATION_CATEGORIES.add( 120 );
56  		INTEGRITY_VIOLATION_CATEGORIES.add( 121 );
57  		INTEGRITY_VIOLATION_CATEGORIES.add( 122 );
58  		INTEGRITY_VIOLATION_CATEGORIES.add( 123 );
59  		INTEGRITY_VIOLATION_CATEGORIES.add( 124 );
60  		INTEGRITY_VIOLATION_CATEGORIES.add( 125 );
61  		INTEGRITY_VIOLATION_CATEGORIES.add( 127 );
62  	}
63  
64  	public CacheSQLExceptionConversionDelegate(ConversionContext conversionContext) {
65  		super( conversionContext );
66  	}
67  
68  	/**
69  	 * Convert the given SQLException into Hibernate's JDBCException hierarchy.
70  	 *
71  	 * @param sqlException The SQLException to be converted.
72  	 * @param message	  An optional error message.
73  	 * @param sql		  Optionally, the sql being performed when the exception occurred.
74  	 * @return The resulting JDBCException; returns null if it could not be converted.
75  	 */
76  	@Override
77  	public JDBCException convert(SQLException sqlException, String message, String sql) {
78  		String sqlStateClassCode = JdbcExceptionHelper.extractSqlStateClassCode( sqlException );
79  		if ( sqlStateClassCode != null ) {
80  			Integer errorCode = JdbcExceptionHelper.extractErrorCode( sqlException );
81  			if ( INTEGRITY_VIOLATION_CATEGORIES.contains( errorCode ) ) {
82  				String constraintName =
83  						getConversionContext()
84  								.getViolatedConstraintNameExtracter()
85  								.extractConstraintName( sqlException );
86  				return new ConstraintViolationException( message, sqlException, sql, constraintName );
87  			}
88  			else if ( DATA_CATEGORIES.contains( sqlStateClassCode ) ) {
89  				return new DataException( message, sqlException, sql );
90  			}
91  		}
92  		return null; // allow other delegates the chance to look
93  	}
94  }

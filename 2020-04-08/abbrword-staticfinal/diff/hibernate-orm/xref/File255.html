24  package org.hibernate.service.jta.platform.internal;
25  
26  import java.util.Map;
27  
28  import org.jboss.logging.Logger;
29  
30  import org.hibernate.cfg.AvailableSettings;
31  import org.hibernate.cfg.Environment;
32  import org.hibernate.internal.CoreMessageLogger;
33  import org.hibernate.internal.util.jndi.JndiHelper;
34  import org.hibernate.service.classloading.spi.ClassLoaderService;
35  import org.hibernate.service.config.spi.ConfigurationService;
36  import org.hibernate.service.jta.platform.spi.JtaPlatform;
37  import org.hibernate.service.jta.platform.spi.JtaPlatformException;
38  import org.hibernate.service.spi.BasicServiceInitiator;
39  import org.hibernate.service.spi.ServiceRegistryImplementor;
40  import org.hibernate.transaction.TransactionManagerLookup;
41  
42  /**
43   * Standard initiator for the standard {@link org.hibernate.service.jta.platform.spi.JtaPlatform}
44   *
45   * @author Steve Ebersole
46   */
47  public class JtaPlatformInitiator implements BasicServiceInitiator<JtaPlatform> {
48  	public static final JtaPlatformInitiator INSTANCE = new JtaPlatformInitiator();
49  
50  	private static final CoreMessageLogger LOG = Logger.getMessageLogger(CoreMessageLogger.class, JtaPlatformInitiator.class.getName());
51  
52  	@Override
53  	public Class<JtaPlatform> getServiceInitiated() {
54  		return JtaPlatform.class;
55  	}
56  
57  	@Override
58  	@SuppressWarnings( {"unchecked"})
59  	public JtaPlatform initiateService(Map configurationValues, ServiceRegistryImplementor registry) {
60  		final Object platform = getConfiguredPlatform( configurationValues, registry );
61  		if ( platform == null ) {
62  			return new NoJtaPlatform();
63  		}
64  		return registry.getService( ConfigurationService.class )
65  				.cast( JtaPlatform.class, platform );
66  
67  	}
68  
69  	private Object getConfiguredPlatform(Map configVales, ServiceRegistryImplementor registry) {
70  		Object platform = configVales.get( AvailableSettings.JTA_PLATFORM );
71  		if ( platform == null ) {
72  			final String transactionManagerLookupImplName = (String) configVales.get( Environment.TRANSACTION_MANAGER_STRATEGY );
73  			if ( transactionManagerLookupImplName != null ) {
74                  LOG.deprecatedTransactionManagerStrategy(TransactionManagerLookup.class.getName(),
75                                                           Environment.TRANSACTION_MANAGER_STRATEGY,
76                                                           JtaPlatform.class.getName(),
77                                                           AvailableSettings.JTA_PLATFORM);
78  				platform = mapLegacyClasses( transactionManagerLookupImplName, configVales, registry );
79                  LOG.debugf("Mapped %s -> %s", transactionManagerLookupImplName, platform);
80  			}
81  		}
82  		return platform;
83  	}
84  
85  	private JtaPlatform mapLegacyClasses(String tmlImplName, Map configVales, ServiceRegistryImplementor registry) {
86  		if ( tmlImplName == null ) {
87  			return null;
88  		}
89  
90          LOG.legacyTransactionManagerStrategy(JtaPlatform.class.getName(), AvailableSettings.JTA_PLATFORM);
91  
92  		if ( "org.hibernate.transaction.BESTransactionManagerLookup".equals( tmlImplName ) ) {
93  			return new BorlandEnterpriseServerJtaPlatform();
94  		}
95  
96  		if ( "org.hibernate.transaction.BTMTransactionManagerLookup".equals( tmlImplName ) ) {
97  			return new BitronixJtaPlatform();
98  		}
99  
100 		if ( "org.hibernate.transaction.JBossTransactionManagerLookup".equals( tmlImplName ) ) {
101 			return new JBossAppServerJtaPlatform();
102 		}
103 
104 		if ( "org.hibernate.transaction.JBossTSStandaloneTransactionManagerLookup".equals( tmlImplName ) ) {
105 			return new JBossStandAloneJtaPlatform();
106 		}
107 
108 		if ( "org.hibernate.transaction.JOnASTransactionManagerLookup".equals( tmlImplName ) ) {
109 			return new JOnASJtaPlatform();
110 		}
111 
112 		if ( "org.hibernate.transaction.JOTMTransactionManagerLookup".equals( tmlImplName ) ) {
113 			return new JOTMJtaPlatform();
114 		}
115 
116 		if ( "org.hibernate.transaction.JRun4TransactionManagerLookup".equals( tmlImplName ) ) {
117 			return new JRun4JtaPlatform();
118 		}
119 
120 		if ( "org.hibernate.transaction.OC4JTransactionManagerLookup".equals( tmlImplName ) ) {
121 			return new OC4JJtaPlatform();
122 		}
123 
124 		if ( "org.hibernate.transaction.OrionTransactionManagerLookup".equals( tmlImplName ) ) {
125 			return new OrionJtaPlatform();
126 		}
127 
128 		if ( "org.hibernate.transaction.ResinTransactionManagerLookup".equals( tmlImplName ) ) {
129 			return new ResinJtaPlatform();
130 		}
131 
132 		if ( "org.hibernate.transaction.SunONETransactionManagerLookup".equals( tmlImplName ) ) {
133 			return new SunOneJtaPlatform();
134 		}
135 
136 		if ( "org.hibernate.transaction.WeblogicTransactionManagerLookup".equals( tmlImplName ) ) {
137 			return new WeblogicJtaPlatform();
138 		}
139 
140 		if ( "org.hibernate.transaction.WebSphereTransactionManagerLookup".equals( tmlImplName ) ) {
141 			return new WebSphereJtaPlatform();
142 		}
143 
144 		if ( "org.hibernate.transaction.WebSphereExtendedJTATransactionLookup".equals( tmlImplName ) ) {
145 			return new WebSphereExtendedJtaPlatform();
146 		}
147 
148 		try {
149 			TransactionManagerLookup lookup = (TransactionManagerLookup) registry.getService( ClassLoaderService.class )
150 					.classForName( tmlImplName )
151 					.newInstance();
152 			return new TransactionManagerLookupBridge( lookup, JndiHelper.extractJndiProperties( configVales ) );
153 		}
154 		catch ( Exception e ) {
155 			throw new JtaPlatformException(
156 					"Unable to build " + TransactionManagerLookupBridge.class.getName() + " from specified " +
157 							TransactionManagerLookup.class.getName() + " implementation: " +
158 							tmlImplName
159 			);
160 		}
161 	}
162 }

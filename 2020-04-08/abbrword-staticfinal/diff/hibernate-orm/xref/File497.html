22  package org.hibernate.ejb.engine.spi;
23  
24  import static org.hibernate.engine.spi.CascadeStyle.STYLES;
25  
26  import org.hibernate.engine.spi.CascadeStyle;
27  import org.hibernate.engine.spi.CascadingAction;
28  
29  /**
30   * Becasue CascadeStyle is not opened and package protected,
31   * I need to subclass and override the persist alias
32   *
33   * Note that This class has to be triggered by EJB3PersistEventListener at class loading time
34   *
35   * TODO get rid of it for 3.3
36   *
37   * @author Emmanuel Bernard
38   */
39  public abstract class EJB3CascadeStyle extends CascadeStyle {
40  
41  	/**
42  	 * cascade using EJB3CascadingAction
43  	 */
44  	public static final CascadeStyle PERSIST_EJB3 = new CascadeStyle() {
45  		public boolean doCascade(CascadingAction action) {
46  			return action== EJB3CascadingAction.PERSIST_SKIPLAZY
47  					|| action==CascadingAction.PERSIST_ON_FLUSH;
48  		}
49  		public String toString() {
50  			return "STYLE_PERSIST_SKIPLAZY";
51  		}
52  	};
53  
54  	static {
55  		STYLES.put( "persist", PERSIST_EJB3 );
56  	}
57  }

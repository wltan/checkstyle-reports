42  /**
43   * Component that initiates leader election based on holding a lock. If the lock has the
44   * right properties (global with expiry), there will never be more than one leader, but
45   * there may occasionally be no leader for short periods. If the lock has stronger
46   * guarantees, and it interrupts the holder's thread when it expires or is stolen, then
47   * you can adjust the parameters to reduce the leaderless period to be limited only by
48   * latency to the lock provider. The election process ties up a thread perpetually while
49   * we hold and try to acquire the lock, so a native leader initiator (not based on a lock)
50   * is likely to be more efficient. If there is no native leader initiator available, but
51   * there is a lock registry (e.g. on a shared database), this implementation is likely to
52   * be useful.
53   *
54   * @author Dave Syer
55   * @author Artem Bilan
56   * @author Vedran Pavic
57   * @author Glenn Renfro
58   * @author Kiel Boatman
59   * @author Gary Russell
60   *
61   * @since 4.3.1
62   */
63  public class LockRegistryLeaderInitiator implements SmartLifecycle, DisposableBean, ApplicationEventPublisherAware {
64  
65  	public static final long DEFAULT_HEART_BEAT_TIME = 500L;
66  
67  	public static final long DEFAULT_BUSY_WAIT_TIME = 50L;
68  
69  	private static final Log logger = LogFactory.getLog(LockRegistryLeaderInitiator.class);
70  
71  	private final Object lifecycleMonitor = new Object();
72  
73  	/**
74  	 * A lock registry. The locks it manages should be global (whatever that means for the
75  	 * system) and expiring, in case the holder dies without notifying anyone.
76  	 */
77  	private final LockRegistry locks;

80  	 * Candidate for leader election. User injects this to receive callbacks on leadership
81  	 * events. Alternatively applications can listen for the
82  	 * {@link org.springframework.integration.leader.event.OnGrantedEvent} and
83  	 * {@link org.springframework.integration.leader.event.OnRevokedEvent}, as long as the
84  	 * {@link #setLeaderEventPublisher(LeaderEventPublisher) leaderEventPublisher} is set.
85  	 */
86  	private final Candidate candidate;

102 	/**
103 	 * Executor service for running leadership daemon.
104 	 */
105 	private ExecutorService executorService =
106 			Executors.newSingleThreadExecutor(new CustomizableThreadFactory("lock-leadership-"));
107 
108 	/**
109 	 * Flag to denote whether the {@link ExecutorService} was provided via the setter and
110 	 * thus should not be shutdown when {@link #destroy()} is called
111 	 */
112 	private boolean executorServiceExplicitlySet;
113 
114 	/**
115 	 * Time in milliseconds to wait in between attempts to re-acquire the lock, once it is
116 	 * held. The heartbeat time has to be less than the remote lock expiry period, if
117 	 * there is one, otherwise other nodes can steal the lock while we are sleeping here.
118 	 * If the remote lock does not expire, or if you know it interrupts the current thread
119 	 * when it expires or is broken, then you can extend the heartbeat to Long.MAX_VALUE.
120 	 */
121 	private long heartBeatMillis = DEFAULT_HEART_BEAT_TIME;
122 
123 	/**
124 	 * Time in milliseconds to wait in between attempts to acquire the lock, if it is not
125 	 * held. The longer this is, the longer the system can be leaderless, if the leader
126 	 * dies. If a leader dies without releasing its lock, the system might still have to
127 	 * wait for the old lock to expire, but after that it should not have to wait longer
128 	 * than the busy wait time to get a new leader. If the remote lock does not expire, or
129 	 * if you know it interrupts the current thread when it expires or is broken, then you
130 	 * can reduce the busy wait to zero.
131 	 */
132 	private long busyWaitMillis = DEFAULT_BUSY_WAIT_TIME;
133 
134 	private boolean publishFailedEvents = false;
135

137 
138 	private ApplicationEventPublisher applicationEventPublisher;
139 
140 	/**
141 	 * Leader event publisher if set.

143 	private LeaderEventPublisher leaderEventPublisher;

168 	 * Create a new leader initiator with the provided lock registry and a default

177 	 * Create a new leader initiator. The candidate implementation is provided by the user

285 	/**
286 	 * Start the registration of the {@link #candidate} for leader election.
287 	 */
288 	@Override
289 	public void start() {
290 		if (this.leaderEventPublisher == null && this.applicationEventPublisher != null) {
291 			this.leaderEventPublisher = new DefaultLeaderEventPublisher(this.applicationEventPublisher);
292 		}
293 		synchronized (this.lifecycleMonitor) {
294 			if (!this.running) {
295 				this.leaderSelector = new LeaderSelector(buildLeaderPath());
296 				this.running = true;
297 				this.future = this.executorService.submit(this.leaderSelector);
298 				logger.debug("Started LeaderInitiator");
299 			}
300 		}
301 	}
302 
303 	@Override
304 	public void destroy() {
305 		stop();
306 		if (!this.executorServiceExplicitlySet) {
307 			this.executorService.shutdown();
308 		}
309 	}
310 
311 	/**
312 	 * Stop the registration of the {@link #candidate} for leader election. If the
313 	 * candidate is currently leader, its leadership will be revoked.
314 	 */
315 	@Override
316 	public void stop() {
317 		synchronized (this.lifecycleMonitor) {
318 			if (this.running) {
319 				this.running = false;
320 				if (this.future != null) {
321 					this.future.cancel(true);
322 				}
323 				this.future = null;
324 				logger.debug("Stopped LeaderInitiator for " + getContext());
325 			}
326 		}
327 	}

329 	/**
330 	 * @return the lock key used by leader election
331 	 */
332 	private String buildLeaderPath() {
333 		return this.candidate.getRole();
334 	}
335 
336 	protected class LeaderSelector implements Callable<Void> {
337 
338 		private final Lock lock;
339 
340 		private final String lockKey;
341 
342 		private final LockContext context = new LockContext();
343 
344 		private volatile boolean locked = false;
345 
346 		LeaderSelector(String lockKey) {
347 			this.lock = LockRegistryLeaderInitiator.this.locks.obtain(lockKey);
348 			this.lockKey = lockKey;
349 		}
350 
351 		@Override
352 		public Void call() {
353 			try {
354 				while (isRunning()) {
355 					try {
356 						tryAcquireLock();
357 					}
358 					catch (Exception e) {
359 						if (handleLockException(e)) {
360 							return null;
361 						}
362 					}
363 				}
364 			}
365 			finally {
366 				if (this.locked) {
367 					this.locked = false;
368 					try {
369 						this.lock.unlock();
370 					}
371 					catch (Exception e) {
372 						logger.debug("Could not unlock during stop for " + this.context
373 								+ " - treat as broken. Revoking...", e);
374 					}
375 					// We are stopping, therefore not leading any more
376 					handleRevoked();
377 				}
378 			}
379 			return null;
380 		}
381 
382 		private void tryAcquireLock() throws InterruptedException {
383 			if (logger.isDebugEnabled()) {
384 				logger.debug("Acquiring the lock for " + this.context);
385 			}
386 			// We always try to acquire the lock, in case it expired
387 			boolean acquired = this.lock.tryLock(LockRegistryLeaderInitiator.this.heartBeatMillis,
388 					TimeUnit.MILLISECONDS);
389 			if (!this.locked) {
390 				if (acquired) {
391 					// Success: we are now leader
392 					this.locked = true;
393 					handleGranted();
394 				}
395 				else if (isPublishFailedEvents()) {
396 					publishFailedToAcquire();
397 				}
398 			}
399 			else if (acquired) {
400 				// If we were able to acquire it but we were already locked we
401 				// should release it
402 				this.lock.unlock();
403 				if (isRunning()) {
404 					// Give it a chance to expire.
405 					Thread.sleep(LockRegistryLeaderInitiator.this.heartBeatMillis);
406 				}
407 			}
408 			else {
409 				this.locked = false;
410 				// We were not able to acquire it, therefore not leading any more
411 				handleRevoked();
412 				if (isRunning()) {
413 					// Try again quickly in case the lock holder dropped it
414 					Thread.sleep(LockRegistryLeaderInitiator.this.busyWaitMillis);
415 				}
416 			}
417 		}
418 
419 		private boolean handleLockException(Exception ex) { // NOSONAR
420 			if (this.locked) {
421 				this.locked = false;
422 				try {
423 					this.lock.unlock();
424 				}
425 				catch (Exception e1) {
426 					logger.debug("Could not unlock - treat as broken " + this.context +
427 							". Revoking " + (isRunning() ? " and retrying..." : "..."), e1);
428 
429 				}
430 				// The lock was broken and we are no longer leader
431 				handleRevoked();
432 			}
433 
434 			if (ex instanceof InterruptedException || Thread.currentThread().isInterrupted()) {
435 				Thread.currentThread().interrupt();
436 				if (isRunning()) {
437 					restartSelectorBecauseOfError(ex);
438 				}
439 				return true;
440 			}
441 			else {
442 				if (isRunning()) {
443 					// Give it a chance to elect some other leader.
444 					try {
445 						Thread.sleep(LockRegistryLeaderInitiator.this.busyWaitMillis);
446 					}
447 					catch (InterruptedException e1) {
448 						// Ignore interruption and let it to be caught on the next cycle.
449 						Thread.currentThread().interrupt();
450 					}
451 				}
452 				if (logger.isDebugEnabled()) {
453 					logger.debug("Error acquiring the lock for " + this.context +
454 							". " + (isRunning() ? "Retrying..." : ""), ex);
455 				}
456 			}
457 			return false;
458 		}
459 
460 		private void restartSelectorBecauseOfError(Exception ex) {
461 			logger.warn("Restarting LeaderSelector for " + this.context + " because of error.", ex);
462 			LockRegistryLeaderInitiator.this.future =
463 					LockRegistryLeaderInitiator.this.executorService.submit(
464 							() -> {
465 								// Give it a chance to elect some other leader.
466 								Thread.sleep(LockRegistryLeaderInitiator.this.busyWaitMillis);
467 								return call();
468 							});
469 		}
470 
471 		public boolean isLeader() {
472 			return this.locked;
473 		}
474 
475 		private void handleGranted() throws InterruptedException {
476 			LockRegistryLeaderInitiator.this.candidate.onGranted(this.context);
477 			if (LockRegistryLeaderInitiator.this.leaderEventPublisher != null) {
478 				try {
479 					LockRegistryLeaderInitiator.this.leaderEventPublisher.publishOnGranted(
480 							LockRegistryLeaderInitiator.this, this.context, this.lockKey);
481 				}
482 				catch (Exception e) {
483 					logger.warn("Error publishing OnGranted event.", e);
484 				}
485 			}
486 		}
487 
488 		private void handleRevoked() {
489 			LockRegistryLeaderInitiator.this.candidate.onRevoked(this.context);
490 			if (LockRegistryLeaderInitiator.this.leaderEventPublisher != null) {
491 				try {
492 					LockRegistryLeaderInitiator.this.leaderEventPublisher.publishOnRevoked(
493 							LockRegistryLeaderInitiator.this, this.context,
494 							LockRegistryLeaderInitiator.this.candidate.getRole());
495 				}
496 				catch (Exception e) {
497 					logger.warn("Error publishing OnRevoked event.", e);
498 				}
499 			}
500 		}
501 
502 		private void publishFailedToAcquire() {
503 			if (LockRegistryLeaderInitiator.this.leaderEventPublisher != null) {
504 				try {
505 					LockRegistryLeaderInitiator.this.leaderEventPublisher.publishOnFailedToAcquire(
506 							LockRegistryLeaderInitiator.this,
507 							this.context,
508 							LockRegistryLeaderInitiator.this.candidate.getRole());
509 				}
510 				catch (Exception e) {
511 					logger.warn("Error publishing OnFailedToAcquire event.", e);
512 				}
513 			}
514 		}
515 
516 	}

518 	/**
519 	 * Implementation of leadership context backed by lock registry.
520 	 */
521 	private class LockContext implements Context {
522 
523 		LockContext() {
524 		}
525 
526 		@Override
527 		public boolean isLeader() {
528 			return LockRegistryLeaderInitiator.this.leaderSelector.isLeader();
529 		}
530 
531 		@Override
532 		public void yield() {
533 			if (logger.isDebugEnabled()) {
534 				logger.debug("Yielding leadership from " + this);
535 			}
536 			if (LockRegistryLeaderInitiator.this.future != null) {
537 				LockRegistryLeaderInitiator.this.future.cancel(true);
538 			}
539 		}
540 
541 		@Override
542 		public String getRole() {
543 			return LockRegistryLeaderInitiator.this.candidate.getRole();
544 		}
545 
546 		@Override
547 		public String toString() {
548 			return "LockContext{role=" + LockRegistryLeaderInitiator.this.candidate.getRole() +
549 					", id=" + LockRegistryLeaderInitiator.this.candidate.getId() +
550 					", isLeader=" + isLeader() + "}";
551 		}

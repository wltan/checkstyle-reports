17  package org.springframework.integration.redis.outbound;
18  
19  import java.util.concurrent.TimeUnit;
20  
21  import org.springframework.data.redis.connection.RedisConnectionFactory;
22  import org.springframework.data.redis.core.BoundListOperations;
23  import org.springframework.data.redis.core.RedisTemplate;
24  import org.springframework.data.redis.serializer.JdkSerializationRedisSerializer;
25  import org.springframework.data.redis.serializer.RedisSerializer;
26  import org.springframework.data.redis.serializer.StringRedisSerializer;
27  import org.springframework.integration.handler.AbstractReplyProducingMessageHandler;
28  import org.springframework.lang.Nullable;
29  import org.springframework.messaging.Message;
30  import org.springframework.util.AlternativeJdkIdGenerator;
31  import org.springframework.util.Assert;
32  import org.springframework.util.IdGenerator;
33  
34  /**
35   * @author David Liu
36   * @author Artem Bilan
37   * @author Gary Russell
38   *
39   * @since 4.1
40   */
41  public class RedisQueueOutboundGateway extends AbstractReplyProducingMessageHandler {
42  
43  	private static final String QUEUE_NAME_SUFFIX = ".reply";
44  
45  	private static final int TIMEOUT = 1000;
46  
47  	private static final IdGenerator defaultIdGenerator = new AlternativeJdkIdGenerator();
48  
49  	private static final RedisSerializer<String> stringSerializer = new StringRedisSerializer();
50  
51  	private final RedisTemplate<String, Object> template = new RedisTemplate<>();
52  
53  	private final BoundListOperations<String, Object> boundListOps;
54  
55  	private boolean extractPayload = true;
56  
57  	private RedisSerializer<?> serializer;
58  
59  	private boolean serializerExplicitlySet;
60  
61  	private int receiveTimeout = TIMEOUT;
62  
63  	public RedisQueueOutboundGateway(String queueName, RedisConnectionFactory connectionFactory) {
64  		Assert.hasText(queueName, "'queueName' is required");
65  		Assert.notNull(connectionFactory, "'connectionFactory' must not be null");
66  		this.template.setConnectionFactory(connectionFactory);
67  		this.template.setEnableDefaultSerializer(false);
68  		this.template.setKeySerializer(new StringRedisSerializer());
69  		this.template.afterPropertiesSet();
70  		this.boundListOps = this.template.boundListOps(queueName);
71  	}
72  
73  	@Override
74  	public void setBeanClassLoader(ClassLoader beanClassLoader) {
75  		super.setBeanClassLoader(beanClassLoader);
76  		if (!this.serializerExplicitlySet) {
77  			this.serializer = new JdkSerializationRedisSerializer(beanClassLoader);
78  		}
79  	}
80  
81  	public void setReceiveTimeout(int timeout) {
82  		this.receiveTimeout = timeout;
83  	}
84  
85  	public void setExtractPayload(boolean extractPayload) {
86  		this.extractPayload = extractPayload;
87  	}
88  
89  	public void setSerializer(RedisSerializer<?> serializer) {
90  		Assert.notNull(serializer, "'serializer' must not be null");
91  		this.serializer = serializer;
92  		this.serializerExplicitlySet = true;
93  	}
94  
95  	@Override
96  	public String getComponentType() {
97  		return "redis:queue-outbound-gateway";
98  	}
99  
100 	@Override
101 	@SuppressWarnings("unchecked")
102 	@Nullable
103 	protected Object handleRequestMessage(Message<?> message) {
104 		Object value = message;
105 
106 		if (this.extractPayload) {
107 			value = message.getPayload();
108 		}
109 		Object beforeSerialization = value;
110 		if (!(value instanceof byte[])) {
111 			if (value instanceof String && !this.serializerExplicitlySet) {
112 				value = stringSerializer.serialize((String) value);
113 			}
114 			else {
115 				value = ((RedisSerializer<Object>) this.serializer).serialize(value);
116 			}
117 		}
118 		if (value == null) {
119 			if (this.logger.isDebugEnabled()) {
120 				this.logger.debug("Serializer produced null for " + beforeSerialization);
121 			}
122 			return null;
123 		}
124 		String uuid = defaultIdGenerator.generateId().toString();
125 
126 		byte[] uuidByte = uuid.getBytes();
127 		this.boundListOps.leftPush(uuidByte);
128 		this.template.boundListOps(uuid).leftPush(value);
129 
130 		BoundListOperations<String, Object> boundListOperations = this.template.boundListOps(uuid + QUEUE_NAME_SUFFIX);
131 		byte[] reply = (byte[]) boundListOperations.rightPop(this.receiveTimeout, TimeUnit.MILLISECONDS);
132 		if (reply != null && reply.length > 0) {
133 			return createReply(reply);
134 		}
135 		return null;
136 	}
137 
138 	@Nullable
139 	private Object createReply(byte[] reply) {
140 		Object replyMessage = this.serializer.deserialize(reply);
141 		if (replyMessage == null) {
142 			return null;
143 		}
144 		if (this.extractPayload) {
145 			return getMessageBuilderFactory()
146 					.withPayload(replyMessage);
147 		}
148 		else {
149 			return replyMessage;
150 		}
151 	}
152 
153 }

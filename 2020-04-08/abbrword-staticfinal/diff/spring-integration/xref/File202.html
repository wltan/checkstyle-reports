17  package org.springframework.integration.ip.tcp.serializer;
18  
19  import java.io.IOException;
20  import java.io.InputStream;
21  import java.io.OutputStream;
22  
23  import org.springframework.integration.mapping.MessageMappingException;
24  
25  /**
26   * Reads data in an InputStream to a byte[]; data must be prefixed by &lt;stx&gt; and
27   * terminated by &lt;etx&gt; (not included in resulting byte[]).
28   * Writes a byte[] to an OutputStream prefixed by &lt;stx&gt; terminated by &lt;etx&gt;
29   *
30   * @author Gary Russell
31   * @since 2.0
32   */
33  public class ByteArrayStxEtxSerializer extends AbstractPooledBufferByteArraySerializer {
34  
35  	/**
36  	 * A single reusable instance.
37  	 */
38  	public static final ByteArrayStxEtxSerializer INSTANCE = new ByteArrayStxEtxSerializer();
39  
40  	public static final int STX = 0x02;
41  
42  	public static final int ETX = 0x03;
43  
44  	/**
45  	 * Reads the data in the inputStream to a byte[]. Data must be prefixed
46  	 * with an ASCII STX character, and terminated with an ASCII ETX character.
47  	 * Throws a {@link SoftEndOfStreamException} if the stream
48  	 * is closed immediately before the STX (i.e. no data is in the process of
49  	 * being read).
50  	 *
51  	 */
52  	@Override
53  	public byte[] doDeserialize(InputStream inputStream, byte[] buffer) throws IOException {
54  		int bite = inputStream.read();
55  		if (bite < 0) {
56  			throw new SoftEndOfStreamException("Stream closed between payloads");
57  		}
58  		int n = 0;
59  		try {
60  			if (bite != STX) {
61  				throw new MessageMappingException("Expected STX to begin message");
62  			}
63  			while ((bite = inputStream.read()) != ETX) {
64  				checkClosure(bite);
65  				buffer[n++] = (byte) bite;
66  				if (n >= getMaxMessageSize()) {
67  					throw new IOException("ETX not found before max message length: "
68  							+ getMaxMessageSize());
69  				}
70  			}
71  			return copyToSizedArray(buffer, n);
72  		}
73  		catch (IOException e) {
74  			publishEvent(e, buffer, n);
75  			throw e;
76  		}
77  		catch (RuntimeException e) {
78  			publishEvent(e, buffer, n);
79  			throw e;
80  		}
81  	}
82  
83  	/**
84  	 * Writes the byte[] to the stream, prefixed by an ASCII STX character and
85  	 * terminated with an ASCII ETX character.
86  	 */
87  	@Override
88  	public void serialize(byte[] bytes, OutputStream outputStream) throws IOException {
89  		outputStream.write(STX);
90  		outputStream.write(bytes);
91  		outputStream.write(ETX);
92  	}
93  
94  }

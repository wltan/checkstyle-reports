296 	@Override
297 	@Nullable
298 	protected Object handleRequestMessage(Message<?> requestMessage) {
299 		HttpMethod httpMethod = determineHttpMethod(requestMessage);
300 		if (this.extractPayloadExplicitlySet && logger.isWarnEnabled() && !shouldIncludeRequestBody(httpMethod)) {
301 			logger.warn("The 'extractPayload' attribute has no relevance for the current request " +
302 					"since the HTTP Method is '" + httpMethod +
303 					"', and no request body will be sent for that method.");
304 		}
305 
306 		Object expectedResponseType = determineExpectedResponseType(requestMessage);
307 
308 		HttpEntity<?> httpRequest = generateHttpRequest(requestMessage, httpMethod);
309 		Object uri = this.uriExpression.getValue(this.evaluationContext, requestMessage);
310 		Assert.state(uri instanceof String || uri instanceof URI,
311 				() -> "'uriExpression' evaluation must result in a 'String' or 'URI' instance, not: "
312 						+ (uri == null ? "null" : uri.getClass()));
313 
314 		Map<String, ?> uriVariables = null;
315 
316 		if (uri instanceof String) {
317 			uriVariables = determineUriVariables(requestMessage);
318 		}
319 
320 		return exchange(uri, httpMethod, httpRequest, expectedResponseType, requestMessage, uriVariables);
321 	}

324 	protected abstract Object exchange(Object uri, HttpMethod httpMethod, HttpEntity<?> httpRequest,
325 			Object expectedResponseType, Message<?> requestMessage, Map<String, ?> uriVariables);

327 	protected Object getReply(ResponseEntity<?> httpResponse) {
328 		HttpHeaders httpHeaders = httpResponse.getHeaders();
329 		Map<String, Object> headers = this.headerMapper.toHeaders(httpHeaders);
330 		if (this.transferCookies) {
331 			doConvertSetCookie(headers);
332 		}
333 
334 		AbstractIntegrationMessageBuilder<?> replyBuilder;
335 		MessageBuilderFactory messageBuilderFactory = getMessageBuilderFactory();
336 		if (httpResponse.hasBody()) {
337 			Object responseBody = httpResponse.getBody();
338 			replyBuilder = (responseBody instanceof Message<?>)
339 					? messageBuilderFactory.fromMessage((Message<?>) responseBody)
340 					: messageBuilderFactory.withPayload(responseBody); // NOSONAR - hasBody()
341 
342 		}
343 		else {
344 			replyBuilder = messageBuilderFactory.withPayload(httpResponse);
345 		}
346 		replyBuilder.setHeader(org.springframework.integration.http.HttpHeaders.STATUS_CODE,
347 				httpResponse.getStatusCode());
348 		return replyBuilder.copyHeaders(headers);
349 	}

351 	/**
352 	 * Convert Set-Cookie to Cookie
353 	 */
354 	private void doConvertSetCookie(Map<String, Object> headers) {
355 		String keyName = null;
356 		for (String key : headers.keySet()) {
357 			if (key.equalsIgnoreCase(HttpHeaders.SET_COOKIE)) {
358 				keyName = key;
359 				break;
360 			}
361 		}
362 		if (keyName != null) {
363 			Object cookies = headers.remove(keyName);
364 			headers.put(HttpHeaders.COOKIE, cookies);
365 			if (logger.isDebugEnabled()) {
366 				logger.debug("Converted Set-Cookie header to Cookie for: " + cookies);
367 			}
368 		}
369 	}

371 	private HttpEntity<?> generateHttpRequest(Message<?> message, HttpMethod httpMethod) {
372 		Assert.notNull(message, "message must not be null");
373 		return this.extractPayload
374 				? createHttpEntityFromPayload(message, httpMethod)
375 				: createHttpEntityFromMessage(message, httpMethod);
376 	}
377 
378 	private HttpEntity<?> createHttpEntityFromPayload(Message<?> message, HttpMethod httpMethod) {
379 		Object payload = message.getPayload();
380 		if (payload instanceof HttpEntity<?>) {
381 			// payload is already an HttpEntity, just return it as-is
382 			return (HttpEntity<?>) payload;
383 		}
384 		HttpHeaders httpHeaders = mapHeaders(message);
385 		if (!shouldIncludeRequestBody(httpMethod)) {
386 			return new HttpEntity<>(httpHeaders);
387 		}
388 		// otherwise, we are creating a request with a body and need to deal with the content-type header as well
389 		if (httpHeaders.getContentType() == null) {
390 			MediaType contentType =
391 					payload instanceof String
392 							? new MediaType("text", "plain", this.charset)
393 							: resolveContentType(payload);
394 			httpHeaders.setContentType(contentType);
395 		}
396 		if ((MediaType.APPLICATION_FORM_URLENCODED.equals(httpHeaders.getContentType()) ||
397 				MediaType.MULTIPART_FORM_DATA.equals(httpHeaders.getContentType()))
398 				&& !(payload instanceof MultiValueMap)) {
399 
400 			Assert.isInstanceOf(Map.class, payload,
401 					() -> "For " + MediaType.APPLICATION_FORM_URLENCODED + " and " +
402 							MediaType.MULTIPART_FORM_DATA + " media types the payload must be an instance of a Map.");
403 			payload = convertToMultiValueMap((Map<?, ?>) payload);
404 		}
405 		return new HttpEntity<>(payload, httpHeaders);
406 	}
407 
408 	private HttpEntity<?> createHttpEntityFromMessage(Message<?> message, HttpMethod httpMethod) {
409 		HttpHeaders httpHeaders = mapHeaders(message);
410 		if (shouldIncludeRequestBody(httpMethod)) {
411 			httpHeaders.setContentType(new MediaType("application", "x-java-serialized-object"));
412 			return new HttpEntity<Object>(message, httpHeaders);
413 		}
414 		return new HttpEntity<>(httpHeaders);
415 	}
416 
417 	protected HttpHeaders mapHeaders(Message<?> message) {
418 		HttpHeaders httpHeaders = new HttpHeaders();
419 		this.headerMapper.fromHeaders(message.getHeaders(), httpHeaders);
420 		return httpHeaders;
421 	}

423 	@SuppressWarnings("unchecked")
424 	private MediaType resolveContentType(Object content) {
425 		MediaType contentType = null;
426 		if (content instanceof byte[]) {
427 			contentType = MediaType.APPLICATION_OCTET_STREAM;
428 		}
429 		else if (content instanceof Source) {
430 			contentType = MediaType.TEXT_XML;
431 		}
432 		else if (content instanceof Map && isFormData((Map<Object, ?>) content)) {
433 			// We need to check separately for MULTIPART as well as URLENCODED simply because
434 			// MultiValueMap<Object, Object> is actually valid content for serialization
435 			if (isMultipart((Map<String, ?>) content)) {
436 				contentType = MediaType.MULTIPART_FORM_DATA;
437 			}
438 			else {
439 				contentType = MediaType.APPLICATION_FORM_URLENCODED;
440 			}
441 		}
442 		if (contentType == null && !(content instanceof Publisher<?>)) {
443 			contentType = new MediaType("application", "x-java-serialized-object");
444 		}
445 		return contentType;
446 	}

448 	private boolean shouldIncludeRequestBody(HttpMethod httpMethod) {
449 		return !(CollectionUtils.containsInstance(NO_BODY_HTTP_METHODS, httpMethod));
450 	}

452 	private MultiValueMap<Object, Object> convertToMultiValueMap(Map<?, ?> simpleMap) {
453 		LinkedMultiValueMap<Object, Object> multipartValueMap = new LinkedMultiValueMap<>();
454 		for (Entry<?, ?> entry : simpleMap.entrySet()) {
455 			Object key = entry.getKey();
456 			Object value = entry.getValue();
457 			if (value instanceof Object[]) {
458 				value = Arrays.asList((Object[]) value);
459 			}
460 			if (value instanceof Collection) {
461 				multipartValueMap.put(key, new ArrayList<>((Collection<?>) value));
462 			}
463 			else {
464 				multipartValueMap.add(key, value);
465 			}
466 		}
467 		return multipartValueMap;
468 	}

470 	/**
471 	 * If all keys and values are Strings, we'll consider the Map to be form data.
472 	 */
473 	private boolean isFormData(Map<Object, ?> map) {
474 		for (Object key : map.keySet()) {
475 			if (!(key instanceof String)) {
476 				return false;
477 			}
478 		}
479 		return true;
480 	}
481 
482 	/**
483 	 * If all keys are Strings, and some values are not Strings we'll consider
484 	 * the Map to be multipart/form-data
485 	 */
486 	private boolean isMultipart(Map<String, ?> map) {
487 		for (Object value : map.values()) {
488 			if (value != null) {
489 				if (value.getClass().isArray()) {
490 					value = CollectionUtils.arrayToList(value);
491 				}
492 				if (value instanceof Collection) {
493 					Collection<?> cValues = (Collection<?>) value;
494 					for (Object cValue : cValues) {
495 						if (cValue != null && !(cValue instanceof String)) {
496 							return true;
497 						}
498 					}
499 				}
500 				else if (!(value instanceof String)) {
501 					return true;
502 				}
503 			}
504 		}
505 		return false;
506 	}

508 	private HttpMethod determineHttpMethod(Message<?> requestMessage) {
509 		Object httpMethod = this.httpMethodExpression.getValue(this.evaluationContext, requestMessage);
510 		Assert.state((httpMethod instanceof String || httpMethod instanceof HttpMethod), () ->
511 				"'httpMethodExpression' evaluation must result in an 'HttpMethod' enum or its String representation, " +
512 						"not: " + (httpMethod == null ? "null" : httpMethod.getClass()));
513 		if (httpMethod instanceof HttpMethod) {
514 			return (HttpMethod) httpMethod;
515 		}
516 		else {
517 			try {
518 				return HttpMethod.valueOf((String) httpMethod);
519 			}
520 			catch (Exception ex) {
521 				throw new IllegalStateException("The 'httpMethodExpression' returned an invalid HTTP Method value: "
522 						+ httpMethod, ex);
523 			}
524 		}
525 	}

527 	private Object determineExpectedResponseType(Message<?> requestMessage) {
528 		return evaluateTypeFromExpression(requestMessage, this.expectedResponseTypeExpression, "expectedResponseType");

531 	@Nullable
532 	protected Object evaluateTypeFromExpression(Message<?> requestMessage, @Nullable Expression expression,
533 			String property) {
534 
535 		Object type = null;
536 		if (expression != null) {
537 			type = expression.getValue(this.evaluationContext, requestMessage);
538 		}
539 		if (type != null) {
540 			Class<?> typeClass = type.getClass();
541 			Assert.state(type instanceof Class<?>
542 							|| type instanceof String
543 							|| type instanceof ParameterizedTypeReference,
544 					() -> "The '" + property + "' can be an instance of 'Class<?>', 'String' " +
545 							"or 'ParameterizedTypeReference<?>'; " +
546 							"evaluation resulted in a " + typeClass + ".");
547 			if (type instanceof String && StringUtils.hasText((String) type)) {
548 				try {
549 					ApplicationContext applicationContext = getApplicationContext();
550 					type = ClassUtils.forName((String) type,
551 							applicationContext == null ? null : applicationContext.getClassLoader());
552 				}
553 				catch (ClassNotFoundException e) {
554 					throw new IllegalStateException("Cannot load class for name: " + type, e);
555 				}
556 			}
557 		}
558 		return type;
559 	}

561 	@SuppressWarnings("unchecked")
562 	private Map<String, ?> determineUriVariables(Message<?> requestMessage) {
563 		Map<String, ?> expressions;
564 
565 		EvaluationContext evaluationContextToUse = this.evaluationContext;
566 		if (this.uriVariablesExpression != null) {
567 			Object expressionsObject = this.uriVariablesExpression.getValue(this.evaluationContext, requestMessage);
568 			Assert.state(expressionsObject instanceof Map,
569 					"The 'uriVariablesExpression' evaluation must result in a 'Map'.");
570 			expressions = (Map<String, ?>) expressionsObject;
571 			if (!this.trustedSpel) {
572 				evaluationContextToUse = this.simpleEvaluationContext;
573 			}
574 		}
575 		else {
576 			expressions = this.uriVariableExpressions;
577 		}
578 
579 		return ExpressionEvalMap.from(expressions)
580 				.usingEvaluationContext(evaluationContextToUse)
581 				.withRoot(requestMessage)
582 				.build();
583 	}

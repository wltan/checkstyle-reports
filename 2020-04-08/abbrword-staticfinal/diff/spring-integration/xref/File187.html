50  /**
51   * TCP outbound gateway that uses a client connection factory. If the factory is configured
52   * for single-use connections, each request is sent on a new connection; if the factory does not use
53   * single use connections, each request is blocked until the previous response is received
54   * (or times out). Asynchronous requests/responses over the same connection are not
55   * supported - use a pair of outbound/inbound adapters for that use case.
56   * <p>
57   * {@link Lifecycle} methods delegate to the underlying {@link AbstractConnectionFactory}
58   *
59   *
60   * @author Gary Russell
61   *
62   * @since 2.0
63   */
64  public class TcpOutboundGateway extends AbstractReplyProducingMessageHandler
65  		implements TcpSender, TcpListener, Lifecycle {
66

140 	@Override
141 	protected Object handleRequestMessage(Message<?> requestMessage) {
142 		Assert.notNull(this.connectionFactory, this.getClass().getName() +
143 				" requires a client connection factory");
144 		boolean haveSemaphore = false;
145 		TcpConnection connection = null;
146 		String connectionId = null;
147 		try {
148 			haveSemaphore = acquireSemaphoreIfNeeded(requestMessage);
149 			connection = this.connectionFactory.getConnection();
150 			Long remoteTimeout = getRemoteTimeout(requestMessage);
151 			AsyncReply reply = new AsyncReply(remoteTimeout);
152 			connectionId = connection.getConnectionId();
153 			this.pendingReplies.put(connectionId, reply);
154 			if (logger.isDebugEnabled()) {
155 				logger.debug("Added pending reply " + connectionId);
156 			}
157 			connection.send(requestMessage);
158 			if (this.closeStreamAfterSend) {
159 				connection.shutdownOutput();
160 			}
161 			return getReply(requestMessage, connection, connectionId, reply);
162 		}
163 		catch (RuntimeException | IOException e) {
164 			logger.error("Tcp Gateway exception", e);
165 			if (e instanceof MessagingException) {
166 				throw (MessagingException) e;
167 			}
168 			throw new MessagingException("Failed to send or receive", e);
169 		}
170 		catch (InterruptedException e) {
171 			Thread.currentThread().interrupt();
172 			throw new MessageHandlingException(requestMessage, "Interrupted in the [" + this + ']', e);
173 		}
174 		finally {
175 			cleanUp(haveSemaphore, connection, connectionId);
176 		}
177 	}
178 
179 	private boolean acquireSemaphoreIfNeeded(Message<?> requestMessage) throws InterruptedException {
180 		if (!this.isSingleUse) {
181 			logger.debug("trying semaphore");
182 			if (!this.semaphore.tryAcquire(this.requestTimeout, TimeUnit.MILLISECONDS)) {
183 				throw new MessageTimeoutException(requestMessage, "Timed out waiting for connection");
184 			}
185 			if (logger.isDebugEnabled()) {
186 				logger.debug("got semaphore");
187 			}
188 			return true;
189 		}
190 		return false;
191 	}
192 
193 	private Long getRemoteTimeout(Message<?> requestMessage) {
194 		Long remoteTimeout = this.remoteTimeoutExpression.getValue(this.evaluationContext, requestMessage,
195 				Long.class);
196 		if (remoteTimeout == null) {
197 			remoteTimeout = DEFAULT_REMOTE_TIMEOUT;
198 			if (logger.isWarnEnabled()) {
199 				logger.warn("remoteTimeoutExpression evaluated to null; falling back to default for message "
200 						+ requestMessage);
201 			}
202 		}
203 		return remoteTimeout;
204 	}
205 
206 	private Message<?> getReply(Message<?> requestMessage, TcpConnection connection, String connectionId,
207 			AsyncReply reply) {
208 
209 		Message<?> replyMessage = reply.getReply();
210 		if (replyMessage == null) {
211 			if (logger.isDebugEnabled()) {
212 				logger.debug("Remote Timeout on " + connectionId);
213 			}
214 			// The connection is dirty - force it closed.
215 			this.connectionFactory.forceClose(connection);
216 			throw new MessageTimeoutException(requestMessage, "Timed out waiting for response");
217 		}
218 		if (logger.isDebugEnabled()) {
219 			logger.debug("Response " + replyMessage);
220 		}
221 		return replyMessage;
222 	}
223 
224 	private void cleanUp(boolean haveSemaphore, TcpConnection connection, String connectionId) {
225 		if (connectionId != null) {
226 			this.pendingReplies.remove(connectionId);
227 			if (logger.isDebugEnabled()) {
228 				logger.debug("Removed pending reply " + connectionId);
229 			}
230 			if (this.isSingleUse) {
231 				connection.close();
232 			}
233 		}
234 		if (haveSemaphore) {
235 			this.semaphore.release();
236 			logger.debug("released semaphore");
237 		}
238 	}

240 	@Override
241 	public boolean onMessage(Message<?> message) {
242 		String connectionId = message.getHeaders().get(IpHeaders.CONNECTION_ID, String.class);
243 		if (connectionId == null) {
244 			logger.error("Cannot correlate response - no connection id");
245 			publishNoConnectionEvent(message, null, "Cannot correlate response - no connection id");
246 			return false;
247 		}
248 		if (logger.isTraceEnabled()) {
249 			logger.trace("onMessage: " + connectionId + "(" + message + ")");
250 		}
251 		AsyncReply reply = this.pendingReplies.get(connectionId);
252 		if (reply == null) {
253 			if (message instanceof ErrorMessage) {
254 				/*
255 				 * Socket errors are sent here so they can be conveyed to any waiting thread.
256 				 * If there's not one, simply ignore.
257 				 */
258 				return false;
259 			}
260 			else {
261 				String errorMessage = "Cannot correlate response - no pending reply for " + connectionId;
262 				logger.error(errorMessage);
263 				publishNoConnectionEvent(message, connectionId, errorMessage);
264 				return false;
265 			}
266 		}
267 		reply.setReply(message);
268 		return false;
269 	}
270 
271 	private void publishNoConnectionEvent(Message<?> message, String connectionId, String errorMessage) {
272 		ApplicationEventPublisher applicationEventPublisher = this.connectionFactory.getApplicationEventPublisher();
273 		if (applicationEventPublisher != null) {
274 			applicationEventPublisher.publishEvent(new TcpConnectionFailedCorrelationEvent(this, connectionId,
275 					new MessagingException(message, errorMessage)));
276 		}
277 	}

354 	/**
355 	 * Class used to coordinate the asynchronous reply to its request.
356 	 *
357 	 * @author Gary Russell
358 	 * @since 2.0
359 	 */
360 	private final class AsyncReply {
361 
362 		private final CountDownLatch latch;
363 
364 		private final CountDownLatch secondChanceLatch;
365 
366 		private final long remoteTimeout;
367 
368 		private volatile Message<?> reply;
369 
370 		private AsyncReply(long remoteTimeout) {
371 			this.latch = new CountDownLatch(1);
372 			this.secondChanceLatch = new CountDownLatch(1);
373 			this.remoteTimeout = remoteTimeout;
374 		}
375 
376 		/**
377 		 * Sender blocks here until the reply is received, or we time out
378 		 * @return The return message or null if we time out
379 		 */
380 		public Message<?> getReply() {
381 			try {
382 				if (!this.latch.await(this.remoteTimeout, TimeUnit.MILLISECONDS)) {
383 					return null;
384 				}
385 			}
386 			catch (@SuppressWarnings("unused") InterruptedException e) {
387 				Thread.currentThread().interrupt();
388 			}
389 			boolean waitForMessageAfterError = true;
390 			while (this.reply instanceof ErrorMessage) {
391 				if (waitForMessageAfterError) {
392 					/*
393 					 * Possible race condition with NIO; we might have received the close
394 					 * before the reply, on a different thread.
395 					 */
396 					logger.debug("second chance");
397 					try {
398 						this.secondChanceLatch
399 								.await(TcpOutboundGateway.this.secondChanceDelay, TimeUnit.SECONDS); // NOSONAR
400 					}
401 					catch (@SuppressWarnings("unused") InterruptedException e) {
402 						Thread.currentThread().interrupt();
403 						doThrowErrorMessagePayload();
404 					}
405 					waitForMessageAfterError = false;
406 				}
407 				else {
408 					doThrowErrorMessagePayload();
409 				}
410 			}
411 			return this.reply;
412 		}
413 
414 		private void doThrowErrorMessagePayload() {
415 			if (this.reply.getPayload() instanceof MessagingException) {
416 				throw (MessagingException) this.reply.getPayload();
417 			}
418 			else {
419 				throw new MessagingException("Exception while awaiting reply", (Throwable) this.reply.getPayload());
420 			}
421 		}
422 
423 		/**
424 		 * We have a race condition when a socket is closed right after the reply is received. The close "error"
425 		 * might arrive before the actual reply. Overwrite an error with a good reply, but not vice-versa.
426 		 * @param reply the reply message.
427 		 */
428 		public void setReply(Message<?> reply) {
429 			if (this.reply == null) {
430 				this.reply = reply;
431 				this.latch.countDown();
432 			}
433 			else if (this.reply instanceof ErrorMessage) {
434 				this.reply = reply;
435 				this.secondChanceLatch.countDown();
436 			}
437 		}

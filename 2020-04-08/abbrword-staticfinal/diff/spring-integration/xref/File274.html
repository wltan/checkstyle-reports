48  /**
49   * Implementation of {@link ExpirableLockRegistry} providing a distributed lock using Redis.
50   * Locks are stored under the key {@code registryKey:lockKey}. Locks expire after
51   * (default 60) seconds. Threads unlocking an
52   * expired lock will get an {@link IllegalStateException}. This should be
53   * considered as a critical error because it is possible the protected
54   * resources were compromised.
55   * <p>
56   * Locks are reentrant.
57   * <p>
58   * <b>However, locks are scoped by the registry; a lock from a different registry with the
59   * same key (even if the registry uses the same 'registryKey') are different
60   * locks, and the second cannot be acquired by the same thread while the first is
61   * locked.</b>
62   * <p>
63   * <b>Note: This is not intended for low latency applications.</b> It is intended
64   * for resource locking across multiple JVMs.
65   * <p>
66   * {@link Condition}s are not supported.
67   *
68   * @author Gary Russell
69   * @author Konstantin Yakimov
70   * @author Artem Bilan
71   * @author Vedran Pavic
72   *
73   * @since 4.0
74   *
75   */
76  public final class RedisLockRegistry implements ExpirableLockRegistry, DisposableBean {
77  
78  	private static final Log LOGGER = LogFactory.getLog(RedisLockRegistry.class);
79  
80  	private static final long DEFAULT_EXPIRE_AFTER = 60000L;
81  
82  	private static final String OBTAIN_LOCK_SCRIPT =
83  			"local lockClientId = redis.call('GET', KEYS[1])\n" +
84  					"if lockClientId == ARGV[1] then\n" +
85  					"  redis.call('PEXPIRE', KEYS[1], ARGV[2])\n" +
86  					"  return true\n" +
87  					"elseif not lockClientId then\n" +
88  					"  redis.call('SET', KEYS[1], ARGV[1], 'PX', ARGV[2])\n" +
89  					"  return true\n" +
90  					"end\n" +
91  					"return false";
92  
93  
94  	private final Map<String, RedisLock> locks = new ConcurrentHashMap<>();
95  
96  	private final String clientId = UUID.randomUUID().toString();
97  
98  	private final String registryKey;
99  
100 	private final StringRedisTemplate redisTemplate;
101 
102 	private final RedisScript<Boolean> obtainLockScript;
103 
104 	private final long expireAfter;
105 
106 	/**
107 	 * An {@link ExecutorService} to call {@link StringRedisTemplate#delete} in
108 	 * the separate thread when the current one is interrupted.
109 	 */
110 	private Executor executor =
111 			Executors.newCachedThreadPool(new CustomizableThreadFactory("redis-lock-registry-"));
112 
113 	/**
114 	 * Flag to denote whether the {@link ExecutorService} was provided via the setter and
115 	 * thus should not be shutdown when {@link #destroy()} is called
116 	 */
117 	private boolean executorExplicitlySet;
118 
119 	/**
120 	 * Constructs a lock registry with the default (60 second) lock expiration.
121 	 * @param connectionFactory The connection factory.
122 	 * @param registryKey The key prefix for locks.
123 	 */
124 	public RedisLockRegistry(RedisConnectionFactory connectionFactory, String registryKey) {
125 		this(connectionFactory, registryKey, DEFAULT_EXPIRE_AFTER);
126 	}
127 
128 	/**
129 	 * Constructs a lock registry with the supplied lock expiration.
130 	 * @param connectionFactory The connection factory.
131 	 * @param registryKey The key prefix for locks.
132 	 * @param expireAfter The expiration in milliseconds.
133 	 */
134 	public RedisLockRegistry(RedisConnectionFactory connectionFactory, String registryKey, long expireAfter) {
135 		Assert.notNull(connectionFactory, "'connectionFactory' cannot be null");
136 		Assert.notNull(registryKey, "'registryKey' cannot be null");
137 		this.redisTemplate = new StringRedisTemplate(connectionFactory);
138 		this.obtainLockScript = new DefaultRedisScript<>(OBTAIN_LOCK_SCRIPT, Boolean.class);
139 		this.registryKey = registryKey;
140 		this.expireAfter = expireAfter;
141 	}
142 
143 	/**
144 	 * Set the {@link Executor}, where is not provided then a default of
145 	 * cached thread pool Executor will be used.
146 	 * @param executor the executor service
147 	 * @since 5.0.5
148 	 */
149 	public void setExecutor(Executor executor) {
150 		this.executor = executor;
151 		this.executorExplicitlySet = true;
152 	}
153 
154 	@Override
155 	public Lock obtain(Object lockKey) {
156 		Assert.isInstanceOf(String.class, lockKey);
157 		String path = (String) lockKey;
158 		return this.locks.computeIfAbsent(path, RedisLock::new);
159 	}
160 
161 	@Override
162 	public void expireUnusedOlderThan(long age) {
163 		Iterator<Map.Entry<String, RedisLock>> iterator = this.locks.entrySet().iterator();
164 		long now = System.currentTimeMillis();
165 		while (iterator.hasNext()) {
166 			Map.Entry<String, RedisLock> entry = iterator.next();
167 			RedisLock lock = entry.getValue();
168 			if (now - lock.getLockedAt() > age && !lock.isAcquiredInThisProcess()) {
169 				iterator.remove();
170 			}
171 		}
172 	}
173 
174 	@Override
175 	public void destroy() {
176 		if (!this.executorExplicitlySet) {
177 			((ExecutorService) this.executor).shutdown();
178 		}
179 	}
180 
181 	private final class RedisLock implements Lock {
182 
183 		private final String lockKey;
184 
185 		private final ReentrantLock localLock = new ReentrantLock();
186 
187 		private volatile boolean unlinkAvailable = true;
188 
189 		private volatile long lockedAt;
190 
191 		private RedisLock(String path) {
192 			this.lockKey = constructLockKey(path);
193 		}
194 
195 		private String constructLockKey(String path) {
196 			return RedisLockRegistry.this.registryKey + ":" + path;
197 		}
198 
199 		public long getLockedAt() {
200 			return this.lockedAt;
201 		}
202 
203 		@Override
204 		public void lock() {
205 			this.localLock.lock();
206 			while (true) {
207 				try {
208 					while (!obtainLock()) {
209 						Thread.sleep(100); //NOSONAR
210 					}
211 					break;
212 				}
213 				catch (InterruptedException e) {
214 					/*
215 					 * This method must be uninterruptible so catch and ignore
216 					 * interrupts and only break out of the while loop when
217 					 * we get the lock.
218 					 */
219 				}
220 				catch (Exception e) {
221 					this.localLock.unlock();
222 					rethrowAsLockException(e);
223 				}
224 			}
225 		}
226 
227 		private void rethrowAsLockException(Exception e) {
228 			throw new CannotAcquireLockException("Failed to lock mutex at " + this.lockKey, e);
229 		}
230 
231 		@Override
232 		public void lockInterruptibly() throws InterruptedException {
233 			this.localLock.lockInterruptibly();
234 			try {
235 				while (!obtainLock()) {
236 					Thread.sleep(100); //NOSONAR
237 				}
238 			}
239 			catch (InterruptedException ie) {
240 				this.localLock.unlock();
241 				Thread.currentThread().interrupt();
242 				throw ie;
243 			}
244 			catch (Exception e) {
245 				this.localLock.unlock();
246 				rethrowAsLockException(e);
247 			}
248 		}
249 
250 		@Override
251 		public boolean tryLock() {
252 			try {
253 				return tryLock(0, TimeUnit.MILLISECONDS);
254 			}
255 			catch (InterruptedException e) {
256 				Thread.currentThread().interrupt();
257 				return false;
258 			}
259 		}
260 
261 		@Override
262 		public boolean tryLock(long time, TimeUnit unit) throws InterruptedException {
263 			long now = System.currentTimeMillis();
264 			if (!this.localLock.tryLock(time, unit)) {
265 				return false;
266 			}
267 			try {
268 				long expire = now + TimeUnit.MILLISECONDS.convert(time, unit);
269 				boolean acquired;
270 				while (!(acquired = obtainLock()) && System.currentTimeMillis() < expire) { //NOSONAR
271 					Thread.sleep(100); //NOSONAR
272 				}
273 				if (!acquired) {
274 					this.localLock.unlock();
275 				}
276 				return acquired;
277 			}
278 			catch (Exception e) {
279 				this.localLock.unlock();
280 				rethrowAsLockException(e);
281 			}
282 			return false;
283 		}
284 
285 		private boolean obtainLock() {
286 			Boolean success =
287 					RedisLockRegistry.this.redisTemplate.execute(RedisLockRegistry.this.obtainLockScript,
288 							Collections.singletonList(this.lockKey), RedisLockRegistry.this.clientId,
289 							String.valueOf(RedisLockRegistry.this.expireAfter));
290 
291 			boolean result = Boolean.TRUE.equals(success);
292 
293 			if (result) {
294 				this.lockedAt = System.currentTimeMillis();
295 			}
296 			return result;
297 		}
298 
299 		@Override
300 		public void unlock() {
301 			if (!this.localLock.isHeldByCurrentThread()) {
302 				throw new IllegalStateException("You do not own lock at " + this.lockKey);
303 			}
304 			if (this.localLock.getHoldCount() > 1) {
305 				this.localLock.unlock();
306 				return;
307 			}
308 			try {
309 				if (!isAcquiredInThisProcess()) {
310 					throw new IllegalStateException("Lock was released in the store due to expiration. " +
311 							"The integrity of data protected by this lock may have been compromised.");
312 				}
313 
314 				if (Thread.currentThread().isInterrupted()) {
315 					RedisLockRegistry.this.executor.execute(this::removeLockKey);
316 				}
317 				else {
318 					removeLockKey();
319 				}
320 
321 				if (LOGGER.isDebugEnabled()) {
322 					LOGGER.debug("Released lock; " + this);
323 				}
324 			}
325 			catch (Exception e) {
326 				ReflectionUtils.rethrowRuntimeException(e);
327 			}
328 			finally {
329 				this.localLock.unlock();
330 			}
331 		}
332 
333 		private void removeLockKey() {
334 			if (this.unlinkAvailable) {
335 				try {
336 					RedisLockRegistry.this.redisTemplate.unlink(this.lockKey);
337 				}
338 				catch (Exception ex) {
339 					LOGGER.warn("The UNLINK command has failed (not supported on the Redis server?); " +
340 							"falling back to the regular DELETE command", ex);
341 					this.unlinkAvailable = false;
342 					RedisLockRegistry.this.redisTemplate.delete(this.lockKey);
343 				}
344 			}
345 			else {
346 				RedisLockRegistry.this.redisTemplate.delete(this.lockKey);
347 			}
348 		}
349 
350 		@Override
351 		public Condition newCondition() {
352 			throw new UnsupportedOperationException("Conditions are not supported");
353 		}
354 
355 		public boolean isAcquiredInThisProcess() {
356 			return RedisLockRegistry.this.clientId.equals(
357 					RedisLockRegistry.this.redisTemplate.boundValueOps(this.lockKey).get());
358 		}
359 
360 		@Override
361 		public String toString() {
362 			SimpleDateFormat dateFormat = new SimpleDateFormat("YYYY-MM-dd@HH:mm:ss.SSS");
363 			return "RedisLock [lockKey=" + this.lockKey
364 					+ ",lockedAt=" + dateFormat.format(new Date(this.lockedAt))
365 					+ ", clientId=" + RedisLockRegistry.this.clientId
366 					+ "]";
367 		}
368 
369 		@Override
370 		public int hashCode() {
371 			final int prime = 31;
372 			int result = 1;
373 			result = prime * result + getOuterType().hashCode();
374 			result = prime * result + ((this.lockKey == null) ? 0 : this.lockKey.hashCode());
375 			result = prime * result + (int) (this.lockedAt ^ (this.lockedAt >>> 32));
376 			result = prime * result + RedisLockRegistry.this.clientId.hashCode();
377 			return result;
378 		}
379 
380 		@Override
381 		public boolean equals(Object obj) {
382 			if (this == obj) {
383 				return true;
384 			}
385 			if (obj == null) {
386 				return false;
387 			}
388 			if (getClass() != obj.getClass()) {
389 				return false;
390 			}
391 			RedisLock other = (RedisLock) obj;
392 			if (!getOuterType().equals(other.getOuterType())) {
393 				return false;
394 			}
395 			if (!this.lockKey.equals(other.lockKey)) {
396 				return false;
397 			}
398 			return this.lockedAt == other.lockedAt;
399 		}
400 
401 		private RedisLockRegistry getOuterType() {
402 			return RedisLockRegistry.this;
403 		}
404 
405 	}

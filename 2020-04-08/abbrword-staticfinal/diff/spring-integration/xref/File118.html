17  package org.springframework.integration.transformer;
18  
19  import java.nio.charset.StandardCharsets;
20  import java.sql.Date;
21  import java.time.LocalDate;
22  import java.time.format.DateTimeFormatter;
23  import java.util.Calendar;
24  import java.util.LinkedHashMap;
25  import java.util.Map;
26  import java.util.regex.Matcher;
27  import java.util.regex.Pattern;
28  
29  import org.springframework.util.Assert;
30  import org.springframework.util.StringUtils;
31  
32  /**
33   * Transforms a packet in Syslog (RFC3164) format to a Map.
34   * If the packet cannot be decoded, the entire packet
35   * is returned as a String under the key {@code UNDECODED}. If the date field can be
36   * parsed, it will be returned as a {@link Date} object; otherwise it is returned as a String.
37   *
38   * @author Gary Russell
39   * @author Artem Bilan
40   * @author Karol Dowbecki
41   *
42   * @since 2.2
43   *
44   */
45  public class SyslogToMapTransformer extends AbstractPayloadTransformer<Object, Map<String, ?>> {
46  
47  	public static final String FACILITY = "FACILITY";
48  
49  	public static final String SEVERITY = "SEVERITY";
50  
51  	public static final String TIMESTAMP = "TIMESTAMP";
52  
53  	public static final String HOST = "HOST";
54  
55  	public static final String TAG = "TAG";
56  
57  	public static final String MESSAGE = "MESSAGE";
58  
59  	public static final String UNDECODED = "UNDECODED";
60  
61  	private final DateTimeFormatter dateTimeFormatter = DateTimeFormatter.ofPattern("MMM dd HH:mm:ss");
62  
63  	private final Pattern pattern = Pattern.compile("<([^>]+)>(.{15}) ([^ ]+) ([a-zA-Z0-9]{0,32})(.*)", Pattern.DOTALL);
64  
65  	@Override
66  	protected Map<String, ?> transformPayload(Object payload) {
67  		boolean isByteArray = payload instanceof byte[];
68  		boolean isString = payload instanceof String;
69  		Assert.isTrue(isByteArray || isString, "payload must be String or byte[]");
70  		if (isByteArray) {
71  			return this.transform((byte[]) payload);
72  		}
73  		else {
74  			return this.transform((String) payload);
75  		}
76  	}
77  
78  	private Map<String, ?> transform(byte[] payloadBytes) {
79  		return transform(new String(payloadBytes, StandardCharsets.UTF_8));
80  	}
81  
82  	private Map<String, ?> transform(String payload) {
83  		Map<String, Object> map = new LinkedHashMap<>();
84  		Matcher matcher = this.pattern.matcher(payload);
85  		if (matcher.matches()) {
86  			parseMatcherToMap(payload, matcher, map);
87  		}
88  		else {
89  			if (logger.isDebugEnabled()) {
90  				logger.debug("Could not decode: " + payload);
91  			}
92  			map.put(UNDECODED, payload);
93  		}
94  		return map;
95  	}
96  
97  	private void parseMatcherToMap(Object payload, Matcher matcher, Map<String, Object> map) {
98  		try {
99  			String facilityString = matcher.group(1); // NOSONAR
100 			int facility = Integer.parseInt(facilityString);
101 			int severity = facility & 0x7;
102 			facility = facility >> 3;
103 			map.put(FACILITY, facility);
104 			map.put(SEVERITY, severity);
105 			String timestamp = matcher.group(2); // NOSONAR
106 			parseTimestampToMap(timestamp, map);
107 			map.put(HOST, matcher.group(3)); // NOSONAR
108 			String tag = matcher.group(4); // NOSONAR
109 			if (StringUtils.hasLength(tag)) {
110 				map.put(TAG, tag);
111 			}
112 			map.put(MESSAGE, matcher.group(5)); // NOSONAR
113 		}
114 		catch (Exception e) {
115 			if (logger.isDebugEnabled()) {
116 				logger.debug("Could not decode: " + payload, e);
117 			}
118 			map.clear();
119 			map.put(UNDECODED, payload);
120 		}
121 	}
122 
123 	private void parseTimestampToMap(String timestamp, Map<String, Object> map) {
124 		try {
125 			LocalDate localDate = this.dateTimeFormatter.parse(timestamp, LocalDate::from);
126 			Calendar calendar = Calendar.getInstance();
127 			int year = calendar.get(Calendar.YEAR);
128 			int month = calendar.get(Calendar.MONTH);
129 			calendar.setTime(Date.valueOf(localDate));
130 			/*
131 			 * syslog date doesn't include a year so we
132 			 * need to insert the current year - adjusted
133 			 * if necessary if close to midnight on Dec 31.
134 			 */
135 			if (month == Calendar.DECEMBER && calendar.get(Calendar.MONTH) == Calendar.JANUARY) {
136 				calendar.set(Calendar.YEAR, year + 1);
137 			}
138 			else if (month == Calendar.JANUARY && calendar.get(Calendar.MONTH) == Calendar.FEBRUARY) {
139 				calendar.set(Calendar.YEAR, year - 1);
140 			}
141 			else {
142 				calendar.set(Calendar.YEAR, year);
143 			}
144 			map.put(TIMESTAMP, calendar.getTime());
145 		}
146 		catch (@SuppressWarnings("unused") Exception e) {
147 			/*
148 			 * If we can't parse the timestamp, return it as an
149 			 * unmodified String. (Postel's law).
150 			 */
151 			map.put(TIMESTAMP, timestamp);
152 		}
153 	}

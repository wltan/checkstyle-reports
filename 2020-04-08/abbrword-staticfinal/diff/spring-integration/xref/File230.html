35  /**
36   * Parser for the &lt;message-driven-channel-adapter&gt; element and the
37   * &lt;inbound-gateway&gt; element of the 'jms' namespace.
38   *
39   * @author Mark Fisher
40   * @author Michael Bannister
41   * @author Gary Russell
42   * @author Artem Bilan
43   */
44  public class JmsMessageDrivenEndpointParser extends AbstractSingleBeanDefinitionParser {
45  
46  	private static final String DEFAULT_REPLY_DESTINATION_ATTRIB = "default-reply-destination";
47  
48  	private static final String DEFAULT_REPLY_QUEUE_NAME_ATTRIB = "default-reply-queue-name";
49  
50  	private static final String DEFAULT_REPLY_TOPIC_NAME_ATTRIB = "default-reply-topic-name";
51  
52  	private static final String REPLY_TIME_TO_LIVE = "reply-time-to-live";
53  
54  	private static final String REPLY_PRIORITY = "reply-priority";
55  
56  	private static final String REPLY_DELIVERY_PERSISTENT = "reply-delivery-persistent";
57  
58  	private static final String EXPLICIT_QOS_ENABLED_FOR_REPLIES = "explicit-qos-enabled-for-replies";
59

61  	private static String[] containerAttributes = new String[] {
62  		JmsParserUtils.CONNECTION_FACTORY_PROPERTY,
63  		JmsParserUtils.DESTINATION_ATTRIBUTE,
64  		JmsParserUtils.DESTINATION_NAME_ATTRIBUTE,
65  		"destination-resolver", "transaction-manager",
66  		"concurrent-consumers", "max-concurrent-consumers",
67  		"acknowledge",
68  		"max-messages-per-task", "selector",
69  		"receive-timeout", "recovery-interval",
70  		"idle-consumer-limit", "idle-task-execution-limit",
71  		"cache-level", "subscription-durable",
72  		"subscription-shared", "subscription-name",
73  		"client-id", "task-executor"
74  	};

77  	private final boolean expectReply;

79  
80  	public JmsMessageDrivenEndpointParser(boolean expectReply) {
81  		this.expectReply = expectReply;
82  	}
83  
84  
85  	@Override
86  	protected String getBeanClassName(Element element) {
87  		return JmsMessageDrivenEndpoint.class.getName();
88  	}
89  
90  	@Override
91  	protected String resolveId(Element element, AbstractBeanDefinition definition, ParserContext parserContext)
92  			throws BeanDefinitionStoreException {
93  		String id = super.resolveId(element, definition, parserContext);
94  
95  		if (!this.expectReply && !element.hasAttribute("channel")) {
96  			// the created channel will get the 'id', so the adapter's bean name includes a suffix
97  			id = id + ".adapter";
98  		}
99  		if (!StringUtils.hasText(id)) {
100 			id = BeanDefinitionReaderUtils.generateBeanName(definition, parserContext.getRegistry());
101 		}
102 
103 		return id;
104 	}
105 
106 	@Override
107 	protected boolean shouldGenerateIdAsFallback() {
108 		return true;
109 	}
110 
111 	@Override
112 	protected void doParse(Element element, ParserContext parserContext, BeanDefinitionBuilder builder) {
113 		String containerBeanName = parseMessageListenerContainer(element, parserContext, builder.getRawBeanDefinition());
114 		String listenerBeanName = parseMessageListener(element, parserContext, builder.getRawBeanDefinition());
115 		builder.addConstructorArgReference(containerBeanName);
116 		builder.addConstructorArgReference(listenerBeanName);
117 		builder.addConstructorArgValue(hasExternalContainer(element));
118 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, IntegrationNamespaceUtils.AUTO_STARTUP);
119 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, IntegrationNamespaceUtils.PHASE);
120 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, IntegrationNamespaceUtils.ROLE);
121 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "acknowledge", "sessionAcknowledgeMode");
122 
123 	}
124 
125 	private String parseMessageListenerContainer(Element element, ParserContext parserContext,
126 			BeanDefinition adapterBeanDefinition) {
127 		String containerClass = element.getAttribute("container-class");
128 		if (hasExternalContainer(element)) {
129 			if (StringUtils.hasText(containerClass)) {
130 				parserContext.getReaderContext().error("Cannot have both 'container' and 'container-class'", element);
131 			}
132 			for (String containerAttribute : containerAttributes) {
133 				if (element.hasAttribute(containerAttribute)) {
134 					parserContext.getReaderContext().error("The '" + containerAttribute +
135 							"' attribute should not be provided when specifying a 'container' reference.", element);
136 				}
137 			}
138 			return element.getAttribute("container");
139 		}
140 		// otherwise, we build a DefaultMessageListenerContainer instance
141 		BeanDefinitionBuilder builder;
142 		if (StringUtils.hasText(containerClass)) {
143 			builder = BeanDefinitionBuilder.genericBeanDefinition(containerClass);
144 		}
145 		else {
146 			builder = BeanDefinitionBuilder.genericBeanDefinition(DefaultMessageListenerContainer.class);
147 		}
148 		String destinationAttribute = this.expectReply ? "request-destination" : "destination";
149 		String destinationNameAttribute = this.expectReply ? "request-destination-name" : "destination-name";
150 		String pubSubDomainAttribute = this.expectReply ? "request-pub-sub-domain" : "pub-sub-domain";
151 		String destination = element.getAttribute(destinationAttribute);
152 		String destinationName = element.getAttribute(destinationNameAttribute);
153 		boolean hasDestination = StringUtils.hasText(destination);
154 		boolean hasDestinationName = StringUtils.hasText(destinationName);
155 		if (hasDestination == hasDestinationName) {
156 			parserContext.getReaderContext().error(
157 					"Exactly one of '" + destinationAttribute +
158 					"' or '" + destinationNameAttribute + "' is required.", element);
159 		}
160 		builder.addPropertyReference(JmsParserUtils.CONNECTION_FACTORY_PROPERTY,
161 				JmsParserUtils.determineConnectionFactoryBeanName(element, parserContext));
162 		if (hasDestination) {
163 			builder.addPropertyReference("destination", destination);
164 		}
165 		else {
166 			builder.addPropertyValue("destinationName", destinationName);
167 			IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, pubSubDomainAttribute, "pubSubDomain");
168 		}
169 
170 		IntegrationNamespaceUtils.setReferenceIfAttributeDefined(builder, element, "destination-resolver");
171 		IntegrationNamespaceUtils.setReferenceIfAttributeDefined(builder, element, "transaction-manager");
172 		IntegrationNamespaceUtils.setReferenceIfAttributeDefined(builder, element, "task-executor");
173 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "selector", "messageSelector");
174 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "concurrent-consumers");
175 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "max-concurrent-consumers");
176 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "max-messages-per-task");
177 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "receive-timeout");
178 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "recovery-interval");
179 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "idle-consumer-limit");
180 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "idle-task-execution-limit");
181 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "cache-level");
182 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "subscription-durable");
183 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "subscription-shared");
184 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "subscription-name");
185 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "client-id");
186 		String beanName = adapterBeanNameRoot(element, parserContext, adapterBeanDefinition)
187 				+ ".container";
188 		parserContext.getRegistry().registerBeanDefinition(beanName, builder.getBeanDefinition());
189 		return beanName;
190 	}
191 
192 
193 	private boolean hasExternalContainer(Element element) {
194 		return element.hasAttribute("container");
195 	}
196 
197 	private String parseMessageListener(Element element, ParserContext parserContext,
198 			BeanDefinition adapterBeanDefinition) {
199 		BeanDefinitionBuilder builder = BeanDefinitionBuilder
200 				.genericBeanDefinition(ChannelPublishingJmsMessageListener.class);
201 		builder.addPropertyValue("expectReply", this.expectReply);
202 		if (this.expectReply) {
203 			IntegrationNamespaceUtils.setReferenceIfAttributeDefined(builder, element, "request-channel");
204 			IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "request-timeout");
205 			IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "reply-timeout");
206 			IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "extract-request-payload");
207 			IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "extract-reply-payload");
208 			IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "correlation-key");
209 			int defaults = 0;
210 			if (StringUtils.hasText(element.getAttribute(DEFAULT_REPLY_DESTINATION_ATTRIB))) {
211 				defaults++;
212 			}
213 			if (StringUtils.hasText(element.getAttribute(DEFAULT_REPLY_QUEUE_NAME_ATTRIB))) {
214 				defaults++;
215 			}
216 			if (StringUtils.hasText(element.getAttribute(DEFAULT_REPLY_TOPIC_NAME_ATTRIB))) {
217 				defaults++;
218 			}
219 			if (defaults > 1) {
220 				parserContext.getReaderContext().error("At most one of '" + DEFAULT_REPLY_DESTINATION_ATTRIB
221 						+ "', '" + DEFAULT_REPLY_QUEUE_NAME_ATTRIB + "', or '" + DEFAULT_REPLY_TOPIC_NAME_ATTRIB
222 						+ "' may be provided.", element);
223 			}
224 			IntegrationNamespaceUtils.setReferenceIfAttributeDefined(builder, element, DEFAULT_REPLY_DESTINATION_ATTRIB);
225 			IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, DEFAULT_REPLY_QUEUE_NAME_ATTRIB);
226 			IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, DEFAULT_REPLY_TOPIC_NAME_ATTRIB);
227 			IntegrationNamespaceUtils.setReferenceIfAttributeDefined(builder, element, "destination-resolver");
228 			IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, REPLY_TIME_TO_LIVE);
229 			IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, REPLY_PRIORITY);
230 			IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, REPLY_DELIVERY_PERSISTENT);
231 			IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, EXPLICIT_QOS_ENABLED_FOR_REPLIES);
232 			IntegrationNamespaceUtils.setReferenceIfAttributeDefined(builder, element, "reply-channel");
233 		}
234 		else {
235 			String channelName = element.getAttribute("channel");
236 			if (!StringUtils.hasText(channelName)) {
237 				channelName = IntegrationNamespaceUtils.createDirectChannel(element, parserContext);
238 			}
239 			builder.addPropertyReference("requestChannel", channelName);
240 			IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "send-timeout", "requestTimeout");
241 			IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "extract-payload", "extractRequestPayload");
242 		}
243 		IntegrationNamespaceUtils.setReferenceIfAttributeDefined(builder, element, "error-channel");
244 		IntegrationNamespaceUtils.setReferenceIfAttributeDefined(builder, element, "message-converter");
245 		IntegrationNamespaceUtils.setReferenceIfAttributeDefined(builder, element, "header-mapper");
246 		String alias = adapterBeanNameRoot(element, parserContext, adapterBeanDefinition)
247 				+ ".listener";
248 		BeanDefinition beanDefinition = builder.getBeanDefinition();
249 		String beanName = BeanDefinitionReaderUtils.generateBeanName(beanDefinition, parserContext.getRegistry());
250 		BeanComponentDefinition component = new BeanComponentDefinition(beanDefinition, beanName, new String[] { alias });
251 		parserContext.registerBeanComponent(component);
252 		return beanName;
253 	}
254 
255 	private String adapterBeanNameRoot(Element element, ParserContext parserContext,
256 			BeanDefinition adapterBeanDefinition) {
257 		String beanName = element.getAttribute(ID_ATTRIBUTE);
258 		if (!StringUtils.hasText(beanName)) {
259 			beanName = BeanDefinitionReaderUtils.generateBeanName(adapterBeanDefinition, parserContext.getRegistry());
260 		}
261 		return beanName;
262 	}

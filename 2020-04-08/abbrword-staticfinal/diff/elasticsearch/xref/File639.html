44  /**
45   * This {@link PostingsFormat} is basically a T-Sink for a default postings
46   * format that is used to store postings on disk fitting the lucene APIs and
47   * builds a suggest FST as an auxiliary data structure next to the actual
48   * postings format. It uses the delegate postings format for simplicity to
49   * handle all the merge operations. The auxiliary suggest FST data structure is
50   * only loaded if a FieldsProducer is requested for reading, for merging it uses
51   * the low memory delegate postings format.
52   */
53  public class Completion090PostingsFormat extends PostingsFormat {
54  
55      public static final String CODEC_NAME = "completion090";
56      public static final int SUGGEST_CODEC_VERSION = 1;
57      public static final int SUGGEST_VERSION_CURRENT = SUGGEST_CODEC_VERSION;
58      public static final String EXTENSION = "cmp";
59  
60      private final static ESLogger logger = Loggers.getLogger(Completion090PostingsFormat.class);
61      private PostingsFormat delegatePostingsFormat;
62      private final static Map<String, CompletionLookupProvider> providers;
63      private CompletionLookupProvider writeProvider;
64  
65  
66      static {
67          final CompletionLookupProvider provider = new AnalyzingCompletionLookupProvider(true, false, true, false);
68          final Builder<String, CompletionLookupProvider> builder = ImmutableMap.builder();
69          providers = builder.put(provider.getName(), provider).build();
70      }
71  
72      public Completion090PostingsFormat(PostingsFormat delegatePostingsFormat, CompletionLookupProvider provider) {
73          super(CODEC_NAME);
74          this.delegatePostingsFormat = delegatePostingsFormat;
75          this.writeProvider = provider;
76          assert delegatePostingsFormat != null && writeProvider != null;
77      }
78  
79      /*
80       * Used only by core Lucene at read-time via Service Provider instantiation
81       * do not use at Write-time in application code.
82       */
83      public Completion090PostingsFormat() {
84          super(CODEC_NAME);
85      }
86  
87      @Override
88      public CompletionFieldsConsumer fieldsConsumer(SegmentWriteState state) throws IOException {
89          if (delegatePostingsFormat == null) {
90              throw new UnsupportedOperationException("Error - " + getClass().getName()
91                      + " has been constructed without a choice of PostingsFormat");
92          }
93          assert writeProvider != null;
94          return new CompletionFieldsConsumer(state);
95      }
96  
97      @Override
98      public CompletionFieldsProducer fieldsProducer(SegmentReadState state) throws IOException {
99          return new CompletionFieldsProducer(state);
100     }
101 
102     private class CompletionFieldsConsumer extends FieldsConsumer {
103 
104         private FieldsConsumer delegatesFieldsConsumer;
105         private FieldsConsumer suggestFieldsConsumer;
106 
107         public CompletionFieldsConsumer(SegmentWriteState state) throws IOException {
108             this.delegatesFieldsConsumer = delegatePostingsFormat.fieldsConsumer(state);
109             String suggestFSTFile = IndexFileNames.segmentFileName(state.segmentInfo.name, state.segmentSuffix, EXTENSION);
110             IndexOutput output = null;
111             boolean success = false;
112             try {
113                 output = state.directory.createOutput(suggestFSTFile, state.context);
114                 CodecUtil.writeHeader(output, CODEC_NAME, SUGGEST_VERSION_CURRENT);
115                 /*
116                  * we write the delegate postings format name so we can load it
117                  * without getting an instance in the ctor
118                  */
119                 output.writeString(delegatePostingsFormat.getName());
120                 output.writeString(writeProvider.getName());
121                 this.suggestFieldsConsumer = writeProvider.consumer(output);
122                 success = true;
123             } finally {
124                 if (!success) {
125                     IOUtils.closeWhileHandlingException(output);
126                 }
127             }
128         }
129 
130         @Override
131         public TermsConsumer addField(final FieldInfo field) throws IOException {
132             final TermsConsumer delegateConsumer = delegatesFieldsConsumer.addField(field);
133             final TermsConsumer suggestTermConsumer = suggestFieldsConsumer.addField(field);
134             final GroupedPostingsConsumer groupedPostingsConsumer = new GroupedPostingsConsumer(delegateConsumer, suggestTermConsumer);
135 
136             return new TermsConsumer() {
137                 @Override
138                 public PostingsConsumer startTerm(BytesRef text) throws IOException {
139                     groupedPostingsConsumer.startTerm(text);
140                     return groupedPostingsConsumer;
141                 }
142 
143                 @Override
144                 public Comparator<BytesRef> getComparator() throws IOException {
145                     return delegateConsumer.getComparator();
146                 }
147 
148                 @Override
149                 public void finishTerm(BytesRef text, TermStats stats) throws IOException {
150                     suggestTermConsumer.finishTerm(text, stats);
151                     delegateConsumer.finishTerm(text, stats);
152                 }
153 
154                 @Override
155                 public void finish(long sumTotalTermFreq, long sumDocFreq, int docCount) throws IOException {
156                     suggestTermConsumer.finish(sumTotalTermFreq, sumDocFreq, docCount);
157                     delegateConsumer.finish(sumTotalTermFreq, sumDocFreq, docCount);
158                 }
159             };
160         }
161 
162         @Override
163         public void close() throws IOException {
164             IOUtils.close(delegatesFieldsConsumer, suggestFieldsConsumer);
165         }
166     }
167 
168     private class GroupedPostingsConsumer extends PostingsConsumer {
169 
170         private TermsConsumer[] termsConsumers;
171         private PostingsConsumer[] postingsConsumers;
172 
173         public GroupedPostingsConsumer(TermsConsumer... termsConsumersArgs) {
174             termsConsumers = termsConsumersArgs;
175             postingsConsumers = new PostingsConsumer[termsConsumersArgs.length];
176         }
177 
178         @Override
179         public void startDoc(int docID, int freq) throws IOException {
180             for (PostingsConsumer postingsConsumer : postingsConsumers) {
181                 postingsConsumer.startDoc(docID, freq);
182             }
183         }
184 
185         @Override
186         public void addPosition(int position, BytesRef payload, int startOffset, int endOffset) throws IOException {
187             for (PostingsConsumer postingsConsumer : postingsConsumers) {
188                 postingsConsumer.addPosition(position, payload, startOffset, endOffset);
189             }
190         }
191 
192         @Override
193         public void finishDoc() throws IOException {
194             for (PostingsConsumer postingsConsumer : postingsConsumers) {
195                 postingsConsumer.finishDoc();
196             }
197         }
198 
199         public void startTerm(BytesRef text) throws IOException {
200             for (int i = 0; i < termsConsumers.length; i++) {
201                 postingsConsumers[i] = termsConsumers[i].startTerm(text);
202             }
203         }
204     }

206     private static class CompletionFieldsProducer extends FieldsProducer {
207 
208         private final FieldsProducer delegateProducer;
209         private final LookupFactory lookupFactory;
210         private final int version;
211 
212         public CompletionFieldsProducer(SegmentReadState state) throws IOException {
213             String suggestFSTFile = IndexFileNames.segmentFileName(state.segmentInfo.name, state.segmentSuffix, EXTENSION);
214             IndexInput input = state.directory.openInput(suggestFSTFile, state.context);
215             version = CodecUtil.checkHeader(input, CODEC_NAME, SUGGEST_CODEC_VERSION, SUGGEST_VERSION_CURRENT);
216             FieldsProducer delegateProducer = null;
217             boolean success = false;
218             try {
219                 PostingsFormat delegatePostingsFormat = PostingsFormat.forName(input.readString());
220                 String providerName = input.readString();
221                 CompletionLookupProvider completionLookupProvider = providers.get(providerName);
222                 if (completionLookupProvider == null) {
223                     throw new ElasticsearchIllegalStateException("no provider with name [" + providerName + "] registered");
224                 }
225                 // TODO: we could clone the ReadState and make it always forward IOContext.MERGE to prevent unecessary heap usage? 
226                 delegateProducer = delegatePostingsFormat.fieldsProducer(state);
227                 /*
228                  * If we are merging we don't load the FSTs at all such that we
229                  * don't consume so much memory during merge
230                  */
231                 if (state.context.context != Context.MERGE) {
232                     // TODO: maybe we can do this in a fully lazy fashion based on some configuration
233                     // eventually we should have some kind of curciut breaker that prevents us from going OOM here
234                     // with some configuration
235                     this.lookupFactory = completionLookupProvider.load(input);
236                 } else {
237                     this.lookupFactory = null;
238                 }
239                 this.delegateProducer = delegateProducer;
240                 success = true;
241             } finally {
242                 if (!success) {
243                     IOUtils.closeWhileHandlingException(delegateProducer, input);
244                 } else {
245                     IOUtils.close(input);
246                 }
247             }
248         }
249 
250         @Override
251         public void close() throws IOException {
252             IOUtils.close(delegateProducer);
253         }
254 
255         @Override
256         public Iterator<String> iterator() {
257             return delegateProducer.iterator();
258         }
259 
260         @Override
261         public Terms terms(String field) throws IOException {
262             final Terms terms = delegateProducer.terms(field);
263             if (terms == null || lookupFactory == null) {
264                 return terms;
265             }
266             return new CompletionTerms(terms, lookupFactory);
267         }
268 
269         @Override
270         public int size() {
271             return delegateProducer.size();
272         }
273 
274         @Override
275         public long ramBytesUsed() {
276             return (lookupFactory == null ? 0 : lookupFactory.ramBytesUsed()) + delegateProducer.ramBytesUsed();
277         }
278 
279         @Override
280         public void checkIntegrity() throws IOException {
281             delegateProducer.checkIntegrity();
282         }
283     }

302     public static abstract class CompletionLookupProvider implements PayloadProcessor, ToFiniteStrings {
303 
304         public static final char UNIT_SEPARATOR = '\u001f';
305 
306         public abstract FieldsConsumer consumer(IndexOutput output) throws IOException;
307 
308         public abstract String getName();
309 
310         public abstract LookupFactory load(IndexInput input) throws IOException;
311 
312         @Override
313         public BytesRef buildPayload(BytesRef surfaceForm, long weight, BytesRef payload) throws IOException {
314             if (weight < -1 || weight > Integer.MAX_VALUE) {
315                 throw new IllegalArgumentException("weight must be >= -1 && <= Integer.MAX_VALUE");
316             }
317             for (int i = 0; i < surfaceForm.length; i++) {
318                 if (surfaceForm.bytes[i] == UNIT_SEPARATOR) {
319                     throw new IllegalArgumentException(
320                             "surface form cannot contain unit separator character U+001F; this character is reserved");
321                 }
322             }
323             ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
324             OutputStreamDataOutput output = new OutputStreamDataOutput(byteArrayOutputStream);
325             output.writeVLong(weight + 1);
326             output.writeVInt(surfaceForm.length);
327             output.writeBytes(surfaceForm.bytes, surfaceForm.offset, surfaceForm.length);
328             output.writeVInt(payload.length);
329             output.writeBytes(payload.bytes, 0, payload.length);
330 
331             output.close();
332             return new BytesRef(byteArrayOutputStream.toByteArray());
333         }
334 
335         @Override
336         public void parsePayload(BytesRef payload, SuggestPayload ref) throws IOException {
337             ByteArrayInputStream byteArrayInputStream = new ByteArrayInputStream(payload.bytes, payload.offset, payload.length);
338             InputStreamDataInput input = new InputStreamDataInput(byteArrayInputStream);
339             ref.weight = input.readVLong() - 1;
340             int len = input.readVInt();
341             ref.surfaceForm.grow(len);
342             ref.surfaceForm.setLength(len);
343             input.readBytes(ref.surfaceForm.bytes(), 0, ref.surfaceForm.length());
344             len = input.readVInt();
345             ref.payload.grow(len);
346             ref.payload.setLength(len);
347             input.readBytes(ref.payload.bytes(), 0, ref.payload.length());
348             input.close();
349         }
350     }

352     public CompletionStats completionStats(IndexReader indexReader, String ... fields) {
353         CompletionStats completionStats = new CompletionStats();
354         for (AtomicReaderContext atomicReaderContext : indexReader.leaves()) {
355             AtomicReader atomicReader = atomicReaderContext.reader();
356             try {
357                 for (String fieldName : atomicReader.fields()) {
358                     Terms terms = atomicReader.fields().terms(fieldName);
359                     if (terms instanceof CompletionTerms) {
360                         CompletionTerms completionTerms = (CompletionTerms) terms;
361                         completionStats.add(completionTerms.stats(fields));
362                     }
363                 }
364             } catch (IOException e) {
365                 logger.error("Could not get completion stats: {}", e, e.getMessage());
366             }
367         }
368 
369         return completionStats;
370     }

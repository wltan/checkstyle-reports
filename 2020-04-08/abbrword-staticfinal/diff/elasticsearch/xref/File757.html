45  /**
46   * A test cluster implementation that holds a fixed set of external nodes as well as a InternalTestCluster
47   * which is used to run mixed version clusters in tests like backwards compatibility tests.
48   * Note: this is an experimental API
49   */
50  public class CompositeTestCluster extends TestCluster {
51      private final InternalTestCluster cluster;
52      private final ExternalNode baseExternalNode;
53      private final List<ExternalNode> externalNodes;
54      private final ExternalClient client = new ExternalClient();
55      private static final String NODE_PREFIX = "external_";
56  
57      public CompositeTestCluster(InternalTestCluster cluster, int numExternalNodes, ExternalNode externalNode) throws IOException {
58          super(cluster.seed());
59          this.cluster = cluster;
60          this.externalNodes = new ArrayList<>();
61          this.baseExternalNode = externalNode;
62  
63          // initialize external nodes with non-started copies
64          for (int i = 0; i < numExternalNodes; i++) {
65              externalNodes.add(baseExternalNode);
66          }
67      }
68  
69      @Override
70      public synchronized void afterTest() throws IOException {
71          cluster.afterTest();
72      }
73  
74      @Override
75      public synchronized void beforeTest(Random random, double transportClientRatio) throws IOException {
76          super.beforeTest(random, transportClientRatio);
77          cluster.beforeTest(random, transportClientRatio);
78          Settings defaultSettings = cluster.getDefaultSettings();
79  
80          if (externalNodes.size() == 0) {
81              // bail out, nothing do here (and no need to initialize a client node)
82              return;
83          }
84  
85          final Client client = cluster.size() > 0 ? cluster.client() : cluster.clientNodeClient();
86          for (int i = 0; i < externalNodes.size(); i++) {
87              if (!externalNodes.get(i).running()) {
88                  try {
89                      externalNodes.set(i, baseExternalNode.start(client, defaultSettings, NODE_PREFIX + i, cluster.getClusterName(), i));
90                  } catch (InterruptedException e) {
91                      Thread.interrupted();
92                      return;
93                  }
94              }
95              externalNodes.get(i).reset(random.nextLong());
96          }
97          if (size() > 0) {
98              client().admin().cluster().prepareHealth().setWaitForNodes(">=" + Integer.toString(this.size())).get();
99          }
100     }
101 
102     private Collection<ExternalNode> runningNodes() {
103         return Collections2.filter(externalNodes, new Predicate<ExternalNode>() {
104             @Override
105             public boolean apply(ExternalNode input) {
106                 return input.running();
107             }
108         });
109     }
110 
111     /**
112      * Upgrades one external running node to a node from the version running the tests. Commonly this is used
113      * to move from a node with version N-1 to a node running version N. This works seamless since they will
114      * share the same data directory. This method will return <tt>true</tt> iff a node got upgraded otherwise if no
115      * external node is running it returns <tt>false</tt>
116      */
117     public synchronized boolean upgradeOneNode() throws InterruptedException, IOException {
118         return upgradeOneNode(ImmutableSettings.EMPTY);
119     }
120 
121     /**
122      * Upgrades all external running nodes to a node from the version running the tests.
123      * All nodes are shut down before the first upgrade happens.
124      *
125      * @return <code>true</code> iff at least one node as upgraded.
126      */
127     public synchronized boolean upgradeAllNodes() throws InterruptedException, IOException {
128         return upgradeAllNodes(ImmutableSettings.EMPTY);
129     }
130 
131 
132     /**
133      * Upgrades all external running nodes to a node from the version running the tests.
134      * All nodes are shut down before the first upgrade happens.
135      *
136      * @param nodeSettings settings for the upgrade nodes
137      * @return <code>true</code> iff at least one node as upgraded.
138      */
139     public synchronized boolean upgradeAllNodes(Settings nodeSettings) throws InterruptedException, IOException {
140         boolean upgradedOneNode = false;
141         while (upgradeOneNode(nodeSettings)) {
142             upgradedOneNode = true;
143         }
144         return upgradedOneNode;
145     }
146 
147     /**
148      * Upgrades one external running node to a node from the version running the tests. Commonly this is used
149      * to move from a node with version N-1 to a node running version N. This works seamless since they will
150      * share the same data directory. This method will return <tt>true</tt> iff a node got upgraded otherwise if no
151      * external node is running it returns <tt>false</tt>
152      */
153     public synchronized boolean upgradeOneNode(Settings nodeSettings) throws InterruptedException, IOException {
154         Collection<ExternalNode> runningNodes = runningNodes();
155         if (!runningNodes.isEmpty()) {
156             final Client existingClient = cluster.client();
157             ExternalNode externalNode = RandomPicks.randomFrom(random, runningNodes);
158             String externalNodeName = externalNode.getName();
159             logger.info("upgrading [{}]", externalNodeName);
160             externalNode.stop();
161             String s = cluster.startNode(nodeSettings);
162             ExternalNode.waitForNode(existingClient, s);
163             logger.info("done upgrading [{}], new node name: [{}]", externalNodeName, s);
164             assertNoTimeout(existingClient.admin().cluster().prepareHealth().setWaitForNodes(Integer.toString(size())).get());
165             return true;
166         }
167         return false;
168     }

172      * Returns the a simple pattern that matches all "new" nodes in the cluster.

174     public String newNodePattern() {
175         return cluster.nodePrefix() + "*";
176     }

178     /**
179      * Returns the a simple pattern that matches all "old" / "backwardss" nodes in the cluster.
180      */
181     public String backwardsNodePattern() {
182         return NODE_PREFIX + "*";
183     }
184 
185     /**
186      * Allows allocation of shards of the given indices on all nodes in the cluster.
187      */
188     public void allowOnAllNodes(String... index) {
189         Settings build = ImmutableSettings.builder().put("index.routing.allocation.exclude._name", "").build();
190         client().admin().indices().prepareUpdateSettings(index).setSettings(build).execute().actionGet();
191     }
192 
193     /**
194      * Allows allocation of shards of the given indices only on "new" nodes in the cluster.
195      * Note: if a shard is allocated on an "old" node and can't be allocated on a "new" node it will only be removed it can
196      * be allocated on some other "new" node.
197      */
198     public void allowOnlyNewNodes(String... index) {
199         Settings build = ImmutableSettings.builder().put("index.routing.allocation.exclude._name", backwardsNodePattern()).build();
200         client().admin().indices().prepareUpdateSettings(index).setSettings(build).execute().actionGet();
201     }

210     /**
211      * Starts a new external node with an old version
212      */
213     public synchronized void startNewExternalNode() throws IOException, InterruptedException {
214         int ordinal = externalNodes.size() + 1;
215         externalNodes.add(baseExternalNode.start(client, cluster.getDefaultSettings(), NODE_PREFIX + ordinal, cluster.getClusterName(), ordinal));
216     }

253     @Override
254     public void ensureEstimatedStats() {
255         if (size() > 0) {
256             NodesStatsResponse nodeStats = client().admin().cluster().prepareNodesStats()
257                     .clear().setBreaker(true).execute().actionGet();
258             for (NodeStats stats : nodeStats.getNodes()) {
259                 assertThat("Fielddata breaker not reset to 0 on node: " + stats.getNode(),
260                         stats.getBreaker().getStats(CircuitBreaker.Name.FIELDDATA).getEstimated(), equalTo(0L));
261             }
262             // CompositeTestCluster does not check the request breaker,
263             // because checking it requires a network request, which in
264             // turn increments the breaker, making it non-0
265         }
266     }

312     private synchronized Client internalClient() {
313         Collection<ExternalNode> externalNodes = runningNodes();
314         return random.nextBoolean() && !externalNodes.isEmpty() ? RandomPicks.randomFrom(random, externalNodes).getClient() : cluster.client();
315     }
316 
317     private final class ExternalClient extends FilterClient {
318 
319         public ExternalClient() {
320             super(null);
321         }
322 
323         @Override
324         protected Client in() {
325             return internalClient();
326         }
327 
328         @Override
329         public ClusterAdminClient cluster() {
330             return new ClusterAdmin(null) {
331 
332                 @Override
333                 protected ClusterAdminClient in() {
334                     return internalClient().admin().cluster();
335                 }
336             };
337         }
338 
339         @Override
340         public IndicesAdminClient indices() {
341             return new IndicesAdmin(null) {
342 
343                 @Override
344                 protected IndicesAdminClient in() {
345                     return internalClient().admin().indices();
346                 }
347             };
348         }
349 
350         @Override
351         public void close() {
352             // never close this client
353         }
354     }

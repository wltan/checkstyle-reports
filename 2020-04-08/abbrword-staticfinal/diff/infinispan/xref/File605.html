1   package org.infinispan.scripting.impl;
2   
3   import java.lang.invoke.MethodHandles;
4   import java.util.ArrayList;
5   import java.util.List;
6   import java.util.concurrent.Future;
7   
8   import org.infinispan.Cache;
9   import org.infinispan.commons.util.concurrent.Futures;
10  import org.infinispan.commons.util.concurrent.NotifyingFuture;
11  import org.infinispan.distexec.DefaultExecutorService;
12  import org.infinispan.scripting.logging.Log;
13  import org.infinispan.util.logging.LogFactory;
14  
15  /**
16   * DistributedRunner.
17   *
18   * @author Tristan Tarrant
19   * @since 7.2
20   */
21  public class DistributedRunner implements ScriptRunner {
22     private static final Log log = LogFactory.getLog(MethodHandles.lookup().lookupClass(), Log.class);
23     public static final DistributedRunner INSTANCE = new DistributedRunner();
24  
25     private DistributedRunner() {
26     }
27  
28     @Override
29     public <T> NotifyingFuture<T> runScript(ScriptingManagerImpl scriptManager, ScriptMetadata metadata, CacheScriptBindings binding) {
30        Cache<?, ?> masterCacheNode = (Cache<?, ?>) binding.get(SystemBindings.CACHE.toString());
31        if (masterCacheNode == null) {
32           throw log.distributedTaskNeedCacheInBinding(metadata.name());
33        }
34        DefaultExecutorService des = new DefaultExecutorService(masterCacheNode);
35        try {
36           List<Future<T>> tasks = des.submitEverywhere(new DistributedScript<T>(metadata));
37           List<NotifyingFuture<T>> notifyingTasks = new ArrayList<>(tasks.size());
38           for (Future<T> task : tasks) {
39                 notifyingTasks.add((NotifyingFuture<T>) task);
40           }
41           NotifyingFuture<List<T>> combinedFutures = Futures.combine(notifyingTasks);
42  
43           return (NotifyingFuture<T>) combinedFutures;
44        } finally {
45           des.shutdown();
46        }
47     }
48  }

53     private void generateKeyList(int numKeys) {
54        // without this we keep getting OutOfMemoryErrors
55        keys = null;
56        keys = new ArrayList<String>(numKeys * LOOP_FACTOR);
57        for (int i = 0; i < numKeys * LOOP_FACTOR; i++) {
58           keys.add("key" + nextIntGaussian(numKeys));
59        }
60     }
61  
62     private int nextIntGaussian(int numKeys) {
63        double gaussian = RANDOM.nextGaussian();
64        if (gaussian < -3 || gaussian > 3)
65           return nextIntGaussian(numKeys);
66  
67        return (int) Math.abs((gaussian + 3) * numKeys / 6);
68     }

70     private Map<String, Integer> synchronizedLinkedHashMap(final int capacity, float loadFactor) {
71        return Collections.synchronizedMap(new LinkedHashMap<String, Integer>(capacity, loadFactor, true) {
72           @Override
73           protected boolean removeEldestEntry(Entry<String, Integer> eldest) {
74              return size() > capacity;
75           }
76        });
77     }

79     private Cache<String, Integer> configureAndBuildCache(int capacity) {
80        ConfigurationBuilder config = new ConfigurationBuilder();
81        config
82           .eviction().maxEntries(capacity).strategy(EvictionStrategy.LRU)
83           .expiration().wakeUpInterval(5000L).maxIdle(120000L);
84        String cacheName = "cache" + capacity;
85        cacheManager.defineConfiguration(cacheName, config.build());
86        return cacheManager.getCache(cacheName);
87     }

113    private Map<String, Map<String, Integer>> createMaps(int capacity, int numKeys, int concurrency) {
114       Map<String, Map<String, Integer>> maps = new TreeMap<String, Map<String, Integer>>();
115       maps.put("BCHMv8:LRU", new BoundedEquivalentConcurrentHashMapV8(
116             (long) capacity, Eviction.LRU,
117             BoundedEquivalentConcurrentHashMapV8.getNullEvictionListener(),
118             AnyEquivalence.STRING, AnyEquivalence.INT));
119       maps.put("BCHMv8:LIRS", new BoundedEquivalentConcurrentHashMapV8(
120             (long) capacity, Eviction.LIRS,
121             BoundedEquivalentConcurrentHashMapV8.getNullEvictionListener(),
122             AnyEquivalence.STRING, AnyEquivalence.INT));
123       maps.put("BCHM:LRU", new BoundedConcurrentHashMap<String, Integer>(
124             capacity, concurrency, BoundedConcurrentHashMap.Eviction.LRU,
125             AnyEquivalence.STRING, AnyEquivalence.INT));
126       maps.put("BCHM:LIRS", new BoundedConcurrentHashMap<String, Integer>(
127             capacity, concurrency, BoundedConcurrentHashMap.Eviction.LIRS,
128             AnyEquivalence.STRING, AnyEquivalence.INT));
129       // CHM doesn't have eviction, so we size it to the capacity to allow for dynamic resize
130       maps.put("CHM", new ConcurrentHashMap<String, Integer>(capacity, MAP_LOAD_FACTOR, concurrency));
131       maps.put("CHMv8", new ConcurrentHashMapV8<String, Integer>(capacity, MAP_LOAD_FACTOR, concurrency));
132       maps.put("ECHMv8", new EquivalentConcurrentHashMapV8<String, Integer>(capacity, 
133             MAP_LOAD_FACTOR, concurrency, AnyEquivalence.STRING, AnyEquivalence.INT));
134       maps.put("SLHM", synchronizedLinkedHashMap(capacity, MAP_LOAD_FACTOR));
135       maps.put("CACHE", configureAndBuildCache(capacity));
136       return maps;
137    }

171    private TotalStats runMapTestReadWriteRemove(final Map<String, Integer> map, int numReaders, int numWriters,
172                                                          int numRemovers, final long runningTimeout) throws Exception {
173       latch = new CountDownLatch(1);
174       final TotalStats perf = new TotalStats();
175       List<Thread> threads = new LinkedList<Thread>();
176 
177       for (int i = 0; i < numReaders; i++) {
178          Thread reader = new WorkerThread(runningTimeout, perf, readOperation(map));
179          threads.add(reader);
180       }
181 
182       for (int i = 0; i < numWriters; i++) {
183          Thread writer = new WorkerThread(runningTimeout, perf, writeOperation(map));
184          threads.add(writer);
185       }
186 
187       for (int i = 0; i < numRemovers; i++) {
188          Thread remover = new WorkerThread(runningTimeout, perf, removeOperation(map));
189          threads.add(remover);
190       }
191 
192       for (Thread t : threads)
193          t.start();
194       latch.countDown();
195 
196       for (Thread t : threads) {
197          t.join();
198       }
199 
200       return perf;
201    }

307    private double computeStdDev(Map<String, Integer> map, int numKeys) {
308       // The keys closest to the mean are suposed to be accessed more often
309       // So we score each map by the standard deviation of the keys in the map
310       // at the end of the test
311       double variance = 0;
312       for (String key : map.keySet()) {
313          double value = Integer.parseInt(key.substring(3));
314          variance += (value - numKeys / 2) * (value - numKeys / 2);
315       }
316       return sqrt(variance / map.size());
317    }

354    private Operation<String, Integer> readWriteOperation(final Map<String, Integer> map, final int readToWriteRatio) {
355       return new Operation<String, Integer>(map, "READ/WRITE:" + readToWriteRatio + "/1") {
356          @Override
357          public boolean call(String key, long run) {
358             if (run % (readToWriteRatio + 1) == 0) {
359                return map.put(key, (int)run) != null;
360             } else {
361                return map.get(key) != null;
362             }
363          }
364       };
365    }

367    private Operation<String, Integer> writeOnMissOperation(final Map<String, Integer> map) {
368       return new Operation<String, Integer>(map, "PUTMISSING") {
369          @Override
370          public boolean call(String key, long run) {
371             boolean hit = map.get(key) != null;
372             if (!hit) {
373                map.put(key, (int)run);
374             }
375             return hit;
376          }
377       };
378    }

385    private class WorkerThread extends Thread {
386       private final long runningTimeout;
387       private final TotalStats perf;
388       private Operation<String, Integer> op;
389 
390       public WorkerThread(long runningTimeout, TotalStats perf, Operation<String, Integer> op) {
391          this.runningTimeout = runningTimeout;
392          this.perf = perf;
393          this.op = op;
394       }
395 
396       @Override
397       public void run() {
398          long startMilis = System.currentTimeMillis();
399          long endMillis = startMilis + runningTimeout;
400          waitForStart();
401          int keyIndex = RANDOM.nextInt(keys.size());
402          long runs = 0;
403          long missCount = 0;
404          while ((runs & 0x3FFF) != 0 || System.currentTimeMillis() < endMillis) {
405             boolean hit = op.call(keys.get(keyIndex), runs);
406             if (!hit) missCount++;
407             keyIndex++;
408             runs++;
409             if (keyIndex >= keys.size()) {
410                keyIndex = 0;
411             }
412          }
413          perf.addStats(op.getName(), runs, System.currentTimeMillis() - startMilis, missCount);
414       }
415    }

417    private static abstract class Operation<K, V> {
418       protected final Map<K, V> map;
419       protected final String name;
420 
421       public Operation(Map<K, V> map, String name) {
422          this.map = map;
423          this.name = name;
424       }
425 
426       /**
427        * @return Return true for a hit, false for a miss.
428        */
429       public abstract boolean call(K key, long run);
430 
431       public String getName() {
432          return name;
433       }
434    }

436    private static class TotalStats {
437       private ConcurrentHashMap<String, OpStats> statsMap = new ConcurrentHashMap<String, OpStats>();
438 
439       public void addStats(String opName, long opCount, long runningTime, long missCount) {
440          OpStats s = new OpStats(opName, opCount, runningTime, missCount);
441          OpStats old = statsMap.putIfAbsent(opName, s);
442          boolean replaced = old == null;
443          while (!replaced) {
444             old = statsMap.get(opName);
445             s = new OpStats(old, opCount, runningTime, missCount);
446             replaced = statsMap.replace(opName, old, s);
447          }
448       }
449 
450       public double getOpsPerSec(String opName) {
451          OpStats s = statsMap.get(opName);
452          if (s == null) return 0;
453          return s.opCount * 1000. / s.runningTime * s.threadCount;
454       }
455 
456       public double getTotalOpsPerSec() {
457          long totalOpCount = 0;
458          long totalRunningTime = 0;
459          long totalThreadCount = 0;
460          for (Map.Entry<String, OpStats> e : statsMap.entrySet()) {
461             OpStats s = e.getValue();
462             totalOpCount += s.opCount;
463             totalRunningTime += s.runningTime;
464             totalThreadCount += s.threadCount;
465          }
466          return totalOpCount * 1000. / totalRunningTime * totalThreadCount;
467       }
468 
469       public double getHitRatio(String opName) {
470          OpStats s = statsMap.get(opName);
471          if (s == null) return 0;
472          return 1 - 1. * s.missCount / s.opCount;
473       }
474 
475       public double getTotalHitRatio() {
476          long totalOpCount = 0;
477          long totalMissCount = 0;
478          for (Map.Entry<String, OpStats> e : statsMap.entrySet()) {
479             OpStats s = e.getValue();
480             totalOpCount += s.opCount;
481             totalMissCount += s.missCount;
482          }
483          return 1 - 1. * totalMissCount / totalOpCount;
484       }
485    }

487    private static class OpStats {
488       public final String opName;
489       public final int threadCount;
490       public final long opCount;
491       public final long runningTime;
492       public final long missCount;
493 
494       private OpStats(String opName, long opCount, long runningTime, long missCount) {
495          this.opName = opName;
496          this.threadCount = 1;
497          this.opCount = opCount;
498          this.runningTime = runningTime;
499          this.missCount = missCount;
500       }
501 
502       private OpStats(OpStats base, long opCount, long runningTime, long missCount) {
503          this.opName = base.opName;
504          this.threadCount = base.threadCount + 1;
505          this.opCount = base.opCount + opCount;
506          this.runningTime = base.runningTime + runningTime;
507          this.missCount = base.missCount + missCount;
508       }

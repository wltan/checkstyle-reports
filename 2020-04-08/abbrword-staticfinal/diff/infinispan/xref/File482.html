1   package org.infinispan.jcache.embedded;
2   
3   import java.net.URI;
4   import java.util.Properties;
5   
6   import javax.cache.CacheManager;
7   import javax.cache.configuration.OptionalFeature;
8   import javax.cache.spi.CachingProvider;
9   
10  import org.infinispan.jcache.AbstractJCachingProvider;
11  import org.infinispan.util.logging.Log;
12  import org.infinispan.util.logging.LogFactory;
13  import org.kohsuke.MetaInfServices;
14  
15  /**
16   * Infinispan's SPI hook up to {@link javax.cache.spi.CachingProvider}.
17   *
18   * @author Vladimir Blagojevic
19   * @author Galder Zamarre√±o
20   * @since 5.3
21   */
22  @MetaInfServices(CachingProvider.class)
23  @SuppressWarnings("unused")
24  public class JCachingProvider extends AbstractJCachingProvider {
25     private static final Log log = LogFactory.getLog(JCachingProvider.class);
26  
27     private static final URI DEFAULT_URI = URI.create(JCachingProvider.class.getName());
28  
29     @Override
30     public URI getDefaultURI() {
31        return DEFAULT_URI;
32     }
33  
34     @Override
35     public boolean isSupported(OptionalFeature optionalFeature) {
36        switch (optionalFeature) {
37           case STORE_BY_REFERENCE:
38              return true;
39           default:
40              return false;
41        }
42     }
43  
44     @Override
45     protected CacheManager createCacheManager(ClassLoader classLoader, URI uri, Properties properties) {
46        return new JCacheManager(uri, classLoader, this, properties);
47     }
48  }

44  @Test(groups = "stress", testName = "iteration.DistributedEntryRetrieverStressTest")
45  public class DistributedEntryRetrieverStressTest extends MultipleCacheManagersTest {
46     protected final String CACHE_NAME = getClass().getName();
47     protected final static int CACHE_COUNT = 5;
48     protected final static int THREAD_MULTIPLIER = 5;
49     protected final static int CACHE_ENTRY_COUNT = 250000;

54        builderUsed = new ConfigurationBuilder();
55        builderUsed.clustering().cacheMode(CacheMode.DIST_SYNC);
56        builderUsed.clustering().hash().numOwners(3);
57        builderUsed.clustering().stateTransfer().chunkSize(25000);
58        // This is increased just for the put all command when doing full tracing
59        builderUsed.clustering().sync().replTimeout(12000000);
60        // This way if an iterator gets stuck we know earlier
61  //      builderUsed.clustering().stateTransfer().timeout(240, TimeUnit.SECONDS);
62        createClusteredCaches(CACHE_COUNT, CACHE_NAME, builderUsed);

65     protected EmbeddedCacheManager addClusterEnabledCacheManager(TransportFlags flags) {
66        GlobalConfigurationBuilder gcb = GlobalConfigurationBuilder.defaultClusteredBuilder();
67        // Amend first so we can increase the transport thread pool
68        TestCacheManagerFactory.amendGlobalConfiguration(gcb, flags);
69        // we need to increase the transport and remote thread pools to default values
70        BlockingThreadPoolExecutorFactory executorFactory = new BlockingThreadPoolExecutorFactory(
71              25, 25, 10000, 30000);
72        gcb.transport().transportThreadPool().threadPoolFactory(executorFactory);
73  
74        gcb.transport().remoteCommandThreadPool().threadPoolFactory(executorFactory);
75  
76        EmbeddedCacheManager cm = TestCacheManagerFactory.newDefaultCacheManager(true, gcb, new ConfigurationBuilder(),                                                                               false);
77        cacheManagers.add(cm);
78        return cm;
79     }

81     @Test
82     public void testStressNodesLeavingWhileMultipleIterators() throws InterruptedException, ExecutionException, TimeoutException {
83        final Map<Integer, Integer> masterValues = new HashMap<Integer, Integer>();
84        // First populate our caches
85        for (int i = 0; i < CACHE_ENTRY_COUNT; ++i) {
86           masterValues.put(i, i);
87        }
88  
89        cache(0, CACHE_NAME).putAll(masterValues);
90  
91        final AtomicBoolean complete = new AtomicBoolean(false);
92        final Exchanger<Exception> exchanger = new Exchanger<Exception>();
93        // Now we spawn off CACHE_COUNT of threads.  All but one will constantly calling to iterator while another
94        // will constantly be killing and adding new caches
95        Future<Void>[] futures = new Future[(CACHE_COUNT - 1) * THREAD_MULTIPLIER + 1];
96        for (int j = 0; j < THREAD_MULTIPLIER; ++j) {
97        // We iterate over all but the last cache since we kill it constantly
98        for (int i = 0; i < CACHE_COUNT - 1; ++i) {
99           final Cache<Integer, Integer> cache = cache(i, CACHE_NAME);
100          futures[i + j * (CACHE_COUNT -1)] = fork(new Callable<Void>() {
101             @Override
102             public Void call() throws Exception {
103                try {
104                   int iteration = 0;
105                   while (!complete.get()) {
106                      log.tracef("Starting iteration %s", iteration);
107                      Map<Integer, Integer> seenValues = new HashMap<Integer, Integer>();
108                      try (CloseableIterable<CacheEntry<Integer, Integer>> iterable = cache.getAdvancedCache().filterEntries(new AllEntriesFilter())) {
109                         for (Map.Entry<Integer, Integer> entry : iterable) {
110                            if (seenValues.containsKey(entry.getKey())) {
111                               log.tracef("Seen values were: %s", seenValues);
112                               throw new IllegalArgumentException(Thread.currentThread() + "-Found duplicate value: " + entry.getKey() + " on iteration " + iteration);
113                            } else if (!masterValues.get(entry.getKey()).equals(entry.getValue())) {
114                               log.tracef("Seen values were: %s", seenValues);
115                               throw new IllegalArgumentException(Thread.currentThread() + "-Found incorrect value: " + entry.getKey() + " with value " + entry.getValue() + " on iteration " + iteration);
116                            }
117                            seenValues.put(entry.getKey(), entry.getValue());
118                         }
119                         if (seenValues.size() != masterValues.size()) {
120                            Map<Integer, Set<Map.Entry<Integer, Integer>>> target = generateEntriesPerSegment(cache.getAdvancedCache().getDistributionManager().getConsistentHash(), masterValues.entrySet());
121                            Map<Integer, Set<Map.Entry<Integer, Integer>>> actual = generateEntriesPerSegment(cache.getAdvancedCache().getDistributionManager().getConsistentHash(), seenValues.entrySet());
122                            for (Map.Entry<Integer, Set<Map.Entry<Integer, Integer>>> entry : target.entrySet()) {
123                               Set<Map.Entry<Integer, Integer>> entrySet = entry.getValue();
124                               Set<Map.Entry<Integer, Integer>> actualEntries = actual.get(entry.getKey());
125                               if (actualEntries != null) {
126                                  entrySet.removeAll(actualEntries);
127                               }
128                               if (!entrySet.isEmpty()) {
129                                  throw new IllegalArgumentException(Thread.currentThread() + "-Found incorrect amount of entries for segment " +
130                                                                           entry.getKey() + " missing entries " + entrySet
131                                                                           + " on iteration " + iteration);
132                               }
133                            }
134                         }
135                         iteration++;
136                      }
137                   }
138                   return null;
139                } catch (Exception e) {
140                   // Stop all the others as well
141                   complete.set(true);
142                   exchanger.exchange(e);
143                   throw e;
144                }
145             }
146          });
147       }
148       }
149 
150       // Then spawn a thread that just constantly kills the last cache and recreates over and over again
151       futures[futures.length - 1] = fork(new Callable<Void>() {
152          @Override
153          public Void call() throws Exception {
154             TestResourceTracker.backgroundTestStarted(DistributedEntryRetrieverStressTest.this);
155             try {
156                Cache<?, ?> cacheToKill = cache(CACHE_COUNT - 1);
157                while (!complete.get()) {
158                   Thread.sleep(1000);
159                   if (cacheManagers.remove(cacheToKill.getCacheManager())) {
160                      log.trace("Killing cache to force rehash");
161                      cacheToKill.getCacheManager().stop();
162                      List<Cache<Object, Object>> caches = caches(CACHE_NAME);
163                      if (caches.size() > 0) {
164                         TestingUtil.blockUntilViewsReceived(60000, false, caches);
165                         TestingUtil.waitForRehashToComplete(caches);
166                      }
167                   } else {
168                      throw new IllegalStateException("Cache Manager " + cacheToKill.getCacheManager() +
169                                                            " wasn't found for some reason!");
170                   }
171 
172                   log.trace("Adding new cache again to force rehash");
173                   // We should only create one so just make it the next cache manager to kill
174                   cacheToKill = createClusteredCaches(1, CACHE_NAME, builderUsed).get(0);
175                   log.trace("Added new cache again to force rehash");
176                }
177                return null;
178             } catch (Exception e) {
179                // Stop all the others as well
180                complete.set(true);
181                exchanger.exchange(e);
182                throw e;
183             }
184          }
185       });
186 
187       try {
188          // If this returns means we had an issue
189          Exception e = exchanger.exchange(null, 5, TimeUnit.MINUTES);
190          fail("Found an exception in at least 1 thread", e);
191       } catch (TimeoutException e) {
192 
193       }
194 
195       complete.set(true);
196 
197       // Make sure they all finish properly
198       for (int i = 0; i < futures.length; ++i) {
199          try {
200             futures[i].get(2, TimeUnit.MINUTES);
201          } catch (TimeoutException e) {
202             System.err.println("Future " + i + " did not complete in time allotted.");
203             throw e;
204          }
205       }
206    }

208    private <K, V> Map<Integer, Set<Map.Entry<K, V>>> generateEntriesPerSegment(ConsistentHash hash, Iterable<Map.Entry<K, V>> entries) {
209       Map<Integer, Set<Map.Entry<K, V>>> returnMap = new HashMap<Integer, Set<Map.Entry<K, V>>>();
210 
211       for (Map.Entry<K, V> value : entries) {
212          int segment = hash.getSegment(value.getKey());
213          Set<Map.Entry<K, V>> set = returnMap.get(segment);
214          if (set == null) {
215             set = new HashSet<Map.Entry<K, V>>();
216             returnMap.put(segment, set);
217          }
218          set.add(new ImmortalCacheEntry(value.getKey(), value.getValue()));
219       }
220       return returnMap;
221    }

161    /**
162     * A function that converts a type into another one.
163     *
164     * @param <E> Input type.
165     * @param <T> Output type.
166     */
167    public static interface Function<E, T> {
168 
169       /**
170        * Transforms an instance of the given input into an instace of the
171        * type to be returned.
172        *
173        * @param input Instance of the input type.
174        * @return Instance of the output type.
175        */
176       T transform(E input);
177    }
178 
179    /**
180     * A function that converts an entry into a key/value pair for use in a map.
181     * @param <K> generated key
182     * @param <V> generated value
183     * @param <E> entry input
184     */
185    public static interface MapMakerFunction<K, V, E> {
186       /**
187        * Transforms the given input into a key/value pair for use in a map
188        * @param input instance of the input type
189        * @return a Map.Entry parameterized with K and V
190        */
191       Map.Entry<K, V> transform(E input);
192    }
193 
194    /**
195     * Given a map of well known key/value types, it makes a shallow copy of it
196     * while at the same time transforming it's value type to a desired output
197     * type. The transformation of the value type is done using a given a
198     * function.
199     *
200     * @param input contains the input key/value pair map
201     * @param f function instance to use to transform the value part of the map
202     * @param <K> input map's key type
203     * @param <V> desired output type of the map's value
204     * @param <E> input map's value type
205     * @return a shallow copy of the input Map with all its values transformed.
206     */
207    public static <K, V, E> Map<K, V> transformMapValue(Map<K, E> input, Function<E, V> f) {
208       // This screams for a map function! Gimme functional programming pleasee...
209       if (input.isEmpty()) return InfinispanCollections.emptyMap();
210       if (input.size() == 1) {
211          Map.Entry<K, E> single = input.entrySet().iterator().next();
212          return singletonMap(single.getKey(), f.transform(single.getValue()));
213       } else {
214          Map<K, V> copy = new HashMap<K, V>(input.size());
215          for (Map.Entry<K, E> entry : input.entrySet())
216             copy.put(entry.getKey(), f.transform(entry.getValue()));
217          return unmodifiableMap(copy);
218       }
219    }
220 
221    /**
222     * Given a collection, transforms the collection to a map given a {@link MapMakerFunction}
223     *
224     * @param input contains a collection of type E
225     * @param f MapMakerFunction instance to use to transform the collection to a key/value pair
226     * @param <K> output map's key type
227     * @param <V> output type of the map's value
228     * @param <E> input collection's entry type
229     * @return a Map with keys and values generated from the input collection
230     */
231    public static <K, V, E> Map<K, V> transformCollectionToMap(Collection<? extends E> input, MapMakerFunction<K, V, ? super E> f) {
232       // This screams for a map function! Gimme functional programming pleasee...
233       if (input.isEmpty()) return InfinispanCollections.emptyMap();
234       if (input.size() == 1) {
235          E single = input.iterator().next();
236          Map.Entry<K, V> entry = f.transform(single);
237          return singletonMap(entry.getKey(), entry.getValue());
238       } else {
239          Map<K, V> map = new HashMap<K, V>(input.size());
240          for (E e : input) {
241             Map.Entry<K, V> entry = f.transform(e);
242             map.put(entry.getKey(), entry.getValue());
243          }
244          return unmodifiableMap(map);
245       }
246    }

248    /**
249     * Returns the elements that are present in s1 but which are not present
250     * in s2, without changing the contents of neither s1, nor s2.
251     *
252     * @param s1 first set
253     * @param s2 second set
254     * @param <E> type of objects in Set
255     * @return the elements in s1 that are not in s2
256     */
257    public static <E> Set<E> difference(Set<? extends E> s1, Set<? extends E> s2) {
258       Set<E> copy1 = new HashSet<E>(s1);
259       copy1.removeAll(new HashSet<E>(s2));
260       return copy1;
261    }

308    public static final <T> boolean containsAny(Collection<T> haystack, Collection<T> needle) {
309       for (T element : needle) {
310          if (haystack.contains(element))
311             return true;
312       }
313       return false;

28  /**
29   * Test to verify that it's possible to use the index using a JdbcStringBasedStore
30   *
31   * @see org.infinispan.persistence.jdbc.stringbased.JdbcStringBasedStore
32   *
33   * @author Sanne Grinovero
34   * @since 4.1
35   */
36  @SuppressWarnings("unchecked")
37  @Test(groups = "functional", testName = "lucene.DatabaseStoredIndexTest")
38  public class DatabaseStoredIndexTest extends SingleCacheManagerTest {
39  
40     private static final String DB_URL = "jdbc:h2:mem:infinispan;DB_CLOSE_DELAY=0";
41  
42     /** The INDEX_NAME */
43     private static final String INDEX_NAME = "testing index";
44  
45     private final HashMap cacheCopy = new HashMap();
46  
47     private Connection connection;
48  
49     public DatabaseStoredIndexTest() throws SQLException {
50        cleanup = CleanupPhase.AFTER_METHOD;
51        connection = DriverManager.getConnection(DB_URL, "sa", "");
52     }
53  
54     @AfterClass(alwaysRun=true)
55     public void closeDB() {
56        try {
57           connection.close();
58        } catch (SQLException e) {
59           AssertJUnit.fail("Could not close the keepalive DB connection");
60        }
61     }
62  
63     @Override
64     protected EmbeddedCacheManager createCacheManager() throws Exception {
65        ConfigurationBuilder cb = TestCacheManagerFactory.getDefaultCacheConfiguration(false);
66        cb.persistence()
67              .addStore(JdbcStringBasedStoreConfigurationBuilder.class)
68              .preload(true)
69              .key2StringMapper(LuceneKey2StringMapper.class)
70              .table()
71                 .idColumnName("ID_COLUMN")
72                 .idColumnType("VARCHAR(255)")
73                 .tableNamePrefix("ISPN_JDBC")
74                 .dataColumnName("DATA_COLUMN")
75                 .dataColumnType("BLOB")
76                 .timestampColumnName("TIMESTAMP_COLUMN")
77                 .timestampColumnType("BIGINT")
78              .simpleConnection()
79                 .driverClass(org.h2.Driver.class)
80                 .connectionUrl(DB_URL)
81                 .username("sa");
82        return TestCacheManagerFactory.createCacheManager(cb);
83     }
84  
85     @Test
86     public void testIndexUsage() throws IOException {
87        cache = cacheManager.getCache();
88        TestSegmentReadLocker testSegmentReadLocker = new TestSegmentReadLocker(cache, cache, cache, INDEX_NAME);
89        Directory dir = DirectoryBuilder.newDirectoryInstance(cache, cache, cache, INDEX_NAME)
90              .overrideSegmentReadLocker(testSegmentReadLocker).create();
91        writeTextToIndex(dir, 0, "hello database");
92        assertTextIsFoundInIds(dir, "hello", 0);
93        writeTextToIndex(dir, 1, "you have to store my index segments");
94        writeTextToIndex(dir, 2, "so that I can shut down all nodes");
95        writeTextToIndex(dir, 3, "and restart later keeping the index around");
96        assertTextIsFoundInIds(dir, "index", 1, 3);
97        removeByTerm(dir, "and");
98        assertTextIsFoundInIds(dir, "index", 1);
99        dir.close();
100       for (Map.Entry me : cache.getAdvancedCache().withFlags(Flag.SKIP_CACHE_LOAD).entrySet()) {
101          cacheCopy.put(me.getKey(), me.getValue());
102       }
103       cache.stop();
104       cacheManager.stop();
105    }
106 
107    @Test(dependsOnMethods="testIndexUsage")
108    public void indexWasStored() throws IOException {
109       cache = cacheManager.getCache();
110       boolean failed = false;
111       for (Object key : cacheCopy.keySet()) {
112          if (key instanceof FileReadLockKey) {
113             System.out.println("Key found in store, shouldn't have persisted this or should have cleaned up all readlocks on directory close:" + key);
114             failed = true;
115          }
116          else {
117             Object expected = cacheCopy.get(key);
118             Object actual = cache.get(key);
119             if (expected==null && actual==null)
120                continue;
121             if (expected instanceof byte[]) {
122                expected = Util.printArray((byte[]) expected, false);
123                actual = Util.printArray((byte[]) actual, false);
124             }
125             if (expected == null || ! expected.equals(actual)) {
126                System.out.println("Failure on key["+key.toString()+"] expected value:\n\t"+expected+"\tactual value:\n\t"+actual);
127                failed = true;
128             }
129          }
130       }
131       AssertJUnit.assertFalse(failed);
132       AssertJUnit.assertEquals("have a different number of keys", cacheCopy.keySet().size(), cache.keySet().size());
133       Directory dir = DirectoryBuilder.newDirectoryInstance(cache, cache, cache, INDEX_NAME).create();
134       assertTextIsFoundInIds(dir, "index", 1);
135       dir.close();
136    }

40   /**
41    * This handler is used to generate, given a JGroups model, a list of JGroups resource commands
42    * which can be used to recreate the model on another host.
43    *
44    * We use a custom handler as the JGroups subsystem makes non-standard use of stack operations
45    * add-protocol,remove-protocol for adding and removing protocol child resources.
46    *
47    * NB THis was required in order to maintain ordering of protocol layers.
48    *
49    * @author Paul Ferraro
50    * @author  Richard Achmatowicz (c) 2011 Red Hat Inc.
51    */
52   public class JGroupsSubsystemDescribe implements OperationStepHandler {
53  
54       public static final JGroupsSubsystemDescribe INSTANCE = new JGroupsSubsystemDescribe();
55  
56       @Override
57       public void execute(OperationContext context, ModelNode operation) throws OperationFailedException {
58           final ModelNode result = new ModelNode();
59           final PathAddress rootAddress = PathAddress.pathAddress(PathAddress.pathAddress(operation.require(ModelDescriptionConstants.OP_ADDR)).getLastElement());
60           final ModelNode subModel = Resource.Tools.readModel(context.readResource(PathAddress.EMPTY_ADDRESS));
61  
62           result.add(JGroupsSubsystemAdd.createOperation(rootAddress.toModelNode(), subModel));
63  
64           if (subModel.hasDefined(ModelKeys.STACK)) {
65               for (final Property stack : subModel.get(ModelKeys.STACK).asPropertyList()) {
66                   // process one stack
67                   final ModelNode stackAddress = rootAddress.toModelNode();
68                   stackAddress.add(ModelKeys.STACK, stack.getName());
69                   result.add(ProtocolStackAdd.createOperation(stackAddress, stack.getValue()));
70  
71                   // process the child resources
72  
73                   // transport=TRANSPORT
74                   if (stack.getValue().get(ModelKeys.TRANSPORT, ModelKeys.TRANSPORT_NAME).isDefined()) {
75                       ModelNode transport = stack.getValue().get(ModelKeys.TRANSPORT, ModelKeys.TRANSPORT_NAME);
76                       ModelNode transportAddress = stackAddress.clone();
77                       transportAddress.add(ModelKeys.TRANSPORT, ModelKeys.TRANSPORT_NAME);
78                       // no optional transport:add parameters will be present, so use attributes list
79                       result.add(createOperation(TransportResourceDefinition.TRANSPORT_ATTRIBUTES, transportAddress, transport));
80                       addProtocolPropertyCommands(transport, transportAddress, result);
81                   }
82                   // protocol=*
83                   if (stack.getValue().get(ModelKeys.PROTOCOL).isDefined()) {
84                       for (Property protocol : ProtocolStackAdd.getOrderedProtocolPropertyList(stack.getValue())) {
85                           // no optional transport:add parameters will be present, so use attributes list
86                           result.add(createProtocolOperation(ProtocolResourceDefinition.PROTOCOL_ATTRIBUTES, stackAddress, protocol.getValue()));
87                           ModelNode protocolAddress = stackAddress.clone();
88                           protocolAddress.add(ModelKeys.PROTOCOL, protocol.getName()) ;
89                           addProtocolPropertyCommands(protocol.getValue(), protocolAddress, result);
90                       }
91                   }
92                   // relay=RELAY
93                   if (stack.getValue().get(ModelKeys.RELAY, ModelKeys.RELAY_NAME).isDefined()) {
94                       ModelNode relay = stack.getValue().get(ModelKeys.RELAY, ModelKeys.RELAY_NAME);
95                       ModelNode relayAddress = stackAddress.clone();
96                       relayAddress.add(ModelKeys.RELAY, ModelKeys.RELAY_NAME);
97                       result.add(createOperation(RelayResourceDefinition.ATTRIBUTES, relayAddress, relay));
98                       addProtocolPropertyCommands(relay, relayAddress, result);
99                       // remote-site=*
100                      if (relay.get(ModelKeys.REMOTE_SITE).isDefined()) {
101                          for (final Property remoteSite : relay.get(ModelKeys.REMOTE_SITE).asPropertyList()) {
102                              ModelNode remoteSiteAddress = relayAddress.clone().add(ModelKeys.REMOTE_SITE, remoteSite.getName());
103                              // no optional transport:add parameters will be present, so use attributes list
104                              result.add(createOperation(RemoteSiteResourceDefinition.ATTRIBUTES, remoteSiteAddress, remoteSite.getValue()));
105                          }
106                      }
107                  }
108                  // sasl=SASL
109                  if (stack.getValue().get(ModelKeys.SASL, ModelKeys.SASL_NAME).isDefined()) {
110                      ModelNode sasl = stack.getValue().get(ModelKeys.SASL, ModelKeys.SASL_NAME);
111                      ModelNode saslAddress = stackAddress.clone();
112                      saslAddress.add(ModelKeys.SASL, ModelKeys.SASL_NAME);
113                      result.add(createOperation(SaslResourceDefinition.ATTRIBUTES, saslAddress, sasl));
114                      addProtocolPropertyCommands(sasl, saslAddress, result);
115                  }
116              }
117          }
118          context.getResult().set(result);
119          context.stepCompleted();
120      }
121 
122      private void addProtocolPropertyCommands(ModelNode protocol, ModelNode address, ModelNode result) throws OperationFailedException {
123 
124          if (protocol.hasDefined(ModelKeys.PROPERTY)) {
125               for (Property property : protocol.get(ModelKeys.PROPERTY).asPropertyList()) {
126                   ModelNode propertyAddress = address.clone().add(ModelKeys.PROPERTY, property.getName());
127                   AttributeDefinition[] ATTRIBUTE = {PropertyResourceDefinition.VALUE} ;
128                   result.add(createOperation(ATTRIBUTE, propertyAddress, property.getValue()));
129               }
130          }
131      }
132 
133      static ModelNode createOperation(AttributeDefinition[] attributes, ModelNode address, ModelNode existing) throws OperationFailedException {
134          ModelNode operation = Util.getEmptyOperation(ADD, address);
135          for(final AttributeDefinition attribute : attributes) {
136              attribute.validateAndSet(existing, operation);
137          }
138          return operation;
139      }
140 
141      static ModelNode createProtocolOperation(AttributeDefinition[] attributes, ModelNode address, ModelNode existing) throws OperationFailedException {
142          ModelNode operation = Util.getEmptyOperation(ModelKeys.ADD_PROTOCOL, address);
143          for(final AttributeDefinition attribute : attributes) {
144              attribute.validateAndSet(existing, operation);
145          }
146          return operation;
147      }

55           "/* @Indexed(false) */\n" +

57           "\toptional string notIndexedField = 1;\n" +

64     @Override
65     protected EmbeddedCacheManager createCacheManager() throws Exception {
66        org.infinispan.configuration.cache.ConfigurationBuilder builder = createConfigBuilder();
67        // the default key equivalence works only for byte[] so we need to override it with one that works for Object
68        builder.dataContainer().keyEquivalence(AnyEquivalence.getInstance());
69  
70        cacheManager = TestCacheManagerFactory.createCacheManager(builder);
71        cache = cacheManager.getCache();
72  
73        hotRodServer = HotRodClientTestingUtil.startHotRodServer(cacheManager);
74  
75        ConfigurationBuilder clientBuilder = new ConfigurationBuilder();
76        clientBuilder.addServer().host("127.0.0.1").port(hotRodServer.getPort());
77        clientBuilder.marshaller(new ProtoStreamMarshaller());
78        remoteCacheManager = new RemoteCacheManager(clientBuilder.build());
79  
80        remoteCache = remoteCacheManager.getCache();
81  
82        //initialize client-side serialization context
83        SerializationContext clientSerCtx = ProtoStreamMarshaller.getSerializationContext(remoteCacheManager);
84        MarshallerRegistration.registerMarshallers(clientSerCtx);
85        clientSerCtx.registerProtoFiles(FileDescriptorSource.fromString("not_indexed.proto", NOT_INDEXED_PROTO_SCHEMA));
86        clientSerCtx.registerMarshaller(new NotIndexedMarshaller());
87  
88        //initialize server-side serialization context
89        ProtobufMetadataManager protobufMetadataManager = cacheManager.getGlobalComponentRegistry().getComponent(ProtobufMetadataManager.class);
90        protobufMetadataManager.registerProtofile("sample_bank_account/bank.proto", Util.read(Util.getResourceAsStream("/sample_bank_account/bank.proto", getClass().getClassLoader())));
91        protobufMetadataManager.registerProtofile("not_indexed.proto", NOT_INDEXED_PROTO_SCHEMA);
92        assertNull(protobufMetadataManager.getFileErrors("sample_bank_account/bank.proto"));
93        assertNull(protobufMetadataManager.getFileErrors("not_indexed.proto"));
94        assertNull(protobufMetadataManager.getFilesWithErrors());
95  
96        protobufMetadataManager.registerMarshaller(new EmbeddedAccountMarshaller());
97        protobufMetadataManager.registerMarshaller(new EmbeddedUserMarshaller());
98        protobufMetadataManager.registerMarshaller(new NotIndexedMarshaller());
99  
100       return cacheManager;
101    }

103    protected org.infinispan.configuration.cache.ConfigurationBuilder createConfigBuilder() {
104       org.infinispan.configuration.cache.ConfigurationBuilder builder = hotRodCacheConfiguration();
105       builder.compatibility().enable().marshaller(new CompatibilityProtoStreamMarshaller());
106       builder.indexing().index(Index.ALL)
107             .addProperty("default.directory_provider", "ram")
108             .addProperty("lucene_version", "LUCENE_CURRENT");
109       return builder;
110    }

118    public void testPutAndGet() throws Exception {
119       Account account = createAccountPB(1);
120       remoteCache.put(1, account);
121 
122       // try to get the object through the local cache interface and check it's the same object we put
123       assertEquals(1, cache.keySet().size());
124       Object key = cache.keySet().iterator().next();
125       Object localObject = cache.get(key);
126       assertAccount((Account) localObject, AccountHS.class);
127 
128       // get the object through the remote cache interface and check it's the same object we put
129       Account fromRemoteCache = remoteCache.get(1);
130       assertAccount(fromRemoteCache, AccountPB.class);
131    }

273    public void testEmbeddedLuceneQuery() throws Exception {
274       Account account = createAccountPB(1);
275       remoteCache.put(1, account);
276 
277       // get account back from local cache via query and check its attributes
278       SearchManager searchManager = org.infinispan.query.Search.getSearchManager(cache);
279       org.apache.lucene.search.Query query = searchManager
280             .buildQueryBuilderForClass(AccountHS.class).get()
281             .keyword().wildcard().onField("description").matching("*test*").createQuery();
282       CacheQuery cacheQuery = searchManager.getQuery(query);
283       List<Object> list = cacheQuery.list();
284 
285       assertNotNull(list);
286       assertEquals(1, list.size());
287       assertAccount((Account) list.get(0), AccountHS.class);
288    }

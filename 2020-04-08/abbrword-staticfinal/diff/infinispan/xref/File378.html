22  @Test(testName = "remoting.rpc.RpcManagerCustomReplicableCommandTest", groups = "functional")
23  public class RpcManagerCustomReplicableCommandTest extends MultipleCacheManagersTest {
24  
25     protected static final String TEST_CACHE = "testCache";
26  
27     protected static final String EXPECTED_RETURN_VALUE = "the-return-value";
28  
29     @Override
30     protected void createCacheManagers() throws Throwable {
31        ConfigurationBuilder builder = getDefaultClusteredCacheConfig(CacheMode.REPL_SYNC, true);
32        createClusteredCaches(2, TEST_CACHE, builder);
33     }
34  
35     protected ReplicableCommand createReplicableCommandForTest(Object arg) {
36        return new CustomReplicableCommand(arg);
37     }
38  
39     /**
40      * Test to make sure that invokeRemotely returns the result from the remote side.
41      */
42     public void testInvokeRemotely() {
43        RpcManager rpcManager = cache(0, "testCache").getAdvancedCache().getRpcManager();
44        ReplicableCommand command = createReplicableCommandForTest(EXPECTED_RETURN_VALUE);
45  
46        Map<Address, Response> remoteResponses = rpcManager.invokeRemotely(null, command, true, true);
47        log.tracef("Responses were: %s", remoteResponses);
48  
49        assertEquals(1, remoteResponses.size());
50        Response response = remoteResponses.values().iterator().next();
51        assertNotNull(response);
52        assertTrue(response.isValid());
53        assertTrue(response.isSuccessful());
54        assertTrue(response instanceof SuccessfulResponse);
55        Object value = ((SuccessfulResponse) response).getResponseValue();
56        assertEquals(EXPECTED_RETURN_VALUE, value);
57     }
58  
59     /**
60      * Test to make sure that invokeRemotely results in a RemoteException.
61      */
62     public void testInvokeRemotelyWithRemoteException() {
63        RpcManager rpcManager = cache(0, "testCache").getAdvancedCache().getRpcManager();
64        ReplicableCommand command = createReplicableCommandForTest(new IllegalArgumentException("exception!"));
65  
66        try {
67           rpcManager.invokeRemotely(null, command, true, true);
68           fail("Expected RemoteException not thrown");
69        } catch (RemoteException e) {
70           assertTrue(e.getCause() instanceof IllegalArgumentException);
71           assertEquals("exception!", e.getCause().getMessage());
72        } catch (Exception ex) {
73           fail("Expected exception not thrown but instead we got : " + ex);
74        }
75     }
76  
77     /**

46  public class InitializationChain extends BytecodeScanningDetector {
47      Set<String> requires = new TreeSet<>();
48  
49      Map<String, Set<String>> classRequires = new TreeMap<>();
50  
51  
52  
53      private final BugReporter bugReporter;
54  
55      private final Map<XMethod, Set<XField>> staticFieldsRead = new HashMap<>();
56      private final Set<XField> staticFieldsReadInAnyConstructor = new HashSet<>();
57      private Set<XField> fieldsReadInThisConstructor = new HashSet<>();
58  
59      private final Set<XMethod> constructorsInvokedInStaticInitializer = new HashSet<>();
60      private final List<InvocationInfo> invocationInfo = new ArrayList<>();
61      private final Set<XField> warningGiven = new HashSet<>();
62  
63      private InvocationInfo lastInvocation;
64  
65      static class InvocationInfo {
66          public InvocationInfo(XMethod constructor, int pc) {
67              this.constructor = constructor;
68              this.pc = pc;
69          }
70          XMethod constructor;
71          int pc;
72          XField field;
73      }
74  
75      private static final boolean DEBUG = SystemProperties.getBoolean("ic.debug");
76  
77      public InitializationChain(BugReporter bugReporter) {
78          this.bugReporter = bugReporter;
79      }
80  
81      @Override
82      protected Iterable<Method> getMethodVisitOrder(JavaClass obj) {
83          ArrayList<Method> visitOrder = new ArrayList<>();
84          Method staticInitializer = null;
85          for(Method m : obj.getMethods()) {
86              String name = m.getName();
87              if (Const.STATIC_INITIALIZER_NAME.equals(name)) {
88                  staticInitializer = m;
89              } else if (Const.CONSTRUCTOR_NAME.equals(name)) {
90                  visitOrder.add(m);
91              }
92  
93          }
94          if (staticInitializer != null) {
95              visitOrder.add(staticInitializer);
96          }
97          return visitOrder;
98      }
99

101     @Override
102     public void visit(Code obj) {
103         fieldsReadInThisConstructor  = new HashSet<>();
104         super.visit(obj);
105         staticFieldsRead.put(getXMethod(), fieldsReadInThisConstructor);
106         requires.remove(getDottedClassName());
107         if ("java.lang.System".equals(getDottedClassName())) {
108             requires.add("java.io.FileInputStream");
109             requires.add("java.io.FileOutputStream");
110             requires.add("java.io.BufferedInputStream");
111             requires.add("java.io.BufferedOutputStream");
112             requires.add("java.io.PrintStream");
113         }
114         if (!requires.isEmpty()) {
115             classRequires.put(getDottedClassName(), requires);
116             requires = new TreeSet<>();
117         }
118     }
119 
120     @Override
121     public void visitAfter(JavaClass obj) {
122 
123         staticFieldsRead.clear();
124 
125         staticFieldsReadInAnyConstructor.clear();
126         fieldsReadInThisConstructor.clear();
127 
128         constructorsInvokedInStaticInitializer.clear();
129         invocationInfo.clear();
130         lastInvocation = null;
131 
132     }
133 
134     @Override
135     public void sawOpcode(int seen) {
136         InvocationInfo prev = lastInvocation;
137         lastInvocation = null;
138         if (Const.CONSTRUCTOR_NAME.equals(getMethodName())) {
139             if (seen == Const.GETSTATIC && getClassConstantOperand().equals(getClassName())) {
140                 staticFieldsReadInAnyConstructor.add(getXFieldOperand());
141                 fieldsReadInThisConstructor.add(getXFieldOperand());
142             }
143             return;
144         }
145 
146         if (seen == Const.INVOKESPECIAL && Const.CONSTRUCTOR_NAME.equals(getNameConstantOperand()) &&  getClassConstantOperand().equals(getClassName())) {
147 
148             XMethod m = getXMethodOperand();
149             Set<XField> read = staticFieldsRead.get(m);
150             if (constructorsInvokedInStaticInitializer.add(m) && read != null && !read.isEmpty()) {
151                 lastInvocation = new InvocationInfo(m, getPC());
152                 invocationInfo.add(lastInvocation);
153 
154             }
155 
156         }
157         if (seen == Const.PUTSTATIC && getClassConstantOperand().equals(getClassName())) {
158             XField f = getXFieldOperand();
159             if (prev != null) {
160                 prev.field = f;
161             }
162             if (staticFieldsReadInAnyConstructor.contains(f) && !warningGiven.contains(f)) {
163                 for(InvocationInfo i : invocationInfo) {
164                     Set<XField> fields = staticFieldsRead.get(i.constructor);
165                     if (fields != null && fields.contains(f)) {
166                         warningGiven.add(f);
167                         BugInstance bug = new BugInstance(this, "SI_INSTANCE_BEFORE_FINALS_ASSIGNED", NORMAL_PRIORITY).addClassAndMethod(this);
168                         if (i.field != null) {
169                             bug.addField(i.field).describe(FieldAnnotation.STORED_ROLE);
170                         }
171                         bug.addMethod(i.constructor).describe(MethodAnnotation.METHOD_CONSTRUCTOR);
172                         bug.addReferencedField(this).describe(FieldAnnotation.VALUE_OF_ROLE).addSourceLine(this, i.pc);
173                         bugReporter.reportBug(bug);
174                         break;
175 
176                     }
177                 }
178             }
179 
180         } else if (seen == Const.PUTSTATIC || seen == Const.GETSTATIC || seen == Const.INVOKESTATIC || seen == Const.NEW) {
181             if (getPC() + 6 < codeBytes.length) {
182                 requires.add(getDottedClassConstantOperand());
183             }
184         }
185     }
186 
187     public void compute() {
188         Set<String> allClasses = classRequires.keySet();
189         Set<String> emptyClasses = new TreeSet<>();
190         for (String c : allClasses) {
191             Set<String> needs = classRequires.get(c);
192             needs.retainAll(allClasses);
193             Set<String> extra = new TreeSet<>();
194             for (String need : needs) {
195                 extra.addAll(classRequires.get(need));
196             }
197             needs.addAll(extra);
198             needs.retainAll(allClasses);
199             classRequires.put(c, needs);
200             if (needs.isEmpty()) {
201                 emptyClasses.add(c);
202             }
203         }
204         for (String c : emptyClasses) {
205             classRequires.remove(c);
206         }
207     }
208 
209     @Override
210     public void report() {
211 
212         if (DEBUG) {
213             System.out.println("Finishing computation");
214         }
215         compute();
216         compute();
217         compute();
218         compute();
219         compute();
220         compute();
221         compute();
222         compute();
223 
224         for (Entry<String, Set<String>> entry : classRequires.entrySet()) {
225             String c = entry.getKey();
226             if (DEBUG) {
227                 System.out.println("Class " + c + " requires:");
228             }
229             for (String needs : entry.getValue()) {
230                 if (DEBUG) {
231                     System.out.println("  " + needs);
232                 }
233                 Set<String> s = classRequires.get(needs);
234                 if (s != null && s.contains(c) && c.compareTo(needs) < 0) {
235                     bugReporter.reportBug(new BugInstance(this, "IC_INIT_CIRCULARITY", NORMAL_PRIORITY).addClass(c).addClass(
236                             needs));
237                 }
238             }
239         }
240     }

29  /**
30   * Abstract base class for the {@link OnSubscribe} interface that helps you build Observable sources one
31   * {@code onNext} at a time, and automatically supports unsubscription and backpressure.
32   * <p>
33   * <h1>Usage rules</h1>
34   * When you implement the {@code next()} method, you
35   * <ul>
36   *  <li>should either
37   *   <ul>
38   *   <li>create the next value and signal it via {@link SubscriptionState#onNext state.onNext()},</li>
39   *   <li>signal a terminal condition via {@link SubscriptionState#onError state.onError()}, or
40   *       {@link SubscriptionState#onCompleted state.onCompleted()}, or</li>
41   *   <li>signal a stop condition via {@link SubscriptionState#stop state.stop()} indicating no further values
42   *       will be sent.</li>
43   *   </ul>
44   *  </li>
45   *  <li>may
46   *   <ul>
47   *   <li>call {@link SubscriptionState#onNext state.onNext()} and either
48   *       {@link SubscriptionState#onError state.onError()} or
49   *       {@link SubscriptionState#onCompleted state.onCompleted()} together, and
50   *   <li>block or sleep.
51   *   </ul>
52   *  </li>
53   *  <li>should not
54   *   <ul>
55   *   <li>do nothing or do async work and not produce any event or request stopping. If neither of 
56   *       the methods are called, an {@link IllegalStateException} is forwarded to the {@code Subscriber} and
57   *       the Observable is terminated;</li>
58   *   <li>call the {@code state.on}<i>foo</i>() methods more than once (yields
59   *       {@link IllegalStateException}).</li>
60   *   </ul>
61   *  </li>
62   * </ul>
63   * 
64   * The {@link SubscriptionState} object features counters that may help implement a state machine:
65   * <ul>
66   * <li>A call counter, accessible via {@link SubscriptionState#calls state.calls()} tells how many times the
67   *     {@code next()} was run (zero based).</li>
68   * <li>You can use a phase counter, accessible via {@link SubscriptionState#phase state.phase}, that helps track
69   *     the current emission phase, in a {@code switch()} statement to implement the state machine. (It is named
70   *     {@code phase} to avoid confusion with the per-subscriber state.)</li>
71   * <li>You can arbitrarily change the current phase with
72   *     {@link SubscriptionState#advancePhase state.advancePhase()}, 
73   *     {@link SubscriptionState#advancePhaseBy(int) state.advancedPhaseBy(int)} and
74   *     {@link SubscriptionState#phase(int) state.phase(int)}.</li>
75   * </ul>
76   * <p>
77   * When you implement {@code AbstractOnSubscribe}, you may override {@link AbstractOnSubscribe#onSubscribe} to
78   * perform special actions (such as registering {@code Subscription}s with {@code Subscriber.add()}) and return
79   * additional state for each subscriber subscribing. You can access this custom state with the
80   * {@link SubscriptionState#state state.state()} method. If you need to do some cleanup, you can override the
81   * {@link #onTerminated} method.
82   * <p>
83   * For convenience, a lambda-accepting static factory method, {@link #create}, is available.
84   * Another convenience is {@link #toObservable} which turns an {@code AbstractOnSubscribe}
85   * instance into an {@code Observable} fluently.
86   * 
87   * <h1>Examples</h1>
88   * Note: these examples use the lambda-helper factories to avoid boilerplate.
89   * 
90   * <h3>Implement: just</h3>
91   * <pre><code>
92   * AbstractOnSubscribe.create(s -> {
93   *   s.onNext(1);
94   *   s.onCompleted();
95   * }).toObservable().subscribe(System.out::println);
96   * </code></pre>
97  
98   * <h3>Implement: from Iterable</h3>
99   * <pre><code>
100  * Iterable<T> iterable = ...;
101  * AbstractOnSubscribe.create(s -> {
102  *   Iterator<T> it = s.state();
103  *   if (it.hasNext()) {
104  *     s.onNext(it.next());
105  *   }
106  *   if (!it.hasNext()) {
107  *     s.onCompleted();
108  *   }
109  * }, u -> iterable.iterator()).subscribe(System.out::println);
110  * </code></pre>
111  *
112  * <h3>Implement source that fails a number of times before succeeding</h3>
113  * <pre><code>
114  * AtomicInteger fails = new AtomicInteger();
115  * int numFails = 50;
116  * AbstractOnSubscribe.create(s -> {
117  *   long c = s.calls();
118  *   switch (s.phase()) {
119  *   case 0:
120  *     s.onNext("Beginning");
121  *     s.onError(new RuntimeException("Oh, failure.");
122  *     if (c == numFails.getAndIncrement()) {
123  *       s.advancePhase();
124  *     }
125  *     break;
126  *   case 1:
127  *     s.onNext("Beginning");
128  *     s.advancePhase();
129  *   case 2:
130  *     s.onNext("Finally working");
131  *     s.onCompleted();
132  *     s.advancePhase();
133  *   default:
134  *     throw new IllegalStateException("How did we get here?");
135  *   }
136  * }).subscribe(System.out::println);
137  * </code></pre>
138 
139  * <h3>Implement: never</h3>
140  * <pre><code>
141  * AbstractOnSubscribe.create(s -> {
142  *   s.stop();
143  * }).toObservable()
144  * .timeout(1, TimeUnit.SECONDS)
145  * .subscribe(System.out::println, Throwable::printStacktrace, () -> System.out.println("Done"));
146  * </code></pre>
147  *
148  * @param <T> the value type
149  * @param <S> the per-subscriber user-defined state type
150  * @since (if this graduates from Experimental/Beta to supported, replace this parenthetical with the release number)
151  * @Experimental
152  */
153 @Experimental
154 public abstract class AbstractOnSubscribe<T, S> implements OnSubscribe<T> {
155     /**
156      * Called when a Subscriber subscribes and lets the implementor create a per-subscriber custom state.
157      * <p>
158      * Override this method to have custom state per-subscriber. The default implementation returns
159      * {@code null}.
160      *
161      * @param subscriber the subscriber who is subscribing
162      * @return the custom state
163      */
164     protected S onSubscribe(Subscriber<? super T> subscriber) {
165         return null;
166     }
167 
168     /**
169      * Called after the terminal emission or when the downstream unsubscribes.
170      * <p>
171      * This is called only once and no {@code onNext} call will run concurrently with it. The default
172      * implementation does nothing.
173      *
174      * @param state the user-provided state
175      */
176     protected void onTerminated(S state) {
177         
178     }
179 
180     /**
181      * Override this method to create an emission state-machine.
182      *
183      * @param state the per-subscriber subscription state
184      */
185     protected abstract void next(SubscriptionState<T, S> state);
186 
187     @Override
188     public final void call(final Subscriber<? super T> subscriber) {
189         final S custom = onSubscribe(subscriber);
190         final SubscriptionState<T, S> state = new SubscriptionState<T, S>(this, subscriber, custom);
191         subscriber.add(new SubscriptionCompleter<T, S>(state));
192         subscriber.setProducer(new SubscriptionProducer<T, S>(state));
193     }
194     
195     /**
196      * Convenience method to create an Observable from this implemented instance.
197      *
198      * @return the created observable
199      */
200     public final Observable<T> toObservable() {
201         return Observable.create(this);
202     }

206         @Override
207         public Object call(Object t1) {
208             return null;
209         }

213      * Creates an {@code AbstractOnSubscribe} instance which calls the provided {@code next} action.
214      * <p>
215      * This is a convenience method to help create {@code AbstractOnSubscribe} instances with the help of
216      * lambdas.
217      *
218      * @param <T> the value type
219      * @param <S> the per-subscriber user-defined state type
220      * @param next the next action to call
221      * @return an {@code AbstractOnSubscribe} instance
222      */
223     public static <T, S> AbstractOnSubscribe<T, S> create(Action1<SubscriptionState<T, S>> next) {
224         @SuppressWarnings("unchecked")
225         Func1<? super Subscriber<? super T>, ? extends S> nullFunc =
226                 (Func1<? super Subscriber<? super T>, ? extends S>)NULL_FUNC1;
227         return create(next, nullFunc, Actions.empty());
228     }

231      * Creates an {@code AbstractOnSubscribe} instance which creates a custom state with the {@code onSubscribe}
232      * function and calls the provided {@code next} action.
233      * <p>
234      * This is a convenience method to help create {@code AbstractOnSubscribe} instances with the help of
235      * lambdas.
236      *
237      * @param <T> the value type
238      * @param <S> the per-subscriber user-defined state type
239      * @param next the next action to call
240      * @param onSubscribe the function that returns a per-subscriber state to be used by {@code next}
241      * @return an {@code AbstractOnSubscribe} instance
242      */
243     public static <T, S> AbstractOnSubscribe<T, S> create(Action1<SubscriptionState<T, S>> next,
244             Func1<? super Subscriber<? super T>, ? extends S> onSubscribe) {
245         return create(next, onSubscribe, Actions.empty());
246     }

248     /**
249      * Creates an {@code AbstractOnSubscribe} instance which creates a custom state with the {@code onSubscribe}
250      * function, calls the provided {@code next} action and calls the {@code onTerminated} action to release the
251      * state when its no longer needed.
252      * <p>
253      * This is a convenience method to help create {@code AbstractOnSubscribe} instances with the help of
254      * lambdas.
255      *
256      * @param <T> the value type
257      * @param <S> the per-subscriber user-defined state type
258      * @param next the next action to call
259      * @param onSubscribe the function that returns a per-subscriber state to be used by {@code next}
260      * @param onTerminated the action to call to release the state created by the {@code onSubscribe} function
261      * @return an {@code AbstractOnSubscribe} instance
262      */
263     public static <T, S> AbstractOnSubscribe<T, S> create(Action1<SubscriptionState<T, S>> next,
264             Func1<? super Subscriber<? super T>, ? extends S> onSubscribe, Action1<? super S> onTerminated) {
265         return new LambdaOnSubscribe<T, S>(next, onSubscribe, onTerminated);
266     }

268     /**
269      * An implementation that forwards the three main methods ({@code next}, {@code onSubscribe}, and
270      * {@code onTermianted}) to functional callbacks.
271      *
272      * @param <T> the value type
273      * @param <S> the per-subscriber user-defined state type
274      */
275     private static final class LambdaOnSubscribe<T, S> extends AbstractOnSubscribe<T, S> {
276         final Action1<SubscriptionState<T, S>> next;
277         final Func1<? super Subscriber<? super T>, ? extends S> onSubscribe;
278         final Action1<? super S> onTerminated;
279         private LambdaOnSubscribe(Action1<SubscriptionState<T, S>> next,
280                 Func1<? super Subscriber<? super T>, ? extends S> onSubscribe, Action1<? super S> onTerminated) {
281             this.next = next;
282             this.onSubscribe = onSubscribe;
283             this.onTerminated = onTerminated;
284         }
285         @Override
286         protected S onSubscribe(Subscriber<? super T> subscriber) {
287             return onSubscribe.call(subscriber);
288         }
289         @Override
290         protected void onTerminated(S state) {
291             onTerminated.call(state);
292         }
293         @Override
294         protected void next(SubscriptionState<T, S> state) {
295             next.call(state);
296         }
297     }

299     /**
300      * Manages unsubscription of the state.
301      *
302      * @param <T> the value type
303      * @param <S> the per-subscriber user-defined state type
304      */
305     private static final class SubscriptionCompleter<T, S> extends AtomicBoolean implements Subscription {
306         private static final long serialVersionUID = 7993888274897325004L;
307         private final SubscriptionState<T, S> state;
308         private SubscriptionCompleter(SubscriptionState<T, S> state) {
309             this.state = state;
310         }
311         @Override
312         public boolean isUnsubscribed() {
313             return get();
314         }
315         @Override
316         public void unsubscribe() {
317             if (compareAndSet(false, true)) {
318                 state.free();
319             }
320         }

323     /**
324      * Contains the producer loop that reacts to downstream requests of work.
325      *
326      * @param <T> the value type
327      * @param <S> the per-subscriber user-defined state type
328      */
329     private static final class SubscriptionProducer<T, S> implements Producer {
330         final SubscriptionState<T, S> state;
331         private SubscriptionProducer(SubscriptionState<T, S> state) {
332             this.state = state;
333         }
334         @Override
335         public void request(long n) {
336             if (n > 0 && BackpressureUtils.getAndAddRequest(state.requestCount, n) == 0) {
337                 if (n == Long.MAX_VALUE) {
338                     // fast-path
339                     for (; !state.subscriber.isUnsubscribed(); ) {
340                         if (!doNext()) {
341                             break;
342                         }
343                     }
344                 } else 
345                 if (!state.subscriber.isUnsubscribed()) {
346                     do {
347                         if (!doNext()) {
348                             break;
349                         }
350                     } while (state.requestCount.decrementAndGet() > 0 && !state.subscriber.isUnsubscribed());
351                 }
352             }
353         }
354 
355         /**
356          * Executes the user-overridden next() method and performs state bookkeeping and
357          * verification.
358          *
359          * @return true if the outer loop may continue
360          */
361         protected boolean doNext() {
362             if (state.use()) {
363                 try {
364                     int p = state.phase();
365                     state.parent.next(state);
366                     if (!state.verify()) {
367                         throw new IllegalStateException("No event produced or stop called @ Phase: " + p + " -> " + state.phase() + ", Calls: " + state.calls());
368                     }
369                     if (state.accept() || state.stopRequested()) {
370                         state.terminate();
371                         return false;
372                     }
373                     state.calls++;
374                 } catch (Throwable t) {
375                     state.terminate();
376                     state.subscriber.onError(t);
377                     return false;
378                 } finally {
379                     state.free();
380                 }
381                 return true;
382             }
383             return false;
384         }
385     }

387     /**
388      * Represents a per-subscription state for the {@code AbstractOnSubscribe} operation. It supports phasing
389      * and counts the number of times a value was requested by the downstream.
390      *
391      * @param <T> the value type 
392      * @param <S> the per-subscriber user-defined state type
393      * @since (if this graduates from Experimental/Beta to supported, replace this parenthetical with the release number)
394      * @Experimental
395      */
396     public static final class SubscriptionState<T, S> {
397         private final AbstractOnSubscribe<T, S> parent;
398         private final Subscriber<? super T> subscriber;
399         private final S state;
400         private final AtomicLong requestCount;
401         private final AtomicInteger inUse;
402         private int phase;
403         private long calls;
404         private T theValue;
405         private boolean hasOnNext;
406         private boolean hasCompleted;
407         private boolean stopRequested;
408         private Throwable theException;
409         private SubscriptionState(AbstractOnSubscribe<T, S> parent, Subscriber<? super T> subscriber, S state) {
410             this.parent = parent;
411             this.subscriber = subscriber;
412             this.state = state;
413             this.requestCount = new AtomicLong();
414             this.inUse = new AtomicInteger(1);
415         }
416 
417         /**
418          * @return the per-subscriber specific user-defined state created via
419          *         {@link AbstractOnSubscribe#onSubscribe}
420          */
421         public S state() {
422             return state;
423         }
424 
425         /**
426          * @return the current phase value
427          */
428         public int phase() {
429             return phase;
430         }
431 
432         /**
433          * Sets a new phase value.
434          *
435          * @param newPhase
436          */
437         public void phase(int newPhase) {
438             phase = newPhase;
439         }
440 
441         /**
442          * Advance the current phase by 1.
443          */
444         public void advancePhase() {
445             advancePhaseBy(1);
446         }
447 
448         /**
449          * Advance the current phase by the given amount (can be negative).
450          *
451          * @param amount the amount to advance the phase
452          */
453         public void advancePhaseBy(int amount) {
454             phase += amount;
455         }
456 
457         /**
458          * @return the number of times {@link AbstractOnSubscribe#next} was called so far, starting at 0 for the
459          *         very first call
460          */
461         public long calls() {
462             return calls;
463         }
464 
465         /**
466          * Call this method to offer the next {@code onNext} value for the subscriber.
467          * 
468          * @param value the value to {@code onNext}
469          * @throws IllegalStateException if there is a value already offered but not taken or a terminal state
470          *         is reached
471          */
472         public void onNext(T value) {
473             if (hasOnNext) {
474                 throw new IllegalStateException("onNext not consumed yet!");
475             } else
476             if (hasCompleted) {
477                 throw new IllegalStateException("Already terminated", theException);
478             }
479             theValue = value;
480             hasOnNext = true;
481         }
482 
483         /**
484          * Call this method to send an {@code onError} to the subscriber and terminate all further activities.
485          * If there is a pending {@code onNext}, that value is emitted to the subscriber followed by this
486          * exception.
487          * 
488          * @param e the exception to deliver to the client
489          * @throws IllegalStateException if the terminal state has been reached already
490          */
491         public void onError(Throwable e) {
492             if (e == null) {
493                 throw new NullPointerException("e != null required");
494             }
495             if (hasCompleted) {
496                 throw new IllegalStateException("Already terminated", theException);
497             }
498             theException = e;
499             hasCompleted = true;
500         }
501 
502         /**
503          * Call this method to send an {@code onCompleted} to the subscriber and terminate all further
504          * activities. If there is a pending {@code onNext}, that value is emitted to the subscriber followed by
505          * this exception.
506          * 
507          * @throws IllegalStateException if the terminal state has been reached already
508          */
509         public void onCompleted() {
510             if (hasCompleted) {
511                 throw new IllegalStateException("Already terminated", theException);
512             }
513             hasCompleted = true;
514         }
515 
516         /**
517          * Signals that there won't be any further events.
518          */
519         public void stop() {
520             stopRequested = true;
521         }
522 
523         /**
524          * Emits the {@code onNext} and/or the terminal value to the actual subscriber.
525          *
526          * @return {@code true} if the event was a terminal event
527          */
528         protected boolean accept() {
529             if (hasOnNext) {
530                 T value = theValue;
531                 theValue = null;
532                 hasOnNext = false;
533                 
534                 try {
535                     subscriber.onNext(value);
536                 } catch (Throwable t) {
537                     hasCompleted = true;
538                     Throwable e = theException;
539                     theException = null;
540                     if (e == null) {
541                         subscriber.onError(t);
542                     } else {
543                         subscriber.onError(new CompositeException(Arrays.asList(t, e)));
544                     }
545                     return true;
546                 }
547             }
548             if (hasCompleted) {
549                 Throwable e = theException;
550                 theException = null;
551                 
552                 if (e != null) {
553                     subscriber.onError(e);
554                 } else {
555                     subscriber.onCompleted();
556                 }
557                 return true;
558             }
559             return false;
560         }
561 
562         /**
563          * Verify if the {@code next()} generated an event or requested a stop.
564          *
565          * @return true if either event was generated or stop was requested
566          */
567         protected boolean verify() {
568             return hasOnNext || hasCompleted || stopRequested;
569         }
570 
571         /** @return true if the {@code next()} requested a stop */
572         protected boolean stopRequested() {
573             return stopRequested;
574         }
575 
576         /**
577          * Request the state to be used by {@code onNext} or returns {@code false} if the downstream has
578          * unsubscribed.
579          *
580          * @return {@code true} if the state can be used exclusively
581          * @throws IllegalStateEception
582          * @warn "throws" section incomplete
583          */
584         protected boolean use() {
585             int i = inUse.get();
586             if (i == 0) {
587                 return false;
588             } else
589             if (i == 1 && inUse.compareAndSet(1, 2)) {
590                 return true;
591             }
592             throw new IllegalStateException("This is not reentrant nor threadsafe!");
593         }
594 
595         /**
596          * Release the state if there are no more interest in it and it is not in use.
597          */
598         protected void free() {
599             int i = inUse.get();
600             if (i <= 0) {
601                 return;
602             } else
603             if (inUse.decrementAndGet() == 0) {
604                 parent.onTerminated(state);
605             }
606         }
607 
608         /**
609          * Terminates the state immediately and calls {@link AbstractOnSubscribe#onTerminated} with the custom
610          * state.
611          */
612         protected void terminate() {
613             for (;;) {
614                 int i = inUse.get();
615                 if (i <= 0) {
616                     return;
617                 }
618                 if (inUse.compareAndSet(i, 0)) {
619                     parent.onTerminated(state);
620                     break;
621                 }
622             }
623         }
624     }

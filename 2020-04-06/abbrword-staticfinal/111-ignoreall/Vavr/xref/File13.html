41      // ------------------------------------------------------------------------
42      // static range, rangeBy, rangeClosed, rangeCloseBy tests
43      //
44      // Basically there are the following tests scenarios:
45      //
46      // * step == 0
47      // * from == to, step < 0
48      // * from == to, step > 0
49      // * from < to, step < 0
50      // * from < to, step > 0
51      // * from > to, step < 0
52      // * from > to, step > 0
53      //
54      // Additionally we have to test these special cases
55      // (where MIN/MAX may also be NEGATIVE_INFINITY, POSITIVE_INFINITY):
56      //
57      // * from = MAX, to = MAX, step < 0
58      // * from = MAX, to = MAX, step > 0
59      // * from = MIN, to = MIN, step < 0
60      // * from = MIN, to = MIN, step > 0
61      // * from = MAX - x, to = MAX, step > 0, 0 < x < step
62      // * from = MAX - step, to = MAX, step > 0
63      // * from = MAX - x, to = MAX, step > 0, x > step
64      // * from = MIN, to = MIN - x, step < 0, 0 > x > step
65      // * from = MIN, to = MIN - step, step < 0
66      // * from = MIN, to = MIN - x, step < 0, x < step
67      //
68      // All of these scenarios are multiplied with
69      //
70      // * the inclusive and exclusive case
71      // * the with and without step case
72      //
73      // ------------------------------------------------------------------------

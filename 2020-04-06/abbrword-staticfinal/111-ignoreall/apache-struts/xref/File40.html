19  package org.apache.struts2.json;
20  
21  import java.util.Map;
22  
23  import junit.framework.TestCase;
24  
25  /**
26   * Test serialization of an Enum in the two supported modes: - Enum as a
27   * name=value pair; - Enum as a bean
28   */
29  public class JSONEnumTest extends TestCase {
30  
31      /**
32       * Asserts that a bean can be serialized to JSON and restored as a map
33       */
34      public void testEnumAsNameValue() throws Exception {
35          Bean bean1 = new Bean();
36  
37          bean1.setStringField("str");
38          bean1.setBooleanField(true);
39          bean1.setCharField('s');
40          bean1.setDoubleField(10.1);
41          bean1.setFloatField(1.5f);
42          bean1.setIntField(10);
43          bean1.setLongField(100);
44          bean1.setEnumField(AnEnum.ValueA);
45          bean1.setEnumBean(AnEnumBean.Two);
46  
47          JSONWriter jsonWriter = new DefaultJSONWriter();
48          jsonWriter.setEnumAsBean(false);
49          String json = jsonWriter.write(bean1);
50  
51          Map result = (Map) JSONUtil.deserialize(json);
52          assertEquals("str", result.get("stringField"));
53          assertEquals(true, result.get("booleanField"));
54          assertEquals("s", result.get("charField")); // note: this is a
55                                                              // String
56          assertEquals(10.1, result.get("doubleField"));
57          assertEquals(1.5, result.get("floatField")); // note: this is a
58                                                              // Double
59          assertEquals(10L, result.get("intField")); // note: this is a
60                                                              // Long
61          assertEquals(AnEnum.ValueA, AnEnum.valueOf((String) result.get("enumField"))); // note:
62                                                                                          // this
63                                                                                          // is a
64                                                                                          // String
65          assertEquals(AnEnumBean.Two, AnEnumBean.valueOf((String) result.get("enumBean"))); // note:
66                                                                                              // this
67                                                                                              // is a
68                                                                                              // String
69      }
70  
71      /**
72       * Asserts that a bean can be serialized to JSON and restored as a map <br>
73       * In this case, the name of the enum is in _name and the two properties of
74       * AnEnumBean are also serialized
75       */
76      public void testEnumAsBean() throws Exception {
77          Bean bean1 = new Bean();
78  
79          bean1.setStringField("str");
80          bean1.setBooleanField(true);
81          bean1.setCharField('s');
82          bean1.setDoubleField(10.1);
83          bean1.setFloatField(1.5f);
84          bean1.setIntField(10);
85          bean1.setLongField(100);
86          bean1.setEnumField(AnEnum.ValueA);
87          bean1.setEnumBean(AnEnumBean.Two);
88  
89          JSONWriter jsonWriter = new DefaultJSONWriter();
90          jsonWriter.setEnumAsBean(true);
91          String json = jsonWriter.write(bean1);
92  
93          Map result = (Map) JSONUtil.deserialize(json);
94          assertEquals("str", result.get("stringField"));
95          assertEquals(true, result.get("booleanField"));
96          assertEquals("s", result.get("charField")); // note: this is a
97                                                              // String
98          assertEquals(10.1, result.get("doubleField"));
99          assertEquals(1.5, result.get("floatField")); // note: this is a
100                                                             // Double
101         assertEquals(10L, result.get("intField")); // note: this is a
102                                                             // Long
103         Map enumBean1 = (Map) result.get("enumField");
104         assertNotNull(enumBean1);
105         assertEquals(AnEnum.ValueA, AnEnum.valueOf((String) enumBean1.get("_name"))); // get
106                                                                                         // the
107                                                                                         // special
108                                                                                         // name
109                                                                                         // property
110         Map enumBean2 = (Map) result.get("enumBean");
111         assertEquals(AnEnumBean.Two, AnEnumBean.valueOf((String) enumBean2.get("_name"))); // get
112                                                                                             // the
113                                                                                             // special
114                                                                                             // name
115                                                                                             // property
116         assertEquals(AnEnumBean.Two.getPropA(), (String) enumBean2.get("propA")); // get
117                                                                                     // the
118                                                                                     // propA
119                                                                                     // property
120         assertEquals(AnEnumBean.Two.getPropB(), (String) enumBean2.get("propB")); // get
121                                                                                     // the
122                                                                                     // propA
123                                                                                     // property
124     }

81      private static void verify(Provider provider, String alg, PublicKey key, byte[] data, byte[] sig, boolean result) throws Exception {
82          Signature s = Signature.getInstance(alg, provider);
83          s.initVerify(key);
84          boolean r;
85          s.update(data);
86          r = s.verify(sig);
87          if (r != result) {
88              throw new Exception("Result mismatch, actual: " + r);
89          }
90          s.update(data);
91          r = s.verify(sig);
92          if (r != result) {
93              throw new Exception("Result mismatch, actual: " + r);
94          }
95          System.out.println("Passed");
96      }

98      private static void sign(Provider provider, String alg, PrivateKey key, byte[] data) throws Exception {
99          Signature s = Signature.getInstance(alg, provider);
100         s.initSign(key);
101         s.update(data);
102         byte[] sig = s.sign();
103         System.out.println(toString(sig));
104     }

157     private void test(Provider provider, String pub, String priv, byte[] sig) throws Exception {
158 
159         KeyFactory kf = KeyFactory.getInstance("EC", provider);
160         X509EncodedKeySpec pubSpec = new X509EncodedKeySpec(parse(pub));
161         PKCS8EncodedKeySpec privSpec = new PKCS8EncodedKeySpec(parse(priv));
162 
163         PrivateKey privateKey = kf.generatePrivate(privSpec);
164         PublicKey publicKey = kf.generatePublic(pubSpec);
165 
166         if (false) {
167             sign(provider, "SHA1withECDSA", privateKey, data1Raw);
168 //          sign(provider, "NONEwithECDSA", privateKey, data1SHA);
169             return;
170         }
171 
172         // verify known-good and known-bad signatures using SHA1withECDSA and NONEwithECDSA
173         verify(provider, "SHA1withECDSA", publicKey, data1Raw, sig, true);
174         verify(provider, "SHA1withECDSA", publicKey, data2Raw, sig, false);
175 
176         verify(provider, "NONEwithECDSA", publicKey, data1SHA, sig, true);
177         verify(provider, "NONEwithECDSA", publicKey, data2SHA, sig, false);
178 
179         testSigning(provider, privateKey, publicKey);
180     }

182     private void testSigning(Provider provider, PrivateKey privateKey,
183             PublicKey publicKey) throws Exception {
184         byte[] data = new byte[2048];
185         new Random().nextBytes(data);
186 
187         // sign random data using SHA1withECDSA and verify using
188         // SHA1withECDSA and NONEwithECDSA
189         Signature s = Signature.getInstance("SHA1withECDSA", provider);
190         s.initSign(privateKey);
191         s.update(data);
192         byte[] s1 = s.sign();
193 
194         s.initVerify(publicKey);
195         s.update(data);
196         if (!s.verify(s1)) {
197             throw new Exception("Sign/verify 1 failed");
198         }
199 
200         s = Signature.getInstance("NONEwithECDSA", provider);
201         MessageDigest md = MessageDigest.getInstance("SHA-1");
202         byte[] digest = md.digest(data);
203         s.initVerify(publicKey);
204         s.update(digest);
205         if (!s.verify(s1)) {
206             throw new Exception("Sign/verify 2 failed");
207         }
208 
209         // sign random data using NONEwithECDSA and verify using
210         // SHA1withECDSA and NONEwithECDSA
211         s.initSign(privateKey);
212         s.update(digest);
213         byte[] s2 = s.sign();
214 
215         s.initVerify(publicKey);
216         s.update(digest);
217         if (!s.verify(s2)) {
218             throw new Exception("Sign/verify 3 failed");
219         }
220 
221         s = Signature.getInstance("SHA1withECDSA", provider);
222         s.initVerify(publicKey);
223         s.update(data);
224         if (!s.verify(s2)) {
225             throw new Exception("Sign/verify 4 failed");
226         }

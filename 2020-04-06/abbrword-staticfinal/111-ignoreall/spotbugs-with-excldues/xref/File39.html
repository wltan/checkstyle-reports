57  /**
58   * A Detector to look for streams that are opened in a method, do not escape the
59   * method, and are not closed on all paths out of the method. Note that "stream"
60   * is a bit misleading, since we also use the detector to look for database
61   * resources that aren't closed.
62   *
63   * @author David Hovemeyer
64   * @author Agustin Toribio atomo@arrakis.es
65   */

279     // List of words that must appear in names of classes which
280     // create possible resources to be tracked. If we don't see a
281     // class containing one of these words, then we don't run the
282     // detector on the class.
283     private static final String[] PRESCREEN_CLASS_LIST = { "Stream", "Reader", "Writer", "ZipFile", "JarFile", "DriverManager",
284             "Connection", "Statement", "Files" };
285 
286     /*
287      * (non-Javadoc)
288      *
289      * @see
290      * edu.umd.cs.findbugs.Detector#visitClassContext(edu.umd.cs.findbugs.ba
291      * .ClassContext)
292      */
293     @Override
294     public void visitClassContext(ClassContext classContext) {
295         JavaClass jclass = classContext.getJavaClass();
296 
297         // Check to see if the class references any other classes
298         // which could be resources we want to track.
299         // If we don't find any such classes, we skip analyzing
300         // the class. (Note: could do this by method.)
301         boolean sawResourceClass = false;
302         for (int i = 0; i < jclass.getConstantPool().getLength(); ++i) {
303             Constant constant = jclass.getConstantPool().getConstant(i);
304             String className = null;
305             if (constant instanceof ConstantMethodref) {
306                 ConstantMethodref cmr = (ConstantMethodref) constant;
307 
308                 int classIndex = cmr.getClassIndex();
309                 className = jclass.getConstantPool().getConstantString(classIndex, Const.CONSTANT_Class);
310             } else if (constant instanceof ConstantInterfaceMethodref) {
311                 ConstantInterfaceMethodref cmr = (ConstantInterfaceMethodref) constant;
312 
313                 int classIndex = cmr.getClassIndex();
314                 className = jclass.getConstantPool().getConstantString(classIndex, Const.CONSTANT_Class);
315             }
316 
317             if (className != null) {
318                 if (DEBUG) {
319                     System.out.println("FindOpenStream: saw class " + className);
320                 }
321 
322                 for (String aPRESCREEN_CLASS_LIST : PRESCREEN_CLASS_LIST) {
323                     if (className.indexOf(aPRESCREEN_CLASS_LIST) >= 0) {
324                         sawResourceClass = true;
325                         break;
326                     }
327                 }
328             }
329 
330         }
331 
332         if (sawResourceClass) {
333             super.visitClassContext(classContext);
334         }
335     }

337     @Override
338     public boolean prescreen(ClassContext classContext, Method method, boolean mightClose) {
339         BitSet bytecodeSet = classContext.getBytecodeSet(method);
340         if (bytecodeSet == null) {
341             return false;
342         }
343         return bytecodeSet.get(Const.NEW) || bytecodeSet.get(Const.INVOKEINTERFACE)
344                 || bytecodeSet.get(Const.INVOKESPECIAL) || bytecodeSet.get(Const.INVOKESTATIC)
345                 || bytecodeSet.get(Const.INVOKEVIRTUAL);
346     }

353     public static boolean isMainMethod(Method method) {
354         return method.isStatic() && "main".equals(method.getName()) && "([Ljava/lang/String;)V".equals(method.getSignature());
355     }

357     @Override
358     public void analyzeMethod(ClassContext classContext, Method method, StreamResourceTracker resourceTracker,
359             ResourceCollection<Stream> resourceCollection) throws CFGBuilderException, DataflowAnalysisException {
360 
361         potentialOpenStreamList.clear();
362 
363         JavaClass javaClass = classContext.getJavaClass();
364         MethodGen methodGen = classContext.getMethodGen(method);
365         if (methodGen == null) {
366             return;
367         }
368         CFG cfg = classContext.getCFG(method);
369 
370         // Add Streams passed into the method as parameters.
371         // These are uninteresting, and should poison
372         // any streams which wrap them.
373         try {
374             Type[] parameterTypeList = Type.getArgumentTypes(methodGen.getSignature());
375             Location firstLocation = new Location(cfg.getEntry().getFirstInstruction(), cfg.getEntry());
376 
377             int local = methodGen.isStatic() ? 0 : 1;
378 
379             for (Type type : parameterTypeList) {
380                 if (type instanceof ObjectType) {
381                     ObjectType objectType = (ObjectType) type;
382 
383                     for (ObjectType streamBase : streamBaseList) {
384                         if (Hierarchy.isSubtype(objectType, streamBase)) {
385                             // OK, found a parameter that is a resource.
386                             // Create a Stream object to represent it.
387                             // The Stream will be uninteresting, so it will
388                             // inhibit reporting for any stream that wraps it.
389                             Stream paramStream = new Stream(firstLocation, objectType.getClassName(), streamBase.getClassName());
390                             paramStream.setIsOpenOnCreation(true);
391                             paramStream.setOpenLocation(firstLocation);
392                             paramStream.setInstanceParam(local);
393                             resourceCollection.addPreexistingResource(paramStream);
394 
395                             break;
396                         }
397                     }
398                 }
399 
400                 switch (type.getType()) {
401                 case Const.T_LONG:
402                 case Const.T_DOUBLE:
403                     local += 2;
404                     break;
405                 default:
406                     local += 1;
407                     break;
408                 }
409             }
410         } catch (ClassNotFoundException e) {
411             bugReporter.reportMissingClass(e);
412         }
413 
414         // Set precomputed map of Locations to Stream creation points.
415         // That way, the StreamResourceTracker won't have to
416         // repeatedly try to figure out where Streams are created.
417         resourceTracker.setResourceCollection(resourceCollection);
418 
419         super.analyzeMethod(classContext, method, resourceTracker, resourceCollection);
420 
421         // Compute streams that escape into other streams:
422         // this takes wrapper streams into account.
423         // This will also compute equivalence classes of streams,
424         // so that if one stream in a class is closed,
425         // they are all considered closed.
426         // (FIXME: this is too simplistic, especially if buffering
427         // is involved. Sometime we should really think harder
428         // about how this should work.)
429         resourceTracker.markTransitiveUninterestingStreamEscapes();
430 
431         // For each stream closed on all paths, mark its equivalence
432         // class as being closed.
433         for (Iterator<Stream> i = resourceCollection.resourceIterator(); i.hasNext();) {
434             Stream stream = i.next();
435             StreamEquivalenceClass equivalenceClass = resourceTracker.getStreamEquivalenceClass(stream);
436             if (stream.isClosed()) {
437                 equivalenceClass.setClosed();
438             }
439         }
440 
441         // Iterate through potential open streams, reporting warnings
442         // for the "interesting" streams that haven't been closed
443         // (and aren't in an equivalence class with another stream
444         // that was closed).
445         for (PotentialOpenStream pos : potentialOpenStreamList) {
446             Stream stream = pos.stream;
447             if (stream.isClosed()) {
448                 // Stream was in an equivalence class with another
449                 // stream that was properly closed.
450                 continue;
451             }
452 
453             if (stream.isUninteresting()) {
454                 continue;
455             }
456 
457             Location openLocation = stream.getOpenLocation();
458             if (openLocation == null) {
459                 continue;
460             }
461 
462             if (IGNORE_WRAPPED_UNINTERESTING_STREAMS && resourceTracker.isUninterestingStreamEscape(stream)) {
463                 continue;
464             }
465 
466             String sourceFile = javaClass.getSourceFileName();
467             String leakClass = stream.getStreamBase();
468             if (isMainMethod(method) && (leakClass.contains("InputStream") || leakClass.contains("Reader"))) {
469                 return;
470             }
471 
472             bugAccumulator.accumulateBug(new BugInstance(this, pos.bugType, pos.priority)
473             .addClassAndMethod(methodGen, sourceFile).addTypeOfNamedClass(leakClass)
474             .describe(TypeAnnotation.CLOSEIT_ROLE), SourceLineAnnotation.fromVisitedInstruction(classContext, methodGen,
475                     sourceFile, stream.getLocation().getHandle()));
476         }
477     }

479     @Override
480     public void inspectResult(ClassContext classContext, MethodGen methodGen, CFG cfg,
481             Dataflow<ResourceValueFrame, ResourceValueAnalysis<Stream>> dataflow, Stream stream) {
482 
483         if (DEBUG) {
484             System.out.printf("Result for %s in %s%n", stream, methodGen);
485             dataflow.dumpDataflow(dataflow.getAnalysis());
486 
487         }
488         ResourceValueFrame exitFrame = dataflow.getResultFact(cfg.getExit());
489 
490         int exitStatus = exitFrame.getStatus();
491         if (exitStatus == ResourceValueFrame.OPEN || exitStatus == ResourceValueFrame.OPEN_ON_EXCEPTION_PATH) {
492 
493             // FIXME: Stream object should be queried for the
494             // priority.
495 
496             String bugType = stream.getBugType();
497             int priority = NORMAL_PRIORITY;
498             if (exitStatus == ResourceValueFrame.OPEN_ON_EXCEPTION_PATH) {
499                 bugType += "_EXCEPTION_PATH";
500                 priority = LOW_PRIORITY;
501             }
502 
503             potentialOpenStreamList.add(new PotentialOpenStream(bugType, priority, stream));
504         } else if (exitStatus == ResourceValueFrame.CLOSED) {
505             // Remember that this stream was closed on all paths.
506             // Later, we will mark all of the streams in its equivalence class
507             // as having been closed.
508             stream.setClosed();
509         }
510     }

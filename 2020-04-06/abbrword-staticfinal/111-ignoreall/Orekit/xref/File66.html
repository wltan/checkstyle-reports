42      @Test
43      public void testLeap2006() throws OrekitException {
44          final UTCScale utc = TimeScalesFactory.getUTC();
45          AbsoluteDate leapDate =
46              new AbsoluteDate(new DateComponents(2006, 1, 1), TimeComponents.H00, utc);
47          AbsoluteDate d1 = leapDate.shiftedBy(-1);
48          AbsoluteDate d2 = leapDate.shiftedBy(+1);
49          Assert.assertEquals(2.0, d2.durationFrom(d1), 1.0e-10);
50  
51          AbsoluteDate d3 = new AbsoluteDate(new DateComponents(2006, 1, 1),
52                                             new TimeComponents(02, 59, 59),
53                                             glonass);
54          Assert.assertEquals(new AbsoluteDate(new DateComponents(2005, 12, 31),
55                                               new TimeComponents(23, 59, 59),
56                                               utc),
57                              d3);
58          AbsoluteDate d4 = new AbsoluteDate(new DateComponents(2006, 1, 1),
59                                             new TimeComponents(3, 0, 1),
60                                             glonass);
61          Assert.assertEquals(new AbsoluteDate(new DateComponents(2006, 1, 1),
62                                               new TimeComponents(0, 0, 1),
63                                               utc),
64                              d4);
65          Assert.assertEquals(3.0, d4.durationFrom(d3), 1.0e-10);
66      }

68      @Test
69      public void testDuringLeap() throws OrekitException {
70          AbsoluteDate d = new AbsoluteDate(new DateComponents(1983, 06, 30),
71                                            new TimeComponents(23, 59, 59),
72                                            TimeScalesFactory.getUTC());
73          Assert.assertEquals("1983-07-01T02:58:59.000", d.shiftedBy(-60).toString(glonass));
74          Assert.assertEquals(60, glonass.minuteDuration(d.shiftedBy(-60)));
75          Assert.assertFalse(glonass.insideLeap(d.shiftedBy(-60)));
76          Assert.assertEquals("1983-07-01T02:59:59.000", d.toString(glonass));
77          Assert.assertEquals(61, glonass.minuteDuration(d));
78          Assert.assertFalse(glonass.insideLeap(d));
79          d = d.shiftedBy(0.251);
80          Assert.assertEquals("1983-07-01T02:59:59.251", d.toString(glonass));
81          Assert.assertEquals(61, glonass.minuteDuration(d));
82          Assert.assertFalse(glonass.insideLeap(d));
83          d = d.shiftedBy(0.251);
84          Assert.assertEquals("1983-07-01T02:59:59.502", d.toString(glonass));
85          Assert.assertEquals(61, glonass.minuteDuration(d));
86          Assert.assertFalse(glonass.insideLeap(d));
87          d = d.shiftedBy(0.251);
88          Assert.assertEquals("1983-07-01T02:59:59.753", d.toString(glonass));
89          Assert.assertEquals(61, glonass.minuteDuration(d));
90          Assert.assertFalse(glonass.insideLeap(d));
91          d = d.shiftedBy( 0.251);
92          Assert.assertEquals("1983-07-01T02:59:60.004", d.toString(glonass));
93          Assert.assertEquals(61, glonass.minuteDuration(d));
94          Assert.assertTrue(glonass.insideLeap(d));
95          d = d.shiftedBy(0.251);
96          Assert.assertEquals("1983-07-01T02:59:60.255", d.toString(glonass));
97          Assert.assertEquals(61, glonass.minuteDuration(d));
98          Assert.assertTrue(glonass.insideLeap(d));
99          d = d.shiftedBy(0.251);
100         Assert.assertEquals("1983-07-01T02:59:60.506", d.toString(glonass));
101         Assert.assertEquals(61, glonass.minuteDuration(d));
102         Assert.assertTrue(glonass.insideLeap(d));
103         d = d.shiftedBy(0.251);
104         Assert.assertEquals("1983-07-01T02:59:60.757", d.toString(glonass));
105         Assert.assertEquals(61, glonass.minuteDuration(d));
106         Assert.assertTrue(glonass.insideLeap(d));
107         d = d.shiftedBy(0.251);
108         Assert.assertEquals("1983-07-01T03:00:00.008", d.toString(glonass));
109         Assert.assertEquals(60, glonass.minuteDuration(d));
110         Assert.assertFalse(glonass.insideLeap(d));
111     }

113     @Test
114     public void testSymmetry() {
115         for (double dt = -10000; dt < 10000; dt += 123.456789) {
116             AbsoluteDate date = AbsoluteDate.J2000_EPOCH.shiftedBy(dt * Constants.JULIAN_DAY);
117             double dt1 = glonass.offsetFromTAI(date);
118             DateTimeComponents components = date.getComponents(glonass);
119             double dt2 = glonass.offsetToTAI(components.getDate(), components.getTime());
120             Assert.assertEquals( 0.0, dt1 + dt2, 1.0e-10);
121         }
122     }

126         AbsoluteDate t = new AbsoluteDate("2015-07-01T02:59:59.999999", glonass);
127         Assert.assertEquals("2015-07-01T02:59:60.000", t.toString(glonass));
128     }

130     @Test
131     public void testMinuteDuration() {
132         final AbsoluteDate t0 = new AbsoluteDate("1983-07-01T02:58:59.000", glonass);
133         for (double dt = 0; dt < 63; dt += 0.3) {
134             if (dt < 1.0) {
135                 // before the minute of the leap
136                 Assert.assertEquals(60, glonass.minuteDuration(t0.shiftedBy(dt)));
137             } else if (dt < 62.0) {
138                 // during the minute of the leap
139                 Assert.assertEquals(61, glonass.minuteDuration(t0.shiftedBy(dt)));
140             } else {
141                 // after the minute of the leap
142                 Assert.assertEquals(60, glonass.minuteDuration(t0.shiftedBy(dt)));                
143             }
144         }
145     }

47      public void testEquinoxVersusCIO() throws OrekitException {
48          Frame itrfEquinox  = FramesFactory.getITRFEquinox(IERSConventions.IERS_1996, true);
49          Frame itrfCIO      = FramesFactory.getITRF(IERSConventions.IERS_2010, true);
50          AbsoluteDate start = new AbsoluteDate(2011, 4, 10, TimeScalesFactory.getUTC());
51          AbsoluteDate end   = new AbsoluteDate(2011, 7,  4, TimeScalesFactory.getUTC());
52          for (AbsoluteDate date = start; date.compareTo(end) < 0; date = date.shiftedBy(10000)) {

54                      itrfEquinox.getTransformTo(itrfCIO, date).getRotation().getAngle();
55              Assert.assertEquals(0, angularOffset / Constants.ARC_SECONDS_TO_RADIANS, 0.07);
56          }

136     @Test
137     public void testSofaCookbook() throws OrekitException {
138 
139         // SOFA cookbook test case:
140         //     date       2007 April 05, 12h00m00s.0 UTC
141         //     xp         +0′′.0349282
142         //     yp         +0′′.4833163
143         //     UT1 − UTC  -0s.072073685
144         //     dψ 1980    -0′′.0550655
145         //     dε 1980    -0′′.0063580
146         //     dX 2000    +0′′.0001725
147         //     dY 2000    -0′′.0002650
148         //     dX 2006    +0′′.0001750
149         //     dY 2006    -0′′.0002259
150 
151         Utils.setLoaders(IERSConventions.IERS_1996,
152                          Utils.buildEOPList(IERSConventions.IERS_1996, new double[][] {
153                              { 54192, -0.072073685,  1.4020, 0.0349282, 0.4833163, -0.0550666, -0.0063580, Double.NaN, Double.NaN },
154                              { 54193, -0.072073685,  1.4020, 0.0349282, 0.4833163, -0.0550666, -0.0063580, Double.NaN, Double.NaN },
155                              { 54194, -0.072073685,  1.4020, 0.0349282, 0.4833163, -0.0550666, -0.0063580, Double.NaN, Double.NaN },
156                              { 54195, -0.072073685,  1.4020, 0.0349282, 0.4833163, -0.0550666, -0.0063580, Double.NaN, Double.NaN },
157                              { 54196, -0.072073685,  1.4020, 0.0349282, 0.4833163, -0.0550666, -0.0063580, Double.NaN, Double.NaN },
158                              { 54197, -0.072073685,  1.4020, 0.0349282, 0.4833163, -0.0550666, -0.0063580, Double.NaN, Double.NaN },
159                              { 54198, -0.072073685,  1.4020, 0.0349282, 0.4833163, -0.0550666, -0.0063580, Double.NaN, Double.NaN },
160                              { 54199, -0.072073685,  1.4020, 0.0349282, 0.4833163, -0.0550666, -0.0063580, Double.NaN, Double.NaN }
161                          }));
162 
163         EOPHistory eopHistory = FramesFactory.getEOPHistory(IERSConventions.IERS_1996, true);
164 
165         TimeScale utc = TimeScalesFactory.getUTC();
166         TTScale tt    = TimeScalesFactory.getTT();
167         UT1Scale ut1  = TimeScalesFactory.getUT1(eopHistory);
168         Frame gcrf    = FramesFactory.getGCRF();
169         Frame itrf    = FramesFactory.getITRFEquinox(IERSConventions.IERS_1996, true);
170         Frame gtod    = itrf.getParent();
171         Frame tod     = gtod.getParent();
172 
173         // time scales checks
174         AbsoluteDate date = new AbsoluteDate(new DateComponents(2007, 4, 5), TimeComponents.H12, utc);
175         Assert.assertEquals(0.50075444444444,
176                             date.getComponents(tt).getTime().getSecondsInUTCDay() / Constants.JULIAN_DAY,
177                             5.0e-15);
178         Assert.assertEquals(0.499999165813831,
179                             date.getComponents(ut1).getTime().getSecondsInUTCDay() / Constants.JULIAN_DAY,
180                             1.0e-15);
181 
182         // sidereal time check
183         double gast = IERSConventions.IERS_1996.getGASTFunction(ut1, eopHistory).value(date).getValue();
184         Assert.assertEquals(13.412402380740 * 3600 * 1.0e6,
185                             radToMicroAS(MathUtils.normalizeAngle(gast, 0)),
186                             25);
187 
188         // nutation/precession/bias matrix check
189         Rotation refNPB = new Rotation(new double[][] {
190             { +0.999998403176203, -0.001639032970562, -0.000712190961847 },
191             { +0.001639000942243, +0.999998655799521, -0.000045552846624 },
192             { +0.000712264667137, +0.000044385492226, +0.999999745354454 }
193         }, 1.0e-13);
194         Rotation npb = gcrf.getTransformTo(tod, date).getRotation();
195         Assert.assertEquals(0.0, radToMicroAS(Rotation.distance(refNPB, npb)), 27.0);
196 
197         // celestial to terrestrial frames matrix, without polar motion
198         Rotation refWithoutPolarMotion = new Rotation(new double[][] {
199             { +0.973104317592265, +0.230363826166883, -0.000703332813776 },
200             { -0.230363798723533, +0.973104570754697, +0.000120888299841 },
201             { +0.000712264667137, +0.000044385492226, +0.999999745354454 }
202         }, 1.0e-13);
203         Rotation withoutPM = gcrf.getTransformTo(gtod, date).getRotation();
204         Assert.assertEquals(0.0, radToMicroAS(Rotation.distance(refWithoutPolarMotion, withoutPM)), 9);
205 
206         // celestial to terrestrial frames matrix, with polar motion
207         Rotation refWithPolarMotion = new Rotation(new double[][] {
208             { +0.973104317712772, +0.230363826174782, -0.000703163477127 },
209             { -0.230363800391868, +0.973104570648022, +0.000118545116892 },
210             { +0.000711560100206, +0.000046626645796, +0.999999745754058 }
211         }, 1.0e-13);
212         Rotation withPM = gcrf.getTransformTo(itrf, date).getRotation();
213         Assert.assertEquals(0.0, radToMicroAS(Rotation.distance(refWithPolarMotion, withPM)), 10);
214

249     private void checkFrames(Frame frame1, Frame frame2, double toleranceMicroAS)
250         throws OrekitException {
251         AbsoluteDate t0 = new AbsoluteDate(2005, 5, 30, TimeScalesFactory.getUTC());
252         for (double dt = 0; dt < Constants.JULIAN_YEAR; dt += Constants.JULIAN_DAY / 4) {
253             AbsoluteDate date = t0.shiftedBy(dt);
254             Transform t = FramesFactory.getNonInterpolatingTransform(frame1, frame2, date);
255             Vector3D a = t.getRotation().getAxis(RotationConvention.VECTOR_OPERATOR);
256             double delta = FastMath.copySign(radToMicroAS(t.getRotation().getAngle()), a.getZ());
257             Assert.assertEquals(0.0, delta, toleranceMicroAS);
258         }
259     }

266     private void checkPV(PVCoordinates reference, PVCoordinates result,
267                          double expectedPositionError, double expectedVelocityError) {
268 
269         Vector3D dP = result.getPosition().subtract(reference.getPosition());
270         Vector3D dV = result.getVelocity().subtract(reference.getVelocity());
271         Assert.assertEquals(expectedPositionError, dP.getNorm(), 0.01 * expectedPositionError);
272         Assert.assertEquals(expectedVelocityError, dV.getNorm(), 0.01 * expectedVelocityError);
273     }
274 
275     double radToMicroAS(double deltaRad) {
276         return deltaRad * 1.0e6 / Constants.ARC_SECONDS_TO_RADIANS;
277     }

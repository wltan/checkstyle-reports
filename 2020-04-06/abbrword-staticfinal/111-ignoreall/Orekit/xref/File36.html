177     @Test
178     public void testEndianness() throws OrekitException {
179         Utils.setDataRoot("inpop");
180         JPLEphemeridesLoader.EphemerisType type = JPLEphemeridesLoader.EphemerisType.MARS;
181         JPLEphemeridesLoader loaderInpopTCBBig =
182                 new JPLEphemeridesLoader("^inpop.*_TCB_.*_bigendian\\.dat$", type);
183         CelestialBody bodysInpopTCBBig = loaderInpopTCBBig.loadCelestialBody(CelestialBodyFactory.MARS);
184         Assert.assertEquals(1.0, loaderInpopTCBBig.getLoadedConstant("TIMESC"), 1.0e-10);
185         JPLEphemeridesLoader loaderInpopTCBLittle =
186                 new JPLEphemeridesLoader("^inpop.*_TCB_.*_littleendian\\.dat$", type);
187         CelestialBody bodysInpopTCBLittle = loaderInpopTCBLittle.loadCelestialBody(CelestialBodyFactory.MARS);
188         Assert.assertEquals(1.0, loaderInpopTCBLittle.getLoadedConstant("TIMESC"), 1.0e-10);
189         AbsoluteDate t0 = new AbsoluteDate(1969, 7, 17, 10, 43, 23.4, TimeScalesFactory.getTT());
190         Frame eme2000   = FramesFactory.getEME2000();
191         for (double dt = 0; dt < 30 * Constants.JULIAN_DAY; dt += 3600) {
192             AbsoluteDate date        = t0.shiftedBy(dt);
193             Vector3D pInpopTCBBig    = bodysInpopTCBBig.getPVCoordinates(date, eme2000).getPosition();
194             Vector3D pInpopTCBLittle = bodysInpopTCBLittle.getPVCoordinates(date, eme2000).getPosition();
195             Assert.assertEquals(0.0, pInpopTCBBig.distance(pInpopTCBLittle), 1.0e-10);
196         }
197         for (String name : DataProvidersManager.getInstance().getLoadedDataNames()) {
198             Assert.assertTrue(name.contains("inpop"));
199         }
200     }

202     @Test
203     public void testInpopvsJPL() throws OrekitException {
204         Utils.setDataRoot("regular-data:inpop");
205         JPLEphemeridesLoader.EphemerisType type = JPLEphemeridesLoader.EphemerisType.MARS;
206         JPLEphemeridesLoader loaderDE405 =
207                 new JPLEphemeridesLoader("^unxp(\\d\\d\\d\\d)\\.405$", type);
208         CelestialBody bodysDE405 = loaderDE405.loadCelestialBody(CelestialBodyFactory.MARS);
209         JPLEphemeridesLoader loaderInpopTDBBig =
210                 new JPLEphemeridesLoader("^inpop.*_TDB_.*_bigendian\\.dat$", type);
211         CelestialBody bodysInpopTDBBig = loaderInpopTDBBig.loadCelestialBody(CelestialBodyFactory.MARS);
212         Assert.assertEquals(0.0, loaderInpopTDBBig.getLoadedConstant("TIMESC"), 1.0e-10);
213         JPLEphemeridesLoader loaderInpopTCBBig =
214                 new JPLEphemeridesLoader("^inpop.*_TCB_.*_bigendian\\.dat$", type);
215         CelestialBody bodysInpopTCBBig = loaderInpopTCBBig.loadCelestialBody(CelestialBodyFactory.MARS);
216         Assert.assertEquals(1.0, loaderInpopTCBBig.getLoadedConstant("TIMESC"), 1.0e-10);
217         AbsoluteDate t0 = new AbsoluteDate(1969, 7, 17, 10, 43, 23.4, TimeScalesFactory.getTT());
218         Frame eme2000   = FramesFactory.getEME2000();
219         for (double dt = 0; dt < 30 * Constants.JULIAN_DAY; dt += 3600) {
220             AbsoluteDate date = t0.shiftedBy(dt);
221             Vector3D pDE405          = bodysDE405.getPVCoordinates(date, eme2000).getPosition();
222             Vector3D pInpopTDBBig    = bodysInpopTDBBig.getPVCoordinates(date, eme2000).getPosition();
223             Vector3D pInpopTCBBig    = bodysInpopTCBBig.getPVCoordinates(date, eme2000).getPosition();
224             Assert.assertTrue(pDE405.distance(pInpopTDBBig) >  650.0);
225             Assert.assertTrue(pDE405.distance(pInpopTDBBig) < 1050.0);
226             Assert.assertTrue(pDE405.distance(pInpopTCBBig) > 1000.0);
227             Assert.assertTrue(pDE405.distance(pInpopTCBBig) < 2000.0);
228         }

232     @Test
233     public void testOverlappingEphemeridesData() throws OrekitException, IOException {
234         Utils.setDataRoot("overlapping-data/data.zip");
235 
236         // the data root contains two ephemerides files (JPL DE 405), which overlap in the period
237         // (1999-12-23T23:58:55.816, 2000-01-24T23:58:55.815)
238         // this test checks that the data in the overlapping and surrounding range is loaded correctly
239         // from both files (see issue #113).
240 
241         // as the bug only manifests if the DataLoader first loads the ephemerides file containing earlier
242         // data points, the data files are zipped to get a deterministic order when listing files
243 
244         CelestialBody moon = CelestialBodyFactory.getMoon();
245 
246         // 1999/12/31 0h00
247         final AbsoluteDate initDate = new AbsoluteDate(1999, 12, 31, 00, 00, 00, TimeScalesFactory.getUTC());
248         moon.getPVCoordinates(initDate, FramesFactory.getGCRF());
249 
250         // 2000/04/01 0h00
251         final AbsoluteDate otherDate = new AbsoluteDate(2000, 02, 01, 00, 00, 00, TimeScalesFactory.getUTC());
252         moon.getPVCoordinates(otherDate, FramesFactory.getGCRF());
253 
254         // 3 years from initDate
255         AbsoluteDate currentDate = new AbsoluteDate(1999, 12, 01, 00, 00, 00, TimeScalesFactory.getTAI());
256         AbsoluteDate finalDate = new AbsoluteDate(2000, 03, 14, 00, 00, 00, TimeScalesFactory.getTAI());
257 
258         while (currentDate.compareTo(finalDate) < 0)  {
259             currentDate = currentDate.shiftedBy(Constants.JULIAN_DAY);
260             moon.getPVCoordinates(currentDate, FramesFactory.getGCRF());
261         }
262 
263     }

265     private void checkDerivative(String supportedNames, AbsoluteDate date, double maxChunkDuration)
266         throws OrekitException {
267         JPLEphemeridesLoader loader =
268             new JPLEphemeridesLoader(supportedNames, JPLEphemeridesLoader.EphemerisType.MERCURY);
269         CelestialBody body = loader.loadCelestialBody(CelestialBodyFactory.MERCURY);
270         double h = 20;
271 
272         // eight points finite differences estimation of the velocity
273         Frame eme2000 = FramesFactory.getEME2000();
274         Vector3D pm4h = body.getPVCoordinates(date.shiftedBy(-4 * h), eme2000).getPosition();
275         Vector3D pm3h = body.getPVCoordinates(date.shiftedBy(-3 * h), eme2000).getPosition();
276         Vector3D pm2h = body.getPVCoordinates(date.shiftedBy(-2 * h), eme2000).getPosition();
277         Vector3D pm1h = body.getPVCoordinates(date.shiftedBy(    -h), eme2000).getPosition();
278         Vector3D pp1h = body.getPVCoordinates(date.shiftedBy(     h), eme2000).getPosition();
279         Vector3D pp2h = body.getPVCoordinates(date.shiftedBy( 2 * h), eme2000).getPosition();
280         Vector3D pp3h = body.getPVCoordinates(date.shiftedBy( 3 * h), eme2000).getPosition();
281         Vector3D pp4h = body.getPVCoordinates(date.shiftedBy( 4 * h), eme2000).getPosition();
282         Vector3D d4   = pp4h.subtract(pm4h);
283         Vector3D d3   = pp3h.subtract(pm3h);
284         Vector3D d2   = pp2h.subtract(pm2h);
285         Vector3D d1   = pp1h.subtract(pm1h);
286         double c = 1.0 / (840 * h);
287         Vector3D estimatedV = new Vector3D(-3 * c, d4, 32 * c, d3, -168 * c, d2, 672 * c, d1);
288 
289         Vector3D loadedV = body.getPVCoordinates(date, eme2000).getVelocity();
290         Assert.assertEquals(0, loadedV.subtract(estimatedV).getNorm(), 3.5e-10 * loadedV.getNorm());
291         Assert.assertEquals(maxChunkDuration, loader.getMaxChunksDuration(), 1.0e-10);
292     }

92      @Test
93      public void testLieske() throws OrekitException {
94  
95          // the following test values correspond to table 1 in the paper:
96          // Precession Matrix Based on IAU (1976) System of Astronomical Constants,
97          // Jay H. Lieske, Astronomy and Astrophysics, vol. 73, no. 3, Mar. 1979, p. 282-284
98          // http://articles.adsabs.harvard.edu/cgi-bin/nph-iarticle_query?1979A%26A....73..282L&defaultprint=YES&filetype=.pdf
99  
100         // published table, with limited accuracy
101         final double publishedEpsilon = 1.0e-6 * Constants.JULIAN_YEAR;
102         checkEpochs(1899.999142, 1900.000000, publishedEpsilon);
103         checkEpochs(1900.000000, 1900.000858, publishedEpsilon);
104         checkEpochs(1950.000000, 1949.999790, publishedEpsilon);
105         checkEpochs(1950.000210, 1950.000000, publishedEpsilon);
106         checkEpochs(2000.000000, 1999.998722, publishedEpsilon);
107         checkEpochs(2000.001278, 2000.000000, publishedEpsilon);
108 
109         // recomputed table, using directly Lieske formulas (i.e. *not* Orekit implementation) with high accuracy
110         final double accurateEpsilon = 1.2e-13 * Constants.JULIAN_YEAR;
111         checkEpochs(1899.99914161068724704, 1900.00000000000000000, accurateEpsilon);
112         checkEpochs(1900.00000000000000000, 1900.00085837097878165, accurateEpsilon);
113         checkEpochs(1950.00000000000000000, 1949.99979044229979466, accurateEpsilon);
114         checkEpochs(1950.00020956217615449, 1950.00000000000000000, accurateEpsilon);
115         checkEpochs(2000.00000000000000000, 1999.99872251362080766, accurateEpsilon);
116         checkEpochs(2000.00127751366506194, 2000.00000000000000000, accurateEpsilon);
117 
118     }

120     private void checkEpochs(final double besselianEpoch, final double julianEpoch, final double epsilon) {
121         final AbsoluteDate b = AbsoluteDate.createBesselianEpoch(besselianEpoch);
122         final AbsoluteDate j = AbsoluteDate.createJulianEpoch(julianEpoch);
123         Assert.assertEquals(0.0, b.durationFrom(j), epsilon);
124     }

188             Assert.assertEquals(minutesFromUTC, dtcInsideLeap.getTime().getMinutesFromUTC());
189             Assert.assertEquals(59.8, dtcBeforeLeap.getTime().getSecond(), 1.0e-10);
190             Assert.assertEquals(60.5, dtcInsideLeap.getTime().getSecond(), 1.0e-10);

341     @Test
342     public void testBeforeAndAfterLeap() {
343         final TimeScale tai = TimeScalesFactory.getTAI();
344         AbsoluteDate leapStart = new AbsoluteDate(1977,  1,  1,  0,  0, 14, tai);
345         AbsoluteDate leapEnd   = new AbsoluteDate(1977,  1,  1,  0,  0, 16, tai);
346         for (int i = -10; i < 10; ++i) {
347             final double dt = 1.1 * (2 * i - 1);
348             AbsoluteDate d1 = leapStart.shiftedBy(dt);
349             AbsoluteDate d2 = new AbsoluteDate(leapStart, dt, tai);
350             AbsoluteDate d3 = new AbsoluteDate(leapStart, dt, utc);
351             AbsoluteDate d4 = new AbsoluteDate(leapEnd,   dt, tai);
352             AbsoluteDate d5 = new AbsoluteDate(leapEnd,   dt, utc);
353             Assert.assertTrue(FastMath.abs(d1.durationFrom(d2)) < 1.0e-10);
354             if (dt < 0) {
355                 Assert.assertTrue(FastMath.abs(d2.durationFrom(d3)) < 1.0e-10);
356                 Assert.assertTrue(d4.durationFrom(d5) > (1.0 - 1.0e-10));
357             } else {
358                 Assert.assertTrue(d2.durationFrom(d3) < (-1.0 + 1.0e-10));
359                 Assert.assertTrue(FastMath.abs(d4.durationFrom(d5)) < 1.0e-10);
360             }
361         }

463     @Test
464     public void testCCSDSUnsegmentedWithExtendedPreamble() throws OrekitException {
465 
466         AbsoluteDate reference = new AbsoluteDate("2095-03-03T22:02:45.789012345678901", utc);
467         int leap = (int) FastMath.rint(utc.offsetFromTAI(reference));
468         double lsb = FastMath.pow(2.0, -48);
469 
470         byte extendedPreamble = (byte) -0x80;
471         byte identification   = (byte)  0x10;
472         byte coarseLength1    = (byte)  0x0C; // four (3 + 1) bytes
473         byte fineLength1      = (byte)  0x03; // 3 bytes
474         byte coarseLength2    = (byte)  0x20; // 1 additional byte for coarse time
475         byte fineLength2      = (byte)  0x0C; // 3 additional bytes for fine time
476         byte[] timeCCSDSEpoch = new byte[] {

643     @Test
644     public void testIterationAccuracy() {
645 
646         final TimeScale tai = TimeScalesFactory.getTAI();
647         final AbsoluteDate t0 = new AbsoluteDate(2010, 6, 21, 18, 42, 0.281, tai);
648 
649         // 0.1 is not representable exactly in double precision
650         // we will accumulate error, between -0.5ULP and -3ULP at each iteration
651         checkIteration(0.1, t0, 10000, 3.0, -1.19, 1.0e-4);
652 
653         // 0.125 is representable exactly in double precision
654         // error will be null
655         checkIteration(0.125, t0, 10000, 1.0e-15, 0.0, 1.0e-15);
656 
657     }
658 
659     private void checkIteration(final double step, final AbsoluteDate t0, final int nMax,
660                                 final double maxErrorFactor,
661                                 final double expectedMean, final double meanTolerance) {
662         final double epsilon = FastMath.ulp(step);
663         AbsoluteDate iteratedDate = t0;
664         double mean = 0;
665         for (int i = 1; i < nMax; ++i) {
666             iteratedDate = iteratedDate.shiftedBy(step);
667             AbsoluteDate directDate = t0.shiftedBy(i * step);
668             final double error = iteratedDate.durationFrom(directDate);
669             mean += error / (i * epsilon);
670             Assert.assertEquals(0.0, iteratedDate.durationFrom(directDate), maxErrorFactor * i * epsilon);
671         }
672         mean /= nMax;
673         Assert.assertEquals(expectedMean, mean, meanTolerance);
674     }

17  package org.springframework.integration.channel;
18  
19  import org.springframework.integration.context.IntegrationContextUtils;
20  import org.springframework.integration.core.ErrorMessagePublisher;
21  import org.springframework.integration.support.ErrorMessageStrategy;
22  import org.springframework.integration.support.MessagingExceptionWrapper;
23  import org.springframework.lang.Nullable;
24  import org.springframework.messaging.Message;
25  import org.springframework.messaging.MessageChannel;
26  import org.springframework.messaging.MessagingException;
27  import org.springframework.messaging.core.DestinationResolver;
28  import org.springframework.messaging.support.ErrorMessage;
29  import org.springframework.util.Assert;
30  import org.springframework.util.ErrorHandler;
31  import org.springframework.util.StringUtils;
32  
33  /**
34   * {@link ErrorHandler} implementation that sends an {@link ErrorMessage} to a
35   * {@link MessageChannel}.
36   *
37   * @author Mark Fisher
38   * @author Iwein Fuld
39   * @author Oleg Zhurakousky
40   * @author Gary Russell
41   * @author Artem Bilan
42   */
43  public class MessagePublishingErrorHandler extends ErrorMessagePublisher implements ErrorHandler {
44  
45  	private static final int DEFAULT_SEND_TIMEOUT = 1000;
46  
47  	private static final ErrorMessageStrategy DEFAULT_ERROR_MESSAGE_STRATEGY = (ex, attrs) -> {
48  		if (ex instanceof MessagingExceptionWrapper) {
49  			return new ErrorMessage(ex.getCause(), ((MessagingExceptionWrapper) ex).getFailedMessage());
50  		}
51  		else {
52  			return new ErrorMessage(ex);
53  		}
54  	};
55  
56  	public MessagePublishingErrorHandler() {
57  		setErrorMessageStrategy(DEFAULT_ERROR_MESSAGE_STRATEGY);
58  		setSendTimeout(DEFAULT_SEND_TIMEOUT);
59  	}
60  
61  	public MessagePublishingErrorHandler(DestinationResolver<MessageChannel> channelResolver) {
62  		this();
63  		setChannelResolver(channelResolver);
64  	}
65  
66  
67  	public void setDefaultErrorChannel(@Nullable MessageChannel defaultErrorChannel) {
68  		setChannel(defaultErrorChannel);
69  	}
70  
71  	/**
72  	 * Return the default error channel for this error handler.
73  	 * @return the error channel.
74  	 * @since 4.3
75  	 */
76  	@Nullable
77  	public MessageChannel getDefaultErrorChannel() {
78  		return getChannel();
79  	}
80  
81  	/**
82  	 * Specify the bean name of default error channel for this error handler.
83  	 * @param defaultErrorChannelName the bean name of the error channel
84  	 * @since 4.3.3
85  	 */
86  	public void setDefaultErrorChannelName(String defaultErrorChannelName) {
87  		setChannelName(defaultErrorChannelName);
88  	}
89  
90  	@Override
91  	public final void handleError(Throwable ex) {
92  		MessageChannel errorChannel = resolveErrorChannel(ex);
93  		boolean sent = false;
94  		if (errorChannel != null) {
95  			try {
96  				getMessagingTemplate().send(errorChannel, getErrorMessageStrategy().buildErrorMessage(ex, null));
97  				sent = true;
98  			}
99  			catch (Exception errorDeliveryError) {
100 				// message will be logged only
101 				if (this.logger.isWarnEnabled()) {
102 					this.logger.warn("Error message was not delivered.", errorDeliveryError);
103 				}
104 			}
105 		}
106 		if (!sent && this.logger.isErrorEnabled()) {
107 			Message<?> failedMessage =
108 					ex instanceof MessagingException
109 							? ((MessagingException) ex).getFailedMessage()
110 							: null;
111 
112 			this.logger.error("failure occurred in messaging task" +
113 					(failedMessage != null ? " with message: " + failedMessage : ""), ex);
114 		}
115 	}
116 
117 	@Nullable
118 	private MessageChannel resolveErrorChannel(Throwable t) {
119 		DestinationResolver<MessageChannel> channelResolver = getChannelResolver();
120 		Throwable actualThrowable = t;
121 		if (t instanceof MessagingExceptionWrapper) {
122 			actualThrowable = t.getCause();
123 		}
124 		Message<?> failedMessage =
125 				actualThrowable instanceof MessagingException
126 						? ((MessagingException) actualThrowable).getFailedMessage()
127 						: null;
128 		if (getDefaultErrorChannel() == null && channelResolver != null) {
129 			setChannel(channelResolver.resolveDestination(IntegrationContextUtils.ERROR_CHANNEL_BEAN_NAME));
130 		}
131 
132 		if (failedMessage != null && failedMessage.getHeaders().getErrorChannel() != null) {
133 			Object errorChannelHeader = failedMessage.getHeaders().getErrorChannel();
134 			if (errorChannelHeader instanceof MessageChannel) {
135 				return (MessageChannel) errorChannelHeader;
136 			}
137 			Assert.isInstanceOf(String.class, errorChannelHeader, () ->
138 					"Unsupported error channel header type. Expected MessageChannel or String, but actual type is [" +
139 							errorChannelHeader.getClass() + "]");
140 			if (channelResolver != null && StringUtils.hasText((String) errorChannelHeader)) {
141 				return channelResolver.resolveDestination((String) errorChannelHeader);
142 			}
143 		}
144 
145 		return getDefaultErrorChannel();
146 	}
147 
148 }

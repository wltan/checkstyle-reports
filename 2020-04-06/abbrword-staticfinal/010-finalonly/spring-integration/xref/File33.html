61  /**
62   * The {@link FactoryBean} implementation for {@link AbstractEndpoint} population.
63   * Controls all the necessary properties and lifecycle.
64   * According the provided {@link MessageChannel} implementation populates
65   * a {@link PollingConsumer} for the {@link PollableChannel},
66   * an {@link EventDrivenConsumer} for the {@link SubscribableChannel}
67   * and {@link ReactiveStreamsConsumer} for all other channel implementations.
68   *
69   * @author Mark Fisher
70   * @author Oleg Zhurakousky
71   * @author Josh Long
72   * @author Gary Russell
73   * @author Artem Bilan
74   */
75  public class ConsumerEndpointFactoryBean
76  		implements FactoryBean<AbstractEndpoint>, BeanFactoryAware, BeanNameAware, BeanClassLoaderAware,
77  		InitializingBean, SmartLifecycle, DisposableBean {

117 	public void setHandler(Object handler) {
118 		Assert.isTrue(handler instanceof MessageHandler || handler instanceof ReactiveMessageHandler,
119 				"'handler' must be an instance of 'MessageHandler' or 'ReactiveMessageHandler'");
120 		synchronized (this.handlerMonitor) {
121 			Assert.isNull(this.handler, "handler cannot be overridden");
122 			if (handler instanceof ReactiveMessageHandler) {
123 				this.handler = new ReactiveMessageHandlerAdapter((ReactiveMessageHandler) handler);
124 			}
125 			else {
126 				this.handler = (MessageHandler) handler;
127 			}
128 		}
129 	}

192 	@Override
193 	public void afterPropertiesSet() {
194 		if (this.beanName == null) {
195 			LOGGER.error("The MessageHandler [" + this.handler + "] will be created without a 'componentName'. " +
196 					"Consider specifying the 'beanName' property on this ConsumerEndpointFactoryBean.");
197 		}
198 		else {
199 			populateComponentNameIfAny();
200 		}
201 
202 		if (!(this.handler instanceof ReactiveMessageHandlerAdapter)) {
203 			this.handler = adviceChain(this.handler);
204 		}
205 		else if (!CollectionUtils.isEmpty(this.adviceChain)) {
206 			this.handler =
207 					new ReactiveMessageHandlerAdapter(
208 							adviceChain(((ReactiveMessageHandlerAdapter) this.handler).getDelegate()));
209 		}
210 		if (this.channelResolver == null) {
211 			this.channelResolver = ChannelResolverUtils.getChannelResolver(this.beanFactory);
212 		}
213 		initializeEndpoint();
214 	}
215 
216 	private void populateComponentNameIfAny() {
217 		try {
218 			if (!this.beanName.startsWith("org.springframework")) {
219 				MessageHandler targetHandler = this.handler;
220 				if (AopUtils.isAopProxy(targetHandler)) {
221 					Object target = ((Advised) targetHandler).getTargetSource().getTarget();
222 					if (target instanceof MessageHandler) {
223 						targetHandler = (MessageHandler) target;
224 					}
225 				}
226 				if (targetHandler instanceof IntegrationObjectSupport) {
227 					((IntegrationObjectSupport) targetHandler).setComponentName(this.beanName);
228 				}
229 			}
230 		}
231 		catch (Exception e) {
232 			if (LOGGER.isDebugEnabled()) {
233 				LOGGER.debug("Could not set component name for handler "
234 						+ this.handler + " for " + this.beanName + " :" + e.getMessage());
235 			}
236 		}
237 	}

239 	@SuppressWarnings("unchecked")
240 	private <H> H adviceChain(H handler) {
241 		H theHandler = handler;
242 		if (!CollectionUtils.isEmpty(this.adviceChain)) {
243 			/*
244 			 *  ARPMHs advise the handleRequestMessage method internally and already have the advice chain injected.
245 			 *  So we only advise handlers that are not reply-producing.
246 			 *  Or if one (or more) of advices is IdempotentReceiverInterceptor.
247 			 *  If the handler is already advised,
248 			 *  add the configured advices to its chain, otherwise create a proxy.
249 			 */
250 			Class<?> targetClass = AopUtils.getTargetClass(theHandler);
251 			boolean replyMessageHandler = AbstractReplyProducingMessageHandler.class.isAssignableFrom(targetClass);
252 
253 			for (Advice advice : this.adviceChain) {
254 				if (!replyMessageHandler || advice instanceof HandleMessageAdvice) {
255 					NameMatchMethodPointcutAdvisor handlerAdvice = new NameMatchMethodPointcutAdvisor(advice);
256 					handlerAdvice.addMethodName("handleMessage");
257 					if (theHandler instanceof Advised) {
258 						((Advised) theHandler).addAdvisor(handlerAdvice);
259 					}
260 					else {
261 						ProxyFactory proxyFactory = new ProxyFactory(theHandler);
262 						proxyFactory.addAdvisor(handlerAdvice);
263 						theHandler = (H) proxyFactory.getProxy(this.beanClassLoader);
264 					}
265 				}
266 			}
267 		}
268 		return theHandler;
269 	}

287 	private void initializeEndpoint() {
288 		synchronized (this.initializationMonitor) {
289 			if (this.initialized) {
290 				return;
291 			}
292 			MessageChannel channel = null;
293 			if (StringUtils.hasText(this.inputChannelName)) {
294 				channel = this.channelResolver.resolveDestination(this.inputChannelName);
295 			}
296 			if (this.inputChannel != null) {
297 				channel = this.inputChannel;
298 			}
299 			Assert.state(channel != null, "one of inputChannelName or inputChannel is required");
300 			if (channel instanceof SubscribableChannel) {
301 				eventDrivenConsumer(channel);
302 			}
303 			else if (channel instanceof PollableChannel) {
304 				pollingConsumer(channel);
305 			}
306 			else {
307 				if (this.handler instanceof ReactiveMessageHandlerAdapter) {
308 					this.endpoint = new ReactiveStreamsConsumer(channel,
309 							((ReactiveMessageHandlerAdapter) this.handler).getDelegate());
310 				}
311 				else {
312 					this.endpoint = new ReactiveStreamsConsumer(channel, this.handler);
313 				}
314 			}
315 			this.endpoint.setBeanName(this.beanName);
316 			this.endpoint.setBeanFactory(this.beanFactory);
317 			smartLifecycle();
318 			this.endpoint.setRole(this.role);
319 			if (this.taskScheduler != null) {
320 				this.endpoint.setTaskScheduler(this.taskScheduler);
321 			}
322 			this.endpoint.afterPropertiesSet();
323 			this.initialized = true;
324 		}
325 	}

327 	private void eventDrivenConsumer(MessageChannel channel) {
328 		Assert.isNull(this.pollerMetadata,
329 				() -> "A poller should not be specified for endpoint '" + this.beanName
330 						+ "', since '" + channel + "' is a SubscribableChannel (not pollable).");
331 		this.endpoint = new EventDrivenConsumer((SubscribableChannel) channel, this.handler);
332 		if (LOGGER.isWarnEnabled()
333 				&& Boolean.FALSE.equals(this.autoStartup)
334 				&& channel instanceof FixedSubscriberChannel) {
335 			LOGGER.warn("'autoStartup=\"false\"' has no effect when using a FixedSubscriberChannel");
336 		}
337 	}
338 
339 	private void pollingConsumer(MessageChannel channel) {
340 		PollingConsumer pollingConsumer = new PollingConsumer((PollableChannel) channel, this.handler);
341 		if (this.pollerMetadata == null) {
342 			this.pollerMetadata = PollerMetadata.getDefaultPollerMetadata(this.beanFactory);
343 			Assert.notNull(this.pollerMetadata,
344 					() -> "No poller has been defined for endpoint '" + this.beanName
345 							+ "', and no default poller is available within the context.");
346 		}
347 		pollingConsumer.setTaskExecutor(this.pollerMetadata.getTaskExecutor());
348 		pollingConsumer.setTrigger(this.pollerMetadata.getTrigger());
349 		pollingConsumer.setAdviceChain(this.pollerMetadata.getAdviceChain());
350 		pollingConsumer.setMaxMessagesPerPoll(this.pollerMetadata.getMaxMessagesPerPoll());
351 
352 		pollingConsumer.setErrorHandler(this.pollerMetadata.getErrorHandler());
353 
354 		pollingConsumer.setReceiveTimeout(this.pollerMetadata.getReceiveTimeout());
355 		pollingConsumer.setTransactionSynchronizationFactory(
356 				this.pollerMetadata.getTransactionSynchronizationFactory());
357 		pollingConsumer.setBeanClassLoader(this.beanClassLoader);
358 		pollingConsumer.setBeanFactory(this.beanFactory);
359 		this.endpoint = pollingConsumer;
360 	}
361 
362 	private void smartLifecycle() {
363 		if (this.autoStartup != null) {
364 			this.endpoint.setAutoStartup(this.autoStartup);
365 		}
366 		int phaseToSet = this.phase;
367 		if (!this.isPhaseSet) {
368 			if (this.endpoint instanceof PollingConsumer) {
369 				phaseToSet = Integer.MAX_VALUE / 2;
370 			}
371 			else {
372 				phaseToSet = Integer.MIN_VALUE;
373 			}
374 		}
375 
376 		this.endpoint.setPhase(phaseToSet);
377 	}

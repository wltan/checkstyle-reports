37  /**
38   * Default implementation of {@link JmsHeaderMapper}.
39   * <p>
40   * This implementation copies JMS API headers (e.g. JMSReplyTo) to and from
41   * Spring Integration Messages. Any user-defined properties will also be copied
42   * from a JMS Message to a Spring Integration Message, and any other headers
43   * on a Spring Integration Message (beyond the JMS API headers) will likewise
44   * be copied to a JMS Message. Those other headers will be copied to the
45   * general properties of a JMS Message whereas the JMS API headers are passed
46   * to the appropriate setter methods (e.g. setJMSReplyTo).
47   * <p>
48   * Constants for the JMS API headers are defined in {@link JmsHeaders}.
49   * Note that the JMSMessageID and JMSRedelivered flag are only copied
50   * <em>from</em> a JMS Message. Those values will <em>not</em> be passed
51   * along from a Spring Integration Message to an outbound JMS Message.
52   *
53   * @author Mark Fisher
54   * @author Gary Russell
55   * @author Artem Bilan
56   */
57  public class DefaultJmsHeaderMapper extends JmsHeaderMapper {
58  
59  	private static final List<Class<?>> SUPPORTED_PROPERTY_TYPES = Arrays.asList(new Class<?>[] {
60  			Boolean.class, Byte.class, Double.class, Float.class, Integer.class, Long.class, Short.class, String.class });
61  
62  
63  	private final Log logger = LogFactory.getLog(this.getClass());

65  	private volatile String inboundPrefix = "";
66  
67  	private volatile String outboundPrefix = "";
68  
69  	private volatile boolean mapInboundPriority = true;
70  
71  	private volatile boolean mapInboundDeliveryMode = false;
72  
73  	private volatile boolean mapInboundExpiration = false;

100 	/**
101 	 * Specify a prefix to be appended to the integration message header name
102 	 * for any JMS property that is being mapped into the MessageHeaders.
103 	 * The Default is an empty string (no prefix).
104 	 * <p>
105 	 * This does not affect the JMS properties covered by the specification/API,
106 	 * such as JMSCorrelationID, etc. The header names used for mapping such
107 	 * properties are all defined in our {@link JmsHeaders}.
108 	 *
109 	 * @param inboundPrefix The inbound prefix.
110 	 */
111 	public void setInboundPrefix(String inboundPrefix) {
112 		this.inboundPrefix = (inboundPrefix != null) ? inboundPrefix : "";
113 	}
114 
115 	/**
116 	 * Specify a prefix to be appended to the JMS property name for any
117 	 * integration message header that is being mapped into the JMS Message.
118 	 * The Default is an empty string (no prefix).
119 	 * <p>
120 	 * This does not affect the JMS properties covered by the specification/API,
121 	 * such as JMSCorrelationID, etc. The header names used for mapping such
122 	 * properties are all defined in our {@link JmsHeaders}.
123 	 *
124 	 * @param outboundPrefix The outbound prefix.
125 	 */
126 	public void setOutboundPrefix(String outboundPrefix) {
127 		this.outboundPrefix = (outboundPrefix != null) ? outboundPrefix : "";
128 	}

130 	@Override
131 	public void fromHeaders(MessageHeaders headers, javax.jms.Message jmsMessage) {
132 		try {
133 			Object jmsCorrelationId = headers.get(JmsHeaders.CORRELATION_ID);
134 			if (jmsCorrelationId instanceof Number) {
135 				jmsCorrelationId = jmsCorrelationId.toString();
136 			}
137 			if (jmsCorrelationId instanceof String) {
138 				try {
139 					jmsMessage.setJMSCorrelationID((String) jmsCorrelationId);
140 				}
141 				catch (Exception e) {
142 					this.logger.info("failed to set JMSCorrelationID, skipping", e);
143 				}
144 			}
145 			Object jmsReplyTo = headers.get(JmsHeaders.REPLY_TO);
146 			if (jmsReplyTo instanceof Destination) {
147 				try {
148 					jmsMessage.setJMSReplyTo((Destination) jmsReplyTo);
149 				}
150 				catch (Exception e) {
151 					this.logger.info("failed to set JMSReplyTo, skipping", e);
152 				}
153 			}
154 			Object jmsType = headers.get(JmsHeaders.TYPE);
155 			if (jmsType instanceof String) {
156 				try {
157 					jmsMessage.setJMSType((String) jmsType);
158 				}
159 				catch (Exception e) {
160 					this.logger.info("failed to set JMSType, skipping", e);
161 				}
162 			}
163 			for (Entry<String, Object> entry : headers.entrySet()) {
164 				String headerName = entry.getKey();
165 				if (StringUtils.hasText(headerName) && !headerName.startsWith(JmsHeaders.PREFIX)
166 						&& jmsMessage.getObjectProperty(headerName) == null) {
167 					Object value = entry.getValue();
168 					if (value != null) {
169 						if (SUPPORTED_PROPERTY_TYPES.contains(value.getClass())) {
170 							try {
171 								String propertyName = this.fromHeaderName(headerName);
172 								jmsMessage.setObjectProperty(propertyName, value);
173 							}
174 							catch (Exception e) {
175 								if (headerName.startsWith("JMSX")
176 										|| headerName.equals(IntegrationMessageHeaderAccessor.PRIORITY)) {
177 									if (this.logger.isTraceEnabled()) {
178 										this.logger.trace("skipping reserved header, it cannot be set by client: "
179 												+ headerName);
180 									}
181 								}
182 								else if (this.logger.isWarnEnabled()) {
183 									this.logger.warn("failed to map Message header '" + headerName + "' to JMS property", e);
184 								}
185 							}
186 						}
187 						else if (IntegrationMessageHeaderAccessor.CORRELATION_ID.equals(headerName)) {
188 							String propertyName = fromHeaderName(headerName);
189 							jmsMessage.setObjectProperty(propertyName, value.toString());
190 						}
191 					}
192 				}
193 			}
194 		}
195 		catch (Exception e) {
196 			if (this.logger.isWarnEnabled()) {
197 				this.logger.warn("error occurred while mapping from MessageHeaders to JMS properties", e);
198 			}
199 		}
200 	}

202 	@Override
203 	public Map<String, Object> toHeaders(javax.jms.Message jmsMessage) {
204 		Map<String, Object> headers = new HashMap<String, Object>();
205 		try {
206 			try {
207 				String messageId = jmsMessage.getJMSMessageID();
208 				if (messageId != null) {
209 					headers.put(JmsHeaders.MESSAGE_ID, messageId);
210 				}
211 			}
212 			catch (Exception e) {
213 				this.logger.info("failed to read JMSMessageID property, skipping", e);
214 			}
215 			try {
216 				Destination destination = jmsMessage.getJMSDestination();
217 				if (destination != null) {
218 					headers.put(JmsHeaders.DESTINATION, destination);
219 				}
220 			}
221 			catch (Exception ex) {
222 				this.logger.info("failed to read JMSDestination property, skipping", ex);
223 			}
224 			try {
225 				String correlationId = jmsMessage.getJMSCorrelationID();
226 				if (correlationId != null) {
227 					headers.put(JmsHeaders.CORRELATION_ID, correlationId);
228 				}
229 			}
230 			catch (Exception e) {
231 				this.logger.info("failed to read JMSCorrelationID property, skipping", e);
232 			}
233 			try {
234 				Destination replyTo = jmsMessage.getJMSReplyTo();
235 				if (replyTo != null) {
236 					headers.put(JmsHeaders.REPLY_TO, replyTo);
237 				}
238 			}
239 			catch (Exception e) {
240 				this.logger.info("failed to read JMSReplyTo property, skipping", e);
241 			}
242 			try {
243 				headers.put(JmsHeaders.REDELIVERED, jmsMessage.getJMSRedelivered());
244 			}
245 			catch (Exception e) {
246 				this.logger.info("failed to read JMSRedelivered property, skipping", e);
247 			}
248 			try {
249 				String type = jmsMessage.getJMSType();
250 				if (type != null) {
251 					headers.put(JmsHeaders.TYPE, type);
252 				}
253 			}
254 			catch (Exception e) {
255 				this.logger.info("failed to read JMSType property, skipping", e);
256 			}
257 			try {
258 				headers.put(JmsHeaders.TIMESTAMP, jmsMessage.getJMSTimestamp());
259 			}
260 			catch (Exception e) {
261 				this.logger.info("failed to read JMSTimestamp property, skipping", e);
262 			}
263 			if (this.mapInboundPriority) {
264 				try {
265 					headers.put(IntegrationMessageHeaderAccessor.PRIORITY, jmsMessage.getJMSPriority());
266 				}
267 				catch (Exception e) {
268 					this.logger.info("failed to read JMSPriority property, skipping", e);
269 				}
270 			}
271 			if (this.mapInboundDeliveryMode) {
272 				try {
273 					headers.put(JmsHeaders.DELIVERY_MODE, jmsMessage.getJMSDeliveryMode());
274 				}
275 				catch (Exception e) {
276 					this.logger.info("failed to read JMSDeliveryMode property, skipping", e);
277 				}
278 			}
279 			if (this.mapInboundExpiration) {
280 				try {
281 					headers.put(JmsHeaders.EXPIRATION, jmsMessage.getJMSExpiration());
282 				}
283 				catch (Exception e) {
284 					this.logger.info("failed to read JMSExpiration property, skipping", e);
285 				}
286 			}
287 			Enumeration<?> jmsPropertyNames = jmsMessage.getPropertyNames();
288 			if (jmsPropertyNames != null) {
289 				while (jmsPropertyNames.hasMoreElements()) {
290 					String propertyName = jmsPropertyNames.nextElement().toString();
291 					try {
292 						String headerName = this.toHeaderName(propertyName);
293 						headers.put(headerName, jmsMessage.getObjectProperty(propertyName));
294 					}
295 					catch (Exception e) {
296 						if (this.logger.isWarnEnabled()) {
297 							this.logger.warn("error occurred while mapping JMS property '"
298 									+ propertyName + "' to Message header", e);
299 						}
300 					}
301 				}
302 			}
303 		}
304 		catch (JMSException e) {
305 			if (this.logger.isWarnEnabled()) {
306 				this.logger.warn("error occurred while mapping from JMS properties to MessageHeaders", e);
307 			}
308 		}
309 		return headers;
310 	}

312 	/**
313 	 * Adds the outbound prefix if necessary.
314 	 * Converts {@link MessageHeaders#CONTENT_TYPE} to content_type for JMS compliance.
315 	 */
316 	private String fromHeaderName(String headerName) {
317 		String propertyName = headerName;
318 		if (StringUtils.hasText(this.outboundPrefix) && !propertyName.startsWith(this.outboundPrefix)) {
319 			propertyName = this.outboundPrefix + headerName;
320 		}
321 		else if (MessageHeaders.CONTENT_TYPE.equals(headerName)) {
322 			propertyName = CONTENT_TYPE_PROPERTY;
323 		}
324 		return propertyName;
325 	}
326 
327 	/**
328 	 * Adds the inbound prefix if necessary.
329 	 * Converts content_type to {@link MessageHeaders#CONTENT_TYPE}.
330 	 */
331 	private String toHeaderName(String propertyName) {
332 		String headerName = propertyName;
333 		if (StringUtils.hasText(this.inboundPrefix) && !headerName.startsWith(this.inboundPrefix)) {
334 			headerName = this.inboundPrefix + propertyName;
335 		}
336 		else if (CONTENT_TYPE_PROPERTY.equals(propertyName)) {
337 			headerName = MessageHeaders.CONTENT_TYPE;
338 		}
339 		return headerName;
340 	}

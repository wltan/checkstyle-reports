40  /**
41   * Adapter that converts and sends Messages to an AMQP Exchange.
42   *
43   * @author Mark Fisher
44   * @author Oleg Zhurakousky
45   * @author Gary Russell
46   * @author Artem Bilan
47   *
48   * @since 2.1
49   */
50  public class AmqpOutboundEndpoint extends AbstractAmqpOutboundEndpoint
51  		implements ConfirmCallback, ReturnCallback {
52  
53  	private static final Duration DEFAULT_CONFIRM_TIMEOUT = Duration.ofSeconds(5);
54  
55  	private final AmqpTemplate amqpTemplate;
56  
57  	private final RabbitTemplate rabbitTemplate;
58  
59  	private boolean expectReply;
60  
61  	private boolean waitForConfirm;
62  
63  	private Duration waitForConfirmTimeout = DEFAULT_CONFIRM_TIMEOUT;
64  
65  	private boolean multiSend;
66  
67  	public AmqpOutboundEndpoint(AmqpTemplate amqpTemplate) {
68  		Assert.notNull(amqpTemplate, "amqpTemplate must not be null");
69  		this.amqpTemplate = amqpTemplate;
70  		if (amqpTemplate instanceof RabbitTemplate) {
71  			setConnectionFactory(((RabbitTemplate) amqpTemplate).getConnectionFactory());
72  			this.rabbitTemplate = (RabbitTemplate) amqpTemplate;
73  		}
74  		else {
75  			this.rabbitTemplate = null;
76  		}
77  	}
78  
79  	/**
80  	 * Set to true if this endpoint is a gateway.
81  	 * @param expectReply true for a gateway.
82  	 */
83  	public void setExpectReply(boolean expectReply) {
84  		this.expectReply = expectReply;
85  	}
86  
87  	/**
88  	 * Set to true if you want to block the calling thread until a publisher confirm has
89  	 * been received. Requires a template configured for returns. If a confirm is not
90  	 * received within the confirm timeout or a negative acknowledgment or returned
91  	 * message is received, an exception will be thrown. Does not apply to the gateway
92  	 * since it blocks awaiting the reply.
93  	 * @param waitForConfirm true to block until the confirmation or timeout is received.
94  	 * @since 5.2
95  	 * @see #setConfirmTimeout(long)
96  	 * @see #setMultiSend(boolean)
97  	 */
98  	public void setWaitForConfirm(boolean waitForConfirm) {
99  		this.waitForConfirm = waitForConfirm;
100 	}
101 
102 	@Override
103 	public String getComponentType() {
104 		return this.expectReply ? "amqp:outbound-gateway" : "amqp:outbound-channel-adapter";
105 	}
106 
107 	/**
108 	 * If true, and the message payload is an {@link Iterable} of {@link Message}, send
109 	 * the messages in a single invocation of the template (same channel) and optionally
110 	 * wait for the confirms or die or perform all sends within a transaction (existing or
111 	 * new).
112 	 * @param multiSend true to send multiple messages.
113 	 * @since 5.3
114 	 * @see #setWaitForConfirm(boolean)
115 	 */
116 	public void setMultiSend(boolean multiSend) {
117 		Assert.isTrue(this.rabbitTemplate != null
118 				&& (!this.waitForConfirm || this.rabbitTemplate.getConnectionFactory().isSimplePublisherConfirms()),
119 				() -> "To use multiSend, " + AmqpOutboundEndpoint.this.amqpTemplate
120 					+ " must be a RabbitTemplate with a ConnectionFactory configured with simple confirms");
121 		this.multiSend = multiSend;
122 	}
123 
124 	@Override
125 	public IntegrationPatternType getIntegrationPatternType() {
126 		return this.expectReply ? super.getIntegrationPatternType() : IntegrationPatternType.outbound_channel_adapter;
127 	}
128 
129 	@Override
130 	public RabbitTemplate getRabbitTemplate() {
131 		return this.rabbitTemplate;
132 	}
133 
134 	@Override
135 	protected void endpointInit() {
136 		if (getConfirmCorrelationExpression() != null) {
137 			Assert.notNull(this.rabbitTemplate,
138 					"RabbitTemplate implementation is required for publisher confirms");
139 			this.rabbitTemplate.setConfirmCallback(this);
140 		}
141 		if (getReturnChannel() != null) {
142 			Assert.notNull(this.rabbitTemplate,
143 					"RabbitTemplate implementation is required for publisher confirms");
144 			this.rabbitTemplate.setReturnCallback(this);
145 		}
146 		Duration confirmTimeout = getConfirmTimeout();
147 		if (confirmTimeout != null) {
148 			this.waitForConfirmTimeout = confirmTimeout;
149 		}
150 	}
151 
152 	@Override
153 	protected void doStop() {
154 		if (this.amqpTemplate instanceof Lifecycle) {
155 			((Lifecycle) this.amqpTemplate).stop();
156 		}
157 	}
158 
159 	@Override
160 	protected Object handleRequestMessage(Message<?> requestMessage) {
161 		CorrelationData correlationData = generateCorrelationData(requestMessage);
162 		String exchangeName = generateExchangeName(requestMessage);
163 		String routingKey = generateRoutingKey(requestMessage);
164 		if (this.expectReply) {
165 			return sendAndReceive(exchangeName, routingKey, requestMessage, correlationData);
166 		}
167 		if (this.multiSend && requestMessage.getPayload() instanceof Iterable) {
168 			multiSend(requestMessage, exchangeName, routingKey);
169 			return null;
170 		}
171 		else {
172 			send(exchangeName, routingKey, requestMessage, correlationData);
173 			if (this.waitForConfirm && correlationData != null) {
174 				waitForConfirm(requestMessage, correlationData);
175 			}
176 			return null;
177 		}
178 	}
179 
180 	@SuppressWarnings("unchecked")
181 	private void multiSend(Message<?> requestMessage, String exchangeName, String routingKey) {
182 		((Iterable<?>) requestMessage.getPayload()).forEach(payload -> {
183 			Assert.state(payload instanceof Message,
184 					"To use multiSend, the payload must be an Iterable<Message<?>>");
185 		});
186 		this.rabbitTemplate.invoke(template -> {
187 			((Iterable<Message<?>>) requestMessage.getPayload()).forEach(message -> {
188 				doRabbitSend(exchangeName, routingKey, message, null, (RabbitTemplate) template);
189 			});
190 			if (this.waitForConfirm) {
191 				template.waitForConfirmsOrDie(this.waitForConfirmTimeout.toMillis());
192 			}
193 			return null;
194 		});
195 	}
196 
197 	private void waitForConfirm(Message<?> requestMessage, CorrelationData correlationData) {
198 		try {
199 			Confirm confirm = correlationData.getFuture().get(this.waitForConfirmTimeout.toMillis(),
200 					TimeUnit.MILLISECONDS);
201 			if (!confirm.isAck()) {
202 				throw new AmqpException("Negative publisher confirm received: " + confirm);
203 			}
204 			if (correlationData.getReturnedMessage() != null) {
205 				throw new AmqpException("Message was returned by the broker");
206 			}
207 		}
208 		catch (@SuppressWarnings("unused") InterruptedException e) {
209 			Thread.currentThread().interrupt();
210 		}
211 		catch (ExecutionException e) {
212 			throw new AmqpException("Failed to get publisher confirm", e);
213 		}
214 		catch (TimeoutException e) {
215 			throw new MessageTimeoutException(requestMessage, this + ": Timed out awaiting publisher confirm", e);
216 		}
217 	}
218 
219 	private void send(String exchangeName, String routingKey,
220 			final Message<?> requestMessage, CorrelationData correlationData) {
221 
222 		if (this.rabbitTemplate != null) {
223 			doRabbitSend(exchangeName, routingKey, requestMessage, correlationData, this.rabbitTemplate);
224 		}
225 		else {
226 			this.amqpTemplate.convertAndSend(exchangeName, routingKey, requestMessage.getPayload(),
227 					message -> {
228 						getHeaderMapper().fromHeadersToRequest(requestMessage.getHeaders(),
229 								message.getMessageProperties());
230 						return message;
231 					});
232 		}
233 	}
234 
235 	private void doRabbitSend(String exchangeName, String routingKey, final Message<?> requestMessage,
236 			CorrelationData correlationData, RabbitTemplate template) {
237 
238 		MessageConverter converter = template.getMessageConverter();
239 		org.springframework.amqp.core.Message amqpMessage = MappingUtils.mapMessage(requestMessage, converter,
240 				getHeaderMapper(), getDefaultDeliveryMode(), isHeadersMappedLast());
241 		addDelayProperty(requestMessage, amqpMessage);
242 		template.send(exchangeName, routingKey, amqpMessage, correlationData);
243 	}
244 
245 	private AbstractIntegrationMessageBuilder<?> sendAndReceive(String exchangeName, String routingKey,
246 			Message<?> requestMessage, CorrelationData correlationData) {
247 
248 		Assert.state(this.rabbitTemplate != null,
249 				"RabbitTemplate implementation is required for publisher confirms");
250 		MessageConverter converter = this.rabbitTemplate.getMessageConverter();
251 		org.springframework.amqp.core.Message amqpMessage = MappingUtils.mapMessage(requestMessage, converter,
252 				getHeaderMapper(), getDefaultDeliveryMode(), isHeadersMappedLast());
253 		addDelayProperty(requestMessage, amqpMessage);
254 		org.springframework.amqp.core.Message amqpReplyMessage =
255 				this.rabbitTemplate.sendAndReceive(exchangeName, routingKey, amqpMessage,
256 						correlationData);
257 
258 		if (amqpReplyMessage == null) {
259 			return null;
260 		}
261 		return buildReply(converter, amqpReplyMessage);
262 	}
263 
264 	@Override
265 	public void confirm(CorrelationData correlationData, boolean ack, String cause) {
266 		handleConfirm(correlationData, ack, cause);
267 	}
268 
269 	@Override
270 	public void returnedMessage(org.springframework.amqp.core.Message message, int replyCode, String replyText,
271 			String exchange, String routingKey) {
272 
273 		// no need for null check; we asserted we have a RabbitTemplate in doInit()
274 		MessageConverter converter = this.rabbitTemplate.getMessageConverter();
275 		Message<?> returned = buildReturnedMessage(message, replyCode, replyText, exchange,
276 				routingKey, converter);
277 		getReturnChannel().send(returned);
278 	}

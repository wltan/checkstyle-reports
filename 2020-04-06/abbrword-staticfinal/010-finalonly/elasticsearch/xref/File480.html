40  /**
41   * CircuitBreakerService that attempts to redistribute space between breakers
42   * if tripped
43   */
44  public class HierarchyCircuitBreakerService extends CircuitBreakerService {
45  
46      private volatile ImmutableMap<CircuitBreaker.Name, CircuitBreaker> breakers;
47  
48      // Old pre-1.4.0 backwards compatible settings
49      public static final String OLD_CIRCUIT_BREAKER_MAX_BYTES_SETTING = "indices.fielddata.breaker.limit";
50      public static final String OLD_CIRCUIT_BREAKER_OVERHEAD_SETTING = "indices.fielddata.breaker.overhead";
51  
52      public static final String TOTAL_CIRCUIT_BREAKER_LIMIT_SETTING = "indices.breaker.total.limit";
53      public static final String DEFAULT_TOTAL_CIRCUIT_BREAKER_LIMIT = "70%";
54  
55      public static final String FIELDDATA_CIRCUIT_BREAKER_LIMIT_SETTING = "indices.breaker.fielddata.limit";
56      public static final String FIELDDATA_CIRCUIT_BREAKER_OVERHEAD_SETTING = "indices.breaker.fielddata.overhead";
57      public static final String FIELDDATA_CIRCUIT_BREAKER_TYPE_SETTING = "indices.breaker.fielddata.type";
58      public static final String DEFAULT_FIELDDATA_BREAKER_LIMIT = "60%";
59      public static final double DEFAULT_FIELDDATA_OVERHEAD_CONSTANT = 1.03;
60  
61      public static final String REQUEST_CIRCUIT_BREAKER_LIMIT_SETTING = "indices.breaker.request.limit";
62      public static final String REQUEST_CIRCUIT_BREAKER_OVERHEAD_SETTING = "indices.breaker.request.overhead";
63      public static final String REQUEST_CIRCUIT_BREAKER_TYPE_SETTING = "indices.breaker.request.type";
64      public static final String DEFAULT_REQUEST_BREAKER_LIMIT = "40%";
65  
66      public static final String DEFAULT_BREAKER_TYPE = "memory";
67  
68      private volatile BreakerSettings parentSettings;
69      private volatile BreakerSettings fielddataSettings;
70      private volatile BreakerSettings requestSettings;
71  
72      // Tripped count for when redistribution was attempted but wasn't successful
73      private final AtomicLong parentTripCount = new AtomicLong(0);
74  
75      @Inject
76      public HierarchyCircuitBreakerService(Settings settings, NodeSettingsService nodeSettingsService) {
77          super(settings);
78  
79          // This uses the old InternalCircuitBreakerService.CIRCUIT_BREAKER_MAX_BYTES_SETTING
80          // setting to keep backwards compatibility with 1.3, it can be safely
81          // removed when compatibility with 1.3 is no longer needed
82          String compatibilityFielddataLimitDefault = DEFAULT_FIELDDATA_BREAKER_LIMIT;
83          ByteSizeValue compatibilityFielddataLimit = settings.getAsMemory(OLD_CIRCUIT_BREAKER_MAX_BYTES_SETTING, null);
84          if (compatibilityFielddataLimit != null) {
85              compatibilityFielddataLimitDefault = compatibilityFielddataLimit.toString();
86          }
87  
88          // This uses the old InternalCircuitBreakerService.CIRCUIT_BREAKER_OVERHEAD_SETTING
89          // setting to keep backwards compatibility with 1.3, it can be safely
90          // removed when compatibility with 1.3 is no longer needed
91          double compatibilityFielddataOverheadDefault = DEFAULT_FIELDDATA_OVERHEAD_CONSTANT;
92          Double compatibilityFielddataOverhead = settings.getAsDouble(OLD_CIRCUIT_BREAKER_OVERHEAD_SETTING, null);
93          if (compatibilityFielddataOverhead != null) {
94              compatibilityFielddataOverheadDefault = compatibilityFielddataOverhead;
95          }
96  
97          this.fielddataSettings = new BreakerSettings(CircuitBreaker.Name.FIELDDATA,
98                  settings.getAsMemory(FIELDDATA_CIRCUIT_BREAKER_LIMIT_SETTING, compatibilityFielddataLimitDefault).bytes(),
99                  settings.getAsDouble(FIELDDATA_CIRCUIT_BREAKER_OVERHEAD_SETTING, compatibilityFielddataOverheadDefault),
100                 CircuitBreaker.Type.parseValue(settings.get(FIELDDATA_CIRCUIT_BREAKER_TYPE_SETTING, DEFAULT_BREAKER_TYPE))
101         );
102 
103         this.requestSettings = new BreakerSettings(CircuitBreaker.Name.REQUEST,
104                 settings.getAsMemory(REQUEST_CIRCUIT_BREAKER_LIMIT_SETTING, DEFAULT_REQUEST_BREAKER_LIMIT).bytes(),
105                 settings.getAsDouble(REQUEST_CIRCUIT_BREAKER_OVERHEAD_SETTING, 1.0),
106                 CircuitBreaker.Type.parseValue(settings.get(REQUEST_CIRCUIT_BREAKER_TYPE_SETTING, DEFAULT_BREAKER_TYPE))
107         );
108 
109         // Validate the configured settings
110         validateSettings(new BreakerSettings[] {this.requestSettings, this.fielddataSettings});
111 
112         this.parentSettings = new BreakerSettings(CircuitBreaker.Name.PARENT,
113                 settings.getAsMemory(TOTAL_CIRCUIT_BREAKER_LIMIT_SETTING, DEFAULT_TOTAL_CIRCUIT_BREAKER_LIMIT).bytes(), 1.0, CircuitBreaker.Type.PARENT);
114         if (logger.isTraceEnabled()) {
115             logger.trace("parent circuit breaker with settings {}", this.parentSettings);
116         }
117 
118         Map<CircuitBreaker.Name, CircuitBreaker> tempBreakers = new HashMap<>();
119 
120         CircuitBreaker fielddataBreaker;
121         if (fielddataSettings.getType() == CircuitBreaker.Type.NOOP) {
122             fielddataBreaker = new NoopCircuitBreaker(CircuitBreaker.Name.FIELDDATA);
123         } else {
124             fielddataBreaker = new ChildMemoryCircuitBreaker(fielddataSettings, logger, this, CircuitBreaker.Name.FIELDDATA);
125         }
126 
127         CircuitBreaker requestBreaker;
128         if (requestSettings.getType() == CircuitBreaker.Type.NOOP) {
129             requestBreaker = new NoopCircuitBreaker(CircuitBreaker.Name.REQUEST);
130         } else {
131             requestBreaker = new ChildMemoryCircuitBreaker(requestSettings, logger, this, CircuitBreaker.Name.REQUEST);
132         }
133 
134         tempBreakers.put(CircuitBreaker.Name.FIELDDATA, fielddataBreaker);
135         tempBreakers.put(CircuitBreaker.Name.REQUEST, requestBreaker);
136         this.breakers = ImmutableMap.copyOf(tempBreakers);
137 
138         nodeSettingsService.addListener(new ApplySettings());
139     }
140 
141     public class ApplySettings implements NodeSettingsService.Listener {
142 
143         @Override
144         public void onRefreshSettings(Settings settings) {
145             boolean changed = false;
146 
147             // Fielddata settings
148             BreakerSettings newFielddataSettings = HierarchyCircuitBreakerService.this.fielddataSettings;
149             ByteSizeValue newFielddataMax = settings.getAsMemory(FIELDDATA_CIRCUIT_BREAKER_LIMIT_SETTING, null);
150             Double newFielddataOverhead = settings.getAsDouble(FIELDDATA_CIRCUIT_BREAKER_OVERHEAD_SETTING, null);
151             if (newFielddataMax != null || newFielddataOverhead != null) {
152                 changed = true;
153                 long newFielddataLimitBytes = newFielddataMax == null ? HierarchyCircuitBreakerService.this.fielddataSettings.getLimit() : newFielddataMax.bytes();
154                 newFielddataOverhead = newFielddataOverhead == null ? HierarchyCircuitBreakerService.this.fielddataSettings.getOverhead() : newFielddataOverhead;
155 
156                 newFielddataSettings = new BreakerSettings(CircuitBreaker.Name.FIELDDATA, newFielddataLimitBytes, newFielddataOverhead,
157                         HierarchyCircuitBreakerService.this.fielddataSettings.getType());
158             }
159 
160             // Request settings
161             BreakerSettings newRequestSettings = HierarchyCircuitBreakerService.this.requestSettings;
162             ByteSizeValue newRequestMax = settings.getAsMemory(REQUEST_CIRCUIT_BREAKER_LIMIT_SETTING, null);
163             Double newRequestOverhead = settings.getAsDouble(REQUEST_CIRCUIT_BREAKER_OVERHEAD_SETTING, null);
164             if (newRequestMax != null || newRequestOverhead != null) {
165                 changed = true;
166                 long newRequestLimitBytes = newRequestMax == null ? HierarchyCircuitBreakerService.this.requestSettings.getLimit() : newRequestMax.bytes();
167                 newRequestOverhead = newRequestOverhead == null ? HierarchyCircuitBreakerService.this.requestSettings.getOverhead() : newRequestOverhead;
168 
169                 newRequestSettings = new BreakerSettings(CircuitBreaker.Name.REQUEST, newRequestLimitBytes, newRequestOverhead,
170                         HierarchyCircuitBreakerService.this.requestSettings.getType());
171             }
172 
173             // Parent settings
174             BreakerSettings newParentSettings = HierarchyCircuitBreakerService.this.parentSettings;
175             long oldParentMax = HierarchyCircuitBreakerService.this.parentSettings.getLimit();
176             ByteSizeValue newParentMax = settings.getAsMemory(TOTAL_CIRCUIT_BREAKER_LIMIT_SETTING, null);
177             if (newParentMax != null && (newParentMax.bytes() != oldParentMax)) {
178                 changed = true;
179                 newParentSettings = new BreakerSettings(CircuitBreaker.Name.PARENT, newParentMax.bytes(), 1.0, CircuitBreaker.Type.PARENT);
180             }
181 
182             if (changed) {
183                 // change all the things
184                 validateSettings(new BreakerSettings[]{newFielddataSettings, newRequestSettings});
185                 logger.info("Updating settings parent: {}, fielddata: {}, request: {}", newParentSettings, newFielddataSettings, newRequestSettings);
186                 HierarchyCircuitBreakerService.this.parentSettings = newParentSettings;
187                 HierarchyCircuitBreakerService.this.fielddataSettings = newFielddataSettings;
188                 HierarchyCircuitBreakerService.this.requestSettings = newRequestSettings;
189 
190                 Map<CircuitBreaker.Name, CircuitBreaker> tempBreakers = new HashMap<>();
191                 CircuitBreaker fielddataBreaker;
192                 if (newFielddataSettings.getType() == CircuitBreaker.Type.NOOP) {
193                     fielddataBreaker = new NoopCircuitBreaker(CircuitBreaker.Name.FIELDDATA);
194                 } else {
195                     fielddataBreaker = new ChildMemoryCircuitBreaker(newFielddataSettings,
196                             (ChildMemoryCircuitBreaker) HierarchyCircuitBreakerService.this.breakers.get(CircuitBreaker.Name.FIELDDATA),
197                             logger, HierarchyCircuitBreakerService.this, CircuitBreaker.Name.FIELDDATA);
198                 }
199 
200                 CircuitBreaker requestBreaker;
201                 if (newRequestSettings.getType() == CircuitBreaker.Type.NOOP) {
202                     requestBreaker = new NoopCircuitBreaker(CircuitBreaker.Name.REQUEST);
203                 } else {
204                     requestBreaker = new ChildMemoryCircuitBreaker(newRequestSettings,
205                             (ChildMemoryCircuitBreaker)HierarchyCircuitBreakerService.this.breakers.get(CircuitBreaker.Name.REQUEST),
206                             logger, HierarchyCircuitBreakerService.this, CircuitBreaker.Name.REQUEST);
207                 }
208 
209                 tempBreakers.put(CircuitBreaker.Name.FIELDDATA, fielddataBreaker);
210                 tempBreakers.put(CircuitBreaker.Name.REQUEST, requestBreaker);
211                 HierarchyCircuitBreakerService.this.breakers = ImmutableMap.copyOf(tempBreakers);
212             }
213         }
214     }

216     /**
217      * Validate that child settings are valid
218      * @throws ElasticsearchIllegalStateException
219      */
220     public static void validateSettings(BreakerSettings[] childrenSettings) throws ElasticsearchIllegalStateException {
221         for (BreakerSettings childSettings : childrenSettings) {
222             // If the child is disabled, ignore it
223             if (childSettings.getLimit() == -1) {
224                 continue;
225             }
226 
227             if (childSettings.getOverhead() < 0) {
228                 throw new ElasticsearchIllegalStateException("Child breaker overhead " + childSettings + " must be non-negative");
229             }
230         }
231     }
232 
233     @Override
234     public CircuitBreaker getBreaker(CircuitBreaker.Name name) {
235         return this.breakers.get(name);
236     }
237 
238     @Override
239     public AllCircuitBreakerStats stats() {
240         long parentEstimated = 0;
241         List<CircuitBreakerStats> allStats = newArrayList();
242         // Gather the "estimated" count for the parent breaker by adding the
243         // estimations for each individual breaker
244         for (CircuitBreaker breaker : this.breakers.values()) {
245             allStats.add(stats(breaker.getName()));
246             parentEstimated += breaker.getUsed();
247         }
248         // Manually add the parent breaker settings since they aren't part of the breaker map
249         allStats.add(new CircuitBreakerStats(CircuitBreaker.Name.PARENT, parentSettings.getLimit(),
250                 parentEstimated, 1.0, parentTripCount.get()));
251         return new AllCircuitBreakerStats(allStats.toArray(new CircuitBreakerStats[allStats.size()]));
252     }
253 
254     @Override
255     public CircuitBreakerStats stats(CircuitBreaker.Name name) {
256         CircuitBreaker breaker = this.breakers.get(name);
257         return new CircuitBreakerStats(breaker.getName(), breaker.getLimit(), breaker.getUsed(), breaker.getOverhead(), breaker.getTrippedCount());
258     }
259 
260     /**
261      * Checks whether the parent breaker has been tripped
262      * @param label
263      * @throws CircuitBreakingException
264      */
265     public void checkParentLimit(String label) throws CircuitBreakingException {
266         long totalUsed = 0;
267         for (CircuitBreaker breaker : this.breakers.values()) {
268             totalUsed += (breaker.getUsed() * breaker.getOverhead());
269         }
270 
271         long parentLimit = this.parentSettings.getLimit();
272         if (totalUsed > parentLimit) {
273             this.parentTripCount.incrementAndGet();
274             throw new CircuitBreakingException("[PARENT] Data too large, data for [" +
275                     label + "] would be larger than limit of [" +
276                     parentLimit + "/" + new ByteSizeValue(parentLimit) + "]",
277                     totalUsed, parentLimit);
278         }
279     }

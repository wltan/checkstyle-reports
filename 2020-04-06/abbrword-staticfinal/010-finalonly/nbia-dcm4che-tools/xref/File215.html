40  package org.dcm4chex.archive.util;

42  import java.io.File;

47  import javax.xml.transform.OutputKeys;
48  import javax.xml.transform.Templates;
49  import javax.xml.transform.Transformer;
50  import javax.xml.transform.TransformerConfigurationException;
51  import javax.xml.transform.TransformerFactory;
52  import javax.xml.transform.sax.SAXResult;
53  import javax.xml.transform.sax.SAXTransformerFactory;
54  import javax.xml.transform.sax.TransformerHandler;
55  import javax.xml.transform.stream.StreamResult;
56  
57  import org.dcm4che.data.Dataset;
58  import org.dcm4che.dict.DictionaryFactory;
59  import org.dcm4che.dict.TagDictionary;
60  import org.dcm4che.net.Association;
61  import org.dcm4che.util.DAFormat;
62  import org.dcm4che.util.TMFormat;
63  
64  /**
65   * @author gunter.zeilinger@tiani.com
66   * @version $Revision: 17587 $ $Date: 2013-01-10 12:04:09 +0100 (Do, 10 JÃ¤n 2013) $
67   * @since Dec 5, 2005

69  public class XSLTUtils {
70  
71      public static final SAXTransformerFactory transformerFactory =
72              (SAXTransformerFactory) TransformerFactory.newInstance();
73  
74      public static void xslt(Dataset ds, Templates tpl, Association a,
75              Dataset out) throws TransformerConfigurationException, IOException {
76          xslt(ds, getTransformerHandler(tpl, a), out);
77      }
78  
79      public static TransformerHandler getTransformerHandler(Templates tpl,
80              Association a) throws TransformerConfigurationException {
81          TransformerHandler th = transformerFactory.newTransformerHandler(tpl);
82          setDateParameters(th);
83          if (a != null) {
84              setAETParameters(th, a);
85          }
86          return th;
87      }
88  
89      public static void xslt(Dataset ds, TransformerHandler th, Dataset out)
90              throws IOException {
91          th.setResult(new SAXResult(out.getSAXHandler2(null)));
92          ds.writeDataset2(th, null, null, 10240, null);
93      }
94  
95      public static void xslt(Dataset ds, Templates tpl, Dataset out)
96              throws TransformerConfigurationException, IOException {
97          xslt(ds, tpl, null, out);        
98      }
99      
100     public static void setDateParameters(TransformerHandler th) {
101         setDateParameters(th, new Date());
102     }
103     
104     public static void setDateParameters(TransformerHandler th, Date date) {
105         Transformer t = th.getTransformer();
106         t.setParameter("date", new DAFormat().format(date));
107         t.setParameter("time", new TMFormat().format(date));
108     }
109 
110     public static void setAETParameters(TransformerHandler th, Association a) {
111         Transformer t = th.getTransformer();
112         t.setParameter("calling", a.getCallingAET());
113         t.setParameter("called", a.getCalledAET());
114     }
115 
116     public static void writeTo(Dataset ds, File f)
117             throws TransformerConfigurationException, IOException {
118         TransformerHandler th = transformerFactory.newTransformerHandler();
119         th.getTransformer().setOutputProperty(OutputKeys.INDENT, "yes");
120         FileOutputStream out = new FileOutputStream(f);
121         TagDictionary dict = DictionaryFactory.getInstance().getDefaultTagDictionary();
122         try {
123             th.setResult(new StreamResult(out));
124             ds.writeDataset2(th, dict, null, 10240, null);		
125         } finally {
126             out.close();
127         }
128     }
129 
130 }

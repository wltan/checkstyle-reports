131         StringBuilder sb = new StringBuilder();
132         for (Code code : rejectionNoteCodes) {
133             sb.append(code).append("\r\n");
134         }
135         return sb.toString();

138     public void setRejectionNoteCodes(String input) {
139         Matcher m = PATTERN.matcher(input);
140         List<Code> tmp = new ArrayList<Code>();
141         while (m.find()) {
142             tmp.add(SRDocumentFactory.getInstance().newCode(m.group(1)));
143         }
144         this.rejectionNoteCodes = tmp;
145     }

179     public boolean scheduleDeleteOnRejectionNoteStored(Dataset ds)
180             throws Exception {
181         if (!UIDs.KeyObjectSelectionDocument.equals(
182                 ds.getString(Tags.SOPClassUID))) {
183             return false;
184         }
185         Dataset title = ds.getItem(Tags.ConceptNameCodeSeq);
186         if (!rejectionNoteCodes.contains(
187                 SRDocumentFactory.getInstance().newCode(title))) {
188             return false;
189         }
190         schedule(new RejectionNoteStoredOrder(ds), deletionDelay != 0 
191                     ? System.currentTimeMillis() + deletionDelay : 0L);
192         return true;
193     }

195     protected void schedule(BaseJmsOrder order, long scheduledTime)
196             throws Exception {
197         if (log.isInfoEnabled()) {
198             String scheduledTimeStr = scheduledTime > 0
199                     ? new Date(scheduledTime).toString()
200                     : "now";
201             log.info("Scheduling job [" + order + "] at "
202                     + scheduledTimeStr + ". Retry times: "
203                     + order.getFailureCount());
204         }
205         jmsDelegate.queue(queueName, order,
206                 Message.DEFAULT_PRIORITY, scheduledTime);
207     }

213     }

218         jmsDelegate.stopListening(queueName);
219     }

222     public void handleNotification(Notification notif, Object handback) {
223         Dataset ds = (Dataset) notif.getUserData();
224         if (UIDs.KeyObjectSelectionDocument.equals(
225                 ds.getString(Tags.SOPClassUID))) {
226             Dataset title = ds.getItem(Tags.ConceptNameCodeSeq);
227             if (rejectionNoteCodes.contains(
228                     SRDocumentFactory.getInstance().newCode(title))) {
229                 RejectionNoteStoredOrder order =
230                     new RejectionNoteStoredOrder(ds);
231                 try {
232                     schedule(order, deletionDelay == 0 ? 0L
233                             : System.currentTimeMillis() + deletionDelay);
234                 } catch (Exception e) {
235                     log.error("Failed to schedule " + order, e);
236                 }
237             }
238         }
239     }

241     public void onMessage(Message msg) {
242         RejectionNoteStoredOrder order;
243         try {
244             order = (RejectionNoteStoredOrder)
245                     ((ObjectMessage) msg).getObject();
246         } catch (Exception e) {
247             log.error("Processing JMS message failed! message:" + msg, e);
248             return;
249         }
250         if (log.isDebugEnabled())
251             log.debug("Processing " + order);
252         try {
253             moveInstancesToTrash(iuidsOf(order.getDataset()));
254             if (log.isDebugEnabled())
255                 log.debug("Finished processing " + order);
256         } catch (Exception e) {
257             final int failureCount = order.getFailureCount() + 1;
258             final long delay = retryIntervals.getIntervall(failureCount);
259             order.setFailureCount(failureCount);
260             if (delay == -1L) {
261                 order.setThrowable(e);
262                 log.error("Give up to process " + order);
263                 try {
264                     jmsDelegate.fail(queueName, order);
265                 } catch (Exception e2) {
266                     log.error("Failed to notify JMSDelgate of failed job!"
267                             + " Give up to process" + order, e2);
268                 }
269             } else {
270                 Throwable thisThrowable = e;
271                 if (e instanceof InvocationTargetException)
272                     thisThrowable = ((InvocationTargetException) e)
273                             .getTargetException();
274 
275                 if (failureCount == 1
276                         || (order.getThrowable() != null && !thisThrowable
277                                 .getClass().equals(
278                                         order.getThrowable().getClass()))) {
279                     // If this happens first time, log as error
280                     log.error("Failed to process JMS job."
281                             +" Will schedule retry... Dumping - "
282                             + order.toLongString(), e);
283                     // Record this exception
284                     order.setThrowable(thisThrowable);
285                 } else {
286                     // otherwise, if it's the same exception as before
287                     log.warn("Failed to process " + order
288                             + ". Details should have been provided."
289                             + " Will schedule retry.");
290                 }
291                 try {
292                     schedule(order, System.currentTimeMillis() + delay);
293                 } catch (Exception e2) {
294                     log.error("Failed to schedule retry! Give up to process "
295                             + order, e2);
296                 }
297             }
298         }
299     }

301     private String[] iuidsOf(Dataset ds) {
302         List<String> iuids;
303         if (checkCallingAET) {
304             ds.setPrivateCreatorID(PrivateTags.CreatorID);
305             String callingAET = ds.getString(PrivateTags.CallingAET);
306             ds.setPrivateCreatorID(null);
307             try {
308                 Storage store = this.getStorage();
309                 iuids = store.getSopIuidsForRejectionNote(ds, callingAET);
310             } catch (Exception x) {
311                 log.error("Get SOPInstance UIDs for RejectionNote failed!", x);
312                 iuids =  new ArrayList<String>();
313             }
314         } else {
315             iuids = new ArrayList<String>();
316             DcmElement stysq = ds.get(Tags.CurrentRequestedProcedureEvidenceSeq);
317             for (int i = 0, styCount = stysq.countItems(); i < styCount; i++) {
318                 DcmElement sersq = stysq.getItem(i).get(Tags.RefSeriesSeq);
319                 for (int j = 0, serCount = sersq.countItems(); j < serCount; j++) {
320                     DcmElement refsopsq = sersq.getItem(j).get(Tags.RefSOPSeq);
321                     for (int k = 0, refsopsqCount = refsopsq.countItems();
322                             k < refsopsqCount; k++) {
323                         iuids.add(refsopsq.getItem(k)
324                                 .getString(Tags.RefSOPInstanceUID));
325                     }
326                 }
327             }
328        }
329        if (!keepRejectionNote) {
330            iuids.add(ds.getString(Tags.SOPInstanceUID));
331        }
332        return iuids.toArray(new String[iuids.size()]);
333     }

335     private void moveInstancesToTrash(String[] iuids) throws Exception {
336         try {
337             server.invoke(contentEditServiceName,
338                     "moveInstancesToTrash",
339                     new Object[] { iuids },
340                     new String[] { String[].class.getName() });
341         } catch (MBeanException e) {
342             throw e.getTargetException();
343         } catch (InstanceNotFoundException e) {
344             throw new ConfigurationException(e);
345         } catch (ReflectionException e) {
346             throw new ConfigurationException(e);
347         }
348     }

181     private void onSeriesStored(final SeriesStored stored) {
182         if (stored.getRetrieveAET() == null) {
183             log.warn("Ignore SeriesStored notification! Reason: Series is not locally retrievable.");
184             return;
185         }
186         log.info("Handle SeriesStored Notification! stored:"+stored);
187         Templates tpl = null;
188         try {
189             tpl = templates.getTemplatesForAET(
190                 stored.getSourceAET(), WADO_PREFETCH_XSL);
191         } catch (Throwable t) {
192             log.error("Failed to get Template for wado-prefetch.xsl!", t );
193         }
194         if (tpl != null) {
195             Dataset ds = DcmObjectFactory.getInstance().newDataset();
196             ds.putAll(stored.getPatientAttrs());
197             ds.putAll(stored.getStudyAttrs());
198             ds.putAll(stored.getSeriesAttrs());
199             log.debug("WADOPrefetch schedule transform input:");log.debug(ds);
200             try {
201                 xslt(stored.getSourceAET(), stored.getRetrieveAET(),
202                         ds, tpl, new DefaultHandler(){
203 
204                     public void startElement(String uri, String localName,
205                             String qName, Attributes attrs) {
206                         if (qName.equals("prefetch")) {
207                             WadoPrefetchOrder order = new WadoPrefetchOrder(attrs.getValue("wadourl"),
208                                     attrs.getValue("exportPath"), stored.getIAN());
209                             try {
210                                 jmsDelegate.queue(queueName, order, 
211                                         Message.DEFAULT_PRIORITY, 0L);
212                             } catch (Exception e) {
213                                 log.error("Failed to schedule " + order, e);
214                             }
215                         }
216                     }});
217             } catch (Exception e) {
218                 log.error("Applying WADO prefetch rules to " + stored + " fails:", e);
219             }
220         }
221     }
222 
223     public void onMessage(Message message) {
224         ObjectMessage om = (ObjectMessage) message;
225         try {
226             WadoPrefetchOrder order = (WadoPrefetchOrder) om.getObject();
227             log.info("Start processing order" + order);
228             try {
229                 process(order);
230                 log.info("Processing finished! order:"+order);
231             } catch (Exception e) {
232                 order.setThrowable(e);
233                 final int failureCount = order.getFailureCount() + 1;
234                 order.setFailureCount(failureCount);
235                 final long delay = retryIntervalls.getIntervall(failureCount);
236                 if (delay == -1L) {
237                     log.error("Give up to process " + order, e);
238                     jmsDelegate.fail(queueName, order);
239                 } else {
240                     log.warn("Failed to process " + order
241                             + ". Scheduling retry.", e);
242                     jmsDelegate.queue(queueName, order, 0, System.currentTimeMillis()
243                             + delay);
244                 }
245             }
246         } catch (Throwable e) {
247             log.error("unexpected error during processing message: " + message,
248                     e);
249         }
250     }
251     
252     private void process(WadoPrefetchOrder order) throws IOException {
253         String wadourl = order.getWadoUrl();
254         String dest = order.getExportPath();
255         Dataset ian = order.getIAN();
256         DcmElement refSopSq = ian.getItem(Tags.RefSeriesSeq).get(Tags.RefSOPSeq);
257         Dataset item;
258         for ( int i = 0 ; i < refSopSq.countItems() ; i++ ) {
259             item = refSopSq.getItem(i);
260             prefetch(wadourl, dest, item.getString(Tags.RefSOPInstanceUID));
261         }
262     }
263 
264     private void prefetch(String baseURL, String dest, String iuid) throws IOException {
265         String url = baseURL+"&objectUID="+iuid;
266         try {
267             URL wadourl = new URL(url);
268             HttpURLConnection conn = (HttpURLConnection) wadourl.openConnection();
269             if (conn.getResponseCode() != HttpServletResponse.SC_OK) {
270                 log.warn("Prefetch WADO URL failed:"+wadourl);
271                 throw new IOException(conn.getResponseCode()+":"+conn.getResponseMessage());
272             }
273             if (dest != null) {
274                 export(dest, iuid, conn);
275             }
276         } catch (MalformedURLException e) {
277             log.error("Prefetch request ignored: Malformed WADO URL! Need configuration change in wado-prefetch.xsl! url:"+url);
278         }
279     }

281     private void export(String dest, String iuid, HttpURLConnection conn)
282             throws IOException, FileNotFoundException {
283         InputStream is = conn.getInputStream();
284         File file = getFile(dest, iuid);
285         log.info("M-WRITE "+file);
286         BufferedOutputStream out = new BufferedOutputStream(
287                 new FileOutputStream(file));
288         byte[] buf = new byte[BUF_LEN];
289         try {
290             int len = is.read(buf);
291             while (len > 0) {
292                 out.write(buf, 0, len);
293                 len = is.read(buf);
294             }
295         } finally {
296             is.close();
297             out.flush();
298             out.close();
299         }
300     }
301 
302     private File getFile(String dest, String iuid) {
303         String fn = MessageFormat.format(dest, new Object[]{iuid});
304         File file = FileUtils.resolve(new File(fn));
305         if (!file.getParentFile().exists()) {
306             file.getParentFile().mkdirs();
307         }
308         return file;
309     }

311     private void xslt(String sourceAET, String retrieveAET,
312             Dataset ds, Templates tpl, ContentHandler ch)
313             throws TransformerConfigurationException, IOException {
314         SAXTransformerFactory tf = (SAXTransformerFactory)
315             TransformerFactory.newInstance();
316         TransformerHandler th = tf.newTransformerHandler(tpl);
317         Transformer t = th.getTransformer();
318         t.setParameter("source-aet", sourceAET);
319         t.setParameter("retrieve-aet", retrieveAET);
320         t.setParameter("wado-baseurl", wadoBaseUrl);
321         t.setParameter("export-path", exportBasePath);
322         th.setResult(new SAXResult(ch));
323         ds.writeDataset2(th, null, null, 64, null);
324     }

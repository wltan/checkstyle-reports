86      private static final int GPSPS_NOT_IN_PROGRESS = 0xA504;
87      private static final int GPSPS_DIFF_TRANS_UID = 0xA505;

89      private PatientLocalHome patHome;
90      private GPSPSLocalHome spsHome;
91      private GPPPSLocalHome ppsHome;

93      public void setSessionContext(SessionContext ctx) {
94          Context jndiCtx = null;
95          try {
96              jndiCtx = new InitialContext();
97              patHome =
98                  (PatientLocalHome) jndiCtx.lookup("java:comp/env/ejb/Patient");
99              spsHome = (GPSPSLocalHome) jndiCtx.lookup("java:comp/env/ejb/GPSPS");
100             ppsHome = (GPPPSLocalHome) jndiCtx.lookup("java:comp/env/ejb/GPPPS");
101         } catch (NamingException e) {
102             throw new EJBException(e);
103         } finally {
104             if (jndiCtx != null) {
105                 try {
106                     jndiCtx.close();
107                 } catch (NamingException ignore) {}
108             }
109         }
110     }
111 
112     public void unsetSessionContext() {
113         spsHome = null;
114         ppsHome = null;
115         patHome = null;
116     }

118     /**
119      * @ejb.interface-method
120      */
121     public void createGPPPS(Dataset ds, PatientMatching matching)
122     throws DcmServiceException {
123         checkDuplicate(ds.getString(Tags.SOPInstanceUID));
124         PatientLocal pat = findOrCreatePatient(ds, matching);
125         Collection gpsps = findRefGpsps(ds, pat);
126         GPPPSLocal pps = doCreate(ds, pat);
127         if (gpsps != null) {
128             pps.setGpsps(gpsps);
129         }
130     }
131 
132     private PatientLocal findOrCreatePatient(Dataset ds,
133             PatientMatching matching) throws DcmServiceException {
134         try {
135             try {
136                 return patHome.selectPatient(ds, matching, true);
137             } catch (ObjectNotFoundException enfe) {
138                 return patHome.create(ds);
139             } catch (NonUniquePatientException onfe) {
140                 return patHome.create(ds);
141             }
142         } catch (Exception e) {
143             throw new DcmServiceException(Status.ProcessingFailure, e);
144         }           
145     }
146 
147     private GPPPSLocal doCreate(Dataset ds, PatientLocal pat)
148     throws DcmServiceException {
149         try {
150             return ppsHome.create(ds, pat);
151         } catch (CreateException e) {
152             log.error("Creation of GP-PPS(iuid=" 
153                     + ds.getString(Tags.SOPInstanceUID) + ") failed: ", e);
154             throw new DcmServiceException(Status.ProcessingFailure);
155         }                
156     }
157 
158     private void checkDuplicate(String ppsiuid) throws DcmServiceException {
159         try {
160             ppsHome.findBySopIuid(ppsiuid);
161             throw new DcmServiceException(Status.DuplicateSOPInstance);
162         } catch (ObjectNotFoundException e) { // Ok           
163         } catch (FinderException e) {
164             log.error("Query for GP-PPS(iuid=" + ppsiuid + ") failed: ", e);
165             throw new DcmServiceException(Status.ProcessingFailure);
166         }
167     }
168 
169     private Collection findRefGpsps(Dataset pps, PatientLocal pat)
170     throws DcmServiceException {
171         DcmElement spssq = pps.get(Tags.RefGPSPSSeq);
172         if (spssq == null) return null;
173         int n = spssq.countItems();
174         ArrayList c = new ArrayList(n);
175         for (int i = 0; i < n; i++) {
176             Dataset refSOP = spssq.getItem(i);
177             String spsiuid = refSOP.getString(Tags.RefSOPInstanceUID);
178             String spstuid = refSOP.getString(Tags.RefGPSPSTransactionUID);
179             GPSPSLocal sps;
180             try {
181                 sps = spsHome.findBySopIuid(spsiuid);
182                 PatientLocal spspat = sps.getPatient();
183                 if (!pat.isIdentical(spspat)) {
184                     log.info("Patient of referenced GP-SPS(iuid=" + spsiuid
185                             + "): " + spspat.asString() 
186                             + " differes from Patient of GP-PPS: "
187                             + pat.asString());                   
188                     throw new DcmServiceException(Status.InvalidAttributeValue,
189                             "GP-SPS PID: " + spspat.getPatientId()
190                             + ", GP-PPS PID: " + pat.getPatientId());                
191                 }
192                 if (!sps.isInProgress()) {
193                     String spsstatus = sps.getGpspsStatus();
194                     log.info("Status of referenced GP-SPS(iuid=" + spsiuid
195                             + ") is not IN PROGRESS, but " + spsstatus);
196                     throw new DcmServiceException(GPSPS_NOT_IN_PROGRESS,
197                             "ref GP-SPS status: " + spsstatus);                
198                 }
199                 String tuid = sps.getTransactionUid();
200                 if (!spstuid.equals(tuid)) {
201                     log.info("Referenced GP-SPS Transaction UID: " + spstuid 
202                             + " does not match the Transaction UID: " + tuid
203                             + " of the N-ACTION request");
204                     throw new DcmServiceException(GPSPS_DIFF_TRANS_UID);                
205                 }
206                 Dataset spsAttrs = sps.getAttributes();
207                 DcmElement ppss = spsAttrs.get(Tags.ResultingGPPPSSeq);
208                 if (ppss == null)
209                     ppss = spsAttrs.putSQ(Tags.ResultingGPPPSSeq);
210                 Dataset refpps = ppss.addNewItem();
211                 refpps.putUI(Tags.RefSOPClassUID,
212                         pps.getString(Tags.SOPClassUID));
213                 refpps.putUI(Tags.RefSOPInstanceUID,
214                         pps.getString(Tags.SOPInstanceUID));
215                 sps.setAttributes(spsAttrs);
216                 c.add(sps);
217             } catch (ObjectNotFoundException e) {
218                 log.info("Referenced GP-SPS(iuid=" + spsiuid
219                         + ") not in provided GP-WL");
220             } catch (FinderException e) {
221                 log.error("Query for GP-SPS(iuid=" + spsiuid
222                         + ") failed: ", e);
223                 throw new DcmServiceException(Status.ProcessingFailure);
224             }
225         }
226         return c;
227     }

229     /**
230      * @ejb.interface-method
231      */
232     public Dataset getGPPPS(String iuid) throws DcmServiceException {
233         GPPPSLocal pps = findBySopIuid(iuid);
234         final PatientLocal pat = pps.getPatient();
235         Dataset attrs = pps.getAttributes();            
236         attrs.putAll(pat.getAttributes(false));
237         return attrs;
238     }
239     
240     /**
241      * @ejb.interface-method
242      */
243     public void updateGPPPS(Dataset ds)
244         throws DcmServiceException {
245         final String iuid = ds.getString(Tags.SOPInstanceUID);
246         GPPPSLocal pps = findBySopIuid(iuid);
247         if (!pps.isInProgress()) {
248             String ppsstatus = pps.getPpsStatus();
249             log.info("Status of GP-PPS(iuid=" + iuid
250                     + ") is not IN PROGRESS, but " + ppsstatus);
251             throw new DcmServiceException(GPPPS_NOT_IN_PROGRESS,
252                     "GP-PPS status: " + ppsstatus);                
253         }
254         Dataset attrs = pps.getAttributes();
255         attrs.putAll(ds);
256         pps.setAttributes(attrs);
257     }
258 
259     private GPPPSLocal findBySopIuid(String iuid) throws DcmServiceException {
260         try {
261             return ppsHome.findBySopIuid(iuid);
262         } catch (ObjectNotFoundException e) {
263             throw new DcmServiceException(Status.NoSuchObjectInstance);
264         } catch (FinderException e) {
265             throw new DcmServiceException(Status.ProcessingFailure, e);
266         }
267     }

1   package org.infinispan.xsite;
2   
3   import org.infinispan.commands.VisitableCommand;
4   import org.infinispan.context.InvocationContext;
5   
6   /**
7    * RPC command to replicate cache operations (such as put, remove, replace, etc.) to the backup site.
8    *
9    * @author Pedro Ruivo
10   * @since 7.0
11   */
12  public class SingleXSiteRpcCommand extends XSiteReplicateCommand {
13  
14     public static final int COMMAND_ID = 40;
15     private VisitableCommand command;
16  
17     public SingleXSiteRpcCommand(String cacheName, VisitableCommand command) {
18        super(cacheName);
19        this.command = command;
20     }
21  
22     public SingleXSiteRpcCommand(String cacheName) {
23        super(cacheName);
24     }
25  
26     public SingleXSiteRpcCommand() {
27        super(null);
28     }
29  
30     @Override
31     public Object performInLocalSite(BackupReceiver receiver) throws Throwable {
32        return receiver.handleRemoteCommand(command);
33     }
34  
35     @Override
36     public Object perform(InvocationContext ctx) throws Throwable {
37        return command.perform(ctx);
38     }
39  
40     @Override
41     public byte getCommandId() {
42        return COMMAND_ID;
43     }
44  
45     @Override
46     public Object[] getParameters() {
47        return new Object[]{command};
48     }
49  
50     @Override
51     public void setParameters(int commandId, Object[] parameters) {
52        if (commandId != COMMAND_ID) {
53           throw new IllegalArgumentException("Unusupported command id:" + commandId);
54        }
55        command = (VisitableCommand) parameters[0];
56     }
57  
58     @Override
59     public boolean isReturnValueExpected() {
60        return command.isReturnValueExpected();
61     }
62  
63     @Override
64     public String toString() {
65        return "SingleXSiteRpcCommand{" +
66              "command=" + command +
67              '}';
68     }
69  }

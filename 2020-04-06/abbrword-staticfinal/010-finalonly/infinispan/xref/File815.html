76          HttpResponse insert = put(fullPathKey, initialXML, "application/octet-stream");
77  
78          assertEquals(0, insert.getEntity().getContentLength());

80          HttpResponse get = get(fullPathKey, initialXML);
81          assertEquals("application/octet-stream", get.getHeaders("Content-Type")[0].getValue());

83          delete(fullPathKey);
84          get(fullPathKey, HttpServletResponse.SC_NOT_FOUND);
85  
86          put(fullPathKey, initialXML, "application/octet-stream");
87          get(fullPathKey, initialXML);
88  
89          delete(fullPathKey(null));
90          get(fullPathKey, HttpServletResponse.SC_NOT_FOUND);
91  
92          ByteArrayOutputStream bout = new ByteArrayOutputStream();
93          ObjectOutputStream oo = new ObjectOutputStream(bout);
94          oo.writeObject(new TestSerializable("CONTENT"));
95          oo.flush();
96  
97          byte[] byteData = bout.toByteArray();
98          put(fullPathKey, byteData, "application/octet-stream");

100         HttpResponse resp = getWithoutClose(fullPathKey);
101         ObjectInputStream oin = new ObjectInputStream(resp.getEntity().getContent());
102         TestSerializable ts = (TestSerializable) oin.readObject();
103         EntityUtils.consume(resp.getEntity());

415     @Test
416     public void testStoreBigObject() throws Exception {
417         int SIZE = 3000000;
418         byte[] bytes = new byte[SIZE];
419         for (int i = 0; i < SIZE; i++) {
420             bytes[i] = (byte) (i % 10);
421         }
422         put(fullPathKey("object"), bytes, "application/octet-stream");
423 
424         HttpResponse resp = getWithoutClose(fullPathKey("object"));
425         InputStream responseStream = resp.getEntity().getContent();
426         byte[] response = new byte[SIZE];
427         byte data;
428         int j = 0;
429         while ((data = (byte) responseStream.read()) != -1) {
430             response[j] = data;
431             j++;
432         }
433 
434         boolean correct = true;
435         for (int i = 0; i < SIZE; i++) {
436             if (bytes[i] != response[i]) {
437                 correct = false;
438             }
439         }
440         EntityUtils.consume(resp.getEntity());
441         assertTrue(correct);
442     }

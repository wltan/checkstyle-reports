23  package org.jboss.as.clustering.infinispan.subsystem;
24  
25  import org.jboss.as.controller.AttributeDefinition;
26  import org.jboss.as.controller.OperationStepHandler;
27  import org.jboss.as.controller.PathElement;
28  import org.jboss.as.controller.ReloadRequiredRemoveStepHandler;
29  import org.jboss.as.controller.ReloadRequiredWriteAttributeHandler;
30  import org.jboss.as.controller.SimpleAttributeDefinition;
31  import org.jboss.as.controller.SimpleAttributeDefinitionBuilder;
32  import org.jboss.as.controller.SimpleResourceDefinition;
33  import org.jboss.as.controller.client.helpers.MeasurementUnit;
34  import org.jboss.as.controller.registry.AttributeAccess;
35  import org.jboss.as.controller.registry.ManagementResourceRegistration;
36  import org.jboss.dmr.ModelNode;
37  import org.jboss.dmr.ModelType;
38  
39  /**
40   * Resource description for the addressable resource /subsystem=infinispan/cache-container=X/transport=TRANSPORT
41   *
42   * @author Richard Achmatowicz (c) 2011 Red Hat Inc.
43   */
44  public class TransportResource extends SimpleResourceDefinition {
45  
46      public static final PathElement TRANSPORT_PATH = PathElement.pathElement(ModelKeys.TRANSPORT, ModelKeys.TRANSPORT_NAME);
47  
48      // attributes
49      static final SimpleAttributeDefinition CLUSTER =
50              new SimpleAttributeDefinitionBuilder(ModelKeys.CLUSTER, ModelType.STRING, true)
51                      .setXmlName(Attribute.CLUSTER.getLocalName())
52                      .setAllowExpression(true)
53                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
54                      .build();
55  
56      static final SimpleAttributeDefinition EXECUTOR =
57              new SimpleAttributeDefinitionBuilder(ModelKeys.EXECUTOR, ModelType.STRING, true)
58                      .setXmlName(Attribute.EXECUTOR.getLocalName())
59                      .setAllowExpression(false)
60                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
61                      .build();
62  
63      static final SimpleAttributeDefinition LOCK_TIMEOUT =
64              new SimpleAttributeDefinitionBuilder(ModelKeys.LOCK_TIMEOUT, ModelType.LONG, true)
65                      .setXmlName(Attribute.LOCK_TIMEOUT.getLocalName())
66                      .setMeasurementUnit(MeasurementUnit.MILLISECONDS)
67                      .setAllowExpression(true)
68                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
69                      .setDefaultValue(new ModelNode().set(240000))
70                      .build();
71  
72      static final SimpleAttributeDefinition REMOTE_COMMAND_EXECUTOR =
73              new SimpleAttributeDefinitionBuilder(ModelKeys.REMOTE_COMMAND_EXECUTOR, ModelType.STRING, true)
74                      .setXmlName(Attribute.REMOTE_COMMAND_EXECUTOR.getLocalName())
75                      .setAllowExpression(false)
76                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
77                      .build();
78  
79      // if stack is null, use default stack
80      static final SimpleAttributeDefinition STACK =
81              new SimpleAttributeDefinitionBuilder(ModelKeys.STACK, ModelType.STRING, true)
82                      .setXmlName(Attribute.STACK.getLocalName())
83                      .setAllowExpression(true)
84                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
85                      .build();
86  
87      static final SimpleAttributeDefinition STRICT_PEER_TO_PEER =
88              new SimpleAttributeDefinitionBuilder(ModelKeys.STRICT_PEER_TO_PEER, ModelType.BOOLEAN, true)
89                      .setXmlName(Attribute.STRICT_PEER_TO_PEER.getLocalName())
90                      .setAllowExpression(true)
91                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
92                      .setDefaultValue(new ModelNode().set(false))
93                      .build();
94  
95      static final SimpleAttributeDefinition TOTAL_ORDER_EXECUTOR =
96              new SimpleAttributeDefinitionBuilder(ModelKeys.TOTAL_ORDER_EXECUTOR, ModelType.STRING, true)
97                      .setXmlName(Attribute.TOTAL_ORDER_EXECUTOR.getLocalName())
98                      .setAllowExpression(false)
99                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
100                     .build();
101 
102     static final AttributeDefinition[] TRANSPORT_ATTRIBUTES = {STACK, CLUSTER, EXECUTOR, LOCK_TIMEOUT,
103             REMOTE_COMMAND_EXECUTOR, STRICT_PEER_TO_PEER, TOTAL_ORDER_EXECUTOR};
104 
105     public TransportResource() {
106         super(TRANSPORT_PATH,
107                 InfinispanExtension.getResourceDescriptionResolver(ModelKeys.TRANSPORT),
108                 CacheConfigOperationHandlers.TRANSPORT_ADD,
109                 ReloadRequiredRemoveStepHandler.INSTANCE);
110     }
111 
112     @Override
113     public void registerAttributes(ManagementResourceRegistration resourceRegistration) {
114         super.registerAttributes(resourceRegistration);
115 
116         // check that we don't need a special handler here?
117         final OperationStepHandler writeHandler = new ReloadRequiredWriteAttributeHandler(TRANSPORT_ATTRIBUTES);
118         for (AttributeDefinition attr : TRANSPORT_ATTRIBUTES) {
119             resourceRegistration.registerReadWriteAttribute(attr, null, writeHandler);
120         }
121     }
122 }

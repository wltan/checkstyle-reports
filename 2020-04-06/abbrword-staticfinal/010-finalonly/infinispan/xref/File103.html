1   package org.infinispan.commons.configuration.attributes;
2   
3   import java.util.HashMap;
4   import java.util.HashSet;
5   
6   import org.infinispan.commons.logging.Log;
7   import org.infinispan.commons.logging.LogFactory;
8   
9   /**
10   * CollectionAttributeCopier. This {@link AttributeCopier} can handle a handful of "known"
11   * collection types ( {@link HashSet}, {@link HashMap} )
12   *
13   * @author Tristan Tarrant
14   * @since 7.2
15   */
16  public class CollectionAttributeCopier<T> implements AttributeCopier<T> {
17     private static final Log log = LogFactory.getLog(CollectionAttributeCopier.class);
18     public static final AttributeCopier<Object> INSTANCE = new CollectionAttributeCopier<>();
19  
20     @SuppressWarnings("unchecked")
21     @Override
22     public T copyAttribute(T attribute) {
23        if (attribute == null) {
24           return null;
25        }
26        Class<?> klass = attribute.getClass();
27        if (HashSet.class.equals(klass)) {
28           return (T) new HashSet<>((HashSet<?>)attribute);
29        } else if (HashMap.class.equals(klass)) {
30           return (T) new HashMap<>((HashMap<?, ?>)attribute);
31        } else {
32           throw log.noAttributeCopierForType(klass);
33        }
34     }
35  
36  }

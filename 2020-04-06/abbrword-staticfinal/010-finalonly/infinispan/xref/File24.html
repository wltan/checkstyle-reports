38      /**
39       * Does the first stage of comparing AnnoatedCallables, however it cannot
40       * compare the method parameters
41       */
42      private static class AnnotatedCallableComparator<T> implements Comparator<AnnotatedCallable<? super T>> {
43  
44          public int compare(AnnotatedCallable<? super T> arg0, AnnotatedCallable<? super T> arg1) {
45              // compare the names first
46              int result = (arg0.getJavaMember().getName().compareTo(arg1.getJavaMember().getName()));
47              if (result != 0) {
48                  return result;
49              }
50              result = arg0.getJavaMember().getDeclaringClass().getName().compareTo(arg1.getJavaMember().getDeclaringClass().getName());
51              if (result != 0) {
52                  return result;
53              }
54              result = arg0.getParameters().size() - arg1.getParameters().size();
55              return result;
56          }
57  
58      }
59  
60      private static class AnnotatedMethodComparator<T> implements Comparator<AnnotatedMethod<? super T>> {
61  
62          public static <T> Comparator<AnnotatedMethod<? super T>> instance() {
63              return new AnnotatedMethodComparator<T>();
64          }
65  
66          private AnnotatedCallableComparator<T> callableComparator = new AnnotatedCallableComparator<T>();
67  
68          public int compare(AnnotatedMethod<? super T> arg0, AnnotatedMethod<? super T> arg1) {
69              int result = callableComparator.compare(arg0, arg1);
70              if (result != 0) {
71                  return result;
72              }
73              for (int i = 0; i < arg0.getJavaMember().getParameterTypes().length; ++i) {
74                  Class<?> p0 = arg0.getJavaMember().getParameterTypes()[i];
75                  Class<?> p1 = arg1.getJavaMember().getParameterTypes()[i];
76                  result = p0.getName().compareTo(p1.getName());
77                  if (result != 0) {
78                      return result;
79                  }
80              }
81              return 0;
82          }
83  
84      }

95              int result = callableComparator.compare(arg0, arg1);

100                 Class<?> p0 = arg0.getJavaMember().getParameterTypes()[i];
101                 Class<?> p1 = arg1.getJavaMember().getParameterTypes()[i];

112     private static class AnnotatedFieldComparator<T> implements Comparator<AnnotatedField<? super T>> {
113 
114         public static <T> Comparator<AnnotatedField<? super T>> instance() {
115             return new AnnotatedFieldComparator<T>();
116         }
117 
118         public int compare(AnnotatedField<? super T> arg0, AnnotatedField<? super T> arg1) {
119             if (arg0.getJavaMember().getName().equals(arg1.getJavaMember().getName())) {
120                 return arg0.getJavaMember().getDeclaringClass().getName().compareTo(arg1.getJavaMember().getDeclaringClass().getName());
121             }
122             return arg0.getJavaMember().getName().compareTo(arg1.getJavaMember().getName());
123         }
124 
125     }

127     private static class AnnotationComparator implements Comparator<Annotation> {
128 
129         public static final Comparator<Annotation> INSTANCE = new AnnotationComparator();

133         }

150     /**
151      * Generates a deterministic signature for an {@link AnnotatedType}. Two
152      * <code>AnnotatedType</code>s that have the same annotations and underlying
153      * type will generate the same signature.
154      * <p/>
155      * This can be used to create a unique bean id for a passivation capable bean
156      * that is added directly through the SPI.
157      *
158      * @param annotatedType The type to generate a signature for
159      * @return A string representation of the annotated type
160      */
161     public static <X> String createTypeId(AnnotatedType<X> annotatedType) {
162         return createTypeId(annotatedType.getJavaClass(), annotatedType.getAnnotations(), annotatedType.getMethods(), annotatedType.getFields(), annotatedType.getConstructors());
163     }
164 
165     /**
166      * Generates a unique signature for a concrete class. Annotations are not
167      * read directly from the class, but are read from the
168      * <code>annotations</code>, <code>methods</code>, <code>fields</code> and
169      * <code>constructors</code> arguments
170      *
171      * @param clazz        The java class tyoe
172      * @param annotations  Annotations present on the java class
173      * @param methods      The AnnotatedMethods to include in the signature
174      * @param fields       The AnnotatedFields to include in the signature
175      * @param constructors The AnnotatedConstructors to include in the signature
176      * @return A string representation of the type
177      */
178     public static <X> String createTypeId(Class<X> clazz, Collection<Annotation> annotations, Collection<AnnotatedMethod<? super X>> methods, Collection<AnnotatedField<? super X>> fields, Collection<AnnotatedConstructor<X>> constructors) {
179         StringBuilder builder = new StringBuilder();
180 
181         builder.append(clazz.getName());
182         builder.append(createAnnotationCollectionId(annotations));
183         builder.append("{");
184 
185         // now deal with the fields
186         List<AnnotatedField<? super X>> sortedFields = new ArrayList<AnnotatedField<? super X>>();
187         sortedFields.addAll(fields);
188         Collections.sort(sortedFields, AnnotatedFieldComparator.<X>instance());
189         for (AnnotatedField<? super X> field : sortedFields) {
190             if (!field.getAnnotations().isEmpty()) {
191                 builder.append(createFieldId(field));
192                 builder.append(SEPERATOR);
193             }
194         }
195 
196         // methods
197         List<AnnotatedMethod<? super X>> sortedMethods = new ArrayList<AnnotatedMethod<? super X>>();
198         sortedMethods.addAll(methods);
199         Collections.sort(sortedMethods, AnnotatedMethodComparator.<X>instance());
200         for (AnnotatedMethod<? super X> method : sortedMethods) {
201             if (!method.getAnnotations().isEmpty() || hasMethodParameters(method)) {
202                 builder.append(createCallableId(method));
203                 builder.append(SEPERATOR);
204             }
205         }
206 
207         // constructors
208         List<AnnotatedConstructor<? super X>> sortedConstructors = new ArrayList<AnnotatedConstructor<? super X>>();
209         sortedConstructors.addAll(constructors);
210         Collections.sort(sortedConstructors, AnnotatedConstructorComparator.<X>instance());
211         for (AnnotatedConstructor<? super X> constructor : sortedConstructors) {
212             if (!constructor.getAnnotations().isEmpty() || hasMethodParameters(constructor)) {
213                 builder.append(createCallableId(constructor));
214                 builder.append(SEPERATOR);
215             }
216         }
217         builder.append("}");
218 
219         return builder.toString();
220     }

222     /**
223      * Generates a deterministic signature for an {@link AnnotatedField}. Two
224      * <code>AnnotatedField</code>s that have the same annotations and
225      * underlying field will generate the same signature.
226      */
227     public static <X> String createFieldId(AnnotatedField<X> field) {
228         return createFieldId(field.getJavaMember(), field.getAnnotations());
229     }

231     /**
232      * Creates a deterministic signature for a {@link Field}.
233      *
234      * @param field       The field to generate the signature for
235      * @param annotations The annotations to include in the signature
236      */
237     public static <X> String createFieldId(Field field, Collection<Annotation> annotations) {
238         StringBuilder builder = new StringBuilder();
239         builder.append(field.getDeclaringClass().getName());
240         builder.append('.');
241         builder.append(field.getName());
242         builder.append(createAnnotationCollectionId(annotations));
243         return builder.toString();
244     }

246     /**
247      * Generates a deterministic signature for an {@link AnnotatedCallable}. Two
248      * <code>AnnotatedCallable</code>s that have the same annotations and
249      * underlying callable will generate the same signature.
250      */
251     public static <X> String createCallableId(AnnotatedCallable<X> method) {
252         StringBuilder builder = new StringBuilder();
253         builder.append(method.getJavaMember().getDeclaringClass().getName());
254         builder.append('.');
255         builder.append(method.getJavaMember().getName());
256         builder.append(createAnnotationCollectionId(method.getAnnotations()));
257         builder.append(createParameterListId(method.getParameters()));
258         return builder.toString();
259     }
260 
261     /**
262      * Generates a unique string representation of a list of
263      * {@link AnnotatedParameter}s.
264      */
265     public static <X> String createParameterListId(List<AnnotatedParameter<X>> parameters) {
266         StringBuilder builder = new StringBuilder();
267         builder.append("(");
268         for (int i = 0; i < parameters.size(); ++i) {
269             AnnotatedParameter<X> ap = parameters.get(i);
270             builder.append(createParameterId(ap));
271             if (i + 1 != parameters.size()) {
272                 builder.append(',');
273             }
274         }
275         builder.append(")");
276         return builder.toString();
277     }

279     /**
280      * Creates a string representation of an {@link AnnotatedParameter}.
281      */
282     public static <X> String createParameterId(AnnotatedParameter<X> annotatedParameter) {
283         return createParameterId(annotatedParameter.getBaseType(), annotatedParameter.getAnnotations());
284     }

286     /**
287      * Creates a string representation of a given type and set of annotations.
288      */
289     public static <X> String createParameterId(Type type, Set<Annotation> annotations) {
290         StringBuilder builder = new StringBuilder();
291         if (type instanceof Class<?>) {
292             Class<?> c = (Class<?>) type;
293             builder.append(c.getName());
294         } else {
295             builder.append(type.toString());
296         }
297         builder.append(createAnnotationCollectionId(annotations));
298         return builder.toString();
299     }
300 
301     private static <X> boolean hasMethodParameters(AnnotatedCallable<X> callable) {
302         for (AnnotatedParameter<X> parameter : callable.getParameters()) {
303             if (!parameter.getAnnotations().isEmpty()) {
304                 return true;
305             }
306         }
307         return false;
308     }

310     private static String createAnnotationCollectionId(Collection<Annotation> annotations) {
311         if (annotations.isEmpty()) {
312             return "";
313         }
314 
315         StringBuilder builder = new StringBuilder();
316         builder.append('[');
317 
318         List<Annotation> annotationList = new ArrayList<Annotation>(annotations.size());
319         annotationList.addAll(annotations);
320         Collections.sort(annotationList, AnnotationComparator.INSTANCE);
321 
322         for (Annotation a : annotationList) {
323             builder.append('@');
324             builder.append(a.annotationType().getName());
325             builder.append('(');
326             Method[] declaredMethods = a.annotationType().getDeclaredMethods();
327             List<Method> methods = new ArrayList<Method>(declaredMethods.length);
328             for (Method m : declaredMethods) {
329                 methods.add(m);
330             }
331             Collections.sort(methods, MethodComparator.INSTANCE);
332 
333             for (int i = 0; i < methods.size(); ++i) {
334                 Method method = methods.get(i);
335                 try {
336                     Object value = method.invoke(a);
337                     builder.append(method.getName());
338                     builder.append('=');
339                     builder.append(value.toString());
340                 } catch (NullPointerException e) {
341                     throw new RuntimeException("NullPointerException accessing annotation member, annotation:" + a.annotationType().getName() + " member: " + method.getName(), e);
342                 } catch (IllegalArgumentException e) {
343                     throw new RuntimeException("IllegalArgumentException accessing annotation member, annotation:" + a.annotationType().getName() + " member: " + method.getName(), e);
344                 } catch (IllegalAccessException e) {
345                     throw new RuntimeException("IllegalAccessException accessing annotation member, annotation:" + a.annotationType().getName() + " member: " + method.getName(), e);
346                 } catch (InvocationTargetException e) {
347                     throw new RuntimeException("InvocationTargetException accessing annotation member, annotation:" + a.annotationType().getName() + " member: " + method.getName(), e);
348                 }
349                 if (i + 1 != methods.size()) {
350                     builder.append(',');
351                 }
352             }
353             builder.append(')');
354         }
355         builder.append(']');
356         return builder.toString();
357     }

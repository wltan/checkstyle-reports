123     /*
124      * Two constraints need to be dealt with here:
125      * 1. There may be no started cache instance available to interrogate. Because of lazy deployment,
126      * a cache instance is only started upon deployment of an application which uses that cache instance.
127      * 2. The attribute name passed in may not correspond to a defined metric
128      *
129      * Read-only attributes have no easy way to throw an exception without negatively impacting other parts
130      * of the system. Therefore in such cases, as message will be logged and a ModelNode of undefined will be returned.
131      */
132     @Override
133     protected void executeRuntimeStep(OperationContext context, ModelNode operation) throws OperationFailedException {
134         final PathAddress address = PathAddress.pathAddress(operation.require(OP_ADDR));
135         final String cacheContainerName = address.getElement(address.size() - 2).getValue();
136         final String cacheName = address.getLastElement().getValue();
137         final String attrName = operation.require(NAME).asString();
138         final ServiceController<?> controller = context.getServiceRegistry(false).getService(CacheService.getServiceName(cacheContainerName, cacheName));
139         Cache<?, ?> cache = (Cache<?, ?>) controller.getValue();
140         CacheMetrics metric = CacheMetrics.getStat(attrName);
141         ModelNode result = new ModelNode();
142 
143         if (metric == null) {
144             context.getFailureDescription().set(String.format("Unknown metric %s", attrName));
145         } else if (cache == null) {
146             context.getFailureDescription().set(String.format("Unavailable cache %s", attrName));
147         } else {
148             AdvancedCache<?, ?> aCache = cache.getAdvancedCache();
149             LockManagerImpl lockManager = (LockManagerImpl) SecurityActions.getLockManager(aCache);
150             RpcManagerImpl rpcManager = (RpcManagerImpl) SecurityActions.getRpcManager(aCache);
151             List<CommandInterceptor> interceptors = SecurityActions.getInterceptorChain(aCache);
152             ComponentRegistry registry = SecurityActions.getComponentRegistry(aCache);
153             ComponentStatus status = SecurityActions.getCacheStatus(aCache);
154             switch (metric) {
155                 case CACHE_STATUS:
156                     result.set(status.toString());
157                     break;
158                 case CONCURRENCY_LEVEL:
159                     result.set(lockManager.getConcurrencyLevel());
160                     break;
161                 case NUMBER_OF_LOCKS_AVAILABLE:
162                     result.set(lockManager.getNumberOfLocksAvailable());
163                     break;
164                 case NUMBER_OF_LOCKS_HELD:
165                     result.set(lockManager.getNumberOfLocksHeld());
166                     break;
167                 case AVERAGE_READ_TIME: {
168                     CacheMgmtInterceptor cacheMgmtInterceptor = getFirstInterceptorWhichExtends(interceptors, CacheMgmtInterceptor.class);
169                     result.set(cacheMgmtInterceptor != null ? cacheMgmtInterceptor.getAverageReadTime() : 0);
170                     break;
171                 }
172                 case AVERAGE_WRITE_TIME: {
173                     CacheMgmtInterceptor cacheMgmtInterceptor = getFirstInterceptorWhichExtends(interceptors, CacheMgmtInterceptor.class);
174                     result.set(cacheMgmtInterceptor != null ? cacheMgmtInterceptor.getAverageWriteTime() : 0);
175                     break;
176                 }
177                 case AVERAGE_REMOVE_TIME: {
178                     CacheMgmtInterceptor cacheMgmtInterceptor = getFirstInterceptorWhichExtends(interceptors, CacheMgmtInterceptor.class);
179                     result.set(cacheMgmtInterceptor != null ? cacheMgmtInterceptor.getAverageRemoveTime() : 0);
180                     break;
181                 }
182                 case TIME_SINCE_START: {
183                     CacheMgmtInterceptor cacheMgmtInterceptor = getFirstInterceptorWhichExtends(interceptors, CacheMgmtInterceptor.class);
184                     result.set(cacheMgmtInterceptor != null ? cacheMgmtInterceptor.getTimeSinceStart() : 0);
185                     break;
186                 }
187                 case EVICTIONS: {
188                     CacheMgmtInterceptor cacheMgmtInterceptor = getFirstInterceptorWhichExtends(interceptors, CacheMgmtInterceptor.class);
189                     result.set(cacheMgmtInterceptor != null ? cacheMgmtInterceptor.getEvictions() : 0);
190                     break;
191                 }
192                 case HIT_RATIO: {
193                     CacheMgmtInterceptor cacheMgmtInterceptor = getFirstInterceptorWhichExtends(interceptors, CacheMgmtInterceptor.class);
194                     result.set(cacheMgmtInterceptor != null ? cacheMgmtInterceptor.getHitRatio() : 0);
195                     break;
196                 }
197                 case HITS: {
198                     CacheMgmtInterceptor cacheMgmtInterceptor = getFirstInterceptorWhichExtends(interceptors, CacheMgmtInterceptor.class);
199                     result.set(cacheMgmtInterceptor != null ? cacheMgmtInterceptor.getHits() : 0);
200                     break;
201                 }
202                 case MISSES: {
203                     CacheMgmtInterceptor cacheMgmtInterceptor = getFirstInterceptorWhichExtends(interceptors, CacheMgmtInterceptor.class);
204                     result.set(cacheMgmtInterceptor != null ? cacheMgmtInterceptor.getMisses() : 0);
205                     break;
206                 }
207                 case NUMBER_OF_ENTRIES: {
208                     CacheMgmtInterceptor cacheMgmtInterceptor = getFirstInterceptorWhichExtends(interceptors, CacheMgmtInterceptor.class);
209                     result.set(cacheMgmtInterceptor != null ? cacheMgmtInterceptor.getNumberOfEntries() : 0);
210                     break;
211                 }
212                 case READ_WRITE_RATIO: {
213                     CacheMgmtInterceptor cacheMgmtInterceptor = getFirstInterceptorWhichExtends(interceptors, CacheMgmtInterceptor.class);
214                     result.set(cacheMgmtInterceptor != null ? cacheMgmtInterceptor.getReadWriteRatio() : 0);
215                     break;
216                 }
217                 case REMOVE_HITS: {
218                     CacheMgmtInterceptor cacheMgmtInterceptor = getFirstInterceptorWhichExtends(interceptors, CacheMgmtInterceptor.class);
219                     result.set(cacheMgmtInterceptor != null ? cacheMgmtInterceptor.getRemoveHits() : 0);
220                     break;
221                 }
222                 case REMOVE_MISSES: {
223                     CacheMgmtInterceptor cacheMgmtInterceptor = getFirstInterceptorWhichExtends(interceptors, CacheMgmtInterceptor.class);
224                     result.set(cacheMgmtInterceptor != null ? cacheMgmtInterceptor.getRemoveMisses() : 0);
225                     break;
226                 }
227                 case STORES: {
228                     CacheMgmtInterceptor cacheMgmtInterceptor = getFirstInterceptorWhichExtends(interceptors, CacheMgmtInterceptor.class);
229                     result.set(cacheMgmtInterceptor != null ? cacheMgmtInterceptor.getStores() : 0);
230                     break;
231                 }
232                 case TIME_SINCE_RESET: {
233                     CacheMgmtInterceptor cacheMgmtInterceptor = getFirstInterceptorWhichExtends(interceptors, CacheMgmtInterceptor.class);
234                     result.set(cacheMgmtInterceptor != null ? cacheMgmtInterceptor.getTimeSinceReset() : 0);
235                     break;
236                 }
237                 case AVERAGE_REPLICATION_TIME: {
238                     result.set(rpcManager.getAverageReplicationTime());
239                     break;
240                 }
241                 case REPLICATION_COUNT:
242                     result.set(rpcManager.getReplicationCount());
243                     break;
244                 case REPLICATION_FAILURES:
245                     result.set(rpcManager.getReplicationFailures());
246                     break;
247                 case SUCCESS_RATIO:
248                     result.set(rpcManager.getSuccessRatioFloatingPoint());
249                     break;
250                 case COMMITS: {
251                     TxInterceptor txInterceptor = getFirstInterceptorWhichExtends(interceptors, TxInterceptor.class);
252                     result.set(txInterceptor != null ? txInterceptor.getCommits() : 0);
253                     break;
254                 }
255                 case PREPARES: {
256                     TxInterceptor txInterceptor = getFirstInterceptorWhichExtends(interceptors, TxInterceptor.class);
257                     result.set(txInterceptor != null ? txInterceptor.getPrepares() : 0);
258                     break;
259                 }
260                 case ROLLBACKS: {
261                     TxInterceptor txInterceptor = getFirstInterceptorWhichExtends(interceptors, TxInterceptor.class);
262                     result.set(txInterceptor != null ? txInterceptor.getRollbacks() : 0);
263                     break;
264                 }
265                 case INVALIDATIONS: {
266                     InvalidationInterceptor invInterceptor = getFirstInterceptorWhichExtends(interceptors, InvalidationInterceptor.class);
267                     result.set(invInterceptor != null ? invInterceptor.getInvalidations() : 0);
268                     break;
269                 }
270                 case PASSIVATIONS: {
271                     PassivationManager manager = registry.getComponent(PassivationManager.class);
272                     result.set(manager != null ? manager.getPassivations() : 0);
273                     break;
274                 }
275                 case ACTIVATIONS: {
276                     ActivationManager manager = registry.getComponent(ActivationManager.class);
277                     result.set(manager != null ? manager.getActivationCount() : 0);
278                     break;
279                 }
280                 case CACHE_LOADER_LOADS: {
281                     ActivationInterceptor interceptor = getFirstInterceptorWhichExtends(interceptors, ActivationInterceptor.class);
282                     result.set(interceptor != null ? interceptor.getCacheLoaderLoads() : 0);
283                     break;
284                 }
285                 case CACHE_LOADER_MISSES: {
286                     ActivationInterceptor interceptor = getFirstInterceptorWhichExtends(interceptors, ActivationInterceptor.class);
287                     result.set(interceptor != null ? interceptor.getCacheLoaderMisses() : 0);
288                     break;
289                 }
290                 case CACHE_LOADER_STORES: {
291                     CacheWriterInterceptor interceptor = getFirstInterceptorWhichExtends(interceptors, CacheWriterInterceptor.class);
292                     result.set(interceptor != null ? interceptor.getWritesToTheStores() : 0);
293                     break;
294                 }
295                 case CACHE_NAME: {
296                     result.set(cache.getName());
297                     break;
298                 }
299                 case VERSION: {
300                     result.set(SecurityActions.getCacheVersion(aCache));
301                     break;
302                 }
303                 default:{
304                     context.getFailureDescription().set(String.format("Unknown metric %s", metric));
305                     break;
306                 }
307             }
308             context.getResult().set(result);
309         }
310         context.stepCompleted();
311     }

313     public void registerCommonMetrics(ManagementResourceRegistration container) {
314         for (CacheMetrics metric : CacheMetrics.values()) {
315             if (!metric.clustered) {
316                 container.registerMetric(metric.definition, this);
317             }
318         }
319     }
320 
321     public void registerClusteredMetrics(ManagementResourceRegistration container) {
322         for (CacheMetrics metric : CacheMetrics.values()) {
323             if (metric.clustered) {
324                 container.registerMetric(metric.definition, this);
325             }
326         }
327     }

329     public static <T extends CommandInterceptor> T getFirstInterceptorWhichExtends(List<CommandInterceptor> interceptors,
330                                                                                     Class<T> interceptorClass) {
331         for (CommandInterceptor interceptor : interceptors) {
332             boolean isSubclass = interceptorClass.isAssignableFrom(interceptor.getClass());
333             if (isSubclass) {
334                 Collections.emptyList();
335                 return (T) interceptor;
336             }
337         }
338         return null;
339     }

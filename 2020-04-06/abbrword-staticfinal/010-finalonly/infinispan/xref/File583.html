31              MIN_NODE_SIZE, MAX_NODE_SIZE, INDEX_QUEUE_LENGTH, SYNC_WRITES, OPEN_FILES_LIMIT, COMPACTION_THRESHOLD);
32     }
33

42     public String indexLocation() {

45  
46     public int indexSegments() {

53  
54     public int minNodeSize() {

64     }

68     }

93           DEFAULT_LOCKING_CACHENAME,

98     /**
99      * Registers the Search interceptor in the cache before it gets started
100     */
101    @Override
102    public void cacheStarting(ComponentRegistry cr, Configuration cfg, String cacheName) {
103       cr.registerComponent(new ReflectionMatcher(null), ReflectionMatcher.class);
104 
105       if (cfg.indexing().index().isEnabled()) {
106          log.registeringQueryInterceptor();
107          SearchIntegrator searchFactory = getSearchFactory(cfg.indexing().properties(), cr);
108          createQueryInterceptorIfNeeded(cr, cfg, searchFactory);
109          EmbeddedCacheManager cacheManager = cr.getGlobalComponentRegistry().getComponent(EmbeddedCacheManager.class);
110          addCacheDependencyIfNeeded(cacheName, cacheManager, cfg.indexing().properties());
111       }
112    }
113 
114    private void addCacheDependencyIfNeeded(String cacheStarting, EmbeddedCacheManager cacheManager, Properties properties) {
115       if (!ClusterRegistryImpl.GLOBAL_REGISTRY_CACHE_NAME.equals(cacheStarting)) {
116          cacheManager.addCacheDependency(cacheStarting, ClusterRegistryImpl.GLOBAL_REGISTRY_CACHE_NAME);
117       }
118       if (hasInfinispanDirectory(properties) && !DEFAULT_CACHES.contains(cacheStarting)) {
119          String metadataCacheName = getMetadataCacheName(properties);
120          String lockingCacheName = getLockingCacheName(properties);
121          String dataCacheName = getDataCacheName(properties);
122          if (!metadataCacheName.equals(cacheStarting)) {
123             cacheManager.addCacheDependency(cacheStarting, metadataCacheName);
124          }
125          if (!lockingCacheName.equals(cacheStarting)) {
126             cacheManager.addCacheDependency(cacheStarting, lockingCacheName);
127          }
128          if (!dataCacheName.equals(cacheStarting)) {
129             cacheManager.addCacheDependency(cacheStarting, dataCacheName);
130          }
131       }
132    }

134    private void createQueryInterceptorIfNeeded(ComponentRegistry cr, Configuration cfg, SearchIntegrator searchFactory) {
135       QueryInterceptor queryInterceptor = cr.getComponent(QueryInterceptor.class);
136       if (queryInterceptor == null) {
137          queryInterceptor = buildQueryInterceptor(cfg, searchFactory);
138 
139          // Interceptor registration not needed, core configuration handling
140          // already does it for all custom interceptors - UNLESS the InterceptorChain already exists in the component registry!
141          InterceptorChain ic = cr.getComponent(InterceptorChain.class);
142 
143          ConfigurationBuilder builder = new ConfigurationBuilder().read(cfg);
144          InterceptorConfigurationBuilder interceptorBuilder = builder.customInterceptors().addInterceptor();
145          interceptorBuilder.interceptor(queryInterceptor);
146 
147          if (!cfg.transaction().transactionMode().isTransactional()) {
148             if (ic != null) ic.addInterceptorAfter(queryInterceptor, NonTransactionalLockingInterceptor.class);
149             interceptorBuilder.after(NonTransactionalLockingInterceptor.class);
150          } else if (cfg.transaction().lockingMode() == LockingMode.OPTIMISTIC) {
151             if (ic != null) ic.addInterceptorAfter(queryInterceptor, OptimisticLockingInterceptor.class);
152             interceptorBuilder.after(OptimisticLockingInterceptor.class);
153          } else {
154             if (ic != null) ic.addInterceptorAfter(queryInterceptor, PessimisticLockingInterceptor.class);
155             interceptorBuilder.after(PessimisticLockingInterceptor.class);
156          }
157          if (ic != null) {
158             cr.registerComponent(queryInterceptor, QueryInterceptor.class);
159             cr.registerComponent(queryInterceptor, queryInterceptor.getClass().getName(), true);
160          }
161          cfg.customInterceptors().interceptors(builder.build().customInterceptors().interceptors());
162       }
163    }
164 
165    private QueryInterceptor buildQueryInterceptor(Configuration cfg, SearchIntegrator searchFactory) {
166       IndexModificationStrategy indexingStrategy = IndexModificationStrategy.configuredStrategy(searchFactory, cfg);
167       return new QueryInterceptor(searchFactory, indexingStrategy);
168    }

170    @Override
171    public void cacheStarted(ComponentRegistry cr, String cacheName) {
172       Configuration configuration = cr.getComponent(Configuration.class);
173       boolean indexingEnabled = configuration.indexing().index().isEnabled();
174       if ( ! indexingEnabled ) {
175          if ( verifyChainContainsQueryInterceptor(cr) ) {
176             throw new IllegalStateException( "It was NOT expected to find the Query interceptor registered in the InterceptorChain as indexing was disabled, but it was found" );
177          }
178          return;
179       }
180       if ( ! verifyChainContainsQueryInterceptor(cr) ) {
181          throw new IllegalStateException( "It was expected to find the Query interceptor registered in the InterceptorChain but it wasn't found" );
182       }
183 
184       // initializing the query module command initializer.
185       // we can t inject Cache and CacheManager with @inject in there
186       Cache<?, ?> cache = cr.getComponent(Cache.class);
187       CommandInitializer initializer = cr.getComponent(CommandInitializer.class);
188       EmbeddedCacheManager cacheManager = cr.getGlobalComponentRegistry().getComponent(EmbeddedCacheManager.class);
189       initializer.setCacheManager(cacheManager);
190 
191       QueryBox queryBox = new QueryBox();
192       queryBox.setCache(cache.getAdvancedCache());
193       cr.registerComponent(queryBox, QueryBox.class);
194 
195       // Register query mbeans
196       registerQueryMBeans(cache.getAdvancedCache(), cr, cacheName);
197    }
198 
199    private void registerQueryMBeans(AdvancedCache cache,
200          ComponentRegistry cr, String cacheName) {
201       Configuration cfg = cache.getCacheConfiguration();
202       SearchIntegrator sf = getSearchFactory(
203             cfg.indexing().properties(), cr);
204 
205       // Resolve MBean server instance
206       GlobalConfiguration globalCfg =
207             cr.getGlobalComponentRegistry().getGlobalConfiguration();
208       mbeanServer = JmxUtil.lookupMBeanServer(globalCfg);
209 
210       // Resolve jmx domain to use for query mbeans
211       String cacheManagerName = cr.getGlobalComponentRegistry().getGlobalConfiguration().globalJmxStatistics().cacheManagerName();
212       String queryGroupName = getQueryGroupName(cacheManagerName, cacheName);
213       jmxDomain = JmxUtil.buildJmxDomain(globalCfg, mbeanServer, queryGroupName);
214 
215       // Register statistics MBean, but only enable if Infinispan config says so
216       InfinispanQueryStatisticsInfo stats = new InfinispanQueryStatisticsInfo(sf);
217       stats.setStatisticsEnabled(cfg.jmxStatistics().enabled());
218       try {
219          ObjectName statsObjName = new ObjectName(
220                jmxDomain + ":" + queryGroupName + ",component=Statistics");
221          JmxUtil.registerMBean(stats, statsObjName, mbeanServer);
222       } catch (Exception e) {
223          throw new CacheException(
224                "Unable to register query module statistics mbean", e);
225       }
226 
227       // Register mass indexer MBean, picking metadata from repo
228       ManageableComponentMetadata massIndexerCompMetadata = cr.getGlobalComponentRegistry().getComponentMetadataRepo()
229             .findComponentMetadata(MassIndexer.class)
230             .toManageableComponentMetadata();
231       try {
232          // TODO: MassIndexer should be some kind of query cache component?
233          DistributedExecutorMassIndexer maxIndexer = new DistributedExecutorMassIndexer(cache, sf);
234          ResourceDMBean mbean = new ResourceDMBean(maxIndexer, massIndexerCompMetadata);
235          ObjectName massIndexerObjName = new ObjectName(jmxDomain + ":"
236                + queryGroupName + ",component=" + massIndexerCompMetadata.getJmxObjectName());
237          JmxUtil.registerMBean(mbean, massIndexerObjName, mbeanServer);
238       } catch (Exception e) {
239          throw new CacheException("Unable to create ", e);
240       }
241    }

243    private String getQueryGroupName(String cacheManagerName, String cacheName) {
244       return "type=Query,manager=" + ObjectName.quote(cacheManagerName) + ",cache=" + ObjectName.quote(cacheName);
245    }

247    private boolean verifyChainContainsQueryInterceptor(ComponentRegistry cr) {
248       InterceptorChain interceptorChain = cr.getComponent(InterceptorChain.class);
249       return interceptorChain.containsInterceptorType(QueryInterceptor.class, true);
250    }

252    private SearchIntegrator getSearchFactory(Properties indexingProperties, ComponentRegistry cr) {
253       Object component = cr.getComponent(SearchIntegrator.class);
254       SearchIntegrator searchFactory = null;
255       if (component instanceof SearchIntegrator) { //could be the placeholder Object REMOVED_REGISTRY_COMPONENT
256          searchFactory = (SearchIntegrator) component;
257       }
258       //defend against multiple initialization:
259       if (searchFactory==null) {
260          GlobalComponentRegistry globalComponentRegistry = cr.getGlobalComponentRegistry();
261          EmbeddedCacheManager uninitializedCacheManager = globalComponentRegistry.getComponent(EmbeddedCacheManager.class);
262          indexingProperties = addProgrammaticMappings(indexingProperties, cr);
263          // Set up the search factory for Hibernate Search first.
264          SearchConfiguration config = new SearchableCacheConfiguration(new Class[0], indexingProperties, uninitializedCacheManager, cr);
265          searchFactory = new SearchFactoryBuilder().configuration(config).buildSearchFactory();
266          cr.registerComponent(searchFactory, SearchIntegrator.class);
267       }
268       return searchFactory;
269    }

271    private Properties addProgrammaticMappings(Properties indexingProperties, ComponentRegistry cr) {
272       Iterator<ProgrammaticSearchMappingProvider> providers = ServiceFinder.load(ProgrammaticSearchMappingProvider.class).iterator();
273       if (providers.hasNext()) {
274          SearchMapping mapping = (SearchMapping) indexingProperties.get(Environment.MODEL_MAPPING);
275          if (mapping == null) {
276             mapping = new SearchMapping();
277             Properties amendedProperties = new Properties();
278             amendedProperties.putAll(indexingProperties);
279             amendedProperties.put(Environment.MODEL_MAPPING, mapping);
280             indexingProperties = amendedProperties;
281          }
282          Cache cache = cr.getComponent(Cache.class);
283          while (providers.hasNext()) {
284             ProgrammaticSearchMappingProvider provider = providers.next();
285             if (log.isDebugEnabled()) {
286                log.debugf("Loading programmatic search mappings for cache %s from provider : %s", cache.getName(), provider.getClass().getName());
287             }
288             provider.defineMappings(cache, mapping);
289          }
290       }
291       return indexingProperties;
292    }

294    @Override
295    public void cacheStopping(ComponentRegistry cr, String cacheName) {
296       final QueryInterceptor queryInterceptor = cr.getComponent(QueryInterceptor.class);
297       if (queryInterceptor != null) {
298          queryInterceptor.prepareForStopping();
299       }
300       //TODO move this to cacheStopped event (won't work right now as the ComponentRegistry is half empty at that point: ISPN-1006)
301       Object searchFactoryIntegrator = cr.getComponent(SearchIntegrator.class);
302       if (searchFactoryIntegrator != null && searchFactoryIntegrator != REMOVED_REGISTRY_COMPONENT) {
303          ((SearchIntegrator) searchFactoryIntegrator).close();
304          //free some memory by de-registering the SearchFactory
305          cr.registerComponent(REMOVED_REGISTRY_COMPONENT, SearchIntegrator.class);
306       }
307 
308       // Unregister MBeans
309       if (mbeanServer != null) {
310          String cacheManagerName = cr.getGlobalComponentRegistry().getGlobalConfiguration().globalJmxStatistics().cacheManagerName();
311          String queryMBeanFilter = jmxDomain + ":" + getQueryGroupName(cacheManagerName, cacheName) + ",*";
312          JmxUtil.unregisterMBeans(queryMBeanFilter, mbeanServer);
313       }
314    }

316    @Override
317    public void cacheStopped(ComponentRegistry cr, String cacheName) {
318       Configuration cfg = cr.getComponent(Configuration.class);
319       removeQueryInterceptorFromConfiguration(cfg);
320    }
321 
322    private void removeQueryInterceptorFromConfiguration(Configuration cfg) {
323 
324       ConfigurationBuilder builder = new ConfigurationBuilder();
325       CustomInterceptorsConfigurationBuilder customInterceptorsBuilder = builder.customInterceptors();
326 
327       for (InterceptorConfiguration interceptorConfig : cfg.customInterceptors().interceptors()) {
328          if (!(interceptorConfig.interceptor() instanceof QueryInterceptor)) {
329             customInterceptorsBuilder.addInterceptor().read(interceptorConfig);
330          }
331       }
332 
333       cfg.customInterceptors().interceptors(builder.build().customInterceptors().interceptors());
334    }

336    @Override
337    public void cacheManagerStarting(GlobalComponentRegistry gcr, GlobalConfiguration globalCfg) {
338       QueryCache queryCache = new QueryCache();
339       gcr.registerComponent(queryCache, QueryCache.class);
340 
341       Map<Integer,AdvancedExternalizer<?>> externalizerMap = globalCfg.serialization().advancedExternalizers();
342       externalizerMap.put(ExternalizerIds.JPA_FILTER_AND_CONVERTER, new JPAFilterAndConverter.JPAFilterAndConverterExternalizer());
343       externalizerMap.put(ExternalizerIds.JPA_FILTER_RESULT, new JPAFilterAndConverter.FilterResultExternalizer());
344       externalizerMap.put(ExternalizerIds.JPA_CACHE_EVENT_FILTER_CONVERTER, new JPACacheEventFilterConverter.Externalizer());
345       externalizerMap.put(ExternalizerIds.LUCENE_QUERY_BOOLEAN, new LuceneBooleanQueryExternalizer());
346       externalizerMap.put(ExternalizerIds.LUCENE_QUERY_TERM, new LuceneTermQueryExternalizer());
347       externalizerMap.put(ExternalizerIds.LUCENE_TERM, new LuceneTermExternalizer());
348       externalizerMap.put(ExternalizerIds.LUCENE_SORT, new LuceneSortExternalizer());
349       externalizerMap.put(ExternalizerIds.LUCENE_SORT_FIELD, new LuceneSortFieldExternalizer());
350       externalizerMap.put(ExternalizerIds.CLUSTERED_QUERY_TOPDOCS, new ClusteredTopDocsExternalizer());
351       externalizerMap.put(ExternalizerIds.LUCENE_TOPDOCS, new LuceneTopDocsExternalizer());
352       externalizerMap.put(ExternalizerIds.LUCENE_FIELD_SCORE_DOC, new LuceneFieldDocExternalizer());
353       externalizerMap.put(ExternalizerIds.LUCENE_SCORE_DOC, new LuceneScoreDocExternalizer());
354       externalizerMap.put(ExternalizerIds.LUCENE_TOPFIELDDOCS, new LuceneTopFieldDocsExternalizer());
355       externalizerMap.put(ExternalizerIds.LUCENE_QUERY_MATCH_ALL, new LuceneMatchAllQueryExternalizer());
356       externalizerMap.put(ExternalizerIds.INDEX_WORKER, new IndexWorker.Externalizer());
357    }

1   package org.infinispan.client.hotrod;
2   
3   import java.util.Map;
4   
5   /**
6    * Defines the possible list of statistics defined by the Hot Rod server.
7    * Can be obtained through {@link RemoteCache#stats()}
8    *
9    * @author Mircea.Markus@jboss.com
10   * @since 4.1
11   */
12  public interface ServerStatistics {
13  
14     /**
15      * Number of seconds since Hot Rod started.
16      */
17     String TIME_SINCE_START = "timeSinceStart";
18  
19     /**
20      * Number of entries currently in the Hot Rod server
21      */
22     String CURRENT_NR_OF_ENTRIES = "currentNumberOfEntries";
23  
24     /**
25      * Number of entries stored in Hot Rod server
26      * since the server started running.
27      */
28     String TOTAL_NR_OF_ENTRIES = "totalNumberOfEntries";
29  
30     /**
31      * Number of put operations.
32      */
33     String STORES = "stores";
34  
35  
36     /**
37      * Number of get operations.
38      */
39     String RETRIEVALS = "retrievals";
40  
41     /**
42      * Number of get hits.
43      */
44     String HITS = "hits";
45  
46     /**
47      * Number of get misses.
48      */
49     String MISSES = "misses";
50  
51  
52     /**
53      * Number of removal hits.
54      */
55     String REMOVE_HITS = "removeHits";
56  
57     /**
58      * Number of removal misses.
59      */
60     String REMOVE_MISSES = "removeMisses";
61  
62     Map<String, String> getStatsMap();
63  
64     String getStatistic(String statsName);
65  
66     Integer getIntStatistic(String statsName);
67  }

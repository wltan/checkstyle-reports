1   package org.jboss.as.clustering.jgroups.subsystem;
2   
3   import org.jboss.as.controller.ReloadRequiredWriteAttributeHandler;
4   import org.jboss.as.controller.SimpleAttributeDefinition;
5   import org.jboss.as.controller.SimpleAttributeDefinitionBuilder;
6   import org.jboss.as.controller.SimpleResourceDefinition;
7   import org.jboss.as.controller.registry.AttributeAccess;
8   import org.jboss.as.controller.registry.ManagementResourceRegistration;
9   import org.jboss.dmr.ModelType;
10  
11  /**
12   * The root resource of the JGroups subsystem.
13   *
14   * @author Richard Achmatowicz (c) 2011 Red Hat Inc.
15   */
16  public class JGroupsSubsystemRootResource extends SimpleResourceDefinition {
17  
18      // attributes
19      static SimpleAttributeDefinition DEFAULT_STACK =
20              new SimpleAttributeDefinitionBuilder(ModelKeys.DEFAULT_STACK, ModelType.STRING, false)
21                      .setXmlName(Attribute.DEFAULT_STACK.getLocalName())
22                      .setAllowExpression(true)
23                      .setFlags(AttributeAccess.Flag.RESTART_ALL_SERVICES)
24                      .build();
25  
26      // registration
27      JGroupsSubsystemRootResource() {
28          super(JGroupsExtension.SUBSYSTEM_PATH,
29                  JGroupsExtension.getResourceDescriptionResolver(),
30                  JGroupsSubsystemAdd.INSTANCE,
31                  JGroupsSubsystemRemove.INSTANCE);
32      }
33  
34      @Override
35      public void registerOperations(ManagementResourceRegistration resourceRegistration) {
36          super.registerOperations(resourceRegistration);
37      }
38  
39      @Override
40      public void registerAttributes(ManagementResourceRegistration resourceRegistration) {
41          super.registerAttributes(resourceRegistration);
42          resourceRegistration.registerReadWriteAttribute(DEFAULT_STACK, null, new ReloadRequiredWriteAttributeHandler(DEFAULT_STACK));
43      }
44  
45      @Override
46      public void registerChildren(ManagementResourceRegistration resourceRegistration) {
47          super.registerChildren(resourceRegistration);
48      }
49  
50  }

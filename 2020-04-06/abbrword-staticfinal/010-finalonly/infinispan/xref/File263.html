1   package org.infinispan.notifications.cachelistener.filter;
2   
3   import org.infinispan.commons.marshall.AbstractExternalizer;
4   import org.infinispan.commons.util.Util;
5   import org.infinispan.factories.ComponentRegistry;
6   import org.infinispan.factories.annotations.Inject;
7   import org.infinispan.filter.KeyValueFilter;
8   import org.infinispan.marshall.core.Ids;
9   import org.infinispan.metadata.Metadata;
10  import org.infinispan.notifications.cachelistener.event.Event;
11  
12  import java.io.IOException;
13  import java.io.ObjectInput;
14  import java.io.ObjectOutput;
15  import java.util.Set;
16  
17  /**
18   * KeyValueFilter that is implemented by using the provided CacheEventFilter.  The provided event type will always be
19   * one that is not retried, post and of type CREATE,  The old value and old metadata in both pre and post events will
20   * be the data that was in the cache before the event occurs.  The new value and new metadata in both pre and post
21   * events will be the data that is in the cache after the event occurs.
22   *
23   * @author wburns
24   * @since 7.0
25   */
26  public class CacheEventFilterAsKeyValueFilter<K, V> implements KeyValueFilter<K, V> {
27     private static final EventType CREATE_EVENT = new EventType(false, false, Event.Type.CACHE_ENTRY_CREATED);
28  
29     private final CacheEventFilter<K, V> filter;
30  
31     public CacheEventFilterAsKeyValueFilter(CacheEventFilter<K, V> filter) {
32        this.filter = filter;
33     }
34  
35     @Override
36     public boolean accept(K key, V value, Metadata metadata) {
37        return filter.accept(key, null, null, value, metadata, CREATE_EVENT);
38     }
39  
40     @Inject
41     protected void injectDependencies(ComponentRegistry cr) {
42        cr.wireDependencies(filter);
43     }
44  
45     public static class Externalizer extends AbstractExternalizer<CacheEventFilterAsKeyValueFilter> {
46        @Override
47        public Set<Class<? extends CacheEventFilterAsKeyValueFilter>> getTypeClasses() {
48           return Util.<Class<? extends CacheEventFilterAsKeyValueFilter>>asSet(CacheEventFilterAsKeyValueFilter.class);
49        }
50  
51        @Override
52        public void writeObject(ObjectOutput output, CacheEventFilterAsKeyValueFilter object) throws IOException {
53           output.writeObject(object.filter);
54        }
55  
56        @Override
57        public CacheEventFilterAsKeyValueFilter readObject(ObjectInput input) throws IOException, ClassNotFoundException {
58           return new CacheEventFilterAsKeyValueFilter((CacheEventFilter)input.readObject());
59        }
60  
61        @Override
62        public Integer getId() {
63           return Ids.CACHE_EVENT_FILTER_AS_KEY_VALUE_FILTER;
64        }
65     }
66  }

22  package org.jboss.as.clustering.jgroups.subsystem;
23  
24  import static org.jboss.as.controller.descriptions.ModelDescriptionConstants.OP_ADDR;
25  import static org.jboss.as.controller.descriptions.ModelDescriptionConstants.REMOVE;
26  
27  import java.util.List;
28  
29  import org.jboss.as.controller.AbstractRemoveStepHandler;
30  import org.jboss.as.controller.OperationContext;
31  import org.jboss.as.controller.OperationFailedException;
32  import org.jboss.as.controller.PathAddress;
33  import org.jboss.as.controller.operations.common.Util;
34  import org.jboss.dmr.ModelNode;
35  import org.jboss.dmr.Property;
36  import org.jboss.msc.service.ServiceName;
37  
38  /**
39   * Handler for JGroups subsystem add operations.
40   *
41   * @author Kabir Khan
42   */
43  public class JGroupsSubsystemRemove extends AbstractRemoveStepHandler {
44  
45      public static final JGroupsSubsystemRemove INSTANCE = new JGroupsSubsystemRemove();
46  
47      @Override
48      protected void performRemove(OperationContext context, ModelNode operation, ModelNode model) throws OperationFailedException {
49          final PathAddress opAddress = PathAddress.pathAddress(operation.require(OP_ADDR));
50          // Add a step *on the top of the stack* to actually remove the subsystem resource
51          ModelNode removeSubsystem = Util.createOperation(REMOVE, opAddress);
52          context.addStep(removeSubsystem, new OriginalSubsystemRemoveHandler(), OperationContext.Stage.MODEL, true);
53  
54          // Now add steps on top of the one added above to remove any existing child stacks
55          if (model.hasDefined(ModelKeys.STACK)) {
56              List<Property> stacks = model.get(ModelKeys.STACK).asPropertyList() ;
57              for (Property stack: stacks) {
58                  PathAddress address = opAddress.append(ModelKeys.STACK, stack.getName());
59                  ModelNode removeStack = Util.createOperation(REMOVE, address);
60                  // remove the stack
61                  context.addStep(removeStack, ProtocolStackRemove.INSTANCE, OperationContext.Stage.MODEL, true);
62              }
63          }
64  
65          context.stepCompleted();
66      }
67  
68      static class OriginalSubsystemRemoveHandler extends AbstractRemoveStepHandler {
69  
70          @Override
71          protected void performRuntime(OperationContext context, ModelNode operation, ModelNode model)
72              throws OperationFailedException {
73              removeRuntimeServices(context, operation, model);
74          }
75  
76          protected void removeRuntimeServices(OperationContext context, ModelNode operation, ModelNode model)
77                  throws OperationFailedException {
78  
79              // remove the ProtocolDefaultsService
80              ServiceName protocolDefaultsService = ProtocolDefaultsService.SERVICE_NAME;
81              context.removeService(protocolDefaultsService);
82  
83              // remove the DefaultChannelFactoryServiceAlias
84              ServiceName defaultChannelFactoryService = ChannelFactoryService.getServiceName(null);
85              context.removeService(defaultChannelFactoryService);
86          }
87      }
88  
89  }

195     @Test
196     public void test_hijrahDateLimits() {
197         HijrahChronology chrono = HijrahChronology.INSTANCE;
198         ValueRange yearRange = chrono.range(YEAR);
199         ValueRange monthRange = chrono.range(MONTH_OF_YEAR);
200         ValueRange dayRange = chrono.range(DAY_OF_MONTH);
201 
202         HijrahDate xx = chrono.date(1434, 1, 1);
203         HijrahDate minDate = chrono.date((int)yearRange.getLargestMinimum(),
204                 (int)monthRange.getMinimum(), (int)dayRange.getMinimum());
205         try {
206             HijrahDate before = minDate.minus(1, ChronoUnit.DAYS);
207             fail("Exception did not occur, minDate: " + minDate + ".minus(1, DAYS) = " + before);
208 
209         } catch (DateTimeException ex) {
210             // ignore, this exception was expected
211         }
212 
213         HijrahDate maxDate = chrono.date((int)yearRange.getSmallestMaximum(),
214                 (int)monthRange.getMaximum(), 1);
215         int monthLen = maxDate.lengthOfMonth();
216         maxDate = maxDate.with(DAY_OF_MONTH, monthLen);
217         try {
218             HijrahDate after = maxDate.plus(1, ChronoUnit.DAYS);
219             fail("Exception did not occur, maxDate: " + maxDate + ".plus(1, DAYS) = " + after);
220         } catch (DateTimeException ex) {
221             // ignore, this exception was expected

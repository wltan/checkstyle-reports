42  @DemoProperties(
43          value = "Slider Demo",
44          category = "Controls",
45          description = "Demonstrates the JSlider, a control which supports linear adjustment",
46          sourceFiles = {
47              "com/sun/swingset3/demos/slider/SliderDemo.java",

53  public class SliderDemo extends JPanel {
54  
55      private static final Dimension HGAP5 = new Dimension(5, 1);
56      private static final Dimension VGAP5 = new Dimension(1, 5);
57      private static final Dimension HGAP10 = new Dimension(10, 1);
58      private static final Dimension VGAP10 = new Dimension(1, 10);
59      private static final Dimension HGAP20 = new Dimension(20, 1);
60      private static final Dimension HGAP25 = new Dimension(25, 1);
61      private final ResourceManager resourceManager = new ResourceManager(this.getClass());
62      public static final String DEMO_TITLE = SliderDemo.class.getAnnotation(DemoProperties.class).value();
63      public static final String HORIZONTAL_PLAIN_SLIDER = "Horizontal Plain Slider";
64      public static final String HORIZONTAL_MAJOR_TICKS_SLIDER = "Horizontal Major Ticks Slider";
65      public static final String HORIZONTAL_MINOR_TICKS_SLIDER = "Horizontal Minor Ticks Slider";
66      public static final String HORIZONTAL_DISABLED_SLIDER = "Horizontal Disabled Slider";
67      public static final String VERTICAL_PLAIN_SLIDER = "Vertical Plain Slider";
68      public static final String VERTICAL_MAJOR_TICKS_SLIDER = "Vertical Major Ticks Slider";
69      public static final String VERTICAL_MINOR_TICKS_SLIDER = "Vertical Minor Ticks Slider";
70      public static final String VERTICAL_DISABLED_SLIDER = "Vertical Disabled Slider";
71  
72      /**
73       * main method allows us to run as a standalone demo.
74       */
75      public static void main(String[] args) {
76          JFrame frame = new JFrame(SliderDemo.class.getAnnotation(DemoProperties.class).value());
77          frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
78          frame.getContentPane().add(new SliderDemo());
79          frame.setPreferredSize(new Dimension(800, 600));
80          frame.pack();
81          frame.setLocationRelativeTo(null);
82          frame.setVisible(true);
83      }

85      /**
86       * SliderDemo Constructor
87       */
88      public SliderDemo() {
89          createSliderDemo();
90      }

92      private void createSliderDemo() {
93          JSlider s;
94          JPanel hp;
95          JPanel vp;
96          GridLayout g;
97          JPanel tp;
98          JLabel tf;
99          ChangeListener listener;
100 
101         setLayout(new BorderLayout());
102         tf = new JLabel(resourceManager.getString("SliderDemo.slidervalue"));
103         add(tf, BorderLayout.SOUTH);
104         tp = new JPanel();
105         g = new GridLayout(1, 2);
106         g.setHgap(5);
107         g.setVgap(5);
108         tp.setLayout(g);
109         add(tp, BorderLayout.CENTER);
110         listener = new SliderListener(tf);
111         BevelBorder border = new BevelBorder(BevelBorder.LOWERED);
112         hp = new JPanel();
113         hp.setLayout(new BoxLayout(hp, BoxLayout.Y_AXIS));
114         hp.setBorder(new TitledBorder(
115                 border,
116                 resourceManager.getString("SliderDemo.horizontal"),
117                 TitledBorder.LEFT,
118                 TitledBorder.ABOVE_TOP));
119         tp.add(hp);
120         vp = new JPanel();
121         vp.setLayout(new BoxLayout(vp, BoxLayout.X_AXIS));
122         vp.setBorder(new TitledBorder(
123                 border,
124                 resourceManager.getString("SliderDemo.vertical"),
125                 TitledBorder.LEFT,
126                 TitledBorder.ABOVE_TOP));
127         tp.add(vp);
128 
129         // Horizontal Slider 1
130         JPanel p = new JPanel();
131         p.setLayout(new BoxLayout(p, BoxLayout.Y_AXIS));
132         p.setBorder(new TitledBorder(resourceManager.getString("SliderDemo.plain")));
133         s = new JSlider(-10, 100, 20);
134         s.getAccessibleContext().setAccessibleName(HORIZONTAL_PLAIN_SLIDER);
135         s.getAccessibleContext().setAccessibleDescription(resourceManager.getString("SliderDemo.a_plain_slider"));
136         s.addChangeListener(listener);
137         p.add(Box.createRigidArea(VGAP5));
138         p.add(s);
139         p.add(Box.createRigidArea(VGAP5));
140         hp.add(p);
141         hp.add(Box.createRigidArea(VGAP10));
142 
143         // Horizontal Slider 2
144         p = new JPanel();
145         p.setLayout(new BoxLayout(p, BoxLayout.Y_AXIS));
146         p.setBorder(new TitledBorder(resourceManager.getString("SliderDemo.majorticks")));
147         s = new JSlider(100, 1000, 400);
148         s.setPaintTicks(true);
149         s.setMajorTickSpacing(100);
150         s.getAccessibleContext().setAccessibleName(HORIZONTAL_MAJOR_TICKS_SLIDER);
151         s.getAccessibleContext().setAccessibleDescription(resourceManager.getString("SliderDemo.majorticksdescription"));
152         s.addChangeListener(listener);
153         p.add(Box.createRigidArea(VGAP5));
154         p.add(s);
155         p.add(Box.createRigidArea(VGAP5));
156         hp.add(p);
157         hp.add(Box.createRigidArea(VGAP10));
158 
159         // Horizontal Slider 3
160         p = new JPanel();
161         p.setLayout(new BoxLayout(p, BoxLayout.Y_AXIS));
162         p.setBorder(new TitledBorder(resourceManager.getString("SliderDemo.ticks")));
163         s = new JSlider(0, 11, 6);
164         s.putClientProperty("JSlider.isFilled", Boolean.TRUE);
165         s.setPaintTicks(true);
166         s.setMajorTickSpacing(5);
167         s.setMinorTickSpacing(1);
168         s.setPaintLabels(true);
169         s.setSnapToTicks(true);
170         s.getLabelTable().put(new Integer(11), new JLabel(Integer.toString(11), JLabel.CENTER));
171         s.setLabelTable(s.getLabelTable());
172         s.getAccessibleContext().setAccessibleName(HORIZONTAL_MINOR_TICKS_SLIDER);
173         s.getAccessibleContext().setAccessibleDescription(resourceManager.getString("SliderDemo.minorticksdescription"));
174         s.addChangeListener(listener);
175         p.add(Box.createRigidArea(VGAP5));
176         p.add(s);
177         p.add(Box.createRigidArea(VGAP5));
178         hp.add(p);
179         hp.add(Box.createRigidArea(VGAP10));
180 
181         // Horizontal Slider 4
182         p = new JPanel();
183         p.setLayout(new BoxLayout(p, BoxLayout.Y_AXIS));
184         p.setBorder(new TitledBorder(resourceManager.getString("SliderDemo.disabled")));
185         BoundedRangeModel brm = new DefaultBoundedRangeModel(80, 0, 0, 100);
186         s = new JSlider(brm);
187         s.setPaintTicks(true);
188         s.setMajorTickSpacing(20);
189         s.setMinorTickSpacing(5);
190         s.setEnabled(false);
191         s.getAccessibleContext().setAccessibleName(HORIZONTAL_DISABLED_SLIDER);
192         s.getAccessibleContext().setAccessibleDescription(resourceManager.getString("SliderDemo.disableddescription"));
193         s.addChangeListener(listener);
194         p.add(Box.createRigidArea(VGAP5));
195         p.add(s);
196         p.add(Box.createRigidArea(VGAP5));
197         hp.add(p);
198 
199         //////////////////////////////////////////////////////////////////////////////
200         // Vertical Slider 1
201         p = new JPanel();
202         p.setLayout(new BoxLayout(p, BoxLayout.X_AXIS));
203         p.setBorder(new TitledBorder(resourceManager.getString("SliderDemo.plain")));
204         s = new JSlider(JSlider.VERTICAL, -10, 100, 20);
205         s.getAccessibleContext().setAccessibleName(VERTICAL_PLAIN_SLIDER);
206         s.getAccessibleContext().setAccessibleDescription(resourceManager.getString("SliderDemo.a_plain_slider"));
207         s.addChangeListener(listener);
208         p.add(Box.createRigidArea(HGAP10));
209         p.add(s);
210         p.add(Box.createRigidArea(HGAP10));
211         vp.add(p);
212         vp.add(Box.createRigidArea(HGAP10));
213 
214         // Vertical Slider 2
215         p = new JPanel();
216         p.setLayout(new BoxLayout(p, BoxLayout.X_AXIS));
217         p.setBorder(new TitledBorder(resourceManager.getString("SliderDemo.majorticks")));
218         s = new JSlider(JSlider.VERTICAL, 100, 1000, 400);
219         s.putClientProperty("JSlider.isFilled", Boolean.TRUE);
220         s.setPaintTicks(true);
221         s.setMajorTickSpacing(100);
222         s.getAccessibleContext().setAccessibleName(VERTICAL_MAJOR_TICKS_SLIDER);
223         s.getAccessibleContext().setAccessibleDescription(resourceManager.getString("SliderDemo.majorticksdescription"));
224         s.addChangeListener(listener);
225         p.add(Box.createRigidArea(HGAP25));
226         p.add(s);
227         p.add(Box.createRigidArea(HGAP25));
228         vp.add(p);
229         vp.add(Box.createRigidArea(HGAP5));
230 
231         // Vertical Slider 3
232         p = new JPanel();
233         p.setLayout(new BoxLayout(p, BoxLayout.X_AXIS));
234         p.setBorder(new TitledBorder(resourceManager.getString("SliderDemo.minorticks")));
235         s = new JSlider(JSlider.VERTICAL, 0, 100, 60);
236         s.setPaintTicks(true);
237         s.setMajorTickSpacing(20);
238         s.setMinorTickSpacing(5);
239         s.setPaintLabels(true);
240         s.getAccessibleContext().setAccessibleName(VERTICAL_MINOR_TICKS_SLIDER);
241         s.getAccessibleContext().setAccessibleDescription(resourceManager.getString("SliderDemo.minorticksdescription"));
242         s.addChangeListener(listener);
243         p.add(Box.createRigidArea(HGAP10));
244         p.add(s);
245         p.add(Box.createRigidArea(HGAP10));
246         vp.add(p);
247         vp.add(Box.createRigidArea(HGAP5));
248 
249         // Vertical Slider 4
250         p = new JPanel();
251         p.setLayout(new BoxLayout(p, BoxLayout.X_AXIS));
252         p.setBorder(new TitledBorder(resourceManager.getString("SliderDemo.disabled")));
253         s = new JSlider(JSlider.VERTICAL, 0, 100, 80);
254         s.setPaintTicks(true);
255         s.setMajorTickSpacing(20);
256         s.setMinorTickSpacing(5);
257         s.setEnabled(false);
258         s.getAccessibleContext().setAccessibleName(VERTICAL_DISABLED_SLIDER);
259         s.getAccessibleContext().setAccessibleDescription(resourceManager.getString("SliderDemo.disableddescription"));
260         s.addChangeListener(listener);
261         p.add(Box.createRigidArea(HGAP20));
262         p.add(s);
263         p.add(Box.createRigidArea(HGAP20));
264         vp.add(p);
265     }
266 
267     private class SliderListener implements ChangeListener {
268 
269         private final JLabel tf;
270 
271         public SliderListener(JLabel f) {
272             tf = f;
273         }
274 
275         public void stateChanged(ChangeEvent e) {
276             JSlider s1 = (JSlider) e.getSource();
277             tf.setText(resourceManager.getString("SliderDemo.slidervalue") + s1.getValue());
278         }
279     }

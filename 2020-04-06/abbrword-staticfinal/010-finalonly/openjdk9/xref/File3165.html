24  /* @test
25   * @bug 6364692
26   * @summary When the "java.rmi.server.randomIDs" system property is
27   * not defined, the ObjID() constructor should behave as if it were
28   * set to "true" and generate random object numbers; if the property
29   * is defined to something other than "true" (ignoring case), then
30   * ObjID() should still generate sequential object numbers.
31   * @author Peter Jones
32   *
33   * @run main/othervm RandomIDs random
34   * @run main/othervm -Djava.rmi.server.randomIDs=true RandomIDs random
35   * @run main/othervm -Djava.rmi.server.randomIDs=false RandomIDs sequential
36   */
37  
38  import java.io.IOException;
39  import java.io.ObjectOutputStream;
40  import java.io.OutputStream;
41  import java.rmi.server.ObjID;
42  
43  public class RandomIDs {
44      private static final int COUNT = 10000;
45  
46      public static void main(String[] args) throws Exception {
47          boolean shouldBeRandom = false;
48          boolean shouldBeSequential = false;
49  
50          String usage = "Usage: java RandomIDs [random|sequential]";
51          if (args.length != 1) {
52              System.err.println(usage);
53              throw new Error("wrong number of arguments");
54          } else if (args[0].equals("random")) {
55              shouldBeRandom = true;
56          } else if (args[0].equals("sequential")) {
57              shouldBeSequential = true;
58          } else {
59              System.err.println(usage);
60              throw new Error("invalid argument");
61          }
62  
63          System.err.println("\nRegression test for bug 6364692\n");
64          String propertyValue = System.getProperty("java.rmi.server.randomIDs");
65          System.err.println(
66              "Value of java.rmi.server.randomIDs system property: " +
67              (propertyValue != null ? "\"" + propertyValue + "\"" : null));
68          System.err.println(
69              "Expecting object numbers of unique ObjIDs to be: " + args[0]);
70  
71          /*
72           * Get the 64-bit "object number" component of COUNT number of
73           * unique (not "well-known") ObjID instances created in
74           * sequence, by writing each to a dummy ObjectOutputStream and
75           * trapping the first writeLong invocation on the stream.
76           */
77          final long[] objnums = new long[COUNT];
78          for (int i = 0; i < COUNT; i++) {
79              final int j = i;
80              class Escape extends RuntimeException { }
81              try {
82                  new ObjID().write(new ObjectOutputStream(new OutputStream() {
83                      public void write(int b) { }
84                  }) {
85                      public void writeLong(long val) throws IOException {
86                          objnums[j] = val;
87                          throw new Escape();
88                      }
89                  });
90                  throw new Error("writeLong not invoked");
91              } catch (Escape e) {
92              }
93          }
94  
95          /*
96           * If the object numbers should be random, then verify that
97           * they are.  (This verification is certainly not a thorough
98           * evaluation of randomness, but it performs a couple of
99           * simple checks to catch mistakes in ObjID's application of a
100          * CSPRNG: are roughly half the bits set, and can the sequence
101          * be used to get a rough Monte Carlo estimate of pi.  Errors
102          * up to 5% are tolerated for both checks.)
103          */
104         if (shouldBeRandom) {
105             int bitCount = 0;
106             int piHitCount = 0;
107             for (int i = 0; i < COUNT; i++) {
108                 bitCount += Long.bitCount(objnums[i]);
109                 double x = ((double) (objnums[i] >>> 32)) / (1L << 32);
110                 double y = ((double) (objnums[i] & 0xFFFFFFFFL)) / (1L << 32);
111                 if (((x * x) + (y * y)) <= 1.0) {
112                     piHitCount++;
113                 }
114             }
115 
116             int bitCountTarget = COUNT * 32;
117             double bitCountError =
118                 ((double) (bitCount - bitCountTarget)) / bitCountTarget;
119             if (Math.abs(bitCountError) > 0.05) { // tolerate 5% error
120                 throw new Error("TEST FAILED: " +
121                                 "bitCount == " + bitCount);
122             }
123 
124             double piEstimate = ((double) piHitCount / COUNT) * 4.0;
125             double piEstimateError = (piEstimate - Math.PI) / Math.PI;
126             if (Math.abs(piEstimateError) > 0.05) { // tolerate 5% error
127                 throw new Error("TEST FAILED: " +
128                                 "piEstimate == " + piEstimate);
129             }
130         }
131 
132         /*
133          * If the object numbers should be sequential, then verify
134          * that they are.
135          */
136         if (shouldBeSequential) {
137             long first = objnums[0];
138             /*
139              * This test currently verifies that the first object
140              * number is zero, but that could be false if one or more
141              * remote objects get exported as part of VM startup-- if
142              * that starts happening, this check could be relaxed.
143              */
144             if (first != 0) {
145                 throw new Error("TEST FAILED: " +
146                                 "first object number == " + first +
147                                 " (not zero)");
148             }
149             for (int i = 1; i < COUNT; i++) {
150                 if (objnums[i] != first + i) {
151                     throw new Error("TEST FAILED: first == " + first + ", " +
152                                     "objnums[" + i + "] == " + objnums[i]);
153                 }
154             }
155         }
156 
157         System.err.println("TEST PASSED");
158     }
159 }

24  /*
25   * @test
26   * @bug     6314913
27   * @summary Basic Test for HotSpotDiagnosticMXBean.setVMOption()
28   *          and getDiagnosticOptions().
29   * @author  Mandy Chung
30   * @author  Jaroslav Bachorik
31   *
32   * @run main/othervm -XX:+HeapDumpOnOutOfMemoryError SetVMOption
33   */
34  
35  import java.lang.management.ManagementFactory;
36  import java.util.*;
37  import com.sun.management.HotSpotDiagnosticMXBean;
38  import com.sun.management.VMOption;
39  import com.sun.management.VMOption.Origin;
40  
41  public class SetVMOption {
42      private static final String HEAP_DUMP_ON_OOM = "HeapDumpOnOutOfMemoryError";
43      private static final String EXPECTED_VALUE = "true";
44      private static final String BAD_VALUE = "yes";
45      private static final String NEW_VALUE = "false";
46      private static final String MANAGEMENT_SERVER = "ManagementServer";
47      private static HotSpotDiagnosticMXBean mbean;
48  
49      public static void main(String[] args) throws Exception {
50          mbean =
51              ManagementFactory.getPlatformMXBean(HotSpotDiagnosticMXBean.class);
52  
53          VMOption option = findHeapDumpOnOomOption();
54          if (!option.getValue().equalsIgnoreCase(EXPECTED_VALUE)) {
55              throw new RuntimeException("Unexpected value: " +
56                  option.getValue() + " expected: " + EXPECTED_VALUE);
57          }
58          if (option.getOrigin() != Origin.VM_CREATION) {
59              throw new RuntimeException("Unexpected origin: " +
60                  option.getOrigin() + " expected: VM_CREATION");
61          }
62          if (!option.isWriteable()) {
63              throw new RuntimeException("Expected " + HEAP_DUMP_ON_OOM +
64                  " to be writeable");
65          }
66  
67          // set VM option to a new value
68          mbean.setVMOption(HEAP_DUMP_ON_OOM, NEW_VALUE);
69  
70          option = findHeapDumpOnOomOption();
71          if (!option.getValue().equalsIgnoreCase(NEW_VALUE)) {
72              throw new RuntimeException("Unexpected value: " +
73                  option.getValue() + " expected: " + NEW_VALUE);
74          }
75          if (option.getOrigin() != Origin.MANAGEMENT) {
76              throw new RuntimeException("Unexpected origin: " +
77                  option.getOrigin() + " expected: MANAGEMENT");
78          }
79          VMOption o = mbean.getVMOption(HEAP_DUMP_ON_OOM);
80          if (!option.getValue().equals(o.getValue())) {
81              throw new RuntimeException("Unmatched value: " +
82                  option.getValue() + " expected: " + o.getValue());
83          }
84          if (!option.getValue().equals(o.getValue())) {
85              throw new RuntimeException("Unmatched value: " +
86                  option.getValue() + " expected: " + o.getValue());
87          }
88          if (option.getOrigin() != o.getOrigin()) {
89              throw new RuntimeException("Unmatched origin: " +
90                  option.getOrigin() + " expected: " + o.getOrigin());
91          }
92          if (option.isWriteable() != o.isWriteable()) {
93              throw new RuntimeException("Unmatched writeable: " +
94                  option.isWriteable() + " expected: " + o.isWriteable());
95          }
96  
97          // check if ManagementServer is not writeable
98          List<VMOption> options = mbean.getDiagnosticOptions();
99          VMOption mgmtServerOption = null;
100         for (VMOption o1 : options) {
101             if (o1.getName().equals(MANAGEMENT_SERVER)) {
102                  mgmtServerOption = o1;
103                  break;
104             }
105         }
106         if (mgmtServerOption != null) {
107             throw new RuntimeException(MANAGEMENT_SERVER +
108                 " is not expected to be writeable");
109         }
110         mgmtServerOption = mbean.getVMOption(MANAGEMENT_SERVER);
111         if (mgmtServerOption == null) {
112             throw new RuntimeException(MANAGEMENT_SERVER +
113                 " should exist.");
114         }
115         if (mgmtServerOption.getOrigin() != Origin.DEFAULT) {
116             throw new RuntimeException(MANAGEMENT_SERVER +
117                 " should have the default value.");
118         }
119         if (mgmtServerOption.isWriteable()) {
120             throw new RuntimeException(MANAGEMENT_SERVER +
121                 " is not expected to be writeable");
122         }
123     }
124 
125     public static VMOption findHeapDumpOnOomOption() {
126         List<VMOption> options = mbean.getDiagnosticOptions();
127         VMOption gcDetails = null;
128         for (VMOption o : options) {
129             if (o.getName().equals(HEAP_DUMP_ON_OOM)) {
130                  gcDetails = o;
131                  break;
132             }
133         }
134         if (gcDetails == null) {
135             throw new RuntimeException("VM option " + HEAP_DUMP_ON_OOM +
136                 " not found");
137         }
138         return gcDetails;
139     }
140 }

557     static class PersonValidator {
558 
559         private final String validNameChars = "[a-zA-Z ]";
560         private final int minAge = 0;
561 
562         public Validation<Seq<String>, Person> validatePerson(String name, int age) {
563             return Validation.combine(validateName(name), validateAge(age)).ap(Person::new);
564         }
565 
566         private Validation<String, String> validateName(String name) {
567             return CharSeq.of(name).replaceAll(validNameChars, "").transform(seq ->
568                     seq.isEmpty() ? Validation.<String, String> valid(name)
569                                   : Validation.<String, String> invalid("Name contains invalid characters: '" + seq.distinct().sorted() + "'"));
570         }
571 
572         private Validation<String, Integer> validateAge(int age) {
573             return (age < minAge) ? Validation.invalid("Age must be greater than 0")
574                                   : Validation.valid(age);
575         }
576     }

578     static class Person {
579 
580         public final String name;
581         public final int age;
582 
583         public Person(String name, int age) {
584             this.name = name;
585             this.age = age;
586         }
587 
588         @Override
589         public boolean equals(Object o) {
590             if (o == this) {
591                 return true;
592             } else if (o instanceof Person) {
593                 final Person person = (Person) o;
594                 return Objects.equals(name, person.name) && age == person.age;
595             } else {
596                 return false;
597             }
598         }
599 
600         @Override
601         public int hashCode() {
602             return Objects.hash(name, age);
603         }
604 
605         @Override
606         public String toString() {
607             return "Person(" + name + ", " + age + ")";
608         }
609     }

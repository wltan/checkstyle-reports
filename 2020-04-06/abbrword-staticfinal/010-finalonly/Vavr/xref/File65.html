16  import static java.lang.Math.floor;

19  import static org.assertj.core.api.Assertions.assertThat;
20  
21  public class Euler39Test {
22  
23      /**
24       * <strong>Problem 39 Integer right triangles</strong>
25       * <p>
26       * If <i>p</i> is the perimeter of a right angle triangle with integral
27       * length sides, {<i>a,b,c</i>}, there are exactly three solutions for
28       * <i>p</i> = 120.
29       * <p>
30       * {20,48,52}, {24,45,51}, {30,40,50}
31       * <p>
32       * For which value of <i>p</i> ≤ 1000, is the number of solutions maximised?
33       * <p>
34       * See also <a href="https://projecteuler.net/problem=39">projecteuler.net
35       * problem 39</a>.
36       */
37      @Test
38      public void shouldSolveProblem39() {
39          assertThat(SOLUTIONS_FOR_PERIMETERS_UP_TO_1000.get(120)).isEqualTo(some(List.of(Tuple.of(20, 48, 52), Tuple.of(24, 45, 51), Tuple.of(30, 40, 50))));
40  
41          assertThat(perimeterUpTo1000WithMaximisedNumberOfSolutions()).isEqualTo(840);
42      }
43  
44      private static int perimeterUpTo1000WithMaximisedNumberOfSolutions() {
45          return SOLUTIONS_FOR_PERIMETERS_UP_TO_1000
46                  .map((perimeter, listOfSolutions) -> Tuple.of(perimeter, listOfSolutions.length()))
47                  .maxBy(Tuple2::_2)
48                  .get()._1;
49      }
50  
51      private static final Map<Integer, List<Tuple3<Integer, Integer, Integer>>> SOLUTIONS_FOR_PERIMETERS_UP_TO_1000
52              = List.rangeClosed(1, 500)
53              .flatMap(a -> List.rangeClosed(a, 500)
54                      .map(b -> Tuple.of(a, b, hypot(a, b))))
55              .filter(t -> floor(t._3) == t._3)
56              .map(t -> t.map3(Double::intValue))
57              .groupBy(t -> t.apply((a, b, c) -> a + b + c))
58              .filterKeys(d -> d <= 1_000);
59  }

107     private <K extends Comparable<? super K>, V> void testBigData(int count, Function<Tuple2<Integer, Integer>, Tuple2<K, V>> mapper) {
108         final Comparator<K, V> cmp = new Comparator<>();
109         final java.util.Map<K, V> rnd = rnd(count, mapper);
110         for (java.util.Map.Entry<K, V> e : rnd.entrySet()) {
111             cmp.set(e.getKey(), e.getValue());
112         }
113         cmp.test();
114         for (K key : new java.util.TreeSet<>(rnd.keySet())) {
115             rnd.remove(key);
116             cmp.remove(key);
117         }
118         cmp.test();
119     }

249     private class WeakInteger implements Comparable<WeakInteger> {
250         final int value;
251 
252         @Override
253         public boolean equals(Object o) {
254             if (this == o) { return true; }
255             if (o == null || getClass() != o.getClass()) { return false; }
256             final WeakInteger that = (WeakInteger) o;
257             return value == that.value;
258         }
259 
260         WeakInteger(int value) {
261             this.value = value;
262         }
263 
264         @Override
265         public int hashCode() {
266             return Math.abs(value) % 10;
267         }
268 
269         @Override
270         public int compareTo(WeakInteger other) {
271             return Integer.compare(value, other.value);
272         }
273     }

275     private class Comparator<K, V> {
276         private final java.util.Map<K, V> classic = new java.util.HashMap<>();
277         private Map<K, V> hamt = HashMap.empty();
278 
279         void test() {
280             assertThat(hamt.size()).isEqualTo(classic.size());
281             hamt.iterator().forEachRemaining(e -> assertThat(classic.get(e._1)).isEqualTo(e._2));
282             classic.forEach((k, v) -> {
283                 assertThat(hamt.get(k).get()).isEqualTo(v);
284                 assertThat(hamt.getOrElse(k, null)).isEqualTo(v);
285             });
286         }
287 
288         void set(K key, V value) {
289             classic.put(key, value);
290             hamt = hamt.put(key, value);
291         }
292 
293         void remove(K key) {
294             classic.remove(key);
295             hamt = hamt.remove(key);
296         }
297     }

299     private <K, V> java.util.Map<K, V> rnd(int count, Function<Tuple2<Integer, Integer>, Tuple2<K, V>> mapper) {
300         final Random r = new Random();
301         final java.util.HashMap<K, V> mp = new java.util.HashMap<>();
302         for (int i = 0; i < count; i++) {
303             final Tuple2<K, V> entry = mapper.apply(Tuple.of(r.nextInt(), r.nextInt()));
304             mp.put(entry._1, entry._2);
305         }
306         return mp;
307     }

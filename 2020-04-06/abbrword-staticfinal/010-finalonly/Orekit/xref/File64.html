35  /** The modified Saastamoinen model. Estimates the path delay imposed to
36   * electro-magnetic signals by the troposphere according to the formula:
37   * <pre>
38   * δ = 2.277e-3 / cos z * (P + (1255 / T + 0.05) * e - B * tan²
39   * z) + δR
40   * </pre>
41   * with the following input data provided to the model:
42   * <ul>
43   * <li>z: zenith angle</li>
44   * <li>P: atmospheric pressure</li>
45   * <li>T: temperature</li>
46   * <li>e: partial pressure of water vapour</li>
47   * <li>B, δR: correction terms</li>
48   * </ul>
49   * <p>
50   * The model supports custom δR correction terms to be read from a
51   * configuration file (saastamoinen-correction.txt) via the
52   * {@link DataProvidersManager}.
53   * </p>
54   * @author Thomas Neidhart
55   * @see "Guochang Xu, GPS - Theory, Algorithms and Applications, Springer, 2007"
56   */
57  public class SaastamoinenModel implements TroposphericModel {

98      /** Create a new Saastamoinen model for the troposphere using the given
99       * environmental conditions.
100      * @param t0 the temperature at the station [K]
101      * @param p0 the atmospheric pressure at the station [mbar]
102      * @param r0 the humidity at the station [fraction] (50% -&gt; 0.5)
103      * @param deltaRFileName regular expression for filename containing δR
104      * correction term table (typically {@link #DELTA_R_FILE_NAME}), if null
105      * default values from the reference book are used
106      * @exception OrekitException if δR correction term table cannot be loaded
107      * @since 7.1
108      */
109     public SaastamoinenModel(final double t0, final double p0, final double r0,
110                              final String deltaRFileName)
111         throws OrekitException {
112         this(t0, p0, r0,
113              deltaRFileName == null ? defaultDeltaR() : loadDeltaR(deltaRFileName));
114     }

116     /** Create a new Saastamoinen model.
117      * @param t0 the temperature at the station [K]
118      * @param p0 the atmospheric pressure at the station [mbar]
119      * @param r0 the humidity at the station [fraction] (50% -> 0.5)
120      * @param deltaR δR correction term function
121      * @since 7.1
122      */
123     private SaastamoinenModel(final double t0, final double p0, final double r0,
124                               final BilinearInterpolatingFunction deltaR) {
125         this.t0             = t0;
126         this.p0             = p0;
127         this.r0             = r0;
128         this.bFunction      = new LinearInterpolator().interpolate(X_VALUES_FOR_B, Y_VALUES_FOR_B);
129         this.eFunction      = new PolynomialFunction(E_COEFFICIENTS);
130         this.deltaRFunction = deltaR;
131     }

133     /** Create a new Saastamoinen model using a standard atmosphere model.
134      *
135      * <ul>
136      * <li>temperature: 18 degree Celsius
137      * <li>pressure: 1013.25 mbar
138      * <li>humidity: 50%
139      * </ul>
140      *
141      * @return a Saastamoinen model with standard environmental values
142      * @exception OrekitException if δR correction term table cannot be loaded
143      */
144     public static SaastamoinenModel getStandardModel()
145         throws OrekitException {
146         return new SaastamoinenModel(273.16 + 18, 1013.25, 0.5, (String) null);
147     }

149     /** {@inheritDoc} */
150     public double pathDelay(final double elevation, final double height) {
151         // the corrected temperature using a temperature gradient of -6.5 K/km
152         final double T = t0 - 6.5e-3 * height;
153         // the corrected pressure
154         final double P = p0 * FastMath.pow(1.0 - 2.26e-5 * height, 5.225);
155         // the corrected humidity
156         final double R = r0 * FastMath.exp(-6.396e-4 * height);
157 
158         // interpolate the b correction term
159         final double B = bFunction.value(height / 1e3);
160         // calculate e
161         final double e = R * FastMath.exp(eFunction.value(T));
162 
163         // calculate the zenith angle from the elevation
164         final double z = FastMath.abs(0.5 * FastMath.PI - elevation);
165 
166         // get correction factor
167         final double deltaR = getDeltaR(height, z);
168 
169         // calculate the path delay in m
170         final double tan = FastMath.tan(z);
171         final double delta = 2.277e-3 / FastMath.cos(z) *
172                              (P + (1255d / T + 5e-2) * e - B * tan * tan) + deltaR;
173 
174         return delta;
175     }

177     /** Calculates the delta R correction term using linear interpolation.
178      * @param height the height of the station in m
179      * @param zenith the zenith angle of the satellite
180      * @return the delta R correction term in m
181      */
182     private double getDeltaR(final double height, final double zenith) {
183         // limit the height to a range of [0, 5000] m
184         final double h = FastMath.min(FastMath.max(0, height), 5000);
185         // limit the zenith angle to 90 degree
186         // Note: the function is symmetric for negative zenith angles
187         final double z = FastMath.min(Math.abs(zenith), 0.5 * FastMath.PI);
188         return deltaRFunction.value(h, z);
189     }

191     /** Load δR function.
192      * @param deltaRFileName regular expression for filename containing δR
193      * correction term table
194      * @return δR function
195      * @exception OrekitException if table cannot be loaded
196      */
197     private static BilinearInterpolatingFunction loadDeltaR(final String deltaRFileName)
198         throws OrekitException {
199 
200         // read the δR interpolation function from the config file
201         final InterpolationTableLoader loader = new InterpolationTableLoader();
202         DataProvidersManager.getInstance().feed(deltaRFileName, loader);
203         if (!loader.stillAcceptsData()) {
204             final double[] elevations = loader.getOrdinateGrid();
205             for (int i = 0; i < elevations.length; ++i) {
206                 elevations[i] = FastMath.toRadians(elevations[i]);
207             }
208             return new BilinearInterpolatingFunction(loader.getAbscissaGrid(), elevations,
209                                                      loader.getValuesSamples());
210         }
211         throw new OrekitException(OrekitMessages.UNABLE_TO_FIND_FILE,
212                                   deltaRFileName.replaceAll("^\\^", "").replaceAll("\\$$", ""));
213     }

315     /**
316      * Function that implements a standard bilinear interpolation.
317      * The interpolation as found
318      * in the Wikipedia reference <a href =
319      * "http://en.wikipedia.org/wiki/Bilinear_interpolation">BiLinear
320      * Interpolation</a>. This is a stand-in until Apache Math has a
321      * bilinear interpolator
322      */
323     private static class BilinearInterpolatingFunction implements BivariateFunction {
324 
325         /**
326          * The minimum number of points that are needed to compute the
327          * function.
328          */
329         private static final int MIN_NUM_POINTS = 2;
330 
331         /** Samples x-coordinates. */
332         private final double[] xval;
333 
334         /** Samples y-coordinates. */
335         private final double[] yval;
336 
337         /** Set of cubic splines patching the whole data grid. */
338         private final double[][] fval;
339 
340         /**
341          * @param x Sample values of the x-coordinate, in increasing order.
342          * @param y Sample values of the y-coordinate, in increasing order.
343          * @param f Values of the function on every grid point. the expected
344          *        number of elements.
345          * @throws MathIllegalArgumentException if the length of x and y don't
346          *         match the row, column height of f, or if any of the arguments
347          *         are null, or if any of the arrays has zero length, or if
348          *         {@code x} or {@code y} are not strictly increasing.
349          */
350         BilinearInterpolatingFunction(final double[] x, final double[] y, final double[][] f)
351                         throws MathIllegalArgumentException {
352 
353             if (x == null || y == null || f == null || f[0] == null) {
354                 throw new IllegalArgumentException("All arguments must be non-null");
355             }
356 
357             final int xLen = x.length;
358             final int yLen = y.length;
359 
360             if (xLen == 0 || yLen == 0 || f.length == 0 || f[0].length == 0) {
361                 throw new MathIllegalArgumentException(LocalizedCoreFormats.NO_DATA);
362             }
363 
364             if (xLen < MIN_NUM_POINTS || yLen < MIN_NUM_POINTS || f.length < MIN_NUM_POINTS ||
365                             f[0].length < MIN_NUM_POINTS) {
366                 throw new MathIllegalArgumentException(LocalizedCoreFormats.INSUFFICIENT_DATA);
367             }
368 
369             if (xLen != f.length) {
370                 throw new MathIllegalArgumentException(LocalizedCoreFormats.DIMENSIONS_MISMATCH,
371                                                        xLen, f.length);
372             }
373 
374             if (yLen != f[0].length) {
375                 throw new MathIllegalArgumentException(LocalizedCoreFormats.DIMENSIONS_MISMATCH,
376                                                        yLen, f[0].length);
377             }
378 
379             MathArrays.checkOrder(x);
380             MathArrays.checkOrder(y);
381 
382             xval = x.clone();
383             yval = y.clone();
384             fval = f.clone();
385         }
386 
387         @Override
388         public double value(final double x, final double y) {
389             final int offset = 1;
390             final int count = offset + 1;
391             final int i = searchIndex(x, xval, offset, count);
392             final int j = searchIndex(y, yval, offset, count);
393 
394             final double x1 = xval[i];
395             final double x2 = xval[i + 1];
396             final double y1 = yval[j];
397             final double y2 = yval[j + 1];
398             final double fQ11 = fval[i][j];
399             final double fQ21 = fval[i + 1][j];
400             final double fQ12 = fval[i][j + 1];
401             final double fQ22 = fval[i + 1][j + 1];
402 
403             final double f = (fQ11 * (x2 - x)  * (y2 - y) +
404                             fQ21 * (x  - x1) * (y2 - y) +
405                             fQ12 * (x2 - x)  * (y  - y1) +
406                             fQ22 * (x  - x1) * (y  - y1)) /
407                             ((x2 - x1) * (y2 - y1));
408 
409             return f;
410         }
411 
412         /**
413          * @param c Coordinate.
414          * @param val Coordinate samples.
415          * @param offset how far back from found value to offset for
416          *        querying
417          * @param count total number of elements forward from beginning that
418          *        will be queried
419          * @return the index in {@code val} corresponding to the interval
420          *         containing {@code c}.
421          * @throws MathIllegalArgumentException if {@code c} is out of the range
422          *         defined by the boundary values of {@code val}.
423          */
424         private int searchIndex(final double c, final double[] val, final int offset, final int count)
425             throws MathIllegalArgumentException {
426             int r = Arrays.binarySearch(val, c);
427 
428             if (r == -1 || r == -val.length - 1) {
429                 throw new MathIllegalArgumentException(LocalizedCoreFormats.OUT_OF_RANGE_SIMPLE,
430                                                        c, val[0], val[val.length - 1]);
431             }
432 
433             if (r < 0) {
434                 // "c" in within an interpolation sub-interval, which
435                 // returns
436                 // negative
437                 // need to remove the negative sign for consistency
438                 r = -r - offset - 1;
439             } else {
440                 r -= offset;
441             }
442 
443             if (r < 0) {
444                 r = 0;
445             }
446 
447             if ((r + count) >= val.length) {
448                 // "c" is the last sample of the range: Return the index
449                 // of the sample at the lower end of the last sub-interval.
450                 r = val.length - count;
451             }
452 
453             return r;
454         }
455 
456     }

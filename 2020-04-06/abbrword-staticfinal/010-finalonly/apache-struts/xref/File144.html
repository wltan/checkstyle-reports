87      public String buildUrl(String action, HttpServletRequest request, HttpServletResponse response, Map<String, Object> params, String urlScheme,
88                             boolean includeContext, boolean encodeResult, boolean forceAddSchemeHostAndPort, boolean escapeAmp) {
89  
90          StringBuilder link = new StringBuilder();
91          boolean changedScheme = false;
92  
93          String scheme = null;
94          if (isValidScheme(urlScheme)) {
95              scheme = urlScheme;
96          }
97  
98          // only append scheme if it is different to the current scheme *OR*
99          // if we explicity want it to be appended by having forceAddSchemeHostAndPort = true
100         if (forceAddSchemeHostAndPort) {
101             String reqScheme = request.getScheme();
102             changedScheme = true;
103             link.append(scheme != null ? scheme : reqScheme);
104             link.append("://");
105             link.append(request.getServerName());
106 
107             if (scheme != null) {
108                 // If switching schemes, use the configured port for the particular scheme.
109                 if (!scheme.equals(reqScheme)) {
110                     appendPort(link, scheme, HTTP_PROTOCOL.equals(scheme) ? httpPort : httpsPort);
111                 // Else use the port from the current request.
112                 } else {
113                     appendPort(link, scheme, request.getServerPort());
114                 }
115             } else {
116                 appendPort(link, reqScheme, request.getServerPort());
117             }
118         } else if ((scheme != null) && !scheme.equals(request.getScheme())) {
119             changedScheme = true;
120             link.append(scheme);
121             link.append("://");
122             link.append(request.getServerName());
123 
124             appendPort(link, scheme, HTTP_PROTOCOL.equals(scheme) ? httpPort : httpsPort);
125         }
126 
127         if (action != null) {
128             // Check if context path needs to be added
129             // Add path to absolute links
130             if (action.startsWith("/") && includeContext) {
131                 String contextPath = request.getContextPath();
132                 if (!contextPath.equals("/")) {
133                     link.append(contextPath);
134                 }
135             } else if (changedScheme) {
136 
137                 // (Applicable to Servlet 2.4 containers)
138                 // If the request was forwarded, the attribute below will be set with the original URL
139                 String uri = (String) request.getAttribute("javax.servlet.forward.request_uri");
140 
141                 // If the attribute wasn't found, default to the value in the request object
142                 if (uri == null) {
143                     uri = request.getRequestURI();
144                 }
145 
146                 link.append(uri.substring(0, uri.lastIndexOf('/') + 1));
147             }
148 
149             // Add page
150             link.append(action);
151         } else {
152             // Go to "same page"
153             String requestURI = (String) request.getAttribute("struts.request_uri");
154 
155             // (Applicable to Servlet 2.4 containers)
156             // If the request was forwarded, the attribute below will be set with the original URL
157             if (requestURI == null) {
158                 requestURI = (String) request.getAttribute("javax.servlet.forward.request_uri");
159             }
160 
161             // If neither request attributes were found, default to the value in the request object
162             if (requestURI == null) {
163                 requestURI = request.getRequestURI();
164             }
165 
166             link.append(requestURI);
167         }
168 
169         //if the action was not explicitly set grab the params from the request
170         if (escapeAmp) {
171             buildParametersString(params, link, AMP, true);
172         } else {
173             buildParametersString(params, link, "&", true);
174         }
175 
176         String result = link.toString();
177 
178         if (StringUtils.containsIgnoreCase(result, "<script")){
179             result = StringEscapeUtils.escapeEcmaScript(result);
180         }
181         try {
182             result = encodeResult ? response.encodeURL(result) : result;
183         } catch (Exception ex) {
184             LOG.debug("Could not encode the URL for some reason, use it unchanged", ex);
185             result = link.toString();
186         }
187 
188         return result;
189     }

191     private void appendPort(StringBuilder link, String scheme, int port) {
192         if ((HTTP_PROTOCOL.equals(scheme) && port != DEFAULT_HTTP_PORT) || (HTTPS_PROTOCOL.equals(scheme) && port != DEFAULT_HTTPS_PORT)) {
193             link.append(":");
194             link.append(port);
195         }
196     }

202     public void buildParametersString(Map<String, Object> params, StringBuilder link, String paramSeparator, boolean encode) {
203         if ((params != null) && (params.size() > 0)) {
204             if (!link.toString().contains("?")) {
205                 link.append("?");
206             } else {
207                 link.append(paramSeparator);
208             }
209 
210             // Set params
211             Iterator<Map.Entry<String, Object>> iter = params.entrySet().iterator();
212             while (iter.hasNext()) {
213                 Map.Entry<String, Object> entry = iter.next();
214                 String name = entry.getKey();
215                 Object value = entry.getValue();
216 
217                 if (value instanceof Iterable) {
218                     for (Iterator iterator = ((Iterable) value).iterator(); iterator.hasNext();) {
219                         Object paramValue = iterator.next();
220                         link.append(buildParameterSubstring(name, paramValue != null ? paramValue.toString() : StringUtils.EMPTY, encode));
221 
222                         if (iterator.hasNext()) {
223                             link.append(paramSeparator);
224                         }
225                     }
226                 } else if (value instanceof Object[]) {
227                     Object[] array = (Object[]) value;
228                     for (int i = 0; i < array.length; i++) {
229                         Object paramValue = array[i];
230                         link.append(buildParameterSubstring(name, paramValue != null ? paramValue.toString() : StringUtils.EMPTY, encode));
231 
232                         if (i < array.length - 1) {
233                             link.append(paramSeparator);
234                         }
235                     }
236                 } else {
237                     link.append(buildParameterSubstring(name, value != null ? value.toString() : StringUtils.EMPTY, encode));
238                 }
239 
240                 if (iter.hasNext()) {
241                     link.append(paramSeparator);
242                 }
243             }
244         }
245     }
246 
247     protected boolean isValidScheme(String scheme) {
248         return HTTP_PROTOCOL.equals(scheme) || HTTPS_PROTOCOL.equals(scheme);
249     }
250 
251     private String buildParameterSubstring(String name, String value, boolean encode) {
252         StringBuilder builder = new StringBuilder();
253         builder.append(encode ? encode(name) : name);
254         builder.append('=');
255         builder.append(encode ? encode(value) : value);
256         return builder.toString();
257     }

259 	/**
260 	 * Encodes the URL using {@link java.net.URLEncoder#encode} with the encoding specified in the configuration.
261 	 *
262 	 * @param input the input to encode
263 	 * @return the encoded string
264 	 */
265 	public String encode( String input ) {
266 		try {
267 			return URLEncoder.encode(input, encoding);
268 		} catch (UnsupportedEncodingException e) {
269 			LOG.warn("Could not encode URL parameter '{}', returning value un-encoded", input);
270 			return input;
271 		}
272 	}

275 	 * Decodes the URL using {@link URLDecoderUtil#decode(String, String)} with the encoding specified in the configuration.
276 	 *
277 	 * @param input the input to decode
278 	 * @return the encoded string

281         return URLDecoderUtil.decode(input, encoding, false);
282 	}

284     /**
285      * Decodes the URL using {@link URLDecoderUtil#decode(String, String, boolean)} with the encoding specified in the configuration.
286      *
287      * @param input the input to decode
288      * @param isQueryString whether input is a query string. If <code>true</code> other decoding rules apply.
289      * @return the encoded string
290      */
291     public String decode( String input, boolean isQueryString ) {
292         try {
293             return URLDecoderUtil.decode(input, encoding, isQueryString);
294         } catch (Exception e) {
295             LOG.warn("Could not decode URL parameter '{}', returning value un-decoded", input);
296         return input;
297         }
298     }

300     public Map<String, Object> parseQueryString(String queryString, boolean forceValueArray) {
301         Map<String, Object> queryParams = new LinkedHashMap<String, Object>();
302         if (queryString != null) {
303             String[] params = queryString.split("&");
304             for (String param : params) {
305                 if (param.trim().length() > 0) {
306                     String[] tmpParams = param.split("=");
307                     String paramName = null;
308                     String paramValue = "";
309                     if (tmpParams.length > 0) {
310                         paramName = tmpParams[0];
311                     }
312                     if (tmpParams.length > 1) {
313                         paramValue = tmpParams[1];
314                     }
315                     if (paramName != null) {
316                         paramName = decode(paramName, true);
317                         String translatedParamValue = decode(paramValue, true);
318 
319                         if (queryParams.containsKey(paramName) || forceValueArray) {
320                             // WW-1619 append new param value to existing value(s)
321                             Object currentParam = queryParams.get(paramName);
322                             if (currentParam instanceof String) {
323                                 queryParams.put(paramName, new String[]{(String) currentParam, translatedParamValue});
324                             } else {
325                                 String currentParamValues[] = (String[]) currentParam;
326                                 if (currentParamValues != null) {
327                                     List<String> paramList = new ArrayList<String>(Arrays.asList(currentParamValues));
328                                     paramList.add(translatedParamValue);
329                                     queryParams.put(paramName, paramList.toArray(new String[paramList.size()]));
330                                 } else {
331                                     queryParams.put(paramName, new String[]{translatedParamValue});
332                                 }
333                             }
334                         } else {
335                             queryParams.put(paramName, translatedParamValue);
336                         }
337                     }
338                 }
339             }
340         }
341         return queryParams;
342     }

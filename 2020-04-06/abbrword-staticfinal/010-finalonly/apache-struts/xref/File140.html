100     /**
101      * Execute this result, using the specified template locationArg.
102      * <p>
103      * The template locationArg has already been interpolated for any variable substitutions
104      * </p>
105      * <p>
106      * this method obtains the freemarker configuration and the object wrapper from the provided hooks.
107      * It them implements the template processing workflow by calling the hooks for
108      * preTemplateProcess and postTemplateProcess
109      * </p>
110      *
111      * @param locationArg location argument
112      * @param invocation the action invocation
113      *
114      * @throws IOException in case of IO errors
115      * @throws TemplateException in case of freemarker template errors
116      */
117     public void doExecute(String locationArg, ActionInvocation invocation) throws IOException, TemplateException {
118         this.location = locationArg;
119         this.invocation = invocation;
120         this.configuration = getConfiguration();
121         this.wrapper = getObjectWrapper();
122 
123         ActionContext ctx = invocation.getInvocationContext();
124         HttpServletRequest req = (HttpServletRequest) ctx.get(ServletActionContext.HTTP_REQUEST);
125 
126         String absoluteLocation;
127         if (location.startsWith("/")) {
128             absoluteLocation = location; 
129         } else { 
130             String namespace = invocation.getProxy().getNamespace();
131             if (namespace == null || namespace.length() == 0 || namespace.equals("/")) {
132                 absoluteLocation = "/" + location;
133             } else if (namespace.startsWith("/")) {
134                 absoluteLocation = namespace + "/" + location;
135             } else {
136                 absoluteLocation = "/" + namespace + "/" + location;
137             }
138         }
139 
140         Template template = configuration.getTemplate(absoluteLocation, deduceLocale());
141         TemplateModel model = createModel();
142 
143         // Give subclasses a chance to hook into preprocessing
144         if (preTemplateProcess(template, model)) {
145             try {
146                 final boolean willWriteIfCompleted;
147                 if (writeIfCompleted != null) {
148                     willWriteIfCompleted = isWriteIfCompleted();
149                 } else {
150                     willWriteIfCompleted = template.getTemplateExceptionHandler() == TemplateExceptionHandler.RETHROW_HANDLER;
151                 }
152 
153                 // Process the template
154                 Writer writer = getWriter();
155                 if (willWriteIfCompleted){
156                     CharArrayWriter parentCharArrayWriter = (CharArrayWriter) req.getAttribute(PARENT_TEMPLATE_WRITER);
157                     boolean isTopTemplate;
158                     if (isTopTemplate = (parentCharArrayWriter == null)) {
159                         //this is the top template
160                         parentCharArrayWriter = new CharArrayWriter();
161                         //set it in the request because when the "action" tag is used a new VS and ActionContext is created
162                         req.setAttribute(PARENT_TEMPLATE_WRITER, parentCharArrayWriter);
163                     }
164 
165                     try {
166                         template.process(model, parentCharArrayWriter);
167 
168                         if (isTopTemplate) {
169                             parentCharArrayWriter.flush();
170                             parentCharArrayWriter.writeTo(writer);
171                         }
172                     } catch (TemplateException | IOException e) {
173                         if (LOG.isErrorEnabled()) {
174                             LOG.error("Error processing Freemarker result!", e);
175                         }
176                         throw e;
177                     } finally {
178                         if (isTopTemplate) {
179                             req.removeAttribute(PARENT_TEMPLATE_WRITER);
180                             parentCharArrayWriter.close();
181                         }
182                     }
183                 } else {
184                     template.process(model, writer);
185                 }
186             } finally {
187                 // Give subclasses a chance to hook into postprocessing
188                 postTemplateProcess(template, model);
189             }
190         }
191     }

193     /**
194      * <p>
195      * This method is called from {@link #doExecute(String, ActionInvocation)} to obtain the
196      * FreeMarker configuration object that this result will use for template loading. This is a
197      * hook that allows you to custom-configure the configuration object in a subclass, or to fetch
198      * it from an IoC container.
199      * </p>
200      *
201      * <p>
202      * <b>

208      * @throws TemplateException in case of freemarker configuration errors

251     /**
252      * Build the instance of the ScopesHashModel, including JspTagLib support
253      * <p>
254      * Objects added to the model are
255      * </p>
256      *
257      * <ul>
258      * <li>Application - servlet context attributes hash model
259      * <li>JspTaglibs - jsp tag lib factory model
260      * <li>Request - request attributes hash model
261      * <li>Session - session attributes hash model
262      * <li>request - the HttpServletRequst object for direct access
263      * <li>response - the HttpServletResponse object for direct access
264      * <li>stack - the OgnLValueStack instance for direct access
265      * <li>ognl - the instance of the OgnlTool
266      * <li>action - the action itself
267      * <li>exception - optional : the JSP or Servlet exception as per the servlet spec (for JSP Exception pages)
268      * <li>struts - instance of the StrutsUtil class
269      * </ul>
270      *
271      * @return TemplateModel returns the created template model
272      * @throws TemplateModelException in case of errors during creating the model
273      */
274     protected TemplateModel createModel() throws TemplateModelException {
275         ServletContext servletContext = ServletActionContext.getServletContext();
276         HttpServletRequest request = ServletActionContext.getRequest();
277         HttpServletResponse response = ServletActionContext.getResponse();
278         ValueStack stack = ServletActionContext.getContext().getValueStack();
279 
280         Object action = null;
281         if(invocation!= null ) action = invocation.getAction(); //Added for NullPointException
282         return freemarkerManager.buildTemplateModel(stack, action, servletContext, request, response, wrapper);
283     }

285     /**
286      * Returns the locale used for the {@link Configuration#getTemplate(String, Locale)} call. The base implementation
287      * simply returns the locale setting of the action (assuming the action implements {@link LocaleProvider}) or, if
288      * the action does not the configuration's locale is returned. Override this method to provide different behaviour,
289      *
290      * @return the locale from action if action implements the {@link LocaleProvider}) or local from configuration
291      */
292     protected Locale deduceLocale() {
293         if (invocation.getAction() instanceof LocaleProvider) {
294             return ((LocaleProvider) invocation.getAction()).getLocale();
295         } else {
296             return configuration.getLocale();
297         }
298     }

311     /**
312      * Called before the execution is passed to template.process().
313      * This is a generic hook you might use in subclasses to perform a specific
314      * action before the template is processed. By default does nothing.
315      * A typical action to perform here is to inject application-specific
316      * objects into the model root
317      *
318      * @param template the freemarker template
319      * @param model the template model
320      * @return true to process the template, false to suppress template processing.
321      * @throws IOException in case of IO errors
322      */
323     protected boolean preTemplateProcess(Template template, TemplateModel model) throws IOException {
324         Object attrContentType = template.getCustomAttribute("content_type");
325 
326         HttpServletResponse response = ServletActionContext.getResponse();
327         if (response.getContentType() == null) {
328             if (attrContentType != null) {
329                 response.setContentType(attrContentType.toString());
330             } else {
331                 String contentType = getContentType();
332 
333                 if (contentType == null) {
334                     contentType = "text/html";
335                 }
336 
337                 String encoding = template.getEncoding();
338 
339                 if (encoding != null) {
340                     contentType = contentType + "; charset=" + encoding;
341                 }
342 
343                 response.setContentType(contentType);
344             }
345         } else if(isInsideActionTag()){
346              //trigger com.opensymphony.module.sitemesh.filter.PageResponseWrapper.deactivateSiteMesh()
347             response.setContentType(response.getContentType());
348         }
349 
350         return true;
351     }

353     private boolean isInsideActionTag() {
354         Object attribute = ServletActionContext.getRequest().getAttribute(StrutsStatics.STRUTS_ACTION_TAG_INVOCATION);
355         return (Boolean) ObjectUtils.defaultIfNull(attribute, Boolean.FALSE);
356     }

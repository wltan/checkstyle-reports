30  /**
31  * <!-- START SNIPPET: description -->
32  *
33  * This result invokes an entire other action, complete with it's own interceptor stack and result.
34  *
35  * <!-- END SNIPPET: description -->
36  *
37  * <b>This result type takes the following parameters:</b>
38  *
39  * <!-- START SNIPPET: params -->
40  *
41  * <ul>
42  *
43  * <li><b>actionName (default)</b> - the name of the action that will be chained to</li>
44  *
45  * <li><b>namespace</b> - used to determine which namespace the Action is in that we're chaining. If namespace is null,
46  * this defaults to the current namespace</li>
47  *
48  * <li><b>method</b> - used to specify another method on target action to be invoked.
49  * If null, this defaults to execute method</li>
50  *
51  * <li><b>skipActions</b> - (optional) the list of comma separated action names for the
52  * actions that could be chained to</li>
53  *
54  * </ul>
55  *
56  * <!-- END SNIPPET: params -->
57  *
58  * <b>Example:</b>
59  *
60  * <pre><!-- START SNIPPET: example -->
61  * &lt;package name="public" extends="struts-default"&gt;
62  *     &lt;!-- Chain creatAccount to login, using the default parameter --&gt;
63  *     &lt;action name="createAccount" class="..."&gt;
64  *         &lt;result type="chain"&gt;login&lt;/result&gt;
65  *     &lt;/action&gt;
66  *
67  *     &lt;action name="login" class="..."&gt;
68  *         &lt;!-- Chain to another namespace --&gt;
69  *         &lt;result type="chain"&gt;
70  *             &lt;param name="actionName"&gt;dashboard&lt;/param&gt;
71  *             &lt;param name="namespace"&gt;/secure&lt;/param&gt;
72  *         &lt;/result&gt;
73  *     &lt;/action&gt;
74  * &lt;/package&gt;
75  *
76  * &lt;package name="secure" extends="struts-default" namespace="/secure"&gt;
77  *     &lt;action name="dashboard" class="..."&gt;
78  *         &lt;result&gt;dashboard.jsp&lt;/result&gt;
79  *     &lt;/action&gt;
80  * &lt;/package&gt;
81  * <!-- END SNIPPET: example --></pre>
82  *
83  * @author <a href='mailto:the_mindstorm[at]evolva[dot]ro'>Alexandru Popescu</a>
84  */
85  public class ActionChainResult implements Result {
86  
87      private static final Logger LOG = LogManager.getLogger(ActionChainResult.class);
88

91       */

96       */

100      * The result parameter name to set the name of the action to chain to.

184     /**
185      * Get the XWork chain history.
186      * The stack is a list of <code>namespace/action!method</code> keys.
187      *
188      * @return the chain history as string list
189      */
190     public static LinkedList<String> getChainHistory() {
191         LinkedList<String> chainHistory = (LinkedList<String>) ActionContext.getContext().get(CHAIN_HISTORY);
192         //  Add if not exists
193         if (chainHistory == null) {
194             chainHistory = new LinkedList<>();
195             ActionContext.getContext().put(CHAIN_HISTORY, chainHistory);
196         }
197 
198         return chainHistory;
199     }
200 
201     /**
202      * @param invocation the DefaultActionInvocation calling the action call stack
203      */
204     public void execute(ActionInvocation invocation) throws Exception {
205         ValueStack stack = ActionContext.getContext().getValueStack();
206         String finalNamespace = this.namespace != null
207                 ? TextParseUtil.translateVariables(namespace, stack)
208                 : invocation.getProxy().getNamespace();
209         String finalActionName = TextParseUtil.translateVariables(actionName, stack);
210         String finalMethodName = this.methodName != null
211                 ? TextParseUtil.translateVariables(this.methodName, stack)
212                 : null;
213 
214         if (isInChainHistory(finalNamespace, finalActionName, finalMethodName)) {
215             addToHistory(finalNamespace, finalActionName, finalMethodName);
216             throw new StrutsException("Infinite recursion detected: " + ActionChainResult.getChainHistory().toString());
217         }
218 
219         if (ActionChainResult.getChainHistory().isEmpty() && invocation != null && invocation.getProxy() != null) {
220             addToHistory(finalNamespace, invocation.getProxy().getActionName(), invocation.getProxy().getMethod());
221         }
222         addToHistory(finalNamespace, finalActionName, finalMethodName);
223 
224         HashMap<String, Object> extraContext = new HashMap<>();
225         extraContext.put(ActionContext.VALUE_STACK, ActionContext.getContext().getValueStack());
226         extraContext.put(ActionContext.PARAMETERS, ActionContext.getContext().getParameters());
227         extraContext.put(CHAIN_HISTORY, ActionChainResult.getChainHistory());
228 
229         LOG.debug("Chaining to action {}", finalActionName);
230 
231         proxy = actionProxyFactory.createActionProxy(finalNamespace, finalActionName, finalMethodName, extraContext);
232         proxy.execute();
233     }
234 
235     @Override public boolean equals(Object o) {
236         if (this == o) return true;
237         if (o == null || getClass() != o.getClass()) return false;
238 
239         final ActionChainResult that = (ActionChainResult) o;
240 
241         if (actionName != null ? !actionName.equals(that.actionName) : that.actionName != null) return false;
242         if (methodName != null ? !methodName.equals(that.methodName) : that.methodName != null) return false;
243         if (namespace != null ? !namespace.equals(that.namespace) : that.namespace != null) return false;
244 
245         return true;
246     }
247 
248     @Override public int hashCode() {
249         int result;
250         result = (actionName != null ? actionName.hashCode() : 0);
251         result = 31 * result + (namespace != null ? namespace.hashCode() : 0);
252         result = 31 * result + (methodName != null ? methodName.hashCode() : 0);
253         return result;
254     }
255 
256     private boolean isInChainHistory(String namespace, String actionName, String methodName) {
257         LinkedList<? extends String> chainHistory = ActionChainResult.getChainHistory();
258 
259         if (chainHistory == null) {
260             return false;
261         } else {
262             //  Actions to skip
263             Set<String> skipActionsList = new HashSet<>();
264             if (skipActions != null && skipActions.length() > 0) {
265                 ValueStack stack = ActionContext.getContext().getValueStack();
266                 String finalSkipActions = TextParseUtil.translateVariables(this.skipActions, stack);
267                 skipActionsList.addAll(TextParseUtil.commaDelimitedStringToSet(finalSkipActions));
268             }
269             if (!skipActionsList.contains(actionName)) {
270                 //  Get if key is in the chain history
271                 return chainHistory.contains(makeKey(namespace, actionName, methodName));
272             }
273 
274             return false;
275         }
276     }
277 
278     private void addToHistory(String namespace, String actionName, String methodName) {
279         List<String> chainHistory = ActionChainResult.getChainHistory();
280         chainHistory.add(makeKey(namespace, actionName, methodName));
281     }
282 
283     private String makeKey(String namespace, String actionName, String methodName) {
284         if (null == methodName) {
285             return namespace + "/" + actionName;
286         }
287 
288         return namespace + "/" + actionName + "!" + methodName;
289     }

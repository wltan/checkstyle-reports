24  package org.hibernate.type.descriptor.java;
25  
26  import java.util.Comparator;
27  import java.util.TimeZone;
28  
29  import org.hibernate.type.descriptor.WrapperOptions;
30  
31  /**
32   * Descriptor for {@link TimeZone} handling.
33   *
34   * @author Steve Ebersole
35   */
36  public class TimeZoneTypeDescriptor extends AbstractTypeDescriptor<TimeZone> {
37  	public static final TimeZoneTypeDescriptor INSTANCE = new TimeZoneTypeDescriptor();
38  
39  	public static class TimeZoneComparator implements Comparator<TimeZone> {
40  		public static final TimeZoneComparator INSTANCE = new TimeZoneComparator();
41  
42  		public int compare(TimeZone o1, TimeZone o2) {
43  			return o1.getID().compareTo( o2.getID() );
44  		}
45  	}
46  
47  	public TimeZoneTypeDescriptor() {
48  		super( TimeZone.class );
49  	}
50  
51  	public String toString(TimeZone value) {
52  		return value.getID();
53  	}
54  
55  	public TimeZone fromString(String string) {
56  		return TimeZone.getTimeZone( string );
57  	}
58  
59  	@Override
60  	public Comparator<TimeZone> getComparator() {
61  		return TimeZoneComparator.INSTANCE;
62  	}
63  
64  	@SuppressWarnings({ "unchecked" })
65  	public <X> X unwrap(TimeZone value, Class<X> type, WrapperOptions options) {
66  		if ( value == null ) {
67  			return null;
68  		}
69  		if ( String.class.isAssignableFrom( type ) ) {
70  			return (X) toString( value );
71  		}
72  		throw unknownUnwrap( type );
73  	}
74  
75  	public <X> TimeZone wrap(X value, WrapperOptions options) {
76  		if ( value == null ) {
77  			return null;
78  		}
79  		if ( String.class.isInstance( value ) ) {
80  			return fromString( (String) value );
81  		}
82  		throw unknownWrap( value.getClass() );
83  	}
84  }

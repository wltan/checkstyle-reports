35  /**
36   * Models SQL aggregation functions (<tt>MIN</tt>, <tt>MAX</tt>, <tt>COUNT</tt>, etc).
37   *
38   * @author Steve Ebersole
39   */
40  public class AggregationFunction<T>
41  		extends ParameterizedFunctionExpression<T>
42  		implements Serializable {
43  
44  	/**
45  	 * Constructs an aggregation function with a single literal argument.
46  	 *
47  	 * @param criteriaBuilder The query builder instance.
48  	 * @param returnType The function return type.
49  	 * @param functionName The name of the function.
50  	 * @param argument The literal argument
51  	 */
52  	@SuppressWarnings({ "unchecked" })
53  	public AggregationFunction(
54  			CriteriaBuilderImpl criteriaBuilder,
55  			Class<T> returnType,
56  			String functionName,
57  			Object argument) {
58  		this( criteriaBuilder, returnType, functionName, new LiteralExpression( criteriaBuilder, argument ) );
59  	}
60  
61  	/**
62  	 * Constructs an aggregation function with a single literal argument.
63  	 *
64  	 * @param criteriaBuilder The query builder instance.
65  	 * @param returnType The function return type.
66  	 * @param functionName The name of the function.
67  	 * @param argument The argument
68  	 */
69  	public AggregationFunction(
70  			CriteriaBuilderImpl criteriaBuilder,
71  			Class<T> returnType,
72  			String functionName,
73  			Expression<?> argument) {
74  		super( criteriaBuilder, returnType, functionName, argument );
75  	}
76  
77  	@Override
78  	public boolean isAggregation() {
79  		return true;
80  	}
81  
82  	/**
83  	 * Implementation of a <tt>COUNT</tt> function providing convenience in construction.
84  	 * <p/>
85  	 * Parameterized as {@link Long} because thats what JPA states
86  	 * that the return from <tt>COUNT</tt> should be.
87  	 */
88  	public static class COUNT extends AggregationFunction<Long> {
89  		public static final String NAME = "count";
90  
91  		private final boolean distinct;
92  
93  		public COUNT(CriteriaBuilderImpl criteriaBuilder, Expression<?> expression, boolean distinct) {
94  			super( criteriaBuilder, Long.class, NAME , expression );
95  			this.distinct = distinct;
96  		}
97  
98  		@Override
99  		protected void renderArguments( StringBuilder buffer,
100 		                                CriteriaQueryCompiler.RenderingContext renderingContext ) {
101 			if (isDistinct()) buffer.append("distinct ");
102 			else {
103 	            // If function specifies a single non-distinct entity with ID, its alias would normally be rendered, which ends up
104 	            // converting to the column(s) associated with the entity's ID in the rendered SQL.  However, some DBs don't support
105 	            // the multiple columns that would end up here for entities with composite IDs.  So, since we modify the query to
106 	            // instead specify star since that's functionally equivalent and supported by all DBs.
107 			    List<Expression<?>> argExprs = getArgumentExpressions();
108 			    if (argExprs.size() == 1) {
109         	        Expression argExpr = argExprs.get(0);
110         	        if (argExpr instanceof Root<?>) {
111         	            Root<?> root = (Root<?>)argExpr;
112         	            if (!root.getModel().hasSingleIdAttribute()) {
113         	                buffer.append('*');
114         	                return;
115         	            }
116         	        }
117 			    }
118 			}
119             super.renderArguments(buffer, renderingContext);
120 		}
121 
122 		public boolean isDistinct() {
123 			return distinct;
124 		}
125 
126 	}

141 	/**
142 	 * Implementation of a <tt>SUM</tt> function providing convenience in construction.
143 	 * <p/>
144 	 * Parameterized as {@link Number N extends Number} because thats what JPA states
145 	 * that the return from <tt>SUM</tt> should be.
146 	 */
147 	public static class SUM<N extends Number> extends AggregationFunction<N> {
148 		public static final String NAME = "sum";
149 
150 		@SuppressWarnings({ "unchecked" })
151 		public SUM(CriteriaBuilderImpl criteriaBuilder, Expression<N> expression) {
152 			super( criteriaBuilder, (Class<N>)expression.getJavaType(), NAME , expression);
153 			// force the use of a ValueHandler
154 			resetJavaType( expression.getJavaType() );
155 		}
156 
157 		public SUM(CriteriaBuilderImpl criteriaBuilder, Expression<? extends Number> expression, Class<N> returnType) {
158 			super( criteriaBuilder, returnType, NAME , expression);
159 			// force the use of a ValueHandler
160 			resetJavaType( returnType );
161 		}
162 	}

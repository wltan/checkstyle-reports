21  abstract class SpmcArrayQueueL1Pad<E> extends ConcurrentCircularArrayQueue<E> {
22      long p10, p11, p12, p13, p14, p15, p16;
23      long p30, p31, p32, p33, p34, p35, p36, p37;
24  
25      public SpmcArrayQueueL1Pad(int capacity) {
26          super(capacity);
27      }
28  }

30  abstract class SpmcArrayQueueProducerField<E> extends SpmcArrayQueueL1Pad<E> {
31      protected final static long P_INDEX_OFFSET;
32      static {
33          try {
34              P_INDEX_OFFSET =
35                      UNSAFE.objectFieldOffset(SpmcArrayQueueProducerField.class.getDeclaredField("producerIndex"));
36          } catch (NoSuchFieldException e) {
37              throw new RuntimeException(e);
38          }
39      }
40      private volatile long producerIndex;
41  
42      protected final long lvProducerIndex() {
43          return producerIndex;
44      }
45  
46      protected final void soTail(long v) {
47          UNSAFE.putOrderedLong(this, P_INDEX_OFFSET, v);
48      }
49  
50      public SpmcArrayQueueProducerField(int capacity) {
51          super(capacity);
52      }
53  }

64  abstract class SpmcArrayQueueConsumerField<E> extends SpmcArrayQueueL2Pad<E> {
65      protected final static long C_INDEX_OFFSET;
66      static {
67          try {
68              C_INDEX_OFFSET =
69                      UNSAFE.objectFieldOffset(SpmcArrayQueueConsumerField.class.getDeclaredField("consumerIndex"));
70          } catch (NoSuchFieldException e) {
71              throw new RuntimeException(e);
72          }
73      }
74      private volatile long consumerIndex;
75  
76      public SpmcArrayQueueConsumerField(int capacity) {
77          super(capacity);
78      }
79  
80      protected final long lvConsumerIndex() {
81          return consumerIndex;
82      }
83  
84      protected final boolean casHead(long expect, long newValue) {
85          return UNSAFE.compareAndSwapLong(this, C_INDEX_OFFSET, expect, newValue);
86      }
87  }

98  abstract class SpmcArrayQueueProducerIndexCacheField<E> extends SpmcArrayQueueMidPad<E> {
99      // This is separated from the consumerIndex which will be highly contended in the hope that this value spends most
100     // of it's time in a cache line that is Shared(and rarely invalidated)
101     private volatile long producerIndexCache;
102 
103     public SpmcArrayQueueProducerIndexCacheField(int capacity) {
104         super(capacity);
105     }
106 
107     protected final long lvProducerIndexCache() {
108         return producerIndexCache;
109     }
110 
111     protected final void svProducerIndexCache(long v) {
112         producerIndexCache = v;
113     }
114 }

116 abstract class SpmcArrayQueueL3Pad<E> extends SpmcArrayQueueProducerIndexCacheField<E> {
117     long p40, p41, p42, p43, p44, p45, p46;
118     long p30, p31, p32, p33, p34, p35, p36, p37;
119 
120     public SpmcArrayQueueL3Pad(int capacity) {
121         super(capacity);
122     }
123 }
124 
125 public final class SpmcArrayQueue<E> extends SpmcArrayQueueL3Pad<E> {
126 
127     public SpmcArrayQueue(final int capacity) {
128         super(capacity);
129     }
130 
131     @Override
132     public boolean offer(final E e) {
133         if (null == e) {
134             throw new NullPointerException("Null is not a valid element");
135         }
136         final E[] lb = buffer;
137         final long lMask = mask;
138         final long currProducerIndex = lvProducerIndex();
139         final long offset = calcElementOffset(currProducerIndex);
140         if (null != lvElement(lb, offset)) {
141             long size = currProducerIndex - lvConsumerIndex();
142             
143             if(size > lMask) {
144                 return false;
145             }
146             else {
147                 // spin wait for slot to clear, buggers wait freedom
148                 while(null != lvElement(lb, offset));
149             }
150         }
151         spElement(lb, offset, e);
152         // single producer, so store ordered is valid. It is also required to correctly publish the element
153         // and for the consumers to pick up the tail value.
154         soTail(currProducerIndex + 1);
155         return true;
156     }
157 
158     @Override
159     public E poll() {
160         long currentConsumerIndex;
161         final long currProducerIndexCache = lvProducerIndexCache();
162         do {
163             currentConsumerIndex = lvConsumerIndex();
164             if (currentConsumerIndex >= currProducerIndexCache) {
165                 long currProducerIndex = lvProducerIndex();
166                 if (currentConsumerIndex >= currProducerIndex) {
167                     return null;
168                 } else {
169                     svProducerIndexCache(currProducerIndex);
170                 }
171             }
172         } while (!casHead(currentConsumerIndex, currentConsumerIndex + 1));
173         // consumers are gated on latest visible tail, and so can't see a null value in the queue or overtake
174         // and wrap to hit same location.
175         final long offset = calcElementOffset(currentConsumerIndex);
176         final E[] lb = buffer;
177         // load plain, element happens before it's index becomes visible
178         final E e = lpElement(lb, offset);
179         // store ordered, make sure nulling out is visible. Producer is waiting for this value.
180         soElement(lb, offset, null);
181         return e;
182     }
183 
184     @Override
185     public E peek() {
186         long currentConsumerIndex;
187         final long currProducerIndexCache = lvProducerIndexCache();
188         E e;
189         do {
190             currentConsumerIndex = lvConsumerIndex();
191             if (currentConsumerIndex >= currProducerIndexCache) {
192                 long currProducerIndex = lvProducerIndex();
193                 if (currentConsumerIndex >= currProducerIndex) {
194                     return null;
195                 } else {
196                     svProducerIndexCache(currProducerIndex);
197                 }
198             }
199         } while (null == (e = lvElement(calcElementOffset(currentConsumerIndex))));
200         return e;
201     }
202 
203     @Override
204     public int size() {
205         /*
206          * It is possible for a thread to be interrupted or reschedule between the read of the producer and consumer
207          * indices, therefore protection is required to ensure size is within valid range. In the event of concurrent
208          * polls/offers to this method the size is OVER estimated as we read consumer index BEFORE the producer index.
209          */
210         long after = lvConsumerIndex();
211         while (true) {
212             final long before = after;
213             final long currentProducerIndex = lvProducerIndex();
214             after = lvConsumerIndex();
215             if (before == after) {
216                 return (int) (currentProducerIndex - after);
217             }
218         }
219     }
220     
221     @Override
222     public boolean isEmpty() {
223         // Order matters! 
224         // Loading consumer before producer allows for producer increments after consumer index is read.
225         // This ensures the correctness of this method at least for the consumer thread. Other threads POV is not really
226         // something we can fix here.
227         return (lvConsumerIndex() == lvProducerIndex());
228     }

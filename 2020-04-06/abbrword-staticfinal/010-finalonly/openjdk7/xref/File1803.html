32  package com.sun.jmx.examples.scandir;
33  
34  import java.net.InetAddress;
35  import java.net.UnknownHostException;
36  import java.util.HashMap;
37  import java.util.Map;
38  import javax.management.MBeanServerConnection;
39  import javax.management.remote.JMXConnector;
40  import javax.management.remote.JMXConnectorFactory;
41  import javax.management.remote.JMXServiceURL;
42  import javax.rmi.ssl.SslRMIClientSocketFactory;
43  
44  /**
45   * The ScanDirClient class is a very simple programmatic client example
46   * which is able to connect to a secured JMX <i>scandir</i> application.
47   * <p>The program initialize the connection environment map with the
48   * appropriate properties and credentials, and then connects to the
49   * secure JMX <i>scandir</i> daemon.</p>
50   * <p>It gets the application's current configuration and prints it on
51   * its <code>System.out</code>.</p>
52   * <p>The {@link #main main} method takes two arguments: the host on which
53   * the server is running (localhost), and the port number
54   * that was configured to start the server RMI Connector (4545).
55   * </p>
56   * @author Sun Microsystems, 2006 - All rights reserved.
57   **/
58  public class ScanDirClient {
59  
60      // This class has only a main.
61      private ScanDirClient() { }
62  
63      /**
64       * The usage string for the ScanDirClient.
65       */
66      public static final String USAGE = ScanDirClient.class.getSimpleName() +
67              " <server-host> <rmi-port-number>";
68  
69      /**
70       * Connects to a secured JMX <i>scandir</i> application.
71       * @param args The {@code main} method takes two parameters:
72       *        <ul>
73       *        <li>args[0] must be the server's host</li>
74       *        <li>args[1] must be the rmi port number at which the
75       *        JMX <i>scandir</i> daemon is listening for connections
76       *        - that is, the port number of its JMX RMI Connector which
77       *        was configured in {@code management.properties}
78       *        </li>
79       *        <ul>
80       **/
81      public static void main(String[] args) {
82          try {
83              // Check args
84              //
85              if (args==null || args.length!=2) {
86                  System.err.println("Bad number of arguments: usage is: \n\t" +
87                          USAGE);
88                  System.exit(1);
89              }
90              try {
91                  InetAddress.getByName(args[0]);
92              } catch (UnknownHostException x) {
93                  System.err.println("No such host: " + args[0]+
94                              "\n usage is: \n\t" + USAGE);
95                  System.exit(2);
96              } catch (Exception x) {
97                  System.err.println("Bad address: " + args[0]+
98                              "\n usage is: \n\t" + USAGE);
99                  System.exit(2);
100             }
101             try {
102                 if (Integer.parseInt(args[1]) <= 0) {
103                     System.err.println("Bad port value: " + args[1]+
104                             "\n usage is: \n\t" + USAGE);
105                     System.exit(2);
106                 }
107             } catch (Exception x) {
108                 System.err.println("Bad argument: " + args[1]+
109                         "\n usage is: \n\t" + USAGE);
110                 System.exit(2);
111             }
112 
113             // Create an environment map to hold connection properties
114             // like credentials etc... We will later pass this map
115             // to the JMX Connector.
116             //
117             System.out.println("\nInitialize the environment map");
118             final Map<String,Object> env = new HashMap<String,Object>();
119 
120             // Provide the credentials required by the server
121             // to successfully perform user authentication
122             //
123             final String[] credentials = new String[] { "guest" , "guestpasswd" };
124             env.put("jmx.remote.credentials", credentials);
125 
126             // Provide the SSL/TLS-based RMI Client Socket Factory required
127             // by the JNDI/RMI Registry Service Provider to communicate with
128             // the SSL/TLS-protected RMI Registry
129             //
130             env.put("com.sun.jndi.rmi.factory.socket",
131                     new SslRMIClientSocketFactory());
132 
133             // Create the RMI connector client and
134             // connect it to the RMI connector server
135             // args[0] is the server's host - localhost
136             // args[1] is the secure server port - 4545
137             //
138             System.out.println("\nCreate the RMI connector client and " +
139                     "connect it to the RMI connector server");
140             final JMXServiceURL url = new JMXServiceURL(
141                     "service:jmx:rmi:///jndi/rmi://"+args[0]+":"+args[1] +
142                     "/jmxrmi");
143 
144             System.out.println("Connecting to: "+url);
145             final JMXConnector jmxc = JMXConnectorFactory.connect(url, env);
146 
147             // Get an MBeanServerConnection
148             //
149             System.out.println("\nGet the MBeanServerConnection");
150             final MBeanServerConnection mbsc = jmxc.getMBeanServerConnection();
151 
152             // Create a proxy for the ScanManager MXBean
153             //
154             final ScanManagerMXBean proxy =
155                     ScanManager.newSingletonProxy(mbsc);
156 
157             // Get the ScanDirConfig MXBean from the scan manager
158             //
159             System.out.println(
160                     "\nGet ScanDirConfigMXBean from ScanManagerMXBean");
161             final ScanDirConfigMXBean configMBean =
162                     proxy.getConfigurationMBean();
163 
164             // Print the scan dir configuration
165             //
166             System.out.println(
167                     "\nGet 'Configuration' attribute on ScanDirConfigMXBean");
168             System.out.println("\nConfiguration:\n" +
169                     configMBean.getConfiguration());
170 
171             // Try to invoke the "close" method on the ScanManager MXBean.
172             //
173             // Should get a SecurityException as the user "guest" doesn't
174             // have readwrite access.
175             //
176             System.out.println("\nInvoke 'close' on ScanManagerMXBean");
177             try {
178                 proxy.close();
179             } catch (SecurityException e) {
180                 System.out.println("\nGot expected security exception: " + e);
181             }
182 
183             // Close MBeanServer connection
184             //
185             System.out.println("\nClose the connection to the server");
186             jmxc.close();
187             System.out.println("\nBye! Bye!");
188         } catch (Exception e) {
189             System.out.println("\nGot unexpected exception: " + e);
190             e.printStackTrace();
191             System.exit(3);
192         }
193     }
194 }

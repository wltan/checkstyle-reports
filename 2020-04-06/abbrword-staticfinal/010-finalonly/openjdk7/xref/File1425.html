66      /**
67       * Retrieve the service ticket for serverPrincipal from caller's Subject
68       * or from Subject obtained by logging in, or if not found, via the
69       * Ticket Granting Service using the TGT obtained from the Subject.
70       *
71       * Caller must have permission to:
72       *    - access and update Subject's private credentials
73       *    - create LoginContext
74       *    - read the auth.login.defaultCallbackHandler security property
75       *
76       * NOTE: This method is used by JSSE Kerberos Cipher Suites
77       */
78      public static KerberosTicket getTicketFromSubjectAndTgs(GSSCaller caller,
79          String clientPrincipal, String serverPrincipal, String tgsPrincipal,
80          AccessControlContext acc)
81          throws LoginException, KrbException, IOException {
82  
83          // 1. Try to find service ticket in acc subject
84          Subject accSubj = Subject.getSubject(acc);
85          KerberosTicket ticket = SubjectComber.find(accSubj,
86              serverPrincipal, clientPrincipal, KerberosTicket.class);
87  
88          if (ticket != null) {
89              return ticket;  // found it
90          }
91  
92          Subject loginSubj = null;
93          if (!GSSUtil.useSubjectCredsOnly(caller)) {
94              // 2. Try to get ticket from login
95              try {
96                  loginSubj = GSSUtil.login(caller, GSSUtil.GSS_KRB5_MECH_OID);
97                  ticket = SubjectComber.find(loginSubj,
98                      serverPrincipal, clientPrincipal, KerberosTicket.class);
99                  if (ticket != null) {
100                     return ticket; // found it
101                 }
102             } catch (LoginException e) {
103                 // No login entry to use
104                 // ignore and continue
105             }
106         }
107 
108         // Service ticket not found in subject or login
109         // Try to get TGT to acquire service ticket
110 
111         // 3. Try to get TGT from acc subject
112         KerberosTicket tgt = SubjectComber.find(accSubj,
113             tgsPrincipal, clientPrincipal, KerberosTicket.class);
114 
115         boolean fromAcc;
116         if (tgt == null && loginSubj != null) {
117             // 4. Try to get TGT from login subject
118             tgt = SubjectComber.find(loginSubj,
119                 tgsPrincipal, clientPrincipal, KerberosTicket.class);
120             fromAcc = false;
121         } else {
122             fromAcc = true;
123         }
124 
125         // 5. Try to get service ticket using TGT
126         if (tgt != null) {
127             Credentials tgtCreds = ticketToCreds(tgt);
128             Credentials serviceCreds = Credentials.acquireServiceCreds(
129                         serverPrincipal, tgtCreds);
130             if (serviceCreds != null) {
131                 ticket = credsToTicket(serviceCreds);
132 
133                 // Store service ticket in acc's Subject
134                 if (fromAcc && accSubj != null && !accSubj.isReadOnly()) {
135                     accSubj.getPrivateCredentials().add(ticket);
136                 }
137             }
138         }
139         return ticket;
140     }

142     /**
143      * Retrieves the ticket corresponding to the client/server principal
144      * pair from the Subject in the specified AccessControlContext.
145      * If the ticket can not be found in the Subject, and if
146      * useSubjectCredsOnly is false, then obtain ticket from
147      * a LoginContext.
148      */
149     static KerberosTicket getTicket(GSSCaller caller,
150         String clientPrincipal, String serverPrincipal,
151         AccessControlContext acc) throws LoginException {
152 
153         // Try to get ticket from acc's Subject
154         Subject accSubj = Subject.getSubject(acc);
155         KerberosTicket ticket =
156             SubjectComber.find(accSubj, serverPrincipal, clientPrincipal,
157                   KerberosTicket.class);
158 
159         // Try to get ticket from Subject obtained from GSSUtil
160         if (ticket == null && !GSSUtil.useSubjectCredsOnly(caller)) {
161             Subject subject = GSSUtil.login(caller, GSSUtil.GSS_KRB5_MECH_OID);
162             ticket = SubjectComber.find(subject,
163                 serverPrincipal, clientPrincipal, KerberosTicket.class);
164         }
165         return ticket;
166     }

168     /**
169      * Retrieves the caller's Subject, or Subject obtained by logging in
170      * via the specified caller.
171      *
172      * Caller must have permission to:
173      *    - access the Subject
174      *    - create LoginContext
175      *    - read the auth.login.defaultCallbackHandler security property
176      *
177      * NOTE: This method is used by JSSE Kerberos Cipher Suites
178      */
179     public static Subject getSubject(GSSCaller caller,
180         AccessControlContext acc) throws LoginException {
181 
182         // Try to get the Subject from acc
183         Subject subject = Subject.getSubject(acc);
184 
185         // Try to get Subject obtained from GSSUtil
186         if (subject == null && !GSSUtil.useSubjectCredsOnly(caller)) {
187             subject = GSSUtil.login(caller, GSSUtil.GSS_KRB5_MECH_OID);
188         }
189         return subject;
190     }

192     // A special KerberosKey, used as keys read from a KeyTab object.
193     // Each time new keys are read from KeyTab objects in the private
194     // credentials set, old ones are removed and new ones added.
195     public static class KeysFromKeyTab extends KerberosKey {
196         public KeysFromKeyTab(KerberosKey key) {
197             super(key.getPrincipal(), key.getEncoded(),
198                     key.getKeyType(), key.getVersionNumber());
199         }
200     }

202     /**
203      * Credentials of a service, the private secret to authenticate its
204      * identity, which can be:
205      *   1. Some KerberosKeys (generated from password)
206      *   2. A KeyTab (for a typical service)
207      *   3. A TGT (for a user2user service. Not supported yet)
208      *
209      * Note that some creds can coexist. For example, a user2user service
210      * can use its keytab (or keys) if the client can successfully obtain a
211      * normal service ticket, otherwise, it can uses the TGT (actually, the
212      * session key of the TGT) if the client can only acquire a service ticket
213      * of ENC-TKT-IN-SKEY style.
214      */
215     public static class ServiceCreds {
216         private KerberosPrincipal kp;
217         private List<KeyTab> ktabs;
218         private List<KerberosKey> kk;
219         private Subject subj;
220         //private KerberosTicket tgt;   // user2user, not supported yet
221 
222         private static ServiceCreds getInstance(
223                 Subject subj, String serverPrincipal) {
224 
225             ServiceCreds sc = new ServiceCreds();
226             sc.subj = subj;
227 
228             for (KerberosPrincipal p: subj.getPrincipals(KerberosPrincipal.class)) {
229                 if (serverPrincipal == null ||
230                         p.getName().equals(serverPrincipal)) {
231                     sc.kp = p;
232                     serverPrincipal = p.getName();
233                     break;
234                 }
235             }
236             if (sc.kp == null) {
237                 // Compatibility with old behavior: even when there is no
238                 // KerberosPrincipal, we can find one from KerberosKeys
239                 List<KerberosKey> keys = SubjectComber.findMany(
240                         subj, null, null, KerberosKey.class);
241                 if (!keys.isEmpty()) {
242                     sc.kp = keys.get(0).getPrincipal();
243                     serverPrincipal = sc.kp.getName();
244                     if (DEBUG) {
245                         System.out.println(">>> ServiceCreds: no kp?"
246                                 + " find one from kk: " + serverPrincipal);
247                     }
248                 } else {
249                     return null;
250                 }
251             }
252             sc.ktabs = SubjectComber.findMany(
253                         subj, null, null, KeyTab.class);
254             sc.kk = SubjectComber.findMany(
255                         subj, serverPrincipal, null, KerberosKey.class);
256             if (sc.ktabs.isEmpty() && sc.kk.isEmpty()) {
257                 return null;
258             }
259             return sc;
260         }
261 
262         public String getName() {
263             return kp.getName();
264         }
265 
266         public KerberosKey[] getKKeys() {
267             if (ktabs.isEmpty()) {
268                 return kk.toArray(new KerberosKey[kk.size()]);
269             } else {
270                 List<KerberosKey> keys = new ArrayList<>();
271                 for (KeyTab ktab: ktabs) {
272                     for (KerberosKey k: ktab.getKeys(kp)) {
273                         keys.add(k);
274                     }
275                 }
276                 // Compatibility: also add keys to privCredSet. Remove old
277                 // ones first, only remove those from keytab.
278                 if (!subj.isReadOnly()) {
279                     Set<Object> pcs = subj.getPrivateCredentials();
280                     synchronized (pcs) {
281                         Iterator<Object> iterator = pcs.iterator();
282                         while (iterator.hasNext()) {
283                             Object obj = iterator.next();
284                             if (obj instanceof KeysFromKeyTab) {
285                                 KerberosKey key = (KerberosKey)obj;
286                                 if (Objects.equals(key.getPrincipal(), kp)) {
287                                     iterator.remove();
288                                 }
289                             }
290                         }
291                     }
292                     for (KerberosKey key: keys) {
293                         subj.getPrivateCredentials().add(new KeysFromKeyTab(key));
294                     }
295                 }
296                 return keys.toArray(new KerberosKey[keys.size()]);
297             }
298         }
299 
300         public EncryptionKey[] getEKeys() {
301             KerberosKey[] kkeys = getKKeys();
302             EncryptionKey[] ekeys = new EncryptionKey[kkeys.length];
303             for (int i=0; i<ekeys.length; i++) {
304                 ekeys[i] =  new EncryptionKey(
305                             kkeys[i].getEncoded(), kkeys[i].getKeyType(),
306                             new Integer(kkeys[i].getVersionNumber()));
307             }
308             return ekeys;
309         }
310 
311         public void destroy() {
312             kp = null;
313             ktabs = null;
314             kk = null;
315         }
316     }
317     /**
318      * Retrieves the ServiceCreds for the specified server principal from
319      * the Subject in the specified AccessControlContext. If not found, and if
320      * useSubjectCredsOnly is false, then obtain from a LoginContext.
321      *
322      * NOTE: This method is also used by JSSE Kerberos Cipher Suites
323      */
324     public static ServiceCreds getServiceCreds(GSSCaller caller,
325         String serverPrincipal, AccessControlContext acc)
326                 throws LoginException {
327 
328         Subject accSubj = Subject.getSubject(acc);
329         ServiceCreds sc = null;
330         if (accSubj != null) {
331             sc = ServiceCreds.getInstance(accSubj, serverPrincipal);
332         }
333         if (sc == null && !GSSUtil.useSubjectCredsOnly(caller)) {
334             Subject subject = GSSUtil.login(caller, GSSUtil.GSS_KRB5_MECH_OID);
335             sc = ServiceCreds.getInstance(subject, serverPrincipal);
336         }
337         return sc;
338     }

340     public static KerberosTicket credsToTicket(Credentials serviceCreds) {
341         EncryptionKey sessionKey =  serviceCreds.getSessionKey();
342         return new KerberosTicket(
343             serviceCreds.getEncoded(),
344             new KerberosPrincipal(serviceCreds.getClient().getName()),
345             new KerberosPrincipal(serviceCreds.getServer().getName(),
346                                 KerberosPrincipal.KRB_NT_SRV_INST),
347             sessionKey.getBytes(),
348             sessionKey.getEType(),
349             serviceCreds.getFlags(),
350             serviceCreds.getAuthTime(),
351             serviceCreds.getStartTime(),
352             serviceCreds.getEndTime(),
353             serviceCreds.getRenewTill(),
354             serviceCreds.getClientAddresses());
355     };
356 
357     public static Credentials ticketToCreds(KerberosTicket kerbTicket)
358         throws KrbException, IOException {
359         return new Credentials(
360             kerbTicket.getEncoded(),
361             kerbTicket.getClient().getName(),
362             kerbTicket.getServer().getName(),
363             kerbTicket.getSessionKey().getEncoded(),
364             kerbTicket.getSessionKeyType(),
365             kerbTicket.getFlags(),
366             kerbTicket.getAuthTime(),
367             kerbTicket.getStartTime(),
368             kerbTicket.getEndTime(),
369             kerbTicket.getRenewTill(),
370             kerbTicket.getClientAddresses());
371     }

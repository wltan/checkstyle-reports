641     /**
642      * Constructs an AES Key according to the specified key size and
643      * round number.
644      * @param len key size in bytes, i.e. 16, 24, or 32
645      * @param rounds round number starting from 0, i.e. valid from 0 to len-1.
646      */
647     private static SecretKey constructAESKey(int len, int rounds)
648         throws IllegalArgumentException {
649         if ((len != 16) && (len != 24) && (len != 32)) {
650             throw new IllegalArgumentException("Wrong Key Length: " + len);
651         }
652         byte[] rawKeyValue = constructKeyValue(len, rounds);
653         SecretKeySpec key = new SecretKeySpec(rawKeyValue, "AES");
654         return key;
655     }

657     private static byte[] constructKeyValue(int keysize, int rounds) {
658         byte[] tempKeyValue = new byte[keysize];
659         Arrays.fill(tempKeyValue, (byte)0);
660 
661         int whichByte = rounds/8;
662         int whichDigit = rounds % 8;
663         if ((whichByte >= keysize) || (whichDigit < 0) ||
664             (whichDigit > 8)) {
665             throw new IllegalArgumentException("Invalid keysize/rounds " +
666                                                "combination :" + keysize +
667                                                "/" + rounds);
668         }
669         switch (whichDigit) {
670         case 0:
671             tempKeyValue[whichByte] = (byte)0x80;
672             break;
673         case 1:
674             tempKeyValue[whichByte] = (byte)0x40;
675             break;
676         case 2:
677             tempKeyValue[whichByte] = (byte)0x20;
678             break;
679         case 3:
680             tempKeyValue[whichByte] = (byte)0x10;
681             break;
682         case 4:
683             tempKeyValue[whichByte] = (byte)0x08;
684             break;
685         case 5:
686             tempKeyValue[whichByte] = (byte)0x04;
687             break;
688         case 6:
689             tempKeyValue[whichByte] = (byte)0x02;
690             break;
691         case 7:
692             tempKeyValue[whichByte] = (byte)0x01;
693             break;
694         }
695         return tempKeyValue;
696     }

698     private static byte[] constructByteArray(String s) {
699         int len = s.length()/2;
700         byte[] tempValue = new byte[len];
701         for (int i = 0; i < len; i++) {
702             tempValue[i] = Integer.valueOf(s.substring(2*i, 2*i+2),
703                                            16).byteValue();
704         }
705         return tempValue;
706     }

708     public boolean execute() throws Exception {
709         String transformation = ALGO+"/"+MODE+"/"+PADDING;
710         Cipher c = Cipher.getInstance(transformation, "SunJCE");
711 
712         for (int i=0; i<KEY_SIZES.length; i++) {
713             if (KEY_SIZES[i]*8 >
714                 Cipher.getMaxAllowedKeyLength(transformation)) {
715                 // skip if this key length is larger than what's
716                 // configured in the jce jurisdiction policy files
717                 continue;
718             }
719             int rounds = KEY_SIZES[i] * 8;
720             byte[] plainText = PT;
721             byte[] cipherText = null;
722             try {
723             for (int j=0; j < rounds; j++) {
724                 SecretKey aesKey = constructAESKey(KEY_SIZES[i], j);
725                 c.init(Cipher.ENCRYPT_MODE, aesKey);
726                 cipherText = c.doFinal(plainText);
727                 byte[] answer = constructByteArray(CTS[i][j]);
728                 if (!Arrays.equals(cipherText, answer)) {
729                     throw new Exception((i+1) + "th known answer test failed for encryption");
730                 }
731                 c.init(Cipher.DECRYPT_MODE, aesKey);
732                 byte[] restored = c.doFinal(cipherText);
733                 if (!Arrays.equals(plainText, restored)) {
734                     throw new Exception((i+1) + "th known answer test failed for decryption");
735                 }
736             }
737             System.out.println("Finished KAT for " + KEY_SIZES[i] + "-byte key");
738             } catch (SecurityException se) {
739                 TestUtil.handleSE(se);
740             }
741         }
742 
743         // passed all tests...hooray!
744         return true;
745     }

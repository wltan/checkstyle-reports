1   package tomcat;
2   
3   import java.io.FileInputStream;
4   import java.io.IOException;
5   import java.security.KeyStore;
6   import java.security.KeyStoreException;
7   import java.security.NoSuchAlgorithmException;
8   import java.security.cert.CertificateException;
9   
10  public class JSSESocketFactory {
11      private KeyStore initKeyStore(String keystoreFile, String keyPass) throws IOException {
12          java.io.InputStream istream = null;
13          KeyStore kstore;
14          try {
15              kstore = KeyStore.getInstance("JKS");
16              istream = new FileInputStream(keystoreFile);
17              kstore.load(istream, keyPass.toCharArray());
18              return kstore;
19          } catch (KeyStoreException e) {
20              throw new IOException(e.getMessage());
21          } catch (NoSuchAlgorithmException e) {
22              throw new IOException(e.getMessage());
23          } catch (CertificateException e) {
24              throw new IOException(e.getMessage());
25          }
26  
27      }
28  
29  }

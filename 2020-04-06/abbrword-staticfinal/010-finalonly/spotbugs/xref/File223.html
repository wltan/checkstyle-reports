143      *
144      * @param methodName
145      *            name of the method
146      * @param methodSig

158      *            the Instruction

161      *
162      * @return true if the instruction is a monitor wait, false if not

164     public static boolean isMonitorWait(Instruction ins, ConstantPoolGen cpg) {
165         if (!(ins instanceof InvokeInstruction)) {
166             return false;
167         }

169             return false;
170         }
171 
172         InvokeInstruction inv = (InvokeInstruction) ins;
173         String methodName = inv.getMethodName(cpg);
174         String methodSig = inv.getSignature(cpg);
175 
176         return isMonitorWait(methodName, methodSig);
177     }

293     /**
294      * <p>Find the least upper bound method in the class hierarchy which could be
295      * called by the given InvokeInstruction. One reason this method is useful
296      * is that it indicates which declared exceptions are thrown by the called
297      * methods.</p>
298      * <ul>
299      * <li>For invokespecial, this is simply an exact lookup.</li>
300      * <li>For invokestatic and invokevirtual, the named class is searched,
301      * followed by superclasses up to the root of the object hierarchy
302      * (java.lang.Object). Yes, invokestatic really is declared to check
303      * superclasses. See VMSpec, 2nd ed, sec. 5.4.3.3.</li>
304      * <li>For invokeinterface, the named class is searched, followed by all
305      * interfaces transitively declared by the class. (Question: is the order
306      * important here? Maybe the VM spec requires that the actual interface
307      * desired is given, so the extended lookup will not be required. Should
308      * check.)</li>
309      * </ul>
310      *
311      * @param inv
312      *            the InvokeInstruction
313      * @param cpg
314      *            the ConstantPoolGen used by the class the InvokeInstruction
315      *            belongs to
316      * @return the JavaClassAndMethod, or null if no matching method can be
317      *         found
318      */
319     public static @CheckForNull
320     JavaClassAndMethod findInvocationLeastUpperBound(InvokeInstruction inv, ConstantPoolGen cpg) throws ClassNotFoundException {
321         return findInvocationLeastUpperBound(inv, cpg, ANY_METHOD);
322     }
323 
324     public static @CheckForNull
325     JavaClassAndMethod findInvocationLeastUpperBound(InvokeInstruction inv, ConstantPoolGen cpg,
326             JavaClassAndMethodChooser methodChooser) throws ClassNotFoundException {
327 
328         if (DEBUG_METHOD_LOOKUP) {
329             System.out.println("Find prototype method for " + SignatureConverter.convertMethodSignature(inv, cpg));
330         }
331 
332         short opcode = inv.getOpcode();
333 
334         if (opcode == Const.INVOKESTATIC) {
335             if (methodChooser == INSTANCE_METHOD) {
336                 return null;
337             }
338         } else {
339             if (methodChooser == STATIC_METHOD) {
340                 return null;
341             }
342         }
343 
344         // Find the method
345         if (opcode == Const.INVOKESPECIAL) {
346             // Non-virtual dispatch
347             return findExactMethod(inv, cpg, methodChooser);
348         }
349         if (opcode == Const.INVOKEDYNAMIC) {
350             return null;
351         }
352         String className = inv.getClassName(cpg);
353         String methodName = inv.getName(cpg);
354         String methodSig = inv.getSignature(cpg);
355         if (DEBUG_METHOD_LOOKUP) {
356             System.out.println("[Class name is " + className + "]");
357             System.out.println("[Method name is " + methodName + "]");
358             System.out.println("[Method signature is " + methodSig + "]");
359         }
360 
361         if (className.startsWith("[")) {
362             // Java 1.5 allows array classes to appear as the class name
363             className = "java.lang.Object";
364         }
365 
366         JavaClass jClass = Repository.lookupClass(className);
367         return findInvocationLeastUpperBound(jClass, methodName, methodSig, methodChooser,
368                 opcode == Const.INVOKEINTERFACE);
369     }

371     public static @CheckForNull
372     JavaClassAndMethod findInvocationLeastUpperBound(JavaClass jClass, String methodName, String methodSig,
373             JavaClassAndMethodChooser methodChooser, boolean invokeInterface) throws ClassNotFoundException {
374         JavaClassAndMethod result = findMethod(jClass, methodName, methodSig, methodChooser);
375         if (result != null) {
376             return result;
377         }
378         if (invokeInterface) {
379             for (JavaClass i : jClass.getInterfaces()) {
380                 result = findInvocationLeastUpperBound(i, methodName, methodSig, methodChooser, invokeInterface);
381                 if (result != null) {
382                     return null;
383                 }
384             }
385         } else {
386             JavaClass sClass = jClass.getSuperClass();
387             if (sClass != null) {
388                 return findInvocationLeastUpperBound(sClass, methodName, methodSig, methodChooser, invokeInterface);
389             }
390         }
391         return null;
392 
393     }

431     public static @CheckForNull
432     JavaClassAndMethod findMethod(JavaClass javaClass, String methodName, String methodSig, JavaClassAndMethodChooser chooser) {
433         if (DEBUG_METHOD_LOOKUP) {
434             System.out.println("Check " + javaClass.getClassName());
435         }
436         Method[] methodList = javaClass.getMethods();
437         for (Method method : methodList) {
438             if (method.getName().equals(methodName) && method.getSignature().equals(methodSig)) {
439                 JavaClassAndMethod m = new JavaClassAndMethod(javaClass, method);
440                 if (chooser.choose(m)) {
441                     if (DEBUG_METHOD_LOOKUP) {
442                         System.out.println("\t==> FOUND: " + method);
443                     }
444                     return m;
445                 }
446             }
447         }
448         if (DEBUG_METHOD_LOOKUP) {
449             System.out.println("\t==> NOT FOUND");
450         }
451         return null;
452     }

690             JavaClassAndMethod m;
691             if ((m = findMethod(cls, methodName, methodSig)) != null && chooser.choose(m)) {
692                 return XFactory.createXMethod(cls, m.getMethod());
693             }
694         }
695         return null;
696     }

698     /**
699      * Resolve possible method call targets. This works for both static and
700      * instance method calls.
701      *
702      * @param invokeInstruction
703      *            the InvokeInstruction
704      * @param typeFrame
705      *            the TypeFrame containing the types of stack values
706      * @param cpg
707      *            the ConstantPoolGen
708      * @return Set of methods which might be called
709      * @throws DataflowAnalysisException
710      * @throws ClassNotFoundException
711      */
712     public static Set<JavaClassAndMethod> resolveMethodCallTargets(InvokeInstruction invokeInstruction, TypeFrame typeFrame,
713             ConstantPoolGen cpg) throws DataflowAnalysisException, ClassNotFoundException {
714 
715         short opcode = invokeInstruction.getOpcode();
716 
717         if (opcode == Const.INVOKESTATIC) {
718             HashSet<JavaClassAndMethod> result = new HashSet<>();
719             JavaClassAndMethod targetMethod = findInvocationLeastUpperBound(invokeInstruction, cpg, CONCRETE_METHOD);
720             if (targetMethod != null) {
721                 result.add(targetMethod);
722             }
723             return result;
724         }
725 
726         if (!typeFrame.isValid()) {
727             return new HashSet<>();
728         }
729 
730         Type receiverType;
731         boolean receiverTypeIsExact;
732 
733         if (opcode == Const.INVOKESPECIAL) {
734             // invokespecial instructions are dispatched to EXACTLY
735             // the class specified by the instruction
736             receiverType = ObjectTypeFactory.getInstance(invokeInstruction.getClassName(cpg));
737             receiverTypeIsExact = false; // Doesn't actually matter
738         } else if (opcode == Const.INVOKEDYNAMIC) {
739             // XXX handle INVOKEDYNAMIC
740             return new HashSet<>();
741         } else {
742             // For invokevirtual and invokeinterface instructions, we have
743             // virtual dispatch. By taking the receiver type (which may be a
744             // subtype of the class specified by the instruction),
745             // we may get a more precise set of call targets.
746             int instanceStackLocation = typeFrame.getInstanceStackLocation(invokeInstruction, cpg);
747             receiverType = typeFrame.getStackValue(instanceStackLocation);
748             if (!(receiverType instanceof ReferenceType)) {
749                 return new HashSet<>();
750             }
751             receiverTypeIsExact = typeFrame.isExact(instanceStackLocation);
752         }
753         if (DEBUG_METHOD_LOOKUP) {
754             System.out.println("[receiver type is " + receiverType + ", " + (receiverTypeIsExact ? "exact]" : " not exact]"));
755         }
756 
757         return resolveMethodCallTargets((ReferenceType) receiverType, invokeInstruction, cpg, receiverTypeIsExact);
758     }

763      *
764      * @param receiverType

766      * @param invokeInstruction

768      * @param cpg

778     /**
779      * Resolve possible instance method call targets.
780      *
781      * @param receiverType
782      *            type of the receiver object
783      * @param invokeInstruction
784      *            the InvokeInstruction
785      * @param cpg
786      *            the ConstantPoolGen
787      * @param receiverTypeIsExact
788      *            if true, the receiver type is known exactly, which should
789      *            allow a precise result
790      * @return Set of methods which might be called
791      * @throws ClassNotFoundException
792      */
793     public static Set<JavaClassAndMethod> resolveMethodCallTargets(ReferenceType receiverType,
794             InvokeInstruction invokeInstruction, ConstantPoolGen cpg, boolean receiverTypeIsExact) throws ClassNotFoundException {
795         HashSet<JavaClassAndMethod> result = new HashSet<>();
796 
797         if (invokeInstruction.getOpcode() == Const.INVOKESTATIC) {
798             throw new IllegalArgumentException();
799         }
800 
801         String methodName = invokeInstruction.getName(cpg);
802         String methodSig = invokeInstruction.getSignature(cpg);
803 
804         // Array method calls aren't virtual.
805         // They should just resolve to Object methods.
806         if (receiverType instanceof ArrayType) {
807             JavaClass javaLangObject = AnalysisContext.currentAnalysisContext().lookupClass("java.lang.Object");
808             JavaClassAndMethod classAndMethod = findMethod(javaLangObject, methodName, methodSig, INSTANCE_METHOD);
809             if (classAndMethod != null) {
810                 result.add(classAndMethod);
811             }
812             return result;
813         }
814 
815         if (receiverType instanceof NullType) {
816             return Collections.<JavaClassAndMethod> emptySet();
817         }
818         AnalysisContext analysisContext = AnalysisContext.currentAnalysisContext();
819 
820         // Get the receiver class.
821         String receiverClassName = ((ObjectType) receiverType).getClassName();
822         JavaClass receiverClass = analysisContext.lookupClass(receiverClassName);
823         ClassDescriptor receiverDesc = DescriptorFactory.createClassDescriptorFromDottedClassName(receiverClassName);
824 
825         // Figure out the upper bound for the method.
826         // This is what will be called if this is not a virtual call site.
827         JavaClassAndMethod upperBound = findMethod(receiverClass, methodName, methodSig, CONCRETE_METHOD);
828         if (upperBound == null) {
829             upperBound = findInvocationLeastUpperBound(receiverClass, methodName, methodSig, CONCRETE_METHOD, false);
830         }
831         if (upperBound != null) {
832             if (DEBUG_METHOD_LOOKUP) {
833                 System.out.println("Adding upper bound: "
834                         + SignatureConverter.convertMethodSignature(upperBound.getJavaClass(), upperBound.getMethod()));
835             }
836             result.add(upperBound);
837         }
838 
839         // Is this a virtual call site?
840         boolean virtualCall = (invokeInstruction.getOpcode() == Const.INVOKEVIRTUAL || invokeInstruction.getOpcode() == Const.INVOKEINTERFACE)
841                 && (upperBound == null || !upperBound.getJavaClass().isFinal() && !upperBound.getMethod().isFinal())
842                 && !receiverTypeIsExact;
843 
844         if (virtualCall) {
845             if (!"java.lang.Object".equals(receiverClassName)) {
846 
847                 // This is a true virtual call: assume that any concrete
848                 // subtype method may be called.
849                 Set<ClassDescriptor> subTypeSet = analysisContext.getSubtypes2().getSubtypes(receiverDesc);
850                 for (ClassDescriptor subtype : subTypeSet) {
851                     XMethod concreteSubtypeMethod = findMethod(subtype, methodName, methodSig, false);
852                     if (concreteSubtypeMethod != null && (concreteSubtypeMethod.getAccessFlags() & Const.ACC_ABSTRACT) == 0) {
853                         result.add(new JavaClassAndMethod(concreteSubtypeMethod));
854                     }
855                 }
856                 if (false && subTypeSet.size() > 500) {
857                     new RuntimeException(receiverClassName + " has " + subTypeSet.size() + " subclasses, " + result.size()
858                             + " of which implement " + methodName + methodSig + " " + invokeInstruction)
859                     .printStackTrace(System.out);
860                 }
861 
862             }
863         }
864         return result;
865     }

880     /**
881      * Find a field with given name defined in given class.
882      *
883      * @param className
884      *            the name of the class
885      * @param fieldName
886      *            the name of the field
887      * @return the Field, or null if no such field could be found
888      */
889     public static Field findField(String className, String fieldName) throws ClassNotFoundException {
890         JavaClass jclass = Repository.lookupClass(className);
891 
892         while (jclass != null) {
893             Field[] fieldList = jclass.getFields();
894             for (Field field : fieldList) {
895                 if (field.getName().equals(fieldName)) {
896                     return field;
897                 }
898             }
899 
900             jclass = jclass.getSuperClass();
901         }
902 
903         return null;
904     }

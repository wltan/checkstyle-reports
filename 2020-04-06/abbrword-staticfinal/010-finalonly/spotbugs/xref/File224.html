68      /**
69       * Look up the method referenced by given InvokeInstruction. This method
70       * does <em>not</em> look for implementations in super or subclasses
71       * according to the virtual dispatch rules.
72       *
73       * @param inv
74       *            the InvokeInstruction
75       * @param cpg
76       *            the ConstantPoolGen used by the class the InvokeInstruction
77       *            belongs to
78       * @param chooser
79       *            JavaClassAndMethodChooser to use to pick the method from among
80       *            the candidates
81       * @return the JavaClassAndMethod, or null if no such method is defined in
82       *         the class
83       */
84      public static XMethod findExactMethod(InvokeInstruction inv, ConstantPoolGen cpg, JavaClassAndMethodChooser chooser)
85      {
86          String className = inv.getClassName(cpg);
87          String methodName = inv.getName(cpg);
88          String methodSig = inv.getSignature(cpg);
89  
90          XMethod result = findMethod(DescriptorFactory.createClassDescriptorFromDottedClassName(className), methodName, methodSig,
91                  inv instanceof INVOKESTATIC);
92  
93          return thisOrNothing(result, chooser);
94      }

96      private static @CheckForNull
97      XMethod thisOrNothing(@CheckForNull XMethod m, JavaClassAndMethodChooser chooser) {
98          if (m == null) {
99              return null;
100         }
101         if (chooser.choose(m)) {
102             return m;
103         }
104         return null;
105     }

107     public static @CheckForNull
108     XMethod findInvocationLeastUpperBound(InvokeInstruction inv, ConstantPoolGen cpg, JavaClassAndMethodChooser methodChooser)
109     {
110 
111         if (DEBUG_METHOD_LOOKUP) {
112             System.out.println("Find prototype method for " + SignatureConverter.convertMethodSignature(inv, cpg));
113         }
114 
115         short opcode = inv.getOpcode();
116 
117         if (opcode == Const.INVOKESTATIC) {
118             if (methodChooser == INSTANCE_METHOD) {
119                 return null;
120             }
121         } else {
122             if (methodChooser == STATIC_METHOD) {
123                 return null;
124             }
125         }
126 
127         // Find the method
128         if (opcode == Const.INVOKESPECIAL) {
129             // Non-virtual dispatch
130             return findExactMethod(inv, cpg, methodChooser);
131         } else {
132             String className = inv.getClassName(cpg);
133             String methodName = inv.getName(cpg);
134             String methodSig = inv.getSignature(cpg);
135             if (DEBUG_METHOD_LOOKUP) {
136                 System.out.println("[Class name is " + className + "]");
137                 System.out.println("[Method name is " + methodName + "]");
138                 System.out.println("[Method signature is " + methodSig + "]");
139             }
140 
141             if (className.startsWith("[")) {
142                 // Java 1.5 allows array classes to appear as the class name
143                 className = "java.lang.Object";
144             }
145 
146             try {
147                 return thisOrNothing(
148                         findInvocationLeastUpperBound(getXClassFromDottedClassName(className), methodName, methodSig,
149                                 opcode == Const.INVOKESTATIC, opcode == Const.INVOKEINTERFACE), methodChooser);
150             } catch (CheckedAnalysisException e) {
151                 return null;
152             }
153 
154         }
155     }

167     public static @CheckForNull
168     XMethod findInvocationLeastUpperBound(XClass jClass, String methodName, String methodSig, boolean invokeStatic,
169             boolean invokeInterface) {
170         XMethod result = findMethod(jClass.getClassDescriptor(), methodName, methodSig, invokeStatic);
171         if (result != null) {
172             return result;
173         }
174         ClassDescriptor sClass = jClass.getSuperclassDescriptor();
175         if (sClass != null) {
176             result = findInvocationLeastUpperBound(sClass, methodName, methodSig, invokeStatic, invokeInterface);
177             if (result != null) {
178                 return result;
179             }
180         }
181 
182         for (ClassDescriptor i : jClass.getInterfaceDescriptorList()) {
183             result = findInvocationLeastUpperBound(i, methodName, methodSig, invokeStatic, invokeInterface);
184             if (result != null) {
185                 return result;
186             }
187         }
188 
189         return null;
190     }

192     public static @CheckForNull
193     XMethod findInvocationLeastUpperBound0(XClass jClass, String methodName, String methodSig, boolean invokeStatic,
194             boolean invokeInterface) {
195         XMethod result = findMethod(jClass.getClassDescriptor(), methodName, methodSig, invokeStatic);
196         if (result != null) {
197             return result;
198         }
199         if (invokeInterface) {
200             for (ClassDescriptor i : jClass.getInterfaceDescriptorList()) {
201                 result = findInvocationLeastUpperBound(i, methodName, methodSig, invokeStatic, invokeInterface);
202                 if (result != null) {
203                     return result;
204                 }
205             }
206         } else {
207             ClassDescriptor sClass = jClass.getSuperclassDescriptor();
208             if (sClass != null) {
209                 return findInvocationLeastUpperBound(sClass, methodName, methodSig, invokeStatic, invokeInterface);
210             }
211         }
212         return null;
213     }

225     public static @CheckForNull XMethod findFirstSuperMethod(XMethod m) {
226 
227         try {
228             @CheckForNull ClassDescriptor c = m.getClassDescriptor();
229             XClass xc = getXClass(c);
230             c = xc.getSuperclassDescriptor();
231             while (c != null) {
232                 xc = getXClass(c);
233                 XMethod xm = xc.findMatchingMethod(m.getMethodDescriptor());
234                 if (xm != null) {
235                     return xm;
236                 }
237                 c = xc.getSuperclassDescriptor();
238             }
239         } catch (CheckedAnalysisException e) {
240             AnalysisContext.logError("Error finding super methods for " + m, e);
241         }
242         return null;
243     }

245     private static void findSuperMethods(@CheckForNull ClassDescriptor c, XMethod m, Set<XMethod> accumulator) {
246         if (c == null) {
247             return;
248         }
249         try {
250             XClass xc = getXClass(c);
251             XMethod xm = xc.findMatchingMethod(m.getMethodDescriptor());
252             if (xm != null && !accumulator.add(xm)) {
253                 return;
254             }
255             findSuperMethods(xc.getSuperclassDescriptor(), m, accumulator);
256             for (ClassDescriptor i : xc.getInterfaceDescriptorList()) {
257                 findSuperMethods(i, m, accumulator);
258             }
259             if (!accumulator.add(m)) {
260                 return;
261             }
262 
263         } catch (CheckedAnalysisException e) {
264             AnalysisContext.logError("Error finding super methods for " + m, e);
265         }
266     }

289     /**
290      * Resolve possible method call targets. This works for both static and
291      * instance method calls.
292      *
293      * @param invokeInstruction
294      *            the InvokeInstruction
295      * @param typeFrame
296      *            the TypeFrame containing the types of stack values
297      * @param cpg
298      *            the ConstantPoolGen
299      * @return Set of methods which might be called
300      * @throws DataflowAnalysisException
301      * @throws ClassNotFoundException
302      */
303     public static @Nonnull
304     Set<XMethod> resolveMethodCallTargets(InvokeInstruction invokeInstruction, TypeFrame typeFrame, ConstantPoolGen cpg)
305             throws DataflowAnalysisException, ClassNotFoundException {
306 
307         short opcode = invokeInstruction.getOpcode();
308 
309         if (opcode == Const.INVOKESTATIC) {
310             return Util.emptyOrNonnullSingleton(findInvocationLeastUpperBound(invokeInstruction, cpg, STATIC_METHOD));
311         }
312 
313         if (!typeFrame.isValid()) {
314             return Collections.<XMethod> emptySet();
315         }
316 
317         // XXX handle INVOKEDYNAMIC
318         if (opcode == Const.INVOKEDYNAMIC) {
319             return Collections.<XMethod> emptySet();
320         }
321 
322         Type receiverType;
323         boolean receiverTypeIsExact;
324 
325         if (opcode == Const.INVOKESPECIAL) {
326             // invokespecial instructions are dispatched to EXACTLY
327             // the class specified by the instruction
328             receiverType = ObjectTypeFactory.getInstance(invokeInstruction.getClassName(cpg));
329             receiverTypeIsExact = false; // Doesn't actually matter
330         } else {
331             // For invokevirtual and invokeinterface instructions, we have
332             // virtual dispatch. By taking the receiver type (which may be a
333             // subtype of the class specified by the instruction),
334             // we may get a more precise set of call targets.
335             int instanceStackLocation = typeFrame.getInstanceStackLocation(invokeInstruction, cpg);
336             receiverType = typeFrame.getStackValue(instanceStackLocation);
337             if (!(receiverType instanceof ReferenceType)) {
338                 return Collections.<XMethod> emptySet();
339             }
340             receiverTypeIsExact = typeFrame.isExact(instanceStackLocation);
341         }
342         if (DEBUG_METHOD_LOOKUP) {
343             System.out.println("[receiver type is " + receiverType + ", " + (receiverTypeIsExact ? "exact]" : " not exact]"));
344         }
345 
346         return resolveMethodCallTargets((ReferenceType) receiverType, invokeInstruction, cpg, receiverTypeIsExact);
347     }

352      *
353      * @param receiverType

355      * @param invokeInstruction

367     /**
368      * Resolve possible instance method call targets.
369      *
370      * @param receiverType
371      *            type of the receiver object
372      * @param invokeInstruction
373      *            the InvokeInstruction
374      * @param cpg
375      *            the ConstantPoolGen
376      * @param receiverTypeIsExact
377      *            if true, the receiver type is known exactly, which should
378      *            allow a precise result
379      * @return Set of methods which might be called
380      * @throws ClassNotFoundException
381      */
382     public static Set<XMethod> resolveMethodCallTargets(ReferenceType receiverType, InvokeInstruction invokeInstruction,
383             ConstantPoolGen cpg, boolean receiverTypeIsExact) throws ClassNotFoundException {
384 
385         if (invokeInstruction.getOpcode() == Const.INVOKESTATIC) {
386             throw new IllegalArgumentException();
387         }
388 
389         String methodName = invokeInstruction.getName(cpg);
390         String methodSig = invokeInstruction.getSignature(cpg);
391 
392         // Array method calls aren't virtual.
393         // They should just resolve to Object methods.
394         if (receiverType instanceof ArrayType) {
395             try {
396                 return Util.emptyOrNonnullSingleton(getXClass(objectDescriptor).findMethod(methodName, methodSig, false));
397             } catch (CheckedAnalysisException e) {
398                 return Collections.<XMethod> emptySet();
399             }
400         }
401 
402         if (receiverType instanceof ObjectType) {
403             // Get the receiver class.
404             String receiverClassName = ((ObjectType) receiverType).getClassName();
405 
406             return resolveVirtualMethodCallTargets(receiverClassName, methodName, methodSig, receiverTypeIsExact,
407                     invokeInstruction instanceof INVOKESPECIAL);
408         }
409         assert receiverType instanceof NullType;
410         return Collections.<XMethod> emptySet();
411 
412     }

426     private static final boolean OPEN_WORLD = SystemProperties.getBoolean("findbugs.openworld", true);
427 
428     private static final boolean OPEN_WORLD_DEBUG = SystemProperties.getBoolean("findbugs.openworld.debug", false);
429 
430     public static Set<XMethod> resolveVirtualMethodCallTargets(ClassDescriptor receiverDesc, String methodName, String methodSig,
431             boolean receiverTypeIsExact, boolean invokeSpecial) throws ClassNotFoundException {
432         // Figure out the upper bound for the method.
433         // This is what will be called if this is not a virtual call site.
434         AnalysisContext analysisContext = AnalysisContext.currentAnalysisContext();
435         XClass xClass;
436         try {
437             xClass = getXClass(receiverDesc);
438         } catch (CheckedAnalysisException e) {
439             return Collections.<XMethod> emptySet();
440         }
441 
442         HashSet<XMethod> result = new LinkedHashSet<>();
443         XMethod upperBound = findMethod(receiverDesc, methodName, methodSig, false);
444         if (upperBound == null) {
445             upperBound = findInvocationLeastUpperBound(xClass, methodName, methodSig, false, false);
446         }
447         if (upperBound != null) {
448             if (DEBUG_METHOD_LOOKUP) {
449                 System.out.println("Adding upper bound: " + upperBound);
450             }
451             result.add(upperBound);
452         }
453 
454         if (OPEN_WORLD_DEBUG) {
455             System.out.println("OWD: " + receiverDesc + "." + methodName + methodSig);
456             if (upperBound != null) {
457                 System.out.println("  upper bound:" + upperBound);
458             }
459         }
460         // Is this a virtual call site?
461         boolean virtualCall = (upperBound == null || !upperBound.isFinal()) && !receiverTypeIsExact && !invokeSpecial;
462 
463         if (virtualCall && !"java/lang/Object".equals(receiverDesc.getClassName())) {
464 
465             // This is a true virtual call: assume that any concrete
466             // subtype method may be called.
467             Set<ClassDescriptor> subTypeSet = analysisContext.getSubtypes2().getSubtypes(receiverDesc);
468             for (ClassDescriptor subtype : subTypeSet) {
469                 XMethod concreteSubtypeMethod = findMethod(subtype, methodName, methodSig, false);
470                 if (concreteSubtypeMethod != null && (OPEN_WORLD || !concreteSubtypeMethod.isAbstract())) {
471                     if (OPEN_WORLD_DEBUG) {
472                         System.out.println("  -> " + concreteSubtypeMethod);
473                     }
474 
475                     result.add(concreteSubtypeMethod);
476                 }
477             }
478 
479         }
480         return result;
481     }

483     /**
484      * Find the declared exceptions for the method called by given instruction.
485      *
486      * @param inv
487      *            the InvokeInstruction
488      * @param cpg
489      *            the ConstantPoolGen used by the class the InvokeInstruction
490      *            belongs to
491      * @return array of ObjectTypes of thrown exceptions, or null if we can't
492      *         find the method implementation
493      */
494     public static @CheckForNull
495     ObjectType[] findDeclaredExceptions(InvokeInstruction inv, ConstantPoolGen cpg)  {
496         XMethod method = findInvocationLeastUpperBound(inv, cpg, inv instanceof INVOKESTATIC ? Hierarchy.STATIC_METHOD
497                 : Hierarchy.INSTANCE_METHOD);
498 
499         if (method == null) {
500             return null;
501         }
502         String[] exceptions = method.getThrownExceptions();
503 
504         if (exceptions == null) {
505             return new ObjectType[0];
506         }
507 
508         ObjectType[] result = new ObjectType[exceptions.length];
509         for (int i = 0; i < exceptions.length; ++i) {
510             result[i] = ObjectTypeFactory.getInstance(ClassName.toDottedClassName(exceptions[i]));
511         }
512         return result;
513     }

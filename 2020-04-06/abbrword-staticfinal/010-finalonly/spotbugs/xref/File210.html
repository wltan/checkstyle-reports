20  package edu.umd.cs.findbugs.ba;
21  
22  import org.apache.bcel.Const;
23  
24  import edu.umd.cs.findbugs.SystemProperties;
25  
26  /**
27   * Scan the raw bytecodes of a method. This is useful in order to find out
28   * quickly whether or not a method uses particular instructions.
29   *
30   * @author David Hovemeyer
31   */
32  public class BytecodeScanner implements org.apache.bcel.Constants {
33      private static final boolean DEBUG = SystemProperties.getBoolean("bs.debug");
34  
35      /**
36       * Callback interface to report scanned instructions.
37       */
38      public interface Callback {
39          /**
40           * Called to indicate that a particular bytecode has been scanned.
41           *
42           * @param opcode
43           *            the opcode of the instruction
44           * @param index
45           *            the bytecode offset of the instruction
46           */
47          public void handleInstruction(int opcode, int index);
48      }
49  
50      /**
51       * Convert the unsigned value of a byte into a short.
52       *
53       * @param value
54       *            the byte
55       * @return the byte's unsigned value as a short
56       */
57      private static short unsignedValueOf(byte value) {
58          short result;
59          if ((value & 0x80) != 0) {
60              result = (short) (value & 0x7F);
61              result |= 0x80;
62          } else {
63              result = value;
64          }
65          return result;
66      }
67  
68      /**
69       * Extract an int from bytes at the given offset in the array.
70       *
71       * @param arr
72       *            the array
73       * @param offset
74       *            the offset in the array
75       */
76      private static int extractInt(byte[] arr, int offset) {
77          return ((arr[offset] & 0xFF) << 24) | ((arr[offset + 1] & 0xFF) << 16) | ((arr[offset + 2] & 0xFF) << 8)
78                  | (arr[offset + 3] & 0xFF);
79      }
80  
81      private static final int PAD[] = { 0, 3, 2, 1 };
82  
83      /**
84       * Scan the raw bytecodes of a method.
85       *
86       * @param instructionList
87       *            the bytecodes
88       * @param callback
89       *            the callback object
90       */
91      public void scan(byte[] instructionList, Callback callback) {
92  
93          boolean wide = false;
94  
95          for (int index = 0; index < instructionList.length;) {
96              short opcode = unsignedValueOf(instructionList[index]);
97              callback.handleInstruction(opcode, index);
98  
99              if (DEBUG) {
100                 System.out.println(index + ": " + Const.getOpcodeName(opcode));
101             }
102 
103             switch (opcode) {
104 
105             // Single byte instructions.
106             case Const.NOP:
107             case Const.ACONST_NULL:
108             case Const.ICONST_M1:
109             case Const.ICONST_0:
110             case Const.ICONST_1:
111             case Const.ICONST_2:
112             case Const.ICONST_3:
113             case Const.ICONST_4:
114             case Const.ICONST_5:
115             case Const.LCONST_0:
116             case Const.LCONST_1:
117             case Const.FCONST_0:
118             case Const.FCONST_1:
119             case Const.FCONST_2:
120             case Const.DCONST_0:
121             case Const.DCONST_1:
122             case Const.ILOAD_0:
123             case Const.ILOAD_1:
124             case Const.ILOAD_2:
125             case Const.ILOAD_3:
126             case Const.LLOAD_0:
127             case Const.LLOAD_1:
128             case Const.LLOAD_2:
129             case Const.LLOAD_3:
130             case Const.FLOAD_0:
131             case Const.FLOAD_1:
132             case Const.FLOAD_2:
133             case Const.FLOAD_3:
134             case Const.DLOAD_0:
135             case Const.DLOAD_1:
136             case Const.DLOAD_2:
137             case Const.DLOAD_3:
138             case Const.ALOAD_0:
139             case Const.ALOAD_1:
140             case Const.ALOAD_2:
141             case Const.ALOAD_3:
142             case Const.IALOAD:
143             case Const.LALOAD:
144             case Const.FALOAD:
145             case Const.DALOAD:
146             case Const.AALOAD:
147             case Const.BALOAD:
148             case Const.CALOAD:
149             case Const.SALOAD:
150             case Const.ISTORE_0:
151             case Const.ISTORE_1:
152             case Const.ISTORE_2:
153             case Const.ISTORE_3:
154             case Const.LSTORE_0:
155             case Const.LSTORE_1:
156             case Const.LSTORE_2:
157             case Const.LSTORE_3:
158             case Const.FSTORE_0:
159             case Const.FSTORE_1:
160             case Const.FSTORE_2:
161             case Const.FSTORE_3:
162             case Const.DSTORE_0:
163             case Const.DSTORE_1:
164             case Const.DSTORE_2:
165             case Const.DSTORE_3:
166             case Const.ASTORE_0:
167             case Const.ASTORE_1:
168             case Const.ASTORE_2:
169             case Const.ASTORE_3:
170             case Const.IASTORE:
171             case Const.LASTORE:
172             case Const.FASTORE:
173             case Const.DASTORE:
174             case Const.AASTORE:
175             case Const.BASTORE:
176             case Const.CASTORE:
177             case Const.SASTORE:
178             case Const.POP:
179             case Const.POP2:
180             case Const.DUP:
181             case Const.DUP_X1:
182             case Const.DUP_X2:
183             case Const.DUP2:
184             case Const.DUP2_X1:
185             case Const.DUP2_X2:
186             case Const.SWAP:
187             case Const.IADD:
188             case Const.LADD:
189             case Const.FADD:
190             case Const.DADD:
191             case Const.ISUB:
192             case Const.LSUB:
193             case Const.FSUB:
194             case Const.DSUB:
195             case Const.IMUL:
196             case Const.LMUL:
197             case Const.FMUL:
198             case Const.DMUL:
199             case Const.IDIV:
200             case Const.LDIV:
201             case Const.FDIV:
202             case Const.DDIV:
203             case Const.IREM:
204             case Const.LREM:
205             case Const.FREM:
206             case Const.DREM:
207             case Const.INEG:
208             case Const.LNEG:
209             case Const.FNEG:
210             case Const.DNEG:
211             case Const.ISHL:
212             case Const.LSHL:
213             case Const.ISHR:
214             case Const.LSHR:
215             case Const.IUSHR:
216             case Const.LUSHR:
217             case Const.IAND:
218             case Const.LAND:
219             case Const.IOR:
220             case Const.LOR:
221             case Const.IXOR:
222             case Const.LXOR:
223             case Const.I2L:
224             case Const.I2F:
225             case Const.I2D:
226             case Const.L2I:
227             case Const.L2F:
228             case Const.L2D:
229             case Const.F2I:
230             case Const.F2L:
231             case Const.F2D:
232             case Const.D2I:
233             case Const.D2L:
234             case Const.D2F:
235             case Const.I2B:
236             case Const.I2C:
237             case Const.I2S:
238             case Const.LCMP:
239             case Const.FCMPL:
240             case Const.FCMPG:
241             case Const.DCMPL:
242             case Const.DCMPG:
243             case Const.IRETURN:
244             case Const.LRETURN:
245             case Const.FRETURN:
246             case Const.DRETURN:
247             case Const.ARETURN:
248             case Const.RETURN:
249             case Const.ARRAYLENGTH:
250             case Const.ATHROW:
251             case Const.MONITORENTER:
252             case Const.MONITOREXIT:
253                 ++index;
254                 break;
255 
256                 // Two byte instructions.
257             case Const.BIPUSH:
258             case Const.LDC:
259             case Const.NEWARRAY:
260                 index += 2;
261                 break;
262 
263                 // Instructions that can be used with the WIDE prefix.
264             case Const.ILOAD:
265             case Const.LLOAD:
266             case Const.FLOAD:
267             case Const.DLOAD:
268             case Const.ALOAD:
269             case Const.ISTORE:
270             case Const.LSTORE:
271             case Const.FSTORE:
272             case Const.DSTORE:
273             case Const.ASTORE:
274             case Const.RET:
275                 if (wide) {
276                     // Skip opcode and two immediate bytes.
277                     index += 3;
278                     wide = false;
279                 } else {
280                     // Skip opcode and one immediate byte.
281                     index += 2;
282                 }
283                 break;
284 
285                 // IINC is a special case Const.for WIDE handling
286             case Const.IINC:
287                 if (wide) {
288                     // Skip opcode, two byte index, and two byte immediate
289                     // value.
290                     index += 5;
291                     wide = false;
292                 } else {
293                     // Skip opcode, one byte index, and one byte immedate value.
294                     index += 3;
295                 }
296                 break;
297 
298                 // Three byte instructions.
299             case Const.SIPUSH:
300             case Const.LDC_W:
301             case Const.LDC2_W:
302             case Const.IFEQ:
303             case Const.IFNE:
304             case Const.IFLT:
305             case Const.IFGE:
306             case Const.IFGT:
307             case Const.IFLE:
308             case Const.IF_ICMPEQ:
309             case Const.IF_ICMPNE:
310             case Const.IF_ICMPLT:
311             case Const.IF_ICMPGE:
312             case Const.IF_ICMPGT:
313             case Const.IF_ICMPLE:
314             case Const.IF_ACMPEQ:
315             case Const.IF_ACMPNE:
316             case Const.GOTO:
317             case Const.JSR:
318             case Const.GETSTATIC:
319             case Const.PUTSTATIC:
320             case Const.GETFIELD:
321             case Const.PUTFIELD:
322             case Const.INVOKEVIRTUAL:
323             case Const.INVOKESPECIAL:
324             case Const.INVOKESTATIC:
325             case Const.NEW:
326             case Const.ANEWARRAY:
327             case Const.CHECKCAST:
328             case Const.INSTANCEOF:
329             case Const.IFNULL:
330             case Const.IFNONNULL:
331                 index += 3;
332                 break;
333 
334                 // Four byte instructions.
335             case Const.MULTIANEWARRAY:
336                 index += 4;
337                 break;
338 
339                 // Five byte instructions.
340             case Const.INVOKEINTERFACE:
341             case Const.INVOKEDYNAMIC:
342             case Const.GOTO_W:
343             case Const.JSR_W:
344                 index += 5;
345                 break;
346 
347                 // TABLESWITCH - variable length.
348             case Const.TABLESWITCH: {
349                 // Skip padding.
350                 int offset = index + 1; // skip the opcode
351                 offset += PAD[offset & 3];
352                 assert (offset & 3) == 0;
353 
354                 // offset should now be posited at the default value
355 
356                 // Extract min and max values.
357                 int low = extractInt(instructionList, offset + 4);
358                 int high = extractInt(instructionList, offset + 8);
359                 int tableSize = (high - low) + 1;
360                 if (DEBUG) {
361                     System.out.println("tableswitch: low=" + low + ", high=" + high + ", tableSize=" + tableSize);
362                 }
363 
364                 // Skip to next instruction.
365                 index = offset + 12 + (tableSize * 4);
366             }
367             break;
368 
369             // LOOKUPSWITCH - variable length.
370             case Const.LOOKUPSWITCH: {
371                 // Skip padding.
372                 int offset = index + 1; // skip the opcode
373                 offset += PAD[offset & 3];
374                 assert (offset & 3) == 0;
375 
376                 // offset should now be posited at the default value
377 
378                 // Extract number of value/offset pairs.
379                 int numPairs = extractInt(instructionList, offset + 4);
380                 if (DEBUG) {
381                     System.out.println("lookupswitch: numPairs=" + numPairs);
382                 }
383 
384                 // Skip to next instruction.
385                 index = offset + 8 + (numPairs * 8);
386             }
387             break;
388 
389             // Wide prefix.
390             case Const.WIDE:
391                 wide = true;
392                 ++index;
393                 break;
394 
395             default:
396                 throw new IllegalArgumentException("Bad opcode " + opcode + " at offset " + index);
397             }
398 
399             if (index < 0) {
400                 throw new IllegalStateException("index=" + index + ", opcode=" + opcode);
401             }
402 
403         }
404     }

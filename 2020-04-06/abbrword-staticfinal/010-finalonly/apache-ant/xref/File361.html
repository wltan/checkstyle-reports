31  import static org.junit.Assert.assertEquals;
32  import static org.junit.Assert.assertTrue;
33  
34  /**
35   * JUnit Testcases for ConcatReader
36   */
37  public class ConcatFilterTest {
38  
39      private static final String lSep = StringUtils.LINE_SEP;
40  
41      private static final String FILE_PREPEND_WITH =
42            "this-should-be-the-first-line" + lSep
43          + "Line  1" + lSep
44          + "Line  2" + lSep
45          + "Line  3" + lSep
46          + "Line  4" + lSep
47      ;
48  
49      private static final String FILE_PREPEND =
50            "Line  1" + lSep
51          + "Line  2" + lSep
52          + "Line  3" + lSep
53          + "Line  4" + lSep
54          + "Line  5" + lSep
55      ;
56  
57      private static final String FILE_APPEND_WITH =
58            "Line 57" + lSep
59          + "Line 58" + lSep
60          + "Line 59" + lSep
61          + "Line 60" + lSep
62          + "this-should-be-the-last-line" + lSep
63      ;
64  
65      private static final String FILE_APPEND =
66            "Line 56" + lSep
67          + "Line 57" + lSep
68          + "Line 58" + lSep
69          + "Line 59" + lSep
70          + "Line 60" + lSep
71      ;
72  
73      @Rule
74      public BuildFileRule buildRule = new BuildFileRule();
75  
76      @Before
77      public void setUp() {
78          buildRule.configureProject("src/etc/testcases/filters/concat.xml");
79      }
80  
81      @Test
82      public void testFilterReaderNoArgs() throws IOException {
83          buildRule.executeTarget("testFilterReaderNoArgs");
84          File expected = new File(buildRule.getProject().getProperty("output"), "concatfilter.test");
85          File result = new File(buildRule.getProject().getProperty("output"),  "concat.FilterReaderNoArgs.test");
86          assertEquals("testFilterReaderNoArgs: Result not like expected", FileUtilities.getFileContents(expected),
87                  FileUtilities.getFileContents(result));
88      }
89  
90      @Test
91      public void testFilterReaderBefore() throws IOException {
92          doTest("testFilterReaderPrepend", FILE_PREPEND_WITH, FILE_APPEND);
93      }
94  
95      @Test
96      public void testFilterReaderAfter() throws IOException {
97          doTest("testFilterReaderAppend", FILE_PREPEND, FILE_APPEND_WITH);
98      }
99  
100     @Test
101     public void testFilterReaderBeforeAfter() throws IOException {
102         doTest("testFilterReaderPrependAppend", FILE_PREPEND_WITH, FILE_APPEND_WITH);
103     }
104 
105     @Test
106     public void testConcatFilter() throws IOException {
107         doTest("testConcatFilter", FILE_PREPEND, FILE_APPEND);
108     }
109 
110     @Test
111     public void testConcatFilterBefore() throws IOException {
112         doTest("testConcatFilterPrepend", FILE_PREPEND_WITH, FILE_APPEND);
113     }
114 
115     @Test
116     public void testConcatFilterAfter() throws IOException {
117         doTest("testConcatFilterAppend", FILE_PREPEND, FILE_APPEND_WITH);
118     }
119 
120     @Test
121     public void testConcatFilterBeforeAfter() throws IOException {
122         doTest("testConcatFilterPrependAppend", FILE_PREPEND_WITH, FILE_APPEND_WITH);
123     }
124 
125 
126     /**
127      * Executes a target and checks the beginning and the ending of a file.
128      * The filename depends on the target name: target name <i>testHelloWorld</i>
129      * will search for a file <i>result/concat.HelloWorld.test</i>.
130      * @param target The target to invoke
131      * @param expectedStart The string which should be at the beginning of the file
132      * @param expectedEnd The string which should be at the end of the file
133      */
134     protected void doTest(String target, String expectedStart, String expectedEnd) throws IOException {
135         buildRule.executeTarget(target);
136         String resultContent = FileUtilities.getFileContents(
137                 new File(buildRule.getProject().getProperty("output") + "/concat." + target.substring(4) + ".test"));
138         assertTrue("First 5 lines differs.", resultContent.startsWith(expectedStart));
139         assertTrue("Last 5 lines differs.", resultContent.endsWith(expectedEnd));
140     }

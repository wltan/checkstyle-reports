19  package org.apache.tools.ant.taskdefs;
20  
21  import static org.apache.tools.ant.AntAssert.assertContains;
22  import static org.junit.Assert.assertEquals;
23  import static org.junit.Assert.assertTrue;
24  import static org.junit.Assert.fail;
25  
26  import org.apache.tools.ant.BuildException;
27  import org.apache.tools.ant.BuildFileRule;
28  import org.apache.tools.ant.util.FileUtils;
29  import org.junit.Assume;
30  import org.junit.Before;
31  import org.junit.Rule;
32  import org.junit.Test;
33  
34  /**
35   */
36  public class PropertyTest {
37  	
38  	@Rule
39  	public BuildFileRule buildRule = new BuildFileRule();
40  
41      /** Utilities used for file operations */
42      private static final FileUtils FILE_UTILS = FileUtils.getFileUtils();
43  
44      @Before
45      public void setUp() {
46          buildRule.configureProject("src/etc/testcases/taskdefs/property.xml");
47          buildRule.executeTarget("setUp");
48      }
49  
50      @Test
51      public void test1() {
52          // should get no output at all
53      	buildRule.executeTarget("test1");
54      	assertEquals("System output should have been empty", "", buildRule.getOutput());
55      	assertEquals("System error should have been empty", "", buildRule.getError());
56      }
57  
58      @Test
59      public void test2() {
60      	buildRule.executeTarget("test2");
61          assertContains("testprop1=aa, testprop3=xxyy, testprop4=aazz", buildRule.getLog());
62      }
63  
64      @Test
65      public void test3() {
66          try {
67              buildRule.executeTarget("test3");
68              fail("Did not throw exception on circular exception");
69          }
70          catch (BuildException e) {
71              assertTrue("Circular definition not detected - ",
72                       e.getMessage().indexOf("was circularly defined") != -1);
73          }
74          
75      }
76  
77      @Test
78      public void test4() {
79      	buildRule.executeTarget("test4");
80      	assertContains("http.url is http://localhost:999", buildRule.getLog());
81      }
82  
83      @Test
84      public void test5() {
85          String baseDir = buildRule.getProject().getProperty("basedir");
86      	String uri = FILE_UTILS.toURI(baseDir + "/property3.properties");
87          buildRule.getProject().setNewProperty("test5.url", uri);
88          
89          buildRule.executeTarget("test5");
90          assertContains("http.url is http://localhost:999", buildRule.getLog());
91      }
92  
93      @Test
94      public void testPrefixSuccess() {
95          buildRule.executeTarget("prefix.success");
96          assertEquals("80", buildRule.getProject().getProperty("server1.http.port"));
97      }
98  
99      @Test
100     public void testPrefixFailure() {
101        try {
102             buildRule.executeTarget("prefix.fail");
103             fail("Did not throw exception on invalid use of prefix");
104         }
105         catch (BuildException e) {
106             assertContains("Prefix allowed on non-resource/file load - ", 
107                      "Prefix is only valid", e.getMessage());
108         }
109     }
110 
111     @Test
112     public void testCircularReference() {
113         try {
114             buildRule.executeTarget("testCircularReference");
115             fail("Did not throw exception on circular exception");
116         } catch (BuildException e) {
117             assertContains("Circular definition not detected - ",
118                          "was circularly defined", e.getMessage());
119         }
120     }
121 
122     @Test
123     public void testThisIsNotACircularReference() {
124     	buildRule.executeTarget("thisIsNotACircularReference");
125         assertContains("b is A/A/A", buildRule.getLog());
126     }
127     
128     @Test
129     public void testXmlProperty() {
130         try {
131             Class.forName("java.lang.Iterable");
132         } catch (ClassNotFoundException e) {
133         	Assume.assumeNoException("XML Loading only on Java 5+", e);
134         }
135         buildRule.executeTarget("testXmlProperty");
136         assertEquals("ONE", buildRule.getProject().getProperty("xml.one"));
137         assertEquals("TWO", buildRule.getProject().getProperty("xml.two"));
138         
139     }
140 
141 }

1   package bugPatterns;
2   
3   import edu.umd.cs.findbugs.annotations.ExpectWarning;
4   import edu.umd.cs.findbugs.annotations.NoWarning;
5   
6   public class AVERAGE_COMPUTED_USING_DIVISION {
7   
8       @ExpectWarning("IM_AVERAGE_COMPUTATION_COULD_OVERFLOW")
9       int bug(int a[], int anyLow, int anyHigh) {
10          int mid = (anyLow + anyHigh) / 2;
11          return a[mid];
12      }
13  
14      @ExpectWarning("IM_AVERAGE_COMPUTATION_COULD_OVERFLOW")
15      long bug(long a[], int anyLow, int anyHigh) {
16          int mid = (anyLow + anyHigh) / 2;
17          return a[mid];
18      }
19  
20      @ExpectWarning("IM_AVERAGE_COMPUTATION_COULD_OVERFLOW")
21      Object bug(Object a[], int anyLow, int anyHigh) {
22          int mid = (anyLow + anyHigh) / 2;
23          return a[mid];
24      }
25  
26      @NoWarning("IM_AVERAGE_COMPUTATION_COULD_OVERFLOW")
27      int notBug(int a[], int anyLow, int anyHigh) {
28          int mid = (anyLow + anyHigh) >>> 1;
29          return a[mid];
30      }
31  }

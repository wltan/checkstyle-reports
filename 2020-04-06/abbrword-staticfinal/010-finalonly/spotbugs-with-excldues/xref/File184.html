97      @Override
98      public String toString() {
99          StringBuilder buf = new StringBuilder();
100         buf.append(getNumClasses()).append(" classes: ");
101         for (PackageStats pStats : getPackageStats()) {
102             for (ClassStats cStats : pStats.getSortedClassStats()) {
103                 buf.append(cStats.getName()).append(" ");
104             }
105         }
106         return buf.toString();
107     }

207     /**
208      * Report that a class has been analyzed.
209      *
210      * @param className
211      *            the full name of the class
212      * @param sourceFile
213      *            TODO
214      * @param isInterface
215      *            true if the class is an interface
216      * @param size
217      *            a normalized class size value; see
218      *            detect/FindBugsSummaryStats.
219      * @param updatePackageStats TODO
220      */
221     public void addClass(@DottedClassName String className, @CheckForNull String sourceFile, boolean isInterface, int size, boolean updatePackageStats) {
222         if(!hasClassStats) {
223             // totalClasses/totalSize might be set from FindBugsSummary before when parsing XML: reset them
224             totalClasses = 0;
225             totalSize = 0;
226         }
227         hasClassStats = true;
228         String packageName;
229         int lastDot = className.lastIndexOf('.');
230         if (lastDot < 0) {
231             packageName = "";
232         } else {
233             packageName = className.substring(0, lastDot);
234         }
235         PackageStats stat = getPackageStats(packageName);
236         stat.addClass(className, sourceFile, isInterface, size, updatePackageStats);
237         totalClasses++;
238         totalSize += size;
239         totalClassesFromPackageStats = 0;
240         totalSizeFromPackageStats = 0;
241     }

243     /**
244      * Report that a class has been analyzed.
245      *
246      * @param className
247      *            the full name of the class
248      */
249     public @CheckForNull
250     ClassStats getClassStats(@DottedClassName String className) {
251         if (hasClassStats) {
252             return null;
253         }
254         String packageName;
255         int lastDot = className.lastIndexOf('.');
256         if (lastDot < 0) {
257             packageName = "";
258         } else {
259             packageName = className.substring(0, lastDot);
260         }
261         PackageStats stat = getPackageStats(packageName);
262         return stat.getClassStatsOrNull(className);
263     }

265     /**
266      * Called when a bug is reported.
267      */
268     public void addBug(BugInstance bug) {
269 
270         SourceLineAnnotation source = bug.getPrimarySourceLineAnnotation();
271         PackageStats stat = getPackageStats(source.getPackageName());
272         stat.addError(bug);
273         ++totalErrors[0];
274         int priority = bug.getPriority();
275         if (priority >= 1) {
276             ++totalErrors[Math.min(priority, totalErrors.length - 1)];
277         }
278     }

292     public void purgeClassesThatDontMatch(Pattern classPattern) {
293         if (hasClassStats) {
294             for (Iterator<Map.Entry<String, PackageStats>> i = packageStatsMap.entrySet().iterator(); i.hasNext();) {
295                 Map.Entry<String, PackageStats> e = i.next();
296                 PackageStats stats = e.getValue();
297                 stats.purgeClassesThatDontMatch(classPattern);
298                 if (stats.getClassStats().isEmpty()) {
299                     i.remove();
300                 }
301             }
302         } else if (hasPackageStats) {
303             boolean matchAny = false;
304             for (String packageName : packageStatsMap.keySet()) {
305                 Matcher m = classPattern.matcher(packageName);
306                 if (m.lookingAt()) {
307                     matchAny = true;
308                     break;
309                 }
310             }
311             if (matchAny) {
312                 for (Iterator<String> i = packageStatsMap.keySet().iterator(); i.hasNext();) {
313                     String packageName = i.next();
314                     Matcher m = classPattern.matcher(packageName);
315                     if (!m.lookingAt()) {
316                         i.remove();
317                     }
318 
319                 }
320             }
321         }
322     }

334             ps.getClassStats().clear();

338     public void purgePackageStats() {
339         hasPackageStats = false;
340         if (totalClassesFromPackageStats == 0) {
341             totalClassesFromPackageStats = totalClasses;

343         if (totalSizeFromPackageStats == 0) {
344             totalSizeFromPackageStats = totalSize;
345         }
346 
347         getPackageStats().clear();
348     }

350     public void recomputeFromComponents() {
351         if (!hasClassStats && !hasPackageStats) {
352             return;
353         }
354         for (int i = 0; i < totalErrors.length; i++) {
355             totalErrors[i] = 0;
356         }
357         totalSize = 0;
358         totalClasses = 0;
359         totalSizeFromPackageStats = 0;
360         totalClassesFromPackageStats = 0;
361 
362         for (PackageStats stats : packageStatsMap.values()) {
363             if (hasClassStats) {
364                 stats.recomputeFromClassStats();
365             }
366             totalSize += stats.size();
367             totalClasses += stats.getNumClasses();
368             for (int i = 0; i < totalErrors.length; i++) {
369                 totalErrors[i] += stats.getBugsAtPriority(i);
370             }
371         }
372     }

391     /**
392      * Output as XML.
393      */
394     public void writeXML(XMLOutput xmlOutput, boolean withMessages) throws IOException {
395         xmlOutput.startTag("FindBugsSummary");
396 
397         xmlOutput.addAttribute("timestamp", new SimpleDateFormat(TIMESTAMP_FORMAT, Locale.ENGLISH).format(analysisTimestamp));
398         xmlOutput.addAttribute("total_classes", String.valueOf(getNumClasses()));
399         xmlOutput.addAttribute("referenced_classes", String.valueOf(referencedClasses));
400 
401         xmlOutput.addAttribute("total_bugs", String.valueOf(totalErrors[0]));
402         xmlOutput.addAttribute("total_size", String.valueOf(getCodeSize()));
403         xmlOutput.addAttribute("num_packages", String.valueOf(packageStatsMap.size()));
404 
405         if (java_version != null) {
406             xmlOutput.addAttribute("java_version", java_version);
407         }
408         if (java_vm_version != null) {
409             xmlOutput.addAttribute("vm_version", java_vm_version);
410         }
411         Footprint delta = new Footprint(baseFootprint);
412         NumberFormat twoPlaces = NumberFormat.getInstance(Locale.ENGLISH);
413         twoPlaces.setMinimumFractionDigits(2);
414         twoPlaces.setMaximumFractionDigits(2);
415         twoPlaces.setGroupingUsed(false);
416         long cpuTime = delta.getCpuTime(); // nanoseconds
417         if (cpuTime >= 0) {
418             xmlOutput.addAttribute("cpu_seconds", twoPlaces.format(cpuTime / 1000000000.0));
419         }
420         long clockTime = delta.getClockTime(); // milliseconds
421         if (clockTime >= 0) {
422             xmlOutput.addAttribute("clock_seconds", twoPlaces.format(clockTime / 1000.0));
423         }
424         long peakMemory = delta.getPeakMemory(); // bytes
425         if (peakMemory >= 0) {
426             xmlOutput.addAttribute("peak_mbytes", twoPlaces.format(peakMemory / (1024.0 * 1024)));
427         }
428         xmlOutput.addAttribute("alloc_mbytes", twoPlaces.format(Runtime.getRuntime().maxMemory() / (1024.0 * 1024)));
429         long gcTime = delta.getCollectionTime(); // milliseconds
430         if (gcTime >= 0) {
431             xmlOutput.addAttribute("gc_seconds", twoPlaces.format(gcTime / 1000.0));
432         }
433 
434         BugCounts.writeBugPriorities(xmlOutput, totalErrors);
435 
436         xmlOutput.stopTag(false);
437 
438         if (withMessages && fileBugHashes != null) {
439             for (String sourceFile : new TreeSet<>(fileBugHashes.getSourceFiles())) {
440                 xmlOutput.startTag("FileStats");
441                 xmlOutput.addAttribute("path", sourceFile);
442                 xmlOutput.addAttribute("bugCount", String.valueOf(fileBugHashes.getBugCount(sourceFile)));
443                 xmlOutput.addAttribute("size", String.valueOf(fileBugHashes.getSize(sourceFile)));
444 
445                 String hash = fileBugHashes.getHash(sourceFile);
446                 if (hash != null) {
447                     xmlOutput.addAttribute("bugHash", hash);
448                 }
449                 xmlOutput.stopTag(true);
450 
451             }
452         }
453 
454         if (!OMIT_PACKAGE_STATS) {
455             for (PackageStats stats : packageStatsMap.values()) {
456                 stats.writeXML(xmlOutput);
457             }
458         }
459 
460         getProfiler().writeXML(xmlOutput);
461         xmlOutput.closeTag("FindBugsSummary");
462     }

464     public Map<String, String> getFileHashes(BugCollection bugs) {
465         if (bugs.getProjectStats() != this) {
466             throw new IllegalArgumentException("Collection doesn't own stats");
467         }
468 
469         if (fileBugHashes == null) {
470             computeFileStats(bugs);
471         }
472 
473         HashMap<String, String> result = new HashMap<>();
474         for (String sourceFile : fileBugHashes.getSourceFiles()) {
475             result.put(sourceFile, fileBugHashes.getHash(sourceFile));
476         }
477         return result;
478 
479     }

484     public void reportSummary(@WillClose OutputStream out) throws IOException {
485         XMLOutput xmlOutput = new OutputStreamXMLOutput(out);
486         try {
487             writeXML(xmlOutput);
488         } finally {
489             xmlOutput.finish();
490         }

493     /**
494      * Transform summary information to HTML.
495      *
496      * @param htmlWriter
497      *            the Writer to write the HTML output to
498      */
499     public void transformSummaryToHTML(Writer htmlWriter) throws IOException, TransformerException {
500 
501         ByteArrayOutputStream summaryOut = new ByteArrayOutputStream(8096);
502         reportSummary(summaryOut);
503 
504         StreamSource in = new StreamSource(new ByteArrayInputStream(summaryOut.toByteArray()));
505         StreamResult out = new StreamResult(htmlWriter);
506         InputStream xslInputStream = this.getClass().getClassLoader().getResourceAsStream("summary.xsl");
507         if (xslInputStream == null) {
508             throw new IOException("Could not load summary stylesheet");
509         }
510         StreamSource xsl = new StreamSource(xslInputStream);
511 
512         TransformerFactory tf = TransformerFactory.newInstance();
513         Transformer transformer = tf.newTransformer(xsl);
514         transformer.transform(in, out);
515 
516         Reader rdr = in.getReader();
517         if (rdr != null) {
518             rdr.close();
519         }
520         htmlWriter.close();
521         InputStream is = xsl.getInputStream();
522         if (is != null) {
523             is.close();
524         }
525     }

531     private PackageStats getPackageStats(String packageName) {
532         PackageStats stat = packageStatsMap.get(packageName);
533         if (stat == null) {
534             stat = new PackageStats(packageName);
535             packageStatsMap.put(packageName, stat);
536         }
537         return stat;
538     }

540     public void putPackageStats(String packageName, int numClasses, int size) {
541         hasPackageStats = true;
542         PackageStats stat = packageStatsMap.get(packageName);
543         if (stat == null) {
544             stat = new PackageStats(packageName, numClasses, size);
545             totalSizeFromPackageStats += size;
546             totalClassesFromPackageStats += numClasses;
547             packageStatsMap.put(packageName, stat);
548 
549         } else {
550             totalSizeFromPackageStats += size - stat.size();
551             totalClassesFromPackageStats += numClasses - stat.getNumClasses();
552 
553             stat.setNumClasses(numClasses);
554             stat.setSize(size);
555         }
556     }

558     /**
559      * @param stats2
560      */
561     public void addStats(ProjectStats stats2) {
562         if (totalSize == totalSizeFromPackageStats) {
563             totalSizeFromPackageStats += stats2.getCodeSize();
564         }
565         totalSize += stats2.getCodeSize();
566         if (totalClasses == totalClassesFromPackageStats) {
567             totalClassesFromPackageStats += stats2.getNumClasses();
568         }
569         totalClasses += stats2.getNumClasses();
570         for (int i = 0; i < totalErrors.length; i++) {
571             totalErrors[i] += stats2.totalErrors[i];
572         }
573 
574         if (stats2.hasPackageStats) {
575             hasPackageStats = true;
576         }
577         if (stats2.hasClassStats) {
578             hasClassStats = true;
579         }
580 
581         for (Map.Entry<String, PackageStats> entry : stats2.packageStatsMap.entrySet()) {
582             String key = entry.getKey();
583             PackageStats pkgStats2 = entry.getValue();
584             if (packageStatsMap.containsKey(key)) {
585                 PackageStats pkgStats = packageStatsMap.get(key);
586                 for (ClassStats classStats : pkgStats2.getClassStats()) {
587                     pkgStats.addClass(classStats, true);
588                 }
589             } else {
590                 packageStatsMap.put(key, pkgStats2);
591             }
592         }
593     }

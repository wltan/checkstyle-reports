1   package bugPatterns;
2   
3   import edu.umd.cs.findbugs.annotations.ExpectWarning;
4   import edu.umd.cs.findbugs.annotations.NoWarning;
5   
6   public class DMI_INVOKING_HASHCODE_ON_ARRAY {
7   
8       /*
9        * Entities: v1: int [] o1: invoke int [].hashCode v2: int Connections: o1
10       * is invoked on v1 v2 is the result of o1
11       */
12      @ExpectWarning("DMI_INVOKING_HASHCODE_ON_ARRAY")
13      void bug(int[] any) {
14          any.hashCode();
15      }
16  
17      /*
18       * Entities: v1: Object [] o1: invoke Object [].hashCode v2: int
19       * Connections: o1 is invoked on v1 v2 is the result of o1
20       */
21      @ExpectWarning("DMI_INVOKING_HASHCODE_ON_ARRAY")
22      void bug(Object[] any) {
23          any.hashCode();
24      }
25  
26      @ExpectWarning("DMI_INVOKING_HASHCODE_ON_ARRAY")
27      void bug(long[] any) {
28          any.hashCode();
29      }
30  
31      @NoWarning("DMI_INVOKING_HASHCODE_ON_ARRAY")
32      void notBug(Object any) {
33          any.hashCode();
34      }
35  
36  }

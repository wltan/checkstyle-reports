44      /**
45       * Encapsulates the algorithms used up to Ant 1.8 as ZipEncoding.
46       */
47      static final ZipEncoding FALLBACK_ENCODING = new ZipEncoding() {
48              public boolean canEncode(String name) { return true; }
49  
50              public ByteBuffer encode(String name) {
51                  final int length = name.length();
52                  byte[] buf = new byte[length];
53  
54                  // copy until end of input or output is reached.
55                  for (int i = 0; i < length; ++i) {
56                      buf[i] = (byte) name.charAt(i);
57                  }
58                  return ByteBuffer.wrap(buf);
59              }
60  
61              public String decode(byte[] buffer) {
62                  final int length = buffer.length;
63                  StringBuffer result = new StringBuffer(length);
64  
65                  for (int i = 0; i < length; ++i) {
66                      byte b = buffer[i];
67                      if (b == 0) { // Trailing null
68                          break;
69                      }
70                      result.append((char) (b & 0xFF)); // Allow for sign-extension
71                  }
72  
73                  return result.toString();
74              }
75          };

81      /**
82       * Parse an octal string from a buffer.
83       *
84       * <p>Leading spaces are ignored.
85       * The buffer must contain a trailing space or NUL,
86       * and may contain an additional trailing space or NUL.</p>
87       *
88       * <p>The input buffer is allowed to contain all NULs,
89       * in which case the method returns 0L
90       * (this allows for missing fields).</p>
91       *
92       * <p>To work-around some tar implementations that insert a
93       * leading NUL this method returns 0 if it detects a leading NUL
94       * since Ant 1.9.</p>
95       *
96       * @param buffer The buffer from which to parse.
97       * @param offset The offset into the buffer from which to parse.
98       * @param length The maximum number of bytes to parse - must be at least 2 bytes.
99       * @return The long value of the octal string.
100      * @throws IllegalArgumentException if the trailing space/NUL is missing or if a invalid byte is detected.
101      */
102     public static long parseOctal(final byte[] buffer, final int offset, final int length) {
103         long    result = 0;
104         int     end = offset + length;
105         int     start = offset;
106 
107         if (length < 2){
108             throw new IllegalArgumentException("Length "+length+" must be at least 2");
109         }
110 
111         if (buffer[start] == 0) {
112             return 0L;
113         }
114 
115         // Skip leading spaces
116         while (start < end){
117             if (buffer[start] == ' '){
118                 start++;
119             } else {
120                 break;
121             }
122         }
123 
124         // Trim all trailing NULs and spaces.
125         // The ustar and POSIX tar specs require a trailing NUL or
126         // space but some implementations use the extra digit for big
127         // sizes/uids/gids ...
128         byte trailer = buffer[end - 1];
129         while (start < end && (trailer == 0 || trailer == ' ')) {
130             end--;
131             trailer = buffer[end - 1];
132         }
133         if (start == end) {
134             throw new IllegalArgumentException(
135                     exceptionMessage(buffer, offset, length, start, trailer));
136         }
137 
138         for ( ;start < end; start++) {
139             final byte currentByte = buffer[start];
140             // CheckStyle:MagicNumber OFF
141             if (currentByte < '0' || currentByte > '7'){
142                 throw new IllegalArgumentException(
143                         exceptionMessage(buffer, offset, length, start, currentByte));
144             }
145             result = (result << 3) + (currentByte - '0'); // convert from ASCII
146             // CheckStyle:MagicNumber ON
147         }
148 
149         return result;
150     }

152     /** 
153      * Compute the value contained in a byte buffer.  If the most
154      * significant bit of the first byte in the buffer is set, this
155      * bit is ignored and the rest of the buffer is interpreted as a
156      * binary number.  Otherwise, the buffer is interpreted as an
157      * octal number as per the parseOctal function above.
158      *
159      * @param buffer The buffer from which to parse.
160      * @param offset The offset into the buffer from which to parse.
161      * @param length The maximum number of bytes to parse.
162      * @return The long value of the octal or binary string.
163      * @throws IllegalArgumentException if the trailing space/NUL is
164      * missing or an invalid byte is detected in an octal number, or
165      * if a binary number would exceed the size of a signed long
166      * 64-bit integer.
167      */
168     public static long parseOctalOrBinary(final byte[] buffer, final int offset,
169                                           final int length) {
170 
171         if ((buffer[offset] & 0x80) == 0) {
172             return parseOctal(buffer, offset, length);
173         }
174         final boolean negative = buffer[offset] == (byte) 0xff;
175         if (length < 9) {
176             return parseBinaryLong(buffer, offset, length, negative);
177         }
178         return parseBinaryBigInteger(buffer, offset, length, negative);
179     }
180 
181     private static long parseBinaryLong(final byte[] buffer, final int offset,
182                                         final int length,
183                                         final boolean negative) {
184         if (length >= 9) {
185             throw new IllegalArgumentException("At offset " + offset + ", "
186                                                + length + " byte binary number"
187                                                + " exceeds maximum signed long"
188                                                + " value");
189         }
190         long val = 0;
191         for (int i = 1; i < length; i++) {
192             val = (val << 8) + (buffer[offset + i] & 0xff);
193         }
194         if (negative) {
195             // 2's complement
196             val--;
197             val ^= ((long) Math.pow(2, (length - 1) * 8) - 1);
198         }
199         return negative ? -val : val;
200     }
201 
202     private static long parseBinaryBigInteger(final byte[] buffer,
203                                               final int offset,
204                                               final int length,
205                                               final boolean negative) {
206         byte[] remainder = new byte[length - 1];
207         System.arraycopy(buffer, offset + 1, remainder, 0, length - 1);
208         BigInteger val = new BigInteger(remainder);
209         if (negative) {
210             // 2's complement
211             val = val.add(BigInteger.valueOf(-1)).not();
212         }
213         if (val.bitLength() > 63) {
214             throw new IllegalArgumentException("At offset " + offset + ", "
215                                                + length + " byte binary number"
216                                                + " exceeds maximum signed long"
217                                                + " value");
218         }
219         return negative ? -val.longValue() : val.longValue();
220     }

222     /**
223      * Parse a boolean byte from a buffer.
224      * Leading spaces and NUL are ignored.
225      * The buffer may contain trailing spaces or NULs.
226      *
227      * @param buffer The buffer from which to parse.
228      * @param offset The offset into the buffer from which to parse.
229      * @return The boolean value of the bytes.
230      * @throws IllegalArgumentException if an invalid byte is detected.
231      */
232     public static boolean parseBoolean(final byte[] buffer, final int offset) {
233         return buffer[offset] == 1;
234     }

236     // Helper method to generate the exception message
237     private static String exceptionMessage(byte[] buffer, final int offset,
238             final int length, int current, final byte currentByte) {
239         String string = new String(buffer, offset, length); // TODO default charset?
240         string=string.replaceAll("\0", "{NUL}"); // Replace NULs to allow string to be printed
241         final String s = "Invalid byte "+currentByte+" at offset "+(current-offset)+" in '"+string+"' len="+length;
242         return s;
243     }

245     /**
246      * Parse an entry name from a buffer.
247      * Parsing stops when a NUL is found
248      * or the buffer length is reached.
249      *
250      * @param buffer The buffer from which to parse.
251      * @param offset The offset into the buffer from which to parse.
252      * @param length The maximum number of bytes to parse.
253      * @return The entry name.
254      */
255     public static String parseName(byte[] buffer, final int offset, final int length) {
256         try {
257             return parseName(buffer, offset, length, DEFAULT_ENCODING);
258         } catch (IOException ex) {
259             try {
260                 return parseName(buffer, offset, length, FALLBACK_ENCODING);
261             } catch (IOException ex2) {
262                 // impossible
263                 throw new RuntimeException(ex2);
264             }
265         }
266     }

268     /**
269      * Parse an entry name from a buffer.
270      * Parsing stops when a NUL is found
271      * or the buffer length is reached.
272      *
273      * @param buffer The buffer from which to parse.
274      * @param offset The offset into the buffer from which to parse.
275      * @param length The maximum number of bytes to parse.
276      * @param encoding name of the encoding to use for file names
277      * @return The entry name.
278      */
279     public static String parseName(byte[] buffer, final int offset,
280                                    final int length,
281                                    final ZipEncoding encoding)
282         throws IOException {
283 
284         int len = length;
285         for (; len > 0; len--) {
286             if (buffer[offset + len - 1] != 0) {
287                 break;
288             }
289         }
290         if (len > 0) {
291             byte[] b = new byte[len];
292             System.arraycopy(buffer, offset, b, 0, len);
293             return encoding.decode(b);
294         }
295         return "";
296     }

300      * Copies characters from the name into the buffer
301      * starting at the specified offset. 
302      * If the buffer is longer than the name, the buffer

327     /**
328      * Copy a name into a buffer.
329      * Copies characters from the name into the buffer
330      * starting at the specified offset. 
331      * If the buffer is longer than the name, the buffer
332      * is filled with trailing NULs.
333      * If the name is longer than the buffer,
334      * the output is truncated.
335      *
336      * @param name The header name from which to copy the characters.
337      * @param buf The buffer where the name is to be stored.
338      * @param offset The starting offset into the buffer
339      * @param length The maximum number of header bytes to copy.
340      * @param encoding name of the encoding to use for file names
341      * @return The updated offset, i.e. offset + length
342      */
343     public static int formatNameBytes(String name, byte[] buf, final int offset,
344                                       final int length,
345                                       final ZipEncoding encoding)
346         throws IOException {
347         int len = name.length();
348         ByteBuffer b = encoding.encode(name);
349         while (b.limit() > length && len > 0) {
350             b = encoding.encode(name.substring(0, --len));
351         }
352         final int limit = b.limit() - b.position();
353         System.arraycopy(b.array(), b.arrayOffset(), buf, offset, limit);
354 
355         // Pad any remaining output bytes with NUL
356         for (int i = limit; i < length; ++i) {
357             buf[offset + i] = 0;
358         }
359 
360         return offset + length;
361     }

363     /**
364      * Fill buffer with unsigned octal number, padded with leading zeroes.
365      * 
366      * @param value number to convert to octal - treated as unsigned
367      * @param buffer destination buffer
368      * @param offset starting offset in buffer
369      * @param length length of buffer to fill
370      * @throws IllegalArgumentException if the value will not fit in the buffer
371      */
372     public static void formatUnsignedOctalString(final long value, byte[] buffer,
373             final int offset, final int length) {
374         int remaining = length;
375         remaining--;
376         if (value == 0) {
377             buffer[offset + remaining--] = (byte) '0';
378         } else {
379             long val = value;
380             for (; remaining >= 0 && val != 0; --remaining) {
381                 // CheckStyle:MagicNumber OFF
382                 buffer[offset + remaining] = (byte) ((byte) '0' + (byte) (val & 7));
383                 val = val >>> 3;
384                 // CheckStyle:MagicNumber ON
385             }
386             if (val != 0){
387                 throw new IllegalArgumentException
388                 (value+"="+Long.toOctalString(value)+ " will not fit in octal number buffer of length "+length);
389             }
390         }
391 
392         for (; remaining >= 0; --remaining) { // leading zeros
393             buffer[offset + remaining] = (byte) '0';
394         }
395     }

397     /**
398      * Write an octal integer into a buffer.
399      *
400      * Uses {@link #formatUnsignedOctalString} to format
401      * the value as an octal string with leading zeros.
402      * The converted number is followed by space and NUL
403      * 
404      * @param value The value to write
405      * @param buf The buffer to receive the output
406      * @param offset The starting offset into the buffer
407      * @param length The size of the output buffer
408      * @return The updated offset, i.e offset+length
409      * @throws IllegalArgumentException if the value (and trailer) will not fit in the buffer
410      */
411     public static int formatOctalBytes(final long value, byte[] buf, final int offset, final int length) {
412 
413         int idx=length-2; // For space and trailing null
414         formatUnsignedOctalString(value, buf, offset, idx);
415 
416         buf[offset + idx++] = (byte) ' '; // Trailing space
417         buf[offset + idx]   = 0; // Trailing null
418 
419         return offset + length;
420     }

446     /**
447      * Write an long integer into a buffer as an octal string if this
448      * will fit, or as a binary number otherwise.
449      * 
450      * Uses {@link #formatUnsignedOctalString} to format
451      * the value as an octal string with leading zeros.
452      * The converted number is followed by a space.
453      * 
454      * @param value The value to write into the buffer.
455      * @param buf The destination buffer.
456      * @param offset The starting offset into the buffer.
457      * @param length The length of the buffer.
458      * @return The updated offset.
459      * @throws IllegalArgumentException if the value (and trailer)
460      * will not fit in the buffer.
461      */
462     public static int formatLongOctalOrBinaryBytes(
463         final long value, byte[] buf, final int offset, final int length) {
464 
465         // Check whether we are dealing with UID/GID or SIZE field
466         final long maxAsOctalChar = length == TarConstants.UIDLEN ? TarConstants.MAXID : TarConstants.MAXSIZE;
467 
468         final boolean negative = value < 0;
469         if (!negative && value <= maxAsOctalChar) { // OK to store as octal chars
470             return formatLongOctalBytes(value, buf, offset, length);
471         }
472 
473         if (length < 9) {
474             formatLongBinary(value, buf, offset, length, negative);
475         }
476         formatBigIntegerBinary(value, buf, offset, length, negative);
477 
478         buf[offset] = (byte) (negative ? 0xff : 0x80);
479         return offset + length;
480     }
481 
482     private static void formatLongBinary(final long value, byte[] buf,
483                                          final int offset, final int length,
484                                          final boolean negative) {
485         final int bits = (length - 1) * 8;
486         final long max = 1l << bits;
487         long val = Math.abs(value);
488         if (val >= max) {
489             throw new IllegalArgumentException("Value " + value +
490                 " is too large for " + length + " byte field.");
491         }
492         if (negative) {
493             val ^= max - 1;
494             val |= 0xff << bits;
495             val++;
496         }
497         for (int i = offset + length - 1; i >= offset; i--) {
498             buf[i] = (byte) val;
499             val >>= 8;
500         }
501     }
502 
503     private static void formatBigIntegerBinary(final long value, byte[] buf,
504                                                final int offset,
505                                                final int length,
506                                                final boolean negative) {
507         BigInteger val = BigInteger.valueOf(value);
508         final byte[] b = val.toByteArray();
509         final int len = b.length;
510         final int off = offset + length - len;
511         System.arraycopy(b, 0, buf, off, len);
512         final byte fill = (byte) (negative ? 0xff : 0);
513         for (int i = offset + 1; i < off; i++) {
514             buf[i] = fill;
515         }
516     }

518     /**
519      * Writes an octal value into a buffer.
520      * 
521      * Uses {@link #formatUnsignedOctalString} to format
522      * the value as an octal string with leading zeros.
523      * The converted number is followed by NUL and then space.
524      *
525      * @param value The value to convert
526      * @param buf The destination buffer
527      * @param offset The starting offset into the buffer.
528      * @param length The size of the buffer.
529      * @return The updated value of offset, i.e. offset+length
530      * @throws IllegalArgumentException if the value (and trailer) will not fit in the buffer
531      */
532     public static int formatCheckSumOctalBytes(final long value, byte[] buf, final int offset, final int length) {
533 
534         int idx=length-2; // for NUL and space
535         formatUnsignedOctalString(value, buf, offset, idx);
536 
537         buf[offset + idx++]   = 0; // Trailing null
538         buf[offset + idx]     = (byte) ' '; // Trailing space
539 
540         return offset + length;
541     }

543     /**
544      * Compute the checksum of a tar entry header.
545      *
546      * @param buf The tar entry's header buffer.
547      * @return The computed checksum.
548      */
549     public static long computeCheckSum(final byte[] buf) {
550         long sum = 0;
551 
552         for (int i = 0; i < buf.length; ++i) {
553             sum += BYTE_MASK & buf[i];
554         }
555 
556         return sum;
557     }

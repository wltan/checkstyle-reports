19  package org.apache.tools.ant.types.selectors;
20  
21  import java.io.File;
22  
23  import org.apache.tools.ant.types.Comparison;
24  import org.apache.tools.ant.types.EnumeratedAttribute;
25  import org.apache.tools.ant.types.Parameter;

27  /**
28   * Selector that filters files based on their size.
29   *
30   * @since 1.5
31   */
32  public class SizeSelector extends BaseExtendSelector {
33  
34      /** Constants for kilo, kibi etc */
35      private static final int  KILO = 1000;
36      private static final int  KIBI = 1024;
37      private static final int  KIBI_POS = 4;
38      private static final int  MEGA = 1000000;
39      private static final int  MEGA_POS = 9;
40      private static final int  MEBI = 1048576;
41      private static final int  MEBI_POS = 13;
42      private static final long GIGA = 1000000000L;
43      private static final int  GIGA_POS = 18;
44      private static final long GIBI = 1073741824L;
45      private static final int  GIBI_POS = 22;
46      private static final long TERA = 1000000000000L;
47      private static final int  TERA_POS = 27;
48      private static final long TEBI = 1099511627776L;
49      private static final int  TEBI_POS = 31;
50      private static final int  END_POS = 36;
51  
52      /** Used for parameterized custom selector */
53      public static final String SIZE_KEY = "value";
54      /** Used for parameterized custom selector */
55      public static final String UNITS_KEY = "units";
56      /** Used for parameterized custom selector */
57      public static final String WHEN_KEY = "when";
58  
59      private long size = -1;
60      private long multiplier = 1;
61      private long sizelimit = -1;
62      private Comparison when = Comparison.EQUAL;
63  
64      /**
65       * Creates a new <code>SizeSelector</code> instance.
66       *
67       */
68      public SizeSelector() {
69      }
70  
71      /**
72       * Returns a <code>String</code> object representing the specified
73       * SizeSelector. This is "{sizeselector value: " + <"compare",
74       * "less", "more", "equal"> + "}".
75       * @return a string describing this object
76       */
77      public String toString() {
78          StringBuilder buf = new StringBuilder("{sizeselector value: ");
79          buf.append(sizelimit);
80          buf.append("compare: ").append(when.getValue());
81          buf.append("}");
82          return buf.toString();
83      }
84  
85      /**
86       * A size selector needs to know what size to base its selecting on.
87       * This will be further modified by the multiplier to get an
88       * actual size limit.
89       *
90       * @param size the size to select against expressed in units.
91       */
92      public void setValue(long size) {
93          this.size = size;
94          if (multiplier != 0 && size > -1) {
95              sizelimit = size * multiplier;
96          }
97      }
98  
99      /**
100      * Sets the units to use for the comparison. This is a little
101      * complicated because common usage has created standards that
102      * play havoc with capitalization rules. Thus, some people will
103      * use "K" for indicating 1000's, when the SI standard calls for
104      * "k". Others have tried to introduce "K" as a multiple of 1024,
105      * but that falls down when you reach "M", since "m" is already
106      * defined as 0.001.
107      * <p>
108      * To get around this complexity, a number of standards bodies
109      * have proposed the 2^10 standard, and at least one has adopted
110      * it. But we are still left with a populace that isn't clear on
111      * how capitalization should work.
112      * <p>
113      * We therefore ignore capitalization as much as possible.
114      * Completely mixed case is not possible, but all upper and lower
115      * forms are accepted for all long and short forms. Since we have
116      * no need to work with the 0.001 case, this practice works here.
117      * <p>
118      * This function translates all the long and short forms that a
119      * unit prefix can occur in and translates them into a single
120      * multiplier.
121      *
122      * @param units The units to compare the size to, using an
123      *        EnumeratedAttribute.
124      */
125     public void setUnits(ByteUnits units) {
126         int i = units.getIndex();
127         multiplier = 0;
128         if (i > -1 && i < KIBI_POS) {
129             multiplier = KILO;
130         } else if (i < MEGA_POS) {
131             multiplier = KIBI;
132         } else if (i < MEBI_POS) {
133             multiplier = MEGA;
134         } else if (i < GIGA_POS) {
135             multiplier = MEBI;
136         } else if (i < GIBI_POS) {
137             multiplier = GIGA;
138         } else if (i < TERA_POS) {
139             multiplier = GIBI;
140         } else if (i < TEBI_POS) {
141             multiplier = TERA;
142         } else if (i < END_POS) {
143             multiplier = TEBI;
144         }
145         if (multiplier > 0 && size > -1) {
146             sizelimit = size * multiplier;
147         }
148     }
149 
150     /**
151      * This specifies when the file should be selected, whether it be
152      * when the file matches a particular size, when it is smaller,
153      * or whether it is larger.
154      *
155      * @param when The comparison to perform, an EnumeratedAttribute.
156      */
157     public void setWhen(SizeComparisons when) {
158         this.when = when;
159     }
160 
161     /**
162      * When using this as a custom selector, this method will be called.
163      * It translates each parameter into the appropriate setXXX() call.
164      *
165      * @param parameters the complete set of parameters for this selector.
166      */
167     public void setParameters(Parameter[] parameters) {
168         super.setParameters(parameters);
169         if (parameters != null) {
170             for (int i = 0; i < parameters.length; i++) {
171                 String paramname = parameters[i].getName();
172                 if (SIZE_KEY.equalsIgnoreCase(paramname)) {
173                     try {
174                         setValue(Long.parseLong(parameters[i].getValue()));
175                     } catch (NumberFormatException nfe) {
176                         setError("Invalid size setting "
177                                 + parameters[i].getValue());
178                     }
179                 } else if (UNITS_KEY.equalsIgnoreCase(paramname)) {
180                     ByteUnits units = new ByteUnits();
181                     units.setValue(parameters[i].getValue());
182                     setUnits(units);
183                 } else if (WHEN_KEY.equalsIgnoreCase(paramname)) {
184                     SizeComparisons scmp = new SizeComparisons();
185                     scmp.setValue(parameters[i].getValue());
186                     setWhen(scmp);
187                 } else {
188                     setError("Invalid parameter " + paramname);
189                 }
190             }
191         }
192     }
193 
194     /**
195      * <p>Checks to make sure all settings are kosher. In this case, it
196      * means that the size attribute has been set (to a positive value),
197      * that the multiplier has a valid setting, and that the size limit
198      * is valid. Since the latter is a calculated value, this can only
199      * fail due to a programming error.
200      * </p>
201      * <p>If a problem is detected, the setError() method is called.
202      * </p>
203      */
204     public void verifySettings() {
205         if (size < 0) {
206             setError("The value attribute is required, and must be positive");
207         } else if (multiplier < 1) {
208             setError("Invalid Units supplied, must be K,Ki,M,Mi,G,Gi,T,or Ti");
209         } else if (sizelimit < 0) {
210             setError("Internal error: Code is not setting sizelimit correctly");
211         }
212     }
213 
214     /**
215      * The heart of the matter. This is where the selector gets to decide
216      * on the inclusion of a file in a particular fileset.
217      *
218      * @param basedir A java.io.File object for the base directory.
219      * @param filename The name of the file to check.
220      * @param file A File object for this filename.
221      * @return whether the file should be selected or not.
222      */
223     public boolean isSelected(File basedir, String filename, File file) {
224 
225         // throw BuildException on error
226         validate();
227 
228         // Directory size never selected for
229         if (file.isDirectory()) {
230             return true;
231         }
232         long diff = file.length() - sizelimit;
233         return when.evaluate(diff == 0 ? 0 : (int) (diff / Math.abs(diff)));
234     }

237     /**
238      * Enumerated attribute with the values for units.
239      * <p>
240      * This treats the standard SI units as representing powers of ten,
241      * as they should. If you want the powers of 2 that approximate
242      * the SI units, use the first two characters followed by a
243      * <code>bi</code>. So 1024 (2^10) becomes <code>kibi</code>,
244      * 1048576 (2^20) becomes <code>mebi</code>, 1073741824 (2^30)
245      * becomes <code>gibi</code>, and so on. The symbols are also
246      * accepted, and these are the first letter capitalized followed
247      * by an <code>i</code>. <code>Ki</code>, <code>Mi</code>,
248      * <code>Gi</code>, and so on. Capitalization variations on these
249      * are also accepted.
250      * <p>
251      * This binary prefix system is approved by the IEC and appears on
252      * its way for approval by other agencies, but it is not an SI
253      * standard. It disambiguates things for us, though.
254      */
255     public static class ByteUnits extends EnumeratedAttribute {
256         /**
257          * @return the values as an array of strings
258          */
259         public String[] getValues() {
260             return new String[]{"K", "k", "kilo", "KILO",
261                                 "Ki", "KI", "ki", "kibi", "KIBI",
262                                 "M", "m", "mega", "MEGA",
263                                 "Mi", "MI", "mi", "mebi", "MEBI",
264                                 "G", "g", "giga", "GIGA",
265                                 "Gi", "GI", "gi", "gibi", "GIBI",
266                                 "T", "t", "tera", "TERA",
267            /* You wish! */      "Ti", "TI", "ti", "tebi", "TEBI"
268             };
269         }
270     }

272     /**
273      * Enumerated attribute with the values for size comparison.
274      */
275     public static class SizeComparisons extends Comparison {
276     }

85      private static char[] HEX_DIGIT = { '0', '1', '2', '3', '4', '5', '6', '7',

88      public static String toHex(int val) {
89          char[] ch8 = new char[8];
90          for (int i = 8; --i >= 0; val >>= 4) {
91              ch8[i] = HEX_DIGIT[val & 0xf];
92          }
93          return String.valueOf(ch8);
94      }

96      /**
97       * Create a new file based on hash code, including the directory structure
98       * if necessary. In case of collision, we will reset hashcode and try again.
99       * 
100      * @param dir
101      *            the directory where the new file will be created
102      * @param hash
103      *            the initial hash value
104      * @return the file created successfully
105      * @throws Exception
106      */
107     public static File createNewFile(File dir, int hash) throws Exception {
108         File f = null;
109         boolean success = false;
110         Exception lastException = null;
111         for (int i = 0; i < MAX_TIMES_CREATE_FILE && !success; i++) {
112             // In UNIX, mkdirs and createNewFile can throw IOException
113             // instead of nicely returning false
114             try {
115                 if (!dir.exists()) {
116                     success = dir.mkdirs();
117                     if (!success)
118                         throw new IOException("Directory creation failed: "
119                                 + dir.getCanonicalPath());
120                 }
121 
122                 // Try to construct a new file name everytime
123                 f = new File(dir, toHex(hash++));
124 
125                 success = f.createNewFile();
126 
127                 if (i > 0)
128                     log.info("file: " + dir.getCanonicalPath()
129                             + " created successfully after retries: " + i);
130 
131                 if (success)
132                     return f;
133                 else
134                     i--; // if not success, we should not increment i which is
135                          // for exceptional retry
136             } catch (Exception e) {
137                 if (lastException == null)
138                     log.warn("failed to create file: " + dir.getCanonicalPath()
139                             + " - retry: " + (i + 1) + " of "
140                             + MAX_TIMES_CREATE_FILE + ". Will retry again.", e);
141                 else
142                     log.warn("failed to create file: " + dir.getCanonicalPath()
143                             + " - got the same exception as above - retry: "
144                             + (i + 1) + " of " + MAX_TIMES_CREATE_FILE
145                             + ". Will retry again");
146                 lastException = e;
147 
148                 success = false;
149 
150                 // Maybe other thread is trying to do the same thing
151                 // Let's wait for INTERVAL_CREATE_FILE ms - this rarely happens
152                 try {
153                     Thread.sleep(INTERVAL_CREATE_FILE);
154                 } catch (InterruptedException e1) {
155                 }
156             }
157         }
158         // Run out of retries, throw original exception
159         throw lastException;
160     }

166     public static File resolve(File f) {
167         if (f.isAbsolute())

191     public static String formatSize(long size) {
192         return (size == -1) ? "UNKNOWN" : (size < GIGA) ? ((float) size / MEGA)
193                 + "MB" : ((float) size / GIGA) + "GB";
194     }

196     public static long parseSize(String s, long minSize) {
197         long u;
198         if (s.endsWith("GB"))
199             u = GIGA;
200         else if (s.endsWith("MB"))
201             u = MEGA;
202         else
203             throw new IllegalArgumentException(s);
204         try {
205             long size = (long) (Float
206                     .parseFloat(s.substring(0, s.length() - 2)) * u);
207             if (size >= minSize)
208                 return size;
209         } catch (IllegalArgumentException e) {
210         }
211         throw new IllegalArgumentException(s);
212     }

214     public static boolean equalsPixelData(File f1, File f2) throws IOException {
215         InputStream in1 = new BufferedInputStream(new FileInputStream(f1));
216         try {
217             InputStream in2 = new BufferedInputStream(new FileInputStream(f2));
218             try {
219                 Dataset attrs = DcmObjectFactory.getInstance().newDataset();
220                 DcmParserFactory pf = DcmParserFactory.getInstance();
221                 DcmParser p1 = pf.newDcmParser(in1);
222                 DcmParser p2 = pf.newDcmParser(in2);
223                 p1.setDcmHandler(attrs.getDcmHandler());
224                 p1.parseDcmFile(FileFormat.DICOM_FILE, Tags.PixelData);
225                 p2.parseDcmFile(FileFormat.DICOM_FILE, Tags.PixelData);
226                 int samples = attrs.getInt(Tags.SamplesPerPixel, 1);
227                 int frames = attrs.getInt(Tags.NumberOfFrames, 1);
228                 int rows = attrs.getInt(Tags.Rows, 1);
229                 int columns = attrs.getInt(Tags.Columns, 1);
230                 int bitsAlloc = attrs.getInt(Tags.BitsAllocated, 8);
231                 int bitsStored = attrs.getInt(Tags.BitsStored, bitsAlloc);
232                 int frameLength = rows * columns * samples * bitsAlloc / 8;
233                 int pixelDataLength = frameLength * frames;
234                 if (pixelDataLength > p1.getReadLength()
235                         || pixelDataLength > p2.getReadLength()) {
236                     return false;
237                 }
238                 byte[] b1 = new byte[BUFFER_SIZE];
239                 byte[] b2 = new byte[BUFFER_SIZE];
240                 int[] mask = { 0xff, 0xff };
241                 int len, len2;
242                 if (bitsAlloc == 16 && bitsStored < 16) {
243                     mask[p1.getDcmDecodeParam().byteOrder == ByteOrder.LITTLE_ENDIAN ? 1
244                             : 0] = 0xff >>> (16 - bitsStored);
245                 }
246                 int pos = 0;
247                 while (pos < pixelDataLength) {
248                     len = in1.read(b1, 0, Math.min(pixelDataLength - pos,
249                             BUFFER_SIZE));
250                     if (len < 0) // EOF
251                         return false;
252                     int off = 0;
253                     while (off < len) {
254                         off += len2 = in2.read(b2, off, len - off);
255                         if (len2 < 0) // EOF
256                             return false;
257                     }
258                     for (int i = 0; i < len; i++, pos++)
259                         if (((b1[i] - b2[i]) & mask[pos & 1]) != 0)
260                             return false;
261                 }
262                 return true;
263             } finally {
264                 in2.close();
265             }
266         } finally {
267             in1.close();
268         }
269     }

271     public static int maxDiffPixelData(File f1, File f2) throws IOException {
272         int maxDiff = 0;
273         InputStream in1 = new BufferedInputStream(new FileInputStream(f1));
274         try {
275             InputStream in2 = new BufferedInputStream(new FileInputStream(f2));
276             try {
277                 DcmObjectFactory df = DcmObjectFactory.getInstance();
278                 DcmParserFactory pf = DcmParserFactory.getInstance();
279                 Dataset attrs1 = df.newDataset();
280                 Dataset attrs2 = df.newDataset();
281                 DcmParser p1 = pf.newDcmParser(in1);
282                 DcmParser p2 = pf.newDcmParser(in2);
283                 p1.setDcmHandler(attrs1.getDcmHandler());
284                 p2.setDcmHandler(attrs2.getDcmHandler());
285                 p1.parseDcmFile(FileFormat.DICOM_FILE, Tags.PixelData);
286                 p2.parseDcmFile(FileFormat.DICOM_FILE, Tags.PixelData);
287                 int samples = attrs1.getInt(Tags.SamplesPerPixel, 1);
288                 int frames = attrs1.getInt(Tags.NumberOfFrames, 1);
289                 int rows = attrs1.getInt(Tags.Rows, 1);
290                 int columns = attrs1.getInt(Tags.Columns, 1);
291                 int bitsAlloc = attrs1.getInt(Tags.BitsAllocated, 8);
292                 int bitsStored = attrs1.getInt(Tags.BitsStored, bitsAlloc);
293                 int bitsStored2 = attrs2.getInt(Tags.BitsStored, bitsAlloc);
294                 int pixelRepresentation1 = attrs1.getInt(Tags.PixelRepresentation, 0);
295                 int pixelRepresentation2 = attrs2.getInt(Tags.PixelRepresentation, 0);
296                 int frameLength = rows * columns * samples * bitsAlloc / 8;
297                 int pixelDataLength = frameLength * frames;
298                 if (pixelDataLength > p1.getReadLength()
299                         || pixelDataLength > p2.getReadLength()) {
300                     return Integer.MAX_VALUE;
301                 }
302                 byte[] b1 = new byte[BUFFER_SIZE];
303                 byte[] b2 = new byte[BUFFER_SIZE];
304                 byte lsb1 = 0, lsb2 = 0;
305                 int w1, w2, len, len2;
306                 int bitmask1 = 0xffff >>> (bitsAlloc - bitsStored);
307                 int bitmask2 = 0xffff >>> (bitsAlloc - bitsStored2);
308                 int signed1 = pixelRepresentation1 != 0 ? (-1 & ~bitmask1)>>1 : 0;
309                 int signed2 = pixelRepresentation2 != 0 ? (-1 & ~bitmask2)>>1 : 0;
310                 int pos = 0;
311                 while (pos < pixelDataLength) {
312                     len = in1.read(b1, 0, Math.min(pixelDataLength - pos,
313                             BUFFER_SIZE));
314                     if (len < 0) // EOF
315                         return Integer.MAX_VALUE;
316                     int off = 0;
317                     while (off < len) {
318                         off += len2 = in2.read(b2, off, len - off);
319                         if (len2 < 0) // EOF
320                             return Integer.MAX_VALUE;
321                     }
322                     if (bitsAlloc == 8)
323                         for (int i = 0; i < len; i++, pos++)
324                             maxDiff = Math.max(maxDiff,
325                                     Math.abs((b1[i] & 0xff) - (b2[i] & 0xff)));
326                     else {
327                         for (int i = 0; i < len; i++, pos++)
328                             // TODO assumes LE Byte Order
329                             if ((pos & 1) == 0) {
330                                 lsb1 = b1[i];
331                                 lsb2 = b2[i];
332                             } else {
333                                 if (((w1 = ((b1[i] << 8) | (lsb1 & 0xff)) & bitmask1) & signed1) != 0)
334                                     w1 |= signed1;
335                                 if (((w2 = ((b2[i] << 8) | (lsb2 & 0xff)) & bitmask2) & signed2) != 0)
336                                     w2 |= signed2;
337                                 maxDiff = Math.max(maxDiff, Math.abs(w1 - w2));
338                             }
339                         
340                     }
341                 }
342                 return maxDiff;
343             } finally {
344                 in2.close();
345             }
346         } finally {
347             in1.close();
348         }
349     }

355     /**
356      * 
357      * @param file - the file to delete
358      * @param deleteEmptyParents - removing empty parent folders flag
359      * @param parentFolderToKeep - if deleteEmptyParents is true, this parameters shows the parent folder 
360      * where removing parents must stop
361      * @return True, if the file has been successfully removed, otherwise returns false
362      */
363     public static boolean delete(File file, boolean deleteEmptyParents, String parentFolderToKeep) {
364         log.info("M-DELETE file: " + file);
365         if (!file.exists()) {
366             log.warn("File: " + file + " was already deleted");
367             return true;
368         }
369         if (!file.delete()) {
370             log.warn("Failed to delete file: " + file);
371             return false;
372         }
373         if (deleteEmptyParents) {
374             File parentToKeep = null;
375             if (parentFolderToKeep != null)
376             	parentToKeep = toFile(parentFolderToKeep);
377             
378         	File parent = file.getParentFile();
379             while (!parent.equals(parentToKeep) && parent.delete()){
380                 log.info("M-DELETE directory: " + parent);
381                 parent = parent.getParentFile();
382             }
383         }
384         return true;
385     }

387     public static void verifyMD5(FileInfo info) throws Exception {
388     	if (info.md5 == null || info.basedir == null
389     			|| info.availability != Availability.ONLINE
390     			|| info.basedir.startsWith("ftp://"))
391     		return;
392     	File file = FileUtils.toFile(info.basedir, info.fileID);
393     	verifyMD5(file, info.md5);
394     }

396     public static void verifyMD5(File file, String originalMd5) throws Exception {
397         log.info("M-READ file:" + file);
398         MessageDigest md = MessageDigest.getInstance("MD5");
399 
400         BufferedInputStream in = new BufferedInputStream(new FileInputStream(file));
401         
402         DigestInputStream dis = new DigestInputStream(in, md);
403         try {
404             DcmParser parser = DcmParserFactory.getInstance().newDcmParser(dis);
405             parser.parseDcmFile(FileFormat.DICOM_FILE, Tags.PixelData);
406             if ((parser.getReadTag() & 0xFFFFFFFFL) >= Tags.PixelData) {
407                 if (parser.getReadLength() == -1) {
408                     while (parser.parseHeader() == Tags.Item) {
409                         readOut(parser.getInputStream(), parser.getReadLength());
410                     }
411                 }
412                 readOut(parser.getInputStream(), parser.getReadLength());
413                 parser.parseDataset(parser.getDcmDecodeParam(), -1);
414             }
415         } finally {
416             try {
417                 dis.close();
418             } catch (IOException ignore) {
419             }
420         }
421         byte[] md5 = md.digest();
422         if (!Arrays.equals(md5, MD5.toBytes(originalMd5))) {
423         	log.error("MD5 for " + file.getAbsolutePath() + " is different that expected.  Has the file been changed or corrupted?");
424             throw new IllegalStateException("MD5 mismatch");
425         }
426     }
427     
428     protected static void readOut(InputStream in, int len) throws IOException {
429         int toRead = len;
430         while (toRead-- > 0) {
431             if (in.read() < 0) {
432                 throw new EOFException();
433             }
434         }

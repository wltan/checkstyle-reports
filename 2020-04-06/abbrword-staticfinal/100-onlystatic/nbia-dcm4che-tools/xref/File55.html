142     public Dataset[] generateMPPS(Long studyPk, boolean ignoreReqAttrIfNoStudyAccNo) throws FinderException {
143         StudyLocal study = studyHome.findByPrimaryKey(studyPk);
144         if (study.getNumberOfReceivingSeries() != 0) {
145             LOG.info("Skip generation of MPPS for study "+study.getStudyIuid()+"! Study contains series with status RECEIVING!");
146             return new Dataset[]{};
147         }
148         String suid = study.getStudyIuid();
149         SeriesLocal series;
150         //Study may contain series with other Modality and/or SourceAET. We create MPPS for all series without ppsIuid to finish whole study in one step.
151         HashMap mppsMap = new HashMap(); 
152         for ( Iterator iter = study.getSeries().iterator() ; iter.hasNext() ; ) {
153             series = (SeriesLocal) iter.next();
154             if ( series.getPpsIuid() == null ) {
155                 addSeries(series, mppsMap, suid, study, ignoreReqAttrIfNoStudyAccNo);
156             }
157         }
158         Dataset[] result = new Dataset[mppsMap.size()];
159         Iterator it = mppsMap.values().iterator();
160         for (int i = 0; i < result.length; ++i) {
161             result[i] = updateSeries((List) it.next());
162         }
163         return result;
164     }

166     private Dataset updateSeries(List list) {
167         Dataset mpps = (Dataset) list.get(0);
168         // use series receive time as fall back for PPS Start/End Date/Time
169         Date ppsStartDT = mpps.getDateTime(Tags.PPSStartDate, Tags.PPSStartTime);
170         Date ppsEndDT = mpps.getDateTime(Tags.PPSEndDate, Tags.PPSEndTime);
171         boolean calcPpsStartDT = ppsStartDT == null;
172         boolean calcPpsEndDT = ppsEndDT == null;
173         if (calcPpsStartDT || calcPpsEndDT) {
174             SeriesLocal series = (SeriesLocal) list.get(1);
175             ppsStartDT = ppsEndDT = series.getCreatedTime();
176             for (int i = 2, n = list.size(); i < n; ++i) {
177                 series = (SeriesLocal) list.get(i);
178                 Date dt = series.getCreatedTime();
179                 if (calcPpsStartDT && ppsStartDT.compareTo(dt) > 0)
180                     ppsStartDT = dt;
181                 if (calcPpsEndDT && ppsEndDT.compareTo(dt) > 0)
182                     ppsEndDT = dt;
183             }
184             if (calcPpsStartDT) {
185                 mpps.putDA(Tags.PPSStartDate, ppsStartDT);
186                 mpps.putTM(Tags.PPSStartTime, ppsStartDT);
187             }
188             if (calcPpsEndDT) {
189                 mpps.putDA(Tags.PPSEndDate, ppsEndDT);
190                 mpps.putTM(Tags.PPSEndTime, ppsEndDT);
191             }
192         }
193         for (int i = 1, n = list.size(); i < n; ++i) {
194             SeriesLocal series = (SeriesLocal) list.get(i);
195             Dataset seriesAttrs = series.getAttributes(false);
196             seriesAttrs.putAll(mpps.subSet(SERIES_PPS_TAGS));
197             Dataset refPPS = seriesAttrs.putSQ(Tags.RefPPSSeq).addNewItem();
198             refPPS.putUI(Tags.RefSOPClassUID, mpps.getString(Tags.SOPClassUID));
199             refPPS.putUI(Tags.RefSOPInstanceUID, mpps.getString(Tags.SOPInstanceUID));            
200             series.setAttributes(seriesAttrs);
201         }
202         return mpps;
203     }

205     private void addSeries(SeriesLocal series, HashMap mppsMap, String suid, StudyLocal study, boolean ignoreReqAttrIfNoStudyAccNo) throws FinderException {
206         final String md = series.getModality() == null ? "OT" : series.getModality();
207         final String srcAet = series.getSourceAET();
208         final Dataset seriesAttrs = series.getAttributes(false);
209         final String key = md + srcAet;
210         List list = (List) mppsMap.get(key);
211         if (list == null) {
212             list = new ArrayList();
213             mppsMap.put(key, list);
214             Dataset mpps = DcmObjectFactory.getInstance().newDataset();
215             list.add(mpps);
216             final Dataset patAttrs = study.getPatient().getAttributes(false);
217             final Dataset studyAttrs = study.getAttributes(false);
218             mpps.putAll(patAttrs.subSet(PATIENT_TAGS));
219             mpps.putAll(studyAttrs.subSet(STUDY_TAGS));
220             checkCodeSeq(mpps, Tags.ProcedureCodeSeq);
221             DcmElement rqaSq = ignoreReqAttrIfNoStudyAccNo && studyAttrs.getString(Tags.AccessionNumber) == null ? 
222                     null : seriesAttrs.get(Tags.RequestAttributesSeq);
223             int rqaSqSize = rqaSq != null ? rqaSq.countItems() : 0;
224             DcmElement ssaSq = mpps.putSQ(Tags.ScheduledStepAttributesSeq);
225             if (rqaSqSize == 0) { // unscheduled case
226                 Dataset ssa = ssaSq.addNewItem();
227                 ssa.putSH(Tags.AccessionNumber);
228                 ssa.putSQ(Tags.RefStudySeq);
229                 ssa.putUI(Tags.StudyInstanceUID,
230                         studyAttrs.getString(Tags.StudyInstanceUID));
231                 ssa.putLO(Tags.RequestedProcedureDescription);
232                 ssa.putSH(Tags.SPSID);
233                 ssa.putLO(Tags.SPSDescription);
234                 ssa.putSQ(Tags.ScheduledProtocolCodeSeq);
235                 ssa.putSH(Tags.RequestedProcedureID);
236             } else {
237                 for (int i = 0, n = rqaSqSize; i < n; i++) {
238                     Dataset ssa = rqaSq.getItem(i);
239                     ssaSq.addItem(ssa);
240                     ssa.putSH(Tags.AccessionNumber,
241                             studyAttrs.getString(Tags.AccessionNumber));
242                     ssa.putUI(Tags.StudyInstanceUID,
243                             studyAttrs.getString(Tags.StudyInstanceUID));
244                     if (!ssa.contains(Tags.RefStudySeq)) {
245                         ssa.putSQ(Tags.RefStudySeq);
246                     }
247                     if (!ssa.contains(Tags.SPSID)) {
248                         ssa.putSH(Tags.SPSID);
249                     }
250                     if (!ssa.contains(Tags.SPSDescription)) {
251                         ssa.putLO(Tags.SPSDescription);
252                     }
253                     checkCodeSeq(ssa, Tags.ScheduledProtocolCodeSeq);
254                     if (!ssa.contains(Tags.RequestedProcedureID)) {
255                         ssa.putSH(Tags.RequestedProcedureID);
256                     }
257                 }
258             }
259             
260             mpps.putUI(Tags.SOPInstanceUID, UIDGenerator.getInstance().createUID());
261             mpps.putUI(Tags.SOPClassUID, UIDs.ModalityPerformedProcedureStep);
262             mpps.putAE(Tags.PerformedStationAET, srcAet);
263             mpps.putSH(Tags.PerformedStationName, seriesAttrs
264                     .getString(Tags.StationName));
265             mpps.putSH(Tags.PerformedLocation);
266             mpps.putCS(Tags.Modality, md);
267             mpps.putSH(Tags.PPSID, makePPSID(md, suid));
268             mpps.putLO(Tags.PerformedProcedureTypeDescription, studyAttrs
269                     .getString(Tags.StudyDescription));
270             mpps.putSQ(Tags.PerformedSeriesSeq);
271         }
272         Dataset mpps = (Dataset) list.get(0);
273         list.add(series);
274         // derive PPS Start/End Date/Time from Series Date/Time
275         Date ppsStartDT = mpps
276                 .getDateTime(Tags.PPSStartDate, Tags.PPSStartTime);
277         Date ppsEndDT = mpps.getDateTime(Tags.PPSEndDate, Tags.PPSEndTime);
278         Date seriesDT = seriesAttrs.getDateTime(Tags.SeriesDate,
279                 Tags.SeriesTime);
280         if (seriesDT != null) {
281             if (ppsStartDT == null || ppsStartDT.compareTo(seriesDT) > 0) {
282                 ppsStartDT = seriesDT;
283                 mpps.putDA(Tags.PPSStartDate, seriesDT);
284                 mpps.putTM(Tags.PPSStartTime, seriesDT);
285             }
286             if (ppsEndDT == null || ppsEndDT.compareTo(seriesDT) < 0) {
287                 ppsEndDT = seriesDT;
288                 mpps.putDA(Tags.PPSEndDate, seriesDT);
289                 mpps.putTM(Tags.PPSEndTime, seriesDT);
290             }
291         }
292         Dataset seriesItem = mpps.get(Tags.PerformedSeriesSeq).addNewItem();
293         seriesItem.putAll(seriesAttrs.subSet(SERIES_TAGS));
294         // TODO put references to non-images into separate
295         // Referenced Non- Image Composite SOP Instance Sequence 
296         DcmElement refImageSq = seriesItem.putSQ(Tags.RefImageSeq);
297         Collection c = instanceHome.findBySeriesPk(series.getPk());
298         for (Iterator it = c.iterator(); it.hasNext();) {
299             InstanceLocal inst = (InstanceLocal) it.next();
300             Dataset refSOP = refImageSq.addNewItem();
301             refSOP.putUI(Tags.RefSOPClassUID, inst.getSopCuid());
302             refSOP.putUI(Tags.RefSOPInstanceUID, inst.getSopIuid());
303         }
304     }

306     private void checkCodeSeq(Dataset ds, int tag) {
307         DcmElement codeSeq = ds.get(tag);
308         if (codeSeq == null) {
309             ds.putSQ(tag);
310         } else if (codeSeq.countItems() > 0) {
311             DcmElement newCodeSeq = ds.putSQ(tag);
312             Dataset item;
313             for (int i = 0, len = codeSeq.countItems() ; i < len ; i++) {
314                 item = codeSeq.getItem(i);
315                 if (item != null && item.containsValue(Tags.CodeValue) && item.containsValue(Tags.CodingSchemeDesignator)) {
316                     newCodeSeq.addItem(item);
317                     if (tag == Tags.ProcedureCodeSeq && len > 1) {
318                         LOG.warn("Study with multiple Procedure Codes - only include first (valid) code item in emulated MPPS");
319                         return;
320                     }
321                 } else {
322                     LOG.warn("Code sequence "+Tags.toString(tag)+":Invalid code item removed! code:");LOG.warn(item);
323                 }
324             }
325         }
326     }

328     private String makePPSID(String md, String suid) {
329         return md.substring(0, 2)
330                 + suid.substring(Math.max(0, suid.length() - 14));
331     }

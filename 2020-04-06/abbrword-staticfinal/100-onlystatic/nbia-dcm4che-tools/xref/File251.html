71  public class FolderPermissionsPropertyFactory extends FolderPermissionsFactory {
72      private static final String RAW_PROPERTIES = "RAW_PROPERTIES";
73      private static final String SUBJECT_CONTEXT_KEY = "javax.security.auth.Subject.container";
74  
75      private Map mapPermissions = new HashMap();
76  
77      private static Logger log = Logger.getLogger(FolderPermissionsPropertyFactory.class.getName());
78  
79      public void init() {
80          try {
81              if ( getInitParameter() == null) {
82                  log.warn("No initString set. Use default");
83                  setInitParameter( "conf/dcm4chee-web/folder.permissions" );
84              }
85              Properties props = new Properties();
86              File f = FileUtils.resolve(new File(getInitParameter()));
87              log.info("read permission properties from file: "+f);
88              props.load(new FileInputStream(f));
89              initMap(props);
90          } catch ( Exception x ) {
91              log.error("Cant initialize FolderPermissionsPropertyFactory!",x);
92          }
93      }
94  
95      /**
96       * @param props
97       * @throws IOException
98       */
99      private void initMap(Properties props) throws IOException {
100         mapPermissions.clear();
101         mapPermissions.put(RAW_PROPERTIES,props);
102         String[] apps = Dcm4cheeFormController.FOLDER_APPLICATIONS;
103         String perm, value, role, methods;
104         int pos;
105         Map map;
106         Set set;
107         for ( int i = 0, len = apps.length ; i < len ; i++ ) {
108             perm = props.getProperty(apps[i]);
109             if ( perm != null ) {
110                 StringTokenizer st = new StringTokenizer(perm, ";");
111                 while ( st.hasMoreTokens() ) {
112                     value = st.nextToken();
113                     pos = value.indexOf('(');
114                     role = value.substring(0,pos);
115                     map = (Map)mapPermissions.get(role);
116                     if ( map == null ) {
117                         map = new HashMap();
118                         mapPermissions.put(role,map);
119                     }
120                     set = (Set) map.get(apps[i]);
121                     if (set == null) {
122                         set = new HashSet();
123                         map.put(apps[i],set);
124                     }
125 
126                     methods = value.substring(++pos,value.length()-1);
127                     addMethods(set, apps[i], methods, props);
128                 }
129             }
130         }
131     }
132 
133     private void addMethods(Set set, String app, String methods, Properties props ) {
134         if ( methods != null ) {
135             StringTokenizer st = new StringTokenizer(methods,",");
136             String method;
137             while ( st.hasMoreTokens() ) {
138                 method = st.nextToken();
139                 if (method.charAt(0)=='*') {
140                     addMethods(set, app, props.getProperty(app+"."+method,"?"), props);
141                 } else {
142                     set.add(method);
143                 }
144             }
145         }		
146     }

148     public FolderPermissions getFolderPermissions(String userID) {
149         log.debug("getFolderPermission for user:"+userID);
150         FolderPermissions permissions = new FolderPermissions();
151         Collection roles = new ArrayList();
152         try {
153             Subject subject = (Subject) PolicyContext.getContext(SUBJECT_CONTEXT_KEY);
154             Set groups = subject.getPrincipals(Group.class);
155             for (Iterator iter = groups.iterator(); iter.hasNext();) {
156                 Group group = (Group) iter.next();
157                 if ("Roles".equals(group.getName())) {
158                     for (Enumeration members = group.members();
159                     members.hasMoreElements();) {
160                         Principal member = (Principal) members.nextElement();
161                         roles.add(member.getName());
162                     }
163                 }
164             }
165         } catch ( Exception x ) {
166             log.error("Cant get roles for user "+userID+"! use role default",x);
167             roles.add("default");
168         }
169         log.debug("Roles:"+roles);
170         Map map;
171         Set set;
172         Map.Entry entry;
173         for ( Iterator iter = roles.iterator() ; iter.hasNext() ; ) {
174             map = (Map) mapPermissions.get(iter.next());//permission for a role (key=app,value=list of methods)
175             log.debug("permission for role:"+map);
176             if ( map != null ) {
177                 for ( Iterator iter1 = map.entrySet().iterator() ; iter1.hasNext() ; ) {
178                     entry = (Map.Entry) iter1.next();
179                     set = (Set) entry.getValue();
180                     permissions.addPermissions((String)entry.getKey(), (String[])set.toArray(new String[set.size()]));
181                 }
182             }
183         }
184         if ( permissions.getNumberOfPrivilegedApps() == 0 ) {
185             log.warn("User "+userID+" has no Permissions for any Folder Application! Set to use folder only!");
186             permissions.addPermissions("folder",null);
187         }
188         Properties rawProps = (Properties) mapPermissions.get(RAW_PROPERTIES);
189         String aeList = rawProps.getProperty(FolderPermissions.AEFILTER+"."+userID);
190         if ( aeList != null ) {
191             permissions.addPermissions(FolderPermissions.AEFILTER, StringUtils.split(aeList, ',') );
192         }
193         applyStationAet(permissions,rawProps, userID);
194         return permissions;
195     }
196 
197     private void applyStationAet(FolderPermissions permissions, Properties rawProps, String userID) {
198         String stationAetGroups = rawProps.getProperty(FolderPermissions.STATION_AET_GROUP_LIST);
199         if ( stationAetGroups != null ) {
200             String[] groups = StringUtils.split(stationAetGroups, ',');
201             permissions.addPermissions(FolderPermissions.STATION_AET_GROUP_LIST, groups );
202             String group, groupAETs;
203             for ( int i = 0 ; i < groups.length ; i++) {
204                 group = FolderPermissions.STATION_AET_GROUP+"."+groups[i];
205                 groupAETs = rawProps.getProperty(group);
206                 if ( groupAETs != null ) {
207                     permissions.addPermissions(group, StringUtils.split(groupAETs, ',') );
208                 }
209             }
210         }
211         String stationAetFilter = rawProps.getProperty(FolderPermissions.STATION_AET_FILTER+"."+userID);
212         if ( stationAetFilter != null ) {
213             permissions.addPermissions(FolderPermissions.STATION_AET_FILTER, StringUtils.split(stationAetFilter, ',') );
214         }
215     }

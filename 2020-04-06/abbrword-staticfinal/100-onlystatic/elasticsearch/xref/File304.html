65      static enum AutoImportDangledState {
66          NO() {
67              @Override
68              public boolean shouldImport() {
69                  return false;
70              }
71          },
72          YES() {
73              @Override
74              public boolean shouldImport() {
75                  return true;
76              }
77          },
78          CLOSED() {
79              @Override
80              public boolean shouldImport() {
81                  return true;
82              }
83          };
84  
85          public abstract boolean shouldImport();
86  
87          public static AutoImportDangledState fromString(String value) {
88              if ("no".equalsIgnoreCase(value)) {
89                  return NO;
90              } else if ("yes".equalsIgnoreCase(value)) {
91                  return YES;
92              } else if ("closed".equalsIgnoreCase(value)) {
93                  return CLOSED;
94              } else {
95                  throw new ElasticsearchIllegalArgumentException("failed to parse [" + value + "], not a valid auto dangling import type");
96              }
97          }
98      }

123     @Inject
124     public LocalGatewayMetaState(Settings settings, ThreadPool threadPool, NodeEnvironment nodeEnv,
125                                  TransportNodesListGatewayMetaState nodesListGatewayMetaState, LocalAllocateDangledIndices allocateDangledIndices,
126                                  IndicesService indicesService, ClusterService clusterService) throws Exception {
127         super(settings);
128         this.nodeEnv = nodeEnv;
129         this.threadPool = threadPool;
130         this.format = XContentType.fromRestContentType(settings.get("format", "smile"));
131         this.allocateDangledIndices = allocateDangledIndices;
132         nodesListGatewayMetaState.init(this);
133 
134         if (this.format == XContentType.SMILE) {
135             Map<String, String> params = Maps.newHashMap();
136             params.put("binary", "true");
137             formatParams = new ToXContent.MapParams(params);
138             Map<String, String> gatewayModeParams = Maps.newHashMap();
139             gatewayModeParams.put("binary", "true");
140             gatewayModeParams.put(MetaData.CONTEXT_MODE_PARAM, MetaData.CONTEXT_MODE_GATEWAY);
141             gatewayModeFormatParams = new ToXContent.MapParams(gatewayModeParams);
142         } else {
143             formatParams = ToXContent.EMPTY_PARAMS;
144             Map<String, String> gatewayModeParams = Maps.newHashMap();
145             gatewayModeParams.put(MetaData.CONTEXT_MODE_PARAM, MetaData.CONTEXT_MODE_GATEWAY);
146             gatewayModeFormatParams = new ToXContent.MapParams(gatewayModeParams);
147         }
148 
149         this.autoImportDangled = AutoImportDangledState.fromString(settings.get("gateway.local.auto_import_dangled", AutoImportDangledState.YES.toString()));
150         this.danglingTimeout = settings.getAsTime("gateway.local.dangling_timeout", TimeValue.timeValueHours(2));
151         this.deleteTimeout = settings.getAsTime("gateway.local.delete_timeout", TimeValue.timeValueSeconds(30));
152 
153         logger.debug("using gateway.local.auto_import_dangled [{}], gateway.local.delete_timeout [{}], with gateway.local.dangling_timeout [{}]",
154                 this.autoImportDangled, this.deleteTimeout, this.danglingTimeout);
155         indexStateFormat = indexStateFormat(format, formatParams);
156         globalStateFormat = globalStateFormat(format, gatewayModeFormatParams);
157         if (DiscoveryNode.masterNode(settings) || DiscoveryNode.dataNode(settings)) {
158             nodeEnv.ensureAtomicMoveSupported();
159         }
160         if (DiscoveryNode.masterNode(settings)) {
161             try {
162                 pre019Upgrade();
163                 long start = System.currentTimeMillis();
164                 loadState();
165                 logger.debug("took {} to load state", TimeValue.timeValueMillis(System.currentTimeMillis() - start));
166             } catch (Exception e) {
167                 logger.error("failed to read local state, exiting...", e);
168                 throw e;
169             }
170         }
171         this.indicesService = indicesService;
172         this.clusterService = clusterService;
173     }

179     @Override
180     public void clusterChanged(ClusterChangedEvent event) {
181         final ClusterState state = event.state();
182         if (state.blocks().disableStatePersistence()) {
183             // reset the current metadata, we need to start fresh...
184             this.currentMetaData = null;
185             return;
186         }
187 
188         MetaData newMetaData = state.metaData();
189         // we don't check if metaData changed, since we might be called several times and we need to check dangling...
190 
191         boolean success = true;
192         // only applied to master node, writing the global and index level states
193         if (state.nodes().localNode().masterNode()) {
194             // check if the global state changed?
195             if (currentMetaData == null || !MetaData.isGlobalStateEquals(currentMetaData, newMetaData)) {
196                 try {
197                     writeGlobalState("changed", newMetaData);
198                 } catch (Throwable e) {
199                     success = false;
200                 }
201             }
202 
203             // check and write changes in indices
204             for (IndexMetaData indexMetaData : newMetaData) {
205                 String writeReason = null;
206                 IndexMetaData currentIndexMetaData;
207                 if (currentMetaData == null) {
208                     // a new event..., check from the state stored
209                     currentIndexMetaData = loadIndexState(indexMetaData.index());
210                 } else {
211                     currentIndexMetaData = currentMetaData.index(indexMetaData.index());
212                 }
213                 if (currentIndexMetaData == null) {
214                     writeReason = "freshly created";
215                 } else if (currentIndexMetaData.version() != indexMetaData.version()) {
216                     writeReason = "version changed from [" + currentIndexMetaData.version() + "] to [" + indexMetaData.version() + "]";
217                 }
218 
219                 // we update the writeReason only if we really need to write it
220                 if (writeReason == null) {
221                     continue;
222                 }
223 
224                 try {
225                     writeIndex(writeReason, indexMetaData, currentIndexMetaData);
226                 } catch (Throwable e) {
227                     success = false;
228                 }
229             }
230         }
231 
232         // handle dangling indices, we handle those for all nodes that have a node file (data or master)
233         if (nodeEnv.hasNodeFile()) {
234             if (danglingTimeout.millis() >= 0) {
235                 synchronized (danglingMutex) {
236                     for (String danglingIndex : danglingIndices.keySet()) {
237                         if (newMetaData.hasIndex(danglingIndex)) {
238                             logger.debug("[{}] no longer dangling (created), removing", danglingIndex);
239                             DanglingIndex removed = danglingIndices.remove(danglingIndex);
240                             FutureUtils.cancel(removed.future);
241                         }
242                     }
243                     // delete indices that are no longer part of the metadata
244                     try {
245                         for (String indexName : nodeEnv.findAllIndices()) {
246                             // if we have the index on the metadata, don't delete it
247                             if (newMetaData.hasIndex(indexName)) {
248                                 continue;
249                             }
250                             if (danglingIndices.containsKey(indexName)) {
251                                 // already dangling, continue
252                                 continue;
253                             }
254                             final IndexMetaData indexMetaData = loadIndexState(indexName);
255                             if (indexMetaData != null) {
256                                 if(autoImportDangled.shouldImport()){
257                                     logger.info("[{}] dangling index, exists on local file system, but not in cluster metadata, auto import to cluster state [{}]", indexName, autoImportDangled);
258                                     danglingIndices.put(indexName, new DanglingIndex(indexName, null));
259                                 } else if (danglingTimeout.millis() == 0) {
260                                     logger.info("[{}] dangling index, exists on local file system, but not in cluster metadata, timeout set to 0, deleting now", indexName);
261                                     indicesService.deleteIndexStore("dangling index with timeout set to 0", indexMetaData, state);
262                                 } else {
263                                     logger.info("[{}] dangling index, exists on local file system, but not in cluster metadata, scheduling to delete in [{}], auto import to cluster state [{}]", indexName, danglingTimeout, autoImportDangled);
264                                     danglingIndices.put(indexName,
265                                             new DanglingIndex(indexName,
266                                                     threadPool.schedule(danglingTimeout,
267                                                             ThreadPool.Names.SAME,
268                                                             new RemoveDanglingIndex(indexMetaData))));
269                                 }
270                             }
271                         }
272                     } catch (Throwable e) {
273                         logger.warn("failed to find dangling indices", e);
274                     }
275                 }
276             }
277             if (autoImportDangled.shouldImport() && !danglingIndices.isEmpty()) {
278                 final List<IndexMetaData> dangled = Lists.newArrayList();
279                 for (String indexName : danglingIndices.keySet()) {
280                     IndexMetaData indexMetaData = loadIndexState(indexName);
281                     if (indexMetaData == null) {
282                         logger.debug("failed to find state for dangling index [{}]", indexName);
283                         continue;
284                     }
285                     // we might have someone copying over an index, renaming the directory, handle that
286                     if (!indexMetaData.index().equals(indexName)) {
287                         logger.info("dangled index directory name is [{}], state name is [{}], renaming to directory name", indexName, indexMetaData.index());
288                         indexMetaData = IndexMetaData.builder(indexMetaData).index(indexName).build();
289                     }
290                     if (autoImportDangled == AutoImportDangledState.CLOSED) {
291                         indexMetaData = IndexMetaData.builder(indexMetaData).state(IndexMetaData.State.CLOSE).build();
292                     }
293                     if (indexMetaData != null) {
294                         dangled.add(indexMetaData);
295                     }
296                 }
297                 IndexMetaData[] dangledIndices = dangled.toArray(new IndexMetaData[dangled.size()]);
298                 try {
299                     allocateDangledIndices.allocateDangled(dangledIndices, new LocalAllocateDangledIndices.Listener() {
300                         @Override
301                         public void onResponse(LocalAllocateDangledIndices.AllocateDangledResponse response) {
302                             logger.trace("allocated dangled");
303                         }
304 
305                         @Override
306                         public void onFailure(Throwable e) {
307                             logger.info("failed to send allocated dangled", e);
308                         }
309                     });
310                 } catch (Throwable e) {
311                     logger.warn("failed to send allocate dangled", e);
312                 }
313             }
314         }
315 
316         if (success) {
317             currentMetaData = newMetaData;
318         }
319     }

356     private void writeIndex(String reason, IndexMetaData indexMetaData, @Nullable IndexMetaData previousIndexMetaData) throws Exception {
357         logger.trace("[{}] writing state, reason [{}]", indexMetaData.index(), reason);
358         try {
359             indexStateFormat.write(indexMetaData, indexMetaData.version(),
360                     nodeEnv.indexLocations(new Index(indexMetaData.index())));
361         } catch (Throwable ex) {
362             logger.warn("[{}]: failed to write index state", ex, indexMetaData.index());
363             throw new IOException("failed to write state for [" + indexMetaData.index() + "]", ex);
364         }
365     }
366 
367     private void writeGlobalState(String reason, MetaData metaData) throws Exception {
368         logger.trace("{} writing state, reason [{}]", GLOBAL_STATE_LOG_TYPE, reason);
369         try {
370             globalStateFormat.write(metaData, metaData.version(), nodeEnv.nodeDataLocations());
371         } catch (Throwable ex) {
372             logger.warn("{}: failed to write global state", ex, GLOBAL_STATE_LOG_TYPE);
373             throw new IOException("failed to write global state", ex);
374         }
375     }

377     private MetaData loadState() throws Exception {
378         MetaData globalMetaData = loadGlobalState();
379         MetaData.Builder metaDataBuilder;
380         if (globalMetaData != null) {
381             metaDataBuilder = MetaData.builder(globalMetaData);
382         } else {
383             metaDataBuilder = MetaData.builder();
384         }
385 
386         final Set<String> indices = nodeEnv.findAllIndices();
387         for (String index : indices) {
388             IndexMetaData indexMetaData = loadIndexState(index);
389             if (indexMetaData == null) {
390                 logger.debug("[{}] failed to find metadata for existing index location", index);
391             } else {
392                 metaDataBuilder.put(indexMetaData, false);
393             }
394         }
395         return metaDataBuilder.build();
396     }

408     private void pre019Upgrade() throws Exception {
409         long index = -1;
410         File metaDataFile = null;
411         MetaData metaData = null;
412         long version = -1;
413         for (File dataLocation : nodeEnv.nodeDataLocations()) {
414             File stateLocation = new File(dataLocation, "_state");
415             File[] stateFiles = stateLocation.listFiles();
416             if (stateFiles == null) {
417                 continue;
418             }
419             for (File stateFile : stateFiles) {
420                 if (logger.isTraceEnabled()) {
421                     logger.trace("[upgrade]: processing [" + stateFile.getName() + "]");
422                 }
423                 String name = stateFile.getName();
424                 if (!name.startsWith("metadata-")) {
425                     continue;
426                 }
427                 long fileIndex = Long.parseLong(name.substring(name.indexOf('-') + 1));
428                 if (fileIndex >= index) {
429                     // try and read the meta data
430                     try {
431                         byte[] data = Streams.copyToByteArray(new FileInputStream(stateFile));
432                         if (data.length == 0) {
433                             continue;
434                         }
435                         try (XContentParser parser = XContentHelper.createParser(data, 0, data.length)) {
436                             String currentFieldName = null;
437                             XContentParser.Token token = parser.nextToken();
438                             if (token != null) {
439                                 while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
440                                     if (token == XContentParser.Token.FIELD_NAME) {
441                                         currentFieldName = parser.currentName();
442                                     } else if (token == XContentParser.Token.START_OBJECT) {
443                                         if ("meta-data".equals(currentFieldName)) {
444                                             metaData = MetaData.Builder.fromXContent(parser);
445                                         }
446                                     } else if (token.isValue()) {
447                                         if ("version".equals(currentFieldName)) {
448                                             version = parser.longValue();
449                                         }
450                                     }
451                                 }
452                             }
453                         }
454                         index = fileIndex;
455                         metaDataFile = stateFile;
456                     } catch (IOException e) {
457                         logger.warn("failed to read pre 0.19 state from [" + name + "], ignoring...", e);
458                     }
459                 }
460             }
461         }
462         if (metaData == null) {
463             return;
464         }
465 
466         logger.info("found old metadata state, loading metadata from [{}] and converting to new metadata location and structure...", metaDataFile.getAbsolutePath());
467 
468         writeGlobalState("upgrade", MetaData.builder(metaData).version(version).build());
469         for (IndexMetaData indexMetaData : metaData) {
470             IndexMetaData.Builder indexMetaDataBuilder = IndexMetaData.builder(indexMetaData).version(version);
471             // set the created version to 0.18
472             indexMetaDataBuilder.settings(ImmutableSettings.settingsBuilder().put(indexMetaData.settings()).put(IndexMetaData.SETTING_VERSION_CREATED, Version.V_0_18_0));
473             writeIndex("upgrade", indexMetaDataBuilder.build(), null);
474         }
475 
476         // rename shards state to backup state
477         File backupFile = new File(metaDataFile.getParentFile(), "backup-" + metaDataFile.getName());
478         if (!metaDataFile.renameTo(backupFile)) {
479             throw new IOException("failed to rename old state to backup state [" + metaDataFile.getAbsolutePath() + "]");
480         }
481 
482         // delete all other shards state files
483         for (File dataLocation : nodeEnv.nodeDataLocations()) {
484             File stateLocation = new File(dataLocation, "_state");
485             File[] stateFiles = stateLocation.listFiles();
486             if (stateFiles == null) {
487                 continue;
488             }
489             for (File stateFile : stateFiles) {
490                 String name = stateFile.getName();
491                 if (!name.startsWith("metadata-")) {
492                     continue;
493                 }
494                 stateFile.delete();
495             }
496         }
497 
498         logger.info("conversion to new metadata location and format done, backup create at [{}]", backupFile.getAbsolutePath());
499     }

501     class RemoveDanglingIndex implements Runnable {
502 
503         private final IndexMetaData metaData;
504 
505         RemoveDanglingIndex(IndexMetaData metaData) {
506             this.metaData = metaData;
507         }
508 
509         @Override
510         public void run() {
511             synchronized (danglingMutex) {
512                 DanglingIndex remove = danglingIndices.remove(metaData.index());
513                 // no longer there...
514                 if (remove == null) {
515                     return;
516                 }
517                 logger.warn("[{}] deleting dangling index", metaData.index());
518                 try {
519                     indicesService.deleteIndexStore("deleting dangling index", metaData, clusterService.state());
520                 } catch (Exception ex) {
521                     logger.debug("failed to delete dangling index", ex);
522                 }
523             }
524         }
525     }

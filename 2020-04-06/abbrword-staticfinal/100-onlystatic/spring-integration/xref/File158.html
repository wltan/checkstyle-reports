336 	@Override
337 	public void synchronizeToLocalDirectory(final File localDirectory, final int maxFetchSize) {
338 		if (maxFetchSize == 0) {
339 			if (this.logger.isDebugEnabled()) {
340 				this.logger.debug("Max Fetch Size is zero - fetch to " + localDirectory.getAbsolutePath() + " ignored");
341 			}
342 			return;
343 		}
344 		if (this.logger.isTraceEnabled()) {
345 			this.logger.trace("Synchronizing " + this.evaluatedRemoteDirectory + " to " + localDirectory);
346 		}
347 		try {
348 			int transferred = this.remoteFileTemplate.execute(session ->
349 					transferFilesFromRemoteToLocal(localDirectory, maxFetchSize, session));
350 			if (this.logger.isDebugEnabled()) {
351 				this.logger.debug(transferred + " files transferred from '" + this.evaluatedRemoteDirectory + "'");
352 			}
353 		}
354 		catch (Exception e) {
355 			throw new MessagingException("Problem occurred while synchronizing '"
356 					+ this.evaluatedRemoteDirectory + "' to local directory", e);
357 		}
358 	}

360 	private Integer transferFilesFromRemoteToLocal(File localDirectory, int maxFetchSize, Session<F> session)
361 			throws IOException {
362 
363 		F[] files = session.list(this.evaluatedRemoteDirectory);
364 		if (!ObjectUtils.isEmpty(files)) {
365 			files = FileUtils.purgeUnwantedElements(files, e -> !isFile(e), this.comparator);
366 		}
367 		if (!ObjectUtils.isEmpty(files)) {
368 			boolean haveFilter = this.filter != null;
369 			boolean filteringOneByOne = haveFilter && this.filter.supportsSingleFileFiltering();
370 			List<F> filteredFiles = applyFilter(files, haveFilter, filteringOneByOne, maxFetchSize);
371 
372 			int copied = filteredFiles.size();
373 			int accepted = 0;
374 
375 			for (F file : filteredFiles) {
376 				if (filteringOneByOne) {
377 					if ((maxFetchSize < 0 || accepted < maxFetchSize) && this.filter
378 							.accept(file)) { // NOSONAR never null
379 						accepted++;
380 					}
381 					else {
382 						file = null;
383 						copied--;
384 					}
385 				}
386 				copied = copyIfNotNull(localDirectory, session, filteringOneByOne, filteredFiles, copied, file);
387 			}
388 			return copied;
389 		}
390 		else {
391 			return 0;
392 		}
393 	}

395 	private int copyIfNotNull(File localDirectory, Session<F> session, boolean filteringOneByOne, List<F> filteredFiles,
396 			int copied, @Nullable F file) throws IOException {
397 
398 		boolean renamedFailed = false;
399 		try {
400 			if (file != null && !copyFileToLocalDirectory(this.evaluatedRemoteDirectory, file,
401 					localDirectory, session)) {
402 				renamedFailed = true;
403 			}
404 		}
405 		catch (RuntimeException | IOException e1) {
406 			if (filteringOneByOne) {
407 				resetFilterIfNecessary(file);
408 			}
409 			else {
410 				rollbackFromFileToListEnd(filteredFiles, file);
411 			}
412 			throw e1;
413 		}
414 		return renamedFailed ? copied - 1 : copied;
415 	}

417 	private List<F> applyFilter(F[] files, boolean haveFilter, boolean filteringOneByOne, int maxFetchSize) {
418 		List<F> filteredFiles;
419 		if (!filteringOneByOne && haveFilter) {
420 			filteredFiles = filterFiles(files);
421 		}
422 		else {
423 			filteredFiles = Arrays.asList(files);
424 		}
425 		if (maxFetchSize >= 0 && filteredFiles.size() > maxFetchSize && !filteringOneByOne) {
426 			if (haveFilter) {
427 				rollbackFromFileToListEnd(filteredFiles, filteredFiles.get(maxFetchSize));
428 			}
429 			filteredFiles = filteredFiles.stream()
430 					.limit(maxFetchSize)
431 					.collect(Collectors.toList());
432 		}
433 		return filteredFiles;
434 	}

436 	protected void rollbackFromFileToListEnd(List<F> filteredFiles, F file) {
437 		if (this.filter instanceof ReversibleFileListFilter) {
438 			((ReversibleFileListFilter<F>) this.filter)
439 					.rollback(file, filteredFiles);
440 		}
441 	}

443 	protected boolean copyFileToLocalDirectory(String remoteDirectoryPath, F remoteFile, // NOSONAR
444 			File localDirectory, Session<F> session) throws IOException {
445 
446 		String remoteFileName = getFilename(remoteFile);
447 		String localFileName = generateLocalFileName(remoteFileName);
448 		String remoteFilePath = remoteDirectoryPath != null
449 				? (remoteDirectoryPath + this.remoteFileSeparator + remoteFileName)
450 				: remoteFileName;
451 
452 		if (!isFile(remoteFile)) {
453 			if (this.logger.isDebugEnabled()) {
454 				this.logger.debug("cannot copy, not a file: " + remoteFilePath);
455 			}
456 			return false;
457 		}
458 
459 		long modified = getModified(remoteFile);
460 
461 		File localFile = new File(localDirectory, localFileName);
462 		boolean exists = localFile.exists();
463 		if (!exists || (this.preserveTimestamp && modified != localFile.lastModified())) {
464 			if (!exists &&
465 					localFileName.replaceAll("/", Matcher.quoteReplacement(File.separator)).contains(File.separator)) {
466 				localFile.getParentFile().mkdirs(); //NOSONAR - will fail on the writing below
467 			}
468 
469 			boolean transfer = true;
470 
471 			if (exists && !localFile.delete()) {
472 				transfer = false;
473 				if (this.logger.isInfoEnabled()) {
474 					this.logger.info("Cannot delete local file '" + localFile +
475 							"' in order to transfer modified remote file '" + remoteFile + "'. " +
476 							"The local file may be busy in some other process.");
477 				}
478 			}
479 
480 			boolean renamed = false;
481 
482 			if (transfer) {
483 				renamed = copyRemoteContentToLocalFile(session, remoteFilePath, localFile);
484 			}
485 
486 			if (renamed) {
487 				if (this.deleteRemoteFiles) {
488 					session.remove(remoteFilePath);
489 					if (this.logger.isDebugEnabled()) {
490 						this.logger.debug("deleted remote file: " + remoteFilePath);
491 					}
492 				}
493 				if (this.preserveTimestamp && !localFile.setLastModified(modified)) {
494 					throw new IllegalStateException("Could not sent last modified on file: " + localFile);
495 				}
496 				String[] hostPort = session.getHostPort().split(":");
497 				try {
498 					String remoteFileMetadata =
499 							new URI(protocol(), null, hostPort[0], Integer.parseInt(hostPort[1]),
500 									'/' + remoteDirectoryPath, null, remoteFileName)
501 									.toString();
502 					this.remoteFileMetadataStore.put(buildMetadataKey(localFile), remoteFileMetadata);
503 				}
504 				catch (URISyntaxException ex) {
505 					throw new IllegalStateException("Cannot create a remote file metadata", ex);
506 				}
507 				return true;
508 			}
509 			else {
510 				resetFilterIfNecessary(remoteFile);
511 			}
512 		}
513 		else if (this.logger.isWarnEnabled()) {
514 			this.logger.warn("The remote file '" + remoteFile + "' has not been transferred " +
515 					"to the existing local file '" + localFile + "'. Consider removing the local file.");
516 		}
517 
518 		return false;
519 	}
520 
521 	private void resetFilterIfNecessary(F remoteFile) {
522 		if (this.filter instanceof ResettableFileListFilter) {
523 			if (this.logger.isInfoEnabled()) {
524 				this.logger.info("Removing the remote file '" + remoteFile +
525 						"' from the filter for a subsequent transfer attempt");
526 			}
527 			((ResettableFileListFilter<F>) this.filter).remove(remoteFile);
528 		}
529 	}

531 	private boolean copyRemoteContentToLocalFile(Session<F> session, String remoteFilePath, File localFile) {
532 		boolean renamed;
533 		String tempFileName = localFile.getAbsolutePath() + this.temporaryFileSuffix;
534 		File tempFile = new File(tempFileName);
535 
536 		try (OutputStream outputStream = new BufferedOutputStream(new FileOutputStream(tempFile))) {
537 			session.read(remoteFilePath, outputStream);
538 		}
539 		catch (RuntimeException e) { // NOSONAR catch and throw
540 			throw e;
541 		}
542 		catch (Exception e) {
543 			throw new MessagingException("Failure occurred while copying '" + remoteFilePath
544 					+ "' from the remote to the local directory", e);
545 		}
546 
547 		renamed = tempFile.renameTo(localFile);
548 
549 		if (!renamed) {
550 			if (localFile.delete()) {
551 				renamed = tempFile.renameTo(localFile);
552 				if (!renamed && this.logger.isInfoEnabled()) {
553 					this.logger.info("Cannot rename '"
554 							+ tempFileName
555 							+ "' to local file '" + localFile + "' after deleting. " +
556 							"The local file may be busy in some other process.");
557 				}
558 			}
559 			else if (this.logger.isInfoEnabled()) {
560 				this.logger.info("Cannot delete local file '" + localFile +
561 						"'. The local file may be busy in some other process.");
562 			}
563 		}
564 		return renamed;
565 	}

567 	private String generateLocalFileName(String remoteFileName) {
568 		if (this.localFilenameGeneratorExpression != null) {
569 			return this.localFilenameGeneratorExpression.getValue(this.evaluationContext, remoteFileName,
570 					String.class);
571 		}
572 		return remoteFileName;
573 	}

575 	protected void evaluateRemoteDirectory() {
576 		if (this.evaluationContext != null) {
577 			this.evaluatedRemoteDirectory = this.remoteDirectoryExpression.getValue(this.evaluationContext,
578 					String.class);
579 			this.evaluationContext.setVariable("remoteDirectory", this.evaluatedRemoteDirectory);
580 		}
581 	}

139 	/**
140 	 * Adds a constructor-arg to the provided bean definition builder
141 	 * with the value of the attribute whose name is provided if that
142 	 * attribute is defined in the given element.
143 	 *
144 	 * @param builder the bean definition builder to be configured
145 	 * @param element the XML element where the attribute should be defined
146 	 * @param attributeName the name of the attribute whose value will be
147 	 */
148 	public static void addConstructorValueIfAttributeDefined(BeanDefinitionBuilder builder,
149 			Element element, String attributeName) {
150 		String attributeValue = element.getAttribute(attributeName);
151 		if (StringUtils.hasText(attributeValue)) {
152 			builder.addConstructorArgValue(attributeValue);
153 		}
154 	}
155 
156 	/**
157 	 * Adds destination configuration to constructor.
158 	 *
159 	 * @param element The element.
160 	 * @param builder The builder.
161 	 * @param parserContext The parser context.
162 	 */
163 	public static void addDestinationConfigToConstructor(Element element,
164 			BeanDefinitionBuilder builder, ParserContext parserContext) {
165 		String destinationExpression = element.getAttribute("destination-expression");
166 		if (StringUtils.hasText(destinationExpression)) {
167 			addDestinationExpressionToConstructor(destinationExpression, element, builder, parserContext);
168 		}
169 		else {
170 			addHostAndPortToConstructor(element, builder, parserContext);
171 		}
172 	}
173 
174 	private static void addDestinationExpressionToConstructor(String destinationExpression,
175 			Element element, BeanDefinitionBuilder builder, ParserContext parserContext) {
176 		String host = element.getAttribute(IpAdapterParserUtils.HOST);
177 		String port = element.getAttribute(IpAdapterParserUtils.PORT);
178 		if (StringUtils.hasText(host) || StringUtils.hasText(port)) {
179 			parserContext.getReaderContext()
180 					.error("The 'destination-expression' is mutually exclusive with 'host/port' pair.", element);
181 		}
182 		builder.addConstructorArgValue(destinationExpression);
183 	}
184 
185 	private static void addHostAndPortToConstructor(Element element,
186 			BeanDefinitionBuilder builder, ParserContext parserContext) {
187 		String host = element.getAttribute(IpAdapterParserUtils.HOST);
188 		if (!StringUtils.hasText(host)) {
189 			parserContext.getReaderContext().error(IpAdapterParserUtils.HOST
190 					+ " is required for IP outbound channel adapters", element);
191 		}
192 		builder.addConstructorArgValue(host);
193 		String port = IpAdapterParserUtils.getPort(element, parserContext);
194 		builder.addConstructorArgValue(port);
195 	}
196 
197 	/**
198 	 * Asserts that a port attribute is supplied.
199 	 * @param element The element.
200 	 * @param parserContext The parser context.
201 	 * @return The value of the attribute.
202 	 * @throws org.springframework.beans.factory.BeanCreationException if attribute is not
203 	 * provided.
204 	 */
205 	static String getPort(Element element, ParserContext parserContext) {
206 		String port = element.getAttribute(IpAdapterParserUtils.PORT);
207 		if (!StringUtils.hasText(port)) {
208 			parserContext.getReaderContext().error(IpAdapterParserUtils.PORT
209 					+ " is required for IP channel adapters", element);
210 		}
211 		return port;
212 	}
213 
214 	/**
215 	 * Gets the multicast attribute, if present; if not returns 'false'.
216 	 * @param element The element.
217 	 * @return The value of the attribute or false.
218 	 */
219 	static String getMulticast(Element element) {
220 		String multicast = element.getAttribute(IpAdapterParserUtils.UDP_MULTICAST);
221 		if (!StringUtils.hasText(multicast)) {
222 			multicast = "false";
223 		}
224 		return multicast;
225 	}

227 	/**
228 	 * Sets the common port attributes on the bean being built (timeout, receive buffer size,
229 	 * send buffer size).
230 	 * @param builder The builder.
231 	 * @param element The element.
232 	 */
233 	static void addCommonSocketOptions(BeanDefinitionBuilder builder, Element element) {
234 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, SO_TIMEOUT);
235 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, SO_RECEIVE_BUFFER_SIZE);
236 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, SO_SEND_BUFFER_SIZE);
237 		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, LOCAL_ADDRESS);
238 	}

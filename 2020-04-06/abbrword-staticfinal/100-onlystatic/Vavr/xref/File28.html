274     /**
275      * Represents arbitrary binary trees of a certain depth n with values of type int.
276      */
277     static class ArbitraryBinaryTree implements Arbitrary<BinaryTree<Integer>> {
278 
279         final int minValue;
280         final int maxValue;
281 
282         ArbitraryBinaryTree(int minValue, int maxValue) {
283             this.minValue = Math.min(minValue, maxValue);
284             this.maxValue = Math.max(minValue, maxValue);
285         }
286 
287         @Override
288         public Gen<BinaryTree<Integer>> apply(int n) {
289             return random -> Gen.choose(minValue, maxValue).flatMap(value -> {
290                         if (n == 0) {
291                             return Gen.of(BinaryTree.leaf(value));
292                         } else {
293                             return Gen.frequency(
294                                     Tuple.of(1, Gen.of(BinaryTree.leaf(value))),
295                                     Tuple.of(4, Gen.of(BinaryTree.branch(apply(n / 2).apply(random), value, apply(n / 2).apply(random))))
296                             );
297                         }
298                     }
299             ).apply(random);
300         }
301     }
302 
303     interface BinaryTree<T> {
304 
305         static <T> Branch<T> branch(BinaryTree<T> left, T value, BinaryTree<T> right) {
306             return new Branch<>(left, value, right);
307         }
308 
309         static <T> Branch<T> leaf(T value) {
310             return new Branch<>(empty(), value, empty());
311         }
312 
313         static <T> Empty<T> empty() {
314             return Empty.instance();
315         }
316 
317         class Branch<T> implements BinaryTree<T> {
318 
319             final BinaryTree<T> left;
320             final T value;
321             final BinaryTree<T> right;
322 
323             Branch(BinaryTree<T> left, T value, BinaryTree<T> right) {
324                 this.left = left;
325                 this.value = value;
326                 this.right = right;
327             }
328         }
329 
330         class Empty<T> implements BinaryTree<T> {
331 
332             private static final Empty<?> INSTANCE = new Empty<>();
333 
334             @SuppressWarnings("unchecked")
335             static <T> Empty<T> instance() {
336                 return (Empty<T>) INSTANCE;
337             }
338         }
339     }

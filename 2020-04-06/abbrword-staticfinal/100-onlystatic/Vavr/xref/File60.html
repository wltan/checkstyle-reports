10  import org.junit.Test;

12  import static org.assertj.core.api.Assertions.assertThat;
13  
14  public class Euler07Test {
15  
16      /**
17       * <strong>Problem 7: 10001st prime</strong>
18       * <p>
19       * By listing the first six prime numbers: 2, 3, 5, 7, 11, and 13, we can see that the 6th prime is 13.
20       * <p>
21       * What is the 10 001st prime number?
22       * <p>
23       * See also <a href="https://projecteuler.net/problem=7">projecteuler.net problem 7</a>.
24       */
25      @Test
26      public void shouldSolveProblem7() {
27          assertThat(prime(1)).isEqualTo(2);
28          assertThat(prime(2)).isEqualTo(3);
29          assertThat(prime(3)).isEqualTo(5);
30          assertThat(prime(4)).isEqualTo(7);
31          assertThat(prime(5)).isEqualTo(11);
32          assertThat(prime(6)).isEqualTo(13);
33          assertThat(prime(10_001)).isEqualTo(104_743);
34      }
35  
36      private static final Array<Integer> PRIMES = Array.ofAll(PrimeNumbers.primes().take(10_001));
37  
38      private static long prime(int index) {
39          return PRIMES.get(index - 1);
40      }

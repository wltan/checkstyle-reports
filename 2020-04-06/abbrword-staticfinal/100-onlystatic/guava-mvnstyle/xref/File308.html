15  package com.google.common.primitives;
16  
17  import static com.google.common.base.Strings.lenientFormat;
18  import static java.lang.Boolean.parseBoolean;
19  
20  import com.google.common.annotations.GwtCompatible;
21  
22  /** Methods factored out so that they can be emulated differently in GWT. */
23  @GwtCompatible(emulated = true)
24  final class Platform {
25    private static final String GWT_RPC_PROPERTY_NAME = "guava.gwt.emergency_reenable_rpc";
26  
27    static void checkGwtRpcEnabled() {
28      if (!parseBoolean(System.getProperty(GWT_RPC_PROPERTY_NAME, "true"))) {
29        throw new UnsupportedOperationException(
30            lenientFormat(
31                "We are removing GWT-RPC support for Guava types. You can temporarily reenable"
32                    + " support by setting the system property %s to true. For more about system"
33                    + " properties, see %s. For more about Guava's GWT-RPC support, see %s.",
34                GWT_RPC_PROPERTY_NAME,
35                "https://stackoverflow.com/q/5189914/28465",
36                "https://groups.google.com/d/msg/guava-announce/zHZTFg7YF3o/rQNnwdHeEwAJ"));
37      }
38    }
39  
40    private Platform() {}
41  }

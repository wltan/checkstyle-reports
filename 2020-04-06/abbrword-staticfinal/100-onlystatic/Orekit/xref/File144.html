150     @Test
151     public void testSerialization() throws OrekitException, IOException, ClassNotFoundException {
152         TIRFProvider provider = new TIRFProvider(FramesFactory.getEOPHistory(IERSConventions.IERS_2010, true));
153         
154         ByteArrayOutputStream bos = new ByteArrayOutputStream();
155         ObjectOutputStream    oos = new ObjectOutputStream(bos);
156         oos.writeObject(provider);
157 
158         Assert.assertTrue(bos.size() > 280000);
159         Assert.assertTrue(bos.size() < 285000);
160 
161         ByteArrayInputStream  bis = new ByteArrayInputStream(bos.toByteArray());
162         ObjectInputStream     ois = new ObjectInputStream(bis);
163         TIRFProvider deserialized  = (TIRFProvider) ois.readObject();
164         for (int i = 0; i < FastMath.min(100, provider.getEOPHistory().getEntries().size()); ++i) {
165             AbsoluteDate date = provider.getEOPHistory().getEntries().get(i).getDate();
166             Transform expectedIdentity = new Transform(date,
167                                                        provider.getTransform(date).getInverse(),
168                                                        deserialized.getTransform(date));
169             Assert.assertEquals(0.0, expectedIdentity.getTranslation().getNorm(), 1.0e-15);
170             Assert.assertEquals(0.0, expectedIdentity.getRotation().getAngle(),   1.0e-15);
171         }
172 
173     }

175     /**
176      * Checks that {@link TIRFProvider#getTransform(AbsoluteDate)} is thread safe.
177      */
178     @Test
179     public void testConcurrentGetTransform()
180         throws OrekitException, InterruptedException, ExecutionException {
181 
182         // subject under test
183         final TIRFProvider tirf = new TIRFProvider(FramesFactory.getEOPHistory(IERSConventions.IERS_2010, false));
184         // arbitrary date
185         final AbsoluteDate start = new AbsoluteDate("2009-09-19T23:59:45.000", TimeScalesFactory.getUTC());
186         // in seconds = 15min
187         final double timeStep = 300;
188         // the number of possible concurrent threads
189         final int nJobs = 24;
190         // the number of calculations per thread
191         final int nPerJob = 1000;
192         // tolerance of comparisons
193         final double absTol = Precision.EPSILON;
194         // the expected result
195         final List<Transform> expecteds = new ArrayList<Transform>();
196         for (int j = 0; j < nPerJob; j++) {
197             final AbsoluteDate date = start.shiftedBy(timeStep * j);
198             // action
199             final Transform expected = tirf.getTransform(date);
200             // verify
201             expecteds.add(expected);
202         }
203 
204         // build jobs for concurrent execution
205         final List<Callable<Boolean>> jobs = new ArrayList<Callable<Boolean>>();
206         for (int i = 0; i < nJobs; i++) {
207             jobs.add(new Callable<Boolean>() {
208                 public Boolean call() throws Exception {
209                     for (int j = 0; j < nPerJob; j++) {
210                         final AbsoluteDate date = start.shiftedBy(timeStep * j);
211                         // action
212                         final Transform actual = tirf.getTransform(date);
213                         // verify
214                         assertTransformEquals(expecteds.get(j), actual, absTol);
215                     }
216                     return true;
217                 }
218             });
219         }
220 
221         // run the jobs
222         runConcurrentlyAndCheck(jobs, nJobs);
223     }
224 
225     /**
226      * Runs several jobs at concurrently in the given number of threads.
227      *
228      * @param jobs
229      *            the jobs to execute
230      * @param threads
231      *            the size of the thread pool
232      */
233     private static void runConcurrentlyAndCheck(List<Callable<Boolean>> jobs, int threads)
234         throws InterruptedException, ExecutionException {
235 
236         // action
237         final List<Future<Boolean>> futures = Executors.newFixedThreadPool(threads).invokeAll(jobs);
238 
239         // verify - necessary to throw AssertionErrors from the Callable
240         for (Future<Boolean> future : futures) {
241             Assert.assertEquals(true, future.get());
242         }
243 
244     }

246     /**
247      * Check the two {@link Transform}s are the same to within an absolute tolerance.
248      */
249     private static void assertTransformEquals(Transform expected, Transform actual, double absTol) {
250         final AngularCoordinates expectedAngular = expected.getAngular();
251         final AngularCoordinates actualAngular = actual.getAngular();
252         final Rotation expectedRotation = expectedAngular.getRotation();
253         final Rotation actualRotation = actualAngular.getRotation();
254         final PVCoordinates expectedPV = expected.getCartesian();
255         final PVCoordinates actualPV = actual.getCartesian();
256 
257         // transform
258         Assert.assertEquals(Rotation.distance(actualRotation, expectedRotation), 0, absTol);
259         Assert.assertEquals(expectedAngular.getRotationRate(), actualAngular.getRotationRate());
260         Assert.assertEquals(expectedPV.getPosition(), actualPV.getPosition());
261         Assert.assertEquals(expectedPV.getVelocity(), actualPV.getVelocity());
262     }

269     private void checkPV(PVCoordinates reference, PVCoordinates result,
270                          double expectedPositionError, double expectedVelocityError) {
271 
272         Vector3D dP = result.getPosition().subtract(reference.getPosition());
273         Vector3D dV = result.getVelocity().subtract(reference.getVelocity());
274         Assert.assertEquals(expectedPositionError, dP.getNorm(), 0.01 * expectedPositionError);
275         Assert.assertEquals(expectedVelocityError, dV.getNorm(), 0.01 * expectedVelocityError);
276     }

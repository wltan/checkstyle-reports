17  package org.orekit.time;
18  
19  /** Geocentric Coordinate Time.
20   * <p>Coordinate time at the center of mass of the Earth.
21   * This time scale depends linearly from {@link TTScale Terrestrial Time}.</p>
22   * <p>This is intended to be accessed thanks to the {@link TimeScalesFactory} class,
23   * so there is no public constructor.</p>
24   * @author Luc Maisonobe
25   * @see AbsoluteDate
26   */
27  public class TCGScale implements TimeScale {
28  
29      /** Serializable UID. */
30      private static final long serialVersionUID = 20131209L;
31  
32      /** LG rate. */
33      private static double LG_RATE = 6.969290134e-10;
34  
35      /** Reference date for TCG.
36       * <p>The reference date is such that the four following instants are equal:</p>
37       * <ul>
38       *   <li>1977-01-01T00:00:32.184 TT</li>
39       *   <li>1977-01-01T00:00:32.184 TCG</li>
40       *   <li>1977-01-01T00:00:32.184 TCB</li>
41       *   <li>1977-01-01T00:00:00.000 TAI</li>
42       * </ul>
43       */
44      private static final AbsoluteDate REFERENCE_DATE =
45          new AbsoluteDate(1977, 01, 01, TimeScalesFactory.getTAI());
46  
47      /** Offset between TT and TAI scales. */
48      private static final double TT_OFFSET =
49          TimeScalesFactory.getTT().offsetFromTAI(REFERENCE_DATE);
50  
51      /** Package private constructor for the factory.
52       */
53      TCGScale() {
54      }
55  
56      /** {@inheritDoc} */
57      @Override
58      public double offsetFromTAI(final AbsoluteDate date) {
59          return TT_OFFSET + LG_RATE * date.durationFrom(REFERENCE_DATE);
60      }
61  
62      /** {@inheritDoc} */
63      public String getName() {
64          return "TCG";
65      }
66  
67      /** {@inheritDoc} */
68      public String toString() {
69          return getName();
70      }
71  
72  }

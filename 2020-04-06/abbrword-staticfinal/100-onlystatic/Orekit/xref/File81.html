17  package org.orekit.propagation.conversion;
18  
19  import org.orekit.errors.OrekitException;
20  import org.orekit.orbits.PositionAngle;
21  import org.orekit.propagation.Propagator;
22  import org.orekit.propagation.SpacecraftState;
23  
24  /** This class converts osculating orbital elements into mean elements.
25   *  <p>
26   *  As this process depends on the force models used to average the orbit,
27   *  a {@link Propagator} is given as input. The force models used will be
28   *  those contained into the propagator. This propagator <em>must</em>
29   *  support its initial state to be reset, and this initial state <em>must</em>
30   *  represent some mean value. This implies that this method will not work
31   *  with {@link org.orekit.propagation.analytical.tle.TLEPropagator TLE propagators}
32   *  because their initial state cannot be reset, and it won't work either with
33   *  {@link org.orekit.propagation.analytical.EcksteinHechlerPropagator Eckstein-Hechler
34   *  propagator} as their initial state is osculating and not mean. As of 6.0, this
35   *  works mainly for {@link org.orekit.propagation.semianalytical.dsst.DSSTPropagator
36   *  DSST propagator}.
37   *  </p>
38   *  @author rdicosta
39   *  @author Pascal Parraud
40   */
41  public class OsculatingToMeanElementsConverter {
42  
43      /** Integrator maximum evaluation. */
44      private static final int      MAX_EVALUATION = 1000;
45  
46      /** Initial orbit to convert. */
47      private final SpacecraftState state;
48  
49      /** Number of satellite revolutions in the averaging interval. */
50      private final int             satelliteRevolution;
51  
52      /** Propagator used to compute mean orbit. */
53      private final Propagator      propagator;
54  
55      /** Scaling factor used for orbital parameters normalization. */
56      private double positionScale;
57  
58      /** Constructor.
59       *  @param state initial orbit to convert
60       *  @param satelliteRevolution number of satellite revolutions in the averaging interval
61       *  @param propagator propagator used to compute mean orbit
62       *  @param positionScale scaling factor used for orbital parameters normalization
63       *  (typically set to the expected standard deviation of the position)
64       */
65      public OsculatingToMeanElementsConverter(final SpacecraftState state,
66                                               final int satelliteRevolution,
67                                               final Propagator propagator,
68                                               final double positionScale) {
69          this.state = state;
70          this.satelliteRevolution = satelliteRevolution;
71          this.propagator = propagator;
72          this.positionScale = positionScale;
73      }
74  
75      /** Convert an osculating orbit into a mean orbit, in DSST sense.
76       *  @return mean orbit state, in DSST sense
77       *  @throws OrekitException if state cannot be propagated throughout range
78       */
79      public final SpacecraftState convert() throws OrekitException {
80  
81          final double timeSpan = state.getKeplerianPeriod() * satelliteRevolution;
82          propagator.resetInitialState(state);
83          final FiniteDifferencePropagatorConverter converter =
84                  new FiniteDifferencePropagatorConverter(new KeplerianPropagatorBuilder(state.getOrbit(),
85                                                                                         PositionAngle.MEAN,
86                                                                                         positionScale),
87                                                          1.e-6, MAX_EVALUATION);
88          final Propagator prop = converter.convert(propagator, timeSpan, satelliteRevolution * 36);
89          return prop.getInitialState();
90      }
91  }

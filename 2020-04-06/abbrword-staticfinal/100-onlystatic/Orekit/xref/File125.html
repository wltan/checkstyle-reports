54      @Test
55      public void testParseOPM1()
56          throws OrekitException {
57          // simple test for OPM file, contains p/v entries and other mandatory
58          // data.
59          final String ex = "/ccsds/OPMExample.txt";
60  
61          final OPMParser parser = new OPMParser().withMu(398600e9).
62                  withConventions(IERSConventions.IERS_2010).
63                  withSimpleEOP(true);
64          
65          final InputStream inEntry = getClass().getResourceAsStream(ex);
66  
67          final OPMFile file = parser.parse(inEntry, "OPMExample.txt");
68          Assert.assertEquals(IERSConventions.IERS_2010, file.getConventions());
69          final SatelliteTimeCoordinate coord = file.getSatelliteCoordinatesOPM();
70  
71          // Check Header Block;
72          Assert.assertEquals(2.0, file.getFormatVersion(), 1.0e-10);
73          Assert.assertEquals(new AbsoluteDate(1998, 11, 06, 9, 23, 57,
74                                               TimeScalesFactory.getUTC()), file
75              .getCreationDate());
76          Assert.assertEquals("JAXA", file.getOriginator());
77          
78          // Check Metadata Block;
79          
80          Assert.assertEquals("GODZILLA 5", file.getMetaData().getObjectName());
81          Assert.assertEquals("1998-057A", file.getMetaData().getObjectID());
82          Assert.assertEquals(1998, file.getMetaData().getLaunchYear());
83          Assert.assertEquals(57, file.getMetaData().getLaunchNumber());
84          Assert.assertEquals("A", file.getMetaData().getLaunchPiece());
85          Assert.assertEquals("EARTH", file.getMetaData().getCenterName());
86          Assert.assertTrue(file.getMetaData().getHasCreatableBody());
87          Assert.assertEquals(CelestialBodyFactory.getEarth(), file.getMetaData().getCenterBody());
88          Assert.assertEquals(CCSDSFrame.ITRF97.toString(), file.getMetaData().getFrame().getName());
89          Assert.assertEquals(TimeSystem.TAI, file.getTimeSystem());
90          Assert.assertFalse(file.hasCovarianceMatrix());
91          
92          // Check State Vector data Block;
93          Assert.assertEquals(new AbsoluteDate(1998, 12, 18, 14, 28, 15.1172,
94                                               TimeScalesFactory.getTAI()), coord.getEpoch());
95          checkPVEntry(new PVCoordinates(new Vector3D(6503514.000, 1239647.000,
96                                                      -717490.000),
97                                         new Vector3D(-873.160, 8740.420,
98                                                      -4191.076)),
99                       coord.getCoordinate());
100 
101         try {
102             file.generateCartesianOrbit();
103             Assert.fail("an exception should have been thrown");
104         } catch(OrekitIllegalArgumentException oiae) {
105             Assert.assertEquals(OrekitMessages.NON_PSEUDO_INERTIAL_FRAME, oiae.getSpecifier());
106             Assert.assertEquals("ITRF97", oiae.getParts()[0]);
107         }
108         try {
109             file.generateKeplerianOrbit();
110             Assert.fail("an exception should have been thrown");
111         } catch(OrekitIllegalArgumentException oiae) {
112             Assert.assertEquals(OrekitMessages.NON_PSEUDO_INERTIAL_FRAME, oiae.getSpecifier());
113             Assert.assertEquals("ITRF97", oiae.getParts()[0]);
114         }
115         try {
116             file.generateSpacecraftState();
117             Assert.fail("an exception should have been thrown");
118         } catch(OrekitIllegalArgumentException oiae) {
119             Assert.assertEquals(OrekitMessages.NON_PSEUDO_INERTIAL_FRAME, oiae.getSpecifier());
120             Assert.assertEquals("ITRF97", oiae.getParts()[0]);
121         }

129         // Keplerian elements, Spacecraft parameters and 2 maneuvers.

131

141         }

143

146         ArrayList<String> headerComment = new ArrayList<String>();
147         headerComment.add("Generated by GSOC, R. Kiehling");
148         headerComment.add("Current intermediate orbit IO2 and maneuver planning data");

150         Assert.assertEquals(new AbsoluteDate(2000, 06, 03, 05, 33, 00,

154

156         
157         Assert.assertEquals("EUTELSAT W4", file.getMetaData().getObjectName());
158         Assert.assertEquals("2000-028A", file.getMetaData().getObjectID());
159         Assert.assertEquals("EARTH", file.getMetaData().getCenterName());
160         Assert.assertTrue(file.getMetaData().getHasCreatableBody());
161         Assert.assertEquals(CelestialBodyFactory.getEarth(), file.getMetaData().getCenterBody());
162         Assert.assertEquals(FramesFactory.getGCRF(), file.getMetaData().getFrame());
163         Assert.assertEquals(TimeSystem.GPS, file.getTimeSystem());
164         Assert.assertEquals(0, file.getMetaDataComment().size());
165 
166         // Check Data State Vector block
167         ArrayList<String> epochComment = new ArrayList<String>();
168         epochComment.add("State Vector");
169         Assert.assertEquals(epochComment, file.getEpochComment());
170         Assert.assertEquals(new AbsoluteDate(2006, 06, 03, 00, 00, 00,
171                                              TimeScalesFactory.getGPS()),
172                             coord.getEpoch());
173         checkPVEntry(new PVCoordinates(new Vector3D(6655994.2, -40218575.1, -82917.7),
174                                        new Vector3D(3115.48208, 470.42605, -1.01495)),
175                      coord.getCoordinate());
176 
177         // Check Data Keplerian Elements block
178         Assert.assertTrue(file.hasKeplerianElements());
179         ArrayList<String> keplerianElementsComment = new ArrayList<String>();
180         keplerianElementsComment.add("Keplerian elements");
181         Assert.assertEquals(keplerianElementsComment, file.getKeplerianElementsComment());
182         Assert.assertEquals(41399512.3, file.getA(), 1e-6);
183         Assert.assertEquals(0.020842611, file.getE(), 1e-10);
184         Assert.assertEquals(FastMath.toRadians(0.117746), file.getI(), 1e-10);
185         Assert.assertEquals(FastMath.toRadians(17.604721), file.getRaan(), 1e-10);
186         Assert.assertEquals(FastMath.toRadians(218.242943), file.getPa(), 1e-10);
187         Assert.assertEquals(PositionAngle.TRUE, file.getAnomalyType());
188         Assert.assertEquals(FastMath.toRadians(41.922339), file.getAnomaly(), 1e-10);
189         Assert.assertEquals(398600.4415 * 1e9, file.getMuParsed(), 1e-10);
190 
191         // Check Data Spacecraft block
192         ArrayList<String> spacecraftComment = new ArrayList<String>();
193         spacecraftComment.add("Spacecraft parameters");
194         Assert.assertEquals(spacecraftComment, file.getSpacecraftComment());        
195         Assert.assertEquals(1913.000, file.getMass(), 1e-10);
196         Assert.assertEquals(10.000, file.getSolarRadArea(), 1e-10);

199         Assert.assertEquals(2.300, file.getDragCoeff(), 1e-10);
200 
201         // Check Data Maneuvers block
202         Assert.assertTrue(file.getHasManeuver());
203         Assert.assertEquals(3, file.getNbManeuvers());
204         ArrayList<String> stateManeuverComment0 = new ArrayList<String>();
205         stateManeuverComment0.add("2 planned maneuvers");
206         stateManeuverComment0.add("First maneuver: AMF-3");
207         stateManeuverComment0.add("Non-impulsive, thrust direction fixed in inertial frame");
208         Assert.assertEquals(stateManeuverComment0, file.getManeuver(0).getComment());
209         Assert.assertEquals(new AbsoluteDate(2000, 06, 03, 9, 00, 34.1,
210                                              TimeScalesFactory.getGPS()),
211                             file.getManeuvers().get(0).getEpochIgnition());
212         Assert.assertEquals(132.6, file.getManeuver(0).getDuration(),1e-10);
213         Assert.assertEquals(-18.418, file.getManeuver(0).getDeltaMass(), 1e-10);
214         Assert.assertNull(file.getManeuver(0).getRefLofType());
215         Assert.assertEquals(FramesFactory.getEME2000(), file.getManeuver(0).getRefFrame());
216         Assert.assertEquals(0.0,
217                             new Vector3D(-23.25700, 16.83160, -8.93444).distance(file.getManeuver(0).getDV()),
218                             1.0e-10);
219 
220         ArrayList<String> stateManeuverComment1 = new ArrayList<String>();
221         stateManeuverComment1.add("Second maneuver: first station acquisition maneuver");
222         stateManeuverComment1.add("impulsive, thrust direction fixed in RTN frame");
223         Assert.assertEquals(stateManeuverComment1, file.getManeuver(1).getComment());
224         Assert.assertEquals(new AbsoluteDate(2000, 06, 05, 18, 59, 21,
225                                              TimeScalesFactory.getGPS()),
226                             file.getManeuvers().get(1).getEpochIgnition());
227         Assert.assertEquals(0.0, file.getManeuver(1).getDuration(), 1e-10);
228         Assert.assertEquals(-1.469, file.getManeuver(1).getDeltaMass(), 1e-10);
229         Assert.assertEquals(LOFType.QSW, file.getManeuver(1).getRefLofType());
230         Assert.assertNull(file.getManeuver(1).getRefFrame());
231         Assert.assertEquals(0.0,
232                             new Vector3D(1.015, -1.873, 0.0).distance(file.getManeuver(1).getDV()),
233                             1.0e-10);
234 
235         Assert.assertTrue(file.getManeuver(2).getComment().isEmpty());
236         Assert.assertEquals(new AbsoluteDate(2000, 06, 05, 18, 59, 51,
237                                              TimeScalesFactory.getGPS()),

242         Assert.assertNull(file.getManeuver(2).getRefFrame());
243         Assert.assertEquals(0.0,
244                             new Vector3D(1.015, -1.873, 0.0).distance(file.getManeuver(2).getDV()),
245                             1.0e-10);
246 
247         file.generateCartesianOrbit();
248         file.generateKeplerianOrbit();
249         file.generateSpacecraftState();

282             throws OrekitException, URISyntaxException {
283         // simple test for OPM file, contains all mandatory information plus
284         // Spacecraft parameters and the position/velocity Covariance Matrix.
285         final String name = getClass().getResource("/ccsds/OPMExample3.txt").toURI().getPath();
286         OPMParser parser = new OPMParser().withConventions(IERSConventions.IERS_2010);

289                                              TimeScalesFactory.getGMST(IERSConventions.IERS_2010, false)),
290                             file.getMetaData().getFrameEpoch());
291         Assert.assertEquals(2, file.getMetaDataComment().size());
292         Assert.assertEquals("GEOCENTRIC, CARTESIAN, EARTH FIXED", file.getMetaDataComment().get(0));

294         Assert.assertEquals(15951238.3495, file.generateKeplerianOrbit().getA(), 0.001);

296         // Check Data Covariance matrix Block
297         ArrayList<String> dataCovMatrixComment = new ArrayList<String>();
298         dataCovMatrixComment.add("toto");
299         dataCovMatrixComment.add("tata");        
300         Assert.assertEquals(dataCovMatrixComment, file.getCovarianceComment());
301         Assert.assertTrue(file.hasCovarianceMatrix());
302         Assert.assertEquals(file.getCovRefFrame(), FramesFactory.getTEME());
303 
304         Array2DRowRealMatrix covMatrix = new Array2DRowRealMatrix(6, 6);
305         double[] column1 = {
306             333.1349476038534, 461.8927349220216,
307             -307.0007847730449, -0.3349365033922630,
308             -0.2211832501084875, -0.3041346050686871

310         double[] column2 = {
311             461.8927349220216, 678.2421679971363,
312             -422.1234189514228, -0.4686084221046758,
313             -0.2864186892102733, -0.4989496988610662
314         };
315         double[] column3 = {
316             -307.0007847730449, -422.1234189514228,
317             323.1931992380369, 0.2484949578400095,
318             0.1798098699846038, 0.3540310904497689
319         };
320         double[] column4 = {
321             -0.3349365033922630, -0.4686084221046758,
322             0.2484949578400095, 0.0004296022805587290,
323             0.0002608899201686016, 0.0001869263192954590
324         };
325         double[] column5 = {
326             -0.2211832501084875, -0.2864186892102733,

475     private void checkPVEntry(final PVCoordinates expected,
476                               final PVCoordinates actual) {
477         final Vector3D expectedPos = expected.getPosition();
478         final Vector3D expectedVel = expected.getVelocity();
479 
480         final Vector3D actualPos = actual.getPosition();
481         final Vector3D actualVel = actual.getVelocity();
482 
483         final double eps = 1e-12;
484 
485         Assert.assertEquals(expectedPos.getX(), actualPos.getX(), eps);
486         Assert.assertEquals(expectedPos.getY(), actualPos.getY(), eps);
487         Assert.assertEquals(expectedPos.getZ(), actualPos.getZ(), eps);
488 
489         Assert.assertEquals(expectedVel.getX(), actualVel.getX(), eps);
490         Assert.assertEquals(expectedVel.getY(), actualVel.getY(), eps);
491         Assert.assertEquals(expectedVel.getZ(), actualVel.getZ(), eps);
492     }

36      public void testReadLimits() throws OrekitException {
37          Utils.setDataRoot("potential");
38          GravityFieldFactory.addPotentialCoefficientsReader(new ICGEMFormatReader("g007_eigen_05c_coef", false));
39          UnnormalizedSphericalHarmonicsProvider provider = GravityFieldFactory.getUnnormalizedProvider(3, 2);

46              // expected
47          } catch (Exception e) {
48              Assert.fail("wrong exception caught: " + e.getLocalizedMessage());
49          }
50          try {
51              harmonics.getUnnormalizedCnm(4, 2);

54              // expected
55          } catch (Exception e) {
56              Assert.fail("wrong exception caught: " + e.getLocalizedMessage());
57          }
58          harmonics.getUnnormalizedCnm(3, 2);
59          Assert.assertEquals(3, provider.getMaxDegree());
60          Assert.assertEquals(2, provider.getMaxOrder());
61      }

63      @Test
64      public void testRegular05cNormalized() throws OrekitException {
65          Utils.setDataRoot("potential");
66          GravityFieldFactory.addPotentialCoefficientsReader(new ICGEMFormatReader("g007_eigen_05c_coef", false));
67          NormalizedSphericalHarmonicsProvider provider = GravityFieldFactory.getNormalizedProvider(5, 5);
68          Assert.assertEquals(TideSystem.TIDE_FREE, provider.getTideSystem());
69          AbsoluteDate refDate = new AbsoluteDate("2004-10-01T12:00:00", TimeScalesFactory.getTT());
70          Assert.assertEquals(refDate, provider.getReferenceDate());
71          AbsoluteDate date = new AbsoluteDate("2013-01-08T10:46:53", TimeScalesFactory.getTT());
72          NormalizedSphericalHarmonics harmonics = provider.onDate(date);
73          Assert.assertEquals(date.durationFrom(refDate), provider.getOffset(date), Precision.SAFE_MIN);
74  
75          double offset     = date.durationFrom(refDate);
76          double offsetYear = offset / Constants.JULIAN_YEAR;
77          Assert.assertEquals(0.957212879862e-06 + offsetYear * 0.490000000000e-11,
78                              harmonics.getNormalizedCnm(3, 0), 1.0e-15);
79          Assert.assertEquals( 0.174804558032e-06, harmonics.getNormalizedCnm(5, 5), 1.0e-15);
80          Assert.assertEquals( 0.0,                harmonics.getNormalizedSnm(4, 0), 1.0e-15);
81          Assert.assertEquals( 0.308816581016e-06, harmonics.getNormalizedSnm(4, 4), 1.0e-15);
82          Assert.assertEquals(0.3986004415E+15, provider.getMu(), 1.0e-20);
83          Assert.assertEquals(0.6378136460E+07, provider.getAe(), 1.0e-20);
84      }

117     public void testRegular05cUnnormalized() throws OrekitException {
118         Utils.setDataRoot("potential");
119         GravityFieldFactory.addPotentialCoefficientsReader(new ICGEMFormatReader("g007_eigen_05c_coef", false));
120         UnnormalizedSphericalHarmonicsProvider provider = GravityFieldFactory.getUnnormalizedProvider(5, 5);

123         Assert.assertEquals(refDate, provider.getReferenceDate());

135                    date, 4, 0, 2004, 10, 1, 0, 0, 0, 0, 0, 0,
136                    maxUlps);
137         checkValue(harmonics.getUnnormalizedSnm(4, 4),
138                    date, 4, 4, 2004, 10, 1, 0.308816581016e-06, 0, 0, 0, 0, 0,
139                    maxUlps);

247     private void checkValue(final double value,
248                             final AbsoluteDate date, final int n, final int m,
249                             final int refYear, final int refMonth, final int refDay,
250                             final double constant, final double trend,
251                             final double cosYear, final double sinYear,
252                             final double cosHalfYear, final double sinHalfYear,
253                             final int maxUlps)
254         throws OrekitException {
255         double factor = GravityFieldFactory.getUnnormalizationFactors(n, m)[n][m];
256         AbsoluteDate refDate = new AbsoluteDate(refYear, refMonth, refDay, 12, 0, 0, TimeScalesFactory.getTT());
257         double dtYear = date.durationFrom(refDate) / Constants.JULIAN_YEAR;
258         double normalized = factor * (constant +
259                                       trend       * dtYear +
260                                       cosYear     * FastMath.cos(MathUtils.TWO_PI * dtYear) +
261                                       sinYear     * FastMath.sin(MathUtils.TWO_PI * dtYear) +
262                                       cosHalfYear * FastMath.cos(MathUtils.TWO_PI * dtYear * 2) +
263                                       sinHalfYear * FastMath.sin(MathUtils.TWO_PI * dtYear * 2));
264         double epsilon = maxUlps * FastMath.ulp(normalized);
265         Assert.assertEquals(normalized, value, epsilon);
266     }

20  package edu.umd.cs.findbugs;
21  
22  import java.lang.reflect.Modifier;
23  
24  /**
25   * Check that the BCEL classes present seem to be the right ones. Specifically,
26   * we check whether the ones extended in FindBugs code are non-final. The
27   * following BCEL classes are extended in FindBugs code:
28   *
29   * org.apache.bcel.generic.ObjectType; org.apache.bcel.generic.Type;
30   * org.apache.bcel.Constants; org.apache.bcel.classfile.EmptyVisitor
31   * org.apache.bcel.Repository;
32   *
33   * @author langmead
34   */
35  
36  public class CheckBcel {
37  
38      /**
39       *
40       */
41      private static final String ORG_APACHE_BCEL_REPOSITORY = "org.apache.bcel.Repository";
42  
43      /**
44       *
45       */
46      private static final String ORG_APACHE_BCEL_CLASSFILE_EMPTY_VISITOR = "org.apache.bcel.classfile.EmptyVisitor";
47  
48      /**
49       *
50       */
51      private static final String ORG_APACHE_BCEL_CONSTANTS = "org.apache.bcel.Constants";
52  
53      /**
54       *
55       */
56      private static final String ORG_APACHE_BCEL_GENERIC_TYPE = "org.apache.bcel.generic.Type";
57  
58      /**
59       *
60       */
61      private static final String ORG_APACHE_BCEL_GENERIC_OBJECT_TYPE = "org.apache.bcel.generic.ObjectType";
62  
63      /**
64       * Check whether given Class is declared final
65       *
66       * @param c
67       *            the class to check
68       * @return true iff Class is declared final
69       */
70      private static boolean isFinal(Class<?> c) {
71          return (c.getModifiers() & Modifier.FINAL) != 0;
72      }
73  
74      /**
75       * Output an appropriate error when a BCEL class looks wrong.
76       *
77       * @param cname
78       *            name of the BCEL class
79       */
80      private static void error(String cname) {
81          System.err.println("BCEL class compatibility error.");
82          System.err.println("The version of class " + cname + " found was not compatible with\n"
83                  + "SpotBugs.  Please remove any BCEL libraries that may be interfering.  This may happen\n"
84                  + "if you have an old version of BCEL or a library that includes an old version of BCEL\n"
85                  + "in an \"endorsed\" directory.");
86      }
87  
88      /**
89       * Check that the BCEL classes present seem to be the right ones.
90       * Specifically, we check whether the ones extended in FindBugs code are
91       * non-final.
92       *
93       * @return true iff all checks passed
94       */
95      public static boolean check() {
96          Class<?> objectType;
97          Class<?>  type;
98          Class<?>  constants;
99          Class<?>  emptyVis;
100         Class<?>  repository;
101         try {
102             objectType = Class.forName(ORG_APACHE_BCEL_GENERIC_OBJECT_TYPE);
103             type = Class.forName(ORG_APACHE_BCEL_GENERIC_TYPE);
104             constants = Class.forName(ORG_APACHE_BCEL_CONSTANTS);
105             emptyVis = Class.forName(ORG_APACHE_BCEL_CLASSFILE_EMPTY_VISITOR);
106             repository = Class.forName(ORG_APACHE_BCEL_REPOSITORY);
107 
108         } catch (ClassNotFoundException e) {
109             System.out.println("One or more required BCEL classes were missing."
110                     + " Ensure that bcel.jar is placed at the same directory with spotbugs.jar");
111             return false;
112         }
113         if (isFinal(objectType)) {
114             error(ORG_APACHE_BCEL_GENERIC_OBJECT_TYPE);
115             return false;
116         }
117         if (isFinal(type)) {
118             error(ORG_APACHE_BCEL_GENERIC_TYPE);
119             return false;
120         }
121         if (isFinal(constants)) {
122             error(ORG_APACHE_BCEL_CONSTANTS);
123             return false;
124         }
125         if (isFinal(emptyVis)) {
126             error(ORG_APACHE_BCEL_CLASSFILE_EMPTY_VISITOR);
127             return false;
128         }
129         if (isFinal(repository)) {
130             error(ORG_APACHE_BCEL_REPOSITORY);
131             return false;
132         }
133         return true;
134     }
135 }

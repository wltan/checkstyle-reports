190     public void acquireDisplayWait() {
191         synchronized (waitLock) {
192             waitCount++;
193             if (GUI2_DEBUG) {
194                 System.err.println("acquiring display wait, count " + waitCount);
195                 Thread.dumpStack();
196             }
197             if (waitCount == 1) {
198                 mainFrameTree.showCard(BugCard.WAITCARD, new Cursor(Cursor.WAIT_CURSOR), this);
199             }
200         }
201     }
202 
203     volatile Exception previousDecrementToZero;
204 
205     public void releaseDisplayWait() {
206         synchronized (waitLock) {
207             if (waitCount <= 0) {
208                 if (previousDecrementToZero != null) {
209                     throw new IllegalStateException("Can't decrease wait count; already zero", previousDecrementToZero);
210                 } else {
211                     throw new IllegalStateException("Can't decrease wait count; never incremented");
212                 }
213             }
214             waitCount--;
215             if (GUI2_DEBUG) {
216                 System.err.println("releasing display wait, count " + waitCount);
217                 Thread.dumpStack();
218             }
219             if (waitCount == 0) {
220                 mainFrameTree.showCard(BugCard.TREECARD, new Cursor(Cursor.DEFAULT_CURSOR), this);
221                 previousDecrementToZero = new Exception("Previously decremented at");
222             }
223         }
224     }

341     /**
342      * This method is called when the application is closing. This is either by
343      * the exit menuItem or by clicking on the window's system menu.
344      */
345     void callOnClose() {
346         if (projectChanged && !SystemProperties.getBoolean("findbugs.skipSaveChangesWarning")) {
347             Object[] options = {
348                     L10N.getLocalString("dlg.save_btn", "Save"),
349                     L10N.getLocalString("dlg.dontsave_btn", "Don't Save"),
350                     L10N.getLocalString("dlg.cancel_btn", "Cancel"),
351             };
352             int value = JOptionPane.showOptionDialog(this, getActionWithoutSavingMsg("closing"),
353                     edu.umd.cs.findbugs.L10N.getLocalString("msg.confirm_save_txt", "Do you want to save?"),
354                     JOptionPane.YES_NO_CANCEL_OPTION, JOptionPane.QUESTION_MESSAGE, null, options, options[0]);
355 
356             if (value == 2 || value == JOptionPane.CLOSED_OPTION) {
357                 return;
358             } else if (value == 0) {
359 
360                 if (saveFile == null) {
361                     if (!mainFrameLoadSaveHelper.saveAs()) {
362                         return;
363                     }
364                 } else {
365                     mainFrameLoadSaveHelper.save();
366                 }
367             }
368         }
369 
370         GUISaveState guiSaveState = GUISaveState.getInstance();
371         guiLayout.saveState();
372         guiSaveState.setFrameBounds(getBounds());
373         guiSaveState.setExtendedWindowState(getExtendedState());
374         guiSaveState.save();
375         System.exit(0);
376     }

432     @SwingThread
433     private void setProjectAndBugCollection(@CheckForNull Project project, @CheckForNull BugCollection bugCollection) {
434         if (GUI2_DEBUG) {
435             if (bugCollection == null) {
436                 System.out.println("Setting bug collection to null");
437             } else {
438                 System.out.println("Setting bug collection; contains " + bugCollection.getCollection().size() + " bugs");
439             }
440 
441         }
442         if (bugCollection != null && bugCollection.getProject() != project) {
443             Project p2 = bugCollection.getProject();
444             throw new IllegalArgumentException(String.format("project %x and bug collection %x don't match",
445                     System.identityHashCode(project), System.identityHashCode(p2)));
446         }
447         acquireDisplayWait();
448         try {
449             // setRebuilding(false);
450             setProject(project);
451             this.bugCollection = bugCollection;
452             BugLoader.addDeadBugMatcher(bugCollection);
453 
454             displayer.clearCache();
455             mainFrameTree.updateBugTree();
456             setProjectChanged(false);
457             Runnable runnable = () -> {
458                 PreferencesFrame.getInstance().updateFilterPanel();
459                 mainFrameMenu.getReconfigMenuItem().setEnabled(true);
460                 mainFrameMenu.setViewMenu();
461                 newProject();
462                 clearSourcePane();
463                 clearSummaryTab();
464 
465                 /*
466                  * This is here due to a threading issue. It can only be
467                  * called after curProject has been changed. Since this
468                  * method is called by both open methods it is put here.
469                  */
470                 updateTitle();
471             };
472             if (SwingUtilities.isEventDispatchThread()) {
473                 runnable.run();
474             } else {
475                 SwingUtilities.invokeLater(runnable);
476             }
477         } finally {
478             releaseDisplayWait();
479         }
480     }

639     /**
640      * Changes the title based on curProject and saveFile.

642     public void updateTitle() {
643         Project project = getProject();
644         String name = project.getProjectName();
645         if ((name == null || "".equals(name.trim())) && saveFile != null) {
646             name = saveFile.getAbsolutePath();
647         }
648         if (name == null) {
649             name = "";//Project.UNNAMED_PROJECT;
650         }
651         String oldTitle = this.getTitle();
652         String newTitle = TITLE_START_TXT + ("".equals(name.trim()) ? "" : " - " + name);
653         if (oldTitle.equals(newTitle)) {
654             return;
655         }
656         this.setTitle(newTitle);
657     }

669     public void selectPackagePrefixByProject() {
670         TreeSet<String> projects = new TreeSet<>(String.CASE_INSENSITIVE_ORDER);
671         Multiset<String> count = new Multiset<>();
672         int total = 0;
673         for (BugInstance b : getBugCollection().getCollection()) {
674             if (shouldDisplayIssueIgnoringPackagePrefixes(b)) {
675                 TreeSet<String> projectsForThisBug = projectPackagePrefixes.getProjects(b.getPrimaryClass().getClassName());
676                 projects.addAll(projectsForThisBug);
677                 count.addAll(projectsForThisBug);
678                 total++;
679             }
680         }
681         if (projects.size() == 0) {
682             JOptionPane.showMessageDialog(this, "No issues in current view");
683             return;
684         }
685         ArrayList<ProjectSelector> selectors = new ArrayList<>(projects.size() + 1);
686         ProjectSelector everything = new ProjectSelector("all projects", "", total);
687         selectors.add(everything);
688         for (String projectName : projects) {
689             ProjectPackagePrefixes.PrefixFilter filter = projectPackagePrefixes.getFilter(projectName);
690             selectors.add(new ProjectSelector(projectName, filter.toString(), count.getCount(projectName)));
691         }
692         ProjectSelector choice = (ProjectSelector) JOptionPane.showInputDialog(null,
693                 "Choose a project to set appropriate package prefix(es)", "Select package prefixes by package",
694                 JOptionPane.QUESTION_MESSAGE, null, selectors.toArray(), everything);
695         if (choice == null) {
696             return;
697         }
698 
699         mainFrameTree.setFieldForPackagesToDisplayText(choice.filter);
700         viewFilter.setPackagesToDisplay(choice.filter);
701         resetViewCache();
702 
703     }

705     private static String join(String s1, String s2) {
706         if (s1 == null || s1.length() == 0) {
707             return s2;
708         }
709         if (s2 == null || s2.length() == 0) {
710             return s1;
711         }
712         return s1 + "; " + s2;
713     }

721             summaryTopPanel.add(mainFrameComponentFactory.bugSummaryComponent(bug.getAbridgedMessage(), bug));
722 
723             for (BugAnnotation b : bug.getAnnotationsForMessage(true)) {
724                 summaryTopPanel.add(mainFrameComponentFactory.bugSummaryComponent(b, bug));
725             }
726 
727 
728             BugPattern bugPattern = bug.getBugPattern();
729             String detailText =
730                     bugPattern.getDetailText()
731                     +"<br><p> <b>Bug kind and pattern: " +
732                     bugPattern.getAbbrev() + " - " + bugPattern.getType();
733             String txt = bugPattern.getDetailHTML(detailText);
734             summaryHtmlArea.setText(txt);
735 
736             summaryTopPanel.add(Box.createVerticalGlue());
737             summaryTopPanel.revalidate();
738 
739             SwingUtilities.invokeLater(() -> summaryHtmlScrollPane.getVerticalScrollBar().setValue(
740                     summaryHtmlScrollPane.getVerticalScrollBar().getMinimum()));
741         });
742     }

750     public void searchSource(int type) {
751         int targetLineNum = -1;
752         String targetString = sourceSearchTextField.getText();
753         switch (type) {
754         case 0:
755             targetLineNum = displayer.find(targetString);
756             break;
757         case 1:
758             targetLineNum = displayer.findNext(targetString);
759             break;
760         case 2:
761             targetLineNum = displayer.findPrevious(targetString);
762             break;
763         default:
764             break;
765         }
766         if (targetLineNum != -1) {
767             displayer.foundItem(targetLineNum);
768         }
769     }

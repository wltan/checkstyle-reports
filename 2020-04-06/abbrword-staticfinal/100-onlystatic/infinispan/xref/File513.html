3   import org.apache.commons.io.FileUtils;
4   import org.apache.commons.io.IOUtils;
5   import org.apache.directory.api.ldap.model.constants.SupportedSaslMechanisms;
6   import org.apache.directory.api.ldap.model.entry.DefaultEntry;
7   import org.apache.directory.api.ldap.model.ldif.LdifEntry;
8   import org.apache.directory.api.ldap.model.ldif.LdifReader;
9   import org.apache.directory.api.ldap.model.schema.SchemaManager;
10  import org.apache.directory.server.annotations.CreateKdcServer;
11  import org.apache.directory.server.annotations.CreateLdapServer;
12  import org.apache.directory.server.annotations.CreateTransport;
13  import org.apache.directory.server.annotations.SaslMechanism;
14  import org.apache.directory.server.core.annotations.AnnotationUtils;
15  import org.apache.directory.server.core.annotations.ContextEntry;
16  import org.apache.directory.server.core.annotations.CreateDS;
17  import org.apache.directory.server.core.annotations.CreateIndex;
18  import org.apache.directory.server.core.annotations.CreatePartition;
19  import org.apache.directory.server.core.api.DirectoryService;
20  import org.apache.directory.server.core.factory.DSAnnotationProcessor;
21  import org.apache.directory.server.core.kerberos.KeyDerivationInterceptor;
22  import org.apache.directory.server.factory.ServerAnnotationProcessor;
23  import org.apache.directory.server.kerberos.kdc.KdcServer;
24  import org.apache.directory.server.ldap.LdapServer;
25  import org.apache.directory.server.ldap.handlers.sasl.cramMD5.CramMd5MechanismHandler;
26  import org.apache.directory.server.ldap.handlers.sasl.digestMD5.DigestMd5MechanismHandler;
27  import org.apache.directory.server.ldap.handlers.sasl.gssapi.GssapiMechanismHandler;
28  import org.apache.directory.server.ldap.handlers.sasl.ntlm.NtlmMechanismHandler;
29  import org.apache.directory.server.ldap.handlers.sasl.plain.PlainMechanismHandler;

31  /** 
32   * @author vjuranek
33   * @since 7.0
34   */
35  public class ApacheDsKrbLdap {
36     
37     
38     public static final int LDAP_PORT = 10389;
39     public static final int KERBEROS_PORT = 6088;
40     public static final String KERBEROS_PRIMARY_REALM = "INFINISPAN.ORG";
41     public static final String LDAP_INIT_FILE = "ldif/ispn-krb-test.ldif";
42     public static final String BASE_DN = "dc=infinispan,dc=org"; 
43  
44     private DirectoryService directoryService;
45     private LdapServer ldapServer;
46     private KdcServer kdcServer;
47     
48     public ApacheDsKrbLdap(String hostname) throws Exception {
49        createDs();
50        createKdc();
51        createLdap(hostname);
52     }
53    
54     public void start() throws Exception {
55        ldapServer.start();
56     }
57     
58     public void stop() throws Exception {
59        kdcServer.stop();
60        ldapServer.stop();
61        directoryService.shutdown();
62        FileUtils.deleteDirectory(directoryService.getInstanceLayout().getInstanceDirectory());
63     }
64     
65     @CreateDS(
66           name = "InfinispanDS",
67           partitions = {
68                 @CreatePartition(
69                       name = "infinispan",
70                       suffix = BASE_DN,
71                       contextEntry = @ContextEntry(
72                             entryLdif =
73                             "dn: " + BASE_DN + "\n" +
74                             "dc: infinispan\n" +
75                             "objectClass: top\n" +
76                             "objectClass: domain\n\n" ),
77                             indexes = {
78                                @CreateIndex( attribute = "objectClass" ),
79                                @CreateIndex( attribute = "dc" ),
80                                @CreateIndex( attribute = "ou" )
81                             }
82                       )
83           },
84           additionalInterceptors = { KeyDerivationInterceptor.class }
85     )
86     public void createDs() throws Exception {
87        directoryService = DSAnnotationProcessor.getDirectoryService();
88     }
89     
90     @CreateKdcServer(
91           primaryRealm = KERBEROS_PRIMARY_REALM,
92           kdcPrincipal = "krbtgt/" + KERBEROS_PRIMARY_REALM + "@" + KERBEROS_PRIMARY_REALM,
93           searchBaseDn = BASE_DN,
94           transports = {@CreateTransport( protocol = "UDP", port = KERBEROS_PORT)}
95     )
96     public void createKdc() throws Exception {
97        kdcServer = ServerAnnotationProcessor.getKdcServer(directoryService, KERBEROS_PORT);
98     }
99     
100    @CreateLdapServer(
101          transports = { @CreateTransport( protocol = "LDAP",  port = LDAP_PORT) },
102          saslRealms = {KERBEROS_PRIMARY_REALM},
103          saslMechanisms = {
104                @SaslMechanism( name=SupportedSaslMechanisms.GSSAPI, implClass=GssapiMechanismHandler.class),
105                @SaslMechanism( name= SupportedSaslMechanisms.PLAIN, implClass=PlainMechanismHandler.class ),
106                @SaslMechanism( name=SupportedSaslMechanisms.CRAM_MD5, implClass=CramMd5MechanismHandler.class),
107                @SaslMechanism( name= SupportedSaslMechanisms.DIGEST_MD5, implClass=DigestMd5MechanismHandler.class),
108                @SaslMechanism( name=SupportedSaslMechanisms.NTLM, implClass=NtlmMechanismHandler.class),
109                @SaslMechanism( name=SupportedSaslMechanisms.GSS_SPNEGO, implClass=NtlmMechanismHandler.class)
110          }
111    )
112    public void createLdap(final String hostname) throws Exception {
113       final String initFile = System.getProperty("ldap.init.file", LDAP_INIT_FILE);
114       final String ldifContent = IOUtils.toString(getClass().getClassLoader().getResource(initFile));
115       final SchemaManager schemaManager = directoryService.getSchemaManager();
116      
117       try {
118          for (LdifEntry ldifEntry : new LdifReader(IOUtils.toInputStream(ldifContent))) {
119             directoryService.getAdminSession().add(new DefaultEntry(schemaManager, ldifEntry.getEntry()));
120          }
121       } catch (Exception e) {
122          e.printStackTrace();
123          throw e;
124       }
125       final CreateLdapServer createLdapServer = (CreateLdapServer) AnnotationUtils.getInstance(CreateLdapServer.class);    
126       ldapServer = ServerAnnotationProcessor.instantiateLdapServer(createLdapServer, directoryService);
127       ldapServer.setSearchBaseDn(BASE_DN);
128       ldapServer.setSaslHost(hostname);
129       ldapServer.setSaslPrincipal("ldap/" + hostname + "@" + KERBEROS_PRIMARY_REALM);
130    }

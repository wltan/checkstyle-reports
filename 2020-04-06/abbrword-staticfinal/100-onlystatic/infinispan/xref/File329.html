5   import org.infinispan.configuration.cache.CacheMode;

9   import org.infinispan.test.MultipleCacheManagersTest;
10  import org.testng.annotations.Test;
11  
12  import static org.testng.Assert.assertEquals;
13  
14  /**
15   * @author gustavonalle
16   * @since 7.0
17   */
18  public abstract class BaseDeltaAwarePreloadTest extends MultipleCacheManagersTest {
19  
20     protected static final int CLUSTER_SIZE = 2;
21  
22     abstract boolean isTx();
23  
24     @Override
25     protected void createCacheManagers() throws Throwable {
26        createCluster(getConfig(isTx()), CLUSTER_SIZE);
27        waitForClusterToForm();
28     }
29  
30     protected ConfigurationBuilder getConfig(boolean tx) throws Throwable {
31        ConfigurationBuilder c = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, tx);
32        c.persistence().addStore(new DummyInMemoryStoreConfigurationBuilder(c.persistence()).storeName(getClass().getSimpleName())).preload(true);
33        return c;
34     }
35  
36     @Test
37     public void testPreloadOnStart() throws PersistenceException {
38        Cache<Object, Object> cache = caches().get(0);
39        TestDeltaAware value = new TestDeltaAware();
40        value.setFirstComponent("1st");
41        value.setSecondComponent("2nd");
42        cache.put(1, value);
43        cache.stop();
44        cache.start();
45  
46        TestDeltaAware deltaAware = (TestDeltaAware) cache.get(1);
47  
48        assertEquals("1st", deltaAware.getFirstComponent());
49        assertEquals("2nd", deltaAware.getSecondComponent());
50     }
51  
52  }

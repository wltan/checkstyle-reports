1   package org.infinispan.spring.provider.sample.resolvers;
2   
3   import java.util.ArrayList;
4   import java.util.Arrays;
5   import java.util.Collection;
6   
7   import org.springframework.beans.factory.annotation.Autowired;
8   import org.springframework.cache.Cache;
9   import org.springframework.cache.CacheManager;
10  import org.springframework.cache.interceptor.CacheOperationInvocationContext;
11  import org.springframework.cache.interceptor.CacheResolver;
12  import org.springframework.stereotype.Component;
13  
14  /**
15   * Simple implementation of {@link org.springframework.cache.interceptor.CacheResolver} interface. It returns a single
16   * instance of {@link org.springframework.cache.Cache} with name 'custom'.
17   *
18   * @author Matej Cimbora (mcimbora@redhat.com)
19   */
20  @Component
21  public class CustomCacheResolver implements CacheResolver {
22  
23     private static final String CACHE_NAME = "custom";
24  
25     @Autowired(required = true)
26     private CacheManager cacheManager;
27  
28     @Override
29     public Collection<? extends Cache> resolveCaches(CacheOperationInvocationContext<?> cacheOperationInvocationContext) {
30        return new ArrayList<Cache>(Arrays.asList(cacheManager.getCache(CACHE_NAME)));
31     }
32  }

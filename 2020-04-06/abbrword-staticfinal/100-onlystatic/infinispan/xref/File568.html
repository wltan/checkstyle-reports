1   package org.infinispan.persistence.jpa.configuration;
2   
3   import org.infinispan.commons.configuration.BuiltBy;
4   import org.infinispan.commons.configuration.ConfigurationFor;
5   import org.infinispan.commons.configuration.attributes.Attribute;
6   import org.infinispan.commons.configuration.attributes.AttributeDefinition;
7   import org.infinispan.commons.configuration.attributes.AttributeSet;
8   import org.infinispan.configuration.cache.AbstractStoreConfiguration;
9   import org.infinispan.configuration.cache.AsyncStoreConfiguration;
10  import org.infinispan.configuration.cache.SingletonStoreConfiguration;
11  import org.infinispan.persistence.jpa.JpaStore;
12  
13  /**
14   *
15   * @author <a href="mailto:rtsang@redhat.com">Ray Tsang</a>
16   *
17   */
18  @BuiltBy(JpaStoreConfigurationBuilder.class)
19  @ConfigurationFor(JpaStore.class)
20  public class JpaStoreConfiguration extends AbstractStoreConfiguration {
21     static final AttributeDefinition<String> PERSISTENCE_UNIT_NAME = AttributeDefinition.builder("persistenceUnitName", null, String.class).immutable().build();
22     static final AttributeDefinition<Class> ENTITY_CLASS = AttributeDefinition.builder("entityClass", null, Class.class).immutable().build();
23     static final AttributeDefinition<Long> BATCH_SIZE = AttributeDefinition.builder("batchSize", 100l).immutable().build();
24     static final AttributeDefinition<Boolean> STORE_METADATA = AttributeDefinition.builder("storeMetadata", true).immutable().build();
25  
26     public static AttributeSet attributeDefinitionSet() {
27        return new AttributeSet(JpaStoreConfiguration.class, AbstractStoreConfiguration.attributeDefinitionSet(), PERSISTENCE_UNIT_NAME, ENTITY_CLASS, BATCH_SIZE, STORE_METADATA);
28     }
29  
30     private final Attribute<String> persistenceUnitName;
31     private final Attribute<Class> entityClass;
32     private final Attribute<Long> batchSize;
33     private final Attribute<Boolean> storeMetadata;
34  
35     protected JpaStoreConfiguration(AttributeSet attributes, AsyncStoreConfiguration async, SingletonStoreConfiguration singletonStore) {
36        super(attributes, async, singletonStore);
37        persistenceUnitName = attributes.attribute(PERSISTENCE_UNIT_NAME);
38        entityClass = attributes.attribute(ENTITY_CLASS);
39        batchSize = attributes.attribute(BATCH_SIZE);
40        storeMetadata = attributes.attribute(STORE_METADATA);
41     }
42  
43     public String persistenceUnitName() {
44        return persistenceUnitName.get();
45     }
46  
47     public Class<?> entityClass() {
48        return entityClass.get();
49     }
50  
51     public long batchSize() {
52        return batchSize.get();
53     }
54  
55     public boolean storeMetadata() {
56        return storeMetadata.get();
57     }
58  }

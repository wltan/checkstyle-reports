23  package org.jboss.as.clustering.jgroups;
24  
25  import java.net.URL;
26  
27  import org.jboss.as.controller.OperationFailedException;
28  import org.jboss.logging.annotations.Message;
29  import org.jboss.logging.annotations.MessageBundle;
30  import org.jboss.logging.Messages;
31  
32  /**
33   * Date: 29.08.2011
34   *
35   * @author <a href="mailto:jperkins@redhat.com">James R. Perkins</a>
36   */
37  @MessageBundle(projectCode = "JBAS")
38  public interface JGroupsMessages {
39      /**
40       * The messages.
41       */
42      JGroupsMessages MESSAGES = Messages.getBundle(JGroupsMessages.class);
43  
44      /**
45       * A message indicating a file could not be parsed.
46       *
47       * @param url the path to the file.
48       *
49       * @return the message.
50       */
51      @Message(id = 10270, value = "Failed to parse %s")
52      String parserFailure(URL url);
53  
54      /**
55       * A message indicating a resource could not be located.
56       *
57       * @param resource the resource that could not be located.
58       *
59       * @return the message.
60       */
61      @Message(id = 10271, value = "Failed to locate %s")
62      String notFound(String resource);
63  
64      @Message(id = 10272, value = "A node named %s already exists in this cluster.  Perhaps there is already a server running on this host?  If so, restart this server with a unique node name, via -Djboss.node.name=<node-name>")
65      IllegalStateException duplicateNodeName(String name);
66  
67      @Message(id = 10273, value = "Transport for stack %s is not defined. Please specify both a transport and protocol list, either as optional parameters to add() or via batching.")
68      OperationFailedException transportNotDefined(String stackName);
69  
70      @Message(id = 10274, value = "Protocol list for stack %s is not defined. Please specify both a transport and protocol list, either as optional parameters to add() or via batching.")
71      OperationFailedException protocolListNotDefined(String stackName);
72  
73      @Message(id = 10275, value = "Protocol with relative path %s is already defined.")
74      OperationFailedException protocolAlreadyDefined(String relativePath);
75  
76      @Message(id = 10276, value = "Protocol with relative path %s is not defined.")
77      OperationFailedException protocolNotDefined(String relativePath);
78  
79      @Message(id = 10277, value = "Property %s for protocol with relative path %s is not defined. ")
80      OperationFailedException propertyNotDefined(String propertyName, String protocolRelativePath);
81  
82      @Message(id = 10299, value = "Unauthorized node %s attempting to join cluster.")
83      SecurityException unauthorizedNodeJoin(String nodeName);
84  }

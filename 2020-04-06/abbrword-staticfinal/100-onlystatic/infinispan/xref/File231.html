34  /**
35   * A registry where components which have been created are stored.  Components are stored as singletons, registered
36   * under a specific name.
37   * <p/>
38   * Components can be retrieved from the registry using {@link #getComponent(Class)}.
39   * <p/>
40   * Components can be registered using {@link #registerComponent(Object, Class)}, which will cause any dependencies to be
41   * wired in as well.  Components that need to be created as a result of wiring will be done using {@link
42   * #getOrCreateComponent(Class)}, which will look up the default factory for the component type (factories annotated
43   * with the appropriate {@link DefaultFactoryFor} annotation.
44   * <p/>
45   * Default factories are treated as components too and will need to be wired before being used.
46   * <p/>
47   * The registry can exist in one of several states, as defined by the {@link org.infinispan.lifecycle.ComponentStatus}
48   * enumeration. In terms of the cache, state changes in the following manner: <ul> <li>INSTANTIATED - when first
49   * constructed</li> <li>CONSTRUCTED - when created using the DefaultCacheFactory</li> <li>STARTED - when {@link
50   * org.infinispan.Cache#start()} is called</li> <li>STOPPED - when {@link org.infinispan.Cache#stop()} is called</li>
51   * </ul>
52   * <p/>
53   * Cache configuration can only be changed and will only be re-injected if the cache is not in the {@link
54   * org.infinispan.lifecycle.ComponentStatus#RUNNING} state.
55   *
56   * Thread Safety: instances of {@link GlobalComponentRegistry} can be concurrently updated so all
57   * the write operations are serialized through class intrinsic lock.
58   *
59   * @author Manik Surtani
60   * @author Galder Zamarre√±o
61   * @since 4.0
62   */
63  @SurvivesRestarts
64  @Scope(Scopes.NAMED_CACHE)
65  public abstract class AbstractComponentRegistry implements Lifecycle, Cloneable {

105    /**
106     * Wires an object instance with dependencies annotated with the {@link Inject} annotation, creating more components
107     * as needed based on the Configuration passed in if these additional components don't exist in the {@link
108     * ComponentRegistry}.  Strictly for components that don't otherwise live in the registry and have a lifecycle, such
109     * as Commands.
110     *
111     * @param target object to wire
112     * @throws CacheConfigurationException if there is a problem wiring the instance
113     */
114    public void wireDependencies(Object target) throws CacheConfigurationException {
115       try {
116          Class<?> targetClass = target.getClass();
117          ComponentMetadata metadata = getComponentMetadataRepo().findComponentMetadata(targetClass);
118          if (metadata != null && metadata.getInjectMethods() != null && metadata.getInjectMethods().length != 0) {
119             // search for anything we need to inject
120             for (ComponentMetadata.InjectMetadata injectMetadata : metadata.getInjectMethods()) {
121                Class<?>[] methodParameters = injectMetadata.getParameterClasses();
122                if (methodParameters == null) {
123                   methodParameters = ReflectionUtil.toClassArray(injectMetadata.getParameters(), getClassLoader());
124                   injectMetadata.setParameterClasses(methodParameters);
125                }
126 
127                Method method = injectMetadata.getMethod();
128                if (method == null) {
129                   method = ReflectionUtil.findMethod(targetClass, injectMetadata.getMethodName(), methodParameters);
130                   injectMetadata.setMethod(method);
131                }
132                invokeInjectionMethod(target, injectMetadata);
133             }
134          }
135       } catch (Exception e) {
136          throw new CacheConfigurationException("Unable to configure component (type: " + target.getClass() + ", instance " + target + ")", e);
137       }
138    }

167    protected synchronized void registerComponentInternal(Object component, String name, boolean nameIsFQCN) {
168       if (component == null)
169          throw new NullPointerException("Cannot register a null component under name [" + name + "]");
170       Component old = componentLookup.get(name);
171 
172       if (old != null) {
173          // if they are equal don't bother
174          if (old.instance.equals(component)) {
175             getLog().tracef("Attempting to register a component equal to one that already exists under the same name (%s).  Not doing anything.", name);
176             return;
177          }
178       }
179 
180       Component c;
181       if (old != null) {
182          getLog().tracef("Replacing old component %s with new instance %s", old, component);
183          old.instance = component;
184          old.methodsScanned = false;
185          c = old;
186       } else {
187          c = new Component();
188          c.name = name;
189          c.instance = component;
190          componentLookup.put(name, c);
191       }
192 
193       c.metadata = getComponentMetadataRepo().findComponentMetadata(component.getClass());
194       try {
195          c.buildInjectionMethodsList();
196       } catch (ClassNotFoundException cnfe) {
197          throw new CacheException("Error injecting dependencies for component " + name, cnfe);
198       }
199       // inject dependencies for this component
200       // we inject dependencies only after the component is already in the map to support cyclical dependencies
201       c.injectDependencies();
202 
203       if (old == null) getLog().tracef("Registering component %s under name %s", c, name);
204       if (state == ComponentStatus.RUNNING) {
205          populateLifeCycleMethods(c);
206          try {
207             invokeStartMethods(Arrays.asList(c.startMethods));
208          } catch (Throwable t) {
209             // the component hasn't started properly, remove its registration
210             componentLookup.remove(name);
211             // the caller will log the exception
212             handleLifecycleTransitionFailure(t);
213          }
214       }
215    }

217    @SuppressWarnings("unchecked")
218    private void invokeInjectionMethod(Object o, ComponentMetadata.InjectMetadata injectMetadata) {
219       Class<?>[] dependencies = injectMetadata.getParameterClasses();
220       if (dependencies.length > 0) {
221          Object[] params = new Object[dependencies.length];
222          if (getLog().isTraceEnabled())
223             getLog().tracef("Injecting dependencies for method [%s] on an instance of [%s].", injectMetadata.getMethod(), o.getClass().getName());
224          for (int i = 0; i < dependencies.length; i++) {
225             String name = injectMetadata.getParameterName(i);
226             boolean nameIsFQCN = !injectMetadata.isParameterNameSet(i);
227             params[i] = getOrCreateComponent(dependencies[i], name, nameIsFQCN);
228          }
229          ReflectionUtil.invokeAccessibly(o, injectMetadata.getMethod(), params);
230       }
231    }

233    /**
234     * Retrieves a component if one exists, and if not, attempts to find a factory capable of constructing the component
235     * (factories annotated with the {@link DefaultFactoryFor} annotation that is capable of creating the component
236     * class).
237     * <p/>
238     * If an instance needs to be constructed, dependencies are then automatically wired into the instance, based on
239     * methods on the component type annotated with {@link Inject}.
240     * <p/>
241     * Summing it up, component retrieval happens in the following order:<br /> 1.  Look for a component that has already
242     * been created and registered. 2.  Look for an appropriate component that exists in the {@link Configuration} that
243     * may be injected from an external system. 3.  Look for a class definition passed in to the {@link Configuration} -
244     * such as an EvictionPolicy implementation 4.  Attempt to create it by looking for an appropriate factory (annotated
245     * with {@link DefaultFactoryFor})
246     * <p/>
247     *
248     * @param componentClass type of component to be retrieved.  Should not be null.
249     * @return a fully wired component instance, or null if one cannot be found or constructed.
250     * @throws CacheConfigurationException if there is a problem with constructing or wiring the instance.
251     */
252    protected synchronized <T> T getOrCreateComponent(Class<T> componentClass) {
253       return getOrCreateComponent(componentClass, componentClass.getName(), true);
254    }

257       return getOrCreateComponent(componentClass, name, false);
258    }

260    @SuppressWarnings("unchecked")
261    protected synchronized <T> T getOrCreateComponent(Class<T> componentClass, String name, boolean nameIsFQCN) {
262       if (DEBUG_DEPENDENCIES) debugStack.push(name);
263 
264       Object component;
265       Component oldWrapper = lookupComponent(componentClass.getName(), name, nameIsFQCN);
266       if (oldWrapper != null) {
267          component = unwrapComponent(oldWrapper);
268       } else {
269          // create this component and add it to the registry
270          AbstractComponentFactory factory = getFactory(componentClass);
271          getLog().tracef("Creating component %s with factory %s", name, factory.getClass());
272          component = factory instanceof NamedComponentFactory ?
273                ((NamedComponentFactory) factory).construct(componentClass, name)
274                : factory.construct(componentClass);
275 
276          if (component != null) {
277             registerComponent(component, name, nameIsFQCN);
278          } else {
279             getLog().tracef("Registering a null for component %s", name);
280             registerNullComponent(name);
281          }
282       }
283 
284       if (DEBUG_DEPENDENCIES) debugStack.pop();
285       return (T) component;
286    }

288    /**
289     * Retrieves a component factory instance capable of constructing components of a specified type.  If the factory
290     * doesn't exist in the registry, one is created.
291     *
292     * @param componentClass type of component to construct
293     * @return component factory capable of constructing such components
294     */
295    protected synchronized AbstractComponentFactory getFactory(Class<?> componentClass) {
296       getLog().tracef("Looking up factory for class %s", componentClass);
297       String cfClass = getComponentMetadataRepo().findFactoryForComponent(componentClass);
298       if (cfClass == null) {
299          throwStackAwareConfigurationException("No registered default factory for component '" + componentClass + "' found!");
300       }
301       // a component factory is a component too!  See if one has been created and exists in the registry
302       AbstractComponentFactory cf = getComponent(cfClass);
303       if (cf == null) {
304          cf = createComponentFactoryInternal(componentClass, cfClass);
305       }
306 
307       // ensure the component factory is in the STARTED state!
308       Component c = lookupComponent(cfClass, cfClass, true);
309       if (c.instance != cf)
310          throwStackAwareConfigurationException("Component factory " + cfClass + " incorrectly registered!");
311       return cf;
312    }
313 
314    protected synchronized AbstractComponentFactory createComponentFactoryInternal(Class<?> componentClass, String cfClass) {
315       //first check as it might have been created in between by another thread
316       AbstractComponentFactory component = getComponent(cfClass);
317       if (component != null) return component;
318 
319       //hasn't yet been created.  Create and put in registry
320       getLog().tracef("Creating factory %s for component %s", cfClass, componentClass);
321       AbstractComponentFactory cf = instantiateFactory(cfClass);
322       if (cf == null)
323          throwStackAwareConfigurationException("Unable to locate component factory for component " + componentClass);
324       // we simply register this factory.  Registration will take care of constructing any dependencies.
325       registerComponent(cf, cfClass);
326       return cf;
327    }

333    /**
334     * No such thing as a meta factory yet.  Factories are created using this method which attempts to use an empty
335     * public constructor.
336     *
337     * @param factoryName classname of factory to be created
338     * @return factory instance
339     */
340    AbstractComponentFactory instantiateFactory(String factoryName) {
341       Class<?> factory = Util.loadClass(factoryName, getClass().getClassLoader());
342       if (AutoInstantiableFactory.class.isAssignableFrom(factory)) {
343          try {
344             return (AbstractComponentFactory) factory.newInstance();
345          } catch (Exception e) {
346             // unable to get a hold of an instance!!
347             throw new CacheConfigurationException("Unable to instantiate factory " + factory + "  Debug stack: " + debugStack, e);
348          }
349       } else {
350          throw new CacheConfigurationException("Cannot auto-instantiate factory " + factory + " as it doesn't implement " + AutoInstantiableFactory.class.getSimpleName() + "!  Debug stack: " + debugStack);
351       }
352    }

446     * Scans each registered component for lifecycle methods, and adds them to the appropriate lists, and then sorts them

450       for (Component c : componentLookup.values()) populateLifeCycleMethods(c);

453    private PrioritizedMethod[] processPrioritizedMethods(ComponentMetadata.PrioritizedMethodMetadata[] methodMetadata,
454                                                          Class<?> componentClass, Component c) {
455       PrioritizedMethod[] retval;
456       int numStartMethods = methodMetadata.length;
457       if (numStartMethods == 0) {
458          retval = EMPTY_PRIO_METHODS;
459       } else {
460          retval = new PrioritizedMethod[numStartMethods];
461          for (int i = 0; i < numStartMethods; i++) {
462             retval[i] = new PrioritizedMethod();
463             retval[i].component = c;
464             retval[i].metadata = methodMetadata[i];
465 
466             if (methodMetadata[i].getMethod() == null) {
467                Method method = ReflectionUtil.findMethod(componentClass, methodMetadata[i].getMethodName());
468                methodMetadata[i].setMethod(method);
469             }
470          }
471          if (retval.length > 1) Arrays.sort(retval);
472       }
473       return retval;
474    }

477    private void populateLifeCycleMethods(Component c) {
478       if (!c.methodsScanned) {
479          c.methodsScanned = true;
480          Class<?> componentClass = c.instance.getClass();
481 
482          // START methods first
483          c.startMethods = processPrioritizedMethods(c.metadata.getStartMethods(), componentClass, c);
484 
485          // And now the STOP methods
486          c.stopMethods = processPrioritizedMethods(c.metadata.getStopMethods(), componentClass, c);
487       }
488    }

490    /**
491     * Removes any components not annotated as @SurvivesRestarts.
492     */
493    public synchronized void resetVolatileComponents() {
494       // destroy all components to clean up resources
495       getLog().tracef("Resetting volatile components");
496       for (Component c : new HashSet<Component>(componentLookup.values())) {
497          // the component is volatile!!
498          if (!c.metadata.isSurvivesRestarts()) {
499             getLog().tracef("Removing volatile component %s", c.name);
500             componentLookup.remove(c.name);
501          }
502       }
503 
504       if (getLog().isTraceEnabled())
505          getLog().tracef("Reset volatile components. Registry now contains %s", componentLookup.keySet());
506    }

511    /**
512     * This starts the components in the cache, connecting to channels, starting service threads, etc.  If the cache is
513     * not in the {@link org.infinispan.lifecycle.ComponentStatus#INITIALIZING} state, it will be initialized first.
514     */
515    @Override
516    public synchronized void start() {
517 
518       if (!state.startAllowed()) {
519          if (state.needToDestroyFailedCache())
520             destroy(); // this will take us back to TERMINATED
521 
522          if (state.needToInitializeBeforeStart()) {
523             rewire();
524          } else
525             return;
526       }
527 
528       state = ComponentStatus.INITIALIZING;
529       try {
530          internalStart();
531       } catch (Throwable t) {
532          handleLifecycleTransitionFailure(t);
533       }
534    }

536    /**
537     * Stops the cache and sets the cache status to {@link org.infinispan.lifecycle.ComponentStatus#TERMINATED} once it
538     * is done.  If the cache is not in the {@link org.infinispan.lifecycle.ComponentStatus#RUNNING} state, this is a
539     * no-op.
540     */
541    @Override
542    public synchronized void stop() {
543       if (!state.stopAllowed()) {
544          getLog().debugf("Ignoring call to stop() as current state is %s", state);
545          return;
546       }
547 
548       // Trying to stop() from FAILED is valid, but may not work
549       boolean failed = state == ComponentStatus.FAILED;
550 
551       try {
552          internalStop();
553       } catch (Throwable t) {
554          if (failed) {
555             getLog().failedToCallStopAfterFailure(t);
556          }
557          failed = true;
558          handleLifecycleTransitionFailure(t);
559       } finally {
560          if (!failed) state = ComponentStatus.TERMINATED;
561       }
562    }

564    /**
565     * Destroys the cache and frees up any resources.  Sets the cache status to {@link
566     * org.infinispan.lifecycle.ComponentStatus#TERMINATED} when it is done.
567     * <p/>
568     * If the cache is in {@link org.infinispan.lifecycle.ComponentStatus#RUNNING} when this method is called, it will
569     * first call {@link #stop()} to stop the cache.
570     */
571    private void destroy() {
572       try {
573          if (state.stopAllowed())
574             stop();
575       } catch (CacheException e) {
576          getLog().stopBeforeDestroyFailed(e);
577       }
578 
579       try {
580          resetVolatileComponents();
581       } finally {
582          // We always progress to destroyed
583          state = ComponentStatus.TERMINATED;
584       }
585    }

590    /**
591     * Sets the cacheStatus to FAILED and re-throws the problem as one of the declared types. Converts any
592     * non-RuntimeException Exception to CacheException.
593     *
594     * @param t throwable thrown during failure
595     */
596    private void handleLifecycleTransitionFailure(Throwable t) {
597       state = ComponentStatus.FAILED;
598       if (t.getCause() != null && t.getCause() instanceof CacheConfigurationException)
599          throw (CacheConfigurationException) t.getCause();
600       else if (t.getCause() != null && t.getCause() instanceof InvocationTargetException && t.getCause().getCause() != null && t.getCause().getCause() instanceof CacheConfigurationException)
601          throw (CacheConfigurationException) t.getCause().getCause();
602       else if (t instanceof CacheException)
603          throw (CacheException) t;
604       else if (t instanceof RuntimeException)
605          throw (RuntimeException) t;
606       else if (t instanceof Error)
607          throw (Error) t;
608       else
609          throw new CacheException(t);
610    }

612    private void internalStart() throws CacheException, IllegalArgumentException {
613       // start all internal components
614       // first cache all start, stop and destroy methods.
615       populateLifecycleMethods();
616 
617       List<PrioritizedMethod> startMethods = new ArrayList<PrioritizedMethod>(componentLookup.size());
618       for (Component c : componentLookup.values()) {
619          Collections.addAll(startMethods, c.startMethods);
620       }
621 
622       // sort the start methods by priority
623       Collections.sort(startMethods);
624 
625       // fire all START methods according to priority
626 
627       invokeStartMethods(startMethods);
628       addShutdownHook();
629 
630       state = ComponentStatus.RUNNING;
631    }
632 
633    private void invokeStartMethods(Collection<PrioritizedMethod> startMethods) {
634       boolean traceEnabled = getLog().isTraceEnabled();
635       for (PrioritizedMethod em : startMethods) {
636          if (traceEnabled)
637             getLog().tracef("Invoking start method %s on component %s", em.metadata.getMethod(), em.component.getName());
638          em.invoke();
639       }
640    }

650    /**
651     * Actual stop
652     */
653    private void internalStop() {
654       state = ComponentStatus.STOPPING;
655       removeShutdownHook();
656 
657       List<PrioritizedMethod> stopMethods = new ArrayList<PrioritizedMethod>(componentLookup.size());
658       for (Component c : componentLookup.values()) {
659          // if one of the components threw an exception during startup
660          // the stop methods list may not have been initialized
661          if (c.stopMethods != null) {
662             Collections.addAll(stopMethods, c.stopMethods);
663          }
664       }
665 
666       Collections.sort(stopMethods);
667 
668       // fire all STOP methods according to priority
669       boolean traceEnabled = getLog().isTraceEnabled();
670       for (PrioritizedMethod em : stopMethods) {
671          if (traceEnabled)
672             getLog().tracef("Invoking stop method %s on component %s", em.metadata.getMethod(), em.component.getName());
673          try {
674             em.invoke();
675          } catch (Throwable t) {
676             getLog().componentFailedToStop(t);
677          }
678       }
679 
680       destroy();
681    }

685    /**
686     * Asserts whether invocations are allowed on the cache or not.  Returns <tt>true</tt> if invocations are to be
687     * allowed, <tt>false</tt> otherwise.  If the origin of the call is remote and the cache status is {@link
688     * org.infinispan.lifecycle.ComponentStatus#INITIALIZING}, this method will block for up to {@link
689     * org.infinispan.configuration.cache.StateTransferConfiguration#timeout()} millis, checking for a valid state.
690     *
691     * @param originLocal true if the call originates locally (i.e., from the {@link org.infinispan.cache.impl.CacheImpl} or false
692     *                    if it originates remotely, i.e., from the {@link org.infinispan.remoting.InboundInvocationHandler}.
693     * @return true if invocations are allowed, false otherwise.
694     */
695    public boolean invocationsAllowed(boolean originLocal) {
696       getLog().trace("Testing if invocations are allowed.");
697       if (state.allowInvocations()) return true;
698 
699       // if this is a locally originating call and the cache is not in a valid state, return false.
700       if (originLocal) return false;
701 
702       getLog().trace("Is remotely originating.");
703 
704       // else if this is a remote call and the status is STARTING, wait until the cache starts.
705       if (state == ComponentStatus.INITIALIZING) {
706          getLog().trace("Cache is initializing; block.");
707          try {
708             blockUntilCacheStarts();
709             return true;
710          } catch (InterruptedException e) {
711             Thread.currentThread().interrupt();
712          }
713       } else {
714          getLog().cacheNotStarted();
715       }
716       return false;
717    }

719    /**
720     * Blocks until the current cache instance is in its {@link org.infinispan.lifecycle.ComponentStatus#RUNNING started}
721     * phase. Blocks for up to {@link org.infinispan.configuration.cache.StateTransferConfiguration#timeout()} milliseconds, throwing an
722     * IllegalStateException if the cache doesn't reach this state even after this maximum wait time.
723     *
724     * @throws InterruptedException  if interrupted while waiting
725     * @throws IllegalStateException if even after waiting the cache has not started.
726     */
727    private void blockUntilCacheStarts() throws InterruptedException, IllegalStateException {
728       int pollFrequencyMS = 20;
729       TimeService timeService = getTimeService();
730       final long startupWaitTime = getConfiguration().clustering().stateTransfer().timeout();
731       final long giveUpTime = timeService.expectedEndTime(startupWaitTime, TimeUnit.MILLISECONDS);
732 
733       while (!timeService.isTimeExpired(giveUpTime)) {
734          if (state.allowInvocations()) break;
735          Thread.sleep(pollFrequencyMS);
736       }
737 
738       // check if we have started.
739       if (!state.allowInvocations())
740          throw new IllegalStateException("Cache not in STARTED state, even after waiting " + startupWaitTime + " millis.");
741    }

762    /**
763     * A wrapper representing a component in the registry
764     */
765    public class Component {
766 
767       /**
768        * A reference to the object instance for this component.
769        */
770       Object instance;
771       /**
772        * The name of the component
773        */
774       String name;
775       boolean methodsScanned;
776       /**
777        * List of injection methods used to inject dependencies into the component
778        */
779       ComponentMetadata.InjectMetadata[] injectionMethods;
780       PrioritizedMethod[] startMethods;
781       PrioritizedMethod[] stopMethods;
782       ComponentMetadata metadata;
783 
784       @Override
785       public String toString() {
786          return "Component{" +
787                "instance=" + instance +
788                ", name=" + name +
789                '}';
790       }
791 
792       /**
793        * Injects dependencies into this component.
794        */
795       public void injectDependencies() {
796          if (injectionMethods != null && injectionMethods.length > 0) {
797             for (ComponentMetadata.InjectMetadata injectMetadata : injectionMethods) invokeInjectionMethod(instance, injectMetadata);
798          }
799       }
800 
801       public Object getInstance() {
802          return instance;
803       }
804 
805       public String getName() {
806          return name;
807       }
808 
809       public ComponentMetadata getMetadata() {
810          return metadata;
811       }
812 
813       public void buildInjectionMethodsList() throws ClassNotFoundException {
814          injectionMethods = metadata.getInjectMethods();
815          if (injectionMethods != null && injectionMethods.length > 0) {
816             Class<?> clazz = instance.getClass();
817             for (ComponentMetadata.InjectMetadata meta: injectionMethods) {
818                Class<?>[] parameterClasses = meta.getParameterClasses();
819                if (parameterClasses == null) {
820                   parameterClasses = ReflectionUtil.toClassArray(meta.getParameters(), getClassLoader());
821                   meta.setParameterClasses(parameterClasses);
822                }
823                Method m = meta.getMethod();
824                if (m == null) {
825                   m = ReflectionUtil.findMethod(clazz, meta.getMethodName(), parameterClasses);
826                   meta.setMethod(m);
827                }
828             }
829          }
830       }
831    }

833    /**
834     * Wrapper to encapsulate a method along with a priority
835     */
836    static class PrioritizedMethod implements Comparable<PrioritizedMethod> {
837       ComponentMetadata.PrioritizedMethodMetadata metadata;
838       Component component;
839 
840       @Override
841       public int compareTo(PrioritizedMethod o) {
842          int thisVal = metadata.getPriority();
843          int anotherVal = o.metadata.getPriority();
844          return (thisVal < anotherVal ? -1 : (thisVal == anotherVal ? 0 : 1));
845       }
846 
847 
848       @Override
849       public boolean equals(Object o) {
850          if (this == o) return true;
851          if (!(o instanceof PrioritizedMethod)) return false;
852 
853          PrioritizedMethod that = (PrioritizedMethod) o;
854 
855          if (component != null ? !component.equals(that.component) : that.component != null) return false;
856          if (metadata != null ? !metadata.equals(that.metadata) : that.metadata != null) return false;
857 
858          return true;
859       }
860 
861       @Override
862       public int hashCode() {
863          int result = metadata != null ? metadata.hashCode() : 0;
864          result = 31 * result + (component != null ? component.hashCode() : 0);
865          return result;
866       }
867 
868       void invoke() {
869          ReflectionUtil.invokeAccessibly(component.instance, metadata.getMethod(), null);
870       }
871 
872       @Override
873       public String toString() {
874          return "PrioritizedMethod{" +
875                "method=" + metadata.getMethod().getName() +
876                ", priority=" + metadata.getPriority() +
877                '}';
878       }
879 
880    }

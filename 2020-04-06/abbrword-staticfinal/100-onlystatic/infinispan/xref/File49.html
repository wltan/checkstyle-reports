1   package org.infinispan.cli.shell;
2   
3   import java.io.BufferedReader;
4   import java.io.BufferedWriter;
5   import java.io.File;
6   import java.io.FileReader;
7   import java.io.FileWriter;
8   import java.io.IOException;
9   import java.io.Reader;
10  import java.io.Writer;
11  import java.util.Properties;
12  
13  import org.infinispan.cli.Config;
14  
15  /**
16   * ConfigImpl.
17   *
18   * @author Tristan Tarrant
19   * @since 5.2
20   */
21  public class ConfigImpl implements Config {
22     private static final String CONFIG_FILE = "ispn-cli.ini";
23     private final String configHome;
24     private boolean colorsEnabled = true;
25     private boolean historyEnabled = true;
26     private String prompt = "[\\c{green}$CONNECTION\\c{yellow}/$CONTAINER\\c{yellow}/$CACHE\\c]> ";
27  
28     public ConfigImpl(String configHome) {
29        this.configHome = configHome;
30     }
31  
32     @Override
33     public void load() {
34        Properties p = new Properties();
35        File configFile = new File(configHome, CONFIG_FILE);
36        if (configFile.exists()) {
37           try {
38              Reader r = new BufferedReader(new FileReader(configFile));
39              p.load(r);
40              r.close();
41              if (p.containsKey("colors")) colorsEnabled = Boolean.parseBoolean(p.getProperty("colors"));
42              if (p.containsKey("history")) historyEnabled = Boolean.parseBoolean(p.getProperty("history"));
43              if (p.containsKey("prompt")) prompt = p.getProperty("prompt");
44           } catch (IOException e) {
45              //FIXME implement me
46           }
47        }
48     }
49  
50     @Override
51     public void save() {
52        Properties p = new Properties();
53        p.setProperty("colors", String.valueOf(colorsEnabled));
54        p.setProperty("history", String.valueOf(historyEnabled));
55        p.setProperty("prompt", prompt);
56        File configDir = new File(configHome);
57        File configFile = new File(configHome, CONFIG_FILE);
58        try {
59           if (!configDir.exists()) {
60              configDir.mkdirs();
61           }
62           Writer w = new BufferedWriter(new FileWriter(configFile));
63           p.store(w, null);
64           w.close();
65        } catch (IOException e) {
66           //FIXME implement me
67        }
68     }
69  
70     @Override
71     public boolean isColorEnabled() {
72        return colorsEnabled;
73     }
74  
75     @Override
76     public boolean isHistoryEnabled() {
77        return historyEnabled;
78     }
79  
80     @Override
81     public String getPrompt() {
82        return prompt;
83     }
84  }

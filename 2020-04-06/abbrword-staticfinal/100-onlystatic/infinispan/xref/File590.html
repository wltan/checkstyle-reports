22  /**
23   * The recursive index structure. References to children are held in soft references,
24   * which allows JVM-handled caching and reduces the amount of reads required while
25   * evading OOMs if the index gets too big.
26   *
27   * @author Radim Vansa &lt;rvansa@redhat.com&gt;
28   */
29  class IndexNode {
30     private static final Log log = LogFactory.getLog(IndexNode.class);
31     private static final boolean trace = log.isTraceEnabled();
32  
33     private static final byte HAS_LEAVES = 1;
34     private static final int INNER_NODE_HEADER_SIZE = 5;
35     private static final int INNER_NODE_REFERENCE_SIZE = 10;
36     private static final int LEAF_NODE_REFERENCE_SIZE = 8;
37  
38     public static final int RESERVED_SPACE
39           = INNER_NODE_HEADER_SIZE + 2 * Math.max(INNER_NODE_REFERENCE_SIZE, LEAF_NODE_REFERENCE_SIZE);
40  
41     private Index.Segment segment;
42     private byte[] prefix;
43     private byte[][] keyParts;
44     InnerNode[] innerNodes;
45     private LeafNode[] leafNodes;
46     private ReadWriteLock lock = new ReentrantReadWriteLock();
47     private long offset = -1;
48     private int contentLength = -1;
49     private int totalLength = -1;
50     private int occupiedSpace;
51  
52     public IndexNode(Index.Segment segment, long offset, int occupiedSpace) throws IOException {
53        this.segment = segment;
54        this.offset = offset;
55        this.occupiedSpace = occupiedSpace;
56  
57        ByteBuffer buffer = loadBuffer(segment.getIndexFile(), offset, occupiedSpace);
58  
59        prefix = new byte[buffer.getShort()];
60        buffer.get(prefix);
61  
62        byte flags = buffer.get();
63        int numKeyParts = buffer.getShort();
64  
65        keyParts = new byte[numKeyParts][];
66        for (int i = 0; i < numKeyParts; ++i) {
67           keyParts[i] = new byte[buffer.getShort()];
68           buffer.get(keyParts[i]);
69        }
70  
71        if ((flags & HAS_LEAVES) != 0) {
72           leafNodes = new LeafNode[numKeyParts + 1];
73           for (int i = 0; i < numKeyParts + 1; ++i) {
74              leafNodes[i] = new LeafNode(buffer.getInt(), buffer.getInt());
75           }
76        } else {
77           innerNodes = new InnerNode[numKeyParts + 1];
78           for (int i = 0; i < numKeyParts + 1; ++i) {
79              innerNodes[i] = new InnerNode(buffer.getLong(), buffer.getShort());
80           }
81        }
82  
83        if (trace) log.tracef("Loaded %08x from %d:%d (length %d)", System.identityHashCode(this), offset, occupiedSpace, length());
84     }
85  
86     private static ByteBuffer loadBuffer(FileChannel indexFile, long offset, int occupiedSpace) throws IOException {
87        ByteBuffer buffer = ByteBuffer.allocate(occupiedSpace);
88        int read = 0;
89        do {
90           int nowRead = indexFile.read(buffer, offset + read);
91           if (nowRead < 0) {
92              throw new IOException("Cannot read record [" + offset + ":" + occupiedSpace + "] (already read "
93                    + read + "), file size is " + indexFile.size());
94           }
95           read += nowRead;
96        } while (read < occupiedSpace);
97        buffer.rewind();
98        return buffer;
99     }

115    @Override
116    public boolean equals(Object o) {
117       if (this == o) return true;
118       if (o == null || getClass() != o.getClass()) return false;
119 
120       IndexNode indexNode = (IndexNode) o;
121 
122       if (!Arrays.equals(innerNodes, indexNode.innerNodes)) return false;
123       if (!Arrays.equals(leafNodes, indexNode.leafNodes)) return false;
124       if (!Arrays.equals(prefix, indexNode.prefix)) return false;
125       if (!Arrays.deepEquals(keyParts, indexNode.keyParts)) return false;
126 
127       return true;
128    }

130    /**
131     * Can be called only from single writer thread (therefore the write lock guards only other readers)
132     * @param other
133     */
134    public void replaceContent(IndexNode other) throws IOException {
135       try {
136          lock.writeLock().lock();
137          this.prefix = other.prefix;
138          this.keyParts = other.keyParts;
139          this.innerNodes = other.innerNodes;
140          this.leafNodes = other.leafNodes;
141          this.contentLength = -1;
142          this.totalLength = -1;
143       } finally {
144          lock.writeLock().unlock();
145       }
146 
147       // don't have to acquire any lock here
148       // the only node with offset < 0 is the root - we can't lose reference to it
149       if (offset >= 0) {
150          store(new Index.IndexSpace(offset, occupiedSpace));
151       }
152    }
153 
154    private void store(Index.IndexSpace indexSpace) throws IOException {
155       this.offset = indexSpace.offset;
156       this.occupiedSpace = indexSpace.length;
157       ByteBuffer buffer = ByteBuffer.allocate(length());
158       buffer.putShort((short) prefix.length);
159       buffer.put(prefix);
160       buffer.put(innerNodes == null ? HAS_LEAVES : 0);
161       buffer.putShort((short) keyParts.length);
162       for (int i = 0; i < keyParts.length; ++i) {
163          buffer.putShort((short) keyParts[i].length);
164          buffer.put(keyParts[i]);
165       }
166       if (innerNodes != null) {
167          for (int i = 0; i < innerNodes.length; ++i) {
168             buffer.putLong(innerNodes[i].offset);
169             buffer.putShort((short) innerNodes[i].length);
170          }
171       } else {
172          for (int i = 0; i < leafNodes.length; ++i) {
173             buffer.putInt(leafNodes[i].file);
174             buffer.putInt(leafNodes[i].offset);
175          }
176       }
177       buffer.flip();
178       segment.getIndexFile().write(buffer, offset);
179 
180       if (trace) log.tracef("Persisted %08x (length %d, %d %s) to %d:%d", System.identityHashCode(this), length(),
181             innerNodes != null ? innerNodes.length : leafNodes.length,
182             innerNodes != null ? "children" : "leaves", offset, occupiedSpace);
183    }

195    public enum ReadOperation {
196       GET_RECORD {
197          @Override
198          protected EntryRecord apply(LeafNode leafNode, byte[] key, FileProvider fileProvider, TimeService timeService) throws IOException, IndexNodeOutdatedException {
199             return leafNode.loadRecord(fileProvider, key, timeService);
200          }
201       },
202       GET_POSITION {
203          @Override
204          protected EntryPosition apply(LeafNode leafNode, byte[] key, FileProvider fileProvider, TimeService timeService) throws IOException, IndexNodeOutdatedException {
205             EntryRecord hak = leafNode.loadHeaderAndKey(fileProvider, timeService);
206             if (hak != null && hak.getKey() != null && Arrays.equals(hak.getKey(), key)) {
207                return leafNode;
208             }
209             return null;
210          }
211       };
212 
213       protected abstract <T> T apply(LeafNode leafNode, byte[] key, FileProvider fileProvider, TimeService timeService) throws IOException, IndexNodeOutdatedException;
214    }

216    public static <T> T applyOnLeaf(Index.Segment segment, byte[] key, Lock rootLock, ReadOperation operation) throws IOException {
217       int attempts = 0;
218       ArrayList<IndexNode> path = new ArrayList<IndexNode>();
219       for (;;) {
220          rootLock.lock();
221          IndexNode node = segment.getRoot();
222          Lock parentLock = rootLock, currentLock = null;
223          try {
224             while (node.innerNodes != null) {
225                path.add(node);
226                currentLock = node.lock.readLock();
227                currentLock.lock();
228                if (parentLock != null) {
229                   parentLock.unlock();
230                }
231                parentLock = currentLock;
232                int insertionPoint = node.getInsertionPoint(key);
233                node = node.innerNodes[insertionPoint].getIndexNode(segment);
234                if (node == null) {
235                   return null;
236                }
237             }
238             currentLock = node.lock.readLock();
239             currentLock.lock();
240             if (node.leafNodes.length == 0) {
241                return null;
242             }
243             int insertionPoint = node.getInsertionPoint(key);
244             return operation.apply(node.leafNodes[insertionPoint], key, segment.getFileProvider(), segment.getTimeService());
245          } catch (IndexNodeOutdatedException e) {
246             try {
247                if (attempts > 10) {
248                   throw new PersistenceException("Index looks corrupt", e);
249                }
250                Thread.sleep(1000);
251                attempts++;
252                path.clear();
253             } catch (InterruptedException e1) {
254             }
255             // noop, we'll simply retry
256          } finally {
257             if (parentLock != currentLock && parentLock != null) parentLock.unlock();
258             if (currentLock != null) currentLock.unlock();
259          }
260       }
261    }

263    public static void setPosition(IndexNode root, byte[] key, int file, int offset, int size, OverwriteHook overwriteHook) throws IOException {
264       IndexNode node = root;
265       Stack<Path> stack = new Stack<Path>();
266       while (node.innerNodes != null) {
267          int insertionPoint = node.getInsertionPoint(key);
268          stack.push(new Path(node, insertionPoint));
269          if (trace) log.tracef("Pushed %08x (length %d, %d children) to stack (insertion point %d)", System.identityHashCode(node), node.length(), node.innerNodes.length, insertionPoint);
270          node = node.innerNodes[insertionPoint].getIndexNode(root.segment);
271       }
272       IndexNode copy = node.copyWith(key, file, offset, size, overwriteHook);
273       if (copy == node) {
274          // no change was executed
275          return;
276       }
277       if (trace) {
278          log.tracef("Created %08x (length %d) from %08x (length %d), stack size %d",
279                System.identityHashCode(copy), copy.length(), System.identityHashCode(node), node.length(), stack.size());
280       }
281 
282       Stack<IndexNode> garbage = new Stack<IndexNode>();
283       try {
284          JoinSplitResult result = manageLength(root.segment, stack, node, copy, garbage);
285          if (result == null) {
286             return;
287          }
288          if (trace)
289             log.tracef("Created (1) %d new nodes, GC %08x", result.newNodes.size(), System.identityHashCode(node));
290          garbage.push(node);
291          for (;;) {
292             if (stack.isEmpty()) {
293                IndexNode newRoot;
294                if (result.newNodes.size() == 1) {
295                   newRoot = result.newNodes.get(0);
296                   if (trace) log.tracef("Setting new root %08x (index has shrunk)", System.identityHashCode(newRoot));
297                } else {
298                   newRoot = IndexNode.emptyWithInnerNodes(root.segment).copyWith(0, 0, result.newNodes);
299                   if (trace) log.tracef("Setting new root %08x (index has grown)", System.identityHashCode(newRoot));
300                }
301                newRoot.segment.setRoot(newRoot);
302                return;
303             }
304             Path path = stack.pop();
305             copy = path.node.copyWith(result.from, result.to, result.newNodes);
306             if (trace) {
307                log.tracef("Created %08x (length %d) from %08x with the %d new nodes (%d - %d)",
308                      System.identityHashCode(copy), copy.length(), System.identityHashCode(path.node), result.newNodes.size(), result.from, result.to);
309             }
310             result = manageLength(path.node.segment, stack, path.node, copy, garbage);
311             if (result == null) {
312                if (trace) log.tracef("No more index updates required");
313                return;
314             }
315             if (trace)
316                log.tracef("Created (2) %d new nodes, GC %08x", result.newNodes.size(), System.identityHashCode(path.node));
317             garbage.push(path.node);
318          }
319       } finally {
320          while (!garbage.isEmpty()) {
321             IndexNode oldNode = garbage.pop();
322             // this will be never unlocked, if the algorithm is correct, this node should be GC'ed soon.
323             oldNode.lock.writeLock().lock();
324             if (oldNode.offset >= 0) {
325                oldNode.segment.freeIndexSpace(oldNode.offset, oldNode.occupiedSpace);
326             }
327          }
328       }
329    }

343    private static JoinSplitResult manageLength(Index.Segment segment, Stack<Path> stack, IndexNode node, IndexNode copy, Stack<IndexNode> garbage) throws IOException {
344       int from, to;
345       if (copy.length() < segment.getMinNodeSize() && !stack.isEmpty()) {
346          Path parent = stack.peek();
347          if (parent.node.innerNodes.length == 1) {
348             // we have no siblings - we can't merge with them even when we're really short
349             if (copy.length() <= node.occupiedSpace) {
350                node.replaceContent(copy);
351                return null;
352             } else {
353                return new JoinSplitResult(parent.index, parent.index, Collections.singletonList(copy));
354             }
355          }
356          int sizeWithLeft = Integer.MAX_VALUE;
357          int sizeWithRight = Integer.MAX_VALUE;
358          if (parent.index > 0) {
359             sizeWithLeft = copy.length() + parent.node.innerNodes[parent.index - 1].length - INNER_NODE_HEADER_SIZE;
360          }
361          if (parent.index < parent.node.innerNodes.length - 1) {
362             sizeWithRight = copy.length() + parent.node.innerNodes[parent.index + 1].length - INNER_NODE_HEADER_SIZE;
363          }
364          int joinWith;
365          // this is just some kind of heuristic, may be changed later
366          if (sizeWithLeft == Integer.MAX_VALUE) {
367             joinWith = parent.index + 1;
368          } else if (sizeWithRight == Integer.MAX_VALUE) {
369             joinWith = parent.index - 1;
370          } else if (sizeWithLeft > segment.getMaxNodeSize() && sizeWithRight > segment.getMaxNodeSize()) {
371             joinWith = sizeWithLeft >= sizeWithRight ? parent.index - 1 : parent.index + 1;
372          } else {
373             joinWith = sizeWithLeft <= sizeWithRight ? parent.index - 1 : parent.index + 1;
374          }
375          if (joinWith < 0 || joinWith >= parent.node.innerNodes.length) {
376             throw new IllegalStateException(String.format("parent %08x, %08x -> %08x: cannot join to %d, with left %d, with right %d, max %d",
377                   System.identityHashCode(parent.node), System.identityHashCode(node), System.identityHashCode(copy),
378                   joinWith, sizeWithLeft, sizeWithRight, segment.getMaxNodeSize()));
379          }
380          IndexNode joiner = parent.node.innerNodes[joinWith].getIndexNode(segment);
381          byte[] middleKey = concat(parent.node.prefix, parent.node.keyParts[joinWith < parent.index ? parent.index - 1 : parent.index]);
382          if (joinWith < parent.index) {
383             copy = join(joiner, middleKey, copy);
384             from = joinWith;
385             to = parent.index;
386          } else {
387             copy = join(copy, middleKey, joiner);
388             from = parent.index;
389             to = joinWith;
390          }
391          garbage.push(joiner);
392       } else if (copy.length() <= node.occupiedSpace) {
393          if (copy.innerNodes != null && copy.innerNodes.length == 1 && stack.isEmpty()) {
394             IndexNode child = copy.innerNodes[0].getIndexNode(copy.segment);
395             return new JoinSplitResult(0, 0, Collections.singletonList(child));
396          } else {
397             // special case where we only overwrite the key
398             node.replaceContent(copy);
399             return null;
400          }
401       } else if (stack.isEmpty()) {
402          from = to = 0;
403       } else {
404          from = to = stack.peek().index;
405       }
406       if (copy.length() <= segment.getMaxNodeSize()) {
407          return new JoinSplitResult(from, to, Collections.singletonList(copy));
408       } else {
409          return new JoinSplitResult(from, to, copy.split());
410       }
411    }

413    private static IndexNode join(IndexNode left, byte[] middleKey, IndexNode right) throws IOException {
414       byte[] newPrefix = commonPrefix(left.prefix, right.prefix);
415       byte[][] newKeyParts = new byte[left.keyParts.length + right.keyParts.length + 1][];
416       newPrefix = commonPrefix(newPrefix == null ? left.prefix : newPrefix, middleKey);
417       copyKeyParts(left.keyParts, 0, newKeyParts, 0, left.keyParts.length, left.prefix, newPrefix);
418       byte[] rightmostKey;
419       try {
420          rightmostKey = left.rightmostKey();
421       } catch (IndexNodeOutdatedException e) {
422          throw new IllegalStateException(e);
423       }
424       int commonLength = Math.abs(compare(middleKey, rightmostKey));
425       newKeyParts[left.keyParts.length] = substring(middleKey, newPrefix.length, commonLength);
426       copyKeyParts(right.keyParts, 0, newKeyParts, left.keyParts.length + 1, right.keyParts.length, right.prefix, newPrefix);
427       if (left.innerNodes != null && right.innerNodes != null) {
428          InnerNode[] newInnerNodes = new InnerNode[left.innerNodes.length + right.innerNodes.length];
429          System.arraycopy(left.innerNodes, 0, newInnerNodes, 0, left.innerNodes.length);
430          System.arraycopy(right.innerNodes, 0, newInnerNodes, left.innerNodes.length, right.innerNodes.length);
431          return new IndexNode(left.segment, newPrefix, newKeyParts, newInnerNodes);
432       } else if (left.leafNodes != null && right.leafNodes != null) {
433          LeafNode[] newLeafNodes = new LeafNode[left.leafNodes.length + right.leafNodes.length];
434          System.arraycopy(left.leafNodes, 0, newLeafNodes, 0, left.leafNodes.length);
435          System.arraycopy(right.leafNodes, 0, newLeafNodes, left.leafNodes.length, right.leafNodes.length);
436          return new IndexNode(left.segment, newPrefix, newKeyParts, newLeafNodes);
437       } else {
438          throw new IllegalArgumentException("Cannot join " + left + " and " + right);
439       }
440    }

442    public IndexNode copyWith(int oldNodesFrom, int oldNodesTo, List<IndexNode> newNodes) throws IOException {
443       InnerNode[] newInnerNodes = new InnerNode[innerNodes.length + newNodes.size() - 1 - oldNodesTo + oldNodesFrom];
444       System.arraycopy(innerNodes, 0, newInnerNodes, 0, oldNodesFrom);
445       System.arraycopy(innerNodes, oldNodesTo + 1, newInnerNodes, oldNodesFrom + newNodes.size(), innerNodes.length - oldNodesTo - 1);
446       for (int i = 0; i < newNodes.size(); ++i) {
447          IndexNode node = newNodes.get(i);
448          Index.IndexSpace space = segment.allocateIndexSpace(node.length());
449          node.store(space);
450          newInnerNodes[i + oldNodesFrom] = new InnerNode(node);
451       }
452       byte[][] newKeys = new byte[newNodes.size() - 1][];
453       byte[] newPrefix = prefix;
454       for (int i = 0; i < newKeys.length; ++i) {
455          try {
456             // TODO: if all keys within the subtree are null (deleted), the new key will be null
457             // will be fixed with proper index reduction
458             newKeys[i] = newNodes.get(i + 1).leftmostKey();
459             if (newKeys[i] == null) {
460                throw new IllegalStateException();
461             }
462          } catch (IndexNodeOutdatedException e) {
463             throw new IllegalStateException("Index cannot be outdated for segment updater thread", e);
464          }
465          newPrefix = commonPrefix(newPrefix, newKeys[i]);
466       }
467       byte[][] newKeyParts = new byte[keyParts.length + newNodes.size() - 1 - oldNodesTo + oldNodesFrom][];
468       copyKeyParts(keyParts, 0, newKeyParts, 0, oldNodesFrom, prefix, newPrefix);
469       copyKeyParts(keyParts, oldNodesTo, newKeyParts, oldNodesFrom + newKeys.length, keyParts.length - oldNodesTo, prefix, newPrefix);
470       for (int i = 0; i < newKeys.length; ++i) {
471          newKeyParts[i + oldNodesFrom] = substring(newKeys[i], newPrefix.length, newKeys[i].length);
472       }
473       return new IndexNode(segment, newPrefix, newKeyParts, newInnerNodes);
474    }

476    private byte[] leftmostKey() throws IOException, IndexNodeOutdatedException {
477       if (innerNodes != null) {
478          for (int i = 0; i < innerNodes.length; ++i) {
479             byte[] key = innerNodes[i].getIndexNode(segment).leftmostKey();
480             if (key != null) return key;
481          }
482       } else {
483          for (int i = 0; i < leafNodes.length; ++i) {
484             EntryRecord hak = leafNodes[i].loadHeaderAndKey(segment.getFileProvider(), segment.getTimeService());
485             if (hak != null && hak.getKey() != null) return hak.getKey();
486          }
487       }
488       return null;
489    }
490 
491    private byte[] rightmostKey() throws IOException, IndexNodeOutdatedException {
492       if (innerNodes != null) {
493          for (int i = innerNodes.length - 1; i >= 0; --i) {
494             byte[] key = innerNodes[i].getIndexNode(segment).rightmostKey();
495             if (key != null) return key;
496          }
497       } else {
498          for (int i = leafNodes.length - 1; i >= 0; --i) {
499             EntryRecord hak = leafNodes[i].loadHeaderAndKey(segment.getFileProvider(), segment.getTimeService());
500             if (hak != null && hak.getKey() != null) return hak.getKey();
501          }
502       }
503       return null;
504    }

506    /**
507     * Called on the most bottom node
508     * @param key
509     * @param file
510     * @param offset
511     * @return
512     */
513    public IndexNode copyWith(byte[] key, int file, int offset, int size, OverwriteHook overwriteHook) throws IOException {
514       if (leafNodes == null) throw new IllegalArgumentException();
515       byte[] newPrefix;
516       byte[][] newKeyParts;
517       LeafNode[] newLeafNodes;
518       if (leafNodes.length == 0) {
519          overwriteHook.setOverwritten(false, -1, -1);
520          if (overwriteHook.check(-1, -1)) {
521             return new IndexNode(segment, prefix, keyParts, new LeafNode[]{ new LeafNode(file, offset)});
522          } else {
523             segment.getCompactor().free(file, size);
524             return this;
525          }
526       }
527 
528       int insertPart = getInsertionPoint(key);
529       EntryRecord hak = null;
530       try {
531          hak = leafNodes[insertPart].loadHeaderAndKey(segment.getFileProvider(), segment.getTimeService());
532       } catch (IndexNodeOutdatedException e) {
533          throw new IllegalStateException("Index cannot be outdated for segment updater thread", e);
534       }
535       int keyComp = Integer.MAX_VALUE;
536       if (hak == null || hak.getKey() == null || (keyComp = compare(hak.getKey(), key)) == 0) {
537          if (offset >= 0) {
538             if (overwriteHook.check(leafNodes[insertPart].file, leafNodes[insertPart].offset)) {
539                newPrefix = prefix;
540                newKeyParts = keyParts;
541                newLeafNodes = new LeafNode[leafNodes.length];
542                System.arraycopy(leafNodes, 0, newLeafNodes, 0, leafNodes.length);
543                if (trace) {
544                   log.trace(String.format("Overwriting %d:%d (%s) with %d:%d",
545                         leafNodes[insertPart].file, leafNodes[insertPart].offset,
546                         hak == null ? "removed" : (hak.getKey() == null ? "expired" : "matching"), file, offset));
547                }
548                newLeafNodes[insertPart] = new LeafNode(file, offset);
549                if (hak != null) {
550                   segment.getCompactor().free(leafNodes[insertPart].file, hak.getHeader().totalLength());
551                }
552                overwriteHook.setOverwritten(true, leafNodes[insertPart].file, leafNodes[insertPart].offset);
553             } else {
554                overwriteHook.setOverwritten(false, -1, -1);
555                segment.getCompactor().free(file, size);
556                return this;
557             }
558          } else {
559             overwriteHook.setOverwritten(keyComp == 0, leafNodes[insertPart].file, leafNodes[insertPart].offset);
560             if (keyParts.length <= 1) {
561                newPrefix = new byte[0];
562                newKeyParts = new byte[0][];
563             } else {
564                newPrefix = prefix;
565                newKeyParts = new byte[keyParts.length - 1][];
566                if (insertPart == keyParts.length) {
567                   System.arraycopy(keyParts, 0, newKeyParts, 0, newKeyParts.length);
568                } else {
569                   System.arraycopy(keyParts, 0, newKeyParts, 0, insertPart);
570                   System.arraycopy(keyParts, insertPart + 1, newKeyParts, insertPart, newKeyParts.length - insertPart);
571                }
572             }
573             if (leafNodes.length > 0) {
574                newLeafNodes = new LeafNode[leafNodes.length - 1];
575                System.arraycopy(leafNodes, 0, newLeafNodes, 0, insertPart);
576                System.arraycopy(leafNodes, insertPart + 1, newLeafNodes, insertPart, newLeafNodes.length - insertPart);
577             } else {
578                newLeafNodes = leafNodes;
579             }
580             if (hak != null) {
581                segment.getCompactor().free(leafNodes[insertPart].file, hak.getHeader().totalLength());
582             }
583          }
584       } else {
585          overwriteHook.setOverwritten(false, -1, -1);
586          if (offset < 0) {
587             // this is a remove request, nothing to do
588             return this;
589          }
590          if (keyParts.length == 0) {
591             // TODO: we may use unnecessarily long keys here and the key is never shortened
592             newPrefix = keyComp > 0 ? key : hak.getKey();
593          } else {
594             newPrefix = commonPrefix(prefix, key);
595          }
596          newKeyParts = new byte[keyParts.length + 1][];
597          newLeafNodes = new LeafNode[leafNodes.length + 1];
598          copyKeyParts(keyParts, 0, newKeyParts, 0, insertPart, prefix, newPrefix);
599          copyKeyParts(keyParts, insertPart, newKeyParts, insertPart + 1, keyParts.length - insertPart, prefix, newPrefix);
600          if (keyComp > 0) {
601             newKeyParts[insertPart] = substring(key, newPrefix.length, keyComp);
602             System.arraycopy(leafNodes, 0, newLeafNodes, 0, insertPart + 1);
603             System.arraycopy(leafNodes, insertPart + 1, newLeafNodes, insertPart + 2, leafNodes.length - insertPart - 1);
604             newLeafNodes[insertPart + 1] = new LeafNode(file, offset);
605          } else {
606             newKeyParts[insertPart] = substring(hak.getKey(), newPrefix.length, -keyComp);
607             System.arraycopy(leafNodes, 0, newLeafNodes, 0, insertPart);
608             System.arraycopy(leafNodes, insertPart, newLeafNodes, insertPart + 1, leafNodes.length - insertPart);
609             newLeafNodes[insertPart] = new LeafNode(file, offset);
610          }
611       }
612       return new IndexNode(segment, newPrefix, newKeyParts, newLeafNodes);
613    }

615    private int getInsertionPoint(byte[] key) {
616       int comp = compare(prefix, key, prefix.length);
617       int insertionPoint;
618       if (comp < 0) {
619          insertionPoint = 0;
620       } else if (comp > 0) {
621          insertionPoint = keyParts.length;
622       } else {
623          byte[] keyPostfix = substring(key, prefix.length, key.length);
624          insertionPoint = Arrays.binarySearch(keyParts, keyPostfix, new Comparator<byte[]>() {
625             @Override
626             public int compare(byte[] o1, byte[] o2) {
627                return IndexNode.this.compare(o2, o1);
628             }
629          });
630          if (insertionPoint < 0) {
631             insertionPoint = -insertionPoint - 1;
632          } else {
633             insertionPoint++; // identical elements must go to the right
634          }
635       }
636       return insertionPoint;
637    }

639    private List<IndexNode> split() {
640       int headerLength = headerLength();
641       int contentLength = contentLength();
642       int maxLength = segment.getMaxNodeSize();
643       int targetParts = contentLength / Math.max(maxLength - headerLength, 1) + 1;
644       int targetLength = contentLength / targetParts + headerLength;
645       List<IndexNode> list = new ArrayList<IndexNode>();
646       int childLength = innerNodes != null ? INNER_NODE_REFERENCE_SIZE : LEAF_NODE_REFERENCE_SIZE;
647       byte[] prefixExtension = keyParts[0]; // the prefix can be only extended
648       int currentLength = INNER_NODE_HEADER_SIZE + prefix.length + prefixExtension.length + 2 * childLength + 2;
649       int nodeFrom = 0;
650       // TODO: under certain circumstances this algorithm can end up by splitting node into very uneven parts
651       // such as having one part with only 1 child, therefore only 15 bytes long
652       for (int i = 1; i < keyParts.length; ++i) {
653          int newLength;
654          byte[] newPrefixExtension = commonPrefix(prefixExtension, keyParts[i]);
655          if (newPrefixExtension.length != prefixExtension.length) {
656             newLength = currentLength + (prefixExtension.length - newPrefixExtension.length) * (i - nodeFrom - 1);
657          } else {
658             newLength = currentLength;
659          }
660          newLength += keyParts[i].length - newPrefixExtension.length + childLength + 2;
661          if (newLength < targetLength) {
662             currentLength = newLength;
663          } else {
664             IndexNode subNode;
665             if (newLength > maxLength) {
666                subNode = subNode(prefixExtension, nodeFrom, i);
667                ++i;
668             } else {
669                subNode = subNode(newPrefixExtension, nodeFrom, i + 1);
670                i += 2;
671             }
672             list.add(subNode);
673             if (i < keyParts.length) {
674                newPrefixExtension = keyParts[i];
675             }
676             currentLength = INNER_NODE_HEADER_SIZE + prefix.length + newPrefixExtension.length + 2 * childLength + 2;
677             nodeFrom = i;
678          }
679          prefixExtension = newPrefixExtension;
680       }
681       if (nodeFrom <= keyParts.length) {
682          list.add(subNode(prefixExtension, nodeFrom, keyParts.length));
683       }
684       return list;
685    }

687    private IndexNode subNode(byte[] newPrefixExtension, int childFrom, int childTo) {
688       // first node takes up to child[to + 1], other do not take the child[from] == child[previousTo + 1]
689       // If the new node has > 1 keyParts, it ignores the first keyPart, otherwise it just sets the first child to be
690       // deleted (empty entry)
691       byte[][] newKeyParts = new byte[childTo - childFrom][];
692       if (newPrefixExtension.length > 0) {
693          for (int i = childFrom; i < childTo; ++i) {
694             newKeyParts[i - childFrom] = substring(keyParts[i], newPrefixExtension.length, keyParts[i].length);
695          }
696       } else {
697          System.arraycopy(keyParts, childFrom, newKeyParts, 0, childTo - childFrom);
698       }
699       byte[] newPrefix = childFrom == childTo ? new byte[0] : concat(prefix, newPrefixExtension);
700       if (innerNodes != null) {
701          InnerNode[] newInnerNodes = new InnerNode[childTo - childFrom + 1];
702          System.arraycopy(innerNodes, childFrom, newInnerNodes, 0, childTo - childFrom + 1);
703          return new IndexNode(segment, newPrefix, newKeyParts, newInnerNodes);
704       } else if (leafNodes != null) {
705          LeafNode[] newLeafNodes = new LeafNode[childTo - childFrom + 1];
706          System.arraycopy(leafNodes, childFrom, newLeafNodes, 0, childTo - childFrom + 1);
707          return new IndexNode(segment, newPrefix, newKeyParts, newLeafNodes);
708       }
709       throw new IllegalStateException();
710    }

712    private static byte[] concat(byte[] first, byte[] second) {
713       if (first == null || first.length == 0) return second;
714       if (second == null || second.length == 0) return first;
715       byte[] result = new byte[first.length + second.length];
716       System.arraycopy(first, 0, result, 0, first.length);
717       System.arraycopy(second, 0, result, first.length, second.length);
718       return result;
719    }

721    private static void copyKeyParts(byte[][] src, int srcIndex, byte[][] dest, int destIndex, int length, byte[] oldPrefix, byte[] common) {
722       if (oldPrefix.length == common.length) {
723          System.arraycopy(src, srcIndex, dest, destIndex, length);
724       } else {
725          for (int i = 0; i < length; ++i) {
726             dest[destIndex + i] = findNewKeyPart(oldPrefix, src[srcIndex + i], common);
727          }
728       }
729    }

731    private static byte[] findNewKeyPart(byte[] oldPrefix, byte[] oldKeyPart, byte[] common) {
732       byte[] newPart = new byte[oldKeyPart.length + oldPrefix.length - common.length];
733       System.arraycopy(oldPrefix, common.length, newPart, 0, oldPrefix.length - common.length);
734       System.arraycopy(oldKeyPart, 0, newPart, oldPrefix.length - common.length, oldKeyPart.length);
735       return newPart;
736    }

738    private static byte[] substring(byte[] key, int begin, int end) {
739       if (end <= begin) return new byte[0];
740       byte[] sub = new byte[end - begin];
741       System.arraycopy(key, begin, sub, 0, end - begin);
742       return sub;
743    }

745    private static byte[] commonPrefix(byte[] oldPrefix, byte[] newKey) {
746       int i;
747       for (i = 0; i < oldPrefix.length && i < newKey.length; ++i) {
748          if (newKey[i] != oldPrefix[i]) break;
749       }
750       if (i == oldPrefix.length) {
751          return oldPrefix;
752       }
753       if (i == newKey.length) {
754          return newKey;
755       }
756       byte[] prefix = new byte[i];
757       for (--i; i >= 0; --i) {
758          prefix[i] = oldPrefix[i];
759       }
760       return prefix;
761    }

763    private static int compare(byte[] first, byte[] second, int length) {
764       for (int i = 0; i < length; ++i) {
765          if (i >= second.length) {
766             return -1;
767          }
768          if (second[i] == first[i]) continue;
769          return second[i] > first[i] ? 1 : -1;
770       }
771       return 0;
772    }
773 
774    private static int compare(byte[] first, byte[] second) {
775       for (int i = 0; i < first.length && i < second.length; ++i) {
776          if (second[i] == first[i]) continue;
777          return second[i] > first[i] ? i + 1 : -i - 1;
778       }
779       return second.length > first.length ? first.length + 1 : (second.length < first.length ? -second.length - 1 : 0);
780    }

782    private int headerLength() {
783       return INNER_NODE_HEADER_SIZE + prefix.length;
784    }

786    private int contentLength() {
787       if (contentLength >= 0) {
788          return contentLength;
789       }
790       int sum = 0;
791       for (byte[] keyPart : keyParts) {
792          sum += 2 + keyPart.length;
793       }
794       if (innerNodes != null) {
795          sum += INNER_NODE_REFERENCE_SIZE * innerNodes.length;
796       } else if (leafNodes != null) {
797          sum += LEAF_NODE_REFERENCE_SIZE * leafNodes.length;
798       } else {
799          throw new IllegalStateException();
800       }
801       return contentLength = sum;
802    }

805       if (totalLength >= 0) return totalLength;
806       return totalLength = headerLength() + contentLength();

826    static class InnerNode extends Index.IndexSpace {
827       private volatile SoftReference<IndexNode> reference;
828 
829       public InnerNode(long offset, short length) {
830          super(offset, length);
831       }
832 
833       public InnerNode(IndexNode node) {
834          super(node.offset, node.occupiedSpace);
835          reference = new SoftReference<IndexNode>(node);
836       }
837 
838       public IndexNode getIndexNode(Index.Segment segment) throws IOException {
839          IndexNode node;
840          if (reference == null || (node = reference.get()) == null) {
841             synchronized (this) {
842                if (reference == null || (node = reference.get()) == null) {
843                   if (offset < 0) return null;
844                   node = new IndexNode(segment, offset, length);
845                   reference = new SoftReference<IndexNode>(node);
846                   if (log.isTraceEnabled()) {
847                      log.trace("Loaded inner node from " + offset + " - " + length);
848                   }
849                }
850             }
851          }
852          return node;
853       }
854    }

856    private static class LeafNode extends EntryPosition {
857 
858       private volatile SoftReference<EntryRecord> keyReference;
859 
860       public LeafNode(int file, int offset) {
861          super(file, offset);
862       }
863 
864       public EntryRecord loadHeaderAndKey(FileProvider fileProvider, TimeService timeService) throws IOException, IndexNodeOutdatedException {
865          if (offset < 0) return null;
866          EntryRecord headerAndKey = getHeaderAndKey(fileProvider, null, timeService);
867          return headerAndKey;
868       }
869 
870       private EntryRecord getHeaderAndKey(FileProvider fileProvider, FileProvider.Handle handle, TimeService timeService) throws IOException, IndexNodeOutdatedException {
871          EntryRecord headerAndKey;
872          if (keyReference == null || (headerAndKey = keyReference.get()) == null) {
873             synchronized (this) {
874                if (keyReference == null || (headerAndKey = keyReference.get()) == null) {
875                   boolean ownHandle = false;
876                   if (handle == null) {
877                      ownHandle = true;
878                      handle = fileProvider.getFile(file);
879                      if (handle == null) {
880                         throw new IndexNodeOutdatedException(file + ":" + offset);
881                      }
882                   }
883                   try {
884                      EntryHeader header = EntryRecord.readEntryHeader(handle, offset);
885                      if (header == null) {
886                         throw new IllegalStateException("Error reading header from " + file + ":" + offset + " | " + handle.getFileSize());
887                      }
888                      headerAndKey = new EntryRecord(header, EntryRecord.readKey(handle, header, offset), null, null);
889                      keyReference = new SoftReference<EntryRecord>(headerAndKey);
890                   } finally {
891                      if (ownHandle) {
892                         handle.close();
893                      }
894                   }
895                }
896             }
897          }
898          if (headerAndKey.getHeader().expiryTime() > 0 && headerAndKey.getHeader().expiryTime() <= timeService.wallClockTime()) {
899             EntryRecord expired = headerAndKey;
900             headerAndKey = new EntryRecord(headerAndKey.getHeader(), null, null, null);
901             synchronized (this) {
902                if (keyReference.get() == expired) {
903                   keyReference = new SoftReference<EntryRecord>(headerAndKey);
904                }
905             }
906          }
907          return headerAndKey;
908       }
909 
910       public EntryRecord loadRecord(FileProvider fileProvider, byte[] key, TimeService timeService) throws IOException, IndexNodeOutdatedException {
911          if (offset < 0) return null;
912          FileProvider.Handle handle = fileProvider.getFile(file);
913          if (handle == null) {
914             throw new IndexNodeOutdatedException(file + ":" + offset);
915          }
916          try {
917             EntryRecord headerAndKey = getHeaderAndKey(fileProvider, handle, timeService);
918             if (!Arrays.equals(key, headerAndKey.getKey())) {
919                return null;
920             }
921             return headerAndKey.loadMetadataAndValue(handle, offset);
922          } finally {
923             handle.close();
924          }
925       }
926    }

32  /**
33   * This class was entirely copied from JGroups 2.7 (same name there). Couldn't simply reuse it because JGroups does not
34   * ship with MBean, ManagedAttribute and ManagedOperation. Once JGroups will ship these classes, the code can be
35   * dynamically reused from there.
36   * <p/>
37   * The original JGroup's ResourceDMBean logic has been modified so that invoke() method checks whether the operation
38   * called has been exposed as a {@link ManagedOperation}, otherwise the call fails. JGroups deviated from this logic on
39   * purpose because they liked the fact that you could expose all class methods by simply annotating class with {@link
40   * MBean} annotation.
41   *
42   * @author Mircea.Markus@jboss.com
43   * @author Galder Zamarre√±o
44   * @since 4.0
45   */
46  public class ResourceDMBean implements DynamicMBean {

48     private static final String MBEAN_DESCRITION = "Dynamic MBean Description";

50     private static final Log log = LogFactory.getLog(ResourceDMBean.class);
51     private static final boolean trace = log.isTraceEnabled();
52     private final Object obj;
53     private final Class<?> objectClass;
54     private final IspnMBeanOperationInfo[] opInfos;
55     private final MBeanAttributeInfo[] attInfos;
56     private final HashMap<String, InvokableMBeanAttributeInfo> atts = new HashMap<String, InvokableMBeanAttributeInfo>(2);
57     private final ManageableComponentMetadata mBeanMetadata;
58  
59     private static final Map<String, Field> FIELD_CACHE = CollectionFactory.makeConcurrentMap(64);
60     private static final Map<String, Method> METHOD_CACHE = CollectionFactory.makeConcurrentMap(64);
61     private static final Map<String[], Class<?>[]> PARAM_TYPE_CACHE = CollectionFactory.makeConcurrentMap(64);

63     public ResourceDMBean(Object instance, ManageableComponentMetadata mBeanMetadata) throws NoSuchFieldException, ClassNotFoundException {
64  
65        if (instance == null)
66           throw new NullPointerException("Cannot make an MBean wrapper for null instance");
67  
68        this.obj = instance;
69        this.objectClass = instance.getClass();
70        this.mBeanMetadata = mBeanMetadata;
71  
72        // Load up all fields.
73        int i = 0;
74        attInfos = new MBeanAttributeInfo[mBeanMetadata.getAttributeMetadata().size()];
75        for (JmxAttributeMetadata attributeMetadata : mBeanMetadata.getAttributeMetadata()) {
76           String attributeName = attributeMetadata.getName();
77           InvokableMBeanAttributeInfo info = toJmxInfo(attributeMetadata);
78           if (atts.containsKey(attributeName)) {
79              throw new IllegalArgumentException("Component " + mBeanMetadata.getName()
80                                                       + " metadata has a duplicate attribute: " + attributeName);
81           }
82           atts.put(attributeName, info);
83           attInfos[i++] = info.getMBeanAttributeInfo();
84           if (trace)
85              log.tracef("Attribute %s [r=%b,w=%b,is=%b,type=%s]", attributeName,
86                         info.getMBeanAttributeInfo().isReadable(), info.getMBeanAttributeInfo().isWritable(),
87                         info.getMBeanAttributeInfo().isIs(), info.getMBeanAttributeInfo().getType());
88        }
89  
90        // And operations
91        IspnMBeanOperationInfo op;
92        opInfos = new IspnMBeanOperationInfo[mBeanMetadata.getOperationMetadata().size()];
93        i = 0;
94        for (JmxOperationMetadata operation : mBeanMetadata.getOperationMetadata()) {
95           op = toJmxInfo(operation);
96           opInfos[i++] = op;
97           if (trace) log.tracef("Operation %s %s", op.getReturnType(), op.getName());
98        }
99     }

101    private static Field findField(Class<?> objectClass, String fieldName) throws NoSuchFieldException {
102       String key = objectClass.getName() + "#" + fieldName;
103       Field f = FIELD_CACHE.get(key);
104       if (f == null) {
105          f = ReflectionUtil.getField(fieldName, objectClass);
106          if (f != null) FIELD_CACHE.put(key, f);
107       }
108       return f;
109    }
110 
111    private static Method findSetter(Class<?> objectClass, String fieldName) throws NoSuchFieldException {
112       String key = objectClass.getName() + "#s#" + fieldName;
113       Method m = METHOD_CACHE.get(key);
114       if (m == null) {
115          m = ReflectionUtil.findSetterForField(objectClass, fieldName);
116          if (m != null) METHOD_CACHE.put(key, m);
117       }
118       return m;
119    }
120 
121    private static Method findGetter(Class<?> objectClass, String fieldName) throws NoSuchFieldException {
122       String key = objectClass.getName() + "#g#" + fieldName;
123       Method m = METHOD_CACHE.get(key);
124       if (m == null) {
125          m = ReflectionUtil.findGetterForField(objectClass, fieldName);
126          if (m != null) METHOD_CACHE.put(key, m);
127       }
128       return m;
129    }

131    private InvokableMBeanAttributeInfo toJmxInfo(JmxAttributeMetadata attributeMetadata) throws NoSuchFieldException {
132       if (!attributeMetadata.isUseSetter()) {
133          Field field = findField(objectClass, attributeMetadata.getName());
134          if (field != null) {
135             return new InvokableFieldBasedMBeanAttributeInfo(attributeMetadata.getName(), attributeMetadata.getType(),
136                                                              attributeMetadata.getDescription(), true, attributeMetadata.isWritable(),
137                                                              attributeMetadata.isIs(), field, this);
138          }
139       }
140 
141       Method setter = attributeMetadata.isWritable() ? findSetter(objectClass, attributeMetadata.getName()) : null;
142       Method getter = findGetter(objectClass, attributeMetadata.getName());
143       return new InvokableSetterBasedMBeanAttributeInfo(attributeMetadata.getName(), attributeMetadata.getType(),
144                                                         attributeMetadata.getDescription(), true, attributeMetadata.isWritable(),
145                                                         attributeMetadata.isIs(), getter, setter, this);
146    }

148    private IspnMBeanOperationInfo toJmxInfo(JmxOperationMetadata operationMetadata) throws ClassNotFoundException {
149       JmxOperationParameter[] parameters = operationMetadata.getMethodParameters();
150       MBeanParameterInfo[] params = new MBeanParameterInfo[parameters.length];
151       for (int i = 0; i < parameters.length; i++) {
152          params[i] = new MBeanParameterInfo(parameters[i].getName(), parameters[i].getType(), parameters[i].getDescription());
153       }
154       return new IspnMBeanOperationInfo(operationMetadata.getMethodName(), operationMetadata.getDescription(), params, operationMetadata.getReturnType(), MBeanOperationInfo.UNKNOWN,
155                                         operationMetadata.getOperationName());
156    }

162    @Override
163    public synchronized MBeanInfo getMBeanInfo() {
164       //the client doesn't know about IspnMBeanOperationInfo so we need to convert first
165       MBeanOperationInfo[] operationInfoForClient = new MBeanOperationInfo[opInfos.length];
166       for (int i = 0; i < opInfos.length; i++) {
167          IspnMBeanOperationInfo current = opInfos[i];
168          operationInfoForClient[i] = new MBeanOperationInfo(current.getOperationName(), current.getDescription(),
169                                                             current.getSignature(), current.getReturnType(), MBeanOperationInfo.UNKNOWN);
170       }
171       return new MBeanInfo(getObject().getClass().getCanonicalName(), mBeanMetadata.getDescription(), attInfos, null, operationInfoForClient, null);
172    }

174    @Override
175    public Object getAttribute(String name) throws AttributeNotFoundException {
176       if (name == null || name.length() == 0)
177          throw new NullPointerException("Invalid attribute requested " + name);
178 
179       Attribute attr = getNamedAttribute(name);
180       if (attr == null) {
181          throw new AttributeNotFoundException("Unknown attribute '" + name
182                                                     + "'. Known attributes names are: " + atts.keySet());
183       }
184       return attr.getValue();
185    }

188    public synchronized void setAttribute(Attribute attribute) throws AttributeNotFoundException, MBeanException {
189       if (attribute == null || attribute.getName() == null)
190          throw new NullPointerException("Invalid attribute requested " + attribute);
191 
192       setNamedAttribute(attribute);
193    }

195    @Override
196    public synchronized AttributeList getAttributes(String[] names) {
197       AttributeList al = new AttributeList();
198       for (String name : names) {
199          Attribute attr = getNamedAttribute(name);
200          if (attr != null) {
201             al.add(attr);
202          } else {
203             log.couldNotFindAttribute(name);
204          }
205       }
206       return al;
207    }

209    @Override
210    public synchronized AttributeList setAttributes(AttributeList list) {
211       AttributeList results = new AttributeList();
212       for (Object aList : list) {
213          Attribute attr = (Attribute) aList;
214 
215          try {
216             setNamedAttribute(attr);
217             results.add(attr);
218          } catch (Exception e) {
219             log.failedToUpdateAttribute(attr.getName(), attr.getValue());
220          }
221       }
222       return results;
223    }

225    @Override
226    public Object invoke(String name, Object[] args, String[] sig) throws MBeanException,
227                                                                          ReflectionException {
228       if (log.isDebugEnabled()) {
229          log.debugf("Invoke method called on %s", name);
230       }
231 
232       MBeanOperationInfo opInfo = null;
233       for (IspnMBeanOperationInfo op : opInfos) {
234          if (op.getOperationName().equals(name)) {
235             opInfo = op;
236             break;
237          }
238       }
239 
240       if (opInfo == null) {
241          final String msg = "Operation " + name + " not amongst operations in " + opInfos;
242          throw new MBeanException(new ServiceNotFoundException(msg), msg);
243       }
244 
245       // Argument type transformation according to signatures
246       for (int i = 0; i < sig.length; i++) {
247          // arguments are taken from RHQ like java.lang.String but we need some fields to be numbers
248          if (args[i] != null) {
249             if (log.isDebugEnabled())
250                log.debugf("Argument value before transformation: %s and its class: %s. " +
251                                 "For method.invoke we need it to be class: %s", args[i], args[i].getClass(), sig[i]);
252             if (sig[i].equals(int.class.getCanonicalName()) || sig[i].equals(Integer.class.getCanonicalName())) {
253                if ((args[i].getClass() != Integer.class) && (args[i].getClass() != int.class))
254                   args[i] = Integer.parseInt((String) args[i]);
255             }
256             if (sig[i].equals(Long.class.getCanonicalName()) || sig[i].equals(long.class.getCanonicalName())) {
257                if ((args[i].getClass() != Long.class) && (args[i].getClass() != long.class))
258                   args[i] = Long.parseLong((String) args[i]);
259             }
260          }
261       }
262 
263       try {
264          Class<?>[] classes = new Class[sig.length];
265          for (int i = 0; i < classes.length; i++) {
266             classes[i] = ReflectionUtil.getClassForName(sig[i], null);
267          }
268          Method method = getObject().getClass().getMethod(opInfo.getName(), classes);
269          return method.invoke(getObject(), args);
270       } catch (Exception e) {
271          throw new MBeanException(new Exception(getRootCause(e)));
272       }
273    }

275    public Throwable getRootCause(Throwable throwable) {
276       Throwable cause;
277       while ((cause = throwable.getCause()) != null) {
278          throwable = cause;
279       }
280       return throwable;
281    }

284    private Attribute getNamedAttribute(String name) {
285       Attribute result = null;
286       if (name.equals(MBEAN_DESCRITION)) {
287          result = new Attribute(MBEAN_DESCRITION, mBeanMetadata.getDescription());
288       } else {
289          InvokableMBeanAttributeInfo i = atts.get(name);
290          if (i == null && name.length() > 0) {
291             // This is legacy.  Earlier versions used an upper-case starting letter for *some* attributes.
292             Character firstChar = name.charAt(0);
293             if (Character.isUpperCase(firstChar)) {
294                name = name.replaceFirst(Character.toString(firstChar), Character.toString(Character.toLowerCase(firstChar)));
295                i = atts.get(name);
296             }
297          }
298          if (i != null) {
299             try {
300                result = new Attribute(name, i.invoke(null));
301                if (trace)
302                   log.tracef("Attribute %s has r=%b,w=%b,is=%b and value %s",
303                              name, i.getMBeanAttributeInfo().isReadable(), i.getMBeanAttributeInfo().isWritable(), i.getMBeanAttributeInfo().isIs(), result.getValue());
304             } catch (Exception e) {
305                log.debugf("Exception while reading value of attribute %s: %s", name, e);
306             }
307          } else {
308             log.queriedAttributeNotFound(name);
309          }
310       }
311       return result;
312    }

314    private void setNamedAttribute(Attribute attribute) throws MBeanException, AttributeNotFoundException {
315       if (log.isDebugEnabled())
316          log.debugf("Invoking set on attribute %s with value %s",
317                     attribute.getName(), attribute.getValue());
318 
319       String name = attribute.getName();
320       InvokableMBeanAttributeInfo i = atts.get(name);
321       if (i == null && name.length() > 0) {
322          // This is legacy.  Earlier versions used an upper-case starting letter for *some* attributes.
323          Character firstChar = name.charAt(0);
324          if (Character.isUpperCase(firstChar)) {
325             name = name.replaceFirst(Character.toString(firstChar), Character.toString(Character.toLowerCase(firstChar)));
326             i = atts.get(name);
327          }
328       }
329 
330       if (i != null) {
331          try {
332             i.invoke(attribute);
333          } catch (Exception e) {
334             log.errorWritingValueForAttribute(name, e);
335             throw new MBeanException(e, "Error invoking setter for attribute " + name);
336          }
337       } else {
338          log.couldNotInvokeSetOnAttribute(name, attribute.getValue());
339          throw new AttributeNotFoundException("Could not find attribute " + name);
340       }
341    }

343    private static abstract class InvokableMBeanAttributeInfo {
344 
345       private final MBeanAttributeInfo attributeInfo;
346 
347       public InvokableMBeanAttributeInfo(String name, String type, String description, boolean isReadable, boolean isWritable, boolean isIs) {
348          attributeInfo = new MBeanAttributeInfo(name, type, description, isReadable, isWritable, isIs);
349       }
350 
351       public abstract Object invoke(Attribute a) throws IllegalAccessException, InvocationTargetException;
352 
353       public MBeanAttributeInfo getMBeanAttributeInfo() {
354          return attributeInfo;
355       }

358    private static class InvokableFieldBasedMBeanAttributeInfo extends InvokableMBeanAttributeInfo {
359       private transient final Field field;
360       private transient final ResourceDMBean resource;
361 
362       public InvokableFieldBasedMBeanAttributeInfo(String name, String type, String description, boolean isReadable, boolean isWritable, boolean isIs, Field field, ResourceDMBean resource) {
363          super(name, type, description, isReadable, isWritable, isIs);
364          this.field = field;
365          this.resource = resource;
366       }
367 
368       @Override
369       public Object invoke(Attribute a) throws IllegalAccessException {
370          if (!Modifier.isPublic(field.getModifiers())) field.setAccessible(true);
371          if (a == null) {
372             return field.get(resource.getObject());
373          } else {
374             field.set(resource.getObject(), a.getValue());
375             return null;
376          }
377       }
378    }

380    private static class InvokableSetterBasedMBeanAttributeInfo extends InvokableMBeanAttributeInfo {
381       private transient final Method setter;
382       private transient final Method getter;
383       private transient final ResourceDMBean resource;
384 
385       public InvokableSetterBasedMBeanAttributeInfo(String name, String type, String description, boolean isReadable, boolean isWritable, boolean isIs, Method getter, Method setter, ResourceDMBean resource) {
386          super(name, type, description, isReadable, isWritable, isIs);
387          this.setter = setter;
388          this.getter = getter;
389          this.resource = resource;
390       }
391 
392       @Override
393       public Object invoke(Attribute a) throws IllegalAccessException, InvocationTargetException {
394          if (a == null) {
395             if (!Modifier.isPublic(getter.getModifiers())) getter.setAccessible(true);
396             return getter.invoke(resource.getObject(), null);
397          } else {
398             if (!Modifier.isPublic(setter.getModifiers())) setter.setAccessible(true);
399             return setter.invoke(resource.getObject(), a.getValue());
400          }
401       }
402    }

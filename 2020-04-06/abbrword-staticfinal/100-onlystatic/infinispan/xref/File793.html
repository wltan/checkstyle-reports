22  package org.jboss.as.clustering.jgroups.subsystem;
23  
24  import static org.jboss.as.clustering.jgroups.JGroupsLogger.ROOT_LOGGER;
25  
26  import java.util.List;
27  
28  import org.jboss.as.clustering.jgroups.ChannelFactory;
29  import org.jboss.as.clustering.jgroups.ProtocolDefaults;
30  import org.jboss.as.controller.AbstractAddStepHandler;
31  import org.jboss.as.controller.OperationContext;
32  import org.jboss.as.controller.OperationFailedException;
33  import org.jboss.as.controller.ServiceVerificationHandler;
34  import org.jboss.as.controller.descriptions.ModelDescriptionConstants;
35  import org.jboss.as.controller.operations.common.Util;
36  import org.jboss.dmr.ModelNode;
37  import org.jboss.msc.service.ServiceBuilder;
38  import org.jboss.msc.service.ServiceController;
39  import org.jboss.msc.service.ServiceName;
40  import org.jboss.msc.service.ServiceTarget;
41  import org.jboss.msc.service.ValueService;
42  import org.jboss.msc.value.InjectedValue;
43  
44  /**
45   * Handler for JGroups subsystem add operations.
46   *
47   * @author Paul Ferraro
48   * @author Richard Achmatowicz (c) 2011 Red Hat, Inc
49   */
50  public class JGroupsSubsystemAdd extends AbstractAddStepHandler {
51  
52      public static final JGroupsSubsystemAdd INSTANCE = new JGroupsSubsystemAdd();
53  
54      static ModelNode createOperation(ModelNode address, ModelNode existing) throws OperationFailedException {
55          ModelNode operation = Util.getEmptyOperation(ModelDescriptionConstants.ADD, address);
56          populate(existing, operation);
57          return operation;
58      }
59  
60      private static void populate(ModelNode source, ModelNode target) throws OperationFailedException {
61  
62          JGroupsSubsystemRootResource.DEFAULT_STACK.validateAndSet(source, target);
63          target.get(ModelKeys.STACK).setEmptyObject();
64      }
65  
66      @Override
67      protected void populateModel(ModelNode operation, ModelNode model) throws OperationFailedException {
68          populate(operation, model);
69      }
70  
71      @Override
72      protected void performRuntime(OperationContext context, ModelNode operation, ModelNode model, ServiceVerificationHandler verificationHandler, List<ServiceController<?>> newControllers)
73              throws OperationFailedException {
74          installRuntimeServices(context, operation, model, verificationHandler, newControllers);
75      }
76  
77      protected void installRuntimeServices(OperationContext context, ModelNode operation, ModelNode model, ServiceVerificationHandler verificationHandler, List<ServiceController<?>> newControllers) throws OperationFailedException {
78  
79          ROOT_LOGGER.activatingSubsystem();
80          ServiceTarget target = context.getServiceTarget();
81  
82          // install the protocol defaults service
83          ServiceController<ProtocolDefaults> pdsController = installProtocolDefaultsService(target, verificationHandler);
84          if (newControllers != null) {
85              newControllers.add(pdsController);
86          }
87  
88          final String stack = JGroupsSubsystemRootResource.DEFAULT_STACK.resolveModelAttribute(context, model).asString() ;
89  
90          // install the default channel factory service
91          ServiceController<ChannelFactory> dcfsController = installDefaultChannelFactoryService(target, stack, verificationHandler);
92          if (newControllers != null) {
93              newControllers.add(dcfsController);
94          }
95      }
96  
97      protected void removeRuntimeServices(OperationContext context, ModelNode operation, ModelNode model)
98              throws OperationFailedException {
99  
100         // remove the ProtocolDefaultsService
101         ServiceName protocolDefaultsService = ProtocolDefaultsService.SERVICE_NAME;
102         context.removeService(protocolDefaultsService);
103 
104         // remove the DefaultChannelFactoryServiceAlias
105         ServiceName defaultChannelFactoryService = ChannelFactoryService.getServiceName(null);
106         context.removeService(defaultChannelFactoryService);
107     }
108 
109     protected ServiceController<ProtocolDefaults> installProtocolDefaultsService(ServiceTarget target, ServiceVerificationHandler verificationHandler) {
110         ServiceBuilder<ProtocolDefaults> protocolDefaultsBuilder =
111                 target.addService(ProtocolDefaultsService.SERVICE_NAME, new ProtocolDefaultsService())
112                 .setInitialMode(ServiceController.Mode.ON_DEMAND) ;
113         return protocolDefaultsBuilder.install() ;
114     }
115 
116     protected ServiceController<ChannelFactory> installDefaultChannelFactoryService(ServiceTarget target, String stack, ServiceVerificationHandler verificationHandler) {
117         final InjectedValue<ChannelFactory> factory = new InjectedValue<ChannelFactory>();
118         return target.addService(ChannelFactoryService.getServiceName(null), new ValueService<ChannelFactory>(factory))
119                 .addDependency(ChannelFactoryService.getServiceName(stack), ChannelFactory.class, factory)
120                 .setInitialMode(ServiceController.Mode.ON_DEMAND)
121                 .install()
122         ;
123     }
124 
125     @Override
126     protected boolean requiresRuntimeVerification() {
127         return false;
128     }
129 }

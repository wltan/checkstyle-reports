1   package org.infinispan.configuration.cache;
2   
3   import org.infinispan.commons.configuration.attributes.Attribute;
4   import org.infinispan.commons.configuration.attributes.AttributeDefinition;
5   import org.infinispan.commons.configuration.attributes.AttributeSet;
6   
7   /**
8    * Defines clustered characteristics of the cache.
9    *
10   * @author pmuir
11   *
12   */
13  public class ClusteringConfiguration {
14     public static final AttributeDefinition<CacheMode> CACHE_MODE = AttributeDefinition.builder("cacheMode",  CacheMode.LOCAL).immutable().build();
15  
16     static AttributeSet attributeDefinitionSet() {
17        return new AttributeSet(ClusteringConfiguration.class, CACHE_MODE);
18     }
19  
20     private final Attribute<CacheMode> cacheMode;
21     private final AsyncConfiguration asyncConfiguration;
22     private final HashConfiguration hashConfiguration;
23     private final L1Configuration l1Configuration;
24     private final StateTransferConfiguration stateTransferConfiguration;
25     private final SyncConfiguration syncConfiguration;
26     private final PartitionHandlingConfiguration partitionHandlingConfiguration;
27     private final AttributeSet attributes;
28  
29     ClusteringConfiguration(AttributeSet attributes, AsyncConfiguration asyncConfiguration, HashConfiguration hashConfiguration,
30           L1Configuration l1Configuration, StateTransferConfiguration stateTransferConfiguration, SyncConfiguration syncConfiguration,
31           PartitionHandlingConfiguration partitionHandlingStrategy) {
32        this.attributes = attributes.checkProtection();
33        this.cacheMode = attributes.attribute(CACHE_MODE);
34        this.asyncConfiguration = asyncConfiguration;
35        this.hashConfiguration = hashConfiguration;
36        this.l1Configuration = l1Configuration;
37        this.stateTransferConfiguration = stateTransferConfiguration;
38        this.syncConfiguration = syncConfiguration;
39        this.partitionHandlingConfiguration  = partitionHandlingStrategy;
40     }
41  
42     /**
43      * Cache mode. See {@link CacheMode} for information on the various cache modes available.
44      */
45     public CacheMode cacheMode() {
46        return cacheMode.get();
47     }
48  
49     /**
50      * Configures cluster's behaviour in the presence of partitions or node failures.
51      */
52     public PartitionHandlingConfiguration partitionHandling() {
53        return partitionHandlingConfiguration;
54     }
55  
56     public String cacheModeString() {
57  
58        return cacheMode() == null ? "none" : cacheMode().toString();
59     }
60  
61     /**
62      * Configure async sub element. Once this method is invoked users cannot subsequently invoke
63      * <code>sync()</code> as two are mutually exclusive
64      */
65     public AsyncConfiguration async() {
66        return asyncConfiguration;
67     }
68  
69     /**
70      * Configure hash sub element
71      */
72     public HashConfiguration hash() {
73        return hashConfiguration;
74     }
75  
76     /**
77      * This method allows configuration of the L1 cache for distributed caches. When this method is
78      * called, it automatically enables L1. So, if you want it to be disabled, make sure you call
79      * {@link org.infinispan.configuration.cache.L1ConfigurationBuilder#disable()}
80      */
81     public L1Configuration l1() {
82        return l1Configuration;
83     }
84  
85     /**
86      * Configure sync sub element. Once this method is invoked users cannot subsequently invoke
87      * <code>async()</code> as two are mutually exclusive
88      */
89     public SyncConfiguration sync() {
90        return syncConfiguration;
91     }
92  
93     public StateTransferConfiguration stateTransfer() {
94        return stateTransferConfiguration;
95     }
96  
97     public AttributeSet attributes() {
98        return attributes;
99     }
100 
101    @Override
102    public String toString() {
103       return "ClusteringConfiguration [asyncConfiguration=" + asyncConfiguration + ", hashConfiguration="
104             + hashConfiguration + ", l1Configuration=" + l1Configuration + ", stateTransferConfiguration="
105             + stateTransferConfiguration + ", syncConfiguration=" + syncConfiguration
106             + ", partitionHandlingConfiguration=" + partitionHandlingConfiguration + ", attributes=" + attributes + "]";
107    }
108 
109    @Override
110    public boolean equals(Object obj) {
111       if (this == obj)
112          return true;
113       if (obj == null)
114          return false;
115       if (getClass() != obj.getClass())
116          return false;
117       ClusteringConfiguration other = (ClusteringConfiguration) obj;
118       if (asyncConfiguration == null) {
119          if (other.asyncConfiguration != null)
120             return false;
121       } else if (!asyncConfiguration.equals(other.asyncConfiguration))
122          return false;
123       if (attributes == null) {
124          if (other.attributes != null)
125             return false;
126       } else if (!attributes.equals(other.attributes))
127          return false;
128       if (hashConfiguration == null) {
129          if (other.hashConfiguration != null)
130             return false;
131       } else if (!hashConfiguration.equals(other.hashConfiguration))
132          return false;
133       if (l1Configuration == null) {
134          if (other.l1Configuration != null)
135             return false;
136       } else if (!l1Configuration.equals(other.l1Configuration))
137          return false;
138       if (partitionHandlingConfiguration == null) {
139          if (other.partitionHandlingConfiguration != null)
140             return false;
141       } else if (!partitionHandlingConfiguration.equals(other.partitionHandlingConfiguration))
142          return false;
143       if (stateTransferConfiguration == null) {
144          if (other.stateTransferConfiguration != null)
145             return false;
146       } else if (!stateTransferConfiguration.equals(other.stateTransferConfiguration))
147          return false;
148       if (syncConfiguration == null) {
149          if (other.syncConfiguration != null)
150             return false;
151       } else if (!syncConfiguration.equals(other.syncConfiguration))
152          return false;
153       return true;
154    }
155 
156    @Override
157    public int hashCode() {
158       final int prime = 31;
159       int result = 1;
160       result = prime * result + ((asyncConfiguration == null) ? 0 : asyncConfiguration.hashCode());
161       result = prime * result + ((attributes == null) ? 0 : attributes.hashCode());
162       result = prime * result + ((hashConfiguration == null) ? 0 : hashConfiguration.hashCode());
163       result = prime * result + ((l1Configuration == null) ? 0 : l1Configuration.hashCode());
164       result = prime * result
165             + ((partitionHandlingConfiguration == null) ? 0 : partitionHandlingConfiguration.hashCode());
166       result = prime * result + ((stateTransferConfiguration == null) ? 0 : stateTransferConfiguration.hashCode());
167       result = prime * result + ((syncConfiguration == null) ? 0 : syncConfiguration.hashCode());
168       return result;
169    }

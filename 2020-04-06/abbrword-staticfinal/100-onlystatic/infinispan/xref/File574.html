1   package org.infinispan.persistence.leveldb.configuration;
2   
3   import org.infinispan.commons.configuration.BuiltBy;
4   import org.infinispan.commons.configuration.ConfigurationFor;
5   import org.infinispan.commons.configuration.attributes.Attribute;
6   import org.infinispan.commons.configuration.attributes.AttributeDefinition;
7   import org.infinispan.commons.configuration.attributes.AttributeSet;
8   import org.infinispan.configuration.cache.AbstractStoreConfiguration;
9   import org.infinispan.configuration.cache.AsyncStoreConfiguration;
10  import org.infinispan.configuration.cache.SingletonStoreConfiguration;
11  import org.infinispan.persistence.leveldb.LevelDBStore;
12  
13  /**
14   *
15   * @author <a href="mailto:rtsang@redhat.com">Ray Tsang</a>
16   *
17   */
18  @ConfigurationFor(LevelDBStore.class)
19  @BuiltBy(LevelDBStoreConfigurationBuilder.class)
20  public class LevelDBStoreConfiguration extends AbstractStoreConfiguration {
21     public enum ImplementationType {
22        AUTO, JAVA, JNI
23     }
24  
25     final static AttributeDefinition<String> LOCATION = AttributeDefinition.builder("location", "Infinispan-LevelDBStore/data").immutable().build();
26     final static AttributeDefinition<String> EXPIRED_LOCATION = AttributeDefinition.builder("expiredLocation", "Infinispan-LevelDBStore/expired").immutable().build();
27     final static AttributeDefinition<ImplementationType> IMPLEMENTATION_TYPE = AttributeDefinition.builder("implementationType", ImplementationType.AUTO).immutable().build();
28     final static AttributeDefinition<CompressionType> COMPRESSION_TYPE = AttributeDefinition.builder("compressionType", CompressionType.NONE).immutable().build();
29     final static AttributeDefinition<Integer> BLOCK_SIZE = AttributeDefinition.builder("blockSize", 0).immutable().build();
30     final static AttributeDefinition<Long> CACHE_SIZE = AttributeDefinition.builder("cacheSize", 0l).immutable().build();
31     final static AttributeDefinition<Integer> EXPIRY_QUEUE_SIZE = AttributeDefinition.builder("expiryQueueSize", 10000).immutable().build();
32     final static AttributeDefinition<Integer> CLEAR_THRESHOLD = AttributeDefinition.builder("clearThreshold", 10000).immutable().build();
33  
34     public static AttributeSet attributeDefinitionSet() {
35        return new AttributeSet(LevelDBStoreConfiguration.class, AbstractStoreConfiguration.attributeDefinitionSet(), LOCATION, EXPIRED_LOCATION, IMPLEMENTATION_TYPE, COMPRESSION_TYPE,
36              BLOCK_SIZE, CACHE_SIZE, EXPIRY_QUEUE_SIZE, CLEAR_THRESHOLD);
37     }
38  
39     private final Attribute<String> location;
40     private final Attribute<String> expiredLocation;
41     private final Attribute<ImplementationType> implementationType;
42     private final Attribute<CompressionType> compressionType;
43     private final Attribute<Integer> blockSize;
44     private final Attribute<Long> cacheSize;
45     private final Attribute<Integer> expiryQueueSize;
46     private final Attribute<Integer> clearThreshold;
47  
48     public LevelDBStoreConfiguration(AttributeSet attributes, AsyncStoreConfiguration async, SingletonStoreConfiguration singletonStore) {
49        super(attributes, async, singletonStore);
50        location = attributes.attribute(LOCATION);
51        expiredLocation = attributes.attribute(EXPIRED_LOCATION);
52        implementationType = attributes.attribute(IMPLEMENTATION_TYPE);
53        compressionType = attributes.attribute(COMPRESSION_TYPE);
54        blockSize = attributes.attribute(BLOCK_SIZE);
55        cacheSize = attributes.attribute(CACHE_SIZE);
56        expiryQueueSize = attributes.attribute(EXPIRY_QUEUE_SIZE);
57        clearThreshold = attributes.attribute(CLEAR_THRESHOLD);
58     }
59  
60     public String location() {
61        return location.get();
62     }
63  
64     public String expiredLocation() {
65        return expiredLocation.get();
66     }
67  
68     public ImplementationType implementationType() {
69        return implementationType.get();
70     }
71  
72     public CompressionType compressionType() {
73        return compressionType.get();
74     }
75  
76     public Integer blockSize() {
77        return blockSize.get();
78     }
79  
80     public Long cacheSize() {
81        return cacheSize.get();
82     }
83  
84     public int expiryQueueSize() {
85        return expiryQueueSize.get();
86     }
87  
88     public int clearThreshold() {
89        return clearThreshold.get();
90     }
91  }

115     public static final String IDS_PATTERN = "\\s*,\\s*";
116 
117     public static final String IDS_SEPARATOR = ", ";
118 
119     public static String encodeIds(Set<String> ids) {
120         StringBuilder sb = new StringBuilder();
121         for (String string : ids) {
122             sb.append(string).append(IDS_SEPARATOR);
123         }
124         if (sb.length() > 0) {
125             sb.setLength(sb.length() - IDS_SEPARATOR.length());
126         }
127         return sb.toString();
128     }
129 
130     public static Set<String> decodeIds(String text) {
131         Set<String> sortedIds = new TreeSet<>();
132         if (text == null || text.trim().length() == 0) {
133             return sortedIds;
134         }
135 
136         String[] strings = text.split(IDS_PATTERN);
137         for (String string : strings) {
138             string = string.trim();
139             if (string.length() > 0) {
140                 sortedIds.add(string);
141             }
142         }
143         return sortedIds;
144     }

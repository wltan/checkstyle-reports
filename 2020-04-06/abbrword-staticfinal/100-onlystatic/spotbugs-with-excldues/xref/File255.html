20  package edu.umd.cs.findbugs.ba.jsr305;
21  
22  import javax.annotation.meta.When;
23  
24  import edu.umd.cs.findbugs.ba.Location;
25  import edu.umd.cs.findbugs.ba.vna.ValueNumber;
26  
27  /**
28   * Information about a source or sink in the type qualifier dataflow analysis.
29   *
30   * @author David Hovemeyer
31   */
32  public class SourceSinkInfo implements Comparable<SourceSinkInfo> {
33      private static final int PRIME = 31;
34  
35      private final SourceSinkType type;
36  
37      private final Location location;
38  
39      private final ValueNumber vn;
40  
41      private final When when;
42  
43      private int parameter;
44  
45      private int local;
46  
47      private Object constantValue;
48  
49      private boolean interproc;
50  
51      /**
52       * Constructor.
53       *
54       * @param type
55       *            type of the source or sink
56       * @param location
57       *            Location of the source or sink
58       * @param vn
59       *            the ValueNumber of the annotated value
60       * @param when
61       *            the When value used (explicitly or implicitly) to annotate
62       *            this source or sink
63       */
64      public SourceSinkInfo(SourceSinkType type, Location location, ValueNumber vn, When when) {
65          this.type = type;
66          this.location = location;
67          this.vn = vn;
68          this.when = when;
69      }
70  
71      /**
72       * @return Returns the type.
73       */
74      public SourceSinkType getType() {
75          return type;
76      }
77  
78      /**
79       * @return Returns the location.
80       */
81      public Location getLocation() {
82          return location;
83      }
84  
85      /**
86       * @return Returns the ValueNumber.
87       */
88      public ValueNumber getValueNumber() {
89          return vn;
90      }
91  
92      /**
93       * @return Returns the when.
94       */
95      public When getWhen() {
96          return when;
97      }
98  
99      /**
100      * @param parameter
101      *            The parameter to set.
102      */
103     public void setParameter(int parameter) {
104         this.parameter = parameter;
105     }
106 
107     /**
108      * @param parameter
109      *            The parameter to set.
110      * @param local
111      *            The local to set.
112      */
113     public void setParameterAndLocal(int parameter, int local) {
114         this.parameter = parameter;
115         this.local = local;
116     }
117 
118     /**
119      * @return Returns the parameter.
120      */
121     public int getParameter() {
122         return parameter;
123     }
124 
125     /**
126      * @return Returns the local.
127      */
128     public int getLocal() {
129         return local;
130     }
131 
132     /**
133      * Set the SourceSinkInfo as having been created based on the results of
134      * type qualifiers computed for a called method (and not explicitly
135      * annotating the called method).
136      *
137      * @param interproc
138      *            true if the SourceSinkInfo results from computed type
139      *            qualifiers for a called method, false otherwise
140      */
141     public void setInterproc(boolean interproc) {
142         this.interproc = interproc;
143     }
144 
145     /**
146      * Return whether or not the SourceSinkInfo was created based on the results
147      * of type qualifiers computed for a called method (and not explicitly
148      * annotating the called method).
149      *
150      * @return true if the SourceSinkInfo results from computed type qualifiers
151      *         for a called method, false otherwise
152      */
153     public boolean getInterproc() {
154         return interproc;
155     }
156 
157     /*
158      * (non-Javadoc)
159      *
160      * @see java.lang.Comparable#compareTo(java.lang.Object)
161      */
162     @Override
163     public int compareTo(SourceSinkInfo o) {
164         return this.location.compareTo(o.location);
165     }
166 
167     public Object getConstantValue() {
168         return constantValue;
169     }
170 
171     public void setConstantValue(Object constantValue) {
172         this.constantValue = constantValue;
173     }
174 
175     @Override
176     public int hashCode() {
177         int result = 1;
178         result = PRIME * result + ((constantValue == null) ? 0 : constantValue.hashCode());
179         result = PRIME * result + (interproc ? 1231 : 1237);
180         result = PRIME * result + local;
181         result = PRIME * result + ((location == null) ? 0 : location.hashCode());
182         result = PRIME * result + parameter;
183         result = PRIME * result + ((type == null) ? 0 : type.hashCode());
184         result = PRIME * result + ((vn == null) ? 0 : vn.hashCode());
185         result = PRIME * result + ((when == null) ? 0 : when.hashCode());
186         return result;
187     }
188 
189     @Override
190     public boolean equals(Object obj) {
191         if (this == obj) {
192             return true;
193         }
194         if (obj == null) {
195             return false;
196         }
197         if (getClass() != obj.getClass()) {
198             return false;
199         }
200         SourceSinkInfo other = (SourceSinkInfo) obj;
201         if (constantValue == null) {
202             if (other.constantValue != null) {
203                 return false;
204             }
205         } else if (!constantValue.equals(other.constantValue)) {
206             return false;
207         }
208         if (interproc != other.interproc) {
209             return false;
210         }
211         if (local != other.local) {
212             return false;
213         }
214         if (location == null) {
215             if (other.location != null) {
216                 return false;
217             }
218         } else if (!location.equals(other.location)) {
219             return false;
220         }
221         if (parameter != other.parameter) {
222             return false;
223         }
224         if (type != other.type) {
225             return false;
226         }
227         if (vn == null) {
228             if (other.vn != null) {
229                 return false;
230             }
231         } else if (!vn.equals(other.vn)) {
232             return false;
233         }
234         if (when != other.when) {
235             return false;
236         }
237         return true;
238     }
239 
240     /*
241      * (non-Javadoc)
242      *
243      * @see java.lang.Object#toString()
244      */
245     @Override
246     public String toString() {
247         return type.toString() + "@" + location.toCompactString() + "[vn=" + vn.getNumber() + ",when=" + when + "]";
248     }
249 }

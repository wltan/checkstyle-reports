19  package edu.umd.cs.findbugs.detect;
20  
21  import edu.umd.cs.findbugs.props.AbstractWarningProperty;
22  import edu.umd.cs.findbugs.props.PriorityAdjustment;
23  
24  /**
25   * Warning properties for null pointer dereference and redundant null comparison
26   * warnings.
27   *
28   * @author David Hovemeyer
29   */
30  public class NullDerefProperty extends AbstractWarningProperty {
31  
32      private NullDerefProperty(String name, PriorityAdjustment priorityAdjustment) {
33          super(name, priorityAdjustment);
34      }
35  
36      /** Redundant null comparison is of a checked null value. */
37      public static final NullDerefProperty CHECKED_VALUE = new NullDerefProperty("CHECKED_VALUE",
38              PriorityAdjustment.RAISE_PRIORITY);
39  
40      /** Redundant null comparison is of a checked null value. */
41      public static final NullDerefProperty LONG_RANGE_NULL_SOURCE = new NullDerefProperty("LONG_RANGE_NULL_SOURCE",
42              PriorityAdjustment.AT_MOST_MEDIUM);
43  
44      /** dereference always on exception path */
45      public static final NullDerefProperty ALWAYS_ON_EXCEPTION_PATH = new NullDerefProperty("ALWAYS_ON_EXCEPTION_PATH",
46              PriorityAdjustment.AT_MOST_MEDIUM);
47  
48      /** Redundant nullcheck of previously dereferenced value. */
49      public static final NullDerefProperty WOULD_HAVE_BEEN_A_KABOOM = new NullDerefProperty("WOULD_HAVE_BEEN_A_KABOOM",
50              PriorityAdjustment.RAISE_PRIORITY);
51  
52      /** Redundant nullcheck created dead code. */
53      public static final NullDerefProperty CREATED_DEAD_CODE = new NullDerefProperty("CREATED_DEAD_CODE",
54              PriorityAdjustment.RAISE_PRIORITY);
55  
56      public static final NullDerefProperty DEREFS_ARE_CLONED = new NullDerefProperty("DEREFS_ARE_CLONED",
57              PriorityAdjustment.AT_MOST_MEDIUM);
58  
59      public static final NullDerefProperty CLOSING_NULL = new NullDerefProperty("CLOSING_NULL", PriorityAdjustment.PEGGED_HIGH);
60  
61      public static final NullDerefProperty DEREFS_ARE_INLINED_FINALLY_BLOCKS = new NullDerefProperty(
62              "DEREFS_ARE_INLINED_FINALLY_BLOCKS", PriorityAdjustment.AT_MOST_MEDIUM);
63  
64      public static final NullDerefProperty DEREFS_IN_CATCH_BLOCKS = new NullDerefProperty("DEREFS_IN_CATCH_BLOCKS",
65              PriorityAdjustment.AT_MOST_MEDIUM);
66  }

52  /**
53   * A TypeQualifierValue is a pair specifying a type qualifier annotation and a
54   * value. Each TypeQualifierValue is effectively a different type qualifier. For
55   * example, if Foo is a type qualifier annotation having an int value, then
56   * Foo(0), Foo(1), etc. are all different type qualifiers which must be checked
57   * separately.
58   *
59   * @author William Pugh
60   */
61  public class TypeQualifierValue<A extends Annotation> {
62      public static final boolean DEBUG = SystemProperties.getBoolean("tqv.debug");
63      public static final boolean DEBUG_CLASSLOADING = SystemProperties.getBoolean("tqv.debug.classloading");
64  
65      private static final ClassDescriptor EXCLUSIVE_ANNOTATION = DescriptorFactory.instance().getClassDescriptor(
66              javax.annotation.meta.Exclusive.class);
67  
68      private static final ClassDescriptor EXHAUSTIVE_ANNOTATION = DescriptorFactory.instance().getClassDescriptor(
69              javax.annotation.meta.Exhaustive.class);
70  
71      public final ClassDescriptor typeQualifier;
72      public final Class<A> typeQualifierClass;
73  
74      public final A proxy;
75  
76      public final @CheckForNull Object value;
77  
78      private final boolean isStrict;
79  
80      private final boolean isExclusive;
81  
82      private final boolean isExhaustive;
83  
84      private final @CheckForNull
85      TypeQualifierValidator<A> validator;
86

89      private TypeQualifierValue(ClassDescriptor typeQualifier, @CheckForNull Object value) {
90          this.typeQualifier = typeQualifier;
91          this.value = value;
92          /**  will be set to true if this is a strict type qualifier value */
93          boolean isStrict1 = false;
94          /**  will be set to true if this is an exclusive type qualifier value */
95          boolean isExclusive1 = false;
96          /** will be set to true if this is an exhaustive type qualifier value */
97          boolean isExhaustive1 = false;
98          TypeQualifierValidator<A> validator1 = null;
99          Class<A> qualifierClass = null;
100         XClass xclass  = null;
101         A proxy1 = null;
102         try {
103             xclass = Global.getAnalysisCache().getClassAnalysis(XClass.class, typeQualifier);
104 
105             // Annotation elements appear as abstract methods in the annotation
106             // class (interface).
107             // So, if the type qualifier annotation has specified a default When
108             // value,
109             // it will appear as an abstract method called "when".
110             XMethod whenMethod = xclass.findMethod("when", "()Ljavax/annotation/meta/When;", false);
111             if (whenMethod == null) {
112                 isStrict1 = true;
113             }
114             for (XMethod xmethod : xclass.getXMethods()) {
115                 if ("value".equals(xmethod.getName()) && xmethod.getSignature().startsWith("()")) {
116                     isExhaustive1 = xmethod.getAnnotation(EXHAUSTIVE_ANNOTATION) != null;
117                     if (isExhaustive1) {
118                         // exhaustive qualifiers are automatically exclusive
119                         isExclusive1 = true;
120                     } else {
121                         // see if there is an explicit @Exclusive annotation
122                         isExclusive1 = xmethod.getAnnotation(EXCLUSIVE_ANNOTATION) != null;
123                     }
124 
125                     break;
126                 }
127             }
128         } catch (MissingClassException e) {
129             AnalysisContext.currentAnalysisContext().getLookupFailureCallback().reportMissingClass(e.getClassNotFoundException());
130         } catch (CheckedAnalysisException e) {
131             AnalysisContext.logError("Error looking up annotation class " + typeQualifier.toDottedClassName(), e);
132         }
133         this.isStrict = isStrict1;
134         this.isExclusive = isExclusive1;
135         this.isExhaustive = isExhaustive1;
136 
137 
138         if (xclass != null) {
139             ClassDescriptor checkerName = DescriptorFactory.createClassDescriptor(typeQualifier.getClassName() + "$Checker");
140 
141             if (!SystemProperties.RUNNING_IN_ECLIPSE) {
142                 /**   don't do this if running in Eclipse; check below is the quick
143                 fix for bug 3599258 (Random obscure Eclipse failures during
144                  analysis) */
145 
146                 try {
147                     Global.getAnalysisCache().getClassAnalysis(ClassData.class, checkerName);
148 
149                     // found it.
150                     SecurityManager m = System.getSecurityManager();
151                     if (m == null) {
152                         if (DEBUG_CLASSLOADING) {
153                             System.out.println("Setting ValidationSecurityManager");
154                         }
155                         System.setSecurityManager(ValidationSecurityManager.INSTANCE);
156                     }
157 
158                     Class<?> c = ValidatorClassLoader.INSTANCE.loadClass(checkerName.getDottedClassName());
159                     if (TypeQualifierValidator.class.isAssignableFrom(c)) {
160 
161                         @SuppressWarnings("unchecked")
162                         Class<? extends TypeQualifierValidator<A>> validatorClass = (Class<? extends TypeQualifierValidator<A>>) c
163                         .asSubclass(TypeQualifierValidator.class);
164                         validator1 = getValidator(validatorClass);
165                         qualifierClass = getQualifierClass(typeQualifier);
166 
167                         InvocationHandler handler = (arg0, arg1, arg2) -> {
168                             if ("value".equals(arg1.getName())) {
169                                 return TypeQualifierValue.this.value;
170                             }
171                             throw new UnsupportedOperationException("Can't handle " + arg1);
172                         };
173 
174                         proxy1 = qualifierClass.cast(Proxy.newProxyInstance(ValidatorClassLoader.INSTANCE,
175                                 new Class[] { qualifierClass }, handler));
176                     }
177 
178                 } catch (ClassNotFoundException e) {
179                     assert true; // ignore
180                 } catch (CheckedAnalysisException e) {
181                     assert true; // ignore
182                 } catch (Exception e) {
183                     AnalysisContext.logError("Unable to construct type qualifier checker " + checkerName, e);
184                 } catch (Throwable e) {
185                     AnalysisContext.logError("Unable to construct type qualifier checker " + checkerName + " due to "
186                             + e.getClass().getSimpleName() + ":" + e.getMessage());
187                 }
188             }
189             else if (DEBUG_CLASSLOADING) {
190                 SecurityManager m = System.getSecurityManager();
191                 if (m == null) {
192                     if (DEBUG_CLASSLOADING) {
193                         System.out.println("Setting ValidationSecurityManager");
194                     }
195                     System.setSecurityManager(ValidationSecurityManager.INSTANCE);
196                 }
197             }
198         }
199         this.validator = validator1;
200         this.typeQualifierClass = qualifierClass;
201         this.proxy = proxy1;
202     }

204     private static <A extends Annotation> TypeQualifierValidator<A> getValidator(
205             Class<? extends TypeQualifierValidator<A>> checkerClass)
206                     throws InstantiationException, IllegalAccessException {
207         return checkerClass.newInstance();
208     }

210     @SuppressWarnings("unchecked")
211     private static <A> Class<A> getQualifierClass(ClassDescriptor typeQualifier) throws ClassNotFoundException {
212         @DottedClassName String className = typeQualifier.getDottedClassName();
213         if (DEBUG_CLASSLOADING) {
214             System.out.println("Getting qualifier class for " + className);
215         }
216         if (className.startsWith("javax.annotation")) {
217             return (Class<A>) Class.forName(className);
218         }
219         try {
220             Global.getAnalysisCache().getClassAnalysis(ClassData.class, typeQualifier);
221         } catch (CheckedAnalysisException e) {
222             throw new ClassNotFoundException("No class data found for " + className);
223         }
224 
225         ValidatorClassLoader validatorLoader = ValidatorClassLoader.INSTANCE;
226         return (Class<A>) validatorLoader.loadClass(typeQualifier.getDottedClassName());
227     }

229     static byte[] loadClassData(String name) throws CheckedAnalysisException {
230         ClassDescriptor d = DescriptorFactory.createClassDescriptorFromDottedClassName(name);
231         ClassData data = Global.getAnalysisCache().getClassAnalysis(ClassData.class, d);
232         return data.getData();
233     }

257     public boolean canValidate(@CheckForNull Object constantValue) {
258         if (validator == null) {
259             return false;
260         }
261         return true;
262     }
263 
264     public When validate(@CheckForNull Object constantValue) {
265         if (validator == null) {
266             throw new IllegalStateException("No validator");
267         }
268         IAnalysisCache analysisCache = Global.getAnalysisCache();
269         Profiler profiler = analysisCache.getProfiler();
270         profiler.start(validator.getClass());
271         try {
272             return ValidationSecurityManager.sandboxedValidation(proxy, validator, constantValue);
273         } catch (Exception e) {
274             AnalysisContext.logError("Error executing custom validator for " + typeQualifier + " " + constantValue, e);
275             return When.UNKNOWN;
276         } finally {
277             profiler.end(validator.getClass());
278         }
279     }

281     /**
282      * Given a ClassDescriptor/value pair, return the interned
283      * TypeQualifierValue representing that pair.
284      *
285      * @param desc
286      *            a ClassDescriptor denoting a type qualifier annotation
287      * @param value
288      *            a value
289      * @return an interned TypeQualifierValue object
290      */
291     @SuppressWarnings("rawtypes")
292     public static @Nonnull
293     TypeQualifierValue<?> getValue(ClassDescriptor desc, @CheckForNull  Object value) {
294         DualKeyHashMap<ClassDescriptor, Object, TypeQualifierValue<?>> map = instance.get().typeQualifierMap;
295         TypeQualifierValue<?> result = map.get(desc, value);
296         if (result != null) {
297             return result;
298         }
299         result = new TypeQualifierValue(desc, value);
300         map.put(desc, value, result);
301         instance.get().allKnownTypeQualifiers.add(result);
302         return result;
303     }
304     @SuppressWarnings("unchecked")
305     public static @Nonnull <A extends Annotation>
306     TypeQualifierValue<A> getValue(Class <A> clazz, @CheckForNull Object value) {
307         return (TypeQualifierValue<A>) getValue(DescriptorFactory.createClassDescriptor(clazz), value);
308     }

319     /**
320      * Get the "complementary" TypeQualifierValues for given exclusive type
321      * qualifier.
322      *
323      * @param tqv
324      *            a type qualifier (which must be exclusive)
325      * @return Collection of complementary exclusive type qualifiers
326      */
327     public static Collection<TypeQualifierValue<?>> getComplementaryExclusiveTypeQualifierValue(TypeQualifierValue<?> tqv) {
328         assert tqv.isExclusiveQualifier();
329 
330         LinkedList<TypeQualifierValue<?>> result = new LinkedList<>();
331 
332         for (TypeQualifierValue<?> t : instance.get().allKnownTypeQualifiers) {
333             //
334             // Any TypeQualifierValue with the same
335             // annotation class but a different value is a complementary
336             // type qualifier.
337             //
338             if (t.typeQualifier.equals(tqv.typeQualifier) && !Objects.equals(t.value, tqv.value)) {
339                 result.add(t);
340             }
341         }
342 
343         return result;
344     }

346     /**
347      * Determine whether or not given TypeQualifierValue has multiple variants.
348      * I.e., if Color is a type qualifier having values RED, GREEN, and BLUE,
349      * then there are 3 variants, Color(RED), Color(GREEN), and COLOR(BLUE).
350      *
351      * @param tqv
352      *            a TypeQualifierValue
353      * @return true if there are multiple variants of this type qualifier, false
354      *         otherwise
355      */
356     public static boolean hasMultipleVariants(TypeQualifierValue<?> tqv) {
357         int count = 0;
358         for (TypeQualifierValue<?> t : instance.get().allKnownTypeQualifiers) {
359             if (t.typeQualifier.equals(tqv.typeQualifier)) {
360                 count++;
361             }
362         }
363         return count > 1;
364     }

404     @Override
405     public int hashCode() {
406         int result = typeQualifier.hashCode();
407         if (value != null) {
408             result += 37 * value.hashCode();
409         }

413     @Override
414     public boolean equals(Object o) {
415         if (!(o instanceof TypeQualifierValue)) {
416             return false;
417         }
418         TypeQualifierValue<?> other = (TypeQualifierValue<?>) o;
419         return typeQualifier.equals(other.typeQualifier) && Objects.equals(value, other.value);
420     }

426         if (value != null) {
427             buf.append(':');
428             buf.append(value.toString());
429         }
430         return buf.toString();

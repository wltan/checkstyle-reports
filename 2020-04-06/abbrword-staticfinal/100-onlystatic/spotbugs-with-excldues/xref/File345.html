124     @Override
125     public void visit(JavaClass obj) {
126         isTigerOrHigher = obj.getMajor() >= Const.MAJOR_1_5;
127         try {
128             Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
129             ClassDescriptor me = getClassDescriptor();
130             if (subtypes2.isSubtype(me, MAP_ENTRY) && subtypes2.isSubtype(me, ITERATOR)) {
131                 bugReporter.reportBug(new BugInstance(this, "PZ_DONT_REUSE_ENTRY_OBJECTS_IN_ITERATORS", NORMAL_PRIORITY)
132                 .addClass(this).addString("shouldn't reuse Iterator as a Map.Entry"));
133             }
134         } catch (ClassNotFoundException e) {
135             AnalysisContext.reportMissingClass(e);
136         }

144     private int adjustPriority(int factor, int priority) {
145         if (factor <= 4) {
146             return LOW_PRIORITY + 2;
147         }
148         if (factor <= 10000) {
149             return priority + 1;
150         }
151         if (factor <= 60 * 60 * 1000) {
152             return priority;
153         }
154         return priority - 1;
155     }
156 
157     private int adjustMultiplier(Object constant, int mul) {
158         if (!(constant instanceof Integer)) {
159             return mul;
160         }
161         return Math.abs(((Integer) constant).intValue()) * mul;

203                     priority = Priorities.NORMAL_PRIORITY;

215         if (seen == Const.INVOKEVIRTUAL && "hashCode".equals(getNameConstantOperand()) && "()I".equals(getSigConstantOperand())
216                 && stack.getStackDepth() > 0) {
217             OpcodeStack.Item item0 = stack.getStackItem(0);
218             if (item0.getSignature().charAt(0) == '[') {
219                 bugReporter.reportBug(new BugInstance(this, "DMI_INVOKING_HASHCODE_ON_ARRAY", NORMAL_PRIORITY)

221             }
222         }
223         if (seen != Const.RETURN && isReturn(seen) && isRegisterStore(getPrevOpcode(1))) {
224 
225             int priority = Priorities.NORMAL_PRIORITY;
226             if (getMethodSig().endsWith(")Z")) {
227                 priority = Priorities.HIGH_PRIORITY;
228             } else {
229                 if (getMethodSig().endsWith(")Ljava/lang/String;")) {
230                     priority = Priorities.LOW_PRIORITY;
231                 }
232                 if (getPC() == getCode().getCode().length - 1) {
233                     priority++;
234                 }
235             }
236             bugReporter.reportBug(new BugInstance(this, "DLS_DEAD_LOCAL_STORE_IN_RETURN", priority).addClassAndMethod(this)
237                     .addSourceLine(this));
238         }

240         // ternaryConversionState);

242             if (imul_distance != 1) {
243                 resetIMulCastLong();
244             }
245             imul_distance = 0;
246             if (stack.getStackDepth() > 1) {
247                 OpcodeStack.Item item0 = stack.getStackItem(0);
248                 OpcodeStack.Item item1 = stack.getStackItem(1);
249                 imul_constant = adjustMultiplier(item0.getConstant(), imul_constant);
250                 imul_constant = adjustMultiplier(item1.getConstant(), imul_constant);
251 
252                 if (item0.isInitialParameter() || item1.isInitialParameter()) {

254                 }
255             }
256         } else {
257             imul_distance++;
258         }

260         if (prevOpCode == Const.IMUL && seen == Const.I2L) {
261             int priority = adjustPriority(imul_constant, NORMAL_PRIORITY);
262             if (priority >= LOW_PRIORITY && imul_constant != 1000 && imul_constant != 60 && imul_operand_is_parameter) {
263                 priority = NORMAL_PRIORITY;
264             }
265             if (priority <= best_priority_for_ICAST_INTEGER_MULTIPLY_CAST_TO_LONG) {
266                 best_priority_for_ICAST_INTEGER_MULTIPLY_CAST_TO_LONG = priority;
267                 bugAccumulator.accumulateBug(
268                         new BugInstance(this, "ICAST_INTEGER_MULTIPLY_CAST_TO_LONG", priority).addClassAndMethod(this), this);
269             }
270         }
271 
272         if (Const.STATIC_INITIALIZER_NAME.equals(getMethodName()) && (seen == Const.PUTSTATIC || seen == Const.GETSTATIC || seen == Const.INVOKESTATIC)) {
273             String clazz = getClassConstantOperand();
274             if (!clazz.equals(getClassName())) {
275                 try {
276                     JavaClass targetClass = Repository.lookupClass(clazz);
277                     if (Repository.instanceOf(targetClass, getThisClass())) {
278                         int priority = NORMAL_PRIORITY;
279                         if (seen == Const.GETSTATIC) {
280                             priority--;
281                         }
282                         if (!targetClass.isPublic()) {
283                             priority++;
284                         }
285                         bugAccumulator.accumulateBug(new BugInstance(this, "IC_SUPERCLASS_USES_SUBCLASS_DURING_INITIALIZATION",

287 
288                     }
289                 } catch (ClassNotFoundException e) {
290                     // ignore it
291                 }
292 
293             }
294         }

332                 && getPrevOpcode(1) == Const.IUSHR && !shiftOfNonnegativeValue

335                     new BugInstance(this, "ICAST_QUESTIONABLE_UNSIGNED_RIGHT_SHIFT", NORMAL_PRIORITY).addClassAndMethod(this),
336                     this);
337         }
338 
339 
340         if (seen == Const.IADD && (getNextOpcode() == Const.ISHL || getNextOpcode() == Const.LSHL) && stack.getStackDepth() >=3) {
341             OpcodeStack.Item l = stack.getStackItem(2);
342             OpcodeStack.Item v = stack.getStackItem(1);
343             Object constantValue = v.getConstant();
344             // Ignore 1 << (const + var) as it's usually intended
345             if (constantValue instanceof Integer && !Integer.valueOf(1).equals(l.getConstant())) {
346                 int c = ((Integer) constantValue).intValue();
347                 int priority = LOW_PRIORITY;
348                 // If (foo << 32 + var) encountered, then ((foo << 32) + var) is absolutely meaningless,
349                 // but (foo << (32 + var)) can be meaningful for negative var values
350                 if (c < 32 || (c < 64 && getNextOpcode() == Const.LSHL)) {
351                     if (c == 8) {
352                         priority--;
353                     }
354                     if (getPrevOpcode(1) == Const.IAND) {
355                         priority--;
356                     }
357                     if (getMethodName().equals("hashCode") && getMethodSig().equals("()I")
358                             && (getCode().getCode()[getNextPC() + 1] & 0xFF) == Const.IRETURN) {
359                         // commonly observed error is hashCode body like "return foo << 16 + bar;"
360                         priority = HIGH_PRIORITY;
361                     }
362                     bugAccumulator.accumulateBug(new BugInstance(this, "BSHIFT_WRONG_ADD_PRIORITY", priority)
363                     .addClassAndMethod(this)
364                     .addInt(c).describe(IntAnnotation.INT_SHIFT)
365                     .addValueSource(stack.getStackItem(2), this)
366                     .addValueSource(stack.getStackItem(0), this)
367                     , this);
368                 }
369             }
370 
371         }
372         constantArgumentToShift = false;
373         shiftOfNonnegativeValue = false;
374         if ((seen == Const.IUSHR || seen == Const.ISHR || seen == Const.ISHL)) {
375             if (stack.getStackDepth() <= 1) {
376                 // don't understand; lie so other detectors won't get concerned
377                 constantArgumentToShift = true;
378                 valueOfConstantArgumentToShift = 8;
379             } else {
380                 Object rightHandSide = stack.getStackItem(0).getConstant();
381 
382                 Object leftHandSide = stack.getStackItem(1).getConstant();
383                 shiftOfNonnegativeValue = stack.getStackItem(1).isNonNegative();
384                 if (rightHandSide instanceof Integer) {
385                     constantArgumentToShift = true;
386                     valueOfConstantArgumentToShift = ((Integer) rightHandSide);
387                     if (valueOfConstantArgumentToShift < 0 || valueOfConstantArgumentToShift >= 32) {
388                         bugAccumulator.accumulateBug(new BugInstance(this, "ICAST_BAD_SHIFT_AMOUNT",
389                                 valueOfConstantArgumentToShift < 0 ? LOW_PRIORITY : (valueOfConstantArgumentToShift == 32
390                                 && "hashCode".equals(getMethodName()) ? NORMAL_PRIORITY : HIGH_PRIORITY))
391                         .addClassAndMethod(this).addInt(valueOfConstantArgumentToShift).describe(IntAnnotation.INT_SHIFT)
392                         .addValueSource(stack.getStackItem(1), this), this);
393                     }
394                 }
395                 if (leftHandSide != null && leftHandSide instanceof Integer && ((Integer) leftHandSide) > 0) {
396                     // boring; lie so other detectors won't get concerned
397                     constantArgumentToShift = true;
398                     valueOfConstantArgumentToShift = 8;
399                 }
400 
401             }
402         }
403 
404         if (seen == Const.INVOKEVIRTUAL && stack.getStackDepth() > 0
405                 && ("java/util/Date".equals(getClassConstantOperand()) || "java/sql/Date".equals(getClassConstantOperand()))
406                 && "setMonth".equals(getNameConstantOperand()) && "(I)V".equals(getSigConstantOperand())) {
407             OpcodeStack.Item item = stack.getStackItem(0);
408             Object o = item.getConstant();
409             if (o != null && o instanceof Integer) {
410                 int v = (Integer) o;
411                 if (v < 0 || v > 11) {
412                     bugReporter.reportBug(new BugInstance(this, "DMI_BAD_MONTH", HIGH_PRIORITY).addClassAndMethod(this).addInt(v)
413                             .describe(IntAnnotation.INT_VALUE).addCalledMethod(this).addSourceLine(this));
414                 }
415             }
416         }
417 
418         if (seen == Const.INVOKEVIRTUAL && stack.getStackDepth() > 1 && "java/util/Calendar".equals(getClassConstantOperand())
419                 && "set".equals(getNameConstantOperand())
420 
421                 || seen == Const.INVOKESPECIAL && stack.getStackDepth() > 1
422                 && "java/util/GregorianCalendar".equals(getClassConstantOperand()) && Const.CONSTRUCTOR_NAME.equals(getNameConstantOperand())
423 
424                 ) {
425             String sig = getSigConstantOperand();
426             if (sig.startsWith("(III")) {
427                 int pos = sig.length() - 5;
428                 OpcodeStack.Item item = stack.getStackItem(pos);
429                 Object o = item.getConstant();
430                 if (o != null && o instanceof Integer) {
431                     int v = (Integer) o;
432                     if (v < 0 || v > 11) {
433                         bugReporter.reportBug(new BugInstance(this, "DMI_BAD_MONTH", NORMAL_PRIORITY).addClassAndMethod(this)
434                                 .addInt(v).describe(IntAnnotation.INT_VALUE).addCalledMethod(this).addSourceLine(this));
435                     }
436                 }
437             }
438         }

441                 && getRegisterOperand() == prevOpcodeIncrementedRegister) {
442             bugAccumulator.accumulateBug(
443                     new BugInstance(this, "DLS_OVERWRITTEN_INCREMENT", HIGH_PRIORITY).addClassAndMethod(this), this);
444 
445         }
446         if (seen == Const.IINC) {
447             prevOpcodeIncrementedRegister = getRegisterOperand();
448         } else {
449             prevOpcodeIncrementedRegister = -1;
450         }

453         // Look for ICONST_2 IREM ICONST_1 IF_ICMPNE L1

455         switch (badlyComputingOddState) {
456         case 0:
457             if (seen == Const.ICONST_2) {
458                 badlyComputingOddState++;
459             }
460             break;
461         case 1:
462             if (seen == Const.IREM) {
463                 OpcodeStack.Item item = stack.getStackItem(1);
464                 if (!item.isNonNegative() && item.getSpecialKind() != OpcodeStack.Item.MATH_ABS) {
465                     badlyComputingOddState++;
466                 } else {
467                     badlyComputingOddState = 0;
468                 }
469             } else {
470                 badlyComputingOddState = 0;
471             }
472             break;
473         case 2:
474             if (seen == Const.ICONST_1) {
475                 badlyComputingOddState++;
476             } else {
477                 badlyComputingOddState = 0;
478             }
479             break;
480         case 3:
481             if (seen == Const.IF_ICMPEQ || seen == Const.IF_ICMPNE) {
482                 bugAccumulator.accumulateBug(
483                         new BugInstance(this, "IM_BAD_CHECK_FOR_ODD", NORMAL_PRIORITY).addClassAndMethod(this), this);
484             }
485             badlyComputingOddState = 0;
486             break;
487         default:
488             break;
489         }

491         // Java Puzzlers, chapter 3, puzzle 12
492         if (seen == Const.INVOKEVIRTUAL
493                 && stack.getStackDepth() > 0
494                 && ("toString".equals(getNameConstantOperand()) && "()Ljava/lang/String;".equals(getSigConstantOperand())
495                         || "append".equals(getNameConstantOperand())
496                         && "(Ljava/lang/Object;)Ljava/lang/StringBuilder;".equals(getSigConstantOperand())
497                         && "java/lang/StringBuilder".equals(getClassConstantOperand())
498                         || "append".equals(getNameConstantOperand())
499                         && "(Ljava/lang/Object;)Ljava/lang/StringBuffer;".equals(getSigConstantOperand())
500                         && "java/lang/StringBuffer".equals(getClassConstantOperand()) || ("print".equals(getNameConstantOperand()) || "println".equals(getNameConstantOperand()))
501                         && "(Ljava/lang/Object;)V".equals(getSigConstantOperand()))) {
502             OpcodeStack.Item item = stack.getStackItem(0);
503             String signature = item.getSignature();
504             if (signature != null && signature.startsWith("[")) {
505                 boolean debuggingContext = "[Ljava/lang/StackTraceElement;".equals(signature);
506 
507                 if (!debuggingContext) {
508                     for (CodeException e : getCode().getExceptionTable()) {
509                         if (e.getHandlerPC() <= getPC() && e.getHandlerPC() + 30 >= getPC()) {
510                             debuggingContext = true;
511                         }
512                     }
513 
514                     for (int i = 1; !debuggingContext && i < stack.getStackDepth(); i++) {
515                         OpcodeStack.Item e = stack.getStackItem(i);
516 
517                         if (e.getSignature().indexOf("Logger") >= 0 || e.getSignature().indexOf("Exception") >= 0) {
518                             debuggingContext = true;
519                         }
520 
521                         XField f = e.getXField();
522                         if (f != null && (SYSTEM_ERR.equals(f.getFieldDescriptor()) || SYSTEM_OUT.equals(f.getFieldDescriptor()))) {
523                             debuggingContext = true;
524                         }
525                     }
526                 }
527                 //                String name = null;
528                 int reg = item.getRegisterNumber();
529                 Collection<BugAnnotation> as = new ArrayList<>();
530                 XField field = item.getXField();
531                 FieldAnnotation fieldAnnotation = null;
532                 if (field != null) {
533                     fieldAnnotation = FieldAnnotation.fromXField(field);
534                     fieldAnnotation.setDescription(FieldAnnotation.LOADED_FROM_ROLE);
535                 }
536 
537                 if (reg != -1) {
538                     LocalVariableAnnotation lva = LocalVariableAnnotation.getLocalVariableAnnotation(getMethod(), reg, getPC(),
539                             getPC() - 1);
540                     if (lva.isNamed()) {
541                         as.add(lva);
542                         if (fieldAnnotation != null) {
543                             as.add(fieldAnnotation);
544                         }
545                     } else {
546                         if (fieldAnnotation != null) {
547                             as.add(fieldAnnotation);
548                         }
549                         as.add(lva);
550                     }
551                 } else if (fieldAnnotation != null) {
552                     as.add(fieldAnnotation);
553                 } else {
554                     XMethod m = item.getReturnValueOf();
555                     if (m != null) {
556                         MethodAnnotation methodAnnotation = MethodAnnotation.fromXMethod(m);
557                         methodAnnotation.setDescription(MethodAnnotation.METHOD_RETURN_VALUE_OF);
558                         as.add(methodAnnotation);
559                     }
560                 }
561                 int priority = debuggingContext ? NORMAL_PRIORITY : HIGH_PRIORITY;
562                 if (!as.isEmpty()) {
563                     bugAccumulator.accumulateBug(new BugInstance(this, "DMI_INVOKING_TOSTRING_ON_ARRAY", priority)
564                     .addClassAndMethod(this).addAnnotations(as), this);
565                 } else {
566                     bugAccumulator.accumulateBug(
567                             new BugInstance(this, "DMI_INVOKING_TOSTRING_ON_ANONYMOUS_ARRAY", priority).addClassAndMethod(this),
568                             this);
569                 }
570             }
571         }
572 
573         if (isTigerOrHigher) {
574             if (previousMethodInvocation != null && prevOpCode == Const.INVOKEVIRTUAL && seen == Const.INVOKESTATIC) {
575                 String classNameForPreviousMethod = previousMethodInvocation.getClassName();
576                 String classNameForThisMethod = getClassConstantOperand();
577                 if (classNameForPreviousMethod.startsWith("java.lang.")
578                         && classNameForPreviousMethod.equals(classNameForThisMethod.replace('/', '.'))
579                         && previousMethodInvocation.getName().endsWith("Value")
580                         && previousMethodInvocation.getSignature().length() == 3
581                         && "valueOf".equals(getNameConstantOperand())
582                         && getSigConstantOperand().charAt(1) == previousMethodInvocation.getSignature().charAt(2)) {
583                     bugAccumulator.accumulateBug(
584                             new BugInstance(this, "BX_UNBOXING_IMMEDIATELY_REBOXED", NORMAL_PRIORITY).addClassAndMethod(this)
585                             .addCalledMethod(this),
586                             this);
587                 }
588 
589             }
590 
591             if (previousMethodInvocation != null && prevOpCode == Const.INVOKESPECIAL && seen == Const.INVOKEVIRTUAL) {
592                 String classNameForPreviousMethod = previousMethodInvocation.getClassName();
593                 String classNameForThisMethod = getClassConstantOperand();
594                 if (classNameForPreviousMethod.startsWith("java.lang.")
595                         && classNameForPreviousMethod.equals(classNameForThisMethod.replace('/', '.'))
596                         && getNameConstantOperand().endsWith("Value") && getSigConstantOperand().length() == 3) {
597                     if (getSigConstantOperand().charAt(2) == previousMethodInvocation.getSignature().charAt(1)) {
598                         bugAccumulator.accumulateBug(
599                                 new BugInstance(this, "BX_BOXING_IMMEDIATELY_UNBOXED", NORMAL_PRIORITY).addClassAndMethod(this),
600                                 this);
601                     } else {
602                         bugAccumulator.accumulateBug(new BugInstance(this, "BX_BOXING_IMMEDIATELY_UNBOXED_TO_PERFORM_COERCION",
603                                 NORMAL_PRIORITY).addClassAndMethod(this), this);
604                     }
605 
606                     ternaryConversionState = 1;
607                 } else {
608                     ternaryConversionState = 0;
609                 }
610 
611             } else if (seen == Const.INVOKEVIRTUAL) {
612                 if (getClassConstantOperand().startsWith("java/lang") && getNameConstantOperand().endsWith("Value")
613                         && getSigConstantOperand().length() == 3) {
614                     ternaryConversionState = 1;
615                 } else {
616                     ternaryConversionState = 0;
617                 }
618             } else if (ternaryConversionState == 1) {
619                 if (Const.I2L < seen && seen <= Const.I2S) {
620                     ternaryConversionState = 2;
621                 } else {
622                     ternaryConversionState = 0;
623                 }
624             } else if (ternaryConversionState == 2) {
625                 ternaryConversionState = 0;
626                 if (seen == Const.GOTO) {
627                     bugReporter.reportBug(new BugInstance(this, "BX_UNBOXED_AND_COERCED_FOR_TERNARY_OPERATOR", NORMAL_PRIORITY)
628                     .addClassAndMethod(this).addSourceLine(this));
629                 }
630             } else {
631                 ternaryConversionState = 0;
632             }
633         }
634 
635         AssertInvokedFromRun: if (seen == Const.INVOKESTATIC) {
636             if ((getNameConstantOperand().startsWith("assert") || getNameConstantOperand().startsWith("fail"))
637                     && "run".equals(getMethodName()) && implementsRunnable(getThisClass())) {
638                 int size1 = Util.getSizeOfSurroundingTryBlock(getConstantPool(), getMethod().getCode(), "java/lang/Throwable",
639                         getPC());
640                 int size2 = Util.getSizeOfSurroundingTryBlock(getConstantPool(), getMethod().getCode(), "java/lang/Error",
641                         getPC());
642                 int size3 = Util.getSizeOfSurroundingTryBlock(getConstantPool(), getMethod().getCode(),
643                         "java/lang/AssertionFailureError", getPC());
644                 int size = Math.min(Math.min(size1, size2), size3);
645                 if (size == Integer.MAX_VALUE) {
646                     String dottedClassName = getClassConstantOperand().replace('/', '.');
647                     if (!dottedClassName.startsWith("junit")) {
648                         try {
649                             JavaClass targetClass = AnalysisContext.currentAnalysisContext().lookupClass(dottedClassName);
650                             if (!targetClass.getSuperclassName().startsWith("junit")) {
651                                 break AssertInvokedFromRun;
652                             }
653                         } catch (ClassNotFoundException e) {
654                             AnalysisContext.reportMissingClass(e);
655                             break AssertInvokedFromRun;
656                         }
657                     }
658 
659                     bugAccumulator.accumulateBug(new BugInstance(this, "IJU_ASSERT_METHOD_INVOKED_FROM_RUN_METHOD",
660                             extendsThread(getThisClass()) ? NORMAL_PRIORITY : LOW_PRIORITY).addClassAndMethod(this), this);
661 
662                 }
663 
664             }
665         }
666         if (seen == Const.INVOKESPECIAL && getClassConstantOperand().startsWith("java/lang/")
667                 && Const.CONSTRUCTOR_NAME.equals(getNameConstantOperand()) && getSigConstantOperand().length() == 4) {
668             previousMethodInvocation = XFactory.createReferencedXMethod(this);
669         } else if (seen == Const.INVOKESTATIC && getClassConstantOperand().startsWith("java/lang/")
670                 && "valueOf".equals(getNameConstantOperand()) && getSigConstantOperand().length() == 4) {
671             previousMethodInvocation = XFactory.createReferencedXMethod(this);
672         } else if (seen == Const.INVOKEVIRTUAL && getClassConstantOperand().startsWith("java/lang/")
673                 && getNameConstantOperand().endsWith("Value")
674                 && getSigConstantOperand().length() == 3) {
675             previousMethodInvocation = XFactory.createReferencedXMethod(this);
676         } else {
677             previousMethodInvocation = null;
678         }
679 
680         if (testingEnabled && seen == Const.IAND || seen == Const.LAND) {
681             OpcodeStack.Item rhs = stack.getStackItem(0);
682             OpcodeStack.Item lhs = stack.getStackItem(1);
683             Object constant = rhs.getConstant();
684             OpcodeStack.Item value = lhs;
685             if (constant == null) {
686                 constant = lhs.getConstant();
687                 value = rhs;
688             }
689             if (constant instanceof Number && (seen == Const.LAND || value.getSpecialKind() == OpcodeStack.Item.RESULT_OF_L2I)) {
690                 long constantValue = ((Number) constant).longValue();
691                 if ((constantValue == 0xEFFFFFFFL || constantValue == 0xEFFFFFFFFFFFFFFFL || seen == Const.IAND
692                         && constantValue == 0xEFFFFFFF)) {
693                     bugAccumulator.accumulateBug(new BugInstance(this, "TESTING", seen == Const.LAND ? HIGH_PRIORITY : NORMAL_PRIORITY)
694                     .addClassAndMethod(this).addString("Possible failed attempt to mask lower 31 bits of an int")
695                     .addValueSource(value, this), this);
696                 }
697 
698             }
699         }
700 
701         if (seen == Const.INEG) {
702             OpcodeStack.Item top = stack.getStackItem(0);
703             XMethod m = top.getReturnValueOf();
704             if (m != null) {
705                 if ("compareTo".equals(m.getName()) || "compare".equals(m.getName())) {
706                     bugAccumulator.accumulateBug(new BugInstance(this, "RV_NEGATING_RESULT_OF_COMPARETO", NORMAL_PRIORITY)
707                     .addClassAndMethod(this)
708                     .addCalledMethod(m).addValueSource(top, this), this);
709                 }
710             }
711 
712         }
713         prevOpCode = seen;
714 
715     }

718         return Subtypes2.instanceOf(obj, "java.lang.Runnable");
719     }

722         return Subtypes2.instanceOf(obj, "java.lang.Thread");
723     }

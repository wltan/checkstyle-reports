25  package com.apple.internal.jobjc.generator.model.coders;
26  
27  import com.apple.internal.jobjc.generator.model.types.JType.JStruct;
28  import com.apple.jobjc.Coder;
29  import com.apple.jobjc.Coder.IDCoder;
30  import com.apple.jobjc.Coder.NSClassCoder;
31  import com.apple.jobjc.Coder.PointerCoder;
32  import com.apple.jobjc.Coder.PrimitivePointerCoder;
33  import com.apple.jobjc.Coder.SELCoder;
34  import com.apple.jobjc.Coder.UnknownCoder;
35  import com.apple.jobjc.Coder.VoidCoder;
36  
37  public class CoderDescriptor {
38      public static final CoderDescriptor VOID_DESC = new CoderDescriptor(VoidCoder.INST, null, null);
39  
40      final Coder coder;
41      private final String coderInstanceName;
42      final String pushName;
43      final String popName;
44  
45      public String mismatchMessage(){ return null; }
46  
47      public CoderDescriptor(final Coder coder, final String pushName, final String popName) {
48          this.coder = coder;
49          this.coderInstanceName = coder.getClass().getCanonicalName() + ".INST";
50          if (coderInstanceName == null) throw new NullPointerException();
51  
52          this.pushName = pushName;
53          this.popName = popName;
54      }
55  
56      public CoderDescriptor(final String pushName, final String popName) {
57          this.coder = null;
58          this.coderInstanceName = null;
59          this.pushName = pushName;
60          this.popName = popName;
61      }
62  
63      public CoderDescriptor(final Coder coder) { this(coder, "push", "pop"); }
64      public CoderDescriptor(){ this("push", "pop"); }
65  
66      public String getPopStatementFor(final String contextName, final String returnValueType, final String returnValueName, final String transform) {
67          return "final " + returnValueType + " " + returnValueName + " = " + "(" + returnValueType + ") " + (transform == null ? "" : transform)
68          + "(" + getCoderInstanceName() + "." + popName + "(" + contextName + "));";
69      }
70  
71      public final String getPushStatementFor(final String contextName, final String argumentName) {
72          return getCoderInstanceName() + "." + pushName + "(" + contextName + ", " + argumentName + ");";
73      }
74  
75      public String getPopAddrStatementFor(final String runtime, final String addr, final String returnValueType, final String returnValueName, final String transform) {
76          return "final " + returnValueType + " " + returnValueName + " = " + "(" + returnValueType + ") " + (transform == null ? "" : transform)
77          + "(" + getCoderInstanceName() + "." + popName + "(" + runtime + ", " + addr + "));";
78      }
79  
80      public final String getPushAddrStatementFor(final String runtime, final String addr, final String argumentName) {
81          return getCoderInstanceName() + "." + pushName + "(" + runtime + ", " + addr + ", " + argumentName + ");";
82      }
83  
84      public String getCoderInstanceName() { return coderInstanceName; }
85      public Coder getCoder() { return coder; }
86      public String getDefaultReturnValue() { return "null"; }
87  
88      //
89      // Specialized
90      //
91  
92      static public class UnknownCoderDescriptor extends CoderDescriptor {
93          public static final CoderDescriptor UNKNOWN_DESC = new UnknownCoderDescriptor();
94          public UnknownCoderDescriptor() { super(UnknownCoder.INST); }
95      }
96  
97      static public class PrimitivePointerCoderDescriptor extends CoderDescriptor {
98          public static final PrimitivePointerCoderDescriptor POINTER_DESC = new PrimitivePointerCoderDescriptor();
99          public PrimitivePointerCoderDescriptor() { super(PrimitivePointerCoder.INST, "push", "popPtr"); }
100     }
101 
102     static public class PointerCoderDescriptor extends CoderDescriptor {
103         public static final PointerCoderDescriptor INST = new PointerCoderDescriptor();
104         public PointerCoderDescriptor() { super(PointerCoder.INST); }
105     }
106 
107     static public class SELCoderDescriptor extends CoderDescriptor {
108         public static final SELCoderDescriptor INST = new SELCoderDescriptor();
109         public SELCoderDescriptor() { super(SELCoder.INST); }
110     }
111 
112     static public class IDCoderDescriptor extends CoderDescriptor {
113         public static final IDCoderDescriptor INST = new IDCoderDescriptor();
114         public IDCoderDescriptor() { super(IDCoder.INST); }
115     }
116 
117     static public class NSClassCoderDescriptor extends CoderDescriptor {
118         public static final NSClassCoderDescriptor INST = new NSClassCoderDescriptor();
119         public NSClassCoderDescriptor() { super(NSClassCoder.INST); }
120     }
121 
122     static public class StructCoderDescriptor extends CoderDescriptor {
123         public final JStruct jstruct;
124         public StructCoderDescriptor(JStruct jst) { this.jstruct = jst; }
125         @Override public String getCoderInstanceName(){ return jstruct.getJavaTypeName() + ".getStructCoder()"; }
126     }
127 }

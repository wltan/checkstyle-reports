60  package tck.java.time;
61  
62  import static org.testng.Assert.assertEquals;
63  import static org.testng.Assert.assertSame;
64  
65  import java.time.Clock;
66  import java.time.Duration;
67  import java.time.Instant;
68  import java.time.LocalDateTime;
69  import java.time.ZoneId;
70  import java.time.ZoneOffset;
71  import java.time.ZonedDateTime;
72  
73  import org.testng.annotations.Test;
74  
75  /**
76   * Test tick clock.
77   */
78  @Test
79  public class TCKClock_Tick extends AbstractTCKTest {
80  
81      private static final ZoneId MOSCOW = ZoneId.of("Europe/Moscow");
82      private static final ZoneId PARIS = ZoneId.of("Europe/Paris");
83      private static final ZonedDateTime ZDT = LocalDateTime.of(2008, 6, 30, 11, 30, 10, 500).atZone(ZoneOffset.ofHours(2));
84  
85      //-----------------------------------------------------------------------
86      public void test_tick_ClockDuration_250millis() {
87          for (int i = 0; i < 1000; i++) {
88              Clock test = Clock.tick(Clock.fixed(ZDT.withNano(i * 1000_000).toInstant(), PARIS), Duration.ofMillis(250));
89              assertEquals(test.instant(), ZDT.withNano((i / 250) * 250_000_000).toInstant());
90              assertEquals(test.getZone(), PARIS);
91          }
92      }
93  
94      public void test_tick_ClockDuration_250micros() {
95          for (int i = 0; i < 1000; i++) {
96              Clock test = Clock.tick(Clock.fixed(ZDT.withNano(i * 1000).toInstant(), PARIS), Duration.ofNanos(250_000));
97              assertEquals(test.instant(), ZDT.withNano((i / 250) * 250_000).toInstant());
98              assertEquals(test.getZone(), PARIS);
99          }
100     }
101 
102     public void test_tick_ClockDuration_20nanos() {
103         for (int i = 0; i < 1000; i++) {
104             Clock test = Clock.tick(Clock.fixed(ZDT.withNano(i).toInstant(), PARIS), Duration.ofNanos(20));
105             assertEquals(test.instant(), ZDT.withNano((i / 20) * 20).toInstant());
106             assertEquals(test.getZone(), PARIS);
107         }
108     }
109 
110     public void test_tick_ClockDuration_zeroDuration() {
111         Clock underlying = Clock.system(PARIS);
112         Clock test = Clock.tick(underlying, Duration.ZERO);
113         assertSame(test, underlying);  // spec says same
114     }
115 
116     public void test_tick_ClockDuration_1nsDuration() {
117         Clock underlying = Clock.system(PARIS);
118         Clock test = Clock.tick(underlying, Duration.ofNanos(1));
119         assertSame(test, underlying);  // spec says same
120     }
121 
122     @Test(expectedExceptions = ArithmeticException.class)
123     public void test_tick_ClockDuration_maxDuration() {
124         Clock.tick(Clock.systemUTC(), Duration.ofSeconds(Long.MAX_VALUE));
125     }
126 
127     @Test(expectedExceptions = IllegalArgumentException.class)
128     public void test_tick_ClockDuration_subMilliNotDivisible_123ns() {
129         Clock.tick(Clock.systemUTC(), Duration.ofSeconds(0, 123));
130     }
131 
132     @Test(expectedExceptions = IllegalArgumentException.class)
133     public void test_tick_ClockDuration_subMilliNotDivisible_999ns() {
134         Clock.tick(Clock.systemUTC(), Duration.ofSeconds(0, 999));
135     }
136 
137     @Test(expectedExceptions = IllegalArgumentException.class)
138     public void test_tick_ClockDuration_subMilliNotDivisible_999_999_999ns() {
139         Clock.tick(Clock.systemUTC(), Duration.ofSeconds(0, 999_999_999));
140     }
141 
142     @Test(expectedExceptions = IllegalArgumentException.class)
143     public void test_tick_ClockDuration_negative1ns() {
144         Clock.tick(Clock.systemUTC(), Duration.ofSeconds(0, -1));
145     }
146 
147     @Test(expectedExceptions = IllegalArgumentException.class)
148     public void test_tick_ClockDuration_negative1s() {
149         Clock.tick(Clock.systemUTC(), Duration.ofSeconds(-1));
150     }
151 
152     @Test(expectedExceptions = NullPointerException.class)
153     public void test_tick_ClockDuration_nullClock() {
154         Clock.tick(null, Duration.ZERO);
155     }
156 
157     @Test(expectedExceptions = NullPointerException.class)
158     public void test_tick_ClockDuration_nullDuration() {
159         Clock.tick(Clock.systemUTC(), null);
160     }
161 
162     //-----------------------------------------------------------------------
163     public void test_tickSeconds_ZoneId() throws Exception {
164         Clock test = Clock.tickSeconds(PARIS);
165         assertEquals(test.getZone(), PARIS);
166         assertEquals(test.instant().getNano(), 0);
167         Thread.sleep(100);
168         assertEquals(test.instant().getNano(), 0);
169     }
170 
171     @Test(expectedExceptions = NullPointerException.class)
172     public void test_tickSeconds_ZoneId_nullZoneId() {
173         Clock.tickSeconds(null);
174     }
175 
176     //-----------------------------------------------------------------------
177     public void test_tickMinutes_ZoneId() {
178         Clock test = Clock.tickMinutes(PARIS);
179         assertEquals(test.getZone(), PARIS);
180         Instant instant = test.instant();
181         assertEquals(instant.getEpochSecond() % 60, 0);
182         assertEquals(instant.getNano(), 0);
183     }
184 
185     @Test(expectedExceptions = NullPointerException.class)
186     public void test_tickMinutes_ZoneId_nullZoneId() {
187         Clock.tickMinutes(null);
188     }
189 
190     //-------------------------------------------------------------------------
191     public void test_withZone() {
192         Clock test = Clock.tick(Clock.system(PARIS), Duration.ofMillis(500));
193         Clock changed = test.withZone(MOSCOW);
194         assertEquals(test.getZone(), PARIS);
195         assertEquals(changed.getZone(), MOSCOW);
196     }
197 
198     public void test_withZone_equal() {
199         Clock test = Clock.tick(Clock.system(PARIS), Duration.ofMillis(500));
200         Clock changed = test.withZone(PARIS);
201         assertEquals(test, changed);
202     }
203 
204     @Test(expectedExceptions = NullPointerException.class)
205     public void test_withZone_null() {
206         Clock.tick(Clock.system(PARIS), Duration.ofMillis(500)).withZone(null);
207     }
208 
209     //-----------------------------------------------------------------------
210     public void test__equals() {
211         Clock a = Clock.tick(Clock.system(PARIS), Duration.ofMillis(500));
212         Clock b = Clock.tick(Clock.system(PARIS), Duration.ofMillis(500));
213         assertEquals(a.equals(a), true);
214         assertEquals(a.equals(b), true);
215         assertEquals(b.equals(a), true);
216         assertEquals(b.equals(b), true);
217 
218         Clock c = Clock.tick(Clock.system(MOSCOW), Duration.ofMillis(500));
219         assertEquals(a.equals(c), false);
220 
221         Clock d = Clock.tick(Clock.system(PARIS), Duration.ofMillis(499));
222         assertEquals(a.equals(d), false);
223 
224         assertEquals(a.equals(null), false);
225         assertEquals(a.equals("other type"), false);
226         assertEquals(a.equals(Clock.systemUTC()), false);
227     }
228 
229     public void test_hashCode() {
230         Clock a = Clock.tick(Clock.system(PARIS), Duration.ofMillis(500));
231         Clock b = Clock.tick(Clock.system(PARIS), Duration.ofMillis(500));
232         assertEquals(a.hashCode(), a.hashCode());
233         assertEquals(a.hashCode(), b.hashCode());
234 
235         Clock c = Clock.tick(Clock.system(MOSCOW), Duration.ofMillis(500));
236         assertEquals(a.hashCode() == c.hashCode(), false);
237 
238         Clock d = Clock.tick(Clock.system(PARIS), Duration.ofMillis(499));
239         assertEquals(a.hashCode() == d.hashCode(), false);
240     }

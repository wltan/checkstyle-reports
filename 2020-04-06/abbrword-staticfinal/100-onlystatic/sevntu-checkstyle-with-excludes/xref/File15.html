28  /**
29   * <p>
30   * This check verifies that large numeric literals are spaced by underscores.
31   * </p>
32   * <p>
33   * Java 7 allows for underscores to delimit numeric literals to enhance readability because
34   * very large numeric literals are hard to read. For example:
35   * </p>
36   * <code>
37   * long creditCardNumber = 1234_5678_1234_5678L;
38   * </code>
39   * <p>
40   * is much easier to read than
41   * </p>
42   * <code>
43   * long creditCardNumber = 1234567812345678L;
44   * </code>
45   * <p>
46   * This check comes with the following parameters:
47   * </p>
48   * <p>
49   * "minDecimalSymbolLength" - The minimum number of symbols in a decimal literal (includes int,
50   * long, float, and double) before the check begins to look for underscores. Numeric literals
51   * with delimiters like decimal points or exponentials will be split before checking the
52   * length. Default is 7.
53   * </p>
54   * <p>
55   * "maxDecimalSymbolsUntilUnderscore" - The maximum number of symbols in a decimal literal
56   * allowed before the check demands an underscore. This does not take postfixes and delimiters
57   * like decimal points and exponentials into account. Default is 3.
58   * </p>
59   * <p>
60   * "minHexSymbolLength" - The minimum number of symbols in a hex literal before the check
61   * begins to look for underscores. Numeric literals with delimiters like decimal points or
62   * exponentials will be split before checking the length. Default is 5.
63   * </p>
64   * <p>
65   * "maxHexSymbolsUntilUnderscore" - The maximum number of symbols in a hex literal allowed
66   * before the check demands an underscore. This does not take the prefix 0x, delimiters like
67   * decimal points and exponentials, and postfixes into account. Default is 4.
68   * </p>
69   * <p>
70   * "minBinarySymbolLength" - The minimum number of symbols in a binary literal before the check
71   * begins to look for underscores. Default is 9.
72   * </p>
73   * <p>
74   * "maxBinarySymbolsUntilUnderscore" - The maximum number of symbols in a binary literal
75   * allowed before the check demands an underscore. This does not take the prefix 0b and
76   * postfixes into account. Default is 8.
77   * </p>
78   * <p>
79   * Examples (assuming default parameters):
80   * </p>
81   *
82   * <pre>
83   * // Ignored because length of token is 6, which is less than minDecimalSymbolLength (7)
84   * int ignoredDecimal = 123456;
85   * // Ignored because the postfix L is not taken into account
86   * long ignoredDecimal = 123456L;
87   * // Ignored because each segment delimited by the decimal point has a length
88   * // less than minDecimalSymbolLength (7)
89   * float ignoredDecimal = 123456.123456f;
90   * // Failed because token does not have underscores every 3 characters
91   * // (maxDecimalSymbolsUntilUnderscore = 3)
92   * int failingDecimal = 1234567;
93   * double failingDecimal = 1.1234567e0d;
94   * int passingDecimal = 1_234_567;
95   * double passingDecimal = 123456.123456e0d;
96   * double passingDecimal = 1.123_456_7e0d;
97   * int ignoredHex = 0xFFFF;
98   * int failingHex = 0xFFFFFFFF;
99   * int passingHex = 0xFFFF_FFFF;
100  * float passingHex = 0xAAAA.BBBBp1f;
101  * int ignoredBinary = 0b01010101;
102  * int failingBinary = 0b0000000000000000;
103  * int passingBinary = 0b00000000_00000000;
104  * </pre>
105  * <p>
106  * An example of how to configure parameters:
107  * </p>
108  *
109  * <pre>
110  * &lt;module name="NumericLiteralNeedsUnderscoreCheck"&gt;
111  *    &lt;property name="minDecimalSymbolLength" value="4"/&gt;
112  *    &lt;property name="maxDecimalSymbolsUntilUnderscore" value="3"/&gt;
113  *    &lt;property name="minHexSymbolLength" value="3"/&gt;
114  *    &lt;property name="maxHexSymbolsUntilUnderscore" value="2"/&gt;
115  *    &lt;property name="minBinarySymbolLength" value="5"/&gt;
116  *    &lt;property name="maxBinarySymbolsUntilUnderscore" value="4"/&gt;
117  * &lt;/module&gt;
118  * </pre>
119  * <p>
120  * Examples (assuming above parameters):
121  * </p>
122  *
123  * <pre>
124  * // Ignored because length of token is 3, which is less than minDecimalSymbolLength (4)
125  * int ignoredDecimal = 123;
126  * // Ignored because each segment delimited by the decimal point has a length
127  * // less than minDecimalSymbolLength (4)
128  * float ignoredDecimal = 123.123f;
129  * // Failed because token does not have underscores every 3 characters
130  * // (maxDecimalSymbolsUntilUnderscore = 3)
131  * int failingDecimal = 1234;
132  * int passingDecimal = 1_234;
133  * int ignoredHex = 0xFF;
134  * int failingHex = 0xFFFF;
135  * int passingHex = 0xFF_FF;
136  * int ignoredBinary = 0b0101;
137  * int failingBinary = 0b00001111;
138  * int passingBinary = 0b0000_1111;
139  * </pre>
140  * @author Cheng-Yu Pai
141  * @since 1.18.0
142  */
143

360     /**
361      * Checks if the provided token is a field.
362      * @param ast
363      *        the token to check
364      * @return whether or not the token is a field
365      */
366     private static boolean isField(final DetailAST ast) {
367         DetailAST current = ast;
368         while (current.getParent() != null && current.getType() != TokenTypes.VARIABLE_DEF) {
369             current = current.getParent();
370         }
371         return current.getType() == TokenTypes.VARIABLE_DEF
372                 && current.findFirstToken(TokenTypes.MODIFIERS)
373                     .findFirstToken(TokenTypes.LITERAL_STATIC) != null
374                 && current.findFirstToken(TokenTypes.MODIFIERS)
375                     .findFirstToken(TokenTypes.FINAL) != null;
376     }
377 
378     /**
379      * Returns the provided field's name.
380      * @param ast
381      *        the field for which the function looks for a name
382      * @return the field's name
383      */
384     private static String getFieldName(final DetailAST ast) {
385         DetailAST current = ast;
386         while (current.getType() != TokenTypes.VARIABLE_DEF) {
387             current = current.getParent();
388         }
389         current = current.getFirstChild();
390         while (current.getType() != TokenTypes.IDENT) {
391             current = current.getNextSibling();
392         }
393         return current.getText();
394     }

396     /**
397      * Returns true if the ast passes the check.
398      * @param ast
399      *        the numeric literal to check
400      * @return if the numeric literal passes the check
401      */
402     private boolean passesCheck(final DetailAST ast) {
403         boolean passing;
404         if (isField(ast) && ignoreFieldNamePattern.matcher(getFieldName(ast)).find()) {
405             passing = true;
406         }
407         else {
408             final String rawLiteral = ast.getText();
409             final NumericType type = getNumericType(rawLiteral);
410             final int minCheckingLength = minSymbolsBeforeChecking(type);
411             final int symbolsUntilUnderscore = maxSymbolsUntilUnderscore(type);
412             final String strippedLiteral = removePrePostfixByType(rawLiteral, type);
413             final String[] numericSegments = getNumericSegments(strippedLiteral, type);
414             passing = true;
415             for (String numericSegment : numericSegments) {
416                 if (!numericSegmentPassesRequirement(numericSegment,
417                         minCheckingLength, symbolsUntilUnderscore)) {
418                     passing = false;
419                     break;
420                 }
421             }
422         }
423         return passing;
424     }

426     /**
427      * Parses the numeric literal to return the minimum checking length for the literal's type.
428      * @param type
429      *        the type of numerical literal
430      * @return minimum length before checking
431      */
432     private int minSymbolsBeforeChecking(NumericType type) {
433         final int minLength;
434         if (type.equals(NumericType.DECIMAL)) {
435             minLength = minDecimalSymbolLength;
436         }
437         else if (type.equals(NumericType.HEX)) {
438             minLength = minHexSymbolLength;
439         }
440         else if (type.equals(NumericType.BINARY)) {
441             minLength = minBinarySymbolLength;
442         }
443         else {
444             throw new IllegalStateException(UNEXPECTED_NUMERIC_TYPE_ERROR
445                     + type.toString());
446         }
447         return minLength;
448     }

450     /**
451      * Parses the numeric literal to return the maximum number of characters before there must
452      * be an underscore for the literal's type.
453      * @param type
454      *        the type of numerical literal
455      * @return maximum number of characters between underscores
456      */
457     private int maxSymbolsUntilUnderscore(NumericType type) {
458         final int maxSymbols;
459         if (type.equals(NumericType.DECIMAL)) {
460             maxSymbols = maxDecimalSymbolsUntilUnderscore;
461         }
462         else if (type.equals(NumericType.HEX)) {
463             maxSymbols = maxHexSymbolsUntilUnderscore;
464         }
465         else if (type.equals(NumericType.BINARY)) {
466             maxSymbols = maxBinarySymbolsUntilUnderscore;
467         }
468         else {
469             throw new IllegalStateException(UNEXPECTED_NUMERIC_TYPE_ERROR
470                     + type.toString());
471         }
472         return maxSymbols;
473     }

475     /**
476      * <p>
477      * Generates easily checkable numeric tokens from the raw literal text, assuming
478      * the numeric type provided.
479      * </p>
480      * <p>
481      * For example: 123.4567 passes because each section itself is not too long and is
482      * perfectly readable.
483      * </p>
484      * Additionally, Java will not compile underscores next to decimal points etc.
485      * @param strippedLiteral
486      *        numeric literal stripped of any prefixes and postfixes
487      * @param type
488      *        the numeric type of the literal
489      * @return numeric tokens (segments) without non-numeric characters
490      */
491     private static String[] getNumericSegments(String strippedLiteral, NumericType type) {
492         final String[] numericSegments;
493         if (type.equals(NumericType.DECIMAL)) {
494             numericSegments = DECIMAL_SPLITTER.split(strippedLiteral);
495         }
496         else if (type.equals(NumericType.HEX)) {
497             numericSegments = HEX_SPLITTER.split(strippedLiteral);
498         }
499         else if (type.equals(NumericType.BINARY)) {
500             numericSegments = new String[1];
501             numericSegments[0] = strippedLiteral;
502         }
503         else {
504             throw new IllegalStateException(UNEXPECTED_NUMERIC_TYPE_ERROR
505                     + type.toString());
506         }
507         return numericSegments;
508     }

510     /**
511      * <p>
512      * Returns the type of numeric literal given the raw text.
513      * </p>
514      * <p>
515      * Decimal literals are normal numbers. Example: 1, 1.0, 1.0e0f, 1L
516      * </p>
517      * <p>
518      * Hex literals are preceded by 0x. Example: 0xDEADBEEF, 0xA.Bp1d
519      * </p>
520      * <p>
521      * Binary literals are preceded by 0b. Example: 0b00001111
522      * </p>
523      * @param rawLiteral
524      *        numeric literal
525      * @return the type of literal (either decimal, hex, or binary)
526      */
527     private static NumericType getNumericType(String rawLiteral) {
528         final NumericType type;
529         if (rawLiteral.length() < PREFIX_LENGTH) {
530             type = NumericType.DECIMAL;
531         }
532         else {
533             final String prefix = rawLiteral.substring(0, PREFIX_LENGTH);
534             if ("0x".equals(prefix)) {
535                 type = NumericType.HEX;
536             }
537             else if ("0b".equals(prefix)) {
538                 type = NumericType.BINARY;
539             }
540             else {
541                 type = NumericType.DECIMAL;
542             }
543         }
544         return type;
545     }

547     /**
548      * Returns whether or not the text passes the underscore requirement given the text and
549      * minimum length.
550      * @param numericSegment
551      *        the numeric segment to check
552      * @param minLength
553      *        minimum length for the numericSegment
554      * @param symbolsUntilUnderscore
555      *        maximum number of characters until there must be an underscore
556      * @return whether or not the numeric segment passes the requirements
557      */
558     private static boolean numericSegmentPassesRequirement(String numericSegment,
559             int minLength, int symbolsUntilUnderscore) {
560         boolean passes = true;
561         if (numericSegment.length() >= minLength) {
562             final char underscore = '_';
563             int symbolCount = 0;
564 
565             for (int i = 0; i < numericSegment.length(); i++) {
566                 final char current = numericSegment.charAt(i);
567                 if (symbolCount >= symbolsUntilUnderscore && current != underscore) {
568                     passes = false;
569                     break;
570                 }
571                 if (current == underscore) {
572                     symbolCount = 0;
573                 }
574                 else {
575                     symbolCount++;
576                 }
577             }
578         }
579         return passes;
580     }

582     /**
583      * Removes 0x, 0b prefixes, and l, L, f, F, d, D postfixes from numeric literals.
584      * @param rawLiteral
585      *        the numeric literal that needs to be stripped of prefixes and postfixes
586      * @param literalType
587      *        the type of the literal being passed in
588      * @return a stripped version of the raw literal
589      */
590     private static String removePrePostfixByType(String rawLiteral, NumericType literalType) {
591         String processedLiteral;
592         if (literalType.equals(NumericType.DECIMAL)) {
593             processedLiteral = removeLetterPostfix(rawLiteral);
594         }
595         else if (literalType.equals(NumericType.HEX)) {
596             processedLiteral = removePrefix(rawLiteral);
597             processedLiteral = removePostfixHex(processedLiteral);
598         }
599         else if (literalType.equals(NumericType.BINARY)) {
600             processedLiteral = removePrefix(rawLiteral);
601             processedLiteral = removeLetterPostfix(processedLiteral);
602         }
603         else {
604             throw new IllegalStateException(UNEXPECTED_NUMERIC_TYPE_ERROR
605                     + literalType.toString());
606         }
607         return processedLiteral;
608     }

610     /**
611      * Removes the prefixes 0x and 0b.
612      * @param text
613      *        the text to remove the prefixes
614      * @return the text without the prefixes
615      */
616     private static String removePrefix(String text) {
617         return text.substring(PREFIX_LENGTH);
618     }
619 
620     /**
621      * Removes the postfix from the text if it exists. Does not handle hex literals correctly,
622      * for that use removePostfixHex.
623      * @param text
624      *        the text to remove the postfixes
625      * @return the text without the postfixes
626      */
627     private static String removeLetterPostfix(String text) {
628         final char lastchar = text.charAt(text.length() - 1);
629         final String noPostfixText;
630         if (Character.isDigit(lastchar)) {
631             noPostfixText = text;
632         }
633         else {
634             noPostfixText = text.substring(0, text.length() - 1);
635         }
636         return noPostfixText;
637     }
638 
639     /**
640      * Removes the postfix from the hex literal text if it exists. Does not handle other
641      * literals correctly, for those use removeLetterPostfix.
642      * @param text
643      *        the text to remove the postfixes
644      * @return the text without the postfixes
645      */
646     private static String removePostfixHex(String text) {
647         final char lastchar = Character.toUpperCase(text.charAt(text.length() - 1));
648         boolean hasPostfix = false;
649         if (lastchar == 'L') {
650             // Example: 0x00FFL
651             hasPostfix = true;
652         }
653         else if (lastchar == 'F' && (text.contains("p") || text.contains("P"))) {
654             // Example: 0x1.0p1f (Hex Float)
655             hasPostfix = true;
656         }
657         final String noPostfixText;
658         if (hasPostfix) {
659             noPostfixText = text.substring(0, text.length() - 1);
660         }
661         else {
662             noPostfixText = text;
663         }
664         return noPostfixText;
665     }

24  package org.hibernate.dialect;
25  
26  import java.sql.Types;
27  
28  import org.hibernate.type.descriptor.sql.BlobTypeDescriptor;
29  import org.hibernate.type.descriptor.sql.ClobTypeDescriptor;
30  import org.hibernate.type.descriptor.sql.SqlTypeDescriptor;
31  
32  
33  /**
34   * All Sybase dialects share an IN list size limit.
35   *
36   * @author Brett Meyer
37   */
38  public class SybaseDialect extends AbstractTransactSQLDialect {
39  	
40  	private static final int PARAM_LIST_SIZE_LIMIT = 250000;
41  
42  	/* (non-Javadoc)
43  		 * @see org.hibernate.dialect.Dialect#getInExpressionCountLimit()
44  		 */
45  	@Override
46  	public int getInExpressionCountLimit() {
47  		return PARAM_LIST_SIZE_LIMIT;
48  	}
49  	
50  	@Override
51  	protected SqlTypeDescriptor getSqlTypeDescriptorOverride(int sqlCode) {
52  		switch (sqlCode) {
53  		case Types.BLOB:
54  			return BlobTypeDescriptor.PRIMITIVE_ARRAY_BINDING;
55  		case Types.CLOB:
56  			// Some Sybase drivers cannot support getClob.  See HHH-7889
57  			return ClobTypeDescriptor.STREAM_BINDING_EXTRACTING;
58  		default:
59  			return super.getSqlTypeDescriptorOverride( sqlCode );
60  		}
61  	}
62  	
63  	@Override
64  	public String getNullColumnString() {
65  		return " null";
66  	}
67  }

215 	public String getAddForeignKeyConstraintString(
216 			String constraintName, 
217 			String[] foreignKey, 
218 			String referencedTable, 
219 			String[] primaryKey, boolean referencesPrimaryKey
220 	) {
221 		String cols = StringHelper.join(", ", foreignKey);
222 		return new StringBuilder(30)
223 			.append(" add index ")
224 			.append(constraintName)
225 			.append(" (")
226 			.append(cols)
227 			.append("), add constraint ")
228 			.append(constraintName)
229 			.append(" foreign key (")
230 			.append(cols)
231 			.append(") references ")
232 			.append(referencedTable)
233 			.append(" (")
234 			.append( StringHelper.join(", ", primaryKey) )
235 			.append(')')
236 			.toString();
237 	}

326 	public ResultSet getResultSet(CallableStatement ps) throws SQLException {
327 		boolean isResultSet = ps.execute(); 
328 		while (!isResultSet && ps.getUpdateCount() != -1) { 
329 			isResultSet = ps.getMoreResults(); 
330 		} 
331 		return ps.getResultSet();
332 	}

338 	@Override
339 	public String renderOrderByElement(String expression, String collation, String order, NullPrecedence nulls) {
340 		final StringBuilder orderByElement = new StringBuilder();
341 		if ( nulls != NullPrecedence.NONE ) {
342 			// Workaround for NULLS FIRST / LAST support.
343 			orderByElement.append( "case when " ).append( expression ).append( " is null then " );
344 			if ( nulls == NullPrecedence.FIRST ) {
345 				orderByElement.append( "0 else 1" );
346 			}
347 			else {
348 				orderByElement.append( "1 else 0" );
349 			}
350 			orderByElement.append( " end, " );
351 		}
352 		// Nulls precedence has already been handled so passing NONE value.
353 		orderByElement.append( super.renderOrderByElement( expression, collation, order, NullPrecedence.NONE ) );
354 		return orderByElement.toString();
355 	}

391 	@Override
392 	public boolean supportsLockTimeouts() {
393 		// yes, we do handle "lock timeout" conditions in the exception conversion delegate,
394 		// but that's a hardcoded lock timeout period across the whole entire MySQL database.
395 		// MySQL does not support specifying lock timeouts as part of the SQL statement, which is really
396 		// what this meta method is asking.
397 		return false;
398 	}

400 	@Override
401 	public SQLExceptionConversionDelegate buildSQLExceptionConversionDelegate() {
402 		return new SQLExceptionConversionDelegate() {
403 			@Override
404 			public JDBCException convert(SQLException sqlException, String message, String sql) {
405 				final String sqlState = JdbcExceptionHelper.extractSqlState( sqlException );
406 
407 				if ( "41000".equals( sqlState ) ) {
408 					return new LockTimeoutException( message, sqlException, sql );
409 				}
410 
411 				if ( "40001".equals( sqlState ) ) {
412 					return new LockAcquisitionException( message, sqlException, sql );
413 				}
414 
415 				return null;
416 			}
417 		};
418 	}

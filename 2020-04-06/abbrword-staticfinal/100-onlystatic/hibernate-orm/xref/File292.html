126 	/**
127 	 * Takes the where condition provided in the mapping attribute and interpolates the alias.
128 	 * Handles sub-selects, quoted identifiers, quoted strings, expressions, SQL functions,
129 	 * named parameters.
130 	 *
131 	 * @param sqlWhereString The string into which to interpolate the placeholder value
132 	 * @param placeholder The value to be interpolated into the the sqlWhereString
133 	 * @param dialect The dialect to apply
134 	 * @param functionRegistry The registry of all sql functions
135 	 * @return The rendered sql fragment
136 	 */
137 	public static String renderWhereStringTemplate(String sqlWhereString, String placeholder, Dialect dialect, SQLFunctionRegistry functionRegistry ) {
138 
139 		// IMPL NOTE : The basic process here is to tokenize the incoming string and to iterate over each token
140 		//		in turn.  As we process each token, we set a series of flags used to indicate the type of context in
141 		// 		which the tokens occur.  Depending on the state of those flags we decide whether we need to qualify
142 		//		identifier references.
143 
144 		String symbols = new StringBuilder()
145 				.append( "=><!+-*/()',|&`" )
146 				.append( StringHelper.WHITESPACE )
147 				.append( dialect.openQuote() )
148 				.append( dialect.closeQuote() )
149 				.toString();
150 		StringTokenizer tokens = new StringTokenizer( sqlWhereString, symbols, true );
151 		StringBuilder result = new StringBuilder();
152 
153 		boolean quoted = false;
154 		boolean quotedIdentifier = false;
155 		boolean beforeTable = false;
156 		boolean inFromClause = false;
157 		boolean afterFromTable = false;
158 
159 		boolean hasMore = tokens.hasMoreTokens();
160 		String nextToken = hasMore ? tokens.nextToken() : null;
161 		while ( hasMore ) {
162 			String token = nextToken;
163 			String lcToken = token.toLowerCase();
164 			hasMore = tokens.hasMoreTokens();
165 			nextToken = hasMore ? tokens.nextToken() : null;
166 
167 			boolean isQuoteCharacter = false;
168 
169 			if ( !quotedIdentifier && "'".equals(token) ) {
170 				quoted = !quoted;
171 				isQuoteCharacter = true;
172 			}
173 
174 			if ( !quoted ) {
175 				boolean isOpenQuote;
176 				if ( "`".equals(token) ) {
177 					isOpenQuote = !quotedIdentifier;
178 					token = lcToken = isOpenQuote
179 							? Character.toString( dialect.openQuote() )
180 							: Character.toString( dialect.closeQuote() );
181 					quotedIdentifier = isOpenQuote;
182 					isQuoteCharacter = true;
183 				}
184 				else if ( !quotedIdentifier && ( dialect.openQuote()==token.charAt(0) ) ) {
185 					isOpenQuote = true;
186 					quotedIdentifier = true;
187 					isQuoteCharacter = true;
188 				}
189 				else if ( quotedIdentifier && ( dialect.closeQuote()==token.charAt(0) ) ) {
190 					quotedIdentifier = false;
191 					isQuoteCharacter = true;
192 					isOpenQuote = false;
193 				}
194 				else {
195 					isOpenQuote = false;
196 				}
197 
198 				if ( isOpenQuote ) {
199 					result.append( placeholder ).append( '.' );
200 				}
201 			}
202 
203 			// Special processing for ANSI SQL EXTRACT function
204 			if ( "extract".equals( lcToken ) && "(".equals( nextToken ) ) {
205 				final String field = extractUntil( tokens, "from" );
206 				final String source = renderWhereStringTemplate(
207 						extractUntil( tokens, ")" ),
208 						placeholder,
209 						dialect,
210 						functionRegistry
211 				);
212 				result.append( "extract(" ).append( field ).append( " from " ).append( source ).append( ')' );
213 
214 				hasMore = tokens.hasMoreTokens();
215 				nextToken = hasMore ? tokens.nextToken() : null;
216 
217 				continue;
218 			}
219 
220 			// Special processing for ANSI SQL TRIM function
221 			if ( "trim".equals( lcToken ) && "(".equals( nextToken ) ) {
222 				List<String> operands = new ArrayList<String>();
223 				StringBuilder builder = new StringBuilder();
224 
225 				boolean hasMoreOperands = true;
226 				String operandToken = tokens.nextToken();
227 				boolean quotedOperand = false;
228 				while ( hasMoreOperands ) {
229 					final boolean isQuote = "'".equals( operandToken );
230 					if ( isQuote ) {
231 						quotedOperand = !quotedOperand;
232 						if ( !quotedOperand ) {
233 							operands.add( builder.append( '\'' ).toString() );
234 							builder.setLength( 0 );
235 						}
236 						else {
237 							builder.append( '\'' );
238 						}
239 					}
240 					else if ( quotedOperand ) {
241 						builder.append( operandToken );
242 					}
243 					else if ( operandToken.length() == 1 && Character.isWhitespace( operandToken.charAt( 0 ) ) ) {
244 						// do nothing
245 					}
246 					else {
247 						operands.add( operandToken );
248 					}
249 					operandToken = tokens.nextToken();
250 					hasMoreOperands = tokens.hasMoreTokens() && ! ")".equals( operandToken );
251 				}
252 
253 				TrimOperands trimOperands = new TrimOperands( operands );
254 				result.append( "trim(" );
255 				if ( trimOperands.trimSpec != null ) {
256 					result.append( trimOperands.trimSpec ).append( ' ' );
257 				}
258 				if ( trimOperands.trimChar != null ) {
259 					if ( trimOperands.trimChar.startsWith( "'" ) && trimOperands.trimChar.endsWith( "'" ) ) {
260 						result.append( trimOperands.trimChar );
261 					}
262 					else {
263 						result.append(
264 								renderWhereStringTemplate( trimOperands.trimSpec, placeholder, dialect, functionRegistry )
265 						);
266 					}
267 					result.append( ' ' );
268 				}
269 				if ( trimOperands.from != null ) {
270 					result.append( trimOperands.from ).append( ' ' );
271 				}
272 				else if ( trimOperands.trimSpec != null || trimOperands.trimChar != null ) {
273 					// I think ANSI SQL says that the 'from' is not optional if either trim-spec or trim-char are specified
274 					result.append( "from " );
275 				}
276 
277 				result.append( renderWhereStringTemplate( trimOperands.trimSource, placeholder, dialect, functionRegistry ) )
278 						.append( ')' );
279 
280 				hasMore = tokens.hasMoreTokens();
281 				nextToken = hasMore ? tokens.nextToken() : null;
282 
283 				continue;
284 			}
285 
286 			boolean quotedOrWhitespace = quoted || quotedIdentifier || isQuoteCharacter
287 					|| Character.isWhitespace( token.charAt(0) );
288 
289 			if ( quotedOrWhitespace ) {
290 				result.append( token );
291 			}
292 			else if ( beforeTable ) {
293 				result.append( token );
294 				beforeTable = false;
295 				afterFromTable = true;
296 			}
297 			else if ( afterFromTable ) {
298 				if ( !"as".equals(lcToken) ) {
299 					afterFromTable = false;
300 				}
301 				result.append(token);
302 			}
303 			else if ( isNamedParameter(token) ) {
304 				result.append(token);
305 			}
306 			else if ( isIdentifier(token, dialect)
307 					&& !isFunctionOrKeyword(lcToken, nextToken, dialect , functionRegistry) ) {
308 				result.append(placeholder)
309 						.append('.')
310 						.append( dialect.quote(token) );
311 			}
312 			else {
313 				if ( BEFORE_TABLE_KEYWORDS.contains(lcToken) ) {
314 					beforeTable = true;
315 					inFromClause = true;
316 				}
317 				else if ( inFromClause && ",".equals(lcToken) ) {
318 					beforeTable = true;
319 				}
320 				result.append(token);
321 			}
322 
323 			//Yuck:
324 			if ( inFromClause
325 					&& KEYWORDS.contains( lcToken ) //"as" is not in KEYWORDS
326 					&& !BEFORE_TABLE_KEYWORDS.contains( lcToken ) ) {
327 				inFromClause = false;
328 			}
329 		}
330 
331 		return result.toString();
332 	}

566 	private static class TrimOperands {
567 		private final String trimSpec;
568 		private final String trimChar;
569 		private final String from;
570 		private final String trimSource;
571 
572 		private TrimOperands(List<String> operands) {
573 			if ( operands.size() == 1 ) {
574 				trimSpec = null;
575 				trimChar = null;
576 				from = null;
577 				trimSource = operands.get(0);
578 			}
579 			else if ( operands.size() == 4 ) {
580 				trimSpec = operands.get(0);
581 				trimChar = operands.get(1);
582 				from = operands.get(2);
583 				trimSource = operands.get(3);
584 			}
585 			else {
586 				if ( operands.size() < 1 || operands.size() > 4 ) {
587 					throw new HibernateException( "Unexpected number of trim function operands : " + operands.size() );
588 				}
589 
590 				// trim-source will always be the last operand
591 				trimSource = operands.get( operands.size() - 1 );
592 
593 				// ANSI SQL says that more than one operand means that the FROM is required
594 				if ( ! "from".equals( operands.get( operands.size() - 2 ) ) ) {
595 					throw new HibernateException( "Expecting FROM, found : " + operands.get( operands.size() - 2 ) );
596 				}
597 				from = operands.get( operands.size() - 2 );
598 
599 				// trim-spec, if there is one will always be the first operand
600 				if ( "leading".equalsIgnoreCase( operands.get(0) )
601 						|| "trailing".equalsIgnoreCase( operands.get(0) )
602 						|| "both".equalsIgnoreCase( operands.get(0) ) ) {
603 					trimSpec = operands.get(0);
604 					trimChar = null;
605 				}
606 				else {
607 					trimSpec = null;
608 					if ( operands.size() - 2 == 0 ) {
609 						trimChar = null;
610 					}
611 					else {
612 						trimChar = operands.get( 0 );
613 					}
614 				}
615 			}
616 		}
617 	}

619 	private static String extractUntil(StringTokenizer tokens, String delimiter) {
620 		StringBuilder valueBuilder = new StringBuilder();
621 		String token = tokens.nextToken();
622 		while ( ! delimiter.equalsIgnoreCase( token ) ) {
623 			valueBuilder.append( token );
624 			token = tokens.nextToken();
625 		}
626 		return valueBuilder.toString().trim();
627 	}

729 	private static boolean isFunctionOrKeyword(String lcToken, String nextToken, Dialect dialect, SQLFunctionRegistry functionRegistry) {
730 		return "(".equals(nextToken) ||
731 			KEYWORDS.contains(lcToken) ||
732 			isFunction(lcToken, nextToken, functionRegistry ) ||
733 			dialect.getKeywords().contains(lcToken) ||
734 			FUNCTION_KEYWORDS.contains(lcToken);
735 	}

737 	private static boolean isFunction(String lcToken, String nextToken, SQLFunctionRegistry functionRegistry) {
738 		// checking for "(" is currently redundant because it is checked before getting here;
739 		// doing the check anyhow, in case that earlier check goes away;
740 		if ( "(".equals( nextToken ) ) {
741 			return true;
742 		}
743 		SQLFunction function = functionRegistry.findSQLFunction(lcToken);
744 		if ( function == null ) {
745 			// lcToken does not refer to a function
746 			return false;
747 		}
748 		// if function.hasParenthesesIfNoArguments() is true, then assume
749 		// lcToken is not a function (since it is not followed by '(')
750 		return ! function.hasParenthesesIfNoArguments();
751 	}

753 	private static boolean isIdentifier(String token, Dialect dialect) {
754 		return token.charAt(0)=='`' || ( //allow any identifier quoted with backtick
755 			Character.isLetter( token.charAt(0) ) && //only recognizes identifiers beginning with a letter
756 			token.indexOf('.') < 0
757 		);
758 	}

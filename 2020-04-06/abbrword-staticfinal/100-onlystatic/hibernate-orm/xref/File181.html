54  		for ( int i=0; i<string.length(); i++ ) {
55  			char character = string.charAt(i);
56  			// Include "_".  See HHH-8073
57  			if ( !Character.isLetter(character) && !('_'==character) ) return i-1;
58  		}
59  		return string.length()-1;
60  	}

140 								 String placeholder,

144 		if ( template == null ) {
145 			return template;
146 		}

148 		if ( loc < 0 ) {

153 			String afterPlaceholder = template.substring( loc + placeholder.length() );

155 		}

159 	public static String replace(String beforePlaceholder,
160 								 String afterPlaceholder,
161 								 String placeholder,
162 								 String replacement,
163 								 boolean wholeWords,
164 								 boolean encloseInParensIfNecessary) {
165 		final boolean actuallyReplace =
166 				! wholeWords ||
167 				afterPlaceholder.length() == 0 ||
168 				! Character.isJavaIdentifierPart( afterPlaceholder.charAt( 0 ) );
169 		boolean encloseInParens =
170 				actuallyReplace &&
171 				encloseInParensIfNecessary &&
172 				! ( getLastNonWhitespaceCharacter( beforePlaceholder ) == '(' ) &&
173 				! ( getFirstNonWhitespaceCharacter( afterPlaceholder ) == ')' );		
174 		StringBuilder buf = new StringBuilder( beforePlaceholder );
175 		if ( encloseInParens ) {
176 			buf.append( '(' );
177 		}
178 		buf.append( actuallyReplace ? replacement : placeholder );
179 		if ( encloseInParens ) {
180 			buf.append( ')' );
181 		}
182 		buf.append(
183 				replace(
184 						afterPlaceholder,
185 						placeholder,
186 						replacement,
187 						wholeWords,
188 						encloseInParensIfNecessary
189 				)
190 		);
191 		return buf.toString();
192 	}

259 	/**
260 	 * Collapses a name.  Mainly intended for use with classnames, where an example might serve best to explain.
261 	 * Imagine you have a class named <samp>'org.hibernate.internal.util.StringHelper'</samp>; calling collapse on that
262 	 * classname will result in <samp>'o.h.u.StringHelper'<samp>.
263 	 *
264 	 * @param name The name to collapse.
265 	 * @return The collapsed name.
266 	 */
267 	public static String collapse(String name) {
268 		if ( name == null ) {
269 			return null;
270 		}
271 		int breakPoint = name.lastIndexOf( '.' );
272 		if ( breakPoint < 0 ) {
273 			return name;
274 		}
275 		return collapseQualifier( name.substring( 0, breakPoint ), true ) + name.substring( breakPoint ); // includes last '.'
276 	}

278 	/**
279 	 * Given a qualifier, collapse it.
280 	 *
281 	 * @param qualifier The qualifier to collapse.
282 	 * @param includeDots Should we include the dots in the collapsed form?
283 	 *
284 	 * @return The collapsed form.
285 	 */
286 	public static String collapseQualifier(String qualifier, boolean includeDots) {
287 		StringTokenizer tokenizer = new StringTokenizer( qualifier, "." );
288 		String collapsed = Character.toString( tokenizer.nextToken().charAt( 0 ) );
289 		while ( tokenizer.hasMoreTokens() ) {
290 			if ( includeDots ) {
291 				collapsed += '.';
292 			}
293 			collapsed += tokenizer.nextToken().charAt( 0 );
294 		}
295 		return collapsed;
296 	}

315 	 * Cross between {@link #collapse} and {@link #partiallyUnqualify}.  Functions much like {@link #collapse}
316 	 * except that only the qualifierBase is collapsed.  For example, with a base of 'org.hibernate' the name
317 	 * 'org.hibernate.internal.util.StringHelper' would become 'o.h.util.StringHelper'.
318 	 *
319 	 * @param name The (potentially) qualified name.
320 	 * @param qualifierBase The qualifier base.

328 		return collapseQualifier( qualifierBase, true ) + name.substring( qualifierBase.length() );

378 		String[] results = new String[replacements.length * strings.length];
379 		int n = 0;
380 		for ( int i = 0; i < replacements.length; i++ ) {
381 			for ( int j = 0; j < strings.length; j++ ) {
382 				results[n++] = replaceOnce( strings[j], placeholder, replacements[i] );
383 			}
384 		}
385 		return results;

388 	public static int countUnquoted(String string, char character) {
389 		if ( '\'' == character ) {
390 			throw new IllegalArgumentException( "Unquoted count of quotes is invalid" );
391 		}
392 		if (string == null)
393 			return 0;
394 		// Impl note: takes advantage of the fact that an escpaed single quote
395 		// embedded within a quote-block can really be handled as two seperate
396 		// quote-blocks for the purposes of this method...
397 		int count = 0;
398 		int stringLength = string.length();
399 		boolean inQuote = false;
400 		for ( int indx = 0; indx < stringLength; indx++ ) {
401 			char c = string.charAt( indx );
402 			if ( inQuote ) {
403 				if ( '\'' == c ) {
404 					inQuote = false;
405 				}
406 			}
407 			else if ( '\'' == c ) {
408 				inQuote = true;
409 			}
410 			else if ( c == character ) {
411 				count++;
412 			}
413 		}
414 		return count;
415 	}

544 	/**
545 	 * Generates a root alias by truncating the "root name" defined by
546 	 * the incoming decription and removing/modifying any non-valid
547 	 * alias characters.
548 	 *
549 	 * @param description The root name from which to generate a root alias.
550 	 * @return The generated root alias.
551 	 */
552 	private static String generateAliasRoot(String description) {
553 		String result = truncate( unqualifyEntityName(description), ALIAS_TRUNCATE_LENGTH )
554 				// Important to use Locale.ENGLISH.  See HHH-8579.  #toLowerCase() uses the default Locale.  Certain DBs
555 				// do not like non-ascii characters in aliases, etc., so ensure consistency/portability here.
556 				.toLowerCase(Locale.ENGLISH)
557 		        .replace( '/', '_' ) // entityNames may now include slashes for the representations
558 				.replace( '$', '_' ); //classname may be an inner class
559 		result = cleanAlias( result );
560 		if ( Character.isDigit( result.charAt(result.length()-1) ) ) {
561 			return result + "x"; //ick!
562 		}
563 		else {
564 			return result;
565 		}
566 	}

568 	/**
569 	 * Clean the generated alias by removing any non-alpha characters from the
570 	 * beginning.
571 	 *
572 	 * @param alias The generated alias to be cleaned.
573 	 * @return The cleaned alias, stripped of any leading non-alpha characters.
574 	 */
575 	private static String cleanAlias(String alias) {
576 		char[] chars = alias.toCharArray();
577 		// short cut check...
578 		if ( !Character.isLetter( chars[0] ) ) {
579 			for ( int i = 1; i < chars.length; i++ ) {
580 				// as soon as we encounter our first letter, return the substring
581 				// from that position
582 				if ( Character.isLetter( chars[i] ) ) {
583 					return alias.substring( i );
584 				}
585 			}
586 		}
587 		return alias;
588 	}

629 	/**
630 	 * Return a representation of the given name ensuring quoting (wrapped with '`' characters).  If already wrapped
631 	 * return name.
632 	 *
633 	 * @param name The name to quote.
634 	 * @return The quoted version.
635 	 */
636 	public static String quote(String name) {
637 		if ( isEmpty( name ) || isQuoted( name ) ) {
638 			return name;
639 		}
640 		// Convert the JPA2 specific quoting character (double quote) to Hibernate's (back tick)
641         else if ( name.startsWith( "\"" ) && name.endsWith( "\"" ) ) {
642             name = name.substring( 1, name.length() - 1 );
643         }
644 
645 		return new StringBuilder( name.length() + 2 ).append('`').append( name ).append( '`' ).toString();
646 	}

651 	 * @param name The name to be unquoted.
652 	 * @return The unquoted version.
653 	 */
654 	public static String unquote(String name) {
655 		return isQuoted( name ) ? name.substring( 1, name.length() - 1 ) : name;
656 	}

658 	/**
659 	 * Determine if the given name is quoted.  It is considered quoted if either:
660 	 * <ol>
661 	 * <li>starts AND ends with backticks (`)</li>
662 	 * <li>starts with dialect-specified {@link org.hibernate.dialect.Dialect#openQuote() open-quote}
663 	 * 		AND ends with dialect-specified {@link org.hibernate.dialect.Dialect#closeQuote() close-quote}</li>
664 	 * </ol>
665 	 *
666 	 * @param name The name to check
667 	 * @param dialect The dialect (to determine the "real" quoting chars).
668 	 *
669 	 * @return True if quoted, false otherwise
670 	 */
671 	public static boolean isQuoted(String name, Dialect dialect) {
672 		return name != null && name.length() != 0 
673 				&& ( ( name.charAt( 0 ) == '`' && name.charAt( name.length() - 1 ) == '`' )
674 						|| ( name.charAt( 0 ) == '"' && name.charAt( name.length() - 1 ) == '"' )
675 						|| ( name.charAt( 0 ) == dialect.openQuote()
676 								&& name.charAt( name.length() - 1 ) == dialect.closeQuote() ) );
677 	}

682 	 * @param name The name to be unquoted.
683 	 * @param dialect The dialect (to determine the "real" quoting chars).
684 	 *
685 	 * @return The unquoted version.
686 	 */
687 	public static String unquote(String name, Dialect dialect) {
688 		return isQuoted( name, dialect ) ? name.substring( 1, name.length() - 1 ) : name;

706 		}
707 		return unquoted;

711 	public static final String BATCH_ID_PLACEHOLDER = "$$BATCH_ID_PLACEHOLDER$$";
712 
713 	public static StringBuilder buildBatchFetchRestrictionFragment(
714 			String alias,
715 			String[] columnNames,
716 			Dialect dialect) {
717 		// the general idea here is to just insert a placeholder that we can easily find later...
718 		if ( columnNames.length == 1 ) {
719 			// non-composite key
720 			return new StringBuilder( StringHelper.qualify( alias, columnNames[0] ) )
721 					.append( " in (" ).append( BATCH_ID_PLACEHOLDER ).append( ")" );
722 		}
723 		else {
724 			// composite key - the form to use here depends on what the dialect supports.
725 			if ( dialect.supportsRowValueConstructorSyntaxInInList() ) {
726 				// use : (col1, col2) in ( (?,?), (?,?), ... )
727 				StringBuilder builder = new StringBuilder();
728 				builder.append( "(" );
729 				boolean firstPass = true;
730 				String deliminator = "";
731 				for ( String columnName : columnNames ) {
732 					builder.append( deliminator ).append( StringHelper.qualify( alias, columnName ) );
733 					if ( firstPass ) {
734 						firstPass = false;
735 						deliminator = ",";
736 					}
737 				}
738 				builder.append( ") in (" );
739 				builder.append( BATCH_ID_PLACEHOLDER );
740 				builder.append( ")" );
741 				return builder;
742 			}
743 			else {
744 				// use : ( (col1 = ? and col2 = ?) or (col1 = ? and col2 = ?) or ... )
745 				//		unfortunately most of this building needs to be held off until we know
746 				//		the exact number of ids :(
747 				return new StringBuilder( "(" ).append( BATCH_ID_PLACEHOLDER ).append( ")" );
748 			}
749 		}
750 	}

752 	public static String expandBatchIdPlaceholder(
753 			String sql,
754 			Serializable[] ids,
755 			String alias,
756 			String[] keyColumnNames,
757 			Dialect dialect) {
758 		if ( keyColumnNames.length == 1 ) {
759 			// non-composite
760 			return StringHelper.replace( sql, BATCH_ID_PLACEHOLDER, repeat( "?", ids.length, "," ) );
761 		}
762 		else {
763 			// composite
764 			if ( dialect.supportsRowValueConstructorSyntaxInInList() ) {
765 				final String tuple = "(" + StringHelper.repeat( "?", keyColumnNames.length, "," );
766 				return StringHelper.replace( sql, BATCH_ID_PLACEHOLDER, repeat( tuple, ids.length, "," ) );
767 			}
768 			else {
769 				final String keyCheck = joinWithQualifier( keyColumnNames, alias, " and " );
770 				return replace( sql, BATCH_ID_PLACEHOLDER, repeat( keyCheck, ids.length, " or " ) );
771 			}
772 		}
773 	}

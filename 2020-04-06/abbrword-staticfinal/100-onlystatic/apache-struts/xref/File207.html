46      public static final String VALID_SCHEME_CHAR = 
47          "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789+.-";

52      public static final int HIGHEST_SPECIAL = '>';

54      public static char[][] specialCharactersRepresentation = new char[HIGHEST_SPECIAL + 1][];
55      
56      static {
57          specialCharactersRepresentation['&'] = "&amp;".toCharArray();
58          specialCharactersRepresentation['<'] = "&lt;".toCharArray();
59          specialCharactersRepresentation['>'] = "&gt;".toCharArray();
60          specialCharactersRepresentation['"'] = "&#034;".toCharArray();
61          specialCharactersRepresentation['\''] = "&#039;".toCharArray();
62      }

64      /**
65       * Converts the given string description of a scope to the corresponding
66       * PageContext constant.
67       *
68       * The validity of the given scope has already been checked by the
69       * appropriate TLV.
70       *
71       * @param scope String description of scope
72       *
73       * @return PageContext constant corresponding to given scope description
74       * 
75       * taken from org.apache.taglibs.standard.tag.common.core.Util  
76       */
77      public static int getScope(String scope){
78          int ret = PageContext.PAGE_SCOPE;
79          
80          if("request".equalsIgnoreCase(scope)){
81              ret = PageContext.REQUEST_SCOPE;
82          }else if("session".equalsIgnoreCase(scope)){
83              ret = PageContext.SESSION_SCOPE;
84          }else if("application".equalsIgnoreCase(scope)){
85              ret = PageContext.APPLICATION_SCOPE;
86          }
87          
88          return ret;
89      }

91      /**
92       * @param url the URL
93       *
94       * @return  <tt>true</tt> if our current URL is absolute,
95       * <tt>false</tt> otherwise.
96       * taken from org.apache.taglibs.standard.tag.common.core.ImportSupport
97       */
98      public static boolean isAbsoluteUrl(String url){
99          if(url == null){
100             return false;
101         }
102         
103         int colonPos = url.indexOf(":");
104         if(colonPos == -1){
105             return false;
106         }
107         
108         for(int i=0;i<colonPos;i++){
109             if(VALID_SCHEME_CHAR.indexOf(url.charAt(i)) == -1){
110                 return false;
111             }
112         }
113         
114         return true;
115     }

117     /**
118      *
119      * @param input input string
120      * @param name name of attribute
121      *
122      * @return the value associated with a content-type attribute.
123      * Syntax defined in RFC 2045, section 5.1.
124      * taken from org.apache.taglibs.standard.tag.common.core.Util
125      */
126     public static String getContentTypeAttribute(String input, String name) {
127         int begin;
128         int end;
129         int index = input.toUpperCase().indexOf(name.toUpperCase());
130         if (index == -1) return null;
131         index = index + name.length(); // positioned after the attribute name
132         index = input.indexOf('=', index); // positioned at the '='
133         if (index == -1) return null;
134         index += 1; // positioned after the '='
135         input = input.substring(index).trim();
136         
137         if (input.charAt(0) == '"') {
138             // attribute value is a quoted string
139             begin = 1;
140             end = input.indexOf('"', begin);
141             if (end == -1) return null;
142         } else {
143             begin = 0;
144             end = input.indexOf(';');
145             if (end == -1) end = input.indexOf(' ');
146             if (end == -1) end = input.length();
147         }
148         return input.substring(begin, end).trim();
149     }

151     /**
152      * Strips a servlet session ID from <tt>url</tt>.  The session ID
153      * is encoded as a URL "path parameter" beginning with "jsessionid=".
154      * We thus remove anything we find between ";jsessionid=" (inclusive)
155      * and either EOS or a subsequent ';' (exclusive).
156      * 
157      * taken from org.apache.taglibs.standard.tag.common.core.ImportSupport
158      *
159      * @param url the URL
160      * @return servlet session ID from URL
161      */
162     public static String stripSession(String url) {
163         StringBuffer u = new StringBuffer(url);
164         int sessionStart;
165         while ((sessionStart = u.toString().indexOf(";" + Constants.SESSION_PARAMETER_NAME + "=")) != -1) {
166             int sessionEnd = u.toString().indexOf(";", sessionStart + 1);
167             if (sessionEnd == -1)
168                 sessionEnd = u.toString().indexOf("?", sessionStart + 1);
169             if (sessionEnd == -1) 				// still
170                 sessionEnd = u.length();
171             u.delete(sessionStart, sessionEnd);
172         }
173         return u.toString();
174     }

177     /**
178      * <p>
179      * Performs the following substring replacements
180      * (to facilitate output to XML/HTML pages):
181      * </p>
182      *
183      * <pre>
184      *    &amp; -&gt; &amp;amp;
185      *    &lt; -&gt; &amp;lt;
186      *    &gt; -&gt; &amp;gt;
187      *    " -&gt; &amp;#034;
188      *    ' -&gt; &amp;#039;
189      * </pre>
190      *
191      * See also OutSupport.writeEscapedXml().
192      * 
193      * taken from org.apache.taglibs.standard.tag.common.core.Util
194      *
195      * @param buffer string
196      * @return escaped string
197      */
198     public static String escapeXml(String buffer) {
199         int start = 0;
200         int length = buffer.length();
201         char[] arrayBuffer = buffer.toCharArray();
202         StringBuffer escapedBuffer = null;
203         
204         for (int i = 0; i < length; i++) {
205             char c = arrayBuffer[i];
206             if (c <= HIGHEST_SPECIAL) {
207                 char[] escaped = specialCharactersRepresentation[c];
208                 if (escaped != null) {
209                     // create StringBuffer to hold escaped xml string
210                     if (start == 0) {
211                         escapedBuffer = new StringBuffer(length + 5);
212                     }
213                     // add unescaped portion
214                     if (start < i) {
215                         escapedBuffer.append(arrayBuffer,start,i-start);
216                     }
217                     start = i + 1;
218                     // add escaped xml
219                     escapedBuffer.append(escaped);
220                 }
221             }
222         }
223         // no xml escaping was necessary
224         if (start == 0) {
225             return buffer;
226         }
227         // add rest of unescaped portion
228         if (start < length) {
229             escapedBuffer.append(arrayBuffer,start,length-start);
230         }
231         return escapedBuffer.toString();
232     }

234     /**
235      * Utility methods
236      * taken from org.apache.taglibs.standard.tag.common.core.UrlSupport
237      *
238      * @param url URL
239      * @param context context
240      * @param pageContext page context
241      *
242      * @return the URL
243      * @throws JspException in case of errors
244      */
245     public static String resolveUrl(
246             String url, String context, PageContext pageContext)
247     throws JspException {
248         // don't touch absolute URLs
249         if (isAbsoluteUrl(url))
250             return url;
251         
252         // normalize relative URLs against a context root
253         HttpServletRequest request =
254             (HttpServletRequest) pageContext.getRequest();
255         if (context == null) {
256             if (url.startsWith("/"))
257                 return (request.getContextPath() + url);
258             else
259                 return url;
260         } else {
261             if (!context.startsWith("/") || !url.startsWith("/")) {
262                 throw new JspTagException(
263                 "In URL tags, when the \"context\" attribute is specified, values of both \"context\" and \"url\" must start with \"/\".");
264             }
265             if (context.equals("/")) {
266                 // Don't produce string starting with '//', many
267                 // browsers interpret this as host name, not as
268                 // path on same host.
269                 return url;
270             } else {
271                 return (context + url);
272             }
273         }
274     }

276     /** Wraps responses to allow us to retrieve results as Strings. 
277      * mainly taken from org.apache.taglibs.standard.tag.common.core.importSupport 
278      */
279     public static class ImportResponseWrapper extends HttpServletResponseWrapper{
280         
281         private StringWriter sw = new StringWriter();
282         private ByteArrayOutputStream bos = new ByteArrayOutputStream();
283         private ServletOutputStream sos = new ServletOutputStream() {
284             public void write(int b) throws IOException {
285                 bos.write(b);
286             }
287 
288             @Override
289             public boolean isReady() {
290                 return (true);
291             }
292 
293             @Override
294             public void setWriteListener(WriteListener arg0) {
295                 return;
296             }
297         };
298         private boolean isWriterUsed;
299         private boolean isStreamUsed;
300         private int status = 200;
301         private String charEncoding;
302         
303         public ImportResponseWrapper(HttpServletResponse arg0) {
304             super(arg0);
305             // TODO Auto-generated constructor stub
306         }
307         
308         public PrintWriter getWriter() {
309             if (isStreamUsed) {
310                 throw new IllegalStateException("Unexpected internal error during &lt;import&gt: Target servlet called getWriter(), then getOutputStream()");
311             }
312             isWriterUsed = true;
313             return new PrintWriter(sw);
314         }
315         
316         public ServletOutputStream getOutputStream() {
317             if (isWriterUsed) {
318                 throw new IllegalStateException("Unexpected internal error during &lt;import&gt: Target servlet called getOutputStream(), then getWriter()");
319             }
320             isStreamUsed = true;
321             return sos;
322         }
323         
324         /**
325          * Has no effect.
326          *
327          * @param x content type
328          */
329         public void setContentType(String x) {
330             // ignore
331         }
332         
333         /**
334          * Has no effect.
335          *
336          * @param x locale
337          */
338         public void setLocale(Locale x) {
339             // ignore
340         }
341         
342         public void setStatus(int status) {
343             this.status = status;
344         }
345         
346         public int getStatus() {
347             return status;
348         }
349         
350         public String getCharEncoding(){
351             return this.charEncoding;
352         }
353         
354         public void setCharEncoding(String ce){
355             this.charEncoding = ce;
356         }
357         
358         public String getString() throws UnsupportedEncodingException {
359             if (isWriterUsed)
360                 return sw.toString();
361             else if (isStreamUsed) {
362                 if (this.charEncoding != null && !this.charEncoding.equals(""))
363                     return bos.toString(charEncoding);
364                 else
365                     return bos.toString("ISO-8859-1");
366             } else
367                 return "";		// target didn't write anything
368         }
369     }

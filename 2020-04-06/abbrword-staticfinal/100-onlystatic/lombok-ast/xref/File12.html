22  package lombok.ast.grammar;
23  
24  import java.util.Arrays;
25  import java.util.Collections;
26  import java.util.List;
27  
28  import lombok.ast.Node;
29  
30  import org.parboiled.BaseParser;
31  import org.parboiled.MatcherContext;
32  import org.parboiled.Rule;
33  import org.parboiled.annotations.SuppressSubnodes;
34  import org.parboiled.matchers.CharSetMatcher;
35  import org.parboiled.support.Characters;
36  
37  /**
38   * Contains the basics of java parsing: Whitespace and comment handling, as well as applying backslash-u escapes.
39   */
40  public class BasicsParser extends BaseParser<Node> {
41  	final ParserGroup group;
42  	final BasicsActions actions;
43  	
44  	public BasicsParser(ParserGroup group) {
45  		this.group = group;
46  		this.actions = new BasicsActions(group.getSource());
47  	}
48  	
49  	/**
50  	 * Eats up any whitespace and comments at the current position.
51  	 */
52  	public Rule optWS() {
53  		return ZeroOrMore(FirstOf(comment(), whitespaceChar())).label("ws");
54  	}
55  	
56  	/**
57  	 * Eats up any whitespace and comments at the current position,
58  	 * but only matches if there is at least one comment or whitespace character to gobble up.
59  	 */
60  	public Rule mandatoryWS() {
61  		return OneOrMore(FirstOf(comment(), whitespaceChar())).label("ws");
62  	}
63  	
64  	public Rule testLexBreak() {
65  		return TestNot(identifierPart());
66  	}
67  	
68  	public Rule identifier() {
69  		return Sequence(
70  				identifierRaw().label("identifier"),
71  				actions.checkIfKeyword(text("identifier")),
72  				set(actions.createIdentifier(text("identifier"), node("identifier"))),
73  				optWS());
74  	}
75  	
76  	public Rule dotIdentifier() {
77  		return Sequence(
78  				Ch('.'), optWS(),
79  				identifierRaw().label("identifier"),
80  				actions.checkIfKeyword(text("identifier")),
81  				set(actions.createIdentifier(text("identifier"), node("identifier"))),
82  				optWS());
83  	}
84  	
85  	/**
86  	 * Technically {@code null}, {@code true} and {@code false} aren't keywords but specific literals, but, from a parser point of view they are keywords.
87  	 * @see <a href="http://java.sun.com/docs/books/jls/third_edition/html/lexical.html#3.9">JLS section 3.9</a>
88  	 */
89  	static final List<String> KEYWORDS = Collections.unmodifiableList(Arrays.asList(
90  			"abstract", "class", "interface", "enum", "static", "final", "volatile", "transient", "strictfp", "native",
91  			"boolean", "byte", "short", "char", "int", "long", "float", "double", "void",
92  			"null", "this", "super", "true", "false",
93  			"continue", "break", "goto", "case", "default", "instanceof",
94  			"if", "do", "while", "for", "else", "synchronized", "switch", "assert", "throw", "try", "catch", "finally", "new", "return",
95  			"throws", "extends", "implements",
96  			"import", "package", "const",
97  			"public", "private", "protected"
98  	));
99  	
100 	@SuppressSubnodes
101 	public Rule identifierRaw() {
102 		return Sequence(new JavaIdentifierStartMatcher(), ZeroOrMore(new JavaIdentifierPartMatcher()));
103 	}
104 	
105 	public Rule identifierPart() {
106 		return new JavaIdentifierPartMatcher();
107 	}
108 	
109 	private static class JavaIdentifierPartMatcher extends CharSetMatcher<Node> {
110 		public JavaIdentifierPartMatcher() {
111 			super(Characters.of("abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789_$"));
112 		}
113 		
114 		@Override public boolean match(MatcherContext<Node> context) {
115 			char current = context.getCurrentChar();
116 			if (Character.isJavaIdentifierPart(current)) {
117 				context.advanceIndex();
118 				context.createNode();
119 				return true;
120 			}
121 			return false;
122 		}
123 	}
124 	
125 	private static class JavaIdentifierStartMatcher extends CharSetMatcher<Node> {
126 		public JavaIdentifierStartMatcher() {
127 			super(Characters.of("abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ_$"));
128 		}
129 		
130 		@Override public boolean match(MatcherContext<Node> context) {
131 			char current = context.getCurrentChar();
132 			if (Character.isJavaIdentifierStart(current)) {
133 				context.advanceIndex();
134 				context.createNode();
135 				return true;
136 			}
137 			return false;
138 		}
139 	}
140 	
141 	
142 	/**
143 	 * Any comment (block, line, or javadoc)
144 	 * 
145 	 * @see <a href="http://java.sun.com/docs/books/jls/third_edition/html/lexical.html#3.7">JLS section 3.7</a>
146 	 */
147 	public Rule comment() {
148 		return Sequence(
149 				FirstOf(lineComment(), blockComment()),
150 				actions.logComment(lastText()));
151 	}
152 	
153 	@SuppressSubnodes
154 	Rule lineComment() {
155 		return Sequence(String("//"), ZeroOrMore(Sequence(TestNot(CharSet("\r\n")), Any())), FirstOf(String("\r\n"), Ch('\r'), Ch('\n'), Test(Eoi())));
156 	}
157 	
158 	@SuppressSubnodes
159 	Rule blockComment() {
160 		return Sequence("/*", ZeroOrMore(Sequence(TestNot("*/"), Any())), "*/");
161 	}
162 	
163 	/**
164 	 * @see <a href="http://java.sun.com/docs/books/jls/third_edition/html/lexical.html#3.4">JLS section 3.4</a>
165 	 */
166 	@SuppressSubnodes
167 	Rule whitespaceChar() {
168 		return FirstOf(Ch(' '), Ch('\t'), Ch('\f'), lineTerminator());
169 	}
170 	
171 	/**
172 	 * @see <a href="http://java.sun.com/docs/books/jls/third_edition/html/lexical.html#3.6">JLS section 3.6</a>
173 	 */
174 	@SuppressSubnodes
175 	public Rule lineTerminator() {
176 		return FirstOf(String("\r\n").label("\\r\\n"), Ch('\r').label("\\r"), Ch('\n').label("\\n"));
177 	}
178 }

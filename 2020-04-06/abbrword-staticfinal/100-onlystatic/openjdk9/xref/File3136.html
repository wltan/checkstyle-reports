39      private static final char[] BAD_PASSWORD = "badpasword".toCharArray();
40      private static final String DIR = System.getProperty("test.src", ".");
41      private static final String CERT_FILE = "trusted.pem";

43      public static final void main(String[] args) throws Exception {

45          // Testing empty keystores

47          init("empty.jks", "JKS");

55          load("empty.p12", "JKS"); // test compatibility mode

59          load("empty.p12", "PKCS12", false); // test without compatibility mode

123     // Instantiate an empty keystore using the supplied keystore type
124     private static void init(String file, String type) throws Exception {
125         KeyStore ks = KeyStore.getInstance(type);
126         ks.load(null, null);
127         try (OutputStream stream = new FileOutputStream(file)) {
128             ks.store(stream, PASSWORD);
129         }
130         System.out.println("Created a " + type + " keystore named '" + file + "'");
131     }
132 
133     // Instantiate a keystore using the supplied keystore type & create an entry
134     private static void init(String file, String type, X509Certificate cert)
135         throws Exception {
136         KeyStore ks = KeyStore.getInstance(type);
137         ks.load(null, null);
138         ks.setEntry("mycert", new KeyStore.TrustedCertificateEntry(cert), null);
139         try (OutputStream stream = new FileOutputStream(file)) {
140             ks.store(stream, PASSWORD);
141         }
142         System.out.println("Created a " + type + " keystore named '" + file + "'");
143     }

145     // Instantiate a keystore using the supplied keystore type & create an entry
146     private static void init(String file, String type, SecretKey key)
147         throws Exception {
148         KeyStore ks = KeyStore.getInstance(type);
149         ks.load(null, null);
150         ks.setEntry("mykey", new KeyStore.SecretKeyEntry(key),
151             new PasswordProtection(PASSWORD));
152         try (OutputStream stream = new FileOutputStream(file)) {
153             ks.store(stream, PASSWORD);
154         }
155         System.out.println("Created a " + type + " keystore named '" + file + "'");
156     }

158     // Instantiate a keystore by probing the supplied file for the keystore type
159     private static void probe(String file, String type) throws Exception {
160         // First try with the correct password
161         KeyStore ks = KeyStore.getInstance(new File(file), PASSWORD);
162         if (!type.equalsIgnoreCase(ks.getType())) {
163             throw new Exception("ERROR: expected a " + type + " keystore, " +
164                 "got a " + ks.getType() + " keystore instead");
165         } else {
166             System.out.println("Probed a " + type + " keystore named '" + file + "'");
167         }
168 
169         // Next try with an incorrect password
170         try {
171             ks = KeyStore.getInstance(new File(file), BAD_PASSWORD);
172             throw new Exception("ERROR: expected an exception but got success");
173         } catch (IOException e) {
174             System.out.println("Failed to load a " + type + " keystore named '" + file + "' (as expected)");
175         }
176     }
177 
178     // Instantiate a keystore by probing the supplied file for the keystore type
179     private static void build(String file, String type, boolean usePassword)
180         throws Exception {
181 
182         Builder builder;
183         if (usePassword) {
184             builder = Builder.newInstance(new File(file),
185                 new PasswordProtection(PASSWORD));
186         } else {
187             builder = Builder.newInstance(new File(file),
188                 new CallbackHandlerProtection(new DummyHandler()));
189         }
190         KeyStore ks = builder.getKeyStore();
191         if (!type.equalsIgnoreCase(ks.getType())) {
192             throw new Exception("ERROR: expected a " + type + " keystore, " +
193                 "got a " + ks.getType() + " keystore instead");
194         } else {
195             System.out.println("Built a " + type + " keystore named '" + file + "'");
196         }
197     }

199     // Load the keystore entries
200     private static void load(String file, String type) throws Exception {
201         KeyStore ks = KeyStore.getInstance(type);
202         try (InputStream stream = new FileInputStream(file)) {
203             ks.load(stream, PASSWORD);
204         }
205         if (!type.equalsIgnoreCase(ks.getType())) {
206             throw new Exception("ERROR: expected a " + type + " keystore, " +
207                 "got a " + ks.getType() + " keystore instead");
208         } else {
209             System.out.println("Loaded a " + type + " keystore named '" + file + "'");
210         }
211     }
212 
213     // Load the keystore entries (with compatibility mode disabled)
214     private static void load(String file, String type, boolean expectFailure)
215         throws Exception {
216         Security.setProperty("keystore.type.compat", "false");
217         try {
218             load(file, type);
219             if (expectFailure) {
220                 throw new Exception("ERROR: expected load to fail but it didn't");
221             }
222         } catch (IOException e) {
223             if (expectFailure) {
224                 System.out.println("Failed to load a " + type + " keystore named '" + file + "' (as expected)");
225             } else {
226                 throw e;
227             }
228         } finally {
229             Security.setProperty("keystore.type.compat", "true");
230         }
231     }
232 
233     // Read an X.509 certificate from the supplied file
234     private static X509Certificate loadCertificate(String certFile)
235         throws Exception {
236         X509Certificate cert = null;
237         try (FileInputStream certStream =
238             new FileInputStream(DIR + "/" + certFile)) {
239             CertificateFactory factory =
240                 CertificateFactory.getInstance("X.509");
241             return (X509Certificate) factory.generateCertificate(certStream);
242         }
243     }
244 
245     // Generate a secret key using the supplied algorithm name and key size
246     private static SecretKey generateSecretKey(String algorithm, int size)
247         throws NoSuchAlgorithmException {
248         KeyGenerator generator = KeyGenerator.getInstance(algorithm);
249         generator.init(size);
250         return generator.generateKey();
251     }

253     private static class DummyHandler implements CallbackHandler {
254         public void handle(Callback[] callbacks)
255             throws IOException, UnsupportedCallbackException {
256             System.out.println("** Callbackhandler invoked");
257             for (int i = 0; i < callbacks.length; i++) {
258                 Callback cb = callbacks[i];
259                 if (cb instanceof PasswordCallback) {
260                     PasswordCallback pcb = (PasswordCallback)cb;
261                     pcb.setPassword(PASSWORD);
262                     break;
263                 }
264             }
265         }
266     }

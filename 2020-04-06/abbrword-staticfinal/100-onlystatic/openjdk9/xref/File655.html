48  /**
49   * An SSL/TLS CipherSuite. Constants for the standard key exchange, cipher,
50   * and mac algorithms are also defined in this class.
51   *
52   * The CipherSuite class and the inner classes defined in this file roughly
53   * follow the type safe enum pattern described in Effective Java. This means:
54   *
55   *  . instances are immutable, classes are final
56   *
57   *  . there is a unique instance of every value, i.e. there are never two
58   *    instances representing the same CipherSuite, etc. This means equality
59   *    tests can be performed using == instead of equals() (although that works
60   *    as well). [A minor exception are *unsupported* CipherSuites read from a
61   *    handshake message, but this is usually irrelevant]
62   *
63   *  . instances are obtained using the static valueOf() factory methods.
64   *
65   *  . properties are defined as final variables and made available as
66   *    package private variables without method accessors
67   *
68   *  . if the member variable allowed is false, the given algorithm is either
69   *    unavailable or disabled at compile time
70   *
71   */
72  final class CipherSuite implements Comparable<CipherSuite> {

182     // See also CipherBox.calculatePacketSize().
183     int calculatePacketSize(int fragmentSize,
184             ProtocolVersion protocolVersion, boolean isDTLS) {
185 
186         int packetSize = fragmentSize;
187         if (cipher != B_NULL) {
188             int blockSize = cipher.ivSize;
189             switch (cipher.cipherType) {
190                 case BLOCK_CIPHER:
191                     packetSize += macAlg.size;
192                     packetSize += 1;        // 1 byte padding length field
193                     packetSize +=           // use the minimal padding
194                             (blockSize - (packetSize % blockSize)) % blockSize;
195                     if (protocolVersion.useTLS11PlusSpec()) {
196                         packetSize += blockSize;        // explicit IV
197                     }
198 
199                     break;
200             case AEAD_CIPHER:
201                 packetSize += cipher.ivSize - cipher.fixedIvSize;   // record IV
202                 packetSize += cipher.tagSize;
203 
204                 break;
205             default:    // NULL_CIPHER or STREAM_CIPHER
206                 packetSize += macAlg.size;
207             }
208         }
209 
210         return packetSize +
211             (isDTLS ? DTLSRecord.headerSize : SSLRecord.headerSize);
212     }
213 
214     // See also CipherBox.calculateFragmentSize().
215     int calculateFragSize(int packetLimit,
216             ProtocolVersion protocolVersion, boolean isDTLS) {
217 
218         int fragSize = packetLimit -
219                 (isDTLS ? DTLSRecord.headerSize : SSLRecord.headerSize);
220         if (cipher != B_NULL) {
221             int blockSize = cipher.ivSize;
222             switch (cipher.cipherType) {
223             case BLOCK_CIPHER:
224                 if (protocolVersion.useTLS11PlusSpec()) {
225                     fragSize -= blockSize;              // explicit IV
226                 }
227                 fragSize -= (fragSize % blockSize);     // cannot hold a block
228                 // No padding for a maximum fragment.
229                 fragSize -= 1;        // 1 byte padding length field: 0x00
230                 fragSize -= macAlg.size;
231 
232                 break;
233             case AEAD_CIPHER:
234                 fragSize -= cipher.tagSize;
235                 fragSize -= cipher.ivSize - cipher.fixedIvSize;     // record IV
236 
237                 break;
238             default:    // NULL_CIPHER or STREAM_CIPHER
239                 fragSize -= macAlg.size;
240             }
241         }
242 
243         return fragSize;
244     }

267     /**
268      * Return a CipherSuite for the given name. The returned CipherSuite
269      * is supported by this implementation but may not actually be
270      * currently useable. See isAvailable().
271      *
272      * @exception IllegalArgumentException if the CipherSuite is unknown or
273      * unsupported.
274      */
275     static CipherSuite valueOf(String s) {
276         if (s == null) {
277             throw new IllegalArgumentException("Name must not be null");
278         }
279 
280         CipherSuite c = nameMap.get(s);
281         if ((c == null) || (c.allowed == false)) {
282             throw new IllegalArgumentException("Unsupported ciphersuite " + s);
283         }
284 
285         return c;
286     }

288     /**
289      * Return a CipherSuite with the given ID. A temporary object is
290      * constructed if the ID is unknown. Use isAvailable() to verify that
291      * the CipherSuite can actually be used.
292      */
293     static CipherSuite valueOf(int id1, int id2) {
294         id1 &= 0xff;
295         id2 &= 0xff;
296         int id = (id1 << 8) | id2;
297         CipherSuite c = idMap.get(id);
298         if (c == null) {
299             String h1 = Integer.toString(id1, 16);
300             String h2 = Integer.toString(id2, 16);
301             c = new CipherSuite("Unknown 0x" + h1 + ":0x" + h2, id);
302         }
303         return c;
304     }

311     /*
312      * Use this method when all of the values need to be specified.
313      * This is primarily used when defining a new ciphersuite for
314      * TLS 1.2+ that doesn't use the "default" PRF.
315      */
316     private static void add(String name, int id, int priority,
317             KeyExchange keyExchange, BulkCipher cipher, MacAlg mac,
318             boolean allowed, int obsoleted, int supported, PRF prf) {
319 
320         CipherSuite c = new CipherSuite(name, id, priority, keyExchange,
321             cipher, mac, allowed, obsoleted, supported, prf);
322         if (idMap.put(id, c) != null) {
323             throw new RuntimeException("Duplicate ciphersuite definition: "
324                                         + id + ", " + name);
325         }
326         if (c.allowed) {
327             if (nameMap.put(name, c) != null) {
328                 throw new RuntimeException("Duplicate ciphersuite definition: "
329                                             + id + ", " + name);
330             }
331         }
332     }

335      * Use this method when there is no lower protocol limit where this
336      * suite can be used, and the PRF is P_SHA256.  That is, the
337      * existing ciphersuites.  From RFC 5246:
338      *
339      *     All cipher suites in this document use P_SHA256.

341     private static void add(String name, int id, int priority,
342             KeyExchange keyExchange, BulkCipher cipher, MacAlg mac,
343             boolean allowed, int obsoleted) {
344         PRF prf = obsoleted < ProtocolVersion.TLS12.v ? P_NONE : P_SHA256;
345 
346         add(name, id, priority, keyExchange, cipher, mac, allowed, obsoleted,
347             ProtocolVersion.LIMIT_MIN_VALUE, prf);
348     }

408             this.name = name;

417                 return true;
418             }

423                 return (allowed && JsseJce.isKerberosAvailable());
424             } else {
425                 return allowed;
426             }
427         }

597         private static boolean isUnlimited(int keySize, String transformation) {
598             int keySizeInBits = keySize * 8;
599             if (keySizeInBits > 128) {    // need the JCE unlimited
600                                           // strength jurisdiction policy
601                 try {
602                     if (Cipher.getMaxAllowedKeyLength(
603                             transformation) < keySizeInBits) {
604 
605                         return false;
606                     }
607                 } catch (Exception e) {
608                     return false;
609                 }
610             }
611 
612             return true;
613         }

672     /**
673      * PRFs (PseudoRandom Function) from TLS specifications.
674      *
675      * TLS 1.1- uses a single MD5/SHA1-based PRF algorithm for generating
676      * the necessary material.

678      * In TLS 1.2+, all existing/known CipherSuites use SHA256, however

683 
684         // PRF algorithms
685         P_NONE(     "NONE",  0,   0),
686         P_SHA256("SHA-256", 32,  64),
687         P_SHA384("SHA-384", 48, 128),
688         P_SHA512("SHA-512", 64, 128);  // not currently used.
689 
690         // PRF characteristics

292     /**
293      * Sets the System security.
294      *
295      * <p> If there is a security manager already installed, this method first
296      * calls the security manager's <code>checkPermission</code> method
297      * with a <code>RuntimePermission("setSecurityManager")</code>
298      * permission to ensure it's ok to replace the existing
299      * security manager.
300      * This may result in throwing a <code>SecurityException</code>.
301      *
302      * <p> Otherwise, the argument is established as the current
303      * security manager. If the argument is <code>null</code> and no
304      * security manager has been established, then no action is taken and
305      * the method simply returns.
306      *
307      * @param      s   the security manager.
308      * @exception  SecurityException  if the security manager has already
309      *             been set and its <code>checkPermission</code> method
310      *             doesn't allow it to be replaced.
311      * @see #getSecurityManager
312      * @see SecurityManager#checkPermission
313      * @see java.lang.RuntimePermission
314      */
315     public static void setSecurityManager(final SecurityManager s) {
316         if (s != null) {
317             try {
318                 s.checkPackageAccess("java.lang");
319             } catch (Exception e) {
320                 // no-op
321             }
322         }
323         setSecurityManager0(s);
324     }
325 
326     private static synchronized
327     void setSecurityManager0(final SecurityManager s) {
328         SecurityManager sm = getSecurityManager();
329         if (sm != null) {
330             // ask the currently installed security manager if we
331             // can replace it.
332             sm.checkPermission(new RuntimePermission
333                                      ("setSecurityManager"));
334         }
335 
336         if ((s != null) && (s.getClass().getClassLoader() != null)) {
337             // New security manager class is not on bootstrap classpath.
338             // Cause policy to get initialized before we install the new
339             // security manager, in order to prevent infinite loops when
340             // trying to initialize the policy (which usually involves
341             // accessing some security and/or system properties, which in turn
342             // calls the installed security manager's checkPermission method
343             // which will loop infinitely if there is a non-system class
344             // (in this case: the new security manager class) on the stack).
345             AccessController.doPrivileged(new PrivilegedAction<>() {
346                 public Object run() {
347                     s.getClass().getProtectionDomain().implies
348                         (SecurityConstants.ALL_PERMISSION);
349                     return null;
350                 }
351             });
352         }
353 
354         security = s;
355     }

432     /**
433      * Copies an array from the specified source array, beginning at the
434      * specified position, to the specified position of the destination array.
435      * A subsequence of array components are copied from the source
436      * array referenced by <code>src</code> to the destination array
437      * referenced by <code>dest</code>. The number of components copied is
438      * equal to the <code>length</code> argument. The components at
439      * positions <code>srcPos</code> through
440      * <code>srcPos+length-1</code> in the source array are copied into
441      * positions <code>destPos</code> through
442      * <code>destPos+length-1</code>, respectively, of the destination
443      * array.
444      * <p>
445      * If the <code>src</code> and <code>dest</code> arguments refer to the
446      * same array object, then the copying is performed as if the
447      * components at positions <code>srcPos</code> through
448      * <code>srcPos+length-1</code> were first copied to a temporary
449      * array with <code>length</code> components and then the contents of
450      * the temporary array were copied into positions
451      * <code>destPos</code> through <code>destPos+length-1</code> of the
452      * destination array.
453      * <p>
454      * If <code>dest</code> is <code>null</code>, then a
455      * <code>NullPointerException</code> is thrown.
456      * <p>
457      * If <code>src</code> is <code>null</code>, then a
458      * <code>NullPointerException</code> is thrown and the destination
459      * array is not modified.
460      * <p>
461      * Otherwise, if any of the following is true, an
462      * <code>ArrayStoreException</code> is thrown and the destination is
463      * not modified:
464      * <ul>
465      * <li>The <code>src</code> argument refers to an object that is not an
466      *     array.
467      * <li>The <code>dest</code> argument refers to an object that is not an
468      *     array.
469      * <li>The <code>src</code> argument and <code>dest</code> argument refer
470      *     to arrays whose component types are different primitive types.
471      * <li>The <code>src</code> argument refers to an array with a primitive
472      *    component type and the <code>dest</code> argument refers to an array
473      *     with a reference component type.
474      * <li>The <code>src</code> argument refers to an array with a reference
475      *    component type and the <code>dest</code> argument refers to an array
476      *     with a primitive component type.
477      * </ul>
478      * <p>
479      * Otherwise, if any of the following is true, an
480      * <code>IndexOutOfBoundsException</code> is
481      * thrown and the destination is not modified:
482      * <ul>
483      * <li>The <code>srcPos</code> argument is negative.
484      * <li>The <code>destPos</code> argument is negative.
485      * <li>The <code>length</code> argument is negative.
486      * <li><code>srcPos+length</code> is greater than
487      *     <code>src.length</code>, the length of the source array.
488      * <li><code>destPos+length</code> is greater than
489      *     <code>dest.length</code>, the length of the destination array.
490      * </ul>
491      * <p>
492      * Otherwise, if any actual component of the source array from
493      * position <code>srcPos</code> through
494      * <code>srcPos+length-1</code> cannot be converted to the component
495      * type of the destination array by assignment conversion, an
496      * <code>ArrayStoreException</code> is thrown. In this case, let
497      * <b><i>k</i></b> be the smallest nonnegative integer less than
498      * length such that <code>src[srcPos+</code><i>k</i><code>]</code>
499      * cannot be converted to the component type of the destination
500      * array; when the exception is thrown, source array components from
501      * positions <code>srcPos</code> through
502      * <code>srcPos+</code><i>k</i><code>-1</code>
503      * will already have been copied to destination array positions
504      * <code>destPos</code> through
505      * <code>destPos+</code><i>k</I><code>-1</code> and no other
506      * positions of the destination array will have been modified.
507      * (Because of the restrictions already itemized, this
508      * paragraph effectively applies only to the situation where both
509      * arrays have component types that are reference types.)
510      *
511      * @param      src      the source array.
512      * @param      srcPos   starting position in the source array.
513      * @param      dest     the destination array.
514      * @param      destPos  starting position in the destination data.
515      * @param      length   the number of array elements to be copied.
516      * @exception  IndexOutOfBoundsException  if copying would cause
517      *               access of data outside array bounds.
518      * @exception  ArrayStoreException  if an element in the <code>src</code>
519      *               array could not be stored into the <code>dest</code> array
520      *               because of a type mismatch.
521      * @exception  NullPointerException if either <code>src</code> or
522      *               <code>dest</code> is <code>null</code>.
523      */
524     @HotSpotIntrinsicCandidate
525     public static native void arraycopy(Object src,  int  srcPos,
526                                         Object dest, int destPos,
527                                         int length);

1348      * The {@code LoggerFinder} service is responsible for creating, managing,
1349      * and configuring loggers to the underlying framework it uses.
1350      * <p>
1351      * A logger finder is a concrete implementation of this class that has a
1352      * zero-argument constructor and implements the abstract methods defined
1353      * by this class.
1354      * The loggers returned from a logger finder are capable of routing log
1355      * messages to the logging backend this provider supports.
1356      * A given invocation of the Java Runtime maintains a single
1357      * system-wide LoggerFinder instance that is loaded as follows:
1358      * <ul>
1359      *    <li>First it finds any custom {@code LoggerFinder} provider
1360      *        using the {@link java.util.ServiceLoader} facility with the
1361      *        {@linkplain ClassLoader#getSystemClassLoader() system class
1362      *        loader}.</li>
1363      *    <li>If no {@code LoggerFinder} provider is found, the system default
1364      *        {@code LoggerFinder} implementation will be used.</li>
1365      * </ul>
1366      * <p>
1367      * An application can replace the logging backend
1368      * <i>even when the java.logging module is present</i>, by simply providing
1369      * and declaring an implementation of the {@link LoggerFinder} service.
1370      * <p>
1371      * <b>Default Implementation</b>
1372      * <p>
1373      * The system default {@code LoggerFinder} implementation uses
1374      * {@code java.util.logging} as the backend framework when the
1375      * {@code java.logging} module is present.
1376      * It returns a {@linkplain System.Logger logger} instance
1377      * that will route log messages to a {@link java.util.logging.Logger
1378      * java.util.logging.Logger}. Otherwise, if {@code java.logging} is not
1379      * present, the default implementation will return a simple logger
1380      * instance that will route log messages of {@code INFO} level and above to
1381      * the console ({@code System.err}).
1382      * <p>
1383      * <b>Logging Configuration</b>
1384      * <p>
1385      * {@linkplain Logger Logger} instances obtained from the
1386      * {@code LoggerFinder} factory methods are not directly configurable by
1387      * the application. Configuration is the responsibility of the underlying
1388      * logging backend, and usually requires using APIs specific to that backend.
1389      * <p>For the default {@code LoggerFinder} implementation
1390      * using {@code java.util.logging} as its backend, refer to
1391      * {@link java.util.logging java.util.logging} for logging configuration.
1392      * For the default {@code LoggerFinder} implementation returning simple loggers
1393      * when the {@code java.logging} module is absent, the configuration
1394      * is implementation dependent.
1395      * <p>
1396      * Usually an application that uses a logging framework will log messages
1397      * through a logger facade defined (or supported) by that framework.
1398      * Applications that wish to use an external framework should log
1399      * through the facade associated with that framework.
1400      * <p>
1401      * A system class that needs to log messages will typically obtain
1402      * a {@link System.Logger} instance to route messages to the logging
1403      * framework selected by the application.
1404      * <p>

1406      * should not attempt to configure loggers by themselves, as that
1407      * would make them dependent from a specific implementation of the
1408      * {@code LoggerFinder} service.
1409      * <p>

1411      * system classes should not be directly configurable through the logging
1412      * backend without requiring permissions.

1415      * the concrete {@code LoggerFinder} implementation to ensure that

1934 
1935         // There are certain system configurations that may be controlled by
1936         // VM options such as the maximum amount of direct memory and
1937         // Integer cache size used to support the object identity semantics
1938         // of autoboxing.  Typically, the library will obtain these values
1939         // from the properties set by the VM.  If the properties are for
1940         // internal implementation use only, these properties should be
1941         // removed from the system properties.
1942         //
1943         // See java.lang.Integer.IntegerCache and the
1944         // VM.saveAndRemoveProperties method for example.
1945         //
1946         // Save a private copy of the system properties object that
1947         // can only be accessed by the internal implementation.  Remove
1948         // certain system properties that are not intended for public access.
1949         VM.saveAndRemoveProperties(props);
1950 
1951         lineSeparator = props.getProperty("line.separator");

1953

2018     /*
2019      * Invoked by VM.  Phase 3 is the final system initialization:
2020      * 1. set security manager
2021      * 2. set system class loader
2022      * 3. set TCCL
2023      *
2024      * This method must be called after the module system initialization.
2025      * The security manager and system class loader may be custom class from
2026      * the application classpath or modulepath.
2027      */
2028     private static void initPhase3() {
2029         // set security manager
2030         String cn = System.getProperty("java.security.manager");
2031         if (cn != null) {
2032             if (cn.isEmpty() || "default".equals(cn)) {
2033                 System.setSecurityManager(new SecurityManager());
2034             } else {
2035                 try {
2036                     Class<?> c = Class.forName(cn, false, ClassLoader.getBuiltinAppClassLoader());
2037                     Constructor<?> ctor = c.getConstructor();
2038                     // Must be a public subclass of SecurityManager with
2039                     // a public no-arg constructor
2040                     if (!SecurityManager.class.isAssignableFrom(c) ||
2041                             !Modifier.isPublic(c.getModifiers()) ||
2042                             !Modifier.isPublic(ctor.getModifiers())) {
2043                         throw new Error("Could not create SecurityManager: " + ctor.toString());
2044                     }
2045                     // custom security manager implementation may be in unnamed module
2046                     // or a named module but non-exported package
2047                     ctor.setAccessible(true);
2048                     SecurityManager sm = (SecurityManager) ctor.newInstance();
2049                     System.setSecurityManager(sm);
2050                 } catch (Exception e) {
2051                     throw new Error("Could not create SecurityManager", e);
2052                 }
2053             }
2054         }
2055 
2056         // initializing the system class loader
2057         VM.initLevel(3);
2058 
2059         // system class loader initialized
2060         ClassLoader scl = ClassLoader.initSystemClassLoader();
2061 
2062         // set TCCL
2063         Thread.currentThread().setContextClassLoader(scl);
2064 
2065         // system is fully initialized
2066         VM.initLevel(4);
2067     }

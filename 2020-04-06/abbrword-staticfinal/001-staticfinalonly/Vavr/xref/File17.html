17  public class Euler43Test {
18  
19      /**
20       * <strong>Problem 43 Sub-string divisibility</strong>
21       * <p>
22       * The number, 1406357289, is a 0 to 9 pandigital number because it is made
23       * up of each of the digits 0 to 9 in some order, but it also has a rather
24       * interesting sub-string divisibility property.</p>
25       * <p>
26       * Let <i>d</i><sub>1</sub> be the 1<sup>st</sup> digit,
27       * <i>d</i><sub>2</sub> be the 2<sup>nd</sup> digit, and so on. In this way,
28       * we note the following:</p>
29       * <pre>
30       * <ul>
31       * <li><i>d</i><sub>2</sub><i>d</i><sub>3</sub><i>d</i><sub>4</sub>=406 is divisible by 2</li>
32       * <li><i>d</i><sub>3</sub><i>d</i><sub>4</sub><i>d</i><sub>5</sub>=063 is divisible by 3</li>
33       * <li><i>d</i><sub>4</sub><i>d</i><sub>5</sub><i>d</i><sub>6</sub>=635 is divisible by 5</li>
34       * <li><i>d</i><sub>5</sub><i>d</i><sub>6</sub><i>d</i><sub>7</sub>=357 is divisible by 7</li>
35       * <li><i>d</i><sub>6</sub><i>d</i><sub>7</sub><i>d</i><sub>8</sub>=572 is divisible by 11</li>
36       * <li><i>d</i><sub>7</sub><i>d</i><sub>8</sub><i>d</i><sub>9</sub>=728 is divisible by 13</li>
37       * <li><i>d</i><sub>8</sub><i>d</i><sub>9</sub><i>d</i><sub>10</sub>=289 is divisible by 17</li>
38       * </ul>
39       * </pre>
40       * <p>
41       * Find the sum of all 0 to 9 pandigital numbers with this property.
42       * <p>
43       * See also <a href="https://projecteuler.net/problem=43">projecteuler.net
44       * problem 43</a>.
45       */
46      @Test
47      public void shouldSolveProblem43() {
48          final Seq<Long> result = tenDigitPandigitalsWithProperty();
49          Assertions.assertThat(result).contains(1406357289L);
50  
51          assertThat(result.sum().longValue()).isEqualTo(16695334890L);
52      }
53  
54      private static Seq<Long> tenDigitPandigitalsWithProperty() {
55          final CharSeq ALL_DIGITS = CharSeq.of("0123456789");
56          final List<Integer> DIVISORS = List.of(2, 3, 5, 7, 11, 13, 17);
57  
58          return ALL_DIGITS
59                  .combinations(2)
60                  .flatMap(CharSeq::permutations)
61                  .flatMap(firstTwoDigits -> DIVISORS
62                          .foldLeft(List.of(firstTwoDigits), (accumulator, divisor) -> accumulator
63                                  .flatMap(digitsSoFar -> ALL_DIGITS
64                                          .removeAll(digitsSoFar)
65                                          .map(nextDigit -> digitsSoFar.append(nextDigit))
66                                  )
67                                  .filter(digitsToTest -> digitsToTest.takeRight(3).parseInt() % divisor == 0)
68                          )
69                  )
70                  .map(tailDigitsWithProperty -> tailDigitsWithProperty
71                          .prepend(ALL_DIGITS
72                                  .removeAll(tailDigitsWithProperty)
73                                  .head()
74                          )
75                  )
76                  .map(CharSeq::parseLong);
77      }

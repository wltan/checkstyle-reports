146     // representation of the type of a single operation parameter TRANSPORT of type OBJECT holding transport attributes
147     // If this type is defined in a resource bundle with prefix X, the type descriptions
148     // should be of the form X.transport.type, X.transport.shared, etc.
149     static final ObjectTypeAttributeDefinition TRANSPORT = ObjectTypeAttributeDefinition.
150                 Builder.of(ModelKeys.TRANSPORT, TRANSPORT_ATTRIBUTES).
151                 setAllowNull(true).
152                 setSuffix(null).
153                 //setSuffix("transport").
154                 build();
155 
156     // operations
157     private static final OperationDefinition TRANSPORT_ADD_DEFINITION = new SimpleOperationDefinitionBuilder(ADD, JGroupsExtension.getResourceDescriptionResolver(ModelKeys.TRANSPORT))
158         .setParameters(TRANSPORT_PARAMETERS)
159         .build();
160 
161     static final OperationStepHandler TRANSPORT_ADD = new TransportLayerAdd(TRANSPORT_PARAMETERS);
162     static final OperationStepHandler TRANSPORT_REMOVE = new TransportLayerRemove();
163     static final TransportResourceDefinition INSTANCE = new TransportResourceDefinition();
164 
165     // registration
166     TransportResourceDefinition() {
167         super(TRANSPORT_PATH,
168                 JGroupsExtension.getResourceDescriptionResolver(ModelKeys.TRANSPORT),
169                 null,  //we register it manualy in #  registerOperations
170                 TRANSPORT_REMOVE);
171     }
172 
173     @Override
174     public void registerOperations(ManagementResourceRegistration registration) {
175         super.registerOperations(registration);
176         registration.registerOperationHandler(TRANSPORT_ADD_DEFINITION, TRANSPORT_ADD);
177     }
178 
179     @Override
180     public void registerAttributes(ManagementResourceRegistration resourceRegistration) {
181         super.registerAttributes(resourceRegistration);
182 
183         final OperationStepHandler writeHandler = new ReloadRequiredWriteAttributeHandler(TRANSPORT_ATTRIBUTES);
184         for (AttributeDefinition attr : TRANSPORT_ATTRIBUTES) {
185             resourceRegistration.registerReadWriteAttribute(attr, null, writeHandler);
186         }
187     }
188 
189     @Override
190     public void registerChildren(ManagementResourceRegistration resourceRegistration) {
191         super.registerChildren(resourceRegistration);
192         resourceRegistration.registerSubModel(PropertyResourceDefinition.INSTANCE);
193     }

23  @Test(groups = "functional", testName = "ShardedTreeMapTest")
24  public class ShardedTreeMapTest extends MultipleCacheManagersTest {
25  
26      private static int NCALLS= 100;
27      private static int NCACHES = 3;
28      private static List<Cache> caches = new ArrayList<Cache>();
29  
30      @Test(enabled = true)
31      public void basicUsageTest() throws  Exception{
32          EmbeddedCacheManager cacheManager = cacheManagers.iterator().next();
33          Cache cache = cacheManager.getCache();
34          AtomicObjectFactory factory = new AtomicObjectFactory(cache);
35  
36          SortedMap<Integer,Integer> map = factory.getInstanceOf(ShardedTreeMap.class,"test",false,null,false,3);
37          for(int i=0; i<NCALLS; i++) {
38              map.put(i,i);
39              map.get(i);
40          }
41          factory.disposeInstanceOf(ShardedTreeMap.class,"test",true);
42          map = factory.getInstanceOf(ShardedTreeMap.class,"test",false,null,false,3);
43          int a = map.size();
44          log.debug(a);
45          assertEquals(NCALLS,map.size());
46          assertEquals(NCALLS/2,map.subMap(0,NCALLS/2).size());
47  
48          SortedMap<Integer,Integer> map2 = factory.getInstanceOf(ShardedTreeMap.class,"test2",false,null,false,3);
49          Map<Integer,Integer> toAdd = new HashMap<Integer, Integer>();
50          for(int i=0; i<NCALLS; i++) {
51              toAdd.put(i, i);
52              toAdd.get(i);
53          }
54          map2.putAll(toAdd);
55          assertEquals(NCALLS,map2.size());
56          assertEquals(map2.lastKey(),Integer.valueOf(NCALLS-1));
57  
58      }
59  
60      //
61      // HELPERS
62      //
63  
64      @Override
65      protected void createCacheManagers() throws Throwable {
66          ConfigurationBuilder builder
67                  = AbstractCacheTest.getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, false);
68          TransportFlags flags = new TransportFlags();
69          createClusteredCaches(NCACHES, builder, flags);
70      }
71  
72      protected void initAndTest() {
73          for (Cache<Object, String> c : caches) assert c.isEmpty();
74          caches.iterator().next().put("k1", "value");
75          assertOnAllCaches(caches,"k1", "value");
76      }

446     /**
447      * Constructs an AES Key according to the specified key size.
448      * @param len key size in bytes, i.e. 16, 24, or 32
449      */
450     private static SecretKey constructAESKey(int len)
451         throws IllegalArgumentException {
452         if ((len != 16) && (len != 24) && (len != 32)) {
453             throw new IllegalArgumentException("Wrong Key Length: " + len);
454         }
455         byte[] keyval = new byte[len];
456         return new SecretKeySpec(keyval, "AES");
457     }
458     /**
459      * Constructs plain text byte array according to the specified round
460      * number.
461      * @param rounds round number starting from 0, i.e. valid from 0
462      * to 127.
463      */
464     private static byte[] constructPT(int rounds) {
465         byte[] tempValue = new byte[16];
466         Arrays.fill(tempValue, (byte)0);
467 
468         int whichByte = rounds/8;
469         int whichDigit = rounds % 8;
470         if ((whichByte >= 16) || (whichDigit < 0) ||
471             (whichDigit > 8)) {
472             throw new IllegalArgumentException("Invalid rounds: " +
473                                                rounds);
474         }
475         switch (whichDigit) {
476         case 0:
477             tempValue[whichByte] = (byte)0x80;
478             break;
479         case 1:
480             tempValue[whichByte] = (byte)0x40;
481             break;
482         case 2:
483             tempValue[whichByte] = (byte)0x20;
484             break;
485         case 3:
486             tempValue[whichByte] = (byte)0x10;
487             break;
488         case 4:
489             tempValue[whichByte] = (byte)0x08;
490             break;
491         case 5:
492             tempValue[whichByte] = (byte)0x04;
493             break;
494         case 6:
495             tempValue[whichByte] = (byte)0x02;
496             break;
497         case 7:
498             tempValue[whichByte] = (byte)0x01;
499             break;
500         }
501         return tempValue;
502     }

504     private static byte[] constructByteArray(String s) {
505         int len = s.length()/2;
506         byte[] tempValue = new byte[len];
507         for (int i = 0; i < len; i++) {
508             tempValue[i] = Integer.valueOf(s.substring(2*i, 2*i+2),
509                                            16).byteValue();
510         }
511         return tempValue;
512     }

514     public boolean execute() throws Exception {
515         String transformation = ALGO+"/"+MODE+"/"+PADDING;
516         Cipher c = Cipher.getInstance(transformation, "SunJCE");
517 
518         for (int i=0; i<KEY_SIZES.length; i++) {
519             if (KEY_SIZES[i]*8 >
520                 Cipher.getMaxAllowedKeyLength(transformation)) {
521                 // skip if this key length is larger than what's
522                 // configured in the jce jurisdiction policy files
523                 continue;
524             }
525             SecretKey aesKey = constructAESKey(KEY_SIZES[i]);
526             byte[] plainText = null;
527             byte[] cipherText = null;
528             try {
529             for (int j=0; j < 128; j++) {
530                 plainText = constructPT(j);
531                 c.init(Cipher.ENCRYPT_MODE, aesKey);
532                 cipherText = c.doFinal(plainText);
533                 byte[] answer = constructByteArray(CTS[i][j]);
534                 if (!Arrays.equals(cipherText, answer)) {
535                     throw new Exception((i+1) +
536                         "th known answer test failed for encryption");
537                 }
538                 c.init(Cipher.DECRYPT_MODE, aesKey);
539                 byte[] restored = c.doFinal(cipherText);
540                 if (!Arrays.equals(plainText, restored)) {
541                     throw new Exception((i+1) +
542                         "th known answer test failed for decryption");
543                 }
544             }
545             System.out.println(
546                 "Finished KAT for " + KEY_SIZES[i] + "-byte key");
547             } catch (SecurityException se) {
548                 TestUtil.handleSE(se);
549             }
550         }
551 
552         // passed all tests...hooray!
553         return true;
554     }

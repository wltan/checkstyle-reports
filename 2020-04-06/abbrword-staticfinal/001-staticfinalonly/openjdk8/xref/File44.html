47  /**
48   * NTypeParser (Native Type Parser) parses type & type64 attributes from BridgeSupport.
49   *
50   * See Obj-C Language: Type Encodings
51   */
52  public abstract class NTypeParser {
53      // ** Parser entry point
54  
55      private static Map<String, NType> cached = new HashMap<String, NType>();
56      public static NType parseFrom(String s) {
57          if(!cached.containsKey(s)) cached.put(s, parseFrom(new StringStream(s)));
58          return cached.get(s);
59      }
60  
61      // ** Parser driver
62  
63      private static List<NTypeParser> PARSERS = new ArrayList<NTypeParser>(
64              Arrays.asList(new NBitfieldParser(), new NPrimitiveParser(), new NVoidParser(),
65              new NPointerParser(), new NStructParser(), new NUnionParser(),
66              new NObjectParser(), new NClassParser(), new NSelectorParser(),
67              new NArrayParser(), new NUnknownParser(), new NSpecifierParser()));
68  
69      protected static NType parseFrom(StringStream ss) {
70          if(ss.left() == 0)
71              return null;
72          try{
73              for(NTypeParser nt : PARSERS)
74                  if(nt.parsePossible(ss))
75                      return nt.parse(ss);
76          }
77          catch(RuntimeException x){
78              throw new RuntimeException("Exception while parsing '" + ss.remainingToString()
79                      + "' from '" + ss.toString() + "'", x);
80          }
81          throw new RuntimeException("Found no parser for '" + ss.remainingToString()
82                  + "' from '" + ss.toString() + "'");
83      }
84  
85      // ** Methods for parsers
86  
87      protected abstract boolean parsePossible(StringStream ss);
88      protected abstract NType parse(StringStream ss);
89  
90      // ** Individual parsers
91  
92      public static class NBitfieldParser extends NTypeParser{
93          @Override protected boolean parsePossible(StringStream ss) {
94              return ss.left() >= 2 && ss.peekAt(0) == 'b' && Character.isDigit(ss.peekAt(1));
95          }
96  
97          @Override protected NType parse(StringStream ss) {
98              assert parsePossible(ss);
99              ss.eat('b');
100             return new NBitfield(Integer.parseInt(ss.readWhileDigits()));
101         }
102     }
103 
104     public static class NPrimitiveParser extends NTypeParser{
105         @Override protected boolean parsePossible(StringStream ss) {
106             return NPrimitive.CODES.contains(ss.peek());
107         }
108 
109         @Override protected NType parse(StringStream ss) {
110             assert parsePossible(ss);
111             return NPrimitive.inst(ss.read());
112         }
113     }

119

121             ss.eat('v');

123         }
124     }
125 
126     public static class NPointerParser extends NTypeParser{
127         private static NPointer CHAR_PTR = new NPointer(NPrimitive.inst('C'));
128 
129         @Override protected boolean parsePossible(StringStream ss) {
130             return (ss.left() >= 2 && ss.peek() == '^') || (ss.peek() == '*');
131         }
132 
133         @Override protected NType parse(StringStream ss) {
134             if(ss.peek() == '*'){
135                 ss.eat('*');
136                 return CHAR_PTR;
137             }
138             else{
139                 ss.eat('^');
140                 return new NPointer(NTypeParser.parseFrom(ss));
141             }
142         }
143     }
144 
145     public static class NStructParser extends NTypeParser{
146         protected char getOpen(){ return '{'; };
147         protected char getClose(){ return '}'; };
148 
149         @Override protected boolean parsePossible(StringStream ss) {
150             return ss.left() >= 2 && ss.peek() == getOpen();
151         }
152 
153         @Override protected NType parse(StringStream ss) {
154             assert parsePossible(ss);
155             // {_NSRect=
156             //   "origin"{_NSPoint="x"f"y"f}
157             //   "size"{_NSSize="width"f"height"f}}
158             ss.eat(getOpen());
159             String cname = ss.readUntilEither("=" + getClose());
160             List<NStruct.NField> fields = new ArrayList<NStruct.NField>();
161             if(ss.peek() == '='){
162                 ss.eat('=');
163                 while(ss.peek() != getClose()){
164                     String fname = "";
165                     if(ss.peek() == '"'){
166                         ss.eat('"');
167                         fname = ss.readUntil('"');
168                         ss.eat('"');
169                     }
170                     NType type = NTypeParser.parseFrom(ss);
171                     fields.add(new NStruct.NField(fname, type));
172                 }
173             }
174             ss.eat(getClose());
175             return getNew(cname, fields);
176         }
177 
178         protected NType getNew(String cname, List<NStruct.NField> fields){
179             return new NStruct(cname, fields);
180         }
181     }
182 
183     // A Union is very much like a Struct.
184     public static class NUnionParser extends NStructParser{
185         @Override protected char getOpen(){ return '('; };
186 
187         @Override protected char getClose(){ return ')'; };
188 
189         @Override protected NType getNew(String cname, List<NStruct.NField> fields){
190             return new NUnion(cname, Fp.map(NUnion.zeroOffsets, fields));
191         }
192     }
193 
194     public static class NArrayParser extends NTypeParser{
195         @Override protected boolean parsePossible(StringStream ss) {
196             return ss.peek() == '[';
197         }
198 
199         @Override protected NType parse(StringStream ss) {
200             ss.eat('[');
201             int size = Integer.parseInt(ss.readWhileDigits());
202             NType type = NTypeParser.parseFrom(ss);
203             ss.eat(']');
204             return new NArray(size, type);
205         }
206     }

252     /**
253      * Specifier     Encoding
254      * const         r
255      * in             n
256      * inout         N
257      * out             o
258      * bycopy         O
259      * oneway         V
260      */
261     public static class NSpecifierParser extends NTypeParser{
262         private static Collection<Character> SPECS = Arrays.asList('r', 'n', 'N', 'o', 'O', 'V');
263         @Override protected boolean parsePossible(StringStream ss) {
264             return SPECS.contains(ss.peek());
265         }
266 
267         @Override protected NType parse(StringStream ss) {
268             assert parsePossible(ss);
269             ss.seek(); // XXX Just ignore specs for now and return the affected type.
270             return NTypeParser.parseFrom(ss);
271         }
272     }

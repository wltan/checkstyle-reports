42  public class IntroTest extends TestCase{
43      // The low-level core makes function calls, sends messages, marshals data, etc.
44      public void testCore(){
45          // pass security check and get ahold of a runtime (should cache this)
46          final JObjCRuntime RUNTIME = JObjCRuntime.getInstance();
47          final NativeArgumentBuffer ARGS = JObjCRuntime.getInstance().getThreadLocalState();
48  
49          // create a funcall (should cache this)
50          final FunCall fc = new FunCall(RUNTIME, "sin", DoubleCoder.INST, DoubleCoder.INST);
51  
52          // start function call
53          fc.init(ARGS);
54          // push an arg
55          DoubleCoder.INST.push(ARGS, 3.14159265 / 2.0);
56          // make the call
57          fc.invoke(ARGS);
58          // read the return value
59          double ret = DoubleCoder.INST.pop(ARGS);
60  
61          assertEquals(1.0, ret);
62      }

64      // Frameworks bridge the Mac OS X frameworks
65      public void testFrameworks(){
66          // First, get an instance of JObjC:
67          final JObjC JOBJC = com.apple.jobjc.JObjC.getInstance();
68  
69          // It's your gateway to the frameworks.
70          final FoundationFramework FND = JOBJC.Foundation();
71          final AppKitFramework APP = JOBJC.AppKit();
72  
73          // From which you can then access...
74  
75          // enums, defines, constants
76          int nsmye = FND.NSMaxYEdge();
77          boolean debug = FND.NSDebugEnabled();
78  
79          // structs
80          NSPoint p = FND.makeNSPoint();
81          p.setX(3);
82          assertEquals(3.0, p.x());
83  
84          // C functions
85          NSPoint p2 = FND.NSMakePoint(12, 34);
86          assertEquals(12.0, p2.x());
87  
88          // ... Let's create an AutoreleasePool before we go on
89          NSAutoreleasePool pool = ((NSAutoreleasePool) FND.NSAutoreleasePool().alloc()).init();
90  
91          // Objective-C classes
92          NSStringClass nsc = FND.NSString();
93  
94          // class-methods
95          NSString nsStringClassDescr = nsc.description();
96  
97          // instances
98          NSString nsi = ((NSString) FND.NSString().alloc()).init();
99  
100         // instance methods
101         NSString d = nsi.description();
102 
103         // The bridge marshals some types for you, but it doesn't
104         // convert between NSString and Java String automatically.
105         // For that we use Utils.get().strings().nsString(String)
106         // and Utils.get().strings().javaString(NSString);
107 
108         assertEquals("NSString", Utils.get().strings().javaString(nsStringClassDescr));
109 
110         NSString format = Utils.get().strings().nsString("Foo bar %d baz");
111 
112         NSString formatted = ((NSString) FND.NSString().alloc()).initWithFormat(format, 34);
113         String jformatted = Utils.get().strings().javaString(formatted);
114 
115         assertEquals("Foo bar 34 baz", jformatted);
116 
117         // Reveal in Finder
118 //        NSString file = Utils.get().strings().nsString(
119 //                "/Applications/Calculator.app/Contents/Resources/Calculator.icns");
120 //        APP.NSWorkspace().sharedWorkspace()
121 //           .selectFile_inFileViewerRootedAtPath(file, null);
122 
123         pool.drain();
124     }

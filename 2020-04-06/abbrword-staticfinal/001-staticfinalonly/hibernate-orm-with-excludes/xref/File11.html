38  /**
39   * Informix dialect.<br>
40   * <br>
41   * Seems to work with Informix Dynamic Server Version 7.31.UD3,  Informix JDBC driver version 2.21JC3.
42   *
43   * @author Steve Molitor
44   */
45  public class InformixDialect extends Dialect {
46  	
47  	private final UniqueDelegate uniqueDelegate;
48  
49  	/**
50  	 * Creates new <code>InformixDialect</code> instance. Sets up the JDBC /
51  	 * Informix type mappings.
52  	 */
53  	public InformixDialect() {
54  		super();
55  
56  		registerColumnType(Types.BIGINT, "int8");
57  		registerColumnType(Types.BINARY, "byte");
58  		registerColumnType(Types.BIT, "smallint"); // Informix doesn't have a bit type
59  		registerColumnType(Types.CHAR, "char($l)");
60  		registerColumnType(Types.DATE, "date");
61  		registerColumnType(Types.DECIMAL, "decimal");
62          registerColumnType(Types.DOUBLE, "float");
63          registerColumnType(Types.FLOAT, "smallfloat");
64  		registerColumnType(Types.INTEGER, "integer");
65  		registerColumnType(Types.LONGVARBINARY, "blob"); // or BYTE
66  		registerColumnType(Types.LONGVARCHAR, "clob"); // or TEXT?
67  		registerColumnType(Types.NUMERIC, "decimal"); // or MONEY
68  		registerColumnType(Types.REAL, "smallfloat");
69  		registerColumnType(Types.SMALLINT, "smallint");
70  		registerColumnType(Types.TIMESTAMP, "datetime year to fraction(5)");
71  		registerColumnType(Types.TIME, "datetime hour to second");
72  		registerColumnType(Types.TINYINT, "smallint");
73  		registerColumnType(Types.VARBINARY, "byte");
74  		registerColumnType(Types.VARCHAR, "varchar($l)");
75  		registerColumnType(Types.VARCHAR, 255, "varchar($l)");
76  		registerColumnType(Types.VARCHAR, 32739, "lvarchar($l)");
77  
78  		registerFunction( "concat", new VarArgsSQLFunction( StandardBasicTypes.STRING, "(", "||", ")" ) );
79  		
80  		uniqueDelegate = new InformixUniqueDelegate( this );
81  	}

108 	/**
109 	 * The syntax used to add a foreign key constraint to a table.
110 	 * Informix constraint name must be at the end.
111 	 * @return String
112 	 */
113 	public String getAddForeignKeyConstraintString(
114 			String constraintName,
115 			String[] foreignKey,
116 			String referencedTable,
117 			String[] primaryKey,
118 			boolean referencesPrimaryKey) {
119 		StringBuilder result = new StringBuilder( 30 )
120 				.append( " add constraint " )
121 				.append( " foreign key (" )
122 				.append( StringHelper.join( ", ", foreignKey ) )
123 				.append( ") references " )
124 				.append( referencedTable );
125 
126 		if ( !referencesPrimaryKey ) {
127 			result.append( " (" )
128 					.append( StringHelper.join( ", ", primaryKey ) )
129 					.append( ')' );
130 		}
131 
132 		result.append( " constraint " ).append( constraintName );
133 
134 		return result.toString();
135 	}

185 	public String getLimitString(String querySelect, int offset, int limit) {
186 		if ( offset > 0 ) {
187 			throw new UnsupportedOperationException( "query result offset is not supported" );
188 		}
189 		return new StringBuilder( querySelect.length() + 8 )
190 				.append( querySelect )
191 				.insert( querySelect.toLowerCase().indexOf( "select" ) + 6, " first " + limit )
192 				.toString();
193 	}

203 	private static ViolatedConstraintNameExtracter EXTRACTER = new TemplatedViolatedConstraintNameExtracter() {
204 
205 		/**
206 		 * Extract the name of the violated constraint from the given SQLException.
207 		 *
208 		 * @param sqle The exception that was the result of the constraint violation.
209 		 * @return The extracted constraint name.
210 		 */
211 		public String extractConstraintName(SQLException sqle) {
212 			String constraintName = null;
213 			
214 			int errorCode = JdbcExceptionHelper.extractErrorCode( sqle );
215 			if ( errorCode == -268 ) {
216 				constraintName = extractUsingTemplate( "Unique constraint (", ") violated.", sqle.getMessage() );
217 			}
218 			else if ( errorCode == -691 ) {
219 				constraintName = extractUsingTemplate( "Missing key in referenced table for referential constraint (", ").", sqle.getMessage() );
220 			}
221 			else if ( errorCode == -692 ) {
222 				constraintName = extractUsingTemplate( "Key value for constraint (", ") is still being referenced.", sqle.getMessage() );
223 			}
224 			
225 			if (constraintName != null) {
226 				// strip table-owner because Informix always returns constraint names as "<table-owner>.<constraint-name>"
227 				int i = constraintName.indexOf('.');
228 				if (i != -1) {
229 					constraintName = constraintName.substring(i + 1);
230 				}
231 			}
232 
233 			return constraintName;
234 		}
235 
236 	};

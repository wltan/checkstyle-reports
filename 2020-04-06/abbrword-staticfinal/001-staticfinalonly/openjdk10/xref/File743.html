24  /* @test @(#)TestOldHangul.java
25   * @summary Verify Old Hangul display
26   * @bug 6886358
27   * @ignore Requires a special font installed.
28   */
29  
30  import javax.swing.*;
31  import javax.swing.border.LineBorder;
32  import java.awt.*;
33  import java.awt.event.ActionEvent;
34  
35  public class TestOldHangul {
36      public static void main(String[] args) {
37          SwingUtilities.invokeLater(new Runnable() {
38              public void run() {
39                  new TestOldHangul().run();
40              }
41          });
42      }
43      public static boolean AUTOMATIC_TEST=true;  // true; run test automatically, else manually at button push
44  
45      private void run()  {
46          Font ourFont = null;
47          final String fontName = "UnBatangOdal.ttf";  // download from http://chem.skku.ac.kr/~wkpark/project/font/GSUB/UnbatangOdal/  and place in {user.home}/fonts/
48          try {
49              ourFont = Font.createFont(Font.TRUETYPE_FONT, new java.io.File(new java.io.File(System.getProperty("user.home"),"fonts"), fontName));
50              ourFont = ourFont.deriveFont((float)48.0);
51          } catch(Throwable t) {
52              t.printStackTrace();
53              System.err.println("Fail: " + t);
54              return;
55          }
56          JFrame frame = new JFrame(System.getProperty("java.version"));
57          frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
58          JPanel panel = new JPanel();
59          final JTextArea label = new JTextArea("(empty)");
60          label.setSize(400, 300);
61          label.setBorder(new LineBorder(Color.black));
62          label.setFont(ourFont);
63          final String str = "\u110A\u119E\u11B7\u0020\u1112\u119E\u11AB\uAE00\u0020\u1100\u119E\u11F9\u0020\u112B\u119E\u11BC\n";
64  
65          if(AUTOMATIC_TEST) {  /* run the test automatically (else, manually) */
66              label.setText(str);
67          } else {
68          JButton button = new JButton("Old Hangul");
69          button.addActionListener(new AbstractAction() {
70              public void actionPerformed(ActionEvent actionEvent) {
71                  label.setText(str);
72              }
73          });
74          panel.add(button);
75          }
76          panel.add(label);
77  
78          frame.getContentPane().add(panel);
79          frame.pack();
80          frame.setVisible(true);
81      }
82  }

24  package org.hibernate.metamodel.source.annotations.xml.filter;
25  
26  import java.util.List;
27  
28  import org.jboss.jandex.AnnotationInstance;
29  import org.jboss.jandex.DotName;
30  
31  /**
32   * if class from index has matched annotations, then remove.
33   *
34   * @author Strong Liu
35   */
36  class NameAnnotationFilter extends AbstractAnnotationFilter {
37  	@Override
38  	protected void process(DotName annName, AnnotationInstance annotationInstance, List<AnnotationInstance> indexedAnnotationInstanceList) {
39  		indexedAnnotationInstanceList.clear();
40  	}
41  
42  	public static NameTargetAnnotationFilter INSTANCE = new NameTargetAnnotationFilter();
43  
44  	@Override
45  	protected DotName[] targetAnnotation() {
46  		return new DotName[] {
47  				CACHEABLE,
48  				TABLE,
49  				EXCLUDE_DEFAULT_LISTENERS,
50  				EXCLUDE_SUPERCLASS_LISTENERS,
51  				ID_CLASS,
52  				INHERITANCE,
53  				DISCRIMINATOR_VALUE,
54  				DISCRIMINATOR_COLUMN,
55  				ENTITY_LISTENERS
56  		};
57  	}
58  }

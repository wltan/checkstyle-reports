24  package org.hibernate.testing.cache;
25  
26  import java.util.Properties;
27  
28  import org.jboss.logging.Logger;
29  
30  import org.hibernate.cache.CacheException;
31  import org.hibernate.cache.spi.CacheDataDescription;
32  import org.hibernate.cache.spi.CollectionRegion;
33  import org.hibernate.cache.spi.EntityRegion;
34  import org.hibernate.cache.spi.NaturalIdRegion;
35  import org.hibernate.cache.spi.QueryResultsRegion;
36  import org.hibernate.cache.spi.RegionFactory;
37  import org.hibernate.cache.spi.TimestampsRegion;
38  import org.hibernate.cache.spi.access.AccessType;
39  import org.hibernate.cfg.Settings;
40  import org.hibernate.internal.CoreMessageLogger;
41  
42  /**
43   * @author Strong Liu
44   */
45  public class CachingRegionFactory implements RegionFactory {
46  	private static final CoreMessageLogger LOG = Logger.getMessageLogger(
47  			CoreMessageLogger.class, CachingRegionFactory.class.getName()
48  	);
49  	public static String DEFAULT_ACCESSTYPE = "DefaultAccessType";
50  	private Settings settings;
51  	private Properties properties;
52  	public CachingRegionFactory() {
53  		LOG.warn( "CachingRegionFactory should be only used for testing." );
54  	}
55  
56  	public CachingRegionFactory(Properties properties) {
57  		//add here to avoid run into catch
58  		LOG.warn( "CachingRegionFactory should be only used for testing." );
59  		this.properties=properties; 
60  	}
61  
62  	@Override
63  	public void start(Settings settings, Properties properties) throws CacheException {
64  		this.settings=settings;
65  		this.properties=properties; 
66  	}
67  
68  	@Override
69  	public void stop() {
70  	}
71  
72  	@Override
73  	public boolean isMinimalPutsEnabledByDefault() {
74  		return false;
75  	}
76  
77  	@Override
78  	public AccessType getDefaultAccessType() {
79  		if (properties != null && properties.get(DEFAULT_ACCESSTYPE) != null) {
80  			return AccessType.fromExternalName(properties.getProperty(DEFAULT_ACCESSTYPE));
81  		}
82  		return AccessType.READ_WRITE;
83  	}
84  
85  	@Override
86  	public long nextTimestamp() {
87  		return Timestamper.next();
88  	}
89  
90  	@Override
91  	public EntityRegion buildEntityRegion(String regionName, Properties properties, CacheDataDescription metadata)
92  			throws CacheException {
93  		return new EntityRegionImpl( regionName, metadata, settings );
94  	}
95  	
96  	@Override
97      public NaturalIdRegion buildNaturalIdRegion(String regionName, Properties properties, CacheDataDescription metadata)
98              throws CacheException {
99          return new NaturalIdRegionImpl( regionName, metadata, settings );
100     }
101 
102     @Override
103 	public CollectionRegion buildCollectionRegion(String regionName, Properties properties, CacheDataDescription metadata)
104 			throws CacheException {
105 		return new CollectionRegionImpl( regionName, metadata, settings );
106 	}
107 
108 	@Override
109 	public QueryResultsRegion buildQueryResultsRegion(String regionName, Properties properties) throws CacheException {
110 		return new QueryResultsRegionImpl( regionName );
111 	}
112 
113 	@Override
114 	public TimestampsRegion buildTimestampsRegion(String regionName, Properties properties) throws CacheException {
115 		return new TimestampsRegionImpl( regionName );
116 	}
117 
118 	private static class QueryResultsRegionImpl extends BaseGeneralDataRegion implements QueryResultsRegion {
119 		QueryResultsRegionImpl(String name) {
120 			super( name );
121 		}
122 	}
123 
124 	private static class TimestampsRegionImpl extends BaseGeneralDataRegion implements TimestampsRegion {
125 		TimestampsRegionImpl(String name) {
126 			super( name );
127 		}
128 	}
129 }

63      /** holds the WADOServiceDelegate instance */
64      protected static WADOServiceDelegate delegate;

69      private static Logger log = Logger.getLogger( WADOServlet.class.getName() );

71      /**
72       * Initialize the WADOServiceDelegator.
73       * <p>
74       * Set the name of the MBean from servlet init param.
75       */
76      public void init() {
77          delegate = new WADOServiceDelegate();
78          delegate.init( getServletConfig() );
79      }

81      /**
82       * Handles the POST requset in the doGet method.
83       * 
84       * @param request 	The http request.
85       * @param response	The http response.
86       */
87      public void doPost( HttpServletRequest request, HttpServletResponse response ){
88          doGet( request, response);
89      }
90  
91      /**
92       * Handles the GET requset.
93       * 
94       * @param request 	The http request.
95       * @param response	The http response.
96       */
97      public void doGet( HttpServletRequest request, HttpServletResponse response ){
98          long twrsp1 = System.currentTimeMillis();
99          log.info(request.getRemoteAddr()+" - WADO URL:"+request.getRequestURI()+"?"+request.getQueryString());
100         BasicRequestObject reqObject = RequestObjectFactory.getRequestObject( request );
101         if ( reqObject == null || ! (reqObject instanceof WADORequestObject) ) {
102             reqObject = RequestObjectFactory.getRequestObject( request );
103             if ( reqObject == null ) {
104                 sendError( response, HttpServletResponse.SC_BAD_REQUEST, "Not A WADO URL" );
105                 return;
106             }
107         }
108         int iErr = reqObject.checkRequest();
109         if ( iErr < 0 ) {
110             sendError( response, HttpServletResponse.SC_BAD_REQUEST, reqObject.getErrorMsg() );//required params missing or invalid!
111             return;
112         }
113         long twget1 = System.currentTimeMillis();
114         WADOResponseObject respObject = delegate.getWADOObject( (WADORequestObject)reqObject );
115         long twget2 = System.currentTimeMillis();
116         int returnCode = respObject.getReturnCode();
117         if ( returnCode == HttpServletResponse.SC_OK ) {
118             sendWADOFile( response, respObject );
119         } else if ( returnCode == HttpServletResponse.SC_TEMPORARY_REDIRECT ) {
120             try {
121                 response.sendRedirect( respObject.getErrorMessage() ); //error message contains redirect host.
122             } catch (IOException e) {
123                 sendError( response, HttpServletResponse.SC_INTERNAL_SERVER_ERROR, "Error: cant send redirect to client! Redirect to host "+respObject.getErrorMessage()+" failed!" );
124             }
125         } else {
126             sendError( response, returnCode, respObject.getErrorMessage() );
127         }
128         long twrsp2 = System.currentTimeMillis();
129         if (log.isDebugEnabled()) 
130             log.debug("TimesToDeliverDICOMObject "+((WADORequestObject)reqObject).getObjectUID()+
131                 ", with requesttype: "+respObject.getContentType()+" to "+((WADORequestObject)reqObject).getRemoteAddr()+
132                 " took total [ms]: " +(twrsp2 -twrsp1)+", FetchObject took [ms]: "+(twget2 -twget1)+
133                 ", Transmission took [ms]: "+((twrsp2 -twrsp1)-(twget2-twget1)));
134     }

137     /**
138      * Send an error response with given response code and message to the client.
139      * <p>
140      * It is recommended that this method is only called once per erequest!<br>
141      * Otherwise a IllegalStateException will be thrown!
142      * 
143      * @param response	The HttpServletResponse of the request.
144      * @param errCode	One of the response code defined in HttpServletResponse.
145      * @param msg		A description for the reason to send the error.
146      */
147     private void sendError( HttpServletResponse response, int errCode, String msg ) {
148         try {
149             response.sendError( errCode, msg );
150         } catch (IOException e) {
151             log.error("Cant perform sendError( "+errCode+", "+msg+" )! reason:"+e.getMessage(), e );
152         }
153     }
154 
155     /**
156      * Send the retrieved file to the client.
157      * <p>
158      * Sets the content type as defined in the WADOResponseObject object.
159      * 
160      * @param response
161      * @param respObject
162      */
163     private void sendWADOFile( HttpServletResponse response, WADOResponseObject respObject ) {
164         response.setHeader("Expires","0");//disables client side caching!!!
165         log.debug("sendResponse:"+respObject);
166         try {
167             if ( respObject != null ) {
168                 log.info("send WADO response: "+respObject.getContentType());
169                 response.setContentType( respObject.getContentType() );
170                 long len = respObject.length();
171                 if ( len != -1 ) 
172                     response.setContentLength((int)len);
173                 final String errMsg = "Exception while writing WADO response to client! reason:";
174                 try {
175                     log.debug("respObject execute");
176                     OutputStream respStream = response.getOutputStream();
177                     try {
178                         respObject.execute(respStream);
179                     } finally {
180                         respStream.close();
181                     }
182                 } catch ( RequestedFrameNumbersOutOfRangeException e ) {
183                     sendError(response, HttpServletResponse.SC_BAD_REQUEST,
184                     "Error: Requested Frame Numbers Out of Range");
185                 } catch ( IOException ioe) {
186                     if(ioe.toString().startsWith("ClientAbortException:")) {
187                         log.debug(errMsg + ioe);
188                     } else {
189                         log.error(errMsg + ioe.getMessage(), ioe);
190                     }
191                 } catch ( Exception e ) {
192                     log.error(errMsg+e.getMessage(), e );
193                 }
194 
195             }
196         } catch ( Exception x ) {
197             log.error("Exception handling WADO, will send error response", x);
198             sendError(	response, HttpServletResponse.SC_INTERNAL_SERVER_ERROR, x.getMessage() );
199         }
200     }

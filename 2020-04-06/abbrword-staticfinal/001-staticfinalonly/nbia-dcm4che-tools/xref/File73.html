202     public final void setConcurrency(int concurrency) throws Exception {
203         if (concurrency <= 0)
204             throw new IllegalArgumentException("Concurrency: " + concurrency);
205         if (this.concurrency != concurrency) {
206             final boolean restart = getState() == STARTED;
207             if (restart)
208                 stop();
209             this.concurrency = concurrency;
210             if (restart)
211                 start();
212         }
213     }

261     public void handleNotification(Notification notif, Object handback) {
262         String spsuid = (String) notif.getUserData();
263         Dataset pps = DcmObjectFactory.getInstance().newDataset();
264         try {
265             Dataset sps;
266             GPWLManager gpwlmgr = getGPWLManager();
267             sps = gpwlmgr.getWorklistItem(spsuid);
268             String ppsiuid = spsuid + ppsuidSuffix;
269             String status = sps.getString(Tags.GPSPSStatus);
270             pps.putCS(Tags.GPPPSStatus, status);
271             pps.putUI(Tags.SOPInstanceUID, ppsiuid);
272             Date now = new Date();
273             if ("IN PROGRESS".equals(status)) {
274                 try {
275                     getGPPPSManager().getGPPPS(ppsiuid);
276                     return; // avoid duplicate N_CREATE
277                 } catch (Exception e) {
278                 }
279                 pps.putSH(Tags.PPSID, "PPS" + ppsiuid.hashCode());
280                 pps.putDA(Tags.PPSStartDate, now);
281                 pps.putTM(Tags.PPSStartTime, now);
282                 pps.putDA(Tags.PPSEndDate);
283                 pps.putTM(Tags.PPSEndTime);
284                 for (int i = 0; i < N_CREATE_TYPE2_ATTRS.length; i++) {
285                     pps.putXX(N_CREATE_TYPE2_ATTRS[i]);
286                 }
287                 pps.putAll(sps.subSet(N_CREATE_SPS_ATTRS));
288                 copyCode(copyWorkitemCode, sps
289                         .getItem(Tags.ScheduledWorkitemCodeSeq), pps
290                         .putSQ(Tags.PerformedWorkitemCodeSeq));
291                 copyCode(copyStationNameCode, sps
292                         .getItem(Tags.ScheduledStationNameCodeSeq), pps
293                         .putSQ(Tags.PerformedStationNameCodeSeq));
294                 copyCode(copyStationClassCode, sps
295                         .getItem(Tags.ScheduledStationClassCodeSeq), pps
296                         .putSQ(Tags.PerformedStationClassCodeSeq));
297                 copyCode(
298                         copyStationGeographicLocationCode,
299                         sps
300                                 .getItem(Tags.ScheduledStationGeographicLocationCodeSeq),
301                         pps
302                                 .putSQ(Tags.PerformedStationGeographicLocationCodeSeq));
303                 copyCode(copyProcessingApplicationsCode, sps
304                         .getItem(Tags.ScheduledProcessingApplicationsCodeSeq),
305                         pps.putSQ(Tags.PerformedProcessingApplicationsCodeSeq));
306             } else if ("COMPLETED".equals(status)
307                     || "DISCONTINUED".equals(status)) {
308                 pps.putDA(Tags.PPSEndDate, now);
309                 pps.putTM(Tags.PPSEndTime, now);
310                 pps.putAll(gpwlmgr.getOutputInformation(spsuid));
311             } else {
312                 return;
313             }
314         } catch (Exception e) {
315             log.error("Failed to access GP-SPS[" + spsuid + "]", e);
316             return;
317         }
318         for (int i = 0; i < destAETs.length; i++) {
319             PPSOrder order = new PPSOrder(pps, destAETs[i]);
320             try {
321                 log.info("Scheduling " + order);
322                 jmsDelegate.queue(queueName, order, Message.DEFAULT_PRIORITY,
323                         0L);
324             } catch (Exception e) {
325                 log.error("Failed to schedule " + order, e);
326             }
327         }
328 
329     }

337     public void onMessage(Message message) {
338         ObjectMessage om = (ObjectMessage) message;
339         try {
340             PPSOrder order = (PPSOrder) om.getObject();
341             log.info("Start processing " + order);
342             try {
343                 sendPPS(order.isCreate(), order.getDataset(), order
344                         .getDestination());
345                 log.info("Finished processing " + order);
346             } catch (Exception e) {
347                 order.setThrowable(e);
348                 final int failureCount = order.getFailureCount() + 1;
349                 order.setFailureCount(failureCount);
350                 final long delay = retryIntervalls.getIntervall(failureCount);
351                 if (delay == -1L) {
352                     log.error("Give up to process " + order, e);
353                     jmsDelegate.fail(queueName, order);
354                 } else {
355                     log.warn("Failed to process " + order
356                             + ". Scheduling retry.", e);
357                     jmsDelegate.queue(queueName, order, 0, System
358                             .currentTimeMillis()
359                             + delay);
360                 }
361             }
362         } catch (JMSException e) {
363             log.error("jms error during processing message: " + message, e);
364         } catch (Throwable e) {
365             log.error("unexpected error during processing message: " + message,
366                     e);
367         }
368     }

370     void sendPPS(boolean create, Dataset pps, String aet)
371             throws Exception {
372         ActiveAssociation aa = openAssociation(aet,
373                 UIDs.GeneralPurposePerformedProcedureStepSOPClass);
374         try {
375             Association a = aa.getAssociation();
376             DcmObjectFactory dof = DcmObjectFactory.getInstance();
377             Command cmdRq = dof.newCommand();
378             final String iuid = pps.getString(Tags.SOPInstanceUID);
379             if (create) {
380                 cmdRq.initNCreateRQ(
381                                 a.nextMsgID(),
382                                 UIDs.GeneralPurposePerformedProcedureStepSOPClass,
383                                 iuid);
384             } else {
385                 cmdRq.initNSetRQ(
386                                 a.nextMsgID(),
387                                 UIDs.GeneralPurposePerformedProcedureStepSOPClass,
388                                 iuid);
389             }
390             Dimse dimseRq = AssociationFactory.getInstance()
391                     .newDimse(PCID_GPPPS, cmdRq, pps.exclude(SOP_IUID));
392             if (log.isDebugEnabled()) {
393                 log.debug("GP-PPS Attributes:");
394                 log.debug(pps);
395             }
396             final Dimse dimseRsp = aa.invoke(dimseRq).get();
397             final Command cmdRsp = dimseRsp.getCommand();
398             final int status = cmdRsp.getStatus();
399             switch (status) {
400             case 0x0000:
401                 break;
402             case 0x0116:
403                 log.warn("Received Warning Status 116H (=Attribute Value Out of Range) from remote AE "
404                                 + aet);
405                 break;
406             default:
407                 throw new DcmServiceException(status,
408                         cmdRsp.getString(Tags.ErrorComment));
409             }
410         } finally {
411             try {
412                 aa.release(true);
413             } catch (Exception e) {
414                 log.warn("Failed to release " + aa.getAssociation());
415             }
416         }

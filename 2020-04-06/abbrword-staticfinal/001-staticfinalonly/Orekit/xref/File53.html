185     @Test
186     public void testIssue229() throws OrekitException {
187         AbsoluteDate initialDate = new AbsoluteDate(2004, 1, 1, 0, 0, 0., TimeScalesFactory.getUTC());
188         Frame frame       = FramesFactory.getEME2000();
189         double rpe         = 160.e3 + Constants.WGS84_EARTH_EQUATORIAL_RADIUS;
190         double rap         = 2000.e3 + Constants.WGS84_EARTH_EQUATORIAL_RADIUS;
191         double inc         = FastMath.toRadians(0.);
192         double aop         = FastMath.toRadians(0.);
193         double raan        = FastMath.toRadians(0.);
194         double mean        = FastMath.toRadians(180.);
195         double mass        = 100.;
196         KeplerianOrbit orbit = new KeplerianOrbit(0.5 * (rpe + rap), (rap - rpe) / (rpe + rap),
197                                                   inc, aop, raan, mean, PositionAngle.MEAN,
198                                                   frame, initialDate, Constants.EIGEN5C_EARTH_MU);
199 
200         IsotropicDrag shape = new IsotropicDrag(10., 2.2);
201 
202         Frame itrf = FramesFactory.getITRF(IERSConventions.IERS_2010, true);
203         BodyShape earthShape = new OneAxisEllipsoid(Constants.WGS84_EARTH_EQUATORIAL_RADIUS, Constants.WGS84_EARTH_FLATTENING, itrf);
204         Atmosphere atmosphere = new SimpleExponentialAtmosphere(earthShape, 2.6e-10, 200000, 26000);
205 
206         double[][]          tolerance  = NumericalPropagator.tolerances(0.1, orbit, OrbitType.CARTESIAN);
207         AbstractIntegrator  integrator = new DormandPrince853Integrator(1.0e-3, 300, tolerance[0], tolerance[1]);
208         NumericalPropagator propagator = new NumericalPropagator(integrator);
209         propagator.setOrbitType(OrbitType.CARTESIAN);
210         propagator.setMu(orbit.getMu());
211         propagator.addForceModel(new DragForce(atmosphere, shape));
212         PartialDerivativesEquations partials = new PartialDerivativesEquations("partials", propagator);
213         propagator.setInitialState(partials.setInitialJacobians(new SpacecraftState(orbit, mass), 6));
214 
215         SpacecraftState state = propagator.propagate(new AbsoluteDate(2004, 1, 1, 1, 30, 0., TimeScalesFactory.getUTC()));
216 
217         double delta = 0.1;
218         Orbit shifted = new CartesianOrbit(new TimeStampedPVCoordinates(orbit.getDate(),
219                                                                         orbit.getPVCoordinates().getPosition().add(new Vector3D(delta, 0, 0)),
220                                                                         orbit.getPVCoordinates().getVelocity()),
221                                            orbit.getFrame(), orbit.getMu());
222         propagator.setInitialState(partials.setInitialJacobians(new SpacecraftState(shifted, mass), 6));
223         SpacecraftState newState = propagator.propagate(new AbsoluteDate(2004, 1, 1, 1, 30, 0., TimeScalesFactory.getUTC()));
224         double[] dPVdX = new double[] {
225             (newState.getPVCoordinates().getPosition().getX() - state.getPVCoordinates().getPosition().getX()) / delta,
226             (newState.getPVCoordinates().getPosition().getY() - state.getPVCoordinates().getPosition().getY()) / delta,
227             (newState.getPVCoordinates().getPosition().getZ() - state.getPVCoordinates().getPosition().getZ()) / delta,
228             (newState.getPVCoordinates().getVelocity().getX() - state.getPVCoordinates().getVelocity().getX()) / delta,
229             (newState.getPVCoordinates().getVelocity().getY() - state.getPVCoordinates().getVelocity().getY()) / delta,
230             (newState.getPVCoordinates().getVelocity().getZ() - state.getPVCoordinates().getVelocity().getZ()) / delta,
231         };
232 
233         double[][] dYdY0 = new double[6][6];
234         partials.getMapper().getStateJacobian(state, dYdY0);
235         for (int i = 0; i < 6; ++i) {
236             Assert.assertEquals(dPVdX[i], dYdY0[i][0], 4.5e-6 * FastMath.abs(dPVdX[i]));
237         }
238 
239     }

106     @Test
107     public void testShift() throws OrekitException {
108         double rate = 2 * FastMath.PI / (12 * 60);
109         TimeStampedAngularCoordinates ac =
110                 new TimeStampedAngularCoordinates(AbsoluteDate.J2000_EPOCH,
111                                                   Rotation.IDENTITY,
112                                                   new Vector3D(rate, Vector3D.PLUS_K), Vector3D.ZERO);
113         Assert.assertEquals(rate, ac.getRotationRate().getNorm(), 1.0e-10);
114         double dt = 10.0;
115         double alpha = rate * dt;
116         TimeStampedAngularCoordinates shifted = ac.shiftedBy(dt);
117         Assert.assertEquals(rate, shifted.getRotationRate().getNorm(), 1.0e-10);
118         Assert.assertEquals(alpha, Rotation.distance(ac.getRotation(), shifted.getRotation()), 1.0e-10);
119 
120         Vector3D xSat = shifted.getRotation().applyInverseTo(Vector3D.PLUS_I);
121         Assert.assertEquals(0.0, xSat.subtract(new Vector3D(FastMath.cos(alpha), FastMath.sin(alpha), 0)).getNorm(), 1.0e-10);
122         Vector3D ySat = shifted.getRotation().applyInverseTo(Vector3D.PLUS_J);
123         Assert.assertEquals(0.0, ySat.subtract(new Vector3D(-FastMath.sin(alpha), FastMath.cos(alpha), 0)).getNorm(), 1.0e-10);
124         Vector3D zSat = shifted.getRotation().applyInverseTo(Vector3D.PLUS_K);
125         Assert.assertEquals(0.0, zSat.subtract(Vector3D.PLUS_K).getNorm(), 1.0e-10);

129     @Test
130     public void testSpin() throws OrekitException {
131         double rate = 2 * FastMath.PI / (12 * 60);
132         TimeStampedAngularCoordinates ac =
133                 new TimeStampedAngularCoordinates(AbsoluteDate.J2000_EPOCH,
134                                                   new Rotation(0.48, 0.64, 0.36, 0.48, false),
135                                                   new Vector3D(rate, Vector3D.PLUS_K), Vector3D.ZERO);
136         Assert.assertEquals(rate, ac.getRotationRate().getNorm(), 1.0e-10);
137         double dt = 10.0;
138         TimeStampedAngularCoordinates shifted = ac.shiftedBy(dt);
139         Assert.assertEquals(rate, shifted.getRotationRate().getNorm(), 1.0e-10);
140         Assert.assertEquals(rate * dt, Rotation.distance(ac.getRotation(), shifted.getRotation()), 1.0e-10);
141 
142         Vector3D shiftedX  = shifted.getRotation().applyInverseTo(Vector3D.PLUS_I);
143         Vector3D shiftedY  = shifted.getRotation().applyInverseTo(Vector3D.PLUS_J);
144         Vector3D shiftedZ  = shifted.getRotation().applyInverseTo(Vector3D.PLUS_K);
145         Vector3D originalX = ac.getRotation().applyInverseTo(Vector3D.PLUS_I);
146         Vector3D originalY = ac.getRotation().applyInverseTo(Vector3D.PLUS_J);
147         Vector3D originalZ = ac.getRotation().applyInverseTo(Vector3D.PLUS_K);
148         Assert.assertEquals( FastMath.cos(rate * dt), Vector3D.dotProduct(shiftedX, originalX), 1.0e-10);
149         Assert.assertEquals( FastMath.sin(rate * dt), Vector3D.dotProduct(shiftedX, originalY), 1.0e-10);
150         Assert.assertEquals( 0.0,                 Vector3D.dotProduct(shiftedX, originalZ), 1.0e-10);
151         Assert.assertEquals(-FastMath.sin(rate * dt), Vector3D.dotProduct(shiftedY, originalX), 1.0e-10);
152         Assert.assertEquals( FastMath.cos(rate * dt), Vector3D.dotProduct(shiftedY, originalY), 1.0e-10);
153         Assert.assertEquals( 0.0,                 Vector3D.dotProduct(shiftedY, originalZ), 1.0e-10);
154         Assert.assertEquals( 0.0,                 Vector3D.dotProduct(shiftedZ, originalX), 1.0e-10);
155         Assert.assertEquals( 0.0,                 Vector3D.dotProduct(shiftedZ, originalY), 1.0e-10);
156         Assert.assertEquals( 1.0,                 Vector3D.dotProduct(shiftedZ, originalZ), 1.0e-10);
157 
158         Vector3D forward = TimeStampedAngularCoordinates.estimateRate(ac.getRotation(), shifted.getRotation(), dt);
159         Assert.assertEquals(0.0, forward.subtract(ac.getRotationRate()).getNorm(), 1.0e-10);
160 
161         Vector3D reversed = TimeStampedAngularCoordinates.estimateRate(shifted.getRotation(), ac.getRotation(), dt);
162         Assert.assertEquals(0.0, reversed.add(ac.getRotationRate()).getNorm(), 1.0e-10);
163 
164     }

292     private double[] interpolationErrors(final TimeStampedAngularCoordinates reference, double dt)
293         throws OrekitException {
294 
295         final OrdinaryDifferentialEquation ode = new OrdinaryDifferentialEquation() {
296             public int getDimension() {
297                 return 4;
298             }
299             public double[] computeDerivatives(final double t, final double[] q) {
300                 final double omegaX = reference.getRotationRate().getX() + t * reference.getRotationAcceleration().getX();
301                 final double omegaY = reference.getRotationRate().getY() + t * reference.getRotationAcceleration().getY();
302                 final double omegaZ = reference.getRotationRate().getZ() + t * reference.getRotationAcceleration().getZ();
303                 return new double[] {
304                     0.5 * MathArrays.linearCombination(-q[1], omegaX, -q[2], omegaY, -q[3], omegaZ),
305                     0.5 * MathArrays.linearCombination( q[0], omegaX, -q[3], omegaY,  q[2], omegaZ),
306                     0.5 * MathArrays.linearCombination( q[3], omegaX,  q[0], omegaY, -q[1], omegaZ),
307                     0.5 * MathArrays.linearCombination(-q[2], omegaX,  q[1], omegaY,  q[0], omegaZ)
308                 };
309             }
310         };
311         final List<TimeStampedAngularCoordinates> complete = new ArrayList<TimeStampedAngularCoordinates>();
312         ODEIntegrator integrator = new DormandPrince853Integrator(1.0e-6, 1.0, 1.0e-12, 1.0e-12);
313         integrator.addStepHandler(new StepNormalizer(dt / 2000, new ODEFixedStepHandler() {
314             public void handleStep(ODEStateAndDerivative state, boolean isLast) {
315                 final double   t = state.getTime();
316                 final double[] q = state.getPrimaryState();
317                 complete.add(new TimeStampedAngularCoordinates(reference.getDate().shiftedBy(t),
318                                                                new Rotation(q[0], q[1], q[2], q[3], true),
319                                                                new Vector3D(1, reference.getRotationRate(),
320                                                                             t, reference.getRotationAcceleration()),
321                                                                reference.getRotationAcceleration()));
322             }
323         }));
324 
325         double[] y = new double[] {
326             reference.getRotation().getQ0(), 
327             reference.getRotation().getQ1(), 
328             reference.getRotation().getQ2(), 
329             reference.getRotation().getQ3()
330         };
331         integrator.integrate(ode, new ODEState(0, y), dt);
332 
333         List<TimeStampedAngularCoordinates> sample = new ArrayList<TimeStampedAngularCoordinates>();
334         sample.add(complete.get(0));
335         sample.add(complete.get(complete.size() / 2));
336         sample.add(complete.get(complete.size() - 1));
337 
338         double maxRotationError     = 0;
339         double maxRateError         = 0;
340         double maxAccelerationError = 0;
341         for (TimeStampedAngularCoordinates acRef : complete) {
342             TimeStampedAngularCoordinates interpolated =
343                     TimeStampedAngularCoordinates.interpolate(acRef.getDate(), AngularDerivativesFilter.USE_RRA, sample);
344             double rotationError     = Rotation.distance(acRef.getRotation(), interpolated.getRotation());
345             double rateError         = Vector3D.distance(acRef.getRotationRate(), interpolated.getRotationRate());
346             double accelerationError = Vector3D.distance(acRef.getRotationAcceleration(), interpolated.getRotationAcceleration());
347             maxRotationError         = FastMath.max(maxRotationError,     rotationError);
348             maxRateError             = FastMath.max(maxRateError,         rateError);
349             maxAccelerationError     = FastMath.max(maxAccelerationError, accelerationError);
350         }
351 
352         return new double[] {
353           maxRotationError, maxRateError, maxAccelerationError  
354         };
355 
356     }

471     private Vector3D randomVector(RandomGenerator random, double norm) {
472         double n = random.nextDouble() * norm;
473         double x = 2 * random.nextDouble() - 1;
474         double y = 2 * random.nextDouble() - 1;
475         double z = 2 * random.nextDouble() - 1;
476         return new Vector3D(n, new Vector3D(x, y, z).normalize());
477     }

479     private PVCoordinates randomPVCoordinates(RandomGenerator random,
480                                               double norm0, double norm1, double norm2) {
481         Vector3D p0 = randomVector(random, norm0);
482         Vector3D p1 = randomVector(random, norm1);
483         Vector3D p2 = randomVector(random, norm2);
484         return new PVCoordinates(p0, p1, p2);
485     }

487     private Rotation randomRotation(RandomGenerator random) {
488         double q0 = random.nextDouble() * 2 - 1;
489         double q1 = random.nextDouble() * 2 - 1;
490         double q2 = random.nextDouble() * 2 - 1;
491         double q3 = random.nextDouble() * 2 - 1;
492         double q  = FastMath.sqrt(q0 * q0 + q1 * q1 + q2 * q2 + q3 * q3);
493         return new Rotation(q0 / q, q1 / q, q2 / q, q3 / q, false);
494     }

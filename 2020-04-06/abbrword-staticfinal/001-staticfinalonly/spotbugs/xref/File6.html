47  /**
48   * Pulldown toolbar action for classifying a FindBugs warning as "bug" or
49   * "not a bug".

52   */
53  public class AccuracyClassificationPulldownAction implements IWorkbenchWindowPulldownDelegate2 {

55      protected static boolean DEBUG;

57      private Menu menu;
58  
59      private MenuItem isBugItem;
60  
61      private MenuItem notBugItem;

63      private IMarker marker;
64  
65      private BugInstance bugInstance;

79      /*
80       * (non-Javadoc)
81       *
82       * @see
83       * org.eclipse.ui.IWorkbenchWindowPulldownDelegate#getMenu(org.eclipse.swt
84       * .widgets.Control)
85       */
86      @Override
87      public Menu getMenu(Control parent) {
88          if (menu == null) {
89              menu = new Menu(parent);
90              fillMenu();
91          }
92          return menu;
93      }

95      /**
96       * Fill the classification menu.
97       */
98      private void fillMenu() {
99          isBugItem = new MenuItem(menu, SWT.RADIO);
100         isBugItem.setText("Bug");
101         notBugItem = new MenuItem(menu, SWT.RADIO);
102         notBugItem.setText("Not Bug");
103 
104         isBugItem.addSelectionListener(new SelectionAdapter() {
105             /*
106              * (non-Javadoc)
107              *
108              * @see
109              * org.eclipse.swt.events.SelectionAdapter#widgetSelected(org.eclipse
110              * .swt.events.SelectionEvent)
111              */
112             @Override
113             public void widgetSelected(SelectionEvent e) {
114                 if (bugInstance != null) {
115                     classifyWarning(bugInstance, true);
116                 }
117             }
118         });
119 
120         notBugItem.addSelectionListener(new SelectionAdapter() {
121             /*
122              * (non-Javadoc)
123              *
124              * @see
125              * org.eclipse.swt.events.SelectionAdapter#widgetSelected(org.eclipse
126              * .swt.events.SelectionEvent)
127              */
128             @Override
129             public void widgetSelected(SelectionEvent e) {
130                 if (bugInstance != null) {
131                     classifyWarning(bugInstance, false);
132                 }
133             }
134         });
135 
136         menu.addMenuListener(new MenuAdapter() {
137             @Override
138             public void menuShown(MenuEvent e) {
139                 // Before showing the menu, sync its contents
140                 // with the current BugInstance (if any)
141                 if (DEBUG) {
142                     System.out.println("Synchronizing menu!");
143                 }
144                 syncMenu();
145             }
146         });
147     }
148 
149     private void classifyWarning(BugInstance warning, boolean isBug) {
150 
151         BugProperty isBugProp = warning.lookupProperty(BugProperty.IS_BUG);
152         if (isBugProp != null) {
153             // Warning was previously classified
154             if (isBugProp.getValueAsBoolean() == isBug) {
155                 // No change
156                 return;
157             }
158         }
159 
160         // Warning is being classified for the first time,
161         // or the classification is being changed
162         warning.setProperty(BugProperty.IS_BUG, isBug ? "true" : "false");
163         try {
164             FindbugsPlugin.markBugCollectionDirty(marker.getResource().getProject(), true);
165         } catch (CoreException e) {
166             FindbugsPlugin.getDefault().logException(e, "Could not mark bug collection dirty");
167         }
168 
169         // Currently, we are displaying a marker for this warning.
170         // If the user has classified it as a false warning,
171         // and false warnings are not being displayed, then we can
172         // remove the marker.
173         if (!isBug) {
174             IProject project = marker.getResource().getProject();
175             try {
176                 UserPreferences userPrefs = FindbugsPlugin.getUserPreferences(project);
177                 if (!MarkerUtil.shouldDisplayWarning(warning, userPrefs.getFilterSettings())) {
178                     if (DEBUG) {
179                         System.out.println("Deleting marker for false warning!");
180                     }
181                     marker.delete();
182                 }
183             } catch (CoreException e) {
184                 FindbugsPlugin.getDefault().logException(e, "Could not get SpotBugs preferences for project");
185             }
186         }
187     }

223     /*
224      * (non-Javadoc)
225      *
226      * @see
227      * org.eclipse.ui.IActionDelegate#selectionChanged(org.eclipse.jface.action
228      * .IAction, org.eclipse.jface.viewers.ISelection)
229      */
230     @Override
231     public void selectionChanged(IAction action, ISelection selection) {
232 
233         bugInstance = null;
234         // TODO learn to deal with ALL elements
235         IMarker markerFromSelection = MarkerUtil.getMarkerFromSingleSelection(selection);
236         marker = markerFromSelection;
237 
238         if (marker == null) {
239             // No marker selected.
240             return;
241         }
242         if (DEBUG) {
243             System.out.println("Found a marker!");
244         }
245 
246         bugInstance = MarkerUtil.findBugInstanceForMarker(marker);
247         if (bugInstance != null) {
248             if (DEBUG) {
249                 System.out.println("Found BugInstance for SpotBugs warning marker!");
250             }
251         }
252     }
253 
254     /**
255      * Update menu to match currently selected BugInstance.
256      */
257     private void syncMenu() {
258         if (bugInstance != null) {
259             isBugItem.setEnabled(true);
260             notBugItem.setEnabled(true);
261 
262             BugProperty isBugProperty = bugInstance.lookupProperty(BugProperty.IS_BUG);
263             if (isBugProperty == null) {
264                 // Unclassified
265                 isBugItem.setSelection(false);
266                 notBugItem.setSelection(false);
267             } else {
268                 boolean isBug = isBugProperty.getValueAsBoolean();
269                 isBugItem.setSelection(isBug);
270                 notBugItem.setSelection(!isBug);
271             }
272         } else {
273             // No bug instance, so uncheck and disable the menu items
274             if (DEBUG) {
275                 System.out.println("No bug instance found, disabling menu items");
276             }
277             isBugItem.setEnabled(false);
278             notBugItem.setEnabled(false);
279             isBugItem.setSelection(false);
280             notBugItem.setSelection(false);
281         }
282     }

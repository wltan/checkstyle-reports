20  package edu.umd.cs.findbugs.jaif;
21  
22  /**
23   * Callbacks for parsing an extenal annotation file.
24   *
25   * @author David Hovemeyer
26   * @see <a
27   *      href="http://groups.csail.mit.edu/pag/jsr308/annotation-file-utilities/">Annotation
28   *      File Utilities/</a>
29   */
30  public interface JAIFEvents {
31  
32      /**
33       * Called to indicate the start of a package definition.
34       *
35       * @param pkgName
36       *            package name
37       */
38      void startPackageDefinition(String pkgName);
39  
40      /**
41       * Called to indicate the end of a package definition.
42       *
43       * @param pkgName
44       */
45      void endPackageDefinition(String pkgName);
46  
47      /**
48       * Called to indicate the start of an annotation.
49       *
50       * @param annotationName
51       *            annotation name
52       */
53      void startAnnotation(String annotationName);
54  
55      /**
56       * Called to indicate the end of an annotation.
57       *
58       * @param annotationName
59       *            annotation name
60       */
61      void endAnnotation(String annotationName);
62  
63      /**
64       * Called to visit an annotation field.
65       *
66       * @param fieldName
67       *            annotation field name
68       * @param constant
69       *            constant value of the annotation field (one of the java.lang
70       *            wrapper types, or a String, or ???)
71       */
72      void annotationField(String fieldName, Object constant);
73  
74      /**
75       * Called to indicate the start of an annotation definition.
76       *
77       * @param annotationName
78       *            name of the annotation
79       * @param retention
80       *            retention: one of "visible", "invisible", or "source"
81       */
82      void startAnnotationDefinition(String annotationName, String retention);
83  
84      /**
85       * Called to indicate the end of an annotation definition.
86       *
87       * @param annotationName
88       *            name of the annotation
89       */
90      void endAnnotationDefinition(String annotationName);
91  
92      /**
93       * Called to visit an annotation field definition.
94       *
95       * @param type
96       *            type of the annotation field (in JVM signature format)
97       * @param fieldName
98       *            name of the annotation field
99       */
100     void annotationFieldDefinition(String type, String fieldName);
101 }

37  /**
38   * @test
39   * @bug 8048830
40   * @summary Test for feature 'support stronger entry protection'. An entry is
41   * stored to keystore with different PasswordProtection objects which are
42   * specified by different PBE algorithms (use -Dseed=X to set PRNG seed)
43   * @library /lib/testlibrary ../
44   * @key randomness
45   */
46  public class EntryProtectionTest {
47      private static final char[] PASSWORD = "passwd".toCharArray();
48      private static final String ALIAS = "testkey";
49      private static final byte[] SALT = new byte[8];
50      private static final int ITERATION_COUNT = 1024;
51      private static final List<KeyStore.PasswordProtection> PASSWORD_PROTECTION
52              = new ArrayList<>();
53      private static final String KEYSTORE_PATH = System.getProperty(
54              "test.classes" + File.separator + "ks.pkcs12",
55              "." + File.separator + "ks.pkcs12");
56  
57      private void runTest() throws Exception {
58              KeyStore ksIn = Utils.loadKeyStore(KEYSTORE_PATH,
59                      Utils.KeyStoreType.pkcs12, PASSWORD);
60              KeyStore ksTest = KeyStore
61                      .getInstance(Utils.KeyStoreType.pkcs12.name());
62              ksTest.load(null);
63              Certificate cert = ksIn.getCertificate(ALIAS);
64              Key key = ksIn.getKey(ALIAS, PASSWORD);
65              KeyStore.Entry keyStoreEntry = new KeyStore.PrivateKeyEntry(
66                      (PrivateKey) key, new Certificate[]{cert});
67              for (KeyStore.PasswordProtection passwordAlgorithm :
68                      PASSWORD_PROTECTION) {
69                  out.println("Try to use: " +
70                          passwordAlgorithm.getProtectionAlgorithm());
71                  ksTest.setEntry(ALIAS, keyStoreEntry, passwordAlgorithm);
72                  KeyStore.Entry entryRead = ksTest.getEntry(ALIAS,
73                          new KeyStore.PasswordProtection(PASSWORD));
74                  if (!isPrivateKeyEntriesEqual((KeyStore.PrivateKeyEntry)
75                          keyStoreEntry, (KeyStore.PrivateKeyEntry)entryRead)) {
76                      err.println("Original entry in KeyStore: " + keyStoreEntry);
77                      err.println("Enc/Dec entry : " + entryRead);
78                      throw new RuntimeException(
79                              String.format(
80                                      "Decrypted & original enities do "
81                                      + "not match. Algo: %s, Actual: %s, "
82                                      + "Expected: %s",
83                                      passwordAlgorithm.getProtectionAlgorithm(),
84                                      entryRead, keyStoreEntry));
85                  }
86                  ksTest.deleteEntry(ALIAS);
87              }
88              out.println("Test Passed");
89      }
90  
91      public static void main(String args[]) throws Exception {
92          EntryProtectionTest entryProtectionTest = new EntryProtectionTest();
93          entryProtectionTest.setUp();
94          entryProtectionTest.runTest();
95      }
96  
97      private void setUp() {
98          out.println("Using KEYSTORE_PATH:"+KEYSTORE_PATH);
99          Utils.createKeyStore(Utils.KeyStoreType.pkcs12, KEYSTORE_PATH, ALIAS);
100         Random rand = RandomFactory.getRandom();
101         rand.nextBytes(SALT);
102         out.print("Salt: ");
103         for (byte b : SALT) {
104             out.format("%02X ", b);
105         }
106         out.println("");
107         PASSWORD_PROTECTION
108                 .add(new KeyStore.PasswordProtection(PASSWORD,
109                                 "PBEWithMD5AndDES", new PBEParameterSpec(SALT,
110                                         ITERATION_COUNT)));
111         PASSWORD_PROTECTION.add(new KeyStore.PasswordProtection(PASSWORD,
112                 "PBEWithSHA1AndDESede", null));
113         PASSWORD_PROTECTION.add(new KeyStore.PasswordProtection(PASSWORD,
114                 "PBEWithSHA1AndRC2_40", null));
115         PASSWORD_PROTECTION.add(new KeyStore.PasswordProtection(PASSWORD,
116                 "PBEWithSHA1AndRC2_128", null));
117         PASSWORD_PROTECTION.add(new KeyStore.PasswordProtection(PASSWORD,
118                 "PBEWithSHA1AndRC4_40", null));
119         PASSWORD_PROTECTION.add(new KeyStore.PasswordProtection(PASSWORD,
120                 "PBEWithSHA1AndRC4_128", null));
121     }
122 
123     /**
124      * Checks whether given two KeyStore.PrivateKeyEntry parameters are equal
125      * The KeyStore.PrivateKeyEntry fields like {privateKey, certificateChain[]}
126      * are checked for equality and another field Set<attributes> is not checked
127      * as default implementation adds few PKCS12 attributes during read
128      * operation
129      * @param  first
130      *         parameter is of type KeyStore.PrivateKeyEntry
131      * @param  second
132      *         parameter is of type KeyStore.PrivateKeyEntry
133      * @return boolean
134      *         true when both the KeyStore.PrivateKeyEntry fields are equal
135     */
136     boolean isPrivateKeyEntriesEqual(KeyStore.PrivateKeyEntry first,
137             KeyStore.PrivateKeyEntry second) {
138         //compare privateKey
139         if (!Arrays.equals(first.getPrivateKey().getEncoded(),
140                 second.getPrivateKey().getEncoded())) {
141             err.println("Mismatch found in privateKey!");
142             return false;
143         }
144         //compare certificateChain[]
145         if (!Arrays.equals(first.getCertificateChain(),
146                 second.getCertificateChain())) {
147             err.println("Mismatch found in certificate chain!");
148             return false;
149         }
150         return true;
151     }

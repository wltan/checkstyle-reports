49  
50  
51  
52  /**
53   * class CK_MECHANISM_INFO provides information about a particular mechanism.
54   * <p>
55   * <B>PKCS#11 structure:</B>
56   * <PRE>
57   * typedef struct CK_MECHANISM_INFO {&nbsp;&nbsp;
58   *   CK_ULONG ulMinKeySize;&nbsp;&nbsp;
59   *   CK_ULONG ulMaxKeySize;&nbsp;&nbsp;
60   *   CK_FLAGS flags;&nbsp;&nbsp;
61   * } CK_MECHANISM_INFO;
62   * </PRE>
63   *
64   * @author Karl Scheibelhofer <Karl.Scheibelhofer@iaik.at>
65   * @author Martin Schlaeffer <schlaeff@sbox.tugraz.at>
66   */
67  public class CK_MECHANISM_INFO {
68  
69      /**
70       * <B>PKCS#11:</B>
71       * <PRE>
72       *   CK_ULONG ulMinKeySize;
73       * </PRE>
74       */
75      public long ulMinKeySize;
76  
77      /**
78       * <B>PKCS#11:</B>
79       * <PRE>
80       *   CK_ULONG ulMaxKeySize;
81       * </PRE>
82       */
83      public long ulMaxKeySize;
84  
85      /**
86       * <B>PKCS#11:</B>
87       * <PRE>
88       *   CK_FLAGS flags;
89       * </PRE>
90       */
91      public long flags;
92  
93      public CK_MECHANISM_INFO(long minKeySize, long maxKeySize,
94                               long flags) {
95          this.ulMinKeySize = minKeySize;
96          this.ulMaxKeySize = maxKeySize;
97          this.flags = flags;
98      }
99  
100     /**
101      * Returns the string representation of CK_MECHANISM_INFO.
102      *
103      * @return the string representation of CK_MECHANISM_INFO
104      */
105     public String toString() {
106         StringBuilder sb = new StringBuilder();
107 
108         sb.append(Constants.INDENT);
109         sb.append("ulMinKeySize: ");
110         sb.append(String.valueOf(ulMinKeySize));
111         sb.append(Constants.NEWLINE);
112 
113         sb.append(Constants.INDENT);
114         sb.append("ulMaxKeySize: ");
115         sb.append(String.valueOf(ulMaxKeySize));
116         sb.append(Constants.NEWLINE);
117 
118         sb.append(Constants.INDENT);
119         sb.append("flags: ");
120         sb.append(String.valueOf(flags));
121         sb.append(" = ");
122         sb.append(Functions.mechanismInfoFlagsToString(flags));
123         //buffer.append(Constants.NEWLINE);
124 
125         return sb.toString() ;
126     }
127 }

56  /**
57   * An Http2Connection. Encapsulates the socket(channel) and any SSLEngine used
58   * over it. Contains an HttpConnection which hides the SocketChannel SSL stuff.
59   *
60   * Http2Connections belong to a Http2ClientImpl, (one of) which belongs
61   * to a HttpClientImpl.
62   *
63   * Creation cases:
64   * 1) upgraded HTTP/1.1 plain tcp connection
65   * 2) prior knowledge directly created plain tcp connection
66   * 3) directly created HTTP/2 SSL connection which uses ALPN.
67   *
68   * Sending is done by writing directly to underlying HttpConnection object which
69   * is operating in async mode. No flow control applies on output at this level
70   * and all writes are just executed as puts to an output Q belonging to HttpConnection
71   * Flow control is implemented by HTTP/2 protocol itself.
72   *
73   * Hpack header compression
74   * and outgoing stream creation is also done here, because these operations
75   * must be synchronized at the socket level. Stream objects send frames simply
76   * by placing them on the connection's output Queue. sendFrame() is called
77   * from a higher level (Stream) thread.
78   *
79   * asyncReceive(ByteBuffer) is always called from the selector thread. It assembles
80   * incoming Http2Frames, and directs them to the appropriate Stream.incoming()
81   * or handles them directly itself. This thread performs hpack decompression
82   * and incoming stream creation (Server push). Incoming frames destined for a
83   * stream are provided by calling Stream.incoming().
84   */
85  class Http2Connection  {
86      /*
87       *  ByteBuffer pooling strategy for HTTP/2 protocol:
88       *
89       * In general there are 4 points where ByteBuffers are used:
90       *  - incoming/outgoing frames from/to ByteBufers plus incoming/outgoing encrypted data
91       *    in case of SSL connection.
92       *
93       * 1. Outgoing frames encoded to ByteBuffers.
94       *    Outgoing ByteBuffers are created with requited size and frequently small (except DataFrames, etc)
95       *    At this place no pools at all. All outgoing buffers should be collected by GC.
96       *
97       * 2. Incoming ByteBuffers (decoded to frames).
98       *    Here, total elimination of BB pool is not a good idea.
99       *    We don't know how many bytes we will receive through network.
100      * So here we allocate buffer of reasonable size. The following life of the BB:
101      * - If all frames decoded from the BB are other than DataFrame and HeaderFrame (and HeaderFrame subclasses)
102      *     BB is returned to pool,
103      * - If we decoded DataFrame from the BB. In that case DataFrame refers to subbuffer obtained by slice() method.
104      *     Such BB is never returned to pool and will be GCed.
105      * - If we decoded HeadersFrame from the BB. Then header decoding is performed inside processFrame method and
106      *     the buffer could be release to pool.
107      *
108      * 3. SLL encrypted buffers. Here another pool was introduced and all net buffers are to/from the pool,
109      *    because of we can't predict size encrypted packets.
110      *
111      */

114     // A small class that allows to control frames with respect to the state of
115     // the connection preface. Any data received before the connection
116     // preface is sent will be buffered.
117     private final class FramesController {
118         volatile boolean prefaceSent;
119         volatile List<ByteBufferReference> pending;
120 
121         boolean processReceivedData(FramesDecoder decoder, ByteBufferReference buf)
122                 throws IOException
123         {
124             // if preface is not sent, buffers data in the pending list
125             if (!prefaceSent) {
126                 synchronized (this) {
127                     if (!prefaceSent) {
128                         if (pending == null) pending = new ArrayList<>();
129                         pending.add(buf);
130                         return false;
131                     }
132                 }
133             }
134 
135             // Preface is sent. Checks for pending data and flush it.
136             // We rely on this method being called from within the readlock,
137             // so we know that no other thread could execute this method
138             // concurrently while we're here.
139             // This ensures that later incoming buffers will not
140             // be processed before we have flushed the pending queue.
141             // No additional synchronization is therefore necessary here.
142             List<ByteBufferReference> pending = this.pending;
143             this.pending = null;
144             if (pending != null) {
145                 // flush pending data
146                 for (ByteBufferReference b : pending) {
147                     decoder.decode(b);
148                 }
149             }
150 
151             // push the received buffer to the frames decoder.
152             decoder.decode(buf);
153             return true;
154         }
155 
156         // Mark that the connection preface is sent
157         void markPrefaceSent() {
158             assert !prefaceSent;
159             synchronized (this) {
160                 prefaceSent = true;
161             }
162         }
163 
164     }

197     private Http2Connection(HttpConnection connection,
198                             Http2ClientImpl client2,
199                             int nextstreamid,
200                             String key) {
201         this.connection = connection;
202         this.client = client2.client();
203         this.client2 = client2;
204         this.nextstreamid = nextstreamid;
205         this.key = key;
206         this.clientSettings = this.client2.getClientSettings();
207         this.framesDecoder = new FramesDecoder(this::processFrame, clientSettings.getParameter(SettingsFrame.MAX_FRAME_SIZE));
208         // serverSettings will be updated by server
209         this.serverSettings = SettingsFrame.getDefaultSettings();
210         this.hpackOut = new Encoder(serverSettings.getParameter(HEADER_TABLE_SIZE));
211         this.hpackIn = new Decoder(clientSettings.getParameter(HEADER_TABLE_SIZE));
212         this.windowUpdater = new ConnectionWindowUpdateSender(this, client.getReceiveBufferSize());
213     }
214         /**
215          * Case 1) Create from upgraded HTTP/1.1 connection.
216          * Is ready to use. Will not be SSL. exchange is the Exchange
217          * that initiated the connection, whose response will be delivered
218          * on a Stream.
219          */
220     Http2Connection(HttpConnection connection,
221                     Http2ClientImpl client2,
222                     Exchange<?> exchange,
223                     ByteBuffer initial)
224         throws IOException, InterruptedException
225     {
226         this(connection,
227                 client2,
228                 3, // stream 1 is registered during the upgrade
229                 keyFor(connection));
230         assert !(connection instanceof SSLConnection);
231         Log.logTrace("Connection send window size {0} ", windowController.connectionWindowSize());
232 
233         Stream<?> initialStream = createStream(exchange);
234         initialStream.registerStream(1);
235         windowController.registerStream(1, getInitialSendWindowSize());
236         initialStream.requestSent();
237         sendConnectionPreface();
238         // start reading and writing
239         // start reading
240         AsyncConnection asyncConn = (AsyncConnection)connection;
241         asyncConn.setAsyncCallbacks(this::asyncReceive, this::shutdown, this::getReadBuffer);
242         connection.configureMode(Mode.ASYNC); // set mode only AFTER setAsyncCallbacks to provide visibility.
243         asyncReceive(ByteBufferReference.of(initial));
244         asyncConn.startReading();
245     }

275         asyncConn.enableCallback();
276         sendConnectionPreface();

279     /**
280      * Throws an IOException if h2 was not negotiated
281      */
282     private void checkSSLConfig() throws IOException {
283         AsyncSSLConnection aconn = (AsyncSSLConnection)connection;
284         SSLEngine engine = aconn.getEngine();
285         String alpn = engine.getApplicationProtocol();
286         if (alpn == null || !alpn.equals("h2")) {
287             String msg;
288             if (alpn == null) {
289                 Log.logSSL("ALPN not supported");
290                 msg = "ALPN not supported";
291             } else switch (alpn) {
292               case "":
293                 Log.logSSL("No ALPN returned");
294                 msg = "No ALPN negotiated";
295                 break;
296               case "http/1.1":
297                 Log.logSSL("HTTP/1.1 ALPN returned");
298                 msg = "HTTP/1.1 ALPN returned";
299                 break;
300               default:
301                 Log.logSSL("unknown ALPN returned");
302                 msg = "Unexpected ALPN: " + alpn;
303                 throw new IOException(msg);
304             }
305             throw new ALPNException(msg, aconn);
306         }
307     }

309     static String keyFor(HttpConnection connection) {
310         boolean isProxy = connection.isProxied();
311         boolean isSecure = connection.isSecure();
312         InetSocketAddress addr = connection.address();
313 
314         return keyString(isSecure, isProxy, addr.getHostString(), addr.getPort());
315     }

317     static String keyFor(URI uri, InetSocketAddress proxy) {
318         boolean isSecure = uri.getScheme().equalsIgnoreCase("https");
319         boolean isProxy = proxy != null;
320 
321         String host;
322         int port;
323 
324         if (isProxy) {
325             host = proxy.getHostString();
326             port = proxy.getPort();
327         } else {
328             host = uri.getHost();
329             port = uri.getPort();
330         }
331         return keyString(isSecure, isProxy, host, port);
332     }

334     // {C,S}:{H:P}:host:port
335     // C indicates clear text connection "http"
336     // S indicates secure "https"
337     // H indicates host (direct) connection
338     // P indicates proxy
339     // Eg: "S:H:foo.com:80"
340     static String keyString(boolean secure, boolean proxy, String host, int port) {
341         return (secure ? "S:" : "C:") + (proxy ? "P:" : "H:") + host + ":" + port;
342     }

352     private static String toHexdump1(ByteBuffer bb) {
353         bb.mark();
354         StringBuilder sb = new StringBuilder(512);
355         Formatter f = new Formatter(sb);
356 
357         while (bb.hasRemaining()) {
358             int i =  Byte.toUnsignedInt(bb.get());
359             f.format("%02x:", i);
360         }
361         sb.deleteCharAt(sb.length()-1);
362         bb.reset();
363         return sb.toString();
364     }

366     private static String toHexdump(ByteBuffer bb) {
367         List<String> words = new ArrayList<>();
368         int i = 0;
369         bb.mark();
370         while (bb.hasRemaining()) {
371             if (i % 2 == 0) {
372                 words.add("");
373             }
374             byte b = bb.get();
375             String hex = Integer.toHexString(256 + Byte.toUnsignedInt(b)).substring(1);
376             words.set(i / 2, words.get(i / 2) + hex);
377             i++;
378         }
379         bb.reset();
380         return words.stream().collect(Collectors.joining(" "));
381     }

383     private void decodeHeaders(HeaderFrame frame, DecodingCallback decoder) {
384         boolean endOfHeaders = frame.getFlag(HeaderFrame.END_HEADERS);
385 
386         ByteBufferReference[] buffers = frame.getHeaderBlock();
387         for (int i = 0; i < buffers.length; i++) {
388             hpackIn.decode(buffers[i].get(), endOfHeaders && (i == buffers.length - 1), decoder);
389         }
390     }

411     public void asyncReceive(ByteBufferReference buffer) {
412         // We don't need to read anything and
413         // we don't want to send anything back to the server
414         // until the connection preface has been sent.
415         // Therefore we're going to wait if needed before reading
416         // (and thus replying) to anything.
417         // Starting to reply to something (e.g send an ACK to a
418         // SettingsFrame sent by the server) before the connection
419         // preface is fully sent might result in the server
420         // sending a GOAWAY frame with 'invalid_preface'.
421         synchronized (readlock) {
422             try {
423                 // the readlock ensures that the order of incoming buffers
424                 // is preserved.
425                 framesController.processReceivedData(framesDecoder, buffer);
426             } catch (Throwable e) {
427                 String msg = Utils.stackTrace(e);
428                 Log.logTrace(msg);
429                 shutdown(e);
430             }
431         }
432     }

435     void shutdown(Throwable t) {
436         Log.logError(t);
437         closed = true;
438         client2.deleteConnection(this);
439         List<Stream<?>> c = new LinkedList<>(streams.values());
440         for (Stream<?> s : c) {
441             s.cancelImpl(t);
442         }
443         connection.close();
444     }

446     /**
447      * Handles stream 0 (common) frames that apply to whole connection and passes
448      * other stream specific frames to that Stream object.
449      *
450      * Invokes Stream.incoming() which is expected to process frame without
451      * blocking.
452      */
453     void processFrame(Http2Frame frame) throws IOException {
454         Log.logFrames(frame, "IN");
455         int streamid = frame.streamid();
456         if (frame instanceof MalformedFrame) {
457             Log.logError(((MalformedFrame) frame).getMessage());
458             if (streamid == 0) {
459                 protocolError(((MalformedFrame) frame).getErrorCode());
460             } else {
461                 resetStream(streamid, ((MalformedFrame) frame).getErrorCode());
462             }
463             return;
464         }
465         if (streamid == 0) {
466             handleConnectionFrame(frame);
467         } else {
468             if (frame instanceof SettingsFrame) {
469                 // The stream identifier for a SETTINGS frame MUST be zero
470                 protocolError(GoAwayFrame.PROTOCOL_ERROR);
471                 return;
472             }
473 
474             Stream<?> stream = getStream(streamid);
475             if (stream == null) {
476                 // Should never receive a frame with unknown stream id
477 
478                 // To avoid looping, an endpoint MUST NOT send a RST_STREAM in
479                 // response to a RST_STREAM frame.
480                 if (!(frame instanceof ResetFrame)) {
481                     resetStream(streamid, ResetFrame.PROTOCOL_ERROR);
482                 }
483                 return;
484             }
485             if (frame instanceof PushPromiseFrame) {
486                 PushPromiseFrame pp = (PushPromiseFrame)frame;
487                 handlePushPromise(stream, pp);
488             } else if (frame instanceof HeaderFrame) {
489                 // decode headers (or continuation)
490                 decodeHeaders((HeaderFrame) frame, stream.rspHeadersConsumer());
491                 stream.incoming(frame);
492             } else {
493                 stream.incoming(frame);
494             }
495         }
496     }
497 
498     private <T> void handlePushPromise(Stream<T> parent, PushPromiseFrame pp)
499         throws IOException
500     {
501         HttpRequestImpl parentReq = parent.request;
502         int promisedStreamid = pp.getPromisedStream();
503         if (promisedStreamid != nextPushStream) {
504             resetStream(promisedStreamid, ResetFrame.PROTOCOL_ERROR);
505             return;
506         } else {
507             nextPushStream += 2;
508         }
509         HeaderDecoder decoder = new HeaderDecoder();
510         decodeHeaders(pp, decoder);
511         HttpHeadersImpl headers = decoder.headers();
512         HttpRequestImpl pushReq = HttpRequestImpl.createPushRequest(parentReq, headers);
513         Exchange<T> pushExch = new Exchange<>(pushReq, parent.exchange.multi);
514         Stream.PushedStream<?,T> pushStream = createPushStream(parent, pushExch);
515         pushExch.exchImpl = pushStream;
516         pushStream.registerStream(promisedStreamid);
517         parent.incoming_pushPromise(pushReq, pushStream);
518     }

520     private void handleConnectionFrame(Http2Frame frame)
521         throws IOException
522     {
523         switch (frame.type()) {
524           case SettingsFrame.TYPE:
525               handleSettings((SettingsFrame)frame);
526               break;
527           case PingFrame.TYPE:
528               handlePing((PingFrame)frame);
529               break;
530           case GoAwayFrame.TYPE:
531               handleGoAway((GoAwayFrame)frame);
532               break;
533           case WindowUpdateFrame.TYPE:
534               handleWindowUpdate((WindowUpdateFrame)frame);
535               break;
536           default:
537             protocolError(ErrorFrame.PROTOCOL_ERROR);
538         }
539     }

541     void resetStream(int streamid, int code) throws IOException {
542         Log.logError(
543             "Resetting stream {0,number,integer} with error code {1,number,integer}",
544             streamid, code);
545         ResetFrame frame = new ResetFrame(streamid, code);
546         sendFrame(frame);
547         closeStream(streamid);
548     }

550     void closeStream(int streamid) {
551         Stream<?> s = streams.remove(streamid);
552         // ## Remove s != null. It is a hack for delayed cancellation,reset
553         if (s != null && !(s instanceof Stream.PushedStream)) {
554             // Since PushStreams have no request body, then they have no
555             // corresponding entry in the window controller.
556             windowController.removeStream(streamid);
557         }
558     }
559     /**
560      * Increments this connection's send Window by the amount in the given frame.
561      */
562     private void handleWindowUpdate(WindowUpdateFrame f)
563         throws IOException
564     {
565         int amount = f.getUpdate();
566         if (amount <= 0) {
567             // ## temporarily disable to workaround a bug in Jetty where it
568             // ## sends Window updates with a 0 update value.
569             //protocolError(ErrorFrame.PROTOCOL_ERROR);
570         } else {
571             boolean success = windowController.increaseConnectionWindow(amount);
572             if (!success) {
573                 protocolError(ErrorFrame.FLOW_CONTROL_ERROR);  // overflow
574             }
575         }
576     }

586     private void handleSettings(SettingsFrame frame)
587         throws IOException
588     {
589         assert frame.streamid() == 0;
590         if (!frame.getFlag(SettingsFrame.ACK)) {
591             int oldWindowSize = serverSettings.getParameter(INITIAL_WINDOW_SIZE);
592             int newWindowSize = frame.getParameter(INITIAL_WINDOW_SIZE);
593             int diff = newWindowSize - oldWindowSize;
594             if (diff != 0) {
595                 windowController.adjustActiveStreams(diff);
596             }
597             serverSettings = frame;
598             sendFrame(new SettingsFrame(SettingsFrame.ACK));
599         }
600     }

640     private static final String CLIENT_PREFACE = "PRI * HTTP/2.0\r\n\r\nSM\r\n\r\n";
641 
642     private static final byte[] PREFACE_BYTES =
643         CLIENT_PREFACE.getBytes(StandardCharsets.ISO_8859_1);
644 
645     /**
646      * Sends Connection preface and Settings frame with current preferred
647      * values
648      */
649     private void sendConnectionPreface() throws IOException {
650         Log.logTrace("{0}: start sending connection preface to {1}",
651                      connection.channel().getLocalAddress(),
652                      connection.address());
653         SettingsFrame sf = client2.getClientSettings();
654         ByteBufferReference ref = framesEncoder.encodeConnectionPreface(PREFACE_BYTES, sf);
655         Log.logFrames(sf, "OUT");
656         // send preface bytes and SettingsFrame together
657         connection.write(ref.get());
658         // mark preface sent.
659         framesController.markPrefaceSent();
660         Log.logTrace("PREFACE_BYTES sent");
661         Log.logTrace("Settings Frame sent");
662 
663         // send a Window update for the receive buffer we are using
664         // minus the initial 64 K specified in protocol
665         final int len = client2.client().getReceiveBufferSize() - (64 * 1024 - 1);
666         windowUpdater.sendWindowUpdate(len);
667         // there will be an ACK to the windows update - which should
668         // cause any pending data stored before the preface was sent to be
669         // flushed (see PrefaceController).
670         Log.logTrace("finished sending connection preface");
671     }

703     /**
704      * Encode the headers into a List<ByteBuffer> and then create HEADERS
705      * and CONTINUATION frames from the list and return the List<Http2Frame>.
706      */
707     private List<HeaderFrame> encodeHeaders(OutgoingHeaders<Stream<?>> frame) {
708         List<ByteBufferReference> buffers = encodeHeadersImpl(
709                 getMaxSendFrameSize(),
710                 frame.getAttachment().getRequestPseudoHeaders(),
711                 frame.getUserHeaders(),
712                 frame.getSystemHeaders());
713 
714         List<HeaderFrame> frames = new ArrayList<>(buffers.size());
715         Iterator<ByteBufferReference> bufIterator = buffers.iterator();
716         HeaderFrame oframe = new HeadersFrame(frame.streamid(), frame.getFlags(), bufIterator.next());
717         frames.add(oframe);
718         while(bufIterator.hasNext()) {
719             oframe = new ContinuationFrame(frame.streamid(), bufIterator.next());
720             frames.add(oframe);
721         }
722         oframe.setFlag(HeaderFrame.END_HEADERS);
723         return frames;
724     }

726     // Dedicated cache for headers encoding ByteBuffer.
727     // There can be no concurrent access to this  buffer as all access to this buffer
728     // and its content happen within a single critical code block section protected
729     // by the sendLock. / (see sendFrame())
730     private ByteBufferPool headerEncodingPool = new ByteBufferPool();
731 
732     private ByteBufferReference getHeaderBuffer(int maxFrameSize) {
733         ByteBufferReference ref = headerEncodingPool.get(maxFrameSize);
734         ref.get().limit(maxFrameSize);
735         return ref;
736     }

738     /*
739      * Encodes all the headers from the given HttpHeaders into the given List
740      * of buffers.
741      *
742      * From https://tools.ietf.org/html/rfc7540#section-8.1.2 :
743      *
744      *     ...Just as in HTTP/1.x, header field names are strings of ASCII
745      *     characters that are compared in a case-insensitive fashion.  However,
746      *     header field names MUST be converted to lowercase prior to their
747      *     encoding in HTTP/2...
748      */
749     private List<ByteBufferReference> encodeHeadersImpl(int maxFrameSize, HttpHeaders... headers) {
750         ByteBufferReference buffer = getHeaderBuffer(maxFrameSize);
751         List<ByteBufferReference> buffers = new ArrayList<>();
752         for(HttpHeaders header : headers) {
753             for (Map.Entry<String, List<String>> e : header.map().entrySet()) {
754                 String lKey = e.getKey().toLowerCase();
755                 List<String> values = e.getValue();
756                 for (String value : values) {
757                     hpackOut.header(lKey, value);
758                     while (!hpackOut.encode(buffer.get())) {
759                         buffer.get().flip();
760                         buffers.add(buffer);
761                         buffer =  getHeaderBuffer(maxFrameSize);
762                     }
763                 }
764             }
765         }
766         buffer.get().flip();
767         buffers.add(buffer);
768         return buffers;
769     }

771     private ByteBufferReference[] encodeHeaders(OutgoingHeaders<Stream<?>> oh, Stream<?> stream) {
772         oh.streamid(stream.streamid);
773         if (Log.headers()) {
774             StringBuilder sb = new StringBuilder("HEADERS FRAME (stream=");
775             sb.append(stream.streamid).append(")\n");
776             Log.dumpHeaders(sb, "    ", oh.getAttachment().getRequestPseudoHeaders());
777             Log.dumpHeaders(sb, "    ", oh.getSystemHeaders());
778             Log.dumpHeaders(sb, "    ", oh.getUserHeaders());
779             Log.logHeaders(sb.toString());
780         }
781         List<HeaderFrame> frames = encodeHeaders(oh);
782         return encodeFrames(frames);
783     }

786         if (Log.frames()) {

788         }
789         return framesEncoder.encodeFrames(frames);

805     private Stream<?> registerNewStream(OutgoingHeaders<Stream<?>> oh) {
806         Stream<?> stream = oh.getAttachment();
807         int streamid = nextstreamid;
808         nextstreamid += 2;
809         stream.registerStream(streamid);
810         // set outgoing window here. This allows thread sending
811         // body to proceed.
812         windowController.registerStream(streamid, getInitialSendWindowSize());
813         return stream;
814     }

816     private final Object sendlock = new Object();
817 
818     void sendFrame(Http2Frame frame) {
819         try {
820             synchronized (sendlock) {
821                 if (frame instanceof OutgoingHeaders) {
822                     @SuppressWarnings("unchecked")
823                     OutgoingHeaders<Stream<?>> oh = (OutgoingHeaders<Stream<?>>) frame;
824                     Stream<?> stream = registerNewStream(oh);
825                     // provide protection from inserting unordered frames between Headers and Continuation
826                     connection.writeAsync(encodeHeaders(oh, stream));
827                 } else {
828                     connection.writeAsync(encodeFrame(frame));
829                 }
830             }
831             connection.flushAsync();
832         } catch (IOException e) {
833             if (!closed) {
834                 Log.logError(e);
835                 shutdown(e);
836             }
837         }
838     }

840     private ByteBufferReference[] encodeFrame(Http2Frame frame) {
841         Log.logFrames(frame, "OUT");
842         return framesEncoder.encodeFrame(frame);

846         try {

858      * Direct call of the method bypasses synchronization on "sendlock" and
859      * allowed only of control frames: WindowUpdateFrame, PingFrame and etc.
860      * prohibited for such frames as DataFrame, HeadersFrame, ContinuationFrame.
861      */
862     void sendUnorderedFrame(Http2Frame frame) {
863         try {
864             connection.writeAsyncUnordered(encodeFrame(frame));
865             connection.flushAsync();
866         } catch (IOException e) {
867             if (!closed) {
868                 Log.logError(e);
869                 shutdown(e);
870             }
871         }

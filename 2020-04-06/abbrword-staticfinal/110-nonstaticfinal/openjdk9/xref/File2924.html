64     private static void deleteFile (String BASE_DIR, int pIndex, String pArchiveName) {
65           java.io.File file = new java.io.File (BASE_DIR, pIndex + pArchiveName );
66           file.delete ();
67     }
68  
69     private static void copyFile( String pBaseDir, String pArchiveName, int pIndex) {
70        try {
71           java.io.File lSource = new java.io.File( pBaseDir, pArchiveName );
72           java.io.File lDestination = new java.io.File( pBaseDir, pIndex + pArchiveName );
73           if( !lDestination.exists() ) {
74              lDestination.createNewFile();
75              java.io.InputStream lInput = new java.io.FileInputStream( lSource );
76              java.io.OutputStream lOutput = new java.io.FileOutputStream( lDestination );
77              byte[] lBuffer = new byte[ 1024 ];
78              int lLength = -1;
79              while( ( lLength = lInput.read( lBuffer ) ) > 0 ) {
80                 lOutput.write( lBuffer, 0, lLength );
81              }
82              lInput.close();
83              lOutput.close();
84           }
85        } catch( Exception e ) {
86           e.printStackTrace();
87        }
88     }

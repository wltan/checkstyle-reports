1   package org.hibernate.test.criteria;
2   import org.hibernate.criterion.Criterion;
3   import org.hibernate.criterion.LikeExpression;
4   
5   public class StringExpression extends LikeExpression {
6       private final static Character ESCAPE_CODE = new Character( '\\' );
7   
8       protected StringExpression( String property, String value,
9               boolean ignoreCase ) {
10          super( property, value, ESCAPE_CODE, ignoreCase );
11      }
12  
13      public static Criterion stringExpression( String propertyName,
14              String value, boolean ignoreCase ) {

16      }
17  }

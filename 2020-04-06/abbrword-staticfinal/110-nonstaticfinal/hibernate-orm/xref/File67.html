34  /**
35   * An SQL dialect for Ingres 9.2.
36   * <p/>
37   * Known limitations:
38   * <ul>
39   * <li> Only supports simple constants or columns on the left side of an IN, making (1,2,3) in (...) or (&lt;subselect&gt;) in (...) non-supported.
40   * <li> Supports only 39 digits in decimal.
41   * <li> Explicitly set USE_GET_GENERATED_KEYS property to false.
42   * <li> Perform string casts to varchar; removes space padding.
43   * </ul>
44   * 
45   * @author Ian Booth
46   * @author Bruce Lunsford
47   * @author Max Rydahl Andersen
48   * @author Raymond Fan
49   */
50  public class IngresDialect extends Dialect {

68  		registerColumnType( Types.LONGVARBINARY, "long byte" );
69  		registerColumnType( Types.CHAR, 32000, "char($l)" );
70  		registerColumnType( Types.VARCHAR, 32000, "varchar($l)" );
71  		registerColumnType( Types.VARCHAR, "long varchar" );
72  		registerColumnType( Types.LONGVARCHAR, "long varchar" );
73  		registerColumnType( Types.DATE, "date" );
74  		registerColumnType( Types.TIME, "time with time zone" );

76  		registerColumnType( Types.BLOB, "blob" );

147 		registerFunction( "str", new SQLFunctionTemplate(StandardBasicTypes.STRING, "cast(?1 as varchar)") );
148         // Ingres driver supports getGeneratedKeys but only in the following
149         // form:
150         // The Ingres DBMS returns only a single table key or a single object
151         // key per insert statement. Ingres does not return table and object
152         // keys for INSERT AS SELECT statements. Depending on the keys that are
153         // produced by the statement executed, auto-generated key parameters in
154         // execute(), executeUpdate(), and prepareStatement() methods are
155         // ignored and getGeneratedKeys() returns a result-set containing no
156         // rows, a single row with one column, or a single row with two columns.
157         // Ingres JDBC Driver returns table and object keys as BINARY values.
158         getDefaultProperties().setProperty(Environment.USE_GET_GENERATED_KEYS, "false");

161         // substitions of true and false.
162         getDefaultProperties().setProperty(Environment.QUERY_SUBSTITUTIONS, "true=1,false=0");

282 	/**
283 	 * Add a <tt>LIMIT</tt> clause to the given SQL <tt>SELECT</tt>
284 	 *
285 	 * @return the modified SQL
286 	 */
287 	public String getLimitString(String querySelect, int offset, int limit) {
288 		if ( offset > 0 ) {
289 			throw new UnsupportedOperationException( "query result offset is not supported" );
290 		}
291 		return new StringBuilder( querySelect.length() + 16 )
292 				.append( querySelect )
293 				.insert( 6, " first " + limit )
294 				.toString();
295 	}

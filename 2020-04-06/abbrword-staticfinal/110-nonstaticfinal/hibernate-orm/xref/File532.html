24  package org.hibernate.ejb;
25  import java.util.HashSet;
26  import java.util.Set;
27  
28  /**
29   * TODO : javadoc
30   *
31   * @author Steve Ebersole
32   */
33  public class QueryHints {
34  	/**
35  	 * @deprecated HINT_TIMEOUT (org.hibernate.timeout),
36  	 * instead use SPEC_HINT_TIMEOUT (javax.persistence.query.timeout)
37  	 */
38  	public static final String HINT_TIMEOUT = "org.hibernate.timeout"; // Query timeout in seconds
39  	public static final String SPEC_HINT_TIMEOUT = "javax.persistence.query.timeout"; // timeout in milliseconds
40  	public static final String HINT_COMMENT = "org.hibernate.comment";
41  	public static final String HINT_FETCH_SIZE = "org.hibernate.fetchSize";
42  	public static final String HINT_CACHE_REGION = "org.hibernate.cacheRegion";
43  	public static final String HINT_CACHEABLE = "org.hibernate.cacheable";
44  	public static final String HINT_READONLY = "org.hibernate.readOnly";
45  	public static final String HINT_CACHE_MODE = "org.hibernate.cacheMode";
46  	public static final String HINT_FLUSH_MODE = "org.hibernate.flushMode";
47  
48  	private static final Set<String> HINTS = buildHintsSet();
49  
50  	private static Set<String> buildHintsSet() {
51  		HashSet<String> hints = new HashSet<String>();
52  		hints.add( HINT_TIMEOUT );
53  		hints.add( SPEC_HINT_TIMEOUT );
54  		hints.add( HINT_COMMENT );
55  		hints.add( HINT_FETCH_SIZE );
56  		hints.add( HINT_CACHE_REGION );
57  		hints.add( HINT_CACHEABLE );
58  		hints.add( HINT_READONLY );
59  		hints.add( HINT_CACHE_MODE );
60  		hints.add( HINT_FLUSH_MODE );
61  		return java.util.Collections.unmodifiableSet( hints );
62  	}
63  
64  	public static Set<String> getDefinedHints() {
65  		return HINTS;
66  	}
67  }

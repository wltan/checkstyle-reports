24  package org.hibernate.dialect;
25  import java.sql.CallableStatement;
26  import java.sql.ResultSet;
27  import java.sql.SQLException;
28  import java.sql.Types;
29  
30  import org.hibernate.dialect.function.NoArgSQLFunction;
31  import org.hibernate.dialect.function.NvlFunction;
32  import org.hibernate.dialect.function.StandardSQLFunction;
33  import org.hibernate.type.StandardBasicTypes;
34  
35  /**
36   * An SQL dialect for Postgres Plus
37   *
38   * @author Jim Mlodgenski
39   */
40  public class PostgresPlusDialect extends PostgreSQLDialect {
41  
42  	public PostgresPlusDialect() {
43  		super();
44  
45  		registerFunction( "ltrim", new StandardSQLFunction( "ltrim" ) );
46  		registerFunction( "rtrim", new StandardSQLFunction( "rtrim" ) );
47  		registerFunction( "soundex", new StandardSQLFunction( "soundex" ) );
48  		registerFunction( "sysdate", new NoArgSQLFunction( "sysdate", StandardBasicTypes.DATE, false ) );
49  		registerFunction( "rowid", new NoArgSQLFunction( "rowid", StandardBasicTypes.LONG, false ) );
50  		registerFunction( "rownum", new NoArgSQLFunction( "rownum", StandardBasicTypes.LONG, false ) );
51  		registerFunction( "instr", new StandardSQLFunction( "instr", StandardBasicTypes.INTEGER ) );
52  		registerFunction( "lpad", new StandardSQLFunction( "lpad", StandardBasicTypes.STRING ) );
53  		registerFunction( "replace", new StandardSQLFunction( "replace", StandardBasicTypes.STRING ) );
54  		registerFunction( "rpad", new StandardSQLFunction( "rpad", StandardBasicTypes.STRING ) );
55  		registerFunction( "translate", new StandardSQLFunction( "translate", StandardBasicTypes.STRING ) );
56  		registerFunction( "substring", new StandardSQLFunction( "substr", StandardBasicTypes.STRING ) );
57  		registerFunction( "coalesce", new NvlFunction() );
58  		registerFunction( "atan2", new StandardSQLFunction( "atan2", StandardBasicTypes.FLOAT ) );
59  		registerFunction( "mod", new StandardSQLFunction( "mod", StandardBasicTypes.INTEGER ) );
60  		registerFunction( "nvl", new StandardSQLFunction( "nvl" ) );
61  		registerFunction( "nvl2", new StandardSQLFunction( "nvl2" ) );
62  		registerFunction( "power", new StandardSQLFunction( "power", StandardBasicTypes.FLOAT ) );
63  		registerFunction( "add_months", new StandardSQLFunction( "add_months", StandardBasicTypes.DATE ) );
64  		registerFunction( "months_between", new StandardSQLFunction( "months_between", StandardBasicTypes.FLOAT ) );
65  		registerFunction( "next_day", new StandardSQLFunction( "next_day", StandardBasicTypes.DATE ) );
66  	}
67  
68  	public String getCurrentTimestampSelectString() {
69  		return "select sysdate";
70  	}
71  
72  	public String getCurrentTimestampSQLFunctionName() {
73  		return "sysdate";
74  	}
75  
76  	public int registerResultSetOutParameter(CallableStatement statement, int col) throws SQLException {
77  		statement.registerOutParameter( col, Types.REF );
78  		col++;
79  		return col;
80  	}
81  
82  	public ResultSet getResultSet(CallableStatement ps) throws SQLException {
83  		ps.execute();
84  		return ( ResultSet ) ps.getObject( 1 );
85  	}
86  
87  	public String getSelectGUIDString() {
88  		return "select uuid_generate_v1";
89  	}
90  
91  }

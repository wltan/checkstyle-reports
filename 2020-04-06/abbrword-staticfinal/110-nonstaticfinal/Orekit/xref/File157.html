58      @Test
59      public void testShift() throws OrekitException {
60          double rate = 2 * FastMath.PI / (12 * 60);
61          TimeStampedFieldAngularCoordinates<DerivativeStructure> angularCoordinates =
62                  new TimeStampedFieldAngularCoordinates<DerivativeStructure>(AbsoluteDate.J2000_EPOCH,
63                                                                              createRotation(1, 0, 0, 0, false),
64                                                                              new FieldVector3D<DerivativeStructure>(rate, createVector(0, 0, 1, 4)),
65                                                                              createVector(0, 0, 0, 4));
66          Assert.assertEquals(rate, angularCoordinates.getRotationRate().getNorm().getReal(), 1.0e-10);
67          double dt = 10.0;
68          double alpha = rate * dt;
69          TimeStampedFieldAngularCoordinates<DerivativeStructure> shifted = angularCoordinates.shiftedBy(dt);
70          Assert.assertEquals(rate, shifted.getRotationRate().getNorm().getReal(), 1.0e-10);
71          Assert.assertEquals(alpha, FieldRotation.distance(angularCoordinates.getRotation(), shifted.getRotation()).getReal(), 1.0e-10);
72  
73          FieldVector3D<DerivativeStructure> xSat = shifted.getRotation().applyInverseTo(createVector(1, 0, 0, 4));
74          Assert.assertEquals(0.0, xSat.subtract(createVector(FastMath.cos(alpha), FastMath.sin(alpha), 0, 4)).getNorm().getReal(), 1.0e-10);
75          FieldVector3D<DerivativeStructure> ySat = shifted.getRotation().applyInverseTo(createVector(0, 1, 0, 4));
76          Assert.assertEquals(0.0, ySat.subtract(createVector(-FastMath.sin(alpha), FastMath.cos(alpha), 0, 4)).getNorm().getReal(), 1.0e-10);
77          FieldVector3D<DerivativeStructure> zSat = shifted.getRotation().applyInverseTo(createVector(0, 0, 1, 4));
78          Assert.assertEquals(0.0, zSat.subtract(createVector(0, 0, 1, 4)).getNorm().getReal(), 1.0e-10);
79  
80      }

98      @Test
99      public void testSpin() throws OrekitException {
100         double rate = 2 * FastMath.PI / (12 * 60);
101         TimeStampedFieldAngularCoordinates<DerivativeStructure> angularCoordinates =
102                 new TimeStampedFieldAngularCoordinates<DerivativeStructure>(AbsoluteDate.J2000_EPOCH,
103                                                                             createRotation(0.48, 0.64, 0.36, 0.48, false),
104                                                                             new FieldVector3D<DerivativeStructure>(rate, createVector(0, 0, 1, 4)),
105 									    createVector(0, 0, 0, 4));
106         Assert.assertEquals(rate, angularCoordinates.getRotationRate().getNorm().getReal(), 1.0e-10);
107         double dt = 10.0;
108         TimeStampedFieldAngularCoordinates<DerivativeStructure> shifted = angularCoordinates.shiftedBy(dt);
109         Assert.assertEquals(rate, shifted.getRotationRate().getNorm().getReal(), 1.0e-10);
110         Assert.assertEquals(rate * dt, FieldRotation.distance(angularCoordinates.getRotation(), shifted.getRotation()).getReal(), 1.0e-10);
111 
112         FieldVector3D<DerivativeStructure> shiftedX  = shifted.getRotation().applyInverseTo(createVector(1, 0, 0, 4));
113         FieldVector3D<DerivativeStructure> shiftedY  = shifted.getRotation().applyInverseTo(createVector(0, 1, 0, 4));
114         FieldVector3D<DerivativeStructure> shiftedZ  = shifted.getRotation().applyInverseTo(createVector(0, 0, 1, 4));
115         FieldVector3D<DerivativeStructure> originalX = angularCoordinates.getRotation().applyInverseTo(createVector(1, 0, 0, 4));
116         FieldVector3D<DerivativeStructure> originalY = angularCoordinates.getRotation().applyInverseTo(createVector(0, 1, 0, 4));
117         FieldVector3D<DerivativeStructure> originalZ = angularCoordinates.getRotation().applyInverseTo(createVector(0, 0, 1, 4));
118         Assert.assertEquals( FastMath.cos(rate * dt), FieldVector3D.dotProduct(shiftedX, originalX).getReal(), 1.0e-10);
119         Assert.assertEquals( FastMath.sin(rate * dt), FieldVector3D.dotProduct(shiftedX, originalY).getReal(), 1.0e-10);
120         Assert.assertEquals( 0.0,                 FieldVector3D.dotProduct(shiftedX, originalZ).getReal(), 1.0e-10);
121         Assert.assertEquals(-FastMath.sin(rate * dt), FieldVector3D.dotProduct(shiftedY, originalX).getReal(), 1.0e-10);
122         Assert.assertEquals( FastMath.cos(rate * dt), FieldVector3D.dotProduct(shiftedY, originalY).getReal(), 1.0e-10);
123         Assert.assertEquals( 0.0,                 FieldVector3D.dotProduct(shiftedY, originalZ).getReal(), 1.0e-10);
124         Assert.assertEquals( 0.0,                 FieldVector3D.dotProduct(shiftedZ, originalX).getReal(), 1.0e-10);
125         Assert.assertEquals( 0.0,                 FieldVector3D.dotProduct(shiftedZ, originalY).getReal(), 1.0e-10);
126         Assert.assertEquals( 1.0,                 FieldVector3D.dotProduct(shiftedZ, originalZ).getReal(), 1.0e-10);
127 
128         FieldVector3D<DerivativeStructure> forward = FieldAngularCoordinates.estimateRate(angularCoordinates.getRotation(), shifted.getRotation(), dt);
129         Assert.assertEquals(0.0, forward.subtract(angularCoordinates.getRotationRate()).getNorm().getReal(), 1.0e-10);
130 
131         FieldVector3D<DerivativeStructure> reversed = FieldAngularCoordinates.estimateRate(shifted.getRotation(), angularCoordinates.getRotation(), dt);
132         Assert.assertEquals(0.0, reversed.add(angularCoordinates.getRotationRate()).getNorm().getReal(), 1.0e-10);
133

204         throws NoSuchMethodException, IllegalAccessException, InvocationTargetException {
205 
206         // use reflection to test the private static methods
207         Method getter  = TimeStampedFieldAngularCoordinates.class.getDeclaredMethod("getModifiedRodrigues",
208                                                                                     new Class<?>[] {
209                                                                                         TimeStampedFieldAngularCoordinates.class,
210                                                                                         double[].class, double.class

214                                                                                     new Class<?>[] {
215                                                                                         RealFieldElement[][].class,
216                                                                                         TimeStampedFieldAngularCoordinates.class
217         });
218         factory.setAccessible(true);
219 
220         // check the two-way conversion result in identity
221         Random random = new Random(0xb1e615aaa8236b52l);
222         double[] previous = new double[] { 1.0, 0.0, 0.0, 0.0 };

233 		new TimeStampedFieldAngularCoordinates<DerivativeStructure>(AbsoluteDate.J2000_EPOCH, rotation, rotationRate, rotationAcceleration);
234             double dt                  = 10.0 * random.nextDouble();
235             DerivativeStructure[][] rodrigues =

240             (TimeStampedFieldAngularCoordinates<DerivativeStructure>) factory.invoke(null, rodrigues, offset.shiftedBy(dt));

242             Assert.assertEquals(0.0, FieldVector3D.distance(rotationRate, rebuilt.getRotationRate()).getReal(), 1.0e-15);
243             Assert.assertEquals(0.0, FieldVector3D.distance(rotationAcceleration, rebuilt.getRotationAcceleration()).getReal(), 1.0e-15);
244         }

472     private FieldVector3D<DerivativeStructure> randomVector(Random random, double norm) {
473         double n = random.nextDouble() * norm;
474         double x = random.nextDouble();
475         double y = random.nextDouble();
476         double z = random.nextDouble();

480     private FieldRotation<DerivativeStructure> randomRotation(Random random) {
481         double q0 = random.nextDouble() * 2 - 1;
482         double q1 = random.nextDouble() * 2 - 1;
483         double q2 = random.nextDouble() * 2 - 1;
484         double q3 = random.nextDouble() * 2 - 1;
485         double q  = FastMath.sqrt(q0 * q0 + q1 * q1 + q2 * q2 + q3 * q3);
486         return createRotation(q0 / q, q1 / q, q2 / q, q3 / q, false);
487     }

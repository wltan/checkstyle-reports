127         /**
128          * Method handleAttributesSubtree
129          * @inheritDoc
130          * @param E
131          * @throws CanonicalizationException
132          */
133         Iterator handleAttributesSubtree(Element E,NameSpaceSymbTable ns)
134                         throws CanonicalizationException {
135                 // System.out.println("During the traversal, I encountered " +
136                 // XMLUtils.getXPath(E));
137                 // result will contain the attrs which have to be outputted
138                 SortedSet result = this.result;
139             result.clear();
140                 NamedNodeMap attrs=null;
141 
142                 int attrsLength = 0;
143         if (E.hasAttributes()) {
144             attrs = E.getAttributes();
145                 attrsLength = attrs.getLength();
146         }
147                 //The prefix visibly utilized(in the attribute or in the name) in the element
148                 SortedSet visiblyUtilized =(SortedSet) _inclusiveNSSet.clone();
149 
150                 for (int i = 0; i < attrsLength; i++) {
151                         Attr N = (Attr) attrs.item(i);
152 
153                         if (XMLNS_URI!=N.getNamespaceURI()) {
154                                 //Not a namespace definition.
155                                 //The Element is output element, add his prefix(if used) to visibyUtilized
156                                 String prefix = N.getPrefix();
157                                 if ( (prefix != null) && (!prefix.equals(XML) && !prefix.equals(XMLNS)) ) {
158                                                 visiblyUtilized.add(prefix);
159                                 }
160                                 //Add to the result.
161                                  result.add(N);
162                                 continue;
163                         }
164                         String NName=N.getLocalName();
165                         String NNodeValue=N.getNodeValue();
166 
167                         if (ns.addMapping(NName, NNodeValue,N)) {
168                                 //New definition check if it is relative.
169                 if (C14nHelper.namespaceIsRelative(NNodeValue)) {
170                     Object exArgs[] = {E.getTagName(), NName,
171                             N.getNodeValue()};
172                     throw new CanonicalizationException(
173                             "c14n.Canonicalizer.RelativeNamespace", exArgs);
174                 }
175             }
176                 }
177                 String prefix;
178                 if (E.getNamespaceURI() != null) {
179                         prefix = E.getPrefix();
180                         if ((prefix == null) || (prefix.length() == 0)) {
181                                 prefix=XMLNS;
182                         }
183 
184                 } else {
185                         prefix=XMLNS;
186                 }
187                 visiblyUtilized.add(prefix);
188 
189                 //This can be optimezed by I don't have time
190                 Iterator it=visiblyUtilized.iterator();
191                 while (it.hasNext()) {
192                         String s=(String)it.next();
193                         Attr key=ns.getMapping(s);
194                         if (key==null) {
195                                 continue;
196                         }
197                         result.add(key);
198                 }
199 
200                 return result.iterator();
201         }

220         /**
221      * @inheritDoc
222          * @param E
223          * @throws CanonicalizationException
224          */
225         final Iterator handleAttributes(Element E, NameSpaceSymbTable ns)
226                         throws CanonicalizationException {
227                 // result will contain the attrs which have to be outputted
228                 SortedSet result = this.result;
229             result.clear();
230                 NamedNodeMap attrs = null;
231                 int attrsLength = 0;
232         if (E.hasAttributes()) {
233             attrs = E.getAttributes();
234                 attrsLength = attrs.getLength();
235         }
236                 //The prefix visibly utilized(in the attribute or in the name) in the element
237                 Set visiblyUtilized =null;
238                 //It's the output selected.
239                 boolean isOutputElement=isVisibleDO(E,ns.getLevel())==1;
240                 if (isOutputElement) {
241                         visiblyUtilized =  (Set) this._inclusiveNSSet.clone();
242                 }
243 
244                 for (int i = 0; i < attrsLength; i++) {
245                         Attr N = (Attr) attrs.item(i);
246 
247 
248                         if (XMLNS_URI!=N.getNamespaceURI()) {
249                                 if ( !isVisible(N) )  {
250                                         //The node is not in the nodeset(if there is a nodeset)
251                                         continue;
252                                 }
253                                 //Not a namespace definition.
254                                 if (isOutputElement) {
255                                         //The Element is output element, add his prefix(if used) to visibyUtilized
256                                         String prefix = N.getPrefix();
257                                         if ((prefix != null) && (!prefix.equals(XML) && !prefix.equals(XMLNS)) ){
258                                                         visiblyUtilized.add(prefix);
259                                         }
260                                         //Add to the result.
261                                     result.add(N);
262                                 }
263                                 continue;
264                         }
265                         String NName=N.getLocalName();
266                         if (isOutputElement && !isVisible(N) && NName!=XMLNS) {
267                         ns.removeMappingIfNotRender(NName);
268                         continue;
269                 }
270                         String NNodeValue=N.getNodeValue();
271 
272                         if (!isOutputElement && isVisible(N) && _inclusiveNSSet.contains(NName) && !ns.removeMappingIfRender(NName)) {
273                                 Node n=ns.addMappingAndRender(NName,NNodeValue,N);
274                                 if (n!=null) {
275                                                 result.add(n);
276                             if (C14nHelper.namespaceIsRelative(N)) {
277                                Object exArgs[] = { E.getTagName(), NName, N.getNodeValue() };
278                                throw new CanonicalizationException(
279                                   "c14n.Canonicalizer.RelativeNamespace", exArgs);
280                            }
281                                  }
282                         }
283 
284 
285 
286                         if (ns.addMapping(NName, NNodeValue,N)) {
287                 //New definiton check if it is relative
288                 if (C14nHelper.namespaceIsRelative(NNodeValue)) {
289                     Object exArgs[] = {E.getTagName(), NName,
290                             N.getNodeValue()};
291                     throw new CanonicalizationException(
292                             "c14n.Canonicalizer.RelativeNamespace", exArgs);
293                 }
294             }
295                 }
296 
297                 if (isOutputElement) {
298            //The element is visible, handle the xmlns definition
299            Attr xmlns = E.getAttributeNodeNS(XMLNS_URI, XMLNS);
300            if ((xmlns!=null) &&  (!isVisible(xmlns))) {
301               //There is a definition but the xmlns is not selected by the xpath.
302               //then xmlns=""
303               ns.addMapping(XMLNS,"",nullNode);
304             }
305 
306                         if (E.getNamespaceURI() != null) {
307                                 String prefix = E.getPrefix();
308                                 if ((prefix == null) || (prefix.length() == 0)) {
309                                         visiblyUtilized.add(XMLNS);
310                                 } else {
311                                         visiblyUtilized.add( prefix);
312                                 }
313                         } else {
314                                 visiblyUtilized.add(XMLNS);
315                         }
316                         //This can be optimezed by I don't have time
317                         //visiblyUtilized.addAll(this._inclusiveNSSet);
318                         Iterator it=visiblyUtilized.iterator();
319                         while (it.hasNext()) {
320                                 String s=(String)it.next();
321                                 Attr key=ns.getMapping(s);
322                                 if (key==null) {
323                                         continue;
324                                 }
325                                 result.add(key);
326                         }
327                 }
328 
329                 return result.iterator();
330         }
331         void circumventBugIfNeeded(XMLSignatureInput input) throws CanonicalizationException, ParserConfigurationException, IOException, SAXException {
332                 if (!input.isNeedsToBeExpanded() || _inclusiveNSSet.isEmpty())
333                         return;
334                 Document doc = null;
335                if (input.getSubNode() != null) {
336                    doc=XMLUtils.getOwnerDocument(input.getSubNode());
337                } else {
338                    doc=XMLUtils.getOwnerDocument(input.getNodeSet());
339                }
340 
341                 XMLUtils.circumventBug2650(doc);
342            }

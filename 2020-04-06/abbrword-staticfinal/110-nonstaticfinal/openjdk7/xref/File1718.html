162     private static void initIIOReadFormats() {
163         List spis = new ArrayList();
164         List shortNames = new ArrayList();
165 
166         ImageIO.scanForPlugins();
167         IIORegistry registry = IIORegistry.getDefaultInstance();
168         java.util.Iterator readerspis =
169             registry.getServiceProviders(ImageReaderSpi.class, false);
170         while (readerspis.hasNext()) {
171             // REMIND: there could be more than one non-core plugin for
172             // a particular format, as is the case for JPEG2000 in the JAI
173             // IIO Tools package, so we should support that somehow
174             ImageReaderSpi spi = (ImageReaderSpi)readerspis.next();
175             String klass = spi.getClass().getName();
176             String format = spi.getFormatNames()[0].toLowerCase();
177             String suffix = spi.getFileSuffixes()[0].toLowerCase();
178             if (suffix == null || suffix.equals("")) {
179                 suffix = format;
180             }
181             String shortName;
182             if (klass.startsWith("com.sun.imageio.plugins")) {
183                 shortName = "core-" + suffix;
184             } else {
185                 shortName = "ext-" + suffix;
186             }
187             spis.add(spi);
188             shortNames.add(shortName);
189         }
190 
191         imageioReaderSpis = new ImageReaderSpi[spis.size()];
192         imageioReaderSpis = (ImageReaderSpi[])spis.toArray(imageioReaderSpis);
193         imageioReadFormatShortNames = new String[shortNames.size()];
194         imageioReadFormatShortNames =
195             (String[])shortNames.toArray(imageioReadFormatShortNames);
196     }

209     private static class Context extends InputTests.Context {
210         String format;
211         BufferedImage image;
212         ImageReader reader;
213         boolean seekForwardOnly;
214         boolean ignoreMetadata;
215 
216         Context(TestEnvironment env, Result result, int testType) {
217             super(env, result);
218 
219             String content = (String)env.getModifier(contentList);
220             if (content == null) {
221                 content = CONTENT_BLANK;
222             }
223             // REMIND: add option for non-opaque images
224             image = createBufferedImage(size, size, content, false);
225 
226             result.setUnits(size*size);
227             result.setUnitName("pixel");
228 
229             if (testType == TEST_IMAGEIO || testType == TEST_IMAGEREADER) {
230                 ImageReaderSpi readerspi =
231                     (ImageReaderSpi)env.getModifier(imageioReadFormatList);
232                 format = readerspi.getFileSuffixes()[0].toLowerCase();
233                 if (testType == TEST_IMAGEREADER) {
234                     seekForwardOnly = env.isEnabled(seekForwardOnlyTog);
235                     ignoreMetadata = env.isEnabled(ignoreMetadataTog);
236                     try {
237                         reader = readerspi.createReaderInstance();
238                     } catch (IOException e) {
239                         System.err.println("error creating reader");
240                         e.printStackTrace();
241                     }
242                     if (env.isEnabled(installListenerTog)) {
243                         reader.addIIOReadProgressListener(
244                             new ReadProgressListener());
245                     }
246                 }
247                 if (format.equals("wbmp")) {
248                     // REMIND: this is a hack to create an image that the
249                     //         WBMPImageWriter can handle (a better approach
250                     //         would involve checking the ImageTypeSpecifier
251                     //         of the writer's default image param)
252                     BufferedImage newimg =
253                         new BufferedImage(size, size,
254                                           BufferedImage.TYPE_BYTE_BINARY);
255                     Graphics g = newimg.createGraphics();
256                     g.drawImage(image, 0, 0, null);
257                     g.dispose();
258                     image = newimg;
259                 }
260             } else if (testType == TEST_TOOLKIT) {
261                 format = (String)env.getModifier(toolkitReadFormatList);
262             } else { // testType == TEST_JPEGCODEC
263                 format = "jpeg";
264             }
265 
266             initInput();
267         }
268 
269         void initContents(File f) throws IOException {
270             ImageIO.write(image, format, f);
271         }
272 
273         void initContents(OutputStream out) throws IOException {
274             ImageIO.write(image, format, out);
275         }
276 
277         void cleanup(TestEnvironment env) {
278             super.cleanup(env);
279             if (reader != null) {
280                 reader.dispose();
281                 reader = null;
282             }
283         }

286     private static class ToolkitCreateImage extends InputImageTests {
287         private static final Component canvas = new Component() {};
288 
289         public ToolkitCreateImage() {
290             super(toolkitTestRoot,
291                   "createImage",
292                   "Toolkit.createImage()");
293             addDependency(generalSourceRoot,
294                 new Modifier.Filter() {
295                     public boolean isCompatible(Object val) {
296                         // Toolkit handles FILE, URL, and ARRAY, but
297                         // not FILECHANNEL
298                         InputType t = (InputType)val;
299                         return (t.getType() != INPUT_FILECHANNEL);
300                     }
301                 });
302             addDependencies(toolkitOptRoot, true);
303         }
304 
305         public Object initTest(TestEnvironment env, Result result) {
306             return new Context(env, result, TEST_TOOLKIT);
307         }
308 
309         public void runTest(Object ctx, int numReps) {
310             final Context ictx = (Context)ctx;
311             final Object input = ictx.input;
312             final int inputType = ictx.inputType;
313             final Toolkit tk = Toolkit.getDefaultToolkit();
314             final MediaTracker mt = new MediaTracker(canvas);
315             switch (inputType) {
316             case INPUT_FILE:
317                 String filename = ((File)input).getAbsolutePath();
318                 do {
319                     try {
320                         Image img = tk.createImage(filename);
321                         mt.addImage(img, 0);
322                         mt.waitForID(0, 0);
323                         mt.removeImage(img, 0);
324                     } catch (Exception e) {
325                         e.printStackTrace();
326                     }
327                 } while (--numReps >= 0);
328                 break;
329             case INPUT_URL:
330                 do {
331                     try {
332                         Image img = tk.createImage((URL)input);
333                         mt.addImage(img, 0);
334                         mt.waitForID(0, 0);
335                         mt.removeImage(img, 0);
336                     } catch (Exception e) {
337                         e.printStackTrace();
338                     }
339                 } while (--numReps >= 0);
340                 break;
341             case INPUT_ARRAY:
342                 do {
343                     try {
344                         Image img = tk.createImage((byte[])input);
345                         mt.addImage(img, 0);
346                         mt.waitForID(0, 0);
347                         mt.removeImage(img, 0);
348                     } catch (Exception e) {
349                         e.printStackTrace();
350                     }
351                 } while (--numReps >= 0);
352                 break;
353             default:
354                 throw new IllegalArgumentException("Invalid input type");
355             }
356         }
357     }

359     private static class ImageIORead extends InputImageTests {
360         public ImageIORead() {
361             super(imageioTestRoot,
362                   "imageioRead",
363                   "ImageIO.read()");
364             addDependency(generalSourceRoot,
365                 new Modifier.Filter() {
366                     public boolean isCompatible(Object val) {
367                         // ImageIO.read() handles FILE, URL, and ARRAY, but
368                         // not FILECHANNEL (well, I suppose we could create
369                         // an ImageInputStream from a FileChannel source,
370                         // but that's not a common use case; FileChannel is
371                         // better handled by the ImageReader tests below)
372                         InputType t = (InputType)val;
373                         return (t.getType() != INPUT_FILECHANNEL);
374                     }
375                 });
376             addDependencies(imageioOptRoot, true);
377         }
378 
379         public Object initTest(TestEnvironment env, Result result) {
380             return new Context(env, result, TEST_IMAGEIO);
381         }
382 
383         public void runTest(Object ctx, int numReps) {
384             final Context ictx = (Context)ctx;
385             final Object input = ictx.input;
386             final int inputType = ictx.inputType;
387             switch (inputType) {
388             case INPUT_FILE:
389                 do {
390                     try {
391                         ImageIO.read((File)input);
392                     } catch (Exception e) {
393                         e.printStackTrace();
394                     }
395                 } while (--numReps >= 0);
396                 break;
397             case INPUT_URL:
398                 do {
399                     try {
400                         ImageIO.read((URL)input);
401                     } catch (Exception e) {
402                         e.printStackTrace();
403                     }
404                 } while (--numReps >= 0);
405                 break;
406             case INPUT_ARRAY:
407                 do {
408                     try {
409                         ByteArrayInputStream bais =
410                             new ByteArrayInputStream((byte[])input);
411                         BufferedInputStream bis =
412                             new BufferedInputStream(bais);
413                         ImageIO.read(bis);
414                         bais.close();
415                     } catch (Exception e) {
416                         e.printStackTrace();
417                     }
418                 } while (--numReps >= 0);
419                 break;
420             default:
421                 throw new IllegalArgumentException("Invalid input type");
422             }
423         }
424     }

441         public void runTest(Object ctx, int numReps) {
442             final Context ictx = (Context)ctx;
443             final ImageReader reader = ictx.reader;
444             final boolean seekForwardOnly = ictx.seekForwardOnly;
445             final boolean ignoreMetadata = ictx.ignoreMetadata;
446             do {
447                 try {
448                     ImageInputStream iis = ictx.createImageInputStream();
449                     reader.setInput(iis, seekForwardOnly, ignoreMetadata);
450                     reader.read(0);
451                     reader.reset();
452                     iis.close();
453                     ictx.closeOriginalStream();
454                 } catch (IOException e) {
455                     e.printStackTrace();
456                 }
457             } while (--numReps >= 0);

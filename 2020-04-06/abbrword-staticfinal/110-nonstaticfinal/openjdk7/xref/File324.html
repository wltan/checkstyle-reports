98      static RegistrationData load(InputStream in) throws IOException {
99          Document document = initializeDocument(in);
100 
101         // Gets the registration URN
102         Element root = getRegistrationDataRoot(document);
103         Element registryRoot =
104                 getSingletonElementFromRoot(root, ST_NODE_REGISTRY);
105         String urn = registryRoot.getAttribute(ST_ATTR_REGISTRY_URN);
106 
107         // Construct a new RegistrationData object from the DOM tree
108         // Initialize the environment map and service tags
109         RegistrationData regData = new RegistrationData(urn);
110         addServiceTags(registryRoot, regData);
111 
112         Element envRoot = getSingletonElementFromRoot(root, ST_NODE_ENVIRONMENT);
113         buildEnvironmentMap(envRoot, regData);
114         return regData;
115     }

117     static void store(OutputStream os, RegistrationData registration)
118             throws IOException {
119         // create a new document with the root node
120         Document document = initializeDocument();
121 
122         // create the nodes for the environment map and the service tags
123         // in the registration data
124         addEnvironmentNodes(document,
125                             registration.getEnvironmentMap());
126         addServiceTagRegistry(document,
127                               registration.getRegistrationURN(),
128                               registration.getServiceTags());
129         transform(document, os);
130     }

132     // initialize a document from an input stream
133     private static Document initializeDocument(InputStream in) throws IOException {
134         DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
135         try {
136             // XML schema for validation
137             SchemaFactory sf = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
138             URL xsdUrl = RegistrationDocument.class.getResource(REGISTRATION_DATA_SCHEMA);
139             Schema schema = sf.newSchema(xsdUrl);
140             Validator validator = schema.newValidator();
141 
142             DocumentBuilder builder = factory.newDocumentBuilder();
143             Document doc = builder.parse(new InputSource(in));
144             validator.validate(new DOMSource(doc));
145             return doc;
146         } catch (SAXException sxe) {
147             IllegalArgumentException e = new IllegalArgumentException("Error generated in parsing");
148             e.initCause(sxe);
149             throw e;
150         } catch (ParserConfigurationException pce) {
151             // Parser with specific options can't be built
152             // should not reach here
153             InternalError x = new InternalError("Error in creating the new document");
154             x.initCause(pce);
155             throw x;
156         }
157     }

159     // initialize a new document for the registration data
160     private static Document initializeDocument() throws IOException {
161         DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
162         try {
163             DocumentBuilder builder = factory.newDocumentBuilder();
164             Document doc = builder.newDocument();
165 
166             // initialize the document with the registration_data root
167             Element root = doc.createElement(ST_NODE_REGISTRATION_DATA);
168             doc.appendChild(root);
169             root.setAttribute(ST_ATTR_REGISTRATION_VERSION, REGISTRATION_DATA_VERSION);
170 
171             return doc;
172         } catch (ParserConfigurationException pce) {
173             // Parser with specified options can't be built
174             // should not reach here
175             InternalError x = new InternalError("Error in creating the new document");
176             x.initCause(pce);
177             throw x;
178         }
179     }

181     // Transform the current DOM tree with the given output stream.
182     private static void transform(Document document, OutputStream os) {
183         try {
184             // Use a Transformer for output
185             TransformerFactory tFactory = TransformerFactory.newInstance();
186             tFactory.setAttribute("indent-number", new Integer(3));
187 
188             Transformer transformer = tFactory.newTransformer();
189 
190             transformer.setOutputProperty(OutputKeys.INDENT, "yes");
191             transformer.setOutputProperty(OutputKeys.METHOD, "xml");
192             transformer.setOutputProperty(OutputKeys.ENCODING, "UTF-8");
193             transformer.setOutputProperty(OutputKeys.STANDALONE, "yes");
194             transformer.transform(new DOMSource(document),
195                 new StreamResult(new BufferedWriter(new OutputStreamWriter(os, "UTF-8"))));
196         } catch (UnsupportedEncodingException ue) {
197             // Should not reach here
198             InternalError x = new InternalError("Error generated during transformation");
199             x.initCause(ue);
200             throw x;
201         } catch (TransformerConfigurationException tce) {
202             // Error generated by the parser
203             // Should not reach here
204             InternalError x = new InternalError("Error in creating the new document");
205             x.initCause(tce);
206             throw x;
207         } catch (TransformerException te) {
208             // Error generated by the transformer
209             InternalError x = new InternalError("Error generated during transformation");
210             x.initCause(te);
211             throw x;
212         }
213     }

215     private static void addServiceTagRegistry(Document document,
216                                               String registryURN,
217                                               Set<ServiceTag> svcTags) {
218         // add service tag registry node and its attributes
219         Element reg = document.createElement(ST_NODE_REGISTRY);
220         reg.setAttribute(ST_ATTR_REGISTRY_URN, registryURN);
221         reg.setAttribute(ST_ATTR_REGISTRY_VERSION, SERVICE_TAG_VERSION);
222 
223         Element root = getRegistrationDataRoot(document);
224         root.appendChild(reg);
225 
226         // adds the elements for the service tags
227         for (ServiceTag st : svcTags) {
228             addServiceTagElement(document, reg, st);
229         }
230     }

232     private static void addServiceTagElement(Document document,
233                                              Element registryRoot,
234                                              ServiceTag st) {
235         Element svcTag = document.createElement(ST_NODE_SERVICE_TAG);
236         registryRoot.appendChild(svcTag);
237         addChildElement(document, svcTag,
238                         ST_NODE_INSTANCE_URN, st.getInstanceURN());
239         addChildElement(document, svcTag,
240                         ST_NODE_PRODUCT_NAME, st.getProductName());
241         addChildElement(document, svcTag,
242                         ST_NODE_PRODUCT_VERSION, st.getProductVersion());
243         addChildElement(document, svcTag,
244                         ST_NODE_PRODUCT_URN, st.getProductURN());
245         addChildElement(document, svcTag,
246                         ST_NODE_PRODUCT_PARENT_URN, st.getProductParentURN());
247         addChildElement(document, svcTag,
248                         ST_NODE_PRODUCT_PARENT, st.getProductParent());
249         addChildElement(document, svcTag,
250                         ST_NODE_PRODUCT_DEFINED_INST_ID,
251                         st.getProductDefinedInstanceID());
252         addChildElement(document, svcTag,
253                         ST_NODE_PRODUCT_VENDOR, st.getProductVendor());
254         addChildElement(document, svcTag,
255                         ST_NODE_PLATFORM_ARCH, st.getPlatformArch());
256         addChildElement(document, svcTag,
257                         ST_NODE_TIMESTAMP, Util.formatTimestamp(st.getTimestamp()));
258         addChildElement(document, svcTag,
259                         ST_NODE_CONTAINER, st.getContainer());
260         addChildElement(document, svcTag,
261                         ST_NODE_SOURCE, st.getSource());
262         addChildElement(document, svcTag,
263                         ST_NODE_INSTALLER_UID,
264                         String.valueOf(st.getInstallerUID()));
265     }
266 
267     private static void addChildElement(Document document, Element root,
268                                         String element, String text) {
269         Element node = document.createElement(element);
270         node.appendChild(document.createTextNode(text));
271         root.appendChild(node);
272     }

274     // Constructs service tags from the document
275     private static void addServiceTags(Element registryRoot,
276                                        RegistrationData registration) {
277         NodeList children = registryRoot.getElementsByTagName(ST_NODE_SERVICE_TAG);
278         int length = (children == null ? 0 : children.getLength());
279         for (int i = 0; i < length; i++) {
280             Element svcTagElement = (Element) children.item(i);
281             ServiceTag st = getServiceTag(svcTagElement);
282             registration.addServiceTag(st);
283         }
284     }

286     // build environment map from the document
287     private static void buildEnvironmentMap(Element envRoot,
288                                          RegistrationData registration) {
289         registration.setEnvironment(ST_NODE_HOSTNAME, getTextValue(envRoot, ST_NODE_HOSTNAME));
290         registration.setEnvironment(ST_NODE_HOST_ID, getTextValue(envRoot, ST_NODE_HOST_ID));
291         registration.setEnvironment(ST_NODE_OS_NAME, getTextValue(envRoot, ST_NODE_OS_NAME));
292         registration.setEnvironment(ST_NODE_OS_VERSION, getTextValue(envRoot, ST_NODE_OS_VERSION));
293         registration.setEnvironment(ST_NODE_OS_ARCH, getTextValue(envRoot, ST_NODE_OS_ARCH));
294         registration.setEnvironment(ST_NODE_SYSTEM_MODEL, getTextValue(envRoot, ST_NODE_SYSTEM_MODEL));
295         registration.setEnvironment(ST_NODE_SYSTEM_MANUFACTURER, getTextValue(envRoot, ST_NODE_SYSTEM_MANUFACTURER));
296         registration.setEnvironment(ST_NODE_CPU_MANUFACTURER, getTextValue(envRoot, ST_NODE_CPU_MANUFACTURER));
297         registration.setEnvironment(ST_NODE_SERIAL_NUMBER, getTextValue(envRoot, ST_NODE_SERIAL_NUMBER));
298     }

300     // add the nodes representing the environment map in the document
301     private static void addEnvironmentNodes(Document document,
302                                             Map<String, String> envMap) {
303         Element root = getRegistrationDataRoot(document);
304         Element env = document.createElement(ST_NODE_ENVIRONMENT);
305         root.appendChild(env);
306         Set<Map.Entry<String, String>> keys = envMap.entrySet();
307         for (Map.Entry<String, String> entry : keys) {
308             addChildElement(document, env, entry.getKey(), entry.getValue());
309         }
310     }

312     private static Element getRegistrationDataRoot(Document doc) {
313         Element root = doc.getDocumentElement();
314         if (!root.getNodeName().equals(ST_NODE_REGISTRATION_DATA)) {
315             throw new IllegalArgumentException("Not a " +
316                     ST_NODE_REGISTRATION_DATA +
317                     " node \"" + root.getNodeName() + "\"");
318         }
319         return root;
320     }
321 
322     private static Element getSingletonElementFromRoot(Element root, String name) {
323         NodeList children = root.getElementsByTagName(name);
324         int length = (children == null ? 0 : children.getLength());
325         if (length != 1) {
326             throw new IllegalArgumentException("Invalid number of " + name +
327                     " nodes = " + length);
328         }
329         Element e = (Element) children.item(0);
330         if (!e.getNodeName().equals(name)) {
331             throw new IllegalArgumentException("Not a  " + name +
332                     " node \"" + e.getNodeName() + "\"");
333         }
334         return e;
335     }

337     // Constructs one ServiceTag instance from a service tag element root
338     private static ServiceTag getServiceTag(Element svcTagElement) {
339         return new ServiceTag(
340             getTextValue(svcTagElement, ST_NODE_INSTANCE_URN),
341             getTextValue(svcTagElement, ST_NODE_PRODUCT_NAME),
342             getTextValue(svcTagElement, ST_NODE_PRODUCT_VERSION),
343             getTextValue(svcTagElement, ST_NODE_PRODUCT_URN),
344             getTextValue(svcTagElement, ST_NODE_PRODUCT_PARENT),
345             getTextValue(svcTagElement, ST_NODE_PRODUCT_PARENT_URN),
346             getTextValue(svcTagElement, ST_NODE_PRODUCT_DEFINED_INST_ID),
347             getTextValue(svcTagElement, ST_NODE_PRODUCT_VENDOR),
348             getTextValue(svcTagElement, ST_NODE_PLATFORM_ARCH),
349             getTextValue(svcTagElement, ST_NODE_CONTAINER),
350             getTextValue(svcTagElement, ST_NODE_SOURCE),
351             Util.getIntValue(getTextValue(svcTagElement, ST_NODE_INSTALLER_UID)),
352             Util.parseTimestamp(getTextValue(svcTagElement, ST_NODE_TIMESTAMP))
353         );
354     }
355 
356     private static String getTextValue(Element e, String tagName) {
357         String value = "";
358         NodeList nl = e.getElementsByTagName(tagName);
359         if (nl != null && nl.getLength() > 0) {
360             Element el = (Element) nl.item(0);
361             Node node = el.getFirstChild();
362             if (node != null) {
363                 value = node.getNodeValue();
364             }
365         }
366         return value;
367     }

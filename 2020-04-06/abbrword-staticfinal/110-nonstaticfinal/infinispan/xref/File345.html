49     public void testLifespanExpiryInPut() throws InterruptedException {
50        Cache<String, String> cache = cm.getCache();
51        long lifespan = EXPIRATION_TIMEOUT;
52        cache.put("k", "v", lifespan, MILLISECONDS);
53  
54        DataContainer dc = cache.getAdvancedCache().getDataContainer();
55        InternalCacheEntry se = dc.get("k");
56        assert se.getKey().equals("k");
57        assert se.getValue().equals("v");
58        assert se.getLifespan() == lifespan;
59        assert se.getMaxIdle() == -1;
60        assert !se.isExpired(TIME_SERVICE.wallClockTime());
61        assert cache.get("k").equals("v");
62        Thread.sleep(lifespan + 100);
63        assert se.isExpired(TIME_SERVICE.wallClockTime());
64        assert cache.get("k") == null;
65     }

85     public void testLifespanExpiryInPutAll() throws InterruptedException {
86        Cache<String, String> cache = cm.getCache();
87        final long startTime = now();
88        final long lifespan = EXPIRATION_TIMEOUT;
89        Map<String, String> m = new HashMap();
90        m.put("k1", "v");
91        m.put("k2", "v");
92        cache.putAll(m, lifespan, MILLISECONDS);
93        while (true) {
94           String v1 = cache.get("k1");
95           String v2 = cache.get("k2");
96           if (moreThanDurationElapsed(startTime, lifespan))
97              break;
98           assertEquals("v", v1);
99           assertEquals("v", v2);
100          Thread.sleep(100);
101       }
102 
103       //make sure that in the next 30 secs data is removed
104       while (!moreThanDurationElapsed(startTime, lifespan + EVICTION_CHECK_TIMEOUT)) {
105          if (cache.get("k1") == null && cache.get("k2") == null) return;
106       }
107       assert cache.get("k1") == null;
108       assert cache.get("k2") == null;
109    }

129       Cache<String, String> cache = cm.getCache();
130       final long startTime = now();
131       final long lifespan = EXPIRATION_TIMEOUT;
132       assert cache.putIfAbsent("k", "v", lifespan, MILLISECONDS) == null;
133       long partial = lifespan / 10;
134       // Sleep some time within the lifespan boundaries
135       Thread.sleep(lifespan - partial);
136       assert "v".equals(cache.get("k")) || moreThanDurationElapsed(startTime, lifespan);
137       // Sleep some time that guarantees that it'll be over the lifespan
138       Thread.sleep(partial * 2);
139       assert cache.get("k") == null;
140 
141       //make sure that in the next 2 secs data is removed
142       while (!moreThanDurationElapsed(startTime, lifespan + EVICTION_CHECK_TIMEOUT)) {
143          if (cache.get("k") == null) break;
144          Thread.sleep(50);
145       }
146       assert cache.get("k") == null;
147 
148       cache.put("k", "v");
149       assert cache.putIfAbsent("k", "v", lifespan, MILLISECONDS) != null;
150    }

269    private void doTestEntrySetAfterExpiryInPut(Method m, CacheContainer cc) throws Exception {
270       Cache<Integer, String> cache = cc.getCache();
271       Set<Map.Entry<Integer, String>> entries;
272       Map dataIn = new HashMap();
273       dataIn.put(1, v(m, 1));
274       dataIn.put(2, v(m, 2));
275       Set entriesIn = dataIn.entrySet();
276 
277       final long startTime = now();
278       final long lifespan = EXPIRATION_TIMEOUT;
279       cache.putAll(dataIn, lifespan, TimeUnit.MILLISECONDS);
280 
281       while (true) {
282          // cache.entrySet is backing so expired entries will be removed so we have to make a copy
283          entries = new HashSet<>(cache.entrySet());
284          if (moreThanDurationElapsed(startTime, lifespan))
285             break;
286          // If the time hasn't lapsed then the entries should be the same
287          assertEquals(entriesIn, entries);
288          Thread.sleep(100);
289       }
290 
291       // Make sure that in the next 20 secs data is removed
292       while (!moreThanDurationElapsed(startTime, lifespan + EVICTION_CHECK_TIMEOUT)) {
293          entries = cache.entrySet();
294          if (entries.size() == 0) break;
295       }
296 
297       assertEquals(0, entries.size());
298    }

300    private void doEntrySetAfterExpiryInTransaction(Method m, CacheContainer cc) throws Exception {
301       Cache<Integer, String> cache = cc.getCache();
302       Set<Map.Entry<Integer, String>> entries;
303       Map dataIn = new HashMap();
304       dataIn.put(1, v(m, 1));
305       dataIn.put(2, v(m, 2));
306 
307       final long startTime = now();
308       final long lifespan = EXPIRATION_TIMEOUT;
309       cache.putAll(dataIn, lifespan, TimeUnit.MILLISECONDS);
310 
311       cache.getAdvancedCache().getTransactionManager().begin();
312       try {
313          Map txDataIn = new HashMap();
314          txDataIn.put(3, v(m, 3));
315          Map allEntriesIn = new HashMap(dataIn);
316          // Update expectations
317          allEntriesIn.putAll(txDataIn);
318          // Add an entry within tx
319          cache.putAll(txDataIn);
320 
321          while (true) {
322             entries = new HashSet<>(cache.entrySet());
323             if (moreThanDurationElapsed(startTime, lifespan))
324                break;
325             assertEquals(allEntriesIn.entrySet(), entries);
326             Thread.sleep(100);
327          }
328 
329          // Make sure that in the next 20 secs data is removed
330          while (!moreThanDurationElapsed(startTime, lifespan + EVICTION_CHECK_TIMEOUT)) {
331             entries = cache.entrySet();
332             if (entries.size() == 1) break;
333          }
334       } finally {
335          cache.getAdvancedCache().getTransactionManager().commit();
336       }
337 
338       assertEquals(1, entries.size());
339    }

468    private void doValuesAfterExpiryInTransaction(Method m, CacheContainer cc) throws Exception {
469       Cache<Integer, String> cache = cc.getCache();
470       // Values come as a Collection, but comparison of HashMap#Values is done
471       // by reference equality, so wrap the collection around to set to make
472       // testing easier, given that we know that there are dup values.
473       Collection<String> values;
474       Map<Integer, String> dataIn = new HashMap<>();
475       dataIn.put(1, v(m, 1));
476       dataIn.put(2, v(m, 2));
477 
478       final long startTime = now();
479       final long lifespan = EXPIRATION_TIMEOUT;
480       cache.putAll(dataIn, lifespan, TimeUnit.MILLISECONDS);
481 
482       cache.getAdvancedCache().getTransactionManager().begin();
483       try {
484          Map<Integer, String> txDataIn = new HashMap<>();
485          txDataIn.put(3, v(m, 3));
486          Collection<String> allValuesIn = new ArrayList<>(dataIn.values());
487          allValuesIn.addAll(txDataIn.values());
488 
489          // Add an entry within tx
490          cache.putAll(txDataIn);
491 
492          while (true) {
493             values = new ArrayList<>(cache.values());
494             if (moreThanDurationElapsed(startTime, lifespan))
495                break;
496             assertTrue(allValuesIn.containsAll(values));
497             Thread.sleep(100);
498          }
499 
500          // Make sure that in the next 20 secs data is removed
501          while (!moreThanDurationElapsed(startTime, lifespan + EVICTION_CHECK_TIMEOUT)) {
502             values = cache.values();
503             if (values.size() == 1) break;
504          }
505       } finally {
506          cache.getAdvancedCache().getTransactionManager().commit();
507       }
508 
509       assertEquals(1, values.size());
510    }

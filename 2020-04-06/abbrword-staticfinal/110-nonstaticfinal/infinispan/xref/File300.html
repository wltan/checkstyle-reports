180    @Override
181    protected void createCacheManagers() throws Throwable {
182       ConfigurationBuilder builder = getDefaultClusteredCacheConfig(cacheMode, true);
183       builder.locking().isolationLevel(IsolationLevel.REPEATABLE_READ).writeSkewCheck(false);
184       builder.transaction().lockingMode(lockingMode);
185       builder.clustering().hash().numOwners(1);
186       createClusteredCaches(2, builder);
187    }

189    private void doIsolationTest(boolean executeOnOwner, boolean initialized, Operation operation) throws Exception {
190       final Cache<Object, Object> ownerCache = cache(0);
191       final Object key = new MagicKey("shared", ownerCache);
192       final Cache<Object, Object> cache = executeOnOwner ? cache(0) : cache(1);
193       final TransactionManager tm = executeOnOwner ? tm(0) : tm(1);
194 
195       assertValueInAllCaches(key, null);
196 
197       final Object initValue = initialized ? INITIAL_VALUE : null;
198       if (initialized) {
199          ownerCache.put(key, initValue);
200          assertValueInAllCaches(key, initValue);
201       }
202 
203       tm.begin();
204       assertEquals("Wrong first get.", initValue, cache.get(key));
205       Transaction tx = tm.suspend();
206 
207       ownerCache.put(key, OTHER_VALUE);
208       assertValueInAllCaches(key, OTHER_VALUE);
209 
210       Object finalValueExpected = null;
211 
212       tm.resume(tx);
213       assertEquals("Wrong second get.", initValue, cache.get(key));
214       switch (operation) {
215          case PUT:
216             finalValueExpected = FINAL_VALUE;
217             assertEquals("Wrong put return value.", initValue, cache.put(key, FINAL_VALUE));
218             assertEquals("Wrong final get.", FINAL_VALUE, cache.get(key));
219             break;
220          case REMOVE:
221             finalValueExpected = null;
222             assertEquals("Wrong remove return value.", initValue, cache.remove(key));
223             assertEquals("Wrong final get.", null, cache.get(key));
224             break;
225          case REPLACE:
226             finalValueExpected = initialized ? FINAL_VALUE : OTHER_VALUE;
227             assertEquals("Wrong replace return value.", initValue, cache.replace(key, FINAL_VALUE));
228             assertEquals("Wrong final get.", initialized ? FINAL_VALUE : initValue, cache.get(key));
229             break;
230          case CONDITIONAL_PUT:
231             finalValueExpected = initialized ? OTHER_VALUE : FINAL_VALUE;
232             assertEquals("Wrong put return value.", initialized ? initValue : null, cache.putIfAbsent(key, FINAL_VALUE));
233             assertEquals("Wrong final get.", initialized ? initValue : FINAL_VALUE, cache.get(key));
234             break;
235          case CONDITIONAL_REMOVE:
236             finalValueExpected = initialized ? null : OTHER_VALUE;
237             assertEquals("Wrong remove return value.", initialized, cache.remove(key, INITIAL_VALUE));
238             assertEquals("Wrong final get.", null, cache.get(key));
239             break;
240          case CONDITIONAL_REPLACE:
241             finalValueExpected = initialized ? FINAL_VALUE : OTHER_VALUE;
242             assertEquals("Wrong replace return value.", initialized, cache.replace(key, INITIAL_VALUE, FINAL_VALUE));
243             assertEquals("Wrong final get.", initialized ? FINAL_VALUE : initValue, cache.get(key));
244             break;
245          default:
246             fail("Unknown operation " + operation);
247             break;
248       }
249       tm.commit();
250 
251       assertValueInAllCaches(key, finalValueExpected);
252       assertNoTransactions();
253    }

255    private void doIsolationTestAfterRemove(boolean executeOnOwner, Operation operation) throws Exception {
256       final Cache<Object, Object> ownerCache = cache(0);
257       final Object key = new MagicKey("shared", ownerCache);
258       final Cache<Object, Object> cache = executeOnOwner ? cache(0) : cache(1);
259       final TransactionManager tm = executeOnOwner ? tm(0) : tm(1);
260 
261       assertValueInAllCaches(key, null);
262 
263       final Object initValue = INITIAL_VALUE;
264       ownerCache.put(key, initValue);
265       assertValueInAllCaches(key, initValue);
266 
267 
268       tm.begin();
269       assertEquals("Wrong first get.", initValue, cache.get(key));
270       Transaction tx = tm.suspend();
271 
272       ownerCache.put(key, OTHER_VALUE);
273       assertValueInAllCaches(key, OTHER_VALUE);
274 
275       Object finalValueExpected = null;
276 
277       tm.resume(tx);
278       assertEquals("Wrong second get.", initValue, cache.get(key));
279       assertEquals("Wrong value after remove.", initValue, cache.remove(key));
280       switch (operation) {
281          case PUT:
282             finalValueExpected = FINAL_VALUE;
283             assertEquals("Wrong put return value.", null, cache.put(key, FINAL_VALUE));
284             assertEquals("Wrong final get.", FINAL_VALUE, cache.get(key));
285             break;
286          case REMOVE:
287             finalValueExpected = null;
288             assertEquals("Wrong remove return value.", null, cache.remove(key));
289             assertEquals("Wrong final get.", null, cache.get(key));
290             break;
291          case REPLACE:
292             finalValueExpected = null;
293             assertEquals("Wrong replace return value.", null, cache.replace(key, FINAL_VALUE));
294             assertEquals("Wrong final get.", null, cache.get(key));
295             break;
296          case CONDITIONAL_PUT:
297             finalValueExpected = FINAL_VALUE;
298             assertEquals("Wrong put return value.", null, cache.putIfAbsent(key, FINAL_VALUE));
299             assertEquals("Wrong final get.", FINAL_VALUE, cache.get(key));
300             break;
301          case CONDITIONAL_REMOVE:
302             finalValueExpected = null;
303             assertEquals("Wrong remove return value.", false, cache.remove(key, INITIAL_VALUE));
304             assertEquals("Wrong final get.", null, cache.get(key));
305             break;
306          case CONDITIONAL_REPLACE:
307             finalValueExpected = null;
308             assertEquals("Wrong replace return value.", false, cache.replace(key, INITIAL_VALUE, FINAL_VALUE));
309             assertEquals("Wrong final get.", null, cache.get(key));
310             break;
311          default:
312             fail("Unknown operation " + operation);
313             break;
314       }
315       tm.commit();
316 
317       assertValueInAllCaches(key, finalValueExpected);
318       assertNoTransactions();
319    }

321    private void assertValueInAllCaches(final Object key, final Object value) {
322       for (Cache<Object, Object> cache : caches()) {
323          assertEquals("Wrong value.", value, cache.get(key));
324       }
325    }

30  @Test(groups = "functional", testName="cli-server.InterpreterTest")
31  public class GrantDenyTest extends SingleCacheManagerTest {
32     static final Subject ADMIN = TestingUtil.makeSubject("admin");
33     private ClusterRoleMapper cpm;
34  
35     @Override
36     protected EmbeddedCacheManager createCacheManager() throws Exception {
37        GlobalConfigurationBuilder global = new GlobalConfigurationBuilder();
38        GlobalAuthorizationConfigurationBuilder globalRoles = global.security().authorization().enable()
39              .principalRoleMapper(new ClusterRoleMapper());
40        ConfigurationBuilder config = TestCacheManagerFactory.getDefaultCacheConfiguration(true);
41        AuthorizationConfigurationBuilder authConfig = config.security().authorization().enable();
42  
43        globalRoles
44           .role("reader").permission(AuthorizationPermission.ALL_READ)
45           .role("writer").permission(AuthorizationPermission.ALL_WRITE)
46           .role("admin").permission(AuthorizationPermission.ALL);
47        authConfig.role("reader").role("writer").role("admin").jmxStatistics().enable();
48        return TestCacheManagerFactory.createCacheManager(global, config);
49     }
50     
51     
52  
53     @Override
54     protected void setup() throws Exception {
55        cpm = Security.doAs(ADMIN, new PrivilegedExceptionAction<ClusterRoleMapper>() {
56           @Override
57           public ClusterRoleMapper run() throws Exception {
58              cacheManager = createCacheManager();
59              cpm = (ClusterRoleMapper) cacheManager.getCacheManagerConfiguration().security().authorization().principalRoleMapper();
60              cpm.grant("admin", "admin");
61              cache = cacheManager.getCache();
62              return cpm;
63           }
64        });
65     }
66  
67     @Override
68     protected void teardown() {
69        Security.doAs(ADMIN, new PrivilegedAction<Void>() {
70           @Override
71           public Void run() {
72              GrantDenyTest.super.teardown();
73              return null;
74           }
75        });
76     }
77  
78     @Override
79     protected void clearContent() {
80        Security.doAs(ADMIN, new PrivilegedAction<Void>() {
81           @Override
82           public Void run() {
83              cacheManager.getCache().clear();
84              return null;
85           }
86        });
87     }
88  
89     private Interpreter getInterpreter() {
90        GlobalComponentRegistry gcr = TestingUtil.extractGlobalComponentRegistry(this.cacheManager);
91        Interpreter interpreter = gcr.getComponent(Interpreter.class);
92        return interpreter;
93     }
94  
95     private Map<String, String> execute(Interpreter interpreter, String sessionId, String commands) throws Exception {
96        Map<String, String> result = interpreter.execute(sessionId, commands);
97        if (result.containsKey(ResultKeys.ERROR.toString())) {
98           fail(String.format("%s\n%s", result.get(ResultKeys.ERROR.toString()), result.get(ResultKeys.STACKTRACE.toString())));
99        }
100 
101       return result;
102    }
103 
104    public void testGrantDeny() throws Exception {
105       Interpreter interpreter = getInterpreter();
106       String sessionId = interpreter.createSessionId(BasicCacheContainer.DEFAULT_CACHE_NAME);
107       execute(interpreter, sessionId, "grant reader to jack;");
108       assertTrue(cpm.list("jack").contains("reader"));
109       execute(interpreter, sessionId, "grant reader to jill;");
110       assertTrue(cpm.list("jill").contains("reader"));
111       execute(interpreter, sessionId, "deny reader to jack;");
112       assertFalse(cpm.list("jack").contains("reader"));
113       Map<String, String> response = execute(interpreter, sessionId, "roles jill;");
114       assertEquals("[reader]", response.get(ResultKeys.OUTPUT.toString()));
115    }
116 }

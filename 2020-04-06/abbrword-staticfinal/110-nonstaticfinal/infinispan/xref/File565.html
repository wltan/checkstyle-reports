1   package org.infinispan.persistence.rest.configuration;
2   
3   import org.infinispan.commons.configuration.BuiltBy;
4   import org.infinispan.commons.configuration.ConfigurationFor;
5   import org.infinispan.commons.configuration.attributes.Attribute;
6   import org.infinispan.commons.configuration.attributes.AttributeDefinition;
7   import org.infinispan.commons.configuration.attributes.AttributeSet;
8   import org.infinispan.configuration.cache.AbstractStoreConfiguration;
9   import org.infinispan.configuration.cache.AsyncStoreConfiguration;
10  import org.infinispan.configuration.cache.SingletonStoreConfiguration;
11  import org.infinispan.persistence.keymappers.MarshalledValueOrPrimitiveMapper;
12  import org.infinispan.persistence.rest.RestStore;
13  import org.infinispan.persistence.rest.metadata.EmbeddedMetadataHelper;
14  
15  /**
16   * RestStoreConfiguration.
17   *
18   * @author Tristan Tarrant
19   * @since 6.0
20   */
21  @BuiltBy(RestStoreConfigurationBuilder.class)
22  @ConfigurationFor(RestStore.class)
23  public class RestStoreConfiguration extends AbstractStoreConfiguration {
24     static final AttributeDefinition<String> KEY2STRING_MAPPER = AttributeDefinition.builder("key2StringMapper", MarshalledValueOrPrimitiveMapper.class.getName()).immutable().build();
25     static final AttributeDefinition<String> METADATA_HELPER = AttributeDefinition.builder("metadataHelper", EmbeddedMetadataHelper.class.getName()).immutable().build();
26     static final AttributeDefinition<String> HOST = AttributeDefinition.builder("host", null, String.class).immutable().build();
27     static final AttributeDefinition<Integer> PORT = AttributeDefinition.builder("port", 80).immutable().build();
28     static final AttributeDefinition<String> PATH = AttributeDefinition.builder("path", "/").immutable().build();
29     static final AttributeDefinition<Boolean> APPEND_CACHE_NAME_TO_PATH = AttributeDefinition.builder("appendCacheNameToPath", false).immutable().build();
30     static final AttributeDefinition<Boolean> RAW_VALUES = AttributeDefinition.builder("rawValues", false).immutable().build();
31  
32     public static AttributeSet attributeDefinitionSet() {
33        return new AttributeSet(RestStoreConfiguration.class, AbstractStoreConfiguration.attributeDefinitionSet(), KEY2STRING_MAPPER, METADATA_HELPER, HOST, PORT, PATH, APPEND_CACHE_NAME_TO_PATH, RAW_VALUES);
34     }
35  
36     private final Attribute<String> key2StringMapper;
37     private final Attribute<String> metadataHelper;
38     private final Attribute<String> host;
39     private final Attribute<Integer> port;
40     private final Attribute<String> path;
41     private final Attribute<Boolean> appendCacheNameToPath;
42     private final Attribute<Boolean> rawValues;
43     private final ConnectionPoolConfiguration connectionPool;
44  
45     public RestStoreConfiguration(AttributeSet attributes,
46                                   AsyncStoreConfiguration async, SingletonStoreConfiguration singletonStore, ConnectionPoolConfiguration connectionPool) {
47        super(attributes, async, singletonStore);
48        key2StringMapper = attributes.attribute(KEY2STRING_MAPPER);
49        metadataHelper = attributes.attribute(METADATA_HELPER);
50        host = attributes.attribute(HOST);
51        port = attributes.attribute(PORT);
52        path = attributes.attribute(PATH);
53        appendCacheNameToPath = attributes.attribute(APPEND_CACHE_NAME_TO_PATH);
54        rawValues = attributes.attribute(RAW_VALUES);
55        this.connectionPool = connectionPool;
56     }
57  
58     public ConnectionPoolConfiguration connectionPool() {
59        return connectionPool;
60     }
61  
62     public String key2StringMapper() {
63        return key2StringMapper.get();
64     }
65  
66     public String metadataHelper() {
67        return metadataHelper.get();
68     }
69  
70     public String host() {
71        return host.get();
72     }
73  
74     public int port() {
75        return port.get();
76     }
77  
78     public String path() {
79        return path.get();
80     }
81  
82     public boolean appendCacheNameToPath() {
83        return appendCacheNameToPath.get();
84     }
85  
86     public boolean rawValues() {
87        return rawValues.get();
88     }
89  
90     @Override
91     public String toString() {
92        return "RestStoreConfiguration [connectionPool=" + connectionPool + ", attributes=" + attributes + "]";
93     }
94  }

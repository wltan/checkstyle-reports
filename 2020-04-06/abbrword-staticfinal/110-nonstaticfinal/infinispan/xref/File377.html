26  /**
27   * Tests if the conditional commands correctly fetch the value from cache loader even with the skip cache load/store
28   * flags.
29   * <p/>
30   * The configuration used is a non-tx distributed cache without passivation.
31   *
32   * @author Pedro Ruivo
33   * @since 7.0
34   */
35  @Test(groups = "functional", testName = "persistence.ClusteredConditionalCommandTest")
36  public class ClusteredConditionalCommandTest extends MultipleCacheManagersTest {
37     private static final String PRIVATE_STORE_CACHE_NAME = "private-store-cache";
38     private static final String SHARED_STORE_CACHE_NAME = "shared-store-cache";
39     private final String key = getClass().getSimpleName() + "-key";
40     private final String value1 = getClass().getSimpleName() + "-value1";
41     private final String value2 = getClass().getSimpleName() + "-value2";
42     private final boolean transactional;
43     private final boolean passivation;
44  
45     public ClusteredConditionalCommandTest() {
46        this(false, false);
47     }
48  
49     protected ClusteredConditionalCommandTest(boolean transactional, boolean passivation) {
50        this.transactional = transactional;
51        this.passivation = passivation;
52     }

54     private static ConfigurationBuilder createConfiguration(String storeName, boolean shared, boolean transactional,
55                                                             boolean passivation, int storePrefix) {
56        ConfigurationBuilder builder = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, transactional);
57        builder.jmxStatistics().enable();
58        builder.clustering()
59              .hash().numOwners(2);
60        builder.persistence()
61              .passivation(passivation)
62              .addStore(DummyInMemoryStoreConfigurationBuilder.class)
63              .storeName(storeName + (shared ? "-shared" : storePrefix))
64              .fetchPersistentState(false)
65              .purgeOnStartup(true)
66              .shared(shared);
67        return builder;
68     }

70     private static <K, V> void writeToStore(CacheHelper<K, V> cacheHelper, Ownership ownership, K key, V value) {
71        cacheHelper.cacheStore(ownership).write(marshalledEntry(key, value, cacheHelper.marshaller(ownership)));
72     }
73  
74     private static <K, V> MarshalledEntry<K, V> marshalledEntry(K key, V value, StreamingMarshaller marshaller) {
75        return new MarshalledEntryImpl<>(key, value, null, marshaller);
76     }
77  
78     private <K, V> CacheHelper<K, V> create(List<Cache<K, V>> cacheList) {
79        CacheHelper<K, V> cacheHelper = new CacheHelper<>();
80        for (Cache<K, V> cache : cacheList) {
81           ClusteringDependentLogic clusteringDependentLogic = extractComponent(cache, ClusteringDependentLogic.class);
82           log.debugf("owners for key %s are %s", key, clusteringDependentLogic.getOwners(key));
83           if (clusteringDependentLogic.localNodeIsPrimaryOwner(key)) {
84              log.debug("Cache " + address(cache) + " is the primary owner");
85              assertTrue(cacheHelper.addCache(Ownership.PRIMARY_OWNER, cache));
86           } else if (clusteringDependentLogic.localNodeIsOwner(key)) {
87              log.debug("Cache " + address(cache) + " is the backup owner");
88              assertTrue(cacheHelper.addCache(Ownership.BACKUP_OWNER, cache));
89           } else {
90              log.debug("Cache " + address(cache) + " is the non owner");
91              assertTrue(cacheHelper.addCache(Ownership.NON_OWNER, cache));
92           }
93        }
94        return cacheHelper;
95     }

97     private void doTest(List<Cache<String, String>> cacheList, ConditionalOperation operation, Ownership ownership,
98                         Flag flag, boolean shared) {
99        waitForRehashToComplete(cacheList);
100       final CacheHelper<String, String> cacheHelper = create(cacheList);
101       final boolean skipLoad = flag == Flag.SKIP_CACHE_LOAD || flag == Flag.SKIP_CACHE_STORE;
102       assertEmpty(cacheList);
103       initStore(cacheHelper, shared);
104 
105       try {
106          if (flag != null) {
107             operation.execute(cacheHelper.cache(ownership).getAdvancedCache().withFlags(flag), key, value1, value2);
108          } else {
109             operation.execute(cacheHelper.cache(ownership), key, value1, value2);
110          }
111       } catch (Exception e) {
112          //some operation are allowed to fail. e.g. putIfAbsent.
113          //we only check the final value
114          log.debug(e);
115       }
116 
117       assertLoadAfterOperation(cacheHelper, operation, ownership, skipLoad);
118 
119       assertEquals(operation.finalValue(value1, value2, skipLoad), cacheHelper.cache(Ownership.PRIMARY_OWNER).get(key));
120       assertEquals(operation.finalValue(value1, value2, skipLoad), cacheHelper.cache(Ownership.BACKUP_OWNER).get(key));
121       //don't test in non_owner because it generates remote gets and they can cross tests causing random failures.
122    }
123 
124    protected <K, V> void assertLoadAfterOperation(CacheHelper<K, V> cacheHelper, ConditionalOperation operation, Ownership ownership, boolean skipLoad) {
125       //in non-transactional caches, only the primary owner will load from store
126       assertLoad(cacheHelper, skipLoad ? 0 : 1, 0, 0);
127    }
128 
129    protected final <K, V> void assertLoad(CacheHelper<K, V> cacheHelper, int primaryOwner, int backupOwner, int nonOwner) {
130       assertEquals("primary owner load", primaryOwner, cacheHelper.loads(Ownership.PRIMARY_OWNER));
131       assertEquals("backup owner load", backupOwner, cacheHelper.loads(Ownership.BACKUP_OWNER));
132       assertEquals("non owner load", nonOwner, cacheHelper.loads(Ownership.NON_OWNER));
133    }
134 
135    private <K, V> void assertEmpty(List<Cache<K, V>> cacheList) {
136       for (Cache<K, V> cache : cacheList) {
137          assertTrue(cache + ".isEmpty()", cache.isEmpty());
138       }
139    }

141    private void initStore(CacheHelper<String, String> cacheHelper, boolean shared) {
142       if (shared) {
143          writeToStore(cacheHelper, Ownership.PRIMARY_OWNER, key, value1);
144          assertTrue(cacheHelper.cacheStore(Ownership.PRIMARY_OWNER).contains(key));
145          assertTrue(cacheHelper.cacheStore(Ownership.BACKUP_OWNER).contains(key));
146          assertTrue(cacheHelper.cacheStore(Ownership.NON_OWNER).contains(key));
147       } else {
148          writeToStore(cacheHelper, Ownership.PRIMARY_OWNER, key, value1);
149          writeToStore(cacheHelper, Ownership.BACKUP_OWNER, key, value1);
150          assertTrue(cacheHelper.cacheStore(Ownership.PRIMARY_OWNER).contains(key));
151          assertTrue(cacheHelper.cacheStore(Ownership.BACKUP_OWNER).contains(key));
152          assertFalse(cacheHelper.cacheStore(Ownership.NON_OWNER).contains(key));
153       }
154 
155       cacheHelper.resetStats(Ownership.PRIMARY_OWNER);
156       cacheHelper.resetStats(Ownership.BACKUP_OWNER);
157       cacheHelper.resetStats(Ownership.NON_OWNER);
158    }

160    @Override
161    protected void createCacheManagers() throws Throwable {
162       createCluster(getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC), 3);
163 
164       int index = 0;
165       for (EmbeddedCacheManager cacheManager : cacheManagers) {
166          cacheManager.defineConfiguration(SHARED_STORE_CACHE_NAME,
167                                           createConfiguration(getClass().getSimpleName(), true, transactional,
168                                                               passivation, index).build());
169          cacheManager.defineConfiguration(PRIVATE_STORE_CACHE_NAME,
170                                           createConfiguration(getClass().getSimpleName(), false, transactional,
171                                                               passivation, index).build());
172          index++;
173       }
174    }

470    protected static enum ConditionalOperation {
471       PUT_IF_ABSENT {
472          @Override
473          public <K, V> void execute(Cache<K, V> cache, K key, V value1, V value2) {
474             cache.putIfAbsent(key, value2);
475          }
476 
477          @Override
478          public <V> V finalValue(V value1, V value2, boolean skipLoad) {
479             return skipLoad ? value2 : value1;
480          }
481       },
482       REPLACE {
483          @Override
484          public <K, V> void execute(Cache<K, V> cache, K key, V value1, V value2) {
485             cache.replace(key, value2);
486          }
487 
488          @Override
489          public <V> V finalValue(V value1, V value2, boolean skipLoad) {
490             return skipLoad ? value1 : value2;
491          }
492       },
493       REPLACE_IF {
494          @Override
495          public <K, V> void execute(Cache<K, V> cache, K key, V value1, V value2) {
496             cache.replace(key, value1, value2);
497          }
498 
499          @Override
500          public <V> V finalValue(V value1, V value2, boolean skipLoad) {
501             return skipLoad ? value1 : value2;
502          }
503       },
504       REMOVE_IF {
505          @Override
506          public <K, V> void execute(Cache<K, V> cache, K key, V value1, V value2) {
507             cache.remove(key, value1);
508          }
509 
510          @Override
511          public <V> V finalValue(V value1, V value2, boolean skipLoad) {
512             return skipLoad ? value1 : null;
513          }
514       };
515 
516       public abstract <K, V> void execute(Cache<K, V> cache, K key, V value1, V value2);
517 
518       public abstract <V> V finalValue(V value1, V value2, boolean skipLoad);
519    }

521    protected static class CacheHelper<K, V> {
522       private final Map<Ownership, Cache<K, V>> cacheEnumMap;
523 
524       private CacheHelper() {
525          cacheEnumMap = new EnumMap<>(Ownership.class);
526       }
527 
528       public boolean addCache(Ownership ownership, Cache<K, V> cache) {
529          boolean contains = cacheEnumMap.containsKey(ownership);
530          if (!contains) {
531             cacheEnumMap.put(ownership, cache);
532          }
533          return !contains;
534       }
535 
536       private Cache<K, V> cache(Ownership ownership) {
537          return cacheEnumMap.get(ownership);
538       }
539 
540       private DummyInMemoryStore cacheStore(Ownership ownership) {
541          return getFirstWriter(cache(ownership));
542       }
543 
544       private StreamingMarshaller marshaller(Ownership ownership) {
545          return cacheEnumMap.get(ownership).getAdvancedCache().getComponentRegistry().getCacheMarshaller();
546       }
547 
548       private long loads(Ownership ownership) {
549          InterceptorChain chain = extractComponent(cache(ownership), InterceptorChain.class);
550          CacheLoaderInterceptor interceptor = (CacheLoaderInterceptor) chain.getInterceptorsWhichExtend(CacheLoaderInterceptor.class).get(0);
551          return interceptor.getCacheLoaderLoads();
552       }
553 
554       private void resetStats(Ownership ownership) {
555          InterceptorChain chain = extractComponent(cache(ownership), InterceptorChain.class);
556          CacheLoaderInterceptor interceptor = (CacheLoaderInterceptor) chain.getInterceptorsWhichExtend(CacheLoaderInterceptor.class).get(0);
557          interceptor.resetStatistics();
558       }
559    }

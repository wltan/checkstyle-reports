89              .setAllowExpression(true)
90              .setAttributeMarshaller(new AttributeMarshaller() {
91                  @Override
92                  public void marshallAsElement(AttributeDefinition attribute, ModelNode resourceModel, boolean marshallDefault, XMLStreamWriter writer) throws XMLStreamException {
93                      resourceModel = resourceModel.get(attribute.getName());
94                      if (!resourceModel.isDefined()) {
95                          return;
96                      }
97                      for (Property property : resourceModel.asPropertyList()) {
98                          writer.writeStartElement(org.jboss.as.controller.parsing.Element.PROPERTY.getLocalName());
99                          writer.writeAttribute(org.jboss.as.controller.parsing.Element.NAME.getLocalName(), property.getName());
100                         writer.writeCharacters(property.getValue().asString());
101                         writer.writeEndElement();
102                     }
103                 }
104             })
105             .build();

223     protected final ResolvePathHandler resolvePathHandler;
224     protected final boolean runtimeRegistration;
225     private final CacheAdd cacheAddHandler;
226 
227     public CacheResource(PathElement pathElement, ResourceDescriptionResolver descriptionResolver, CacheAdd cacheAddHandler, OperationStepHandler removeHandler, ResolvePathHandler resolvePathHandler, boolean runtimeRegistration) {
228         super(pathElement, descriptionResolver, cacheAddHandler, removeHandler);
229         this.cacheAddHandler = cacheAddHandler;
230         this.resolvePathHandler = resolvePathHandler;
231         this.runtimeRegistration = runtimeRegistration;
232     }
233 
234     public CacheAdd getCacheAddHandler() {
235         return cacheAddHandler;
236     }
237 
238     public boolean isRuntimeRegistration() {
239         return runtimeRegistration;
240     }
241 
242     @Override
243     public void registerAttributes(ManagementResourceRegistration resourceRegistration) {
244         super.registerAttributes(resourceRegistration);
245 
246         final OperationStepHandler restartWriteHandler = new RestartCacheWriteAttributeHandler(getPathElement().getKey(), cacheAddHandler, CACHE_ATTRIBUTES);
247         for (AttributeDefinition attr : CACHE_ATTRIBUTES) {
248             resourceRegistration.registerReadWriteAttribute(attr, CacheReadAttributeHandler.INSTANCE, restartWriteHandler);
249         }
250     }
251 
252     @Override
253     public void registerOperations(ManagementResourceRegistration resourceRegistration) {
254         super.registerOperations(resourceRegistration);
255         if (runtimeRegistration) {
256             resourceRegistration.registerOperationHandler(CacheResource.CLEAR_CACHE, CacheCommands.ClearCacheCommand.INSTANCE);
257             resourceRegistration.registerOperationHandler(CacheResource.START_CACHE, CacheCommands.StartCacheCommand.INSTANCE);
258             resourceRegistration.registerOperationHandler(CacheResource.STOP_CACHE, CacheCommands.StopCacheCommand.INSTANCE);
259             resourceRegistration.registerOperationHandler(CacheResource.RESET_STATISTICS, CacheCommands.ResetCacheStatisticsCommand.INSTANCE);
260             resourceRegistration.registerOperationHandler(CacheResource.RESET_ACTIVATION_STATISTICS, CacheCommands.ResetActivationStatisticsCommand.INSTANCE);
261             resourceRegistration.registerOperationHandler(CacheResource.RESET_INVALIDATION_STATISTICS, CacheCommands.ResetInvalidationStatisticsCommand.INSTANCE);
262             resourceRegistration.registerOperationHandler(CacheResource.RESET_PASSIVATION_STATISTICS, CacheCommands.ResetPassivationStatisticsCommand.INSTANCE);
263             resourceRegistration.registerOperationHandler(CacheResource.RESET_RPC_MANAGER_STATISTICS, CacheCommands.ResetRpcManagerStatisticsCommand.INSTANCE);
264             resourceRegistration.registerOperationHandler(CacheResource.DISCONNECT_SOURCE, CacheCommands.DisconnectSourceCommand.INSTANCE);
265             resourceRegistration.registerOperationHandler(CacheResource.RECORD_KNOWN_GLOBAL_KEYSET, CacheCommands.RecordGlobalKeySetCommand.INSTANCE);
266             resourceRegistration.registerOperationHandler(CacheResource.SYNCHRONIZE_DATA, CacheCommands.SynchronizeDataCommand.INSTANCE);
267             resourceRegistration.registerOperationHandler(CacheResource.MASS_REINDEX, CacheCommands.MassReindexCommand.INSTANCE);
268         }
269     }

271     @Override
272     public void registerChildren(ManagementResourceRegistration resourceRegistration) {
273         super.registerChildren(resourceRegistration);
274 
275         resourceRegistration.registerSubModel(new LockingResource(this));

277         resourceRegistration.registerSubModel(new EvictionResource(this));

279         resourceRegistration.registerSubModel(new CompatibilityResource(this));
280         resourceRegistration.registerSubModel(new LoaderResource(this));
281         resourceRegistration.registerSubModel(new ClusterLoaderResource(this));
282         resourceRegistration.registerSubModel(new BackupSiteResource(this));
283         resourceRegistration.registerSubModel(new StoreResource(this));
284         resourceRegistration.registerSubModel(new FileStoreResource(this, resolvePathHandler));
285         resourceRegistration.registerSubModel(new StringKeyedJDBCStoreResource(this));
286         resourceRegistration.registerSubModel(new BinaryKeyedJDBCStoreResource(this));
287         resourceRegistration.registerSubModel(new MixedKeyedJDBCStoreResource(this));
288         resourceRegistration.registerSubModel(new RemoteStoreResource(this));
289         resourceRegistration.registerSubModel(new LevelDBStoreResource(this, resolvePathHandler));
290         resourceRegistration.registerSubModel(new RestStoreResource(this));
291         resourceRegistration.registerSubModel(new CacheSecurityResource(this));
292     }

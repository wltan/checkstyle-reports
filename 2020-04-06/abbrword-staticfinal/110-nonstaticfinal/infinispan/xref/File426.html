168    public static void replaceField(Object newValue, String fieldName, Object owner, Class baseType) {
169       Field field;
170       try {
171          field = baseType.getDeclaredField(fieldName);
172          field.setAccessible(true);
173          field.set(owner, newValue);
174       }
175       catch (Exception e) {
176          throw new RuntimeException(e);//just to simplify exception handling
177       }

181    public static Object extractField(Class type, Object target, String fieldName) {
182       while (true) {
183          Field field;
184          try {
185             field = type.getDeclaredField(fieldName);
186             field.setAccessible(true);
187             return field.get(target);
188          }
189          catch (Exception e) {
190             if (type.equals(Object.class)) {
191                e.printStackTrace();
192                return null;
193             } else {
194                // try with superclass!!
195                type = type.getSuperclass();
196             }
197          }
198       }
199    }

208    public static void waitForRehashToComplete(Cache... caches) {
209       final int REHASH_TIMEOUT_SECONDS = 60; //Needs to be rather large to prevent sporadic failures on CI
210       final long giveup = System.nanoTime() + TimeUnit.SECONDS.toNanos(REHASH_TIMEOUT_SECONDS);
211       for (Cache c : caches) {
212          if (c instanceof SecureCacheImpl) {
213             c = (Cache) extractField(SecureCacheImpl.class, c, "delegate");
214          }
215          StateTransferManager stateTransferManager = extractComponent(c, StateTransferManager.class);
216          Address cacheAddress = c.getAdvancedCache().getRpcManager().getAddress();
217          while (true) {
218             CacheTopology cacheTopology = stateTransferManager.getCacheTopology();
219             ConsistentHash currentCH = cacheTopology.getCurrentCH();
220             boolean rebalanceInProgress = cacheTopology.getPendingCH() != null;
221             boolean chContainsAllMembers = currentCH.getMembers().size() == caches.length;
222             boolean currentChIsBalanced = true;
223             int actualNumOwners = Math.min(currentCH.getNumOwners(), currentCH.getMembers().size());
224             for (int i = 0; i < currentCH.getNumSegments(); i++) {
225                if (currentCH.locateOwnersForSegment(i).size() < actualNumOwners) {
226                   currentChIsBalanced = false;
227                   break;
228                }
229             }
230             if (chContainsAllMembers && !rebalanceInProgress && currentChIsBalanced)
231                break;
232 
233             if (System.nanoTime() - giveup > 0) {
234                String message;
235                if (!chContainsAllMembers) {
236                   Address[] addresses = new Address[caches.length];
237                   for (int i = 0; i < caches.length; i++) {
238                      addresses[i] = caches[i].getCacheManager().getAddress();
239                   }
240                   message = String.format("Timed out waiting for rebalancing to complete on node %s, " +
241                         "expected member list is %s, current member list is %s!",
242                         cacheAddress, Arrays.toString(addresses), cacheTopology.getCurrentCH().getMembers());
243                } else {
244                   message = String.format("Timed out waiting for rebalancing to complete on node %s, " +
245                         "current topology is %s. rebalanceInProgress=%s, currentChIsBalanced=%s", c.getCacheManager().getAddress(),
246                                           cacheTopology, rebalanceInProgress, currentChIsBalanced);
247                }
248                log.error(message);
249                throw new RuntimeException(message);
250             }
251 
252             LockSupport.parkNanos(TimeUnit.MILLISECONDS.toNanos(100));
253          }
254          log.trace("Node " + cacheAddress + " finished state transfer.");
255       }
256    }

262    /**
263     * Loops, continually calling {@link #areCacheViewsComplete(Cache[])} until it either returns true or
264     * <code>timeout</code> ms have elapsed.
265     *
266     * @param caches  caches which must all have consistent views
267     * @param timeout max number of ms to loop
268     *
269     * @throws RuntimeException if <code>timeout</code> ms have elapse without all caches having the same number of
270     *                          members.
271     */
272    public static void blockUntilViewsReceived(Cache[] caches, long timeout) {
273       long failTime = System.currentTimeMillis() + timeout;
274 
275       while (System.currentTimeMillis() < failTime) {
276          sleepThread(100);
277          if (areCacheViewsComplete(caches)) {
278             return;
279          }
280       }
281 
282       viewsTimedOut(caches);
283    }

285    private static void viewsTimedOut(Cache[] caches) {
286       CacheContainer[] cacheContainers = new CacheContainer[caches.length];
287       for (int i = 0; i < caches.length; i++) {
288          cacheContainers[i] = caches[i].getCacheManager();
289       }
290       viewsTimedOut(cacheContainers);
291    }
292    private static void viewsTimedOut(CacheContainer[] cacheContainers) {
293       int length = cacheContainers.length;
294       List<View> incompleteViews = new ArrayList<>(length);
295       for (int i = 0; i < length; i++) {
296          EmbeddedCacheManager cm = (EmbeddedCacheManager) cacheContainers[i];
297          if (cm.getMembers().size() != cacheContainers.length) {
298             incompleteViews.add(((JGroupsTransport) cm.getTransport()).getChannel().getView());
299             log.warnf("Manager %s has an incomplete view: %s", cm.getAddress(), cm.getMembers());
300          }
301       }
302 
303       throw new TimeoutException(String.format(
304          "Timed out before caches had complete views.  Expected %d members in each view.  Views are as follows: %s",
305          cacheContainers.length, incompleteViews));
306    }

482    public static boolean isCacheViewComplete(List members, Address address, int memberCount, boolean barfIfTooManyMembers) {
483       if (members == null || memberCount > members.size()) {
484          return false;
485       } else if (memberCount < members.size()) {
486          if (barfIfTooManyMembers) {
487             // This is an exceptional condition
488             StringBuilder sb = new StringBuilder("Cache at address ");
489             sb.append(address);
490             sb.append(" had ");
491             sb.append(members.size());
492             sb.append(" members; expecting ");
493             sb.append(memberCount);
494             sb.append(". Members were (");
495             for (int j = 0; j < members.size(); j++) {
496                if (j > 0) {
497                   sb.append(", ");
498                }
499                sb.append(members.get(j));
500             }
501             sb.append(')');
502 
503             throw new IllegalStateException(sb.toString());
504          } else return false;
505       }
506 
507       return true;
508    }

510    /**
511     * This method blocks until the given caches have a view of whose size
512     * matches the desired value. This method is particularly useful for
513     * discovering that members have been split, or that they have joined back
514     * again.
515     *
516     * @param timeout max number of milliseconds to block for
517     * @param finalViewSize desired final view size
518     * @param caches caches representing current, or expected members in the cluster.
519     */
520    public static void blockUntilViewsChanged(long timeout, int finalViewSize, Cache... caches) {
521       blockUntilViewsChanged(caches, timeout, finalViewSize);
522    }

524    private static void blockUntilViewsChanged(Cache[] caches, long timeout, int finalViewSize) {
525       long failTime = System.currentTimeMillis() + timeout;
526 
527       while (System.currentTimeMillis() < failTime) {
528          sleepThread(100);
529          if (areCacheViewsChanged(caches, finalViewSize)) {
530             return;
531          }
532       }
533 
534       List<List<Address>> allViews = new ArrayList<List<Address>>(caches.length);
535       for (int i = 0; i < caches.length; i++) {
536          allViews.add(caches[i].getCacheManager().getMembers());
537       }
538 
539       throw new RuntimeException(String.format(
540             "Timed out before caches had changed views (%s) to contain %d members",
541             allViews, finalViewSize));
542    }

544    private static boolean areCacheViewsChanged(Cache[] caches, int finalViewSize) {
545       int memberCount = caches.length;
546 
547       for (int i = 0; i < memberCount; i++) {
548          EmbeddedCacheManager cacheManager = caches[i].getCacheManager();
549          if (!isCacheViewChanged(cacheManager.getMembers(), finalViewSize)) {
550             return false;
551          }
552       }
553 
554       return true;
555    }

557    private static boolean isCacheViewChanged(List members, int finalViewSize) {
558       if (members == null || finalViewSize != members.size())
559          return false;
560       else
561          return true;
562    }

611    private static void recursiveDelete(File f) {
612       File absoluteFile = f.getAbsoluteFile();
613       if (absoluteFile.isDirectory()) {
614          File[] files = absoluteFile.listFiles();
615          if (files != null) {
616             for (File file : files) {
617                recursiveDelete(file);
618             }
619          }
620       }
621       absoluteFile.delete();
622    }

672    public static void clearContent(EmbeddedCacheManager cacheContainer) {
673       if (cacheContainer != null && cacheContainer.getStatus().allowInvocations()) {
674          Set<Cache> runningCaches = getRunningCaches(cacheContainer);
675          for (Cache cache : runningCaches) {
676             clearRunningTx(cache);
677          }
678 
679          if (!cacheContainer.getStatus().allowInvocations()) return;
680 
681          Cache<?, ?> registryCache = cacheContainer.getCache(ClusterRegistryImpl.GLOBAL_REGISTRY_CACHE_NAME, false);
682          if (registryCache != null && registryCache.getStatus().allowInvocations()) {
683             runningCaches.add(registryCache);
684          }
685          for (Cache cache : runningCaches) {
686             clearReplicationQueues(cache);
687             clearCacheLoader(cache);
688             removeInMemoryData(cache);
689          }
690       }
691    }

693    protected static Set<Cache> getRunningCaches(EmbeddedCacheManager cacheContainer) {
694       Set<Cache> running = new HashSet<Cache>();
695       if (cacheContainer == null || !cacheContainer.getStatus().allowInvocations())
696          return running;
697 
698       for (String cacheName : cacheContainer.getCacheNames()) {
699          if (cacheContainer.isRunning(cacheName)) {
700             Cache c = cacheContainer.getCache(cacheName, false);
701             if (c != null && c.getStatus().allowInvocations()) {
702                running.add(c);
703             }
704          }
705       }
706 
707       if (cacheContainer.isDefaultRunning()) {
708          Cache defaultCache = cacheContainer.getCache(EmbeddedCacheManager.DEFAULT_CACHE_NAME, false);
709          if (defaultCache != null && defaultCache.getStatus().allowInvocations()) {
710             running.add(defaultCache);
711          }
712       }
713 
714       return running;
715    }

777     * Kills a cache - stops it and rolls back any associated txs
778     */
779    public static void killCaches(boolean clear, Collection<Cache> caches) {
780       for (Cache c : caches) {
781          try {
782             if (c != null && c.getStatus() == ComponentStatus.RUNNING) {
783                TransactionManager tm = c.getAdvancedCache().getTransactionManager();
784                if (tm != null) {
785                   try {
786                      tm.rollback();
787                   }
788                   catch (Exception e) {
789                      // don't care
790                   }
791                }
792                if (c.getAdvancedCache().getRpcManager() != null) {
793                   log.tracef("Cache contents on %s before stopping: %s", c.getAdvancedCache().getRpcManager().getAddress(),
794                              c.getAdvancedCache().withFlags(Flag.CACHE_MODE_LOCAL).entrySet());
795                } else {
796                   log.tracef("Cache contents before stopping: %s", c.getAdvancedCache().withFlags(Flag.CACHE_MODE_LOCAL).entrySet());
797                }
798                if (clear) {
799                   try {
800                      c.clear();
801                   } catch (Exception ignored) {}
802                }
803                c.stop();
804             }
805          }
806          catch (Throwable t) {
807 
808          }
809       }
810    }

1301    /**
1302     * Returns a callable that will call the provided callable within a transaction.  This method guarantees that the
1303     * right pattern is used to make sure that the transaction is always either committed or rollbacked around
1304     * the callable.
1305     *
1306     * @param tm transaction manager
1307     * @param c callable instance to run within a transaction
1308     * @param <T> tyep of callable to return
1309     * @return The callable to invoke.  Note as long as the provided callable is thread safe this callable will be as well
1310     */
1311    public static <T> Callable<T> withTxCallable(final TransactionManager tm, final Callable<? extends T> c) {
1312       return new Callable<T>() {
1313          @Override
1314          public T call() throws Exception {
1315             tm.begin();
1316             try {
1317                return c.call();
1318             } catch (Exception e) {
1319                tm.setRollbackOnly();
1320                throw e;
1321             } finally {
1322                if (tm.getStatus() == Status.STATUS_ACTIVE) tm.commit();
1323                else tm.rollback();
1324             }
1325          }
1326       };
1327    }

1564    public static void expectCause(Throwable t, Class<? extends Throwable> c, String messageRegex) throws Exception {
1565       for (;;) {
1566          if (c.isAssignableFrom(t.getClass())) {
1567             if (messageRegex != null && !Pattern.matches(messageRegex, t.getMessage())) {
1568                throw new RuntimeException(String.format("Exception message '%s' does not match regex '%s'", t.getMessage(), messageRegex));
1569             }
1570             return;
1571          }
1572          Throwable cause = t.getCause();
1573          if (cause == null || cause == t) {
1574             throw new RuntimeException("Cannot find a cause of type " + c.getName(), cause);
1575          } else {
1576             t = cause;
1577          }
1578       }
1579    }

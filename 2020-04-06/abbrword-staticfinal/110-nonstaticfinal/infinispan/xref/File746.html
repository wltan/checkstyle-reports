23  package org.jboss.as.clustering.infinispan.subsystem;
24  
25  import static org.jboss.as.controller.descriptions.ModelDescriptionConstants.ADD;
26  
27  import org.jboss.as.controller.AttributeDefinition;
28  import org.jboss.as.controller.ObjectListAttributeDefinition;
29  import org.jboss.as.controller.ObjectTypeAttributeDefinition;
30  import org.jboss.as.controller.OperationDefinition;
31  import org.jboss.as.controller.OperationStepHandler;
32  import org.jboss.as.controller.PathElement;
33  import org.jboss.as.controller.SimpleAttributeDefinition;
34  import org.jboss.as.controller.SimpleAttributeDefinitionBuilder;
35  import org.jboss.as.controller.SimpleOperationDefinitionBuilder;
36  import org.jboss.as.controller.client.helpers.MeasurementUnit;
37  import org.jboss.as.controller.registry.AttributeAccess;
38  import org.jboss.as.controller.registry.ManagementResourceRegistration;
39  import org.jboss.as.controller.registry.OperationEntry;
40  import org.jboss.dmr.ModelNode;
41  import org.jboss.dmr.ModelType;
42  
43  /**
44   * Resource description for the addressable resource /subsystem=infinispan/cache-container=X/cache=Y/remote-store=REMOTE_STORE
45   *
46   * @author Richard Achmatowicz (c) 2011 Red Hat Inc.
47   */
48  public class RemoteStoreResource extends BaseStoreResource {
49  
50      public static final PathElement REMOTE_STORE_PATH = PathElement.pathElement(ModelKeys.REMOTE_STORE);
51  
52      // attributes
53      static final SimpleAttributeDefinition CACHE =
54              new SimpleAttributeDefinitionBuilder(ModelKeys.CACHE, ModelType.STRING, true)
55                      .setXmlName(Attribute.CACHE.getLocalName())
56                      .setAllowExpression(true)
57                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
58                      .build();
59      static final SimpleAttributeDefinition HOTROD_WRAPPING =
60              new SimpleAttributeDefinitionBuilder(ModelKeys.HOTROD_WRAPPING, ModelType.BOOLEAN, true)
61                      .setXmlName(Attribute.HOTROD_WRAPPING.getLocalName())
62                      .setAllowExpression(true)
63                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
64                      .setDefaultValue(new ModelNode().set(false))
65                      .build();
66      static final SimpleAttributeDefinition RAW_VALUES =
67              new SimpleAttributeDefinitionBuilder(ModelKeys.RAW_VALUES, ModelType.BOOLEAN, true)
68                      .setXmlName(Attribute.RAW_VALUES.getLocalName())
69                      .setAllowExpression(false)
70                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
71                      .setDefaultValue(new ModelNode().set(false))
72                      .build();
73      static final SimpleAttributeDefinition TCP_NO_DELAY =
74              new SimpleAttributeDefinitionBuilder(ModelKeys.TCP_NO_DELAY, ModelType.BOOLEAN, true)
75                      .setXmlName(Attribute.TCP_NO_DELAY.getLocalName())
76                      .setAllowExpression(true)
77                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
78                      .setDefaultValue(new ModelNode().set(true))
79                      .build();
80      static final SimpleAttributeDefinition SOCKET_TIMEOUT =
81              new SimpleAttributeDefinitionBuilder(ModelKeys.SOCKET_TIMEOUT, ModelType.LONG, true)
82                      .setXmlName(Attribute.SOCKET_TIMEOUT.getLocalName())
83                      .setMeasurementUnit(MeasurementUnit.MILLISECONDS)
84                      .setAllowExpression(true)
85                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
86                      .setDefaultValue(new ModelNode().set(60000))
87                      .build();
88  
89      static final SimpleAttributeDefinition OUTBOUND_SOCKET_BINDING = new SimpleAttributeDefinition("outbound-socket-binding", ModelType.STRING, true);
90  
91      static final SimpleAttributeDefinition NAME =
92              new SimpleAttributeDefinitionBuilder(BaseStoreResource.NAME)
93                     .setDefaultValue(new ModelNode().set(ModelKeys.REMOTE_STORE_NAME))
94                     .build();
95  
96      static final ObjectTypeAttributeDefinition REMOTE_SERVER = ObjectTypeAttributeDefinition.
97              Builder.of(ModelKeys.REMOTE_SERVER, OUTBOUND_SOCKET_BINDING).
98              setAllowNull(true).
99              setSuffix("remote-server").
100             build();
101 
102     static final ObjectListAttributeDefinition REMOTE_SERVERS = ObjectListAttributeDefinition.Builder.of(ModelKeys.REMOTE_SERVERS, REMOTE_SERVER).
103             setAllowNull(false).
104             build();
105 
106     static final AttributeDefinition[] REMOTE_STORE_ATTRIBUTES = {CACHE, HOTROD_WRAPPING, TCP_NO_DELAY, RAW_VALUES, SOCKET_TIMEOUT, REMOTE_SERVERS};
107 
108     // operations
109     private static final OperationDefinition REMOTE_STORE_ADD_DEFINITION = new SimpleOperationDefinitionBuilder(ADD, InfinispanExtension.getResourceDescriptionResolver(ModelKeys.REMOTE_STORE))
110         .setParameters(COMMON_STORE_PARAMETERS)
111         .addParameter(CACHE)
112         .addParameter(HOTROD_WRAPPING)
113         .addParameter(RAW_VALUES)
114         .addParameter(TCP_NO_DELAY)
115         .addParameter(SOCKET_TIMEOUT)
116         .addParameter(REMOTE_SERVERS)
117         .setAttributeResolver(InfinispanExtension.getResourceDescriptionResolver(ModelKeys.REMOTE_STORE))
118         .build();
119 
120     public RemoteStoreResource(CacheResource cacheResource) {
121         super(REMOTE_STORE_PATH, ModelKeys.REMOTE_STORE, cacheResource, REMOTE_STORE_ATTRIBUTES);
122     }
123 
124     // override the add operation to provide a custom definition (for the optional PROPERTIES parameter to add())
125     @Override
126     protected void registerAddOperation(final ManagementResourceRegistration registration, final OperationStepHandler handler, OperationEntry.Flag... flags) {
127         registration.registerOperationHandler(REMOTE_STORE_ADD_DEFINITION, handler);
128     }
129 }

19  package org.infinispan.server.endpoint.subsystem;
20  
21  import java.util.List;
22  
23  import org.jboss.as.controller.AbstractAddStepHandler;
24  import org.jboss.as.controller.AttributeDefinition;
25  import org.jboss.as.controller.OperationContext;
26  import org.jboss.as.controller.OperationFailedException;
27  import org.jboss.as.controller.PathAddress;
28  import org.jboss.as.controller.ServiceVerificationHandler;
29  import org.jboss.as.controller.registry.Resource;
30  import org.jboss.as.controller.services.path.PathManager;
31  import org.jboss.as.controller.services.path.PathManagerService;
32  import org.jboss.dmr.ModelNode;
33  import org.jboss.msc.service.ServiceBuilder;
34  import org.jboss.msc.service.ServiceController;
35  
36  /**
37   * RestSubsystemAdd.
38   *
39   * @author Tristan Tarrant
40   * @since 5.1
41   */
42  class RestSubsystemAdd extends AbstractAddStepHandler {
43     static final RestSubsystemAdd INSTANCE = new RestSubsystemAdd();
44  
45     @Override
46     protected void performRuntime(OperationContext context, ModelNode operation, ModelNode model, ServiceVerificationHandler verificationHandler, List<ServiceController<?>> newControllers)
47              throws OperationFailedException {
48        // Read the full model
49        ModelNode config = Resource.Tools.readModel(context.readResource(PathAddress.EMPTY_ADDRESS));
50  
51        // Create the service
52        final RestService service = new RestService(config);
53  
54        // Setup the various dependencies with injectors and install the service
55        ServiceBuilder<?> builder = context.getServiceTarget().addService(EndpointUtils.getServiceName(operation, "rest"), service);
56        EndpointUtils.addCacheContainerDependency(builder, service.getCacheContainerName(), service.getCacheManager());
57        EndpointUtils.addSocketBindingDependency(builder, getSocketBindingName(operation), service.getSocketBinding());
58        builder
59           .addDependency(PathManagerService.SERVICE_NAME, PathManager.class, service.getPathManagerInjector());
60        if (service.getSecurityDomain()!=null) {
61           EndpointUtils.addSecurityDomainDependency(builder, service.getSecurityDomain(), service.getSecurityDomainContextInjector());
62        }
63        builder.addListener(verificationHandler);
64        builder.setInitialMode(ServiceController.Mode.ACTIVE);
65        builder.install();
66     }
67  
68     protected String getSocketBindingName(ModelNode config) {
69        return config.hasDefined(ModelKeys.SOCKET_BINDING) ? config.get(ModelKeys.SOCKET_BINDING).asString() : null;
70     }
71  
72     @Override
73     protected void populateModel(ModelNode source, ModelNode target) throws OperationFailedException {
74        populate(source, target);
75     }
76  
77     private static void populate(ModelNode source, ModelNode target) throws OperationFailedException {
78        for(AttributeDefinition attr : ProtocolServerConnectorResource.COMMON_CONNECTOR_ATTRIBUTES) {
79           attr.validateAndSet(source, target);
80        }
81        for(AttributeDefinition attr : RestConnectorResource.REST_ATTRIBUTES) {
82           attr.validateAndSet(source, target);
83        }
84     }
85  
86     @Override
87     protected boolean requiresRuntimeVerification() {
88        return false;
89     }
90  }

11  /**
12   * The 2nd phase command of 2PC, used when a transaction must be aborted. This implementation is used when Total Order
13   * based protocol is used
14   *
15   * @author Pedro Ruivo
16   * @since 5.3
17   */
18  public class TotalOrderRollbackCommand extends RollbackCommand {
19  
20     public static final byte COMMAND_ID = 37;
21     private static final Log log = LogFactory.getLog(TotalOrderRollbackCommand.class);
22  
23     public TotalOrderRollbackCommand(String cacheName, GlobalTransaction globalTransaction) {
24        super(cacheName, globalTransaction);
25     }
26  
27     public TotalOrderRollbackCommand(String cacheName) {
28        super(cacheName);
29     }
30  
31     private TotalOrderRollbackCommand() {
32        super(null); // For command id uniqueness test
33     }
34  
35     @Override
36     public byte getCommandId() {
37        return COMMAND_ID;
38     }
39  
40     @Override
41     protected RemoteTransaction getRemoteTransaction() {
42        return txTable.getOrCreateRemoteTransaction(globalTx, null);
43     }
44  }

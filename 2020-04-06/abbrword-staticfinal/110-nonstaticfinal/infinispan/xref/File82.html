1   package org.infinispan.client.hotrod;
2   
3   import org.infinispan.client.hotrod.test.MultiHotRodServersTest;
4   import org.infinispan.configuration.cache.CacheMode;
5   import org.infinispan.configuration.cache.ConfigurationBuilder;
6   import org.infinispan.persistence.dummy.DummyInMemoryStoreConfigurationBuilder;
7   import org.testng.annotations.Test;
8   
9   import static org.testng.AssertJUnit.assertEquals;
10  
11  @Test(groups = "functional", testName = "client.hotrod.SizeTest")
12  public class SizeTest extends MultiHotRodServersTest {
13  
14     static final int NUM_SERVERS = 3;
15     static final int SIZE = 20;
16  
17     @Override
18     protected void createCacheManagers() throws Throwable {
19        createHotRodServers(NUM_SERVERS, new ConfigurationBuilder());
20     }
21  
22     public void testLocalCacheSize() {
23        String cacheName = "local-size";
24        defineInAll(cacheName, new ConfigurationBuilder());
25        // Create a brand new client so that as a local cache it does not do load balancing
26        // This is important for size assertions since there's data is not clustered
27        RemoteCache<Integer, Integer> remote = createClient(0).getCache(cacheName);
28        populateCache(remote);
29        assertEquals(SIZE, remote.size());
30     }
31  
32     public void testReplicatedCacheSize() {
33        String cacheName = "replicated-size";
34        defineInAll(cacheName, getDefaultClusteredCacheConfig(CacheMode.REPL_SYNC, false));
35        populateCache(cacheName);
36        assertEquals(SIZE, clients.get(0).getCache(cacheName).size());
37     }
38  
39     public void testDistributeCacheSize() {
40        String cacheName = "distributed-size";
41        defineInAll(cacheName, getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, false));
42        populateCache(cacheName);
43        assertEquals(SIZE, clients.get(0).getCache(cacheName).size());
44     }
45  
46     public void testPersistentDistributedCacheSize() {
47        String cacheName = "persistent-distributed-size";
48        ConfigurationBuilder builder = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, false);
49        builder.eviction().maxEntries(1);
50        builder.persistence()
51              .passivation(true)
52              .addStore(DummyInMemoryStoreConfigurationBuilder.class)
53              .storeName(getClass().getName())
54              .shared(true);
55        defineInAll(cacheName, builder);
56        populateCache(cacheName);
57        assertEquals(SIZE, clients.get(0).getCache(cacheName).size());
58     }
59  
60     private void populateCache(String cacheName) {
61        for (int i = 0; i < SIZE; i++)
62           clients.get(i % NUM_SERVERS).getCache(cacheName).put(i, i);
63     }
64  
65     private void populateCache(RemoteCache<Integer, Integer> remote) {
66        for (int i = 0; i < SIZE; i++)
67           remote.put(i, i);
68     }
69  
70  }

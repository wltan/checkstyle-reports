1   package org.infinispan.it.osgi.util;
2   
3   public class PaxURLUtils {
4      public static final String PROP_PAX_URL_LOCAL_REPO = "org.ops4j.pax.url.mvn.localRepository";
5   
6      private static final String JAVA_URL_HANDLERS_PROPERTY = "java.protocol.handler.pkgs";
7      private static final String PAX_URL_PACKAGE = "org.ops4j.pax.url";
8   
9      public static void registerURLHandlers() {
10        String protocolHandlers = System.getProperty(JAVA_URL_HANDLERS_PROPERTY);
11  
12        if (protocolHandlers == null) {
13           System.setProperty(JAVA_URL_HANDLERS_PROPERTY, PAX_URL_PACKAGE);
14        } else if (!protocolHandlers.contains(PAX_URL_PACKAGE)) {
15           System.setProperty(JAVA_URL_HANDLERS_PROPERTY, String.format("%s|%s",protocolHandlers, PAX_URL_PACKAGE));
16        }
17     }
18  
19     public static void configureLocalMavenRepo() {
20        String localRepo = null;
21        try {
22           localRepo = MavenUtils.getLocalRepository();
23        } catch (Exception ex) {
24        }
25        if (localRepo == null) {
26           return;
27        }
28        System.setProperty(PROP_PAX_URL_LOCAL_REPO, localRepo);
29     }
30  }

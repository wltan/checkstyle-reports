18  /**
19   * Verifies the Flags affect both local and remote nodes.
20   * 
21   * @author Sanne Grinovero <sanne@hibernate.org> (C) 2011 Red Hat Inc.
22   * @since 4.2.1
23   */
24  @Test(groups = "functional", testName = FlagsReplicationTest.TEST_NAME)
25  public class FlagsReplicationTest extends BaseDistFunctionalTest<Object, String> {
26     
27     static final String TEST_NAME = "replication.FlagsReplicationTest";
28     static final String DATA_PROVIDER = TEST_NAME + ".dataprovider";
29     private ExecutorService threadPool;
30     
31     private final Integer one = 1;
32     private final String key = TEST_NAME;
33     
34     public FlagsReplicationTest() {
35        tx = true;
36        cacheName = TEST_NAME;
37        cleanup = CleanupPhase.AFTER_METHOD;
38        lockingMode = LockingMode.PESSIMISTIC;
39        lockTimeout = 1;
40     }
41     
42     @DataProvider(name = DATA_PROVIDER)
43     public Object[][] createTestConfigurations() {
44        return new Object[][] {
45                 { true,  true  },
46                 { false, false },
47                 { false, true  },
48                 { true,  false },
49           };
50     }
51  
52     @Test(dataProvider = DATA_PROVIDER)
53     public void testScenario(boolean cache1IsOwner, boolean cache2IsOwner) throws Throwable {
54        log.tracef("Start cache1IsOwner = %s, cache2IsOwner %s", cache1IsOwner, cache2IsOwner);
55        AdvancedCache cache1 = (cache1IsOwner ? getFirstOwner(key) : getFirstNonOwner(key)).getAdvancedCache();
56        AdvancedCache cache2 = (cache2IsOwner ? getFirstOwner(key) : getFirstNonOwner(key)).getAdvancedCache();
57  
58        assert null == cache1.put(key, one);
59  
60        haveSecondaryThreadTakeLock(cache2);
61  
62        cache1.getTransactionManager().begin();
63        boolean locked = cache1.withFlags(Flag.ZERO_LOCK_ACQUISITION_TIMEOUT, Flag.FAIL_SILENTLY).lock(key);
64        assert !locked;
65        Object removed = cache1.withFlags(Flag.SKIP_LOCKING).remove(key);
66        assert one.equals(removed);
67  
68        haveSecondaryThreadReleaseLock(cache2);
69        cache1.getTransactionManager().commit();
70        assert null == cache2.get(key);
71        log.tracef("End cache1IsOwner = %s, cache2IsOwner %s", cache1IsOwner, cache2IsOwner);
72     }
73  
74     private void haveSecondaryThreadTakeLock(final AdvancedCache viaCache) throws InterruptedException, ExecutionException {
75        AtomicBoolean noerrors = new AtomicBoolean(true);
76        Future<?> submit = threadPool.submit(new LockingThread(viaCache, noerrors));
77        submit.get(); //wait to be done
78        assert noerrors.get();
79     }
80     
81     private void haveSecondaryThreadReleaseLock(final AdvancedCache viaCache) throws InterruptedException, ExecutionException {
82        AtomicBoolean noerrors = new AtomicBoolean(true);
83        Future<?> submit = threadPool.submit(new CommitThread(viaCache, noerrors));
84        submit.get(); //wait to be done
85        assert noerrors.get();
86     }
87     
88     private class LockingThread implements Runnable {
89        
90        private final AdvancedCache cache;
91        private final AtomicBoolean allok;
92  
93        LockingThread(AdvancedCache cache, AtomicBoolean allok) {
94           this.cache = cache;
95           this.allok = allok;
96        }
97  
98        @Override
99        public void run() {
100          try {
101             log.trace("About to try to acquire a lock.");
102             cache.getTransactionManager().begin();
103             if (! cache.lock(key)) {
104                allok.set(false);
105                log.trace("Could not acquire lock");
106             }
107          } catch (Throwable e) {
108             log.trace("Error", e);
109             allok.set(false);
110          }
111       }
112    }
113    
114    static private class CommitThread implements Runnable {
115       
116       private final AdvancedCache cache;
117       private final AtomicBoolean allok;
118 
119       CommitThread(AdvancedCache cache, AtomicBoolean allok) {
120          this.cache = cache;
121          this.allok = allok;
122       }
123 
124       @Override
125       public void run() {
126          try {
127             cache.getTransactionManager().commit();
128          } catch (Throwable e) {
129             allok.set(false);
130          }
131       }
132    }
133    
134    @BeforeClass
135    protected void startThreadPool() {
136       threadPool = Executors.newFixedThreadPool(1);
137    }
138    
139    @AfterClass
140    protected void stopThreadPool() {
141       threadPool.shutdownNow();
142    }

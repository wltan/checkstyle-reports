26  package com.sun.tools.jdi;
27  
28  import java.io.IOException;
29  import java.util.Map;
30  import java.util.Properties;
31  
32  import com.sun.jdi.Bootstrap;
33  import com.sun.jdi.VirtualMachine;
34  import com.sun.jdi.connect.AttachingConnector;
35  import com.sun.jdi.connect.Connector;
36  import com.sun.jdi.connect.IllegalConnectorArgumentsException;
37  import com.sun.jdi.connect.Transport;
38  import com.sun.jdi.connect.spi.Connection;
39  import com.sun.jdi.connect.spi.TransportService;
40  
41  /*
42   * An AttachingConnector that connects to a debuggee by specifying the process
43   * id (pid) as the connector argument. If the process is a debuggee listening
44   * on a transport address then this connector reads the transport address
45   * and attempts to attach to it using the appropriate transport.
46   */
47  
48  public class ProcessAttachingConnector
49          extends ConnectorImpl implements AttachingConnector
50  {
51      /*
52       * The arguments that this connector supports
53       */
54      static final String ARG_PID = "pid";
55      static final String ARG_TIMEOUT = "timeout";
56  
57      com.sun.tools.attach.VirtualMachine vm;
58      Transport transport;
59  
60      public ProcessAttachingConnector() {
61          addStringArgument(
62              ARG_PID,
63              getString("process_attaching.pid.label"),
64              getString("process_attaching.pid"),
65              "",
66              true);
67  
68          addIntegerArgument(
69              ARG_TIMEOUT,
70              getString("generic_attaching.timeout.label"),       // use generic keys to keep
71              getString("generic_attaching.timeout"),             // resource bundle small
72              "",
73              false,
74              0, Integer.MAX_VALUE);
75  
76          transport = new Transport() {
77              public String name() {
78                  return "local";
79              }
80          };
81      }
82  
83      /**
84       * Attach to a target VM using the specified address and Connector arguments.
85       */
86      public VirtualMachine attach(Map<String,? extends Connector.Argument> args)
87                  throws IOException, IllegalConnectorArgumentsException
88      {
89          String pid = argument(ARG_PID, args).value();
90          String t = argument(ARG_TIMEOUT, args).value();
91          int timeout = 0;
92          if (t.length() > 0) {
93              timeout = Integer.decode(t).intValue();
94          }
95  
96          // Use Attach API to attach to target VM and read value of
97          // sun.jdwp.listenAddress property.
98  
99          String address = null;
100         com.sun.tools.attach.VirtualMachine vm = null;
101         try {
102             vm = com.sun.tools.attach.VirtualMachine.attach(pid);
103             Properties props = vm.getAgentProperties();
104             address = props.getProperty("sun.jdwp.listenerAddress");
105         } catch (Exception x) {
106             throw new IOException(x.getMessage());
107         } finally {
108             if (vm != null) vm.detach();
109         }
110 
111         // check that the property value is formatted correctly
112 
113         if (address == null) {
114             throw new IOException("Not a debuggee, or not listening for debugger to attach");
115         }
116         int pos = address.indexOf(':');
117         if (pos < 1) {
118             throw new IOException("Unable to determine transport endpoint");
119         }
120 
121         // parse into transport library name and address
122 
123         final String lib = address.substring(0, pos);
124         address = address.substring(pos+1, address.length());
125 
126         TransportService ts = null;
127         if (lib.equals("dt_socket")) {
128             ts = new SocketTransportService();
129         } else {
130             if (lib.equals("dt_shmem")) {
131                 try {
132                     Class<?> c = Class.forName("com.sun.tools.jdi.SharedMemoryTransportService");
133                     @SuppressWarnings("deprecation")
134                     Object tmp = c.newInstance();
135                     ts = (TransportService)tmp;
136                 } catch (Exception x) { }
137             }
138         }
139         if (ts == null) {
140             throw new IOException("Transport " + lib + " not recognized");
141         }
142 
143         // connect to the debuggee
144 
145         Connection connection = ts.attach(address, timeout, 0);
146         return Bootstrap.virtualMachineManager().createVirtualMachine(connection);
147     }
148 
149     public String name() {
150         return "com.sun.jdi.ProcessAttach";
151     }
152 
153     public String description() {
154         return getString("process_attaching.description");
155     }
156 
157     public Transport transport() {
158         if (transport == null) {
159             return new Transport() {
160                 public String name() {
161                     return "local";
162                 }
163             };
164         }
165         return transport;
166     }
167 }

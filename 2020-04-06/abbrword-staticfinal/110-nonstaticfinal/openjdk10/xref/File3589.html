24  import javax.net.ssl.SSLContext;
25  import javax.net.ssl.SSLEngine;
26  import javax.net.ssl.SSLException;
27  
28  /**
29   * Testing SSLEngines handshake using each of the supported cipher suites with
30   * different maximum fragment length. Testing of MFLN extension.
31   */
32  public class MFLNTest extends SSLEngineTestCase {
33  
34      public static void main(String[] args) {
35          setUpAndStartKDCIfNeeded();
36          System.setProperty("jsse.enableMFLNExtension", "true");
37          for (int mfl = 4096; mfl >= 256; mfl /= 2) {
38              System.out.println("=============================================="
39                      + "==============");
40              System.out.printf("Testsing DTLS handshake with MFL = %d%n", mfl);
41              MFLNTest test = new MFLNTest(mfl);
42              test.runTests();
43          }
44      }
45  
46      protected MFLNTest(int maxPacketSize) {
47          super(maxPacketSize);
48      }
49  
50      @Override
51      protected void testOneCipher(String cipher) throws SSLException {
52          SSLContext context = getContext();
53          int maxPacketSize = getMaxPacketSize();
54          boolean useSNI = !TEST_MODE.equals("norm");
55          SSLEngine clientEngine = getClientSSLEngine(context, useSNI);
56          SSLEngine serverEngine = getServerSSLEngine(context, useSNI);
57          clientEngine.setEnabledCipherSuites(new String[]{cipher});
58          serverEngine.setEnabledCipherSuites(new String[]{cipher});
59          serverEngine.setNeedClientAuth(!cipher.contains("anon"));
60          doHandshake(clientEngine, serverEngine, maxPacketSize,
61                  HandshakeMode.INITIAL_HANDSHAKE);
62      }
63  }

541      * correctness for a String containing a substantial

545     private static void roundTrip(String testStr, byte[] expectBytes,
546                                   String csName)
547     throws Exception {
548         byte[] encodedBytes = testStr.getBytes(csName);
549 
550         if (encodedBytes.length != expectBytes.length) {
551             throw new Exception(csName + " Encoder error");
552         }
553 
554         for (int i = 0; i < expectBytes.length; i++) {
555             if (encodedBytes[i] != expectBytes[i])  {
556                 throw new Exception(csName + " Encoder error");
557             }
558         }
559         String decoded = new String(encodedBytes, csName);
560 
561         if (!decoded.equals(testStr)) {
562             throw new Exception(csName + " Decoder error");
563         }
564         String decoded2 = new String(repeatingEscapes, csName);
565         if (!decoded2.equals(MIXEDCONTENT)) {
566             throw new Exception(csName + " Decoder error");
567         }
568      }

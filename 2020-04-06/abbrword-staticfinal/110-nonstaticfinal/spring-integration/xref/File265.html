17  package org.springframework.integration.mongodb.metadata;
18  
19  import java.util.HashMap;
20  import java.util.Map;
21  
22  import org.bson.Document;
23  
24  import org.springframework.data.mongodb.MongoDatabaseFactory;
25  import org.springframework.data.mongodb.core.FindAndModifyOptions;
26  import org.springframework.data.mongodb.core.MongoTemplate;
27  import org.springframework.data.mongodb.core.query.Criteria;
28  import org.springframework.data.mongodb.core.query.Query;
29  import org.springframework.data.mongodb.core.query.Update;
30  import org.springframework.integration.metadata.ConcurrentMetadataStore;
31  import org.springframework.util.Assert;
32  
33  /**
34   * MongoDbMetadataStore implementation of {@link ConcurrentMetadataStore}.
35   * Use this {@link org.springframework.integration.metadata.MetadataStore} to
36   * achieve meta-data persistence shared across application instances and
37   * restarts.
38   *
39   * @author Senthil Arumugam, Samiraj Panneer Selvam
40   * @author Artem Bilan
41   * @author Gary Russell
42   *
43   * @since 4.2
44   *
45   */
46  public class MongoDbMetadataStore implements ConcurrentMetadataStore {
47  
48  	private static final String KEY_MUST_NOT_BE_EMPTY = "'key' must not be empty.";
49  
50  	private static final String DEFAULT_COLLECTION_NAME = "metadataStore";
51  
52  	private static final String ID_FIELD = "_id";
53  
54  	private static final String VALUE = "value";
55  
56  	private final MongoTemplate template;
57  
58  	private final String collectionName;
59  
60  	/**
61  	 * Configure the MongoDbMetadataStore by provided {@link MongoDatabaseFactory} and
62  	 * default collection name - {@link #DEFAULT_COLLECTION_NAME}.
63  	 * @param factory the mongodb factory
64  	 */
65  	public MongoDbMetadataStore(MongoDatabaseFactory factory) {
66  		this(factory, DEFAULT_COLLECTION_NAME);
67  	}
68  
69  	/**
70  	 * Configure the MongoDbMetadataStore by provided {@link MongoDatabaseFactory} and
71  	 * collection name
72  	 * @param factory the mongodb factory
73  	 * @param collectionName the collection name where it persists the data
74  	 */
75  	public MongoDbMetadataStore(MongoDatabaseFactory factory, String collectionName) {
76  		this(new MongoTemplate(factory), collectionName);
77  	}
78  
79  	/**
80  	 * Configure the MongoDbMetadataStore by provided {@link MongoTemplate} and
81  	 * default collection name - {@link #DEFAULT_COLLECTION_NAME}.
82  	 * @param template the mongodb template
83  	 */
84  	public MongoDbMetadataStore(MongoTemplate template) {
85  		this(template, DEFAULT_COLLECTION_NAME);
86  	}
87  
88  	/**
89  	 * Configure the MongoDbMetadataStore by provided {@link MongoTemplate} and collection name.
90  	 * @param template the mongodb template
91  	 * @param collectionName the collection name where it persists the data
92  	 */
93  	public MongoDbMetadataStore(MongoTemplate template, String collectionName) {
94  		Assert.notNull(template, "'template' must not be null.");
95  		Assert.hasText(collectionName, "'collectionName' must not be empty.");
96  		this.template = template;
97  		this.collectionName = collectionName;
98  	}
99  
100 	/**
101 	 * Store a metadata {@code value} under provided {@code key} to the configured
102 	 * {@link #collectionName}.
103 	 * <p>
104 	 * If a document does not exist with the specified {@code key}, the method performs an {@code insert}.
105 	 * If a document exists with the specified {@code key}, the method performs an {@code update}.
106 	 * @param key the metadata entry key
107 	 * @param value the metadata entry value
108 	 * @see MongoTemplate#execute(String, org.springframework.data.mongodb.core.CollectionCallback)
109 	 */
110 	@Override
111 	public void put(String key, String value) {
112 		Assert.hasText(key, KEY_MUST_NOT_BE_EMPTY);
113 		Assert.hasText(value, "'value' must not be empty.");
114 		final Map<String, Object> entry = new HashMap<>();
115 		entry.put(ID_FIELD, key);
116 		entry.put(VALUE, value);
117 		this.template.save(new Document(entry), this.collectionName);
118 	}
119 
120 	/**
121 	 * Get the {@code value} for the provided {@code key} performing {@code findOne} MongoDB operation.
122 	 * @param key the metadata entry key
123 	 * @return the metadata entry value or null if doesn't exist.
124 	 * @see MongoTemplate#findOne(Query, Class, String)
125 	 */
126 	@Override
127 	public String get(String key) {
128 		Assert.hasText(key, KEY_MUST_NOT_BE_EMPTY);
129 		Query query = new Query(Criteria.where(ID_FIELD).is(key));
130 		query.fields().exclude(ID_FIELD);
131 		@SuppressWarnings("unchecked")
132 		Map<String, String> result = this.template.findOne(query, Map.class, this.collectionName);
133 		return result == null ? null : result.get(VALUE);
134 
135 	}
136 
137 	/**
138 	 * Remove the metadata entry for the provided {@code key} and return its {@code value}, if any,
139 	 * using {@code findAndRemove} MongoDB operation.
140 	 * @param key the metadata entry key
141 	 * @return the metadata entry value or null if doesn't exist.
142 	 * @see MongoTemplate#findAndRemove(Query, Class, String)
143 	 */
144 	@Override
145 	public String remove(String key) {
146 		Assert.hasText(key, KEY_MUST_NOT_BE_EMPTY);
147 		Query query = new Query(Criteria.where(ID_FIELD).is(key));
148 		query.fields().exclude(ID_FIELD);
149 		@SuppressWarnings("unchecked")
150 		Map<String, String> result = this.template.findAndRemove(query, Map.class, this.collectionName);
151 		return result == null ? null : result.get(VALUE);
152 	}
153 
154 	/**
155 	 * If the specified key is not already associated with a value, associate it with the given value.
156 	 * This is equivalent to
157 	 * <pre> {@code
158 	 * if (!map.containsKey(key))
159 	 *   return map.put(key, value);
160 	 * else
161 	 *   return map.get(key);
162 	 * }</pre>
163 	 * except that the action is performed atomically.
164 	 * <p>
165 	 * @param key the metadata entry key
166 	 * @param value the metadata entry value to store
167 	 * @return null if successful, the old value otherwise.
168 	 * @see java.util.concurrent.ConcurrentMap#putIfAbsent(Object, Object)
169 	 */
170 	@Override
171 	public String putIfAbsent(String key, String value) {
172 		Assert.hasText(key, KEY_MUST_NOT_BE_EMPTY);
173 		Assert.hasText(value, "'value' must not be empty.");
174 
175 		Query query = new Query(Criteria.where(ID_FIELD).is(key));
176 		query.fields().exclude(ID_FIELD);
177 		@SuppressWarnings("unchecked")
178 		Map<String, String> result = this.template.findAndModify(query, new Update().setOnInsert(VALUE, value),
179 				new FindAndModifyOptions().upsert(true), Map.class, this.collectionName);
180 		return result == null ? null : result.get(VALUE);
181 	}
182 
183 	/**
184 	 * Replace an existing metadata entry {@code value} with a new one. Otherwise does nothing.
185 	 * Performs {@code updateFirst} if a document for the provided {@code key} and {@code oldValue}
186 	 * exists in the {@link #collectionName}.
187 	 * @param key the metadata entry key
188 	 * @param oldValue the metadata entry old value to replace
189 	 * @param newValue the metadata entry new value to put
190 	 * @return {@code true} if replace was successful, {@code false} otherwise.
191 	 * @see MongoTemplate#updateFirst(Query, org.springframework.data.mongodb.core.query.UpdateDefinition, String)
192 	 */
193 	@Override
194 	public boolean replace(String key, String oldValue, String newValue) {
195 		Assert.hasText(key, KEY_MUST_NOT_BE_EMPTY);
196 		Assert.hasText(oldValue, "'oldValue' must not be empty.");
197 		Assert.hasText(newValue, "'newValue' must not be empty.");
198 		Query query = new Query(Criteria.where(ID_FIELD).is(key).and(VALUE).is(oldValue));
199 		return this.template.updateFirst(query, Update.update(VALUE, newValue), this.collectionName)
200 				.getModifiedCount() > 0;
201 	}

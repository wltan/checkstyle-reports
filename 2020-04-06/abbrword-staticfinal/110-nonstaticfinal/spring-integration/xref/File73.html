194 	@Override
195 	protected void onInit() {
196 		synchronized (this.initializationMonitor) {
197 			if (this.initialized) {
198 				return;
199 			}
200 			Assert.notNull(this.trigger, "Trigger is required");
201 			if (this.taskExecutor != null && !(this.taskExecutor instanceof ErrorHandlingTaskExecutor)) {
202 				if (this.errorHandler == null) {
203 					this.errorHandler = ChannelUtils.getErrorHandler(getBeanFactory());
204 					this.errorHandlerIsDefault = true;
205 				}
206 				this.taskExecutor = new ErrorHandlingTaskExecutor(this.taskExecutor, this.errorHandler);
207 			}
208 			if (this.transactionSynchronizationFactory == null && this.adviceChain != null &&
209 					this.adviceChain.stream().anyMatch(TransactionInterceptor.class::isInstance)) {
210 				this.transactionSynchronizationFactory = new PassThroughTransactionSynchronizationFactory();
211 			}
212 			this.initialized = true;
213 		}
214 		try {
215 			super.onInit();
216 		}
217 		catch (Exception e) {
218 			throw new BeanInitializationException("Cannot initialize: " + this, e);
219 		}
220 	}

224 	@Override // guarded by super#lifecycleLock
225 	protected void doStart() {
226 		if (!this.initialized) {
227 			onInit();
228 		}
229 
230 		this.pollingTask = createPollingTask();
231 
232 		if (isReactive()) {
233 			this.pollingFlux = createFluxGenerator();
234 		}
235 		else {
236 			Assert.state(getTaskScheduler() != null, "unable to start polling, no taskScheduler available");
237 
238 			this.runningTask =
239 					getTaskScheduler()
240 							.schedule(createPoller(), this.trigger);
241 		}
242 	}
243 
244 	@SuppressWarnings("unchecked")
245 	private Callable<Message<?>> createPollingTask() {
246 		List<Advice> receiveOnlyAdviceChain = null;
247 		if (!CollectionUtils.isEmpty(this.adviceChain)) {
248 			receiveOnlyAdviceChain = this.adviceChain.stream()
249 					.filter(this::isReceiveOnlyAdvice)
250 					.collect(Collectors.toList());
251 		}
252 
253 		Callable<Message<?>> task = this::doPoll;
254 
255 		List<Advice> advices = this.adviceChain;
256 		if (!CollectionUtils.isEmpty(advices)) {
257 			ProxyFactory proxyFactory = new ProxyFactory(task);
258 			if (!CollectionUtils.isEmpty(advices)) {
259 				advices.stream()
260 						.filter(advice -> !isReceiveOnlyAdvice(advice))
261 						.forEach(proxyFactory::addAdvice);
262 			}
263 			task = (Callable<Message<?>>) proxyFactory.getProxy(this.beanClassLoader);
264 		}
265 		if (!CollectionUtils.isEmpty(receiveOnlyAdviceChain)) {
266 			applyReceiveOnlyAdviceChain(receiveOnlyAdviceChain);
267 		}
268 
269 		return task;
270 	}
271 
272 	private Runnable createPoller() {
273 		return () ->
274 				this.taskExecutor.execute(() -> {
275 					int count = 0;
276 					while (this.initialized && (this.maxMessagesPerPoll <= 0 || count < this.maxMessagesPerPoll)) {
277 						if (pollForMessage() == null) {
278 							break;
279 						}
280 						count++;
281 					}
282 				});
283 	}

285 	private Flux<Message<?>> createFluxGenerator() {
286 		SimpleTriggerContext triggerContext = new SimpleTriggerContext();
287 
288 		return Flux
289 				.<Duration>generate(sink -> {
290 					Date date = this.trigger.nextExecutionTime(triggerContext);
291 					if (date != null) {
292 						triggerContext.update(date, null, null);
293 						long millis = date.getTime() - System.currentTimeMillis();
294 						sink.next(Duration.ofMillis(millis));
295 					}
296 					else {
297 						sink.complete();
298 					}
299 				})
300 				.concatMap(duration ->
301 						Mono.delay(duration)
302 								.doOnNext(l ->
303 										triggerContext.update(triggerContext.lastScheduledExecutionTime(),
304 												new Date(), null))
305 								.flatMapMany(l ->
306 										Flux
307 												.<Message<?>>generate(fluxSink -> {
308 													Message<?> message = pollForMessage();
309 													if (message != null) {
310 														fluxSink.next(message);
311 													}
312 													else {
313 														fluxSink.complete();
314 													}
315 												})
316 												.take(this.maxMessagesPerPoll)
317 												.subscribeOn(Schedulers.fromExecutor(this.taskExecutor))
318 												.doOnComplete(() ->
319 														triggerContext
320 																.update(triggerContext.lastScheduledExecutionTime(),
321 																		triggerContext.lastActualExecutionTime(),
322 																		new Date())
323 												)), 1)
324 				.repeat(this::isRunning)
325 				.doOnSubscribe(subs -> this.subscription = subs);
326 	}

328 	private Message<?> pollForMessage() {
329 		try {
330 			return this.pollingTask.call();
331 		}
332 		catch (Exception e) {
333 			if (e instanceof MessagingException) { // NOSONAR
334 				throw (MessagingException) e;
335 			}
336 			else {
337 				Message<?> failedMessage = null;
338 				if (this.transactionSynchronizationFactory != null) {
339 					Object resource = TransactionSynchronizationManager.getResource(getResourceToBind());
340 					if (resource instanceof IntegrationResourceHolder) {
341 						failedMessage = ((IntegrationResourceHolder) resource).getMessage();
342 					}
343 				}
344 				throw new MessagingException(failedMessage, e); // NOSONAR (null failedMessage)
345 			}
346 		}
347 		finally {
348 			if (this.transactionSynchronizationFactory != null) {
349 				Object resource = getResourceToBind();
350 				if (TransactionSynchronizationManager.hasResource(resource)) {
351 					TransactionSynchronizationManager.unbindResource(resource);
352 				}
353 			}
354 		}
355 	}
356 
357 	private Message<?> doPoll() {
358 		IntegrationResourceHolder holder = bindResourceHolderIfNecessary(getResourceKey(), getResourceToBind());
359 		Message<?> message;
360 		try {
361 			message = receiveMessage();
362 		}
363 		catch (Exception e) {
364 			if (Thread.interrupted()) {
365 				if (logger.isDebugEnabled()) {
366 					logger.debug("Poll interrupted - during stop()? : " + e.getMessage());
367 				}
368 				return null;
369 			}
370 			else {
371 				throw (RuntimeException) e;
372 			}
373 		}
374 
375 		if (message == null) {
376 			if (this.logger.isDebugEnabled()) {
377 				this.logger.debug("Received no Message during the poll, returning 'false'");
378 			}
379 			return null;
380 		}
381 		else {
382 			messageReceived(holder, message);
383 		}
384 
385 		return message;
386 	}
387 
388 	private void messageReceived(IntegrationResourceHolder holder, Message<?> message) {
389 		if (this.logger.isDebugEnabled()) {
390 			this.logger.debug("Poll resulted in Message: " + message);
391 		}
392 		if (holder != null) {
393 			holder.setMessage(message);
394 		}
395 
396 		if (!isReactive()) {
397 			try {
398 				handleMessage(message);
399 			}
400 			catch (MessagingException ex) {
401 				throw new MessagingExceptionWrapper(message, ex);
402 			}
403 			catch (Exception ex) {
404 				throw new MessagingException(message, ex);
405 			}
406 		}
407 	}

409 	@Override // guarded by super#lifecycleLock
410 	protected void doStop() {
411 		if (this.runningTask != null) {
412 			this.runningTask.cancel(true);
413 		}
414 		this.runningTask = null;
415 
416 		if (this.subscription != null) {
417 			this.subscription.cancel();
418 		}
419 	}

455 	private IntegrationResourceHolder bindResourceHolderIfNecessary(String key, Object resource) {
456 		if (this.transactionSynchronizationFactory != null && resource != null &&
457 				TransactionSynchronizationManager.isActualTransactionActive()) {
458 
459 			TransactionSynchronization synchronization = this.transactionSynchronizationFactory.create(resource);
460 			if (synchronization != null) {
461 				TransactionSynchronizationManager.registerSynchronization(synchronization);
462 
463 				if (synchronization instanceof IntegrationResourceHolderSynchronization) {
464 					IntegrationResourceHolderSynchronization integrationSynchronization =
465 							((IntegrationResourceHolderSynchronization) synchronization);
466 					integrationSynchronization.setShouldUnbindAtCompletion(false);
467 
468 					if (!TransactionSynchronizationManager.hasResource(resource)) {
469 						TransactionSynchronizationManager.bindResource(resource,
470 								integrationSynchronization.getResourceHolder());
471 					}
472 				}
473 			}
474 
475 			Object resourceHolder = TransactionSynchronizationManager.getResource(resource);
476 			if (resourceHolder instanceof IntegrationResourceHolder) {
477 				IntegrationResourceHolder integrationResourceHolder = (IntegrationResourceHolder) resourceHolder;
478 				if (key != null) {
479 					integrationResourceHolder.addAttribute(key, resource);
480 				}
481 				return integrationResourceHolder;
482 			}
483 		}
484 
485 		return null;
486 	}

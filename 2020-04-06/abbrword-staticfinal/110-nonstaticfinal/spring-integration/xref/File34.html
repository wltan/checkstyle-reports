113 	@Override
114 	public void postProcessBeanFactory(ConfigurableListableBeanFactory beanFactory) throws BeansException {
115 		if (beanFactory instanceof BeanDefinitionRegistry) {
116 			this.beanFactory = beanFactory;
117 			this.registry = (BeanDefinitionRegistry) beanFactory;
118 
119 			registerBeanFactoryChannelResolver();
120 			registerMessagePublishingErrorHandler();
121 			registerNullChannel();
122 			registerErrorChannel();
123 			registerIntegrationEvaluationContext();
124 			registerTaskScheduler();
125 			registerIdGeneratorConfigurer();
126 			registerIntegrationProperties();
127 			registerBuiltInBeans();
128 			registerRoleController();
129 			registerMessageBuilderFactory();
130 			registerHeaderChannelRegistry();
131 			registerGlobalChannelInterceptorProcessor();
132 			registerDefaultDatatypeChannelMessageConverter();
133 			registerArgumentResolverMessageConverter();
134 			registerMessageHandlerMethodFactory();
135 			registerListMessageHandlerMethodFactory();
136 		}
137 		else if (LOGGER.isWarnEnabled()) {
138 			LOGGER.warn("BeanFactory is not a BeanDefinitionRegistry. " +
139 					"The default Spring Integration infrastructure beans are not going to be registered");
140 		}
141 	}

153 		}

171 	/**
172 	 * Register a null channel in the application context.
173 	 * The bean name is defined by the constant {@link IntegrationContextUtils#NULL_CHANNEL_BEAN_NAME}.
174 	 */
175 	private void registerNullChannel() {
176 		if (this.beanFactory.containsBean(IntegrationContextUtils.NULL_CHANNEL_BEAN_NAME)) {
177 			BeanDefinition nullChannelDefinition = null;
178 			BeanFactory beanFactoryToUse = this.beanFactory;
179 			do {
180 				if (beanFactoryToUse instanceof ConfigurableListableBeanFactory) {
181 					ConfigurableListableBeanFactory listable = (ConfigurableListableBeanFactory) beanFactoryToUse;
182 					if (listable.containsBeanDefinition(IntegrationContextUtils.NULL_CHANNEL_BEAN_NAME)) {
183 						nullChannelDefinition = listable
184 								.getBeanDefinition(IntegrationContextUtils.NULL_CHANNEL_BEAN_NAME);
185 					}
186 				}
187 				if (beanFactoryToUse instanceof HierarchicalBeanFactory) {
188 					beanFactoryToUse = ((HierarchicalBeanFactory) beanFactoryToUse).getParentBeanFactory();
189 				}
190 			}
191 			while (nullChannelDefinition == null);
192 
193 			if (!NullChannel.class.getName().equals(nullChannelDefinition.getBeanClassName())) {
194 				throw new IllegalStateException("The bean name '" + IntegrationContextUtils.NULL_CHANNEL_BEAN_NAME
195 						+ "' is reserved.");
196 			}
197 		}
198 		else {
199 			this.registry.registerBeanDefinition(IntegrationContextUtils.NULL_CHANNEL_BEAN_NAME,
200 					new RootBeanDefinition(NullChannel.class));
201 		}
202 	}

204 	/**
205 	 * Register an error channel in the application context.
206 	 * The bean name is defined by the constant {@link IntegrationContextUtils#ERROR_CHANNEL_BEAN_NAME}.
207 	 * Also a {@link IntegrationContextUtils#ERROR_LOGGER_BEAN_NAME} is registered as a subscriber for this
208 	 * error channel.
209 	 */
210 	private void registerErrorChannel() {
211 		if (!this.beanFactory.containsBean(IntegrationContextUtils.ERROR_CHANNEL_BEAN_NAME)) {
212 			if (LOGGER.isInfoEnabled()) {
213 				LOGGER.info("No bean named '" + IntegrationContextUtils.ERROR_CHANNEL_BEAN_NAME +
214 						"' has been explicitly defined. " +
215 						"Therefore, a default PublishSubscribeChannel will be created.");
216 			}
217 			this.registry.registerBeanDefinition(IntegrationContextUtils.ERROR_CHANNEL_BEAN_NAME,
218 					new RootBeanDefinition(PublishSubscribeChannel.class));
219 
220 			BeanDefinition loggingHandler =
221 					BeanDefinitionBuilder.genericBeanDefinition(LoggingHandler.class)
222 							.addConstructorArgValue("ERROR")
223 							.getBeanDefinition();
224 
225 			String errorLoggerBeanName =
226 					IntegrationContextUtils.ERROR_LOGGER_BEAN_NAME + IntegrationConfigUtils.HANDLER_ALIAS_SUFFIX;
227 			this.registry.registerBeanDefinition(errorLoggerBeanName, loggingHandler);
228 
229 			BeanDefinitionBuilder loggingEndpointBuilder =
230 					BeanDefinitionBuilder.genericBeanDefinition(ConsumerEndpointFactoryBean.class)
231 							.addPropertyReference("handler", errorLoggerBeanName)
232 							.addPropertyValue("inputChannelName", IntegrationContextUtils.ERROR_CHANNEL_BEAN_NAME);
233 
234 			BeanComponentDefinition componentDefinition =
235 					new BeanComponentDefinition(loggingEndpointBuilder.getBeanDefinition(),
236 							IntegrationContextUtils.ERROR_LOGGER_BEAN_NAME);
237 			BeanDefinitionReaderUtils.registerBeanDefinition(componentDefinition, this.registry);
238 		}
239 	}

273 	/**
274 	 * Register an {@link IdGeneratorConfigurer} in the application context.
275 	 */
276 	private void registerIdGeneratorConfigurer() {
277 		Class<?> clazz = IdGeneratorConfigurer.class;
278 		String className = clazz.getName();
279 		String[] definitionNames = this.registry.getBeanDefinitionNames();
280 		for (String definitionName : definitionNames) {
281 			BeanDefinition definition = this.registry.getBeanDefinition(definitionName);
282 			if (className.equals(definition.getBeanClassName())) {
283 				if (LOGGER.isInfoEnabled()) {
284 					LOGGER.info(className + " is already registered and will be used");
285 				}
286 				return;
287 			}
288 		}
289 		RootBeanDefinition beanDefinition = new RootBeanDefinition(clazz);
290 		beanDefinition.setRole(BeanDefinition.ROLE_INFRASTRUCTURE);
291 		BeanDefinitionReaderUtils.registerWithGeneratedName(beanDefinition, this.registry);
292 	}

298 		if (!this.beanFactory.containsBean(IntegrationContextUtils.TASK_SCHEDULER_BEAN_NAME)) {

309 					.addPropertyReference("errorHandler",
310 							ChannelUtils.MESSAGE_PUBLISHING_ERROR_HANDLER_BEAN_NAME)
311 					.getBeanDefinition();

313 			this.registry.registerBeanDefinition(IntegrationContextUtils.TASK_SCHEDULER_BEAN_NAME, scheduler);
314 		}

317 	/**
318 	 * Register an {@code integrationGlobalProperties} bean if necessary.
319 	 */
320 	private void registerIntegrationProperties() {
321 		if (!this.beanFactory.containsBean(IntegrationContextUtils.INTEGRATION_GLOBAL_PROPERTIES_BEAN_NAME)) {
322 			ResourcePatternResolver resourceResolver = new PathMatchingResourcePatternResolver(this.classLoader);
323 			try {
324 				Resource[] defaultResources =
325 						resourceResolver.getResources("classpath*:META-INF/spring.integration.default.properties");
326 				Resource[] userResources =
327 						resourceResolver.getResources("classpath*:META-INF/spring.integration.properties");
328 
329 				List<Resource> resources = new LinkedList<>(Arrays.asList(defaultResources));
330 				resources.addAll(Arrays.asList(userResources));
331 
332 				BeanDefinitionBuilder integrationPropertiesBuilder = BeanDefinitionBuilder
333 						.genericBeanDefinition(PropertiesFactoryBean.class)
334 						.addPropertyValue("locations", resources);
335 
336 				this.registry.registerBeanDefinition(IntegrationContextUtils.INTEGRATION_GLOBAL_PROPERTIES_BEAN_NAME,
337 						integrationPropertiesBuilder.getBeanDefinition());
338 			}
339 			catch (IOException e) {
340 				LOGGER.warn("Cannot load 'spring.integration.properties' Resources.", e);
341 			}
342 		}
343 	}

356 	private void jsonPath(int registryId) throws LinkageError {
357 		String jsonPathBeanName = "jsonPath";
358 		if (!this.beanFactory.containsBean(jsonPathBeanName) && !REGISTRIES_PROCESSED.contains(registryId)) {
359 			Class<?> jsonPathClass = null;
360 			try {
361 				jsonPathClass = ClassUtils.forName("com.jayway.jsonpath.JsonPath", this.classLoader);
362 			}
363 			catch (@SuppressWarnings("unused") ClassNotFoundException e) {
364 				LOGGER.debug("The '#jsonPath' SpEL function cannot be registered: " +
365 						"there is no jayway json-path.jar on the classpath.");
366 			}
367 
368 			if (jsonPathClass != null) {
369 				try {
370 					ClassUtils.forName("com.jayway.jsonpath.Predicate", this.classLoader);
371 				}
372 				catch (ClassNotFoundException e) {
373 					jsonPathClass = null;
374 					LOGGER.warn("The '#jsonPath' SpEL function cannot be registered. " +
375 							"An old json-path.jar version is detected in the classpath." +
376 							"At least 2.4.0 is required; see version information at: " +
377 							"https://github.com/jayway/JsonPath/releases", e);
378 
379 				}
380 			}
381 
382 			if (jsonPathClass != null) {
383 				IntegrationConfigUtils.registerSpelFunctionBean(this.registry, jsonPathBeanName,
384 						IntegrationConfigUtils.BASE_PACKAGE + ".json.JsonPathUtils", "evaluate");
385 			}
386 		}
387 	}

389 	private void xpath(int registryId) throws LinkageError {
390 		String xpathBeanName = "xpath";
391 		if (!this.beanFactory.containsBean(xpathBeanName) && !REGISTRIES_PROCESSED.contains(registryId)) {
392 			Class<?> xpathClass = null;
393 			try {
394 				xpathClass = ClassUtils.forName(IntegrationConfigUtils.BASE_PACKAGE + ".xml.xpath.XPathUtils",
395 						this.classLoader);
396 			}
397 			catch (@SuppressWarnings("unused") ClassNotFoundException e) {
398 				LOGGER.debug("SpEL function '#xpath' isn't registered: " +
399 						"there is no spring-integration-xml.jar on the classpath.");
400 			}
401 
402 			if (xpathClass != null) {
403 				IntegrationConfigUtils.registerSpelFunctionBean(this.registry, xpathBeanName,
404 						IntegrationConfigUtils.BASE_PACKAGE + ".xml.xpath.XPathUtils", "evaluate");
405 			}
406 		}
407 	}

541 	private ManagedList<BeanDefinition> buildArgumentResolvers(boolean listCapable) {
542 		ManagedList<BeanDefinition> resolvers = new ManagedList<>();
543 		resolvers.add(new RootBeanDefinition(PayloadExpressionArgumentResolver.class));

58  /**
59   * A {@link MessageHandler} that is capable of delaying the continuation of a Message flow
60   * based on the result of evaluation {@code delayExpression} on an inbound {@link Message}
61   * or a default delay value configured on this handler. Note that the continuation of the
62   * flow is delegated to a {@link TaskScheduler}, and therefore, the calling thread does
63   * not block. The advantage of this approach is that many delays can be managed
64   * concurrently, even very long delays, without producing a buildup of blocked Threads.
65   * <p>
66   * One thing to keep in mind, however, is that any active transactional context will not
67   * propagate from the original sender to the eventual recipient. This is a side-effect of
68   * passing the Message to the output channel after the delay with a different Thread in
69   * control.
70   * <p>
71   * When this handler's {@code delayExpression} property is configured, that evaluation
72   * result value will take precedence over the handler's {@code defaultDelay} value. The
73   * actual evaluation result value may be a long, a String that can be parsed as a long, or
74   * a Date. If it is a long, it will be interpreted as the length of time to delay in
75   * milliseconds counting from the current time (e.g. a value of 5000 indicates that the
76   * Message can be released as soon as five seconds from the current time). If the value is
77   * a Date, it will be delayed at least until that Date occurs (i.e. the delay in that case
78   * is equivalent to {@code headerDate.getTime() - new Date().getTime()}).
79   *
80   * @author Mark Fisher
81   * @author Artem Bilan
82   * @author Gary Russell
83   *
84   * @since 1.0.3
85   */
86

305 	private MessageHandler createReleaseMessageTask() {
306 		ReleaseMessageHandler handler = new ReleaseMessageHandler();
307 
308 		if (!CollectionUtils.isEmpty(this.delayedAdviceChain)) {
309 			ProxyFactory proxyFactory = new ProxyFactory(handler);
310 			for (Advice advice : this.delayedAdviceChain) {
311 				proxyFactory.addAdvice(advice);
312 			}
313 			return (MessageHandler) proxyFactory.getProxy(getApplicationContext().getClassLoader());
314 		}
315 		return handler;
316 	}

323 	/**
324 	 * Checks if 'requestMessage' wasn't delayed before ({@link #releaseMessageAfterDelay}
325 	 * and {@link DelayHandler.DelayedMessageWrapper}). Than determine 'delay' for
326 	 * 'requestMessage' ({@link #determineDelayForMessage}) and if {@code delay > 0}
327 	 * schedules 'releaseMessage' task after 'delay'.
328 	 * @param requestMessage - the Message which may be delayed.
329 	 * @return - {@code null} if 'requestMessage' is delayed, otherwise - 'payload' from
330 	 * 'requestMessage'.
331 	 * @see #releaseMessage
332 	 */
333 	@Override
334 	protected Object handleRequestMessage(Message<?> requestMessage) {
335 		boolean delayed = requestMessage.getPayload() instanceof DelayedMessageWrapper;
336 
337 		if (!delayed) {
338 			long delay = determineDelayForMessage(requestMessage);
339 			if (delay > 0) {
340 				releaseMessageAfterDelay(requestMessage, delay);
341 				return null;
342 			}
343 		}
344 
345 		// no delay
346 		return delayed ? ((DelayedMessageWrapper) requestMessage.getPayload()).getOriginal() : requestMessage;
347 	}

349 	private long determineDelayForMessage(Message<?> message) {
350 		if (this.delayExpression != null) {
351 			return determineDelayFromExpression(message);
352 		}
353 		else {
354 			return this.defaultDelay;
355 		}
356 	}
357 
358 	private long determineDelayFromExpression(Message<?> message) {
359 		long delay = this.defaultDelay;
360 		DelayedMessageWrapper delayedMessageWrapper = null;
361 		if (message.getPayload() instanceof DelayedMessageWrapper) {
362 			delayedMessageWrapper = (DelayedMessageWrapper) message.getPayload();
363 		}
364 		Exception delayValueException = null;
365 		Object delayValue = null;
366 		try {
367 			delayValue = this.delayExpression.getValue(this.evaluationContext,
368 					delayedMessageWrapper != null ? delayedMessageWrapper.getOriginal() : message);
369 		}
370 		catch (EvaluationException e) {
371 			delayValueException = e;
372 		}
373 		if (delayValue instanceof Date) {
374 			long current = delayedMessageWrapper != null
375 					? delayedMessageWrapper.getRequestDate()
376 					: System.currentTimeMillis();
377 			delay = ((Date) delayValue).getTime() - current;
378 		}
379 		else if (delayValue != null) {
380 			try {
381 				delay = Long.parseLong(delayValue.toString());
382 			}
383 			catch (NumberFormatException e) {
384 				delayValueException = e;
385 			}
386 		}
387 		if (delayValueException != null) {
388 			handleDelayValueException(delayValueException);
389 		}
390 		return delay;
391 	}
392 
393 	private void handleDelayValueException(Exception delayValueException) {
394 		if (this.ignoreExpressionFailures) {
395 			if (logger.isDebugEnabled()) {
396 				logger.debug("Failed to get delay value from 'delayExpression': " +
397 						delayValueException.getMessage() +
398 						". Will fall back to default delay: " + this.defaultDelay);
399 			}
400 		}
401 		else {
402 			throw new IllegalStateException("Error occurred during 'delay' value determination",
403 					delayValueException);
404 		}
405 	}

407 	private void releaseMessageAfterDelay(final Message<?> message, long delay) {
408 		Message<?> delayedMessage = message;
409 
410 		DelayedMessageWrapper messageWrapper = null;
411 		if (message.getPayload() instanceof DelayedMessageWrapper) {
412 			messageWrapper = (DelayedMessageWrapper) message.getPayload();
413 		}
414 		else {
415 			messageWrapper = new DelayedMessageWrapper(message, System.currentTimeMillis());
416 			delayedMessage = getMessageBuilderFactory()
417 					.withPayload(messageWrapper)
418 					.copyHeaders(message.getHeaders())
419 					.build();
420 			this.messageStore.addMessageToGroup(this.messageGroupId, delayedMessage);
421 		}
422 
423 		Runnable releaseTask;
424 
425 		if (this.messageStore instanceof SimpleMessageStore) {
426 			final Message<?> messageToSchedule = delayedMessage;
427 
428 			releaseTask = () -> releaseMessage(messageToSchedule);
429 		}
430 		else {
431 			final UUID messageId = delayedMessage.getHeaders().getId();
432 
433 			releaseTask = () -> {
434 				Message<?> messageToRelease = getMessageById(messageId);
435 				if (messageToRelease != null) {
436 					releaseMessage(messageToRelease);
437 				}
438 			};
439 		}
440 
441 		getTaskScheduler().schedule(releaseTask, new Date(messageWrapper.getRequestDate() + delay));
442 	}
443 
444 	private Message<?> getMessageById(UUID messageId) {
445 		Message<?> theMessage = ((MessageStore) this.messageStore).getMessage(messageId);
446 
447 		if (theMessage == null) {
448 			if (logger.isDebugEnabled()) {
449 				logger.debug("No message in the Message Store for id: " + messageId +
450 						". Likely another instance has already released it.");
451 			}
452 			return null;
453 		}
454 		else {
455 			return theMessage;
456 		}
457 	}

459 	private void releaseMessage(Message<?> message) {
460 		String identity = ObjectUtils.getIdentityHexString(message);
461 		this.deliveries.putIfAbsent(identity, new AtomicInteger());
462 		try {
463 			this.releaseHandler.handleMessage(message);
464 			this.deliveries.remove(identity);
465 		}
466 		catch (Exception e) {
467 			if (getErrorChannel() != null) {
468 				ErrorMessage errorMessage = new ErrorMessage(e,
469 						Collections.singletonMap(IntegrationMessageHeaderAccessor.DELIVERY_ATTEMPT,
470 								new AtomicInteger(this.deliveries.get(identity).get() + 1)),
471 						message);
472 				try {
473 					if (!(getErrorChannel().send(errorMessage))) {
474 						if (this.logger.isDebugEnabled()) {
475 							this.logger.debug("Failed to send error message: " + errorMessage);
476 						}
477 						rescheduleForRetry(message, identity);
478 					}
479 					else {
480 						this.deliveries.remove(identity);
481 					}
482 				}
483 				catch (Exception e1) {
484 					if (this.logger.isDebugEnabled()) {
485 						logger.debug("Error flow threw an exception for message: " + message, e1);
486 					}
487 					rescheduleForRetry(message, identity);
488 				}
489 			}
490 			else {
491 				if (this.logger.isDebugEnabled()) {
492 					logger.debug("Release flow threw an exception for message: " + message, e);
493 				}
494 				if (!rescheduleForRetry(message, identity)) {
495 					throw e; // there might be an error handler on the scheduler
496 				}
497 			}
498 		}
499 	}

501 	private boolean rescheduleForRetry(Message<?> message, String identity) {
502 		if (this.deliveries.get(identity).incrementAndGet() >= this.maxAttempts) {
503 			this.logger.error("Discarding; maximum release attempts reached for: " + message);
504 			this.deliveries.remove(identity);
505 			return false;
506 		}
507 		if (this.retryDelay <= 0) {
508 			rescheduleNow(message);
509 		}
510 		else {
511 			rescheduleAt(message, new Date(System.currentTimeMillis() + this.retryDelay));
512 		}
513 		return true;
514 	}
515 
516 	private void rescheduleNow(final Message<?> message) {
517 		rescheduleAt(message, new Date());
518 	}
519 
520 	protected void rescheduleAt(final Message<?> message, Date startTime) {
521 		getTaskScheduler()
522 				.schedule(() -> releaseMessage(message), startTime);
523 	}
524 
525 	private void doReleaseMessage(Message<?> message) {
526 		if (removeDelayedMessageFromMessageStore(message)
527 				|| this.deliveries.get(ObjectUtils.getIdentityHexString(message)).get() > 0) {
528 			if (!(this.messageStore instanceof SimpleMessageStore)) {
529 				this.messageStore.removeMessagesFromGroup(this.messageGroupId, message);
530 			}
531 			handleMessageInternal(message);
532 		}
533 		else if (logger.isDebugEnabled()) {
534 			logger.debug("No message in the Message Store to release: " + message +
535 					". Likely another instance has already released it.");
536 		}
537 	}
538 
539 	private boolean removeDelayedMessageFromMessageStore(Message<?> message) {
540 		if (this.messageStore instanceof SimpleMessageStore) {
541 			synchronized (this.messageGroupId) {
542 				Collection<Message<?>> messages = this.messageStore.getMessageGroup(this.messageGroupId).getMessages();
543 				if (messages.contains(message)) {
544 					this.messageStore.removeMessagesFromGroup(this.messageGroupId, message);
545 					return true;
546 				}
547 				else {
548 					return false;
549 				}
550 			}
551 		}
552 		else {
553 			return ((MessageStore) this.messageStore).removeMessage(message.getHeaders().getId()) != null;
554 		}
555 	}

562 	/**
563 	 * Used for reading persisted Messages in the 'messageStore' to reschedule them e.g.
564 	 * upon application restart. The logic is based on iteration over
565 	 * {@code messageGroup.getMessages()} and schedules task for 'delay' logic. This
566 	 * behavior is dictated by the avoidance of invocation thread overload.
567 	 */
568 	@Override
569 	public synchronized void reschedulePersistedMessages() {
570 		MessageGroup messageGroup = this.messageStore.getMessageGroup(this.messageGroupId);
571 		for (final Message<?> message : messageGroup.getMessages()) {
572 			getTaskScheduler()
573 					.schedule(() -> {
574 						// This is fine to keep the reference to the message,
575 						// because the scheduled task is performed immediately.
576 						long delay = determineDelayForMessage(message);
577 						if (delay > 0) {
578 							releaseMessageAfterDelay(message, delay);
579 						}
580 						else {
581 							releaseMessage(message);
582 						}
583 					}, new Date());
584 		}
585 	}

587 	/**
588 	 * Handles {@link ContextRefreshedEvent} to invoke
589 	 * {@link #reschedulePersistedMessages} as late as possible after application context
590 	 * startup. Also it checks {@link #initialized} to ignore other
591 	 * {@link ContextRefreshedEvent}s which may be published in the 'parent-child'
592 	 * contexts, e.g. in the Spring-MVC applications.
593 	 * @param event - {@link ContextRefreshedEvent} which occurs after Application context
594 	 * is completely initialized.
595 	 * @see #reschedulePersistedMessages
596 	 */
597 	@Override
598 	public void onApplicationEvent(ContextRefreshedEvent event) {
599 		if (event.getApplicationContext().equals(getApplicationContext())
600 				&& !this.initialized.getAndSet(true)) {
601 			reschedulePersistedMessages();
602 		}
603 	}

625 
626 		private static final long serialVersionUID = -4739802369074947045L;
627 
628 		private final long requestDate;
629 
630 		private final Message<?> original;
631 
632 		DelayedMessageWrapper(Message<?> original, long requestDate) {
633 			this.original = original;
634 			this.requestDate = requestDate;
635 		}

644 
645 		@Override
646 		public boolean equals(Object o) {
647 			if (this == o) {
648 				return true;
649 			}
650 			if (o == null || getClass() != o.getClass()) {
651 				return false;

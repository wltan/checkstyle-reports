62      @Test
63      public void testNonExistingFile() {
64          PropertyFile props = new PropertyFile();
65          props.setProject( buildRule.getProject() );
66          File file = new File("this-file-does-not-exist.properties");
67          props.setFile(file);
68          assertFalse("Properties file exists before test.", file.exists());
69          props.execute();
70          assertTrue("Properties file does not exist after test.", file.exists());
71          file.delete();
72      }

74      /**
75       *  A unit test for JUnit- Exercises the propertyfile tasks ability to
76       *  update properties that are already defined-
77       */
78      @Test
79      public void testUpdatesExistingProperties() throws Exception {
80          Properties beforeUpdate = getTestProperties();
81          assertEquals(FNAME, beforeUpdate.getProperty(FNAME_KEY));
82          assertEquals(LNAME, beforeUpdate.getProperty(LNAME_KEY));
83          assertEquals(EMAIL, beforeUpdate.getProperty(EMAIL_KEY));
84          assertEquals(null, beforeUpdate.getProperty(PHONE_KEY));
85          assertEquals(null, beforeUpdate.getProperty(AGE_KEY));
86          assertEquals(null, beforeUpdate.getProperty(DATE_KEY));
87  
88          // ask ant to update the properties...
89          buildRule.executeTarget("update-existing-properties");
90  
91          Properties afterUpdate = getTestProperties();
92          assertEquals(NEW_FNAME, afterUpdate.getProperty(FNAME_KEY));
93          assertEquals(NEW_LNAME, afterUpdate.getProperty(LNAME_KEY));
94          assertEquals(NEW_EMAIL, afterUpdate.getProperty(EMAIL_KEY));
95          assertEquals(NEW_PHONE, afterUpdate.getProperty(PHONE_KEY));
96          assertEquals(NEW_AGE, afterUpdate.getProperty(AGE_KEY));
97          assertEquals(NEW_DATE, afterUpdate.getProperty(DATE_KEY));
98      }

100     @Test
101     public void testDeleteProperties() throws Exception {
102         Properties beforeUpdate = getTestProperties();
103         assertEquals("Property '" + FNAME_KEY + "' should exist before deleting",
104             FNAME, beforeUpdate.getProperty(FNAME_KEY));
105         assertEquals("Property '" + LNAME_KEY + "' should exist before deleting",
106             LNAME, beforeUpdate.getProperty(LNAME_KEY));
107         
108         buildRule.executeTarget("delete-properties");
109         Properties afterUpdate = getTestProperties();
110 
111         assertEquals("Property '" + LNAME_KEY + "' should exist after deleting",
112             LNAME, afterUpdate.getProperty(LNAME_KEY));
113         assertNull("Property '" + FNAME_KEY + "' should be deleted",
114             afterUpdate.getProperty(FNAME_KEY));
115     }

165     private Properties getTestProperties() throws Exception {
166         Properties testProps = new Properties();
167         FileInputStream propsFile = new FileInputStream(new File(buildRule.getOutputDir(), testPropsFilePath));
168         testProps.load(propsFile);
169         propsFile.close();
170         return testProps;
171     }

174     private void initTestPropFile() throws IOException {
175         Properties testProps = new Properties();
176         testProps.put(FNAME_KEY, FNAME);
177         testProps.put(LNAME_KEY, LNAME);
178         testProps.put(EMAIL_KEY, EMAIL);
179         testProps.put("existing.prop", "37");
180 
181         FileOutputStream fos = new FileOutputStream(new File(buildRule.getOutputDir(), testPropsFilePath));
182         testProps.store(fos, "defaults");
183         fos.close();
184     }

198         buildProps.store(fos, null);

200     }

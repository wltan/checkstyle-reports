39  /**
40   * Helper class that collects the methods a task or nested element
41   * holds to set attributes, create nested elements or hold PCDATA
42   * elements.
43   *
44   * It contains hashtables containing classes that use introspection
45   * to handle all the invocation of the project-component specific methods.
46   *
47   * This class is somewhat complex, as it implements the O/X mapping between
48   * Ant XML and Java class instances. This is not the best place for someone new
49   * to Ant to start contributing to the codebase, as a change here can break the
50   * entire system in interesting ways. Always run a full test of Ant before checking
51   * in/submitting changes to this file.
52   *
53   * The class is final and has a private constructor.
54   * To get an instance for a specific (class,project) combination,
55   * use {@link #getHelper(Project,Class)}.
56   * This may return an existing version, or a new one
57   * ...do not make any assumptions about its uniqueness, or its validity after the Project
58   * instance has finished its build.
59   *
60   */

129     /**
130      * Sole constructor, which is private to ensure that all
131      * IntrospectionHelpers are created via {@link #getHelper(Class) getHelper}.
132      * Introspects the given class for bean-like methods.
133      * Each method is examined in turn, and the following rules are applied:
134      * <p>
135      * <ul>
136      * <li>If the method is <code>Task.setLocation(Location)</code>,
137      * <code>Task.setTaskType(String)</code>
138      * or <code>TaskContainer.addTask(Task)</code>, it is ignored. These
139      * methods are handled differently elsewhere.
140      * <li><code>void addText(String)</code> is recognised as the method for
141      * adding PCDATA to a bean.
142      * <li><code>void setFoo(Bar)</code> is recognised as a method for
143      * setting the value of attribute <code>foo</code>, so long as
144      * <code>Bar</code> is non-void and is not an array type.
145      * As of Ant 1.8, a Resource or FileProvider parameter overrides a java.io.File parameter;
146      * in practice the only effect of this is to allow objects rendered from
147      * the 1.8 PropertyHelper implementation to be used as Resource parameters,
148      * since Resources set from Strings are resolved as project-relative files
149      * to preserve backward compatibility.  Beyond this, non-String
150      * parameter types always overload String parameter types; these are
151      * the only guarantees made in terms of priority.
152      * <li><code>Foo createBar()</code> is recognised as a method for
153      * creating a nested element called <code>bar</code> of type
154      * <code>Foo</code>, so long as <code>Foo</code> is not a primitive or
155      * array type.
156      * <li><code>void addConfiguredFoo(Bar)</code> is recognised as a
157      * method for storing a pre-configured element called
158      * <code>foo</code> and of type <code>Bar</code>, so long as
159      * <code>Bar</code> is not an array, primitive or String type.
160      * <code>Bar</code> must have an accessible constructor taking no
161      * arguments.
162      * <li><code>void addFoo(Bar)</code> is recognised as a method for storing
163      * an element called <code>foo</code> and of type <code>Bar</code>, so
164      * long as <code>Bar</code> is not an array, primitive or String type.
165      * <code>Bar</code> must have an accessible constructor taking no
166      * arguments. This is distinct from the 'addConfigured' idiom in that
167      * the nested element is added to the parent immediately after it is
168      * constructed; in practice this means that <code>addFoo(Bar)</code> should
169      * do little or nothing with its argument besides storing it for later use.
170      * </ul>
171      * Note that only one method is retained to create/set/addConfigured/add
172      * any element or attribute.
173      *
174      * @param bean The bean type to introspect.
175      *             Must not be <code>null</code>.
176      *
177      * @see #getHelper(Class)
178      */
179     private IntrospectionHelper(final Class<?> bean) {
180         this.bean = bean;
181         Method[] methods = bean.getMethods();
182         Method addTextMethod = null;
183         for (int i = 0; i < methods.length; i++) {
184             final Method m = methods[i];
185             final String name = m.getName();
186             Class<?> returnType = m.getReturnType();
187             Class<?>[] args = m.getParameterTypes();
188 
189             // check of add[Configured](Class) pattern
190             if (args.length == 1 && java.lang.Void.TYPE.equals(returnType)
191                     && ("add".equals(name) || "addConfigured".equals(name))) {
192                 insertAddTypeMethod(m);
193                 continue;
194             }
195             // not really user settable properties on tasks/project components
196             if (org.apache.tools.ant.ProjectComponent.class.isAssignableFrom(bean)
197                     && args.length == 1 && isHiddenSetMethod(name, args[0])) {
198                 continue;
199             }
200             // hide addTask for TaskContainers
201             if (isContainer() && args.length == 1 && "addTask".equals(name)
202                     && org.apache.tools.ant.Task.class.equals(args[0])) {
203                 continue;
204             }
205             if ("addText".equals(name) && java.lang.Void.TYPE.equals(returnType)
206                     && args.length == 1 && java.lang.String.class.equals(args[0])) {
207                 addTextMethod = methods[i];
208             } else if (name.startsWith("set") && java.lang.Void.TYPE.equals(returnType)
209                     && args.length == 1 && !args[0].isArray()) {
210                 String propName = getPropertyName(name, "set");
211                 AttributeSetter as = (AttributeSetter) attributeSetters.get(propName);
212                 if (as != null) {
213                     if (java.lang.String.class.equals(args[0])) {
214                         /*
215                             Ignore method m, as there is an overloaded
216                             form of this method that takes in a
217                             non-string argument, which gains higher
218                             priority.
219                         */
220                         continue;
221                     }
222                     if (java.io.File.class.equals(args[0])) {
223                         // Ant Resources/FileProviders override java.io.File
224                         if (Resource.class.equals(as.type) || FileProvider.class.equals(as.type)) {
225                             continue;
226                         }
227                     }
228                     /*
229                         In cases other than those just explicitly covered,
230                         we just override that with the new one.
231                         This mechanism does not guarantee any specific order
232                         in which the methods will be selected: so any code
233                         that depends on the order in which "set" methods have
234                         been defined, is not guaranteed to be selected in any
235                         particular order.
236                     */
237                 }
238                 as = createAttributeSetter(m, args[0], propName);
239                 if (as != null) {
240                     attributeTypes.put(propName, args[0]);
241                     attributeSetters.put(propName, as);
242                 }
243             } else if (name.startsWith("create") && !returnType.isArray()
244                     && !returnType.isPrimitive() && args.length == 0) {
245 
246                 String propName = getPropertyName(name, "create");
247                 // Check if a create of this property is already present
248                 // add takes preference over create for CB purposes
249                 if (nestedCreators.get(propName) == null) {
250                     nestedTypes.put(propName, returnType);
251                     nestedCreators.put(propName, new CreateNestedCreator(m));
252                 }
253             } else if (name.startsWith("addConfigured")
254                     && java.lang.Void.TYPE.equals(returnType) && args.length == 1
255                     && !java.lang.String.class.equals(args[0])
256                     && !args[0].isArray() && !args[0].isPrimitive()) {
257                 try {
258                     Constructor<?> constructor = null;
259                     try {
260                         constructor = args[0].getConstructor();
261                     } catch (NoSuchMethodException ex) {
262                         constructor = args[0].getConstructor(Project.class);
263                     }
264                     String propName = getPropertyName(name, "addConfigured");
265                     nestedTypes.put(propName, args[0]);
266                     nestedCreators.put(propName, new AddNestedCreator(m,
267                         constructor, AddNestedCreator.ADD_CONFIGURED));
268                 } catch (NoSuchMethodException nse) {
269                     // ignore
270                 }
271             } else if (name.startsWith("add")
272                     && java.lang.Void.TYPE.equals(returnType) && args.length == 1
273                     && !java.lang.String.class.equals(args[0])
274                     && !args[0].isArray() && !args[0].isPrimitive()) {
275                 try {
276                     Constructor<?> constructor = null;
277                     try {
278                         constructor = args[0].getConstructor();
279                     } catch (NoSuchMethodException ex) {
280                         constructor = args[0].getConstructor(Project.class);
281                     }
282                     String propName = getPropertyName(name, "add");
283                     if (nestedTypes.get(propName) != null) {
284                         /*
285                          *  Ignore this method as there is an addConfigured
286                          *  form of this method that has a higher
287                          *  priority
288                          */
289                         continue;
290                     }
291                     nestedTypes.put(propName, args[0]);
292                     nestedCreators.put(propName, new AddNestedCreator(m,
293                             constructor, AddNestedCreator.ADD));
294                 } catch (NoSuchMethodException nse) {
295                     // ignore
296                 }
297             }
298         }
299         addText = addTextMethod;
300     }

333     /**
334      * Returns a helper for the given class, either from the cache
335      * or by creating a new instance.
336      *
337      * The method will make sure the helper will be cleaned up at the end of
338      * the project, and only one instance will be created for each class.
339      *
340      * @param p the project instance. Can be null, in which case the helper is not cached.
341      * @param c The class for which a helper is required.
342      *          Must not be <code>null</code>.
343      *
344      * @return a helper for the specified class
345      */
346     public synchronized static IntrospectionHelper getHelper(Project p, Class<?> c) {
347         IntrospectionHelper ih = HELPERS.get(c.getName());
348         // If a helper cannot be found, or if the helper is for another
349         // classloader, create a new IH
350         if (ih == null || ih.bean != c) {
351             ih = new IntrospectionHelper(c);
352             if (p != null) {
353                 // #30162: do *not* cache this if there is no project, as we
354                 // cannot guarantee that the cache will be cleared.
355                 HELPERS.put(c.getName(), ih);
356             }
357         }
358         return ih;
359     }

361     /**
362      * Sets the named attribute in the given element, which is part of the
363      * given project.
364      *
365      * @param p The project containing the element. This is used when files
366      *          need to be resolved. Must not be <code>null</code>.
367      * @param element The element to set the attribute in. Must not be
368      *                <code>null</code>.
369      * @param attributeName The name of the attribute to set. Must not be
370      *                      <code>null</code>.
371      * @param value The value to set the attribute to. This may be interpreted
372      *              or converted to the necessary type if the setter method
373      *              doesn't accept an object of the supplied type.
374      *
375      * @exception BuildException if the introspected class doesn't support
376      *                           the given attribute, or if the setting
377      *                           method fails.
378      */
379     public void setAttribute(Project p, Object element, String attributeName,
380             Object value) throws BuildException {
381         AttributeSetter as = (AttributeSetter) attributeSetters.get(
382                 attributeName.toLowerCase(Locale.ENGLISH));
383         if (as == null && value != null) {
384             if (element instanceof DynamicAttributeNS) {
385                 DynamicAttributeNS dc = (DynamicAttributeNS) element;
386                 String uriPlusPrefix = ProjectHelper.extractUriFromComponentName(attributeName);
387                 String uri = ProjectHelper.extractUriFromComponentName(uriPlusPrefix);
388                 String localName = ProjectHelper.extractNameFromComponentName(attributeName);
389                 String qName = "".equals(uri) ? localName : uri + ":" + localName;
390                 dc.setDynamicAttribute(uri, localName, qName, value.toString());
391                 return;
392             }
393             if (element instanceof DynamicObjectAttribute) {
394                 DynamicObjectAttribute dc = (DynamicObjectAttribute) element;
395                 dc.setDynamicAttribute(attributeName.toLowerCase(Locale.ENGLISH), value);
396                 return;
397             }
398             if (element instanceof DynamicAttribute) {
399                 DynamicAttribute dc = (DynamicAttribute) element;
400                 dc.setDynamicAttribute(attributeName.toLowerCase(Locale.ENGLISH), value.toString());
401                 return;
402             }
403             if (attributeName.indexOf(':') >= 0) {
404                 return; // Ignore attribute from unknown uri's
405             }
406             String msg = getElementName(p, element)
407                     + " doesn't support the \"" + attributeName + "\" attribute.";
408             throw new UnsupportedAttributeException(msg, attributeName);
409         }
410         try {
411             as.setObject(p, element, value);
412         } catch (IllegalAccessException ie) {
413             // impossible as getMethods should only return public methods
414             throw new BuildException(ie);
415         } catch (InvocationTargetException ite) {
416             throw extractBuildException(ite);
417         }
418     }

443     /**
444      * Adds PCDATA to an element, using the element's
445      * <code>void addText(String)</code> method, if it has one. If no
446      * such method is present, a BuildException is thrown if the
447      * given text contains non-whitespace.
448      *
449      * @param project The project which the element is part of.
450      *                Must not be <code>null</code>.
451      * @param element The element to add the text to.
452      *                Must not be <code>null</code>.
453      * @param text    The text to add.
454      *                Must not be <code>null</code>.
455      *
456      * @exception BuildException if non-whitespace text is provided and no
457      *                           method is available to handle it, or if
458      *                           the handling method fails.
459      */
460     public void addText(Project project, Object element, String text)
461         throws BuildException {
462         if (addText == null) {
463             text = text.trim();
464             // Element doesn't handle text content
465             if (text.length() == 0) {
466                 // Only whitespace - ignore
467                 return;
468             }
469             // Not whitespace - fail
470             throw new BuildException(project.getElementName(element)
471                     + " doesn't support nested text data (\"" + condenseText(text) + "\").");
472         }
473         try {
474             addText.invoke(element, new Object[] {text});
475         } catch (IllegalAccessException ie) {
476             // impossible as getMethods should only return public methods
477             throw new BuildException(ie);
478         } catch (InvocationTargetException ite) {
479             throw extractBuildException(ite);
480         }
481     }

512     /**
513      * Get the specific NestedCreator for a given project/parent/element combination
514      * @param project ant project
515      * @param parentUri URI of the parent.
516      * @param parent the parent class
517      * @param elementName element to work with. This can contain
518      *  a URI,localname tuple of of the form uri:localname
519      * @param child the bit of XML to work with
520      * @return a nested creator that can handle the child elements.
521      * @throws BuildException if the parent does not support child elements of that name
522      */
523     private NestedCreator getNestedCreator(
524         Project project, String parentUri, Object parent,
525         String elementName, UnknownElement child) throws BuildException {
526 
527         String uri = ProjectHelper.extractUriFromComponentName(elementName);
528         String name = ProjectHelper.extractNameFromComponentName(elementName);
529 
530         if (uri.equals(ProjectHelper.ANT_CORE_URI)) {
531             uri = "";
532         }
533         if (parentUri.equals(ProjectHelper.ANT_CORE_URI)) {
534             parentUri = "";
535         }
536         NestedCreator nc = null;
537         if (uri.equals(parentUri) || uri.length() == 0) {
538             nc = (NestedCreator) nestedCreators.get(name.toLowerCase(Locale.ENGLISH));
539         }
540         if (nc == null) {
541             nc = createAddTypeCreator(project, parent, elementName);
542         }
543         if (nc == null &&
544             (parent instanceof DynamicElementNS
545              || parent instanceof DynamicElement)
546             ) {
547             String qName = child == null ? name : child.getQName();
548             final Object nestedElement =
549                 createDynamicElement(parent,
550                                      child == null ? "" : child.getNamespace(),
551                                      name, qName);
552             if (nestedElement != null) {
553                 nc = new NestedCreator(null) {
554                     Object create(Project project, Object parent, Object ignore) {
555                         return nestedElement;
556                     }
557                 };
558             }
559         }
560         if (nc == null) {
561             throwNotSupported(project, parent, elementName);
562         }
563         return nc;
564     }

568      * order to obtain a child element by name.
569      *
570      * @since Ant 1.8.0.

572     private Object createDynamicElement(Object parent, String ns,

574         Object nestedElement = null;
575         if (parent instanceof DynamicElementNS) {
576             DynamicElementNS dc = (DynamicElementNS) parent;

582                 dc.createDynamicElement(localName.toLowerCase(Locale.ENGLISH));
583         }
584         return nestedElement;

587     /**
588      * Creates a named nested element. Depending on the results of the
589      * initial introspection, either a method in the given parent instance
590      * or a simple no-arg constructor is used to create an instance of the
591      * specified element type.
592      *
593      * @param project Project to which the parent object belongs.
594      *                Must not be <code>null</code>. If the resulting
595      *                object is an instance of ProjectComponent, its
596      *                Project reference is set to this parameter value.
597      * @param parent  Parent object used to create the instance.
598      *                Must not be <code>null</code>.
599      * @param elementName Name of the element to create an instance of.
600      *                    Must not be <code>null</code>.
601      *
602      * @return an instance of the specified element type
603      * @deprecated since 1.6.x.
604      *             This is not a namespace aware method.
605      *
606      * @exception BuildException if no method is available to create the
607      *                           element instance, or if the creating method fails.
608      */
609     public Object createElement(Project project, Object parent, String elementName)
610             throws BuildException {
611         NestedCreator nc = getNestedCreator(project, "", parent, elementName, null);
612         try {
613             Object nestedElement = nc.create(project, parent, null);
614             if (project != null) {
615                 project.setProjectReference(nestedElement);
616             }
617             return nestedElement;
618         } catch (IllegalAccessException ie) {
619             // impossible as getMethods should only return public methods
620             throw new BuildException(ie);
621         } catch (InstantiationException ine) {
622             // impossible as getMethods should only return public methods

1005     /**
1006      * Creates an implementation of AttributeSetter for the given
1007      * attribute type. Conversions (where necessary) are automatically
1008      * made for the following types:
1009      * <ul>
1010      * <li>String (left as it is)
1011      * <li>Character/char (first character is used)
1012      * <li>Boolean/boolean
1013      * ({@link Project#toBoolean(String) Project.toBoolean(String)} is used)
1014      * <li>Class (Class.forName is used)
1015      * <li>File (resolved relative to the appropriate project)
1016      * <li>Path (resolve relative to the appropriate project)
1017      * <li>Resource (resolved as a FileResource relative to the appropriate project)
1018      * <li>FileProvider (resolved as a FileResource relative to the appropriate project)
1019      * <li>EnumeratedAttribute (uses its own
1020      * {@link EnumeratedAttribute#setValue(String) setValue} method)
1021      * <li>Other primitive types (wrapper classes are used with constructors
1022      * taking String)
1023      * </ul>
1024      *
1025      * If none of the above covers the given parameters, a constructor for the
1026      * appropriate class taking a String parameter is used if it is available.
1027      *
1028      * @param m The method to invoke on the bean when the setter is invoked.
1029      *          Must not be <code>null</code>.
1030      * @param arg The type of the single argument of the bean's method.
1031      *            Must not be <code>null</code>.
1032      * @param attrName the name of the attribute for which the setter is being
1033      *                 created.
1034      *
1035      * @return an appropriate AttributeSetter instance, or <code>null</code>
1036      *         if no appropriate conversion is available.
1037      */
1038     private AttributeSetter createAttributeSetter(final Method m,
1039                                                   Class<?> arg,
1040                                                   final String attrName) {
1041         // use wrappers for primitive classes, e.g. int and
1042         // Integer are treated identically
1043         final Class<?> reflectedArg = PRIMITIVE_TYPE_MAP.containsKey(arg)
1044             ? PRIMITIVE_TYPE_MAP.get(arg) : arg;
1045 
1046         // Object.class - it gets handled differently by AttributeSetter
1047         if (java.lang.Object.class == reflectedArg) {
1048             return new AttributeSetter(m, arg) {
1049                 public void set(Project p, Object parent, String value)
1050                         throws InvocationTargetException,
1051                     IllegalAccessException {
1052                     throw new BuildException(
1053                         "Internal ant problem - this should not get called");
1054                 }
1055             };
1056         }
1057         // simplest case - setAttribute expects String
1058         if (java.lang.String.class.equals(reflectedArg)) {
1059             return new AttributeSetter(m, arg) {
1060                 public void set(Project p, Object parent, String value)
1061                         throws InvocationTargetException, IllegalAccessException {
1062                     m.invoke(parent, (Object[]) new String[] {value});
1063                 }
1064             };
1065         }
1066         // char and Character get special treatment - take the first character
1067         if (java.lang.Character.class.equals(reflectedArg)) {
1068             return new AttributeSetter(m, arg) {
1069                 public void set(Project p, Object parent, String value)
1070                         throws InvocationTargetException, IllegalAccessException {
1071                     if (value.length() == 0) {
1072                         throw new BuildException("The value \"\" is not a "
1073                                 + "legal value for attribute \"" + attrName + "\"");
1074                     }
1075                     m.invoke(parent, (Object[]) new Character[] {new Character(value.charAt(0))});
1076                 }
1077             };
1078         }
1079         // boolean and Boolean get special treatment because we have a nice method in Project
1080         if (java.lang.Boolean.class.equals(reflectedArg)) {
1081             return new AttributeSetter(m, arg) {
1082                 public void set(Project p, Object parent, String value)
1083                         throws InvocationTargetException, IllegalAccessException {
1084                     m.invoke(parent, (Object[]) new Boolean[] {
1085                             Project.toBoolean(value) ? Boolean.TRUE : Boolean.FALSE });
1086                 }
1087             };
1088         }
1089         // Class doesn't have a String constructor but a decent factory method
1090         if (java.lang.Class.class.equals(reflectedArg)) {
1091             return new AttributeSetter(m, arg) {
1092                 public void set(Project p, Object parent, String value)
1093                         throws InvocationTargetException, IllegalAccessException, BuildException {
1094                     try {
1095                         m.invoke(parent, new Object[] {Class.forName(value)});
1096                     } catch (ClassNotFoundException ce) {
1097                         throw new BuildException(ce);
1098                     }
1099                 }
1100             };
1101         }
1102         // resolve relative paths through Project
1103         if (java.io.File.class.equals(reflectedArg)) {
1104             return new AttributeSetter(m, arg) {
1105                 public void set(Project p, Object parent, String value)
1106                         throws InvocationTargetException, IllegalAccessException {
1107                     m.invoke(parent, new Object[] {p.resolveFile(value)});
1108                 }
1109             };
1110         }
1111         // resolve Resources/FileProviders as FileResources relative to Project:
1112         if (Resource.class.equals(reflectedArg) || FileProvider.class.equals(reflectedArg)) {
1113             return new AttributeSetter(m, arg) {
1114                 void set(Project p, Object parent, String value) throws InvocationTargetException,
1115                         IllegalAccessException, BuildException {
1116                     m.invoke(parent, new Object[] { new FileResource(p, p.resolveFile(value)) });
1117                 };
1118             };
1119         }
1120         // EnumeratedAttributes have their own helper class
1121         if (EnumeratedAttribute.class.isAssignableFrom(reflectedArg)) {
1122             return new AttributeSetter(m, arg) {
1123                 public void set(Project p, Object parent, String value)
1124                         throws InvocationTargetException, IllegalAccessException, BuildException {
1125                     try {
1126                         EnumeratedAttribute ea = (EnumeratedAttribute) reflectedArg.newInstance();
1127                         ea.setValue(value);
1128                         m.invoke(parent, new Object[] {ea});
1129                     } catch (InstantiationException ie) {
1130                         throw new BuildException(ie);
1131                     }
1132                 }
1133             };
1134         }
1135 
1136         AttributeSetter setter = getEnumSetter(reflectedArg, m, arg);
1137         if (setter != null) {
1138             return setter;
1139         }
1140 
1141         if (java.lang.Long.class.equals(reflectedArg)) {
1142             return new AttributeSetter(m, arg) {
1143                 public void set(Project p, Object parent, String value)
1144                         throws InvocationTargetException, IllegalAccessException, BuildException {
1145                     try {
1146                         m.invoke(parent, new Object[] {
1147                                 new Long(StringUtils.parseHumanSizes(value)) });
1148                     } catch (NumberFormatException e) {
1149                         throw new BuildException("Can't assign non-numeric"
1150                                                  + " value '" + value + "' to"
1151                                                  + " attribute " + attrName);
1152                     } catch (InvocationTargetException e) {
1153                         throw e;
1154                     } catch (IllegalAccessException e) {
1155                         throw e;
1156                     } catch (Exception e) {
1157                         throw new BuildException(e);
1158                     }
1159                 }
1160             };
1161         }
1162         // worst case. look for a public String constructor and use it
1163         // also supports new Whatever(Project, String) as for Path or Reference
1164         // This is used (deliberately) for all primitives/wrappers other than
1165         // char, boolean, and long.
1166         boolean includeProject;
1167         Constructor<?> c;
1168         try {
1169             // First try with Project.
1170             c = reflectedArg.getConstructor(Project.class, String.class);
1171             includeProject = true;
1172         } catch (NoSuchMethodException nme) {
1173             // OK, try without.
1174             try {
1175                 c = reflectedArg.getConstructor(String.class);
1176                 includeProject = false;
1177             } catch (NoSuchMethodException nme2) {
1178                 // Well, no matching constructor.
1179                 return null;
1180             }
1181         }
1182         final boolean finalIncludeProject = includeProject;
1183         final Constructor<?> finalConstructor = c;
1184 
1185         return new AttributeSetter(m, arg) {
1186             public void set(Project p, Object parent, String value)
1187                     throws InvocationTargetException, IllegalAccessException, BuildException {
1188                 try {
1189                     Object[] args = finalIncludeProject
1190                             ? new Object[] {p, value} : new Object[] {value};
1191 
1192                     Object attribute = finalConstructor.newInstance(args);
1193                     if (p != null) {
1194                         p.setProjectReference(attribute);
1195                     }
1196                     m.invoke(parent, new Object[] {attribute});
1197                 } catch (InvocationTargetException e) {
1198                     Throwable cause = e.getCause();
1199                     if (cause instanceof IllegalArgumentException) {
1200                         throw new BuildException("Can't assign value '" + value
1201                                                  + "' to attribute " + attrName
1202                                                  + ", reason: "
1203                                                  + cause.getClass()
1204                                                  + " with message '"
1205                                                  + cause.getMessage() + "'");
1206                     }
1207                     throw e;
1208                 } catch (InstantiationException ie) {
1209                     throw new BuildException(ie);
1210                 }
1211             }
1212         };
1213     }

1506     /**
1507      * Create a NestedCreator for the given element.
1508      * @param project owning project
1509      * @param parent Parent object used to create the instance.
1510      * @param elementName name of the element
1511      * @return a nested creator, or null if there is no component of the given name, or it
1512      *        has no matching add type methods
1513      * @throws BuildException
1514      */
1515     private NestedCreator createAddTypeCreator(
1516             Project project, Object parent, String elementName) throws BuildException {
1517         if (addTypeMethods.size() == 0) {
1518             return null;
1519         }
1520         ComponentHelper helper = ComponentHelper.getComponentHelper(project);
1521 
1522         MethodAndObject restricted =  createRestricted(
1523             helper, elementName, addTypeMethods);
1524         MethodAndObject topLevel = createTopLevel(
1525             helper, elementName, addTypeMethods);
1526 
1527         if (restricted == null && topLevel == null) {
1528             return null;
1529         }
1530 
1531         if (restricted != null && topLevel != null) {
1532             throw new BuildException(
1533                 "ambiguous: type and component definitions for "
1534                 + elementName);
1535         }
1536 
1537         MethodAndObject methodAndObject
1538             = restricted != null ? restricted : topLevel;
1539 
1540         Object rObject = methodAndObject.object;
1541         if (methodAndObject.object instanceof PreSetDef.PreSetDefinition) {
1542             rObject = ((PreSetDef.PreSetDefinition) methodAndObject.object)
1543                 .createObject(project);
1544         }
1545         final Object nestedObject = methodAndObject.object;
1546         final Object realObject = rObject;
1547 
1548         return new NestedCreator(methodAndObject.method) {
1549             Object create(Project project, Object parent, Object ignore)
1550                     throws InvocationTargetException, IllegalAccessException {
1551                 if (!getMethod().getName().endsWith("Configured")) {
1552                     getMethod().invoke(parent, new Object[] {realObject});
1553                 }
1554                 return nestedObject;
1555             }
1556 
1557             Object getRealObject() {
1558                 return realObject;
1559             }
1560 
1561             void store(Object parent, Object child) throws InvocationTargetException,
1562                     IllegalAccessException, InstantiationException {
1563                 if (getMethod().getName().endsWith("Configured")) {
1564                     getMethod().invoke(parent, new Object[] {realObject});
1565                 }
1566             }
1567         };
1568     }

1570     /**
1571      * Inserts an add or addConfigured method into
1572      * the addTypeMethods array. The array is
1573      * ordered so that the more derived classes are first.
1574      * If both add and addConfigured are present, the addConfigured will take priority.
1575      * @param method the <code>Method</code> to insert.
1576      */
1577     private void insertAddTypeMethod(Method method) {
1578         Class<?> argClass = method.getParameterTypes()[0];
1579         final int size = addTypeMethods.size();
1580         for (int c = 0; c < size; ++c) {
1581             Method current = (Method) addTypeMethods.get(c);
1582             if (current.getParameterTypes()[0].equals(argClass)) {
1583                 if (method.getName().equals("addConfigured")) {
1584                     // add configured replaces the add method
1585                     addTypeMethods.set(c, method);
1586                 }
1587                 return; // Already present
1588             }
1589             if (current.getParameterTypes()[0].isAssignableFrom(argClass)) {
1590                 addTypeMethods.add(c, method);
1591                 return; // higher derived
1592             }
1593         }
1594         addTypeMethods.add(method);
1595     }

1597     /**
1598      * Search the list of methods to find the first method
1599      * that has a parameter that accepts the nested element object.
1600      * @param paramClass the <code>Class</code> type to search for.
1601      * @param methods the <code>List</code> of methods to search.
1602      * @return a matching <code>Method</code>; null if none found.
1603      */
1604     private Method findMatchingMethod(Class<?> paramClass, List<Method> methods) {
1605         if (paramClass == null) {
1606             return null;
1607         }
1608         Class<?> matchedClass = null;
1609         Method matchedMethod = null;
1610 
1611         final int size = methods.size();
1612         for (int i = 0; i < size; ++i) {
1613             Method method = methods.get(i);
1614             Class<?> methodClass = method.getParameterTypes()[0];
1615             if (methodClass.isAssignableFrom(paramClass)) {
1616                 if (matchedClass == null) {
1617                     matchedClass = methodClass;
1618                     matchedMethod = method;
1619                 } else if (!methodClass.isAssignableFrom(matchedClass)) {
1620                     throw new BuildException("ambiguous: types " + matchedClass.getName() + " and "
1621                             + methodClass.getName() + " match " + paramClass.getName());
1622                 }
1623             }
1624         }
1625         return matchedMethod;
1626     }

1649     private AntTypeDefinition findRestrictedDefinition(
1650         ComponentHelper helper, String componentName, List<Method> methods) {
1651         AntTypeDefinition definition = null;
1652         Class<?> matchedDefinitionClass = null;
1653 
1654         List<AntTypeDefinition> definitions = helper.getRestrictedDefinitions(componentName);
1655         if (definitions == null) {
1656             return null;
1657         }
1658         synchronized (definitions) {
1659             final int size = definitions.size();
1660             for (int i = 0; i < size; ++i) {
1661                 AntTypeDefinition d = definitions.get(i);
1662                 Class<?> exposedClass = d.getExposedClass(helper.getProject());
1663                 if (exposedClass == null) {
1664                     continue;
1665                 }
1666                 Method method  = findMatchingMethod(exposedClass, methods);
1667                 if (method == null) {
1668                     continue;
1669                 }
1670                 if (matchedDefinitionClass != null) {
1671                     throw new BuildException(
1672                         "ambiguous: restricted definitions for "
1673                         + componentName + " "
1674                         + matchedDefinitionClass + " and " + exposedClass);
1675                 }
1676                 matchedDefinitionClass = exposedClass;
1677                 definition = d;
1678             }
1679         }
1680         return definition;
1681     }

1683     private MethodAndObject createRestricted(
1684         ComponentHelper helper, String elementName, List<Method> addTypeMethods) {
1685 
1686         Project project = helper.getProject();
1687 
1688         AntTypeDefinition restrictedDefinition =
1689             findRestrictedDefinition(helper, elementName, addTypeMethods);
1690 
1691         if (restrictedDefinition == null) {
1692             return null;
1693         }
1694 
1695         Method addMethod = findMatchingMethod(
1696             restrictedDefinition.getExposedClass(project), addTypeMethods);
1697         if (addMethod == null) {
1698             throw new BuildException(
1699                 "Ant Internal Error - contract mismatch for "
1700                 + elementName);
1701         }
1702         Object addedObject = restrictedDefinition.create(project);
1703         if (addedObject == null) {
1704             throw new BuildException(
1705                 "Failed to create object " + elementName
1706                 + " of type " + restrictedDefinition.getTypeClass(project));
1707         }
1708         return new MethodAndObject(addMethod, addedObject);
1709     }

1711     private MethodAndObject createTopLevel(
1712         ComponentHelper helper, String elementName, List<Method> methods) {
1713         Class<?> clazz = helper.getComponentClass(elementName);
1714         if (clazz == null) {
1715             return null;
1716         }
1717         Method addMethod = findMatchingMethod(clazz, addTypeMethods);
1718         if (addMethod == null) {
1719             return null;
1720         }
1721         Object addedObject = helper.createComponent(elementName);
1722         return new MethodAndObject(addMethod, addedObject);

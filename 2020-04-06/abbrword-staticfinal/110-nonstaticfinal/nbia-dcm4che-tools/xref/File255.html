100     private void initRefReqSq( Dataset ds ) {
101         refRequests = new ArrayList();
102         DcmElement e = ds.get( Tags.RefRequestSeq );
103         if ( e != null ) {
104             for ( int i=0, len=e.countItems() ; i < len ; i++ ) {
105                 refRequests.add( new RefRequest( e.getItem(i)));
106             }
107         }
108     }

114             for ( int i=0, len=e.countItems() ; i < len ; i++ ) {
115                 gpspsList.add( new GPSPS( e.getItem(i)));
116             }
117         }

119     }

121     private void initResults( Dataset ds ) {
122         results = new ArrayList();
123         DcmElement e = ds.get( Tags.OutputInformationSeq );
124         if ( e != null ) {
125             for ( int i=0, len=e.countItems() ; i < len ; i++ ) {
126                 results.addAll( getRefIUIDs( e.getItem(i)));
127             }
128         }
129     }

131     private List getRefIUIDs(Dataset item) {
132         List l = new ArrayList();
133         DcmElement series = ds.get( Tags.RefSeriesSeq );
134         DcmElement instances;
135         if ( series != null ) {
136             for ( int i=0, len=series.countItems() ; i < len ; i++ ) {
137                 instances = series.getItem(i).get(Tags.RefSOPSeq);
138                 if ( instances != null ) {
139                     for ( int j = 0, len1=instances.countItems() ; j < len1 ; j++ ) {
140                         l.add( instances.getItem(j).getString(Tags.RefSOPInstanceUID));
141                     }
142                 }
143             }
144         }
145         return l;
146     }

271     private String getCodeValue( DcmElement elem ) {
272         if ( elem == null || elem.countItems() < 1) return null;
273         StringBuffer sb = new StringBuffer();
274         Dataset dsCode = elem.getItem();
275         sb.append( dsCode.getString( Tags.CodeMeaning ) ).append("[");
276         sb.append( dsCode.getString( Tags.CodeValue ) ).append("]");
277         return sb.toString();
278 
279     }
280     private String[] getCodeValues( DcmElement elem ) {
281         if ( elem == null || elem.countItems() < 1 ) return null;
282         int len = elem.countItems();
283         Dataset dsCode;
284         StringBuffer sb = new StringBuffer();
285         String[] sa = new String[len];
286         for ( int i = 0 ; i < len ; i++ ) {
287             dsCode = elem.getItem( i );
288             sb.setLength(0);
289             sb.append( dsCode.getString( Tags.CodeMeaning ) ).append("[");
290             sb.append( dsCode.getString( Tags.CodeValue ) ).append("]");
291             sa[i] = sb.toString();
292         }
293         return sa;
294 
295     }

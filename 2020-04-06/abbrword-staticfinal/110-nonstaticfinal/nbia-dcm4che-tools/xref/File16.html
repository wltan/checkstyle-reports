41  import java.util.ArrayList;
42  import java.util.Arrays;
43  
44  import javax.xml.parsers.SAXParserFactory;
45  
46  import org.dcm4che.dict.Tags;
47  import org.dcm4che.dict.VRs;
48  import org.dcm4che2.soundex.FuzzyStr;
49  import org.dcm4chex.archive.exceptions.ConfigurationException;
50  import org.xml.sax.Attributes;
51  import org.xml.sax.SAXException;
52  import org.xml.sax.helpers.DefaultHandler;

54  /**
55   * @author Gunter Zeilinger <gunterze@gmail.com>
56   * @version $Id: AttributeFilterLoader.java 14967 2011-02-28 09:42:54Z gunterze $
57   * @since Jul 4, 2006

59  class AttributeFilterLoader extends DefaultHandler {
60      private static final int[] INST_SUPPL_TAGS = { Tags.RetrieveAET,
61              Tags.InstanceAvailability, Tags.StorageMediaFileSetID,
62              Tags.StorageMediaFileSetUID };
63      private final ArrayList<String> tagList = new ArrayList<String>();
64      private final ArrayList<String> vrList = new ArrayList<String>();
65      private final ArrayList<String> noCoerceList = new ArrayList<String>();
66      private final ArrayList<String> iCaseList = new ArrayList<String>();
67      private final ArrayList<String> fieldTagList = new ArrayList<String>();
68      private final ArrayList<String> fieldList = new ArrayList<String>();
69      private AttributeFilter filter;
70  
71      public static void loadFrom(String url) throws ConfigurationException {
72          AttributeFilterLoader h = new AttributeFilterLoader();
73          try {
74              SAXParserFactory.newInstance().newSAXParser().parse(url, h);
75          } catch (Exception e) {
76              throw new ConfigurationException(
77                      "Failed to load attribute filter from " + url, e);
78          }
79      }
80  
81      public void startElement(String uri, String localName, String qName,
82              Attributes attributes) throws SAXException {
83          if (qName.equals("attr")) {
84              String tag = attributes.getValue("tag");
85              if (tag != null) {
86                  if (attributes.getValue("seq") == null) {
87                      tagList.add(tag);
88                      String field = attributes.getValue("field");
89                      if (field != null) {
90                          fieldTagList.add(tag);
91                          fieldList.add(field);
92                      }
93                      if ("false".equalsIgnoreCase(attributes.getValue("coerce")))
94                          noCoerceList.add(tag);
95                  }
96                  if ("false".equalsIgnoreCase(attributes.getValue("case-sensitive")))
97                      iCaseList.add(tag);
98              } else {
99                  String vr = attributes.getValue("vr");
100                 if (vr != null) {
101                     vrList.add(vr);
102                 }
103             }
104         } else if (qName.equals("instance")) {
105             String cuid = attributes.getValue("cuid");
106             if (AttributeFilter.instanceFilters.containsKey(cuid)) {
107                 throw new SAXException(
108                         "more than one instance element with cuid=" + cuid);
109             }
110             AttributeFilter.instanceFilters.put(cuid,
111                     filter = makeFilter(attributes));
112             String s = attributes.getValue("content-item-text-value-max-length");
113             try {
114                 filter.setContentItemTextValueMaxLength(Integer.parseInt(s));
115             } catch (IllegalArgumentException e) {
116                 throw new SAXException(
117                         "illegal content-item-text-value-max-length: " + s);
118             }
119         } else if (qName.equals("series")) {
120             if (AttributeFilter.seriesFilter != null) {
121                 throw new SAXException("more than one series element");
122             }
123             AttributeFilter.seriesFilter = filter = makeFilter(attributes);
124         } else if (qName.equals("study")) {
125             if (AttributeFilter.studyFilter != null) {
126                 throw new SAXException("more than one study element");
127             }
128             AttributeFilter.studyFilter = filter = makeFilter(attributes);
129         } else if (qName.equals("patient")) {
130             if (AttributeFilter.patientFilter != null) {
131                 throw new SAXException("more than one patient element");
132             }
133             AttributeFilter.patientFilter = filter = makeFilter(attributes);
134         } else if (qName.equals("exclude-patient")) {
135             if (AttributeFilter.excludePatientFilter != null) {
136                 throw new SAXException("more than one exclude-patient element");
137             }
138             AttributeFilter.excludePatientFilter = filter = makeFilter(attributes);
139         } else if (qName.equals("soundex")) {
140             String clazz = attributes.getValue("class");
141             try {
142                 AttributeFilter.soundex =
143                     (FuzzyStr) Class.forName(clazz).newInstance();
144                 AttributeFilter.soundexWithTrailingWildCard = 
145                     "true".equalsIgnoreCase(attributes.getValue("trailing-wildcard"));
146                 
147             } catch (Exception e) {
148                 throw new SAXException(
149                         "Failed to instantiate Soundex implementation: "
150                         + clazz, e);
151             }
152         }
153     }
154 
155     private AttributeFilter makeFilter(Attributes attributes) {
156         String strategy = attributes.getValue("update-strategy");
157         return new AttributeFilter(attributes.getValue("tsuid"), 
158                 "true".equalsIgnoreCase(attributes.getValue("exclude")),
159                 "true".equalsIgnoreCase(attributes.getValue("excludePrivate")),
160                 strategy.startsWith("overwrite"), strategy.endsWith("merge"));
161     }
162 
163     public void endElement(String uri, String localName, String qName)
164             throws SAXException {
165         if (qName.equals("attr")) {
166             return;
167         }
168         boolean inst = qName.equals("instance");
169         if (inst || qName.equals("series") || qName.equals("study")
170                 || qName.equals("patient") || qName.equals("exclude-patient")) {
171             int[] tags = parseInts(tagList, true);
172             int[] vrs = parseVRs(vrList);
173             if (inst && filter.isExclude()) {
174                 if (AttributeFilter.patientFilter == null) {
175                     throw new SAXException(
176                             "missing patient before instance element");
177                 }
178                 if (AttributeFilter.studyFilter == null) {
179                     throw new SAXException(
180                             "missing study before instance element");
181                 }
182                 if (AttributeFilter.seriesFilter == null) {
183                     throw new SAXException(
184                             "missing series before instance element");
185                 }
186                 tags = merge(INST_SUPPL_TAGS, AttributeFilter.patientFilter
187                         .getTags(), AttributeFilter.studyFilter.getTags(),
188                         AttributeFilter.seriesFilter.getTags(), tags);
189                 vrs = merge(new int[] {}, AttributeFilter.patientFilter
190                         .getVRs(), AttributeFilter.studyFilter.getVRs(),
191                         AttributeFilter.seriesFilter.getVRs(), vrs);
192             }
193             filter.setTags(tags);
194             filter.setFieldTags(parseInts(fieldTagList, false));
195             filter.setFields((String[]) fieldList.toArray(new String[] {}));
196             filter.setNoCoercion(parseInts(noCoerceList, true));
197             filter.setICase(parseInts(iCaseList, true));
198             filter.setVRs(vrs);
199             tagList.clear();
200             fieldTagList.clear();
201             fieldList.clear();
202             noCoerceList.clear();
203             vrList.clear();
204             filter = null;
205         }
206     }
207 
208     private int[] merge(int[] a, int[] b, int[] c, int[] d, int[] e) {
209         int[] dst = new int[a.length + b.length + c.length + d.length
210                 + e.length];
211         System.arraycopy(a, 0, dst, 0, a.length);
212         System.arraycopy(b, 0, dst, a.length, b.length);
213         System.arraycopy(c, 0, dst, a.length + b.length, c.length);
214         System.arraycopy(d, 0, dst, a.length + b.length + c.length, d.length);
215         System.arraycopy(e, 0, dst, a.length + b.length + c.length + d.length,
216                 e.length);
217         Arrays.sort(dst);
218         return dst;
219     }
220 
221     private static int[] parseInts(ArrayList<String> list, boolean sort) {
222         int[] array = new int[list.size()];
223         for (int i = 0; i < array.length; i++) {
224             array[i] = Integer.parseInt(list.get(i), 16);
225         }
226         if (sort) {
227             Arrays.sort(array);
228         }
229         return array;
230     }
231 
232     private static int[] parseVRs(ArrayList<String> list) {
233         int[] array = new int[list.size()];
234         for (int i = 0; i < array.length; i++) {
235             array[i] = VRs.valueOf(list.get(i));
236         }
237         return array;
238     }
239 
240     public void endDocument() throws SAXException {
241         if (AttributeFilter.excludePatientFilter == null) {
242             throw new SAXException("missing exclude-patient element");
243         }
244         if (AttributeFilter.patientFilter == null) {
245             throw new SAXException("missing patient element");
246         }
247         if (AttributeFilter.studyFilter == null) {
248             throw new SAXException("missing study element");
249         }
250         if (AttributeFilter.seriesFilter == null) {
251             throw new SAXException("missing series element");
252         }
253         if (AttributeFilter.instanceFilters.get(null) == null) {
254             throw new SAXException("missing instance element");
255         }
256     }

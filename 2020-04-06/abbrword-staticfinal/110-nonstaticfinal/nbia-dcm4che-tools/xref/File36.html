58  /**
59   * Create command for use with Firebird that uses a sequence (generator).
60   * 
61   * The sequence (generator) is called by the parameter attribute "sequence_name".
62   * As an example, the sequence_name could be %%t_sequence to use %lt;table_name&gt;_sequence
63   * for each distinct table.
64   * 
65   * <i>The code was derived from:
66   *    <ul>
67   *     <li>JBoss class org.jboss.ejb.plugins.cmp.jdbc.keygen.JDBCOracleSequenceCreateCommand
68   *     <li>Class from Petr Kalina - http://www.jboss.com/index.html?module=bb&op=viewtopic&t=77075 
69   *    </ul>
70   * </i>
71   * 
72   * @author Jan Pechanec <jpechanec@orcz.cz>
73   * @author Petr Kalina <petr.kalina@jlabs.cz>
74   */
75  public class JDBCFirebirdSequenceCreateCommand extends JDBCInsertPKCreateCommand
76  {
77      private String sequence_name;
78      private String sequenceSQL;
79  
80      protected JDBCCMPFieldBridge pkField;
81      
82      /*private int pkIndex;
83      private int jdbcType;*/
84  
85      public JDBCFirebirdSequenceCreateCommand() {
86          super();
87      }
88  
89      protected void initGeneratedFields() throws DeploymentException
90      {
91         super.initGeneratedFields();
92         pkField = getGeneratedPKField();
93      }
94  
95      public void init(JDBCStoreManager manager) throws DeploymentException
96      {
97         super.init(manager);
98      }
99  
100     protected void initEntityCommand(JDBCEntityCommandMetaData entityCommand) throws DeploymentException
101     {
102        super.initEntityCommand(entityCommand);
103 
104        sequence_name = entityCommand.getAttribute("sequence_name");
105        if (sequence_name == null) {
106           throw new DeploymentException("sequence_name attribute must be specified inside <entity-command>");
107        }
108 
109        String sequence_name_inst = replaceTable(sequence_name,entity.getTableName());
110        
111        sequenceSQL = "SELECT GEN_ID("+sequence_name_inst+",1) FROM RDB$DATABASE;";
112        if (debug) {
113           log.debug("SEQUENCE SQL is :"+sequenceSQL);
114        }
115     }
116    
117     /**
118       * Replace %%t in the sql command with the current table name
119       *
120       * @param in sql statement with possible %%t to substitute with table name
121       * @param table the table name
122       * @return String with sql statement
123       */
124     private static String replaceTable(String in, String table)
125     {
126         int pos;
127 
128         pos = in.indexOf("%%t");
129         // No %%t -> return input
130         if(pos == -1)
131             return in;
132 
133         String first = in.substring(0, pos);
134         String last = in.substring(pos + 3);
135 
136         return first + table + last;
137     }
138    
139     protected void generateFields(EntityEnterpriseContext ctx) throws CreateException
140     {
141        super.generateFields(ctx);
142 
143        Connection con = null;
144        Statement s = null;
145        ResultSet rs = null;
146        try
147        {
148           if(debug)
149           {
150              log.debug("Executing SQL: " + sequenceSQL);
151           }
152 
153           DataSource dataSource = entity.getDataSource();
154           con = dataSource.getConnection();
155           s = con.createStatement();
156 
157           rs = s.executeQuery(sequenceSQL);
158           if(!rs.next())
159           {
160              throw new CreateException("Error fetching next primary key value: result set contains no rows");
161           }
162           pkField.loadInstanceResults(rs, 1, ctx);
163        }
164        catch(SQLException e)
165        {
166           log.error("Error fetching the next primary key value", e);
167           throw new CreateException("Error fetching the next primary key value:" + e);
168        }
169        finally
170        {
171           JDBCUtil.safeClose(rs);
172           JDBCUtil.safeClose(s);
173           JDBCUtil.safeClose(con);
174        }
175     }
176

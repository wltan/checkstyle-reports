33  /**
34   * <!-- START SNIPPET: javadoc -->
35   * <p>
36   * Create a option transfer select component which is basically two &lt;select ...&gt;
37   * tag with buttons in the middle of them allowing options in each of the
38   * &lt;select ...&gt; to be moved between themselves. Will auto-select all its
39   * elements upon its containing form submision.
40   * </p>
41   * <!-- END SNIPPET: javadoc -->
42   *
43   * <!-- START SNIPPET: notice -->
44   * <p>
45   * NOTE: The id and doubleId need not be supplied as they will generated provided
46   * that the optiontransferselect tag is being used in a form tag. The generated id
47   * and doubleId will be &lt;form_id&gt;_&lt;optiontransferselect_nameame&gt; and
48   * &lt;form_id&gt;_&lt;optiontransferselect_doubleName&gt; respectively.
49   * </p>
50   *
51   * <!-- END SNIPPET: notice -->
52   *
53   * <pre>
54   * <!-- START SNIPPET: example -->
55   *
56   * &lt;-- minimum configuration --&gt;
57   * &lt;s:optiontransferselect
58   *      label="Favourite Cartoons Characters"
59   *      name="leftSideCartoonCharacters"
60   *      list="{'Popeye', 'He-Man', 'Spiderman'}"
61   *      doubleName="rightSideCartoonCharacters"
62   *      doubleList="{'Superman', 'Mickey Mouse', 'Donald Duck'}"
63   *  /&gt;
64   *
65   *  &lt;-- possible configuration --&gt;
66   *  &lt;s:optiontransferselect
67   *      label="Favourite Cartoons Characters"
68   *      name="leftSideCartoonCharacters"
69   *      leftTitle="Left Title"
70   *      rightTitle="Right Title"
71   *      list="{'Popeye', 'He-Man', 'Spiderman'}"
72   *      multiple="true"
73   *      headerKey="headerKey"
74   *      headerValue="--- Please Select ---"
75   *      emptyOption="true"
76   *      doubleList="{'Superman', 'Mickey Mouse', 'Donald Duck'}"
77   *      doubleName="rightSideCartoonCharacters"
78   *      doubleHeaderKey="doubleHeaderKey"
79   *      doubleHeaderValue="--- Please Select ---"
80   *      doubleEmptyOption="true"
81   *      doubleMultiple="true"
82   *  /&gt;
83   *
84   * <!-- END SNIPPET: example -->
85   * </pre>
86   *
87   */

137     public void evaluateExtraParams() {
138         super.evaluateExtraParams();
139 
140         Object doubleValue = null;
141 
142         // override DoubleListUIBean's
143         if (doubleList != null) {
144             doubleValue = findValue(doubleList);
145             addParameter("doubleList", doubleValue);
146         }
147         if (StringUtils.isBlank(size)) {
148             addParameter("size", "15");
149         }
150         if (StringUtils.isBlank(doubleSize)) {
151             addParameter("doubleSize", "15");
152         }
153         if (StringUtils.isBlank(multiple)) {
154             addParameter("multiple", Boolean.TRUE);
155         }
156         if (StringUtils.isBlank(doubleMultiple)) {
157             addParameter("doubleMultiple", Boolean.TRUE);
158         }
159 
160         // buttonCssClass
161         if (StringUtils.isNotBlank(buttonCssClass)) {
162             addParameter("buttonCssClass", buttonCssClass);
163         }
164 
165         // buttonCssStyle
166         if (StringUtils.isNotBlank(buttonCssStyle)) {
167             addParameter("buttonCssStyle", buttonCssStyle);
168         }
169 
170         // allowSelectAll
171         addParameter("allowSelectAll",
172                 allowSelectAll != null ? findValue(allowSelectAll, Boolean.class) : Boolean.TRUE);
173 
174         // allowAddToLeft
175         addParameter("allowAddToLeft",
176                 allowAddToLeft != null ? findValue(allowAddToLeft, Boolean.class) : Boolean.TRUE);
177 
178         // allowAddToRight
179         addParameter("allowAddToRight",
180                 allowAddToRight != null ? findValue(allowAddToRight, Boolean.class) : Boolean.TRUE);
181 
182         // allowAddAllToLeft
183         addParameter("allowAddAllToLeft",
184                 allowAddAllToLeft != null ? findValue(allowAddAllToLeft, Boolean.class) : Boolean.TRUE);
185 
186         // allowAddAllToRight
187         addParameter("allowAddAllToRight",
188                 allowAddAllToRight != null ? findValue(allowAddAllToRight, Boolean.class) : Boolean.TRUE);
189 
190         // allowUpDownOnLeft
191         addParameter("allowUpDownOnLeft",
192                 allowUpDownOnLeft != null ? findValue(allowUpDownOnLeft, Boolean.class) : Boolean.TRUE);
193 
194         // allowUpDownOnRight
195         addParameter("allowUpDownOnRight",
196                 allowUpDownOnRight != null ? findValue(allowUpDownOnRight, Boolean.class) : Boolean.TRUE);
197 
198 
199         // leftTitle
200         if (leftTitle != null) {
201             addParameter("leftTitle", findValue(leftTitle, String.class));
202         }
203 
204         // rightTitle
205         if (rightTitle != null) {
206             addParameter("rightTitle", findValue(rightTitle, String.class));
207         }
208 
209 
210         // addToLeftLabel
211         addParameter("addToLeftLabel",
212                 addToLeftLabel != null ? findValue(addToLeftLabel, String.class) : "<-" );
213 
214         // addToRightLabel
215         addParameter("addToRightLabel",
216                 addToRightLabel != null ? findValue(addToRightLabel, String.class) : "->");
217 
218         // addAllToLeftLabel
219         addParameter("addAllToLeftLabel",
220                 addAllToLeftLabel != null ? findValue(addAllToLeftLabel, String.class) : "<<--");
221 
222         // addAllToRightLabel
223         addParameter("addAllToRightLabel",
224                 addAllToRightLabel != null ? findValue(addAllToRightLabel, String.class) : "-->>");
225 
226         // selectAllLabel
227         addParameter("selectAllLabel",
228                 selectAllLabel != null ? findValue(selectAllLabel, String.class) : "<*>");
229 
230         // leftUpLabel
231         addParameter("leftUpLabel",
232                 leftUpLabel != null ? findValue(leftUpLabel, String.class) : "^");
233 
234 
235         // leftDownLabel
236         addParameter("leftDownLabel",
237                 leftDownlabel != null ? findValue(leftDownlabel, String.class) : "v");
238 
239 
240         // rightUpLabel
241         addParameter("rightUpLabel",
242                 rightUpLabel != null ? findValue(rightUpLabel, String.class) : "^");
243 
244 
245         // rightDownlabel
246         addParameter("rightDownLabel",
247                 rightDownLabel != null ? findValue(rightDownLabel, String.class) : "v");
248 
249 
250         // selectAllOnclick
251         addParameter("selectAllOnclick",
252                 selectAllOnclick != null ? findValue(selectAllOnclick, String.class) : "");
253 
254         // addToLeftOnclick
255         addParameter("addToLeftOnclick",
256                 addToLeftOnclick != null ? findValue(addToLeftOnclick, String.class) : "");
257 
258         // addToRightOnclick
259         addParameter("addToRightOnclick",
260                 addToRightOnclick != null ? findValue(addToRightOnclick, String.class) : "");
261 
262         // addAllToLeftOnclick
263         addParameter("addAllToLeftOnclick",
264                 addAllToLeftOnclick != null ? findValue(addAllToLeftOnclick, String.class) : "");
265 
266         // addAllToRightOnclick
267         addParameter("addAllToRightOnclick",
268                 addAllToRightOnclick != null ? findValue(addAllToRightOnclick, String.class) : "");
269 
270         // upDownOnLeftOnclick
271         addParameter("upDownOnLeftOnclick",
272                 upDownOnLeftOnclick != null ? findValue(upDownOnLeftOnclick, String.class) : "");
273 
274         // upDownOnRightOnclick
275         addParameter("upDownOnRightOnclick",
276                 upDownOnRightOnclick != null ? findValue(upDownOnRightOnclick, String.class) : "");
277 
278 
279 
280         // inform the form component our select tag infos, so they know how to select
281         // its elements upon onsubmit
282         Form formAncestor = (Form) findAncestor(Form.class);
283         if (formAncestor != null) {
284 
285             // inform ancestor form that we are having a customOnsubmit (see form-close.ftl [simple theme])
286             enableAncestorFormCustomOnsubmit();
287 
288 
289             // key -> select tag id, value -> headerKey (if exists)
290             Map formOptiontransferselectIds = (Map) formAncestor.getParameters().get("optiontransferselectIds");
291             Map formOptiontransferselectDoubleIds = (Map) formAncestor.getParameters().get("optiontransferselectDoubleIds");
292 
293             // init lists
294             if (formOptiontransferselectIds == null) {
295                 formOptiontransferselectIds = new LinkedHashMap();
296             }
297             if (formOptiontransferselectDoubleIds == null) {
298                 formOptiontransferselectDoubleIds = new LinkedHashMap();
299             }
300 
301 
302             // id
303             String tmpId = (String) getParameters().get("id");
304             String tmpHeaderKey = (String) getParameters().get("headerKey");
305             if (tmpId != null && (! formOptiontransferselectIds.containsKey(tmpId))) {
306                 formOptiontransferselectIds.put(tmpId, tmpHeaderKey);
307             }
308 
309             // doubleId
310             String tmpDoubleId = (String) getParameters().get("doubleId");
311             String tmpDoubleHeaderKey = (String) getParameters().get("doubleHeaderKey");
312             if (tmpDoubleId != null && (! formOptiontransferselectDoubleIds.containsKey(tmpDoubleId))) {
313                 formOptiontransferselectDoubleIds.put(tmpDoubleId, tmpDoubleHeaderKey);
314             }
315 
316             formAncestor.getParameters().put("optiontransferselectIds", formOptiontransferselectIds);
317             formAncestor.getParameters().put("optiontransferselectDoubleIds", formOptiontransferselectDoubleIds);
318 
319         }
320         else {
321             if (LOG.isWarnEnabled()) {
322         	LOG.warn("form enclosing optiontransferselect "+this+" not found, auto select upon form submit of optiontransferselect will not work");
323             }
324         }
325     }

22  package lombok.ast.grammar;
23  
24  import java.util.List;
25  import java.util.Stack;
26  
27  import com.google.common.collect.Lists;
28  
29  import lombok.ast.AstException;
30  import lombok.ast.Node;
31  import lombok.ast.Position;
32  import lombok.ast.printer.TextFormatter;
33  
34  public class PositionCheckingFormatter extends TextFormatter {
35  	private Stack<Node> nodeStack = new Stack<Node>();
36  	private List<AstException> problems = Lists.newArrayList();
37  	private Source source;
38  	
39  	public PositionCheckingFormatter(Source source) {
40  		this.source = source;
41  		if ("\r\n".equals(System.getProperty("line.separator", "\n")) && source.getRawInput().contains("\r\n")) {
42  			setNewlineText("\r\n");
43  		}
44  	}
45  	
46  	public List<AstException> getProblems() {
47  		List<AstException> p = Lists.newArrayList(problems);
48  		if (!nodeStack.isEmpty()) p.add(new AstException(nodeStack.peek(), "This node was never closed"));
49  		return p;
50  	}
51  	
52  	private void registerNode(Node node) {
53  		nodeStack.add(node);
54  		if (node == null) return;
55  		
56  		Position p = node.getPosition();
57  		if (p.isUnplaced()) problems.add(new AstException(node, String.format(
58  				"this node is unplaced [%s]", node.getClass().getSimpleName())));
59  		else {
60  			int delta = p.getStart() - getCurrentPosition(true);
61  			if (delta != 0) {
62  				int actualPos = getCurrentPosition(true);
63  				int repPos = p.getStart();
64  				reportError(node, "start", actualPos, repPos);
65  			}
66  		}
67  	}
68  	
69  	@Override public void buildBlock(Node node) {
70  		super.buildBlock(node);
71  		registerNode(node);
72  	}
73  	
74  	@Override public void buildInline(Node node) {
75  		super.buildInline(node);
76  		registerNode(node);
77  	}
78  	
79  	@Override public void closeBlock() {
80  		super.closeBlock();
81  		checkNodeClose();
82  	}
83  	
84  	@Override public void closeInline() {
85  		super.closeInline();
86  		checkNodeClose();
87  	}
88  	
89  	private void checkNodeClose() {
90  		Node node = nodeStack.pop();
91  		if (node == null) return;
92  		Position p = node.getPosition();
93  		if (p.isUnplaced()) return;	//opener already generated error.
94  		if (p.getStart() > p.getEnd()) problems.add(new AstException(node, "Node ends before it starts."));
95  		int delta = p.getEnd() - Math.max(getCurrentPosition(false), p.getStart());
96  		if (delta != 0) {
97  			int actualPos = getCurrentPosition(false);
98  			int repPos = p.getEnd();
99  			reportError(node, "end", actualPos, repPos);
100 		}
101 	}
102 	
103 	private static final int RANGE = 12;
104 	private void reportError(Node node, String description, int actualPos, int repPos) {
105 		int delta = repPos - actualPos;
106 		String raw = source.getRawInput();
107 		String actualPrefix = getCharsBeforePosition(actualPos, raw);
108 		String actualPostfix = getCharsAfterPosition(actualPos, raw);
109 		String reportedPrefix = getCharsBeforePosition(repPos, raw);
110 		String reportedPostfix = getCharsAfterPosition(repPos, raw);
111 		problems.add(new AstException(node, String.format(
112 			"this[%s] node's " + description + " is misreported with %+d\nactual(%d): \"%s⊚%s\"\nreported(%d): \"%s⊚%s\"\n%s",
113 			node.getClass().getSimpleName(),
114 			delta, actualPos, actualPrefix, actualPostfix, repPos, reportedPrefix, reportedPostfix, node)));
115 	}
116 	
117 	private String getCharsAfterPosition(int pos, String raw) {
118 		int max = raw.length();
119 		if (pos < 0) pos = 0;
120 		return pos >= max ? "" : raw.substring(pos, Math.min(pos + RANGE, max));
121 	}
122 	
123 	private String getCharsBeforePosition(int pos, String raw) {
124 		int max = raw.length();
125 		return pos > RANGE ? raw.substring(
126 				Math.min(max, pos-RANGE), Math.min(max, pos)) :
127 				raw.substring(0, Math.min(max, pos));
128 	}
129 }

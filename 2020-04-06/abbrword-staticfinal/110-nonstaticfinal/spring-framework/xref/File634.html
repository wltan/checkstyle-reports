18  
19  import java.io.ByteArrayOutputStream;
20  import java.io.UnsupportedEncodingException;
21  import java.util.regex.Matcher;

26  /**
27   * Utility class for URI encoding and decoding based on RFC 3986.
28   * Offers encoding methods for the various URI components.
29   *
30   * <p>All {@code encode*(String, String} methods in this class operate in a similar way:
31   * <ul>
32   * <li>Valid characters for the specific URI component as defined in RFC 3986 stay the same.</li>
33   * <li>All other characters are converted into one or more bytes in the given encoding scheme.
34   * Each of the resulting bytes is written as a hexadecimal string in the "{@code %<i>xy</i>}"
35   * format.</li>
36   * </ul>
37   *
38   * @author Arjen Poutsma
39   * @since 3.0
40   * @see <a href="http://www.ietf.org/rfc/rfc3986.txt">RFC 3986</a>
41   */
42  public abstract class UriUtils {
43  
44  	private static final String SCHEME_PATTERN = "([^:/?#]+):";
45  
46  	private static final String HTTP_PATTERN = "(http|https):";
47  
48  	private static final String USERINFO_PATTERN = "([^@/]*)";
49  
50  	private static final String HOST_PATTERN = "([^/?#:]*)";
51  
52  	private static final String PORT_PATTERN = "(\\d*)";
53  
54  	private static final String PATH_PATTERN = "([^?#]*)";
55  
56  	private static final String QUERY_PATTERN = "([^#]*)";
57  
58  	private static final String LAST_PATTERN = "(.*)";
59  
60  	// Regex patterns that matches URIs. See RFC 3986, appendix B
61  	private static final Pattern URI_PATTERN = Pattern.compile(
62  			"^(" + SCHEME_PATTERN + ")?" + "(//(" + USERINFO_PATTERN + "@)?" + HOST_PATTERN + "(:" + PORT_PATTERN +
63  					")?" + ")?" + PATH_PATTERN + "(\\?" + QUERY_PATTERN + ")?" + "(#" + LAST_PATTERN + ")?");
64  
65  	private static final Pattern HTTP_URL_PATTERN = Pattern.compile(
66  			"^" + HTTP_PATTERN + "(//(" + USERINFO_PATTERN + "@)?" + HOST_PATTERN + "(:" + PORT_PATTERN + ")?" + ")?" +
67  					PATH_PATTERN + "(\\?" + LAST_PATTERN + ")?");
68

72  	/**
73  	 * Encodes the given source URI into an encoded String. All various URI components are
74  	 * encoded according to their respective valid character sets.
75  	 * <p><strong>Note</strong> that this method does not attempt to encode "=" and "&"
76  	 * characters in query parameter names and query parameter values because they cannot
77  	 * be parsed in a reliable way. Instead use:
78  	 * <pre class="code">
79  	 * UriComponents uriComponents = UriComponentsBuilder.fromUri("/path?name={value}").buildAndExpand("a=b");
80  	 * String encodedUri = uriComponents.encode().toUriString();
81  	 * </pre>
82  	 * @param uri the URI to be encoded
83  	 * @param encoding the character encoding to encode to
84  	 * @return the encoded URI
85  	 * @throws IllegalArgumentException when the given uri parameter is not a valid URI
86  	 * @throws UnsupportedEncodingException when the given encoding parameter is not supported
87  	 * @deprecated in favor of {@link UriComponentsBuilder}; see note about query param encoding
88  	 */
89  	@Deprecated
90  	public static String encodeUri(String uri, String encoding) throws UnsupportedEncodingException {
91  		Assert.notNull(uri, "URI must not be null");
92  		Assert.hasLength(encoding, "Encoding must not be empty");
93  		Matcher matcher = URI_PATTERN.matcher(uri);
94  		if (matcher.matches()) {
95  			String scheme = matcher.group(2);
96  			String authority = matcher.group(3);
97  			String userinfo = matcher.group(5);
98  			String host = matcher.group(6);
99  			String port = matcher.group(8);
100 			String path = matcher.group(9);
101 			String query = matcher.group(11);
102 			String fragment = matcher.group(13);
103 			return encodeUriComponents(scheme, authority, userinfo, host, port, path, query, fragment, encoding);
104 		}
105 		else {
106 			throw new IllegalArgumentException("[" + uri + "] is not a valid URI");
107 		}
108 	}
109 
110 	/**
111 	 * Encodes the given HTTP URI into an encoded String. All various URI components are
112 	 * encoded according to their respective valid character sets.
113 	 * <p><strong>Note</strong> that this method does not support fragments ({@code #}),
114 	 * as these are not supposed to be sent to the server, but retained by the client.
115 	 * <p><strong>Note</strong> that this method does not attempt to encode "=" and "&"
116 	 * characters in query parameter names and query parameter values because they cannot
117 	 * be parsed in a reliable way. Instead use:
118 	 * <pre class="code">
119 	 * UriComponents uriComponents = UriComponentsBuilder.fromHttpUrl("/path?name={value}").buildAndExpand("a=b");
120 	 * String encodedUri = uriComponents.encode().toUriString();
121 	 * </pre>
122 	 * @param httpUrl the HTTP URL to be encoded
123 	 * @param encoding the character encoding to encode to
124 	 * @return the encoded URL
125 	 * @throws IllegalArgumentException when the given uri parameter is not a valid URI
126 	 * @throws UnsupportedEncodingException when the given encoding parameter is not supported
127 	 * @deprecated in favor of {@link UriComponentsBuilder}; see note about query param encoding
128 	 */
129 	@Deprecated
130 	public static String encodeHttpUrl(String httpUrl, String encoding) throws UnsupportedEncodingException {
131 		Assert.notNull(httpUrl, "HTTP URL must not be null");
132 		Assert.hasLength(encoding, "Encoding must not be empty");
133 		Matcher matcher = HTTP_URL_PATTERN.matcher(httpUrl);
134 		if (matcher.matches()) {
135 			String scheme = matcher.group(1);
136 			String authority = matcher.group(2);
137 			String userinfo = matcher.group(4);
138 			String host = matcher.group(5);
139 			String portString = matcher.group(7);
140 			String path = matcher.group(8);
141 			String query = matcher.group(10);
142 			return encodeUriComponents(scheme, authority, userinfo, host, portString, path, query, null, encoding);
143 		}
144 		else {
145 			throw new IllegalArgumentException("[" + httpUrl + "] is not a valid HTTP URL");
146 		}
147 	}
148 
149 	/**
150 	 * Encodes the given source URI components into an encoded String. All various URI components
151 	 * are optional, but encoded according to their respective valid character sets.
152 	 * @param scheme the scheme
153 	 * @param authority the authority
154 	 * @param userInfo the user info
155 	 * @param host the host
156 	 * @param port the port
157 	 * @param path the path
158 	 * @param query the query
159 	 * @param fragment the fragment
160 	 * @param encoding the character encoding to encode to
161 	 * @return the encoded URI
162 	 * @throws IllegalArgumentException when the given uri parameter is not a valid URI
163 	 * @throws UnsupportedEncodingException when the given encoding parameter is not supported
164 	 * @deprecated in favor of {@link UriComponentsBuilder}
165 	 */
166 	@Deprecated
167 	public static String encodeUriComponents(String scheme, String authority, String userInfo,
168 			String host, String port, String path, String query, String fragment, String encoding)
169 			throws UnsupportedEncodingException {
170 
171 		Assert.hasLength(encoding, "Encoding must not be empty");
172 		StringBuilder sb = new StringBuilder();
173 
174 		if (scheme != null) {
175 			sb.append(encodeScheme(scheme, encoding));
176 			sb.append(':');
177 		}
178 
179 		if (authority != null) {
180 			sb.append("//");
181 			if (userInfo != null) {
182 				sb.append(encodeUserInfo(userInfo, encoding));
183 				sb.append('@');
184 			}
185 			if (host != null) {
186 				sb.append(encodeHost(host, encoding));
187 			}
188 			if (port != null) {
189 				sb.append(':');
190 				sb.append(encodePort(port, encoding));
191 			}
192 		}
193 
194 		sb.append(encodePath(path, encoding));
195 
196 		if (query != null) {
197 			sb.append('?');
198 			sb.append(encodeQuery(query, encoding));
199 		}
200 
201 		if (fragment != null) {
202 			sb.append('#');
203 			sb.append(encodeFragment(fragment, encoding));
204 		}
205 
206 		return sb.toString();
207 	}

325 	/**
326 	 * Decodes the given encoded source String into an URI. Based on the following rules:
327 	 * <ul>
328 	 * <li>Alphanumeric characters {@code "a"} through {@code "z"}, {@code "A"} through {@code "Z"}, and
329 	 * {@code "0"} through {@code "9"} stay the same.</li>
330 	 * <li>Special characters {@code "-"}, {@code "_"}, {@code "."}, and {@code "*"} stay the same.</li>
331 	 * <li>A sequence "{@code %<i>xy</i>}" is interpreted as a hexadecimal representation of the character.</li>
332 	 * </ul>
333 	 * @param source the source string
334 	 * @param encoding the encoding
335 	 * @return the decoded URI
336 	 * @throws IllegalArgumentException when the given source contains invalid encoded sequences
337 	 * @throws UnsupportedEncodingException when the given encoding parameter is not supported
338 	 * @see java.net.URLDecoder#decode(String, String)
339 	 */
340 	public static String decode(String source, String encoding) throws UnsupportedEncodingException {
341 		Assert.notNull(source, "Source must not be null");
342 		Assert.hasLength(encoding, "Encoding must not be empty");
343 		int length = source.length();
344 		ByteArrayOutputStream bos = new ByteArrayOutputStream(length);
345 		boolean changed = false;
346 		for (int i = 0; i < length; i++) {
347 			int ch = source.charAt(i);
348 			if (ch == '%') {
349 				if ((i + 2) < length) {
350 					char hex1 = source.charAt(i + 1);
351 					char hex2 = source.charAt(i + 2);
352 					int u = Character.digit(hex1, 16);
353 					int l = Character.digit(hex2, 16);
354 					if (u == -1 || l == -1) {
355 						throw new IllegalArgumentException("Invalid encoded sequence \"" + source.substring(i) + "\"");
356 					}
357 					bos.write((char) ((u << 4) + l));
358 					i += 2;
359 					changed = true;
360 				}
361 				else {
362 					throw new IllegalArgumentException("Invalid encoded sequence \"" + source.substring(i) + "\"");
363 				}
364 			}
365 			else {
366 				bos.write(ch);
367 			}
368 		}
369 		return (changed ? new String(bos.toByteArray(), encoding) : source);
370 	}

9   import io.vavr.collection.Stream;
10  import io.vavr.collection.List;
11  import org.assertj.core.api.Assertions;
12  import org.junit.Test;
13  
14  import static org.assertj.core.api.Assertions.assertThat;
15  
16  public class Euler45Test {
17  
18      /**
19       * <strong>Problem 45 Triangular, pentagonal, and hexagonal</strong>
20       *
21       * <p>Triangle, pentagonal, and hexagonal numbers are generated by the following formulae:</p>
22       * <p>Triangle      Tn=n(n+1)/2     1, 3,  6, 10, 15, ...</p>
23       * <p>Pentagonal    Pn=n(3n−1)/2    1, 5, 12, 22, 35, ...</p>
24       * <p>Hexagonal     Hn=n(2n−1)      1, 6, 15, 28, 45, ...</p>
25       *
26       * <p>
27       * It can be verified that T285 = P165 = H143 = 40755.
28       * Find the next triangle number that is also pentagonal and hexagonal.
29       * </p>
30       *
31       * See also <a href="https://projecteuler.net/problem=45">projecteuler.net
32       * problem 45</a>.
33       */
34  
35      @Test
36      public void shouldSolveProblem45() {
37          assertThat(List.of(5, 12, 22, 35)).allMatch(Euler45Test::isPentagonal);
38          assertThat(List.of(3, 11, 21, 36)).allMatch(i -> !isPentagonal(i));
39          Assertions.assertThat(HEXAGONAL.take(5)).containsExactly(6L, 15L, 28L, 45L, 66L);
40  
41          assertThat(HEXAGONAL
42                  .filter(Euler45Test::isPentagonal)
43                  .head())
44                  .isEqualTo(40755);
45  
46          assertThat(HEXAGONAL
47                  .filter(Euler45Test::isPentagonal)
48                  .tail()
49                  .head())
50                  .isEqualTo(1533776805L);
51      }
52  
53      private static final Stream<Long> HEXAGONAL = Stream.from(2L).map(i -> i*(2*i -1));
54  
55      private static boolean isPentagonal(long i) {
56          // If a number k is pentagonal then n(3n−1)/2 = k; for some integer n
57          // by the quadratic formula (1+sqrt(1+4*3*2k))/6 = n
58          long discriminant = 1+24*i;
59          return isPerfectSquare(discriminant) && (1 + flooredRoot(discriminant)) % 6 == 0;
60      }
61  
62      private static boolean isPerfectSquare(long i) {
63          long sqrtFloor = flooredRoot(i);
64          return sqrtFloor*sqrtFloor == i;
65      }
66  
67      private static long flooredRoot(long i) {
68          return (long)Math.sqrt(i+0.5);
69      }
70  }

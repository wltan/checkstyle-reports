80      private static int maxProduct() {
81          return List.of(Tuple.of(0, 1), Tuple.of(1, 1), Tuple.of(1, 0), Tuple.of(1, -1))
82                  .map(d -> maxProduct(d._1, d._2)).max().get();
83      }
84  
85      private static int maxProduct(int dr, int dc) {
86          return range(dr).crossProduct(range(dc)).map(rc -> product(rc._1, rc._2, dr, dc)).max().get();
87      }
88  
89      private static List<Integer> range(int d) {
90          return List.range(0, 20).filter(i -> i + 3 * d >= 0 && i + 3 * d < 20);
91      }
92  
93      private static int product(int row, int col, int dr, int dc) {
94          return Stream.range(0, 4).map(i -> MATRIX.get(row + i * dr).get(col + i * dc)).reduce((i1, i2) -> i1 * i2);
95      }

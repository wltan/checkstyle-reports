7   package io.vavr.collection.euler;
8   
9   import io.vavr.collection.CharSeq;
10  import io.vavr.collection.Stream;
11  import org.junit.Test;
12  
13  import static org.assertj.core.api.Assertions.assertThat;
14  
15  public class Euler40Test {
16  
17      /**
18       * <strong>Problem 40 Champernowne's constant</strong>
19       * <p>
20       * An irrational decimal fraction is created by concatenating the positive
21       * integers:
22       * <p>
23       * 0.12345678910<b>1</b>112131415161718192021...
24       * <p>
25       * It can be seen that the 12<sup>th</sup> digit of the fractional part is
26       * 1.
27       * <p>
28       * If <i>d</i><sub>n</sub> represents the <i>n</i><sup>th</sup> digit of the
29       * fractional part, find the value of the following expression.
30       * <pre>
31       * <i>d</i><sub>1</sub> × <i>d</i><sub>10</sub> × <i>d</i><sub>100</sub> × <i>d</i><sub>1000</sub> × <i>d</i><sub>10000</sub> × <i>d</i><sub>100000</sub> × <i>d</i><sub>1000000</sub>
32       * </pre>
33       * <p>
34       * See also <a href="https://projecteuler.net/problem=40">projecteuler.net
35       * problem 40</a>.
36       */
37      @Test
38      public void shouldSolveProblem40() {
39          assertThat(decimalDigitAtPosition(12)).isEqualTo('1');
40  
41          assertThat(solution()).isEqualTo(210);
42      }
43  
44      private static int solution() {
45          return Stream.iterate(1, i -> i * 10)
46                  .takeWhile(i -> i <= 1_000_000)
47                  .map(Euler40Test::decimalDigitAtPosition)
48                  .map(c -> Character.digit(c, 10))
49                  .fold(1, (i1, i2) -> i1 * i2);
50      }
51  
52      private static char decimalDigitAtPosition(int num) {
53          return FIRST_1_000_000_DECIMALS.get(num - 1);
54      }
55  
56      private static final Stream<Character> FIRST_1_000_000_DECIMALS = Stream.from(1)
57              .map(i -> i.toString())
58              .flatMap(CharSeq::of)
59              .take(1_000_000);
60  }

27  import java.util.List;
28  import java.util.UUID;
29  
30  import org.junit.Test;
31  
32  import org.hibernate.Session;
33  import org.hibernate.Transaction;
34  import org.hibernate.dialect.SQLServerDialect;

37  
38  import static org.junit.Assert.assertFalse;
39  
40  /**
41   * HHH-5715 bug test case: Duplicated entries when using select distinct with join and pagination. The bug has to do
42   * with new {@link SQLServerDialect} that uses row_number function for pagination
43   * 
44   * @author Valotasios Yoryos
45   */
46  @TestForIssue( jiraKey = "HHH-5715" )
47  public class DistinctSelectTest extends BaseCoreFunctionalTestCase {
48  	private static final int NUM_OF_USERS = 30;
49  
50  	@Override
51  	public String[] getMappings() {
52  		return new String[] { "pagination/EntryTag.hbm.xml" };
53  	}
54  
55  	public void feedDatabase() {
56  		List<Tag> tags = new ArrayList<Tag>();
57  
58  		Session s = openSession();
59  		Transaction t = s.beginTransaction();
60  
61  		for (int i = 0; i < 5; i++) {
62  			Tag tag = new Tag("Tag: " + UUID.randomUUID().toString());
63  			tags.add(tag);
64  			s.save(tag);
65  		}
66  
67  		for (int i = 0; i < NUM_OF_USERS; i++) {
68  			Entry e = new Entry("Entry: " + UUID.randomUUID().toString());
69  			e.getTags().addAll(tags);
70  			s.save(e);
71  		}
72  		t.commit();
73  		s.close();
74  	}
75  
76  	@SuppressWarnings( {"unchecked"})
77  	@Test
78  	public void testDistinctSelectWithJoin() {
79  		feedDatabase();
80  
81  		Session s = openSession();
82  
83  		List<Entry> entries = s.createQuery("select distinct e from Entry e join e.tags t where t.surrogate != null order by e.name").setFirstResult(10).setMaxResults(5).list();
84  
85  		// System.out.println(entries);
86  		Entry firstEntry = entries.remove(0);
87  		assertFalse("The list of entries should not contain dublicated Entry objects as we've done a distinct select", entries.contains(firstEntry));
88  
89  		s.close();
90  	}
91  }

24  package org.hibernate;
25  
26  import java.io.Serializable;
27  import java.sql.Connection;
28  import java.util.Map;
29  import java.util.Set;
30  import javax.naming.Referenceable;
31  
32  import org.hibernate.engine.spi.FilterDefinition;
33  import org.hibernate.metadata.ClassMetadata;
34  import org.hibernate.metadata.CollectionMetadata;
35  import org.hibernate.proxy.EntityNotFoundDelegate;
36  import org.hibernate.stat.Statistics;
37  
38  /**
39   * The main contract here is the creation of {@link Session} instances.  Usually
40   * an application has a single {@link SessionFactory} instance and threads
41   * servicing client requests obtain {@link Session} instances from this factory.
42   * <p/>
43   * The internal state of a {@link SessionFactory} is immutable.  Once it is created
44   * this internal state is set.  This internal state includes all of the metadata
45   * about Object/Relational Mapping.
46   * <p/>
47   * Implementors <strong>must</strong> be threadsafe.
48   *
49   * @see org.hibernate.cfg.Configuration
50   *
51   * @author Gavin King
52   * @author Steve Ebersole
53   */
54  public interface SessionFactory extends Referenceable, Serializable {
55  
56  	public interface SessionFactoryOptions {
57  		Interceptor getInterceptor();
58  		EntityNotFoundDelegate getEntityNotFoundDelegate();
59  	}
60  
61  	public SessionFactoryOptions getSessionFactoryOptions();
62  
63  	/**
64  	 * Obtain a {@link Session} builder.
65  	 *
66  	 * @return The session builder
67  	 */
68  	public SessionBuilder withOptions();
69  
70  	/**
71  	 * Open a {@link Session}.
72  	 * <p/>
73  	 * JDBC {@link Connection connection(s} will be obtained from the
74  	 * configured {@link org.hibernate.service.jdbc.connections.spi.ConnectionProvider} as needed
75  	 * to perform requested work.
76  	 *
77  	 * @return The created session.
78  	 *
79  	 * @throws HibernateException Indicates a problem opening the session; pretty rare here.
80  	 */
81  	public Session openSession() throws HibernateException;
82  
83  	/**
84  	 * Obtains the current session.  The definition of what exactly "current"
85  	 * means controlled by the {@link org.hibernate.context.spi.CurrentSessionContext} impl configured
86  	 * for use.
87  	 * <p/>
88  	 * Note that for backwards compatibility, if a {@link org.hibernate.context.spi.CurrentSessionContext}
89  	 * is not configured but JTA is configured this will default to the {@link org.hibernate.context.internal.JTASessionContext}
90  	 * impl.
91  	 *
92  	 * @return The current session.
93  	 *
94  	 * @throws HibernateException Indicates an issue locating a suitable current session.
95  	 */
96  	public Session getCurrentSession() throws HibernateException;
97  
98  	/**
99  	 * Obtain a {@link StatelessSession} builder.
100 	 *
101 	 * @return The stateless session builder
102 	 */
103 	public StatelessSessionBuilder withStatelessOptions();
104 
105 	/**
106 	 * Open a new stateless session.
107 	 *
108 	 * @return The created stateless session.
109 	 */
110 	public StatelessSession openStatelessSession();
111 
112 	/**
113 	 * Open a new stateless session, utilizing the specified JDBC
114 	 * {@link Connection}.
115 	 *
116 	 * @param connection Connection provided by the application.
117 	 *
118 	 * @return The created stateless session.
119 	 */
120 	public StatelessSession openStatelessSession(Connection connection);
121 
122 	/**
123 	 * Retrieve the {@link ClassMetadata} associated with the given entity class.
124 	 *
125 	 * @param entityClass The entity class
126 	 *
127 	 * @return The metadata associated with the given entity; may be null if no such
128 	 * entity was mapped.
129 	 *
130 	 * @throws HibernateException Generally null is returned instead of throwing.
131 	 */
132 	public ClassMetadata getClassMetadata(Class entityClass);
133 
134 	/**
135 	 * Retrieve the {@link ClassMetadata} associated with the given entity class.
136 	 *
137 	 * @param entityName The entity class
138 	 *
139 	 * @return The metadata associated with the given entity; may be null if no such
140 	 * entity was mapped.
141 	 *
142 	 * @throws HibernateException Generally null is returned instead of throwing.
143 	 * @since 3.0
144 	 */
145 	public ClassMetadata getClassMetadata(String entityName);
146 
147 	/**
148 	 * Get the {@link CollectionMetadata} associated with the named collection role.
149 	 *
150 	 * @param roleName The collection role (in form [owning-entity-name].[collection-property-name]).
151 	 *
152 	 * @return The metadata associated with the given collection; may be null if no such
153 	 * collection was mapped.
154 	 *
155 	 * @throws HibernateException Generally null is returned instead of throwing.
156 	 */
157 	public CollectionMetadata getCollectionMetadata(String roleName);
158 
159 	/**
160 	 * Retrieve the {@link ClassMetadata} for all mapped entities.
161 	 *
162 	 * @return A map containing all {@link ClassMetadata} keyed by the
163 	 * corresponding {@link String} entity-name.
164 	 *
165 	 * @throws HibernateException Generally empty map is returned instead of throwing.
166 	 *
167 	 * @since 3.0 changed key from {@link Class} to {@link String}.
168 	 */
169 	public Map<String,ClassMetadata> getAllClassMetadata();
170 
171 	/**
172 	 * Get the {@link CollectionMetadata} for all mapped collections
173 	 *
174 	 * @return a map from <tt>String</tt> to <tt>CollectionMetadata</tt>
175 	 *
176 	 * @throws HibernateException Generally empty map is returned instead of throwing.
177 	 */
178 	public Map getAllCollectionMetadata();
179 
180 	/**
181 	 * Retrieve the statistics fopr this factory.
182 	 *
183 	 * @return The statistics.
184 	 */
185 	public Statistics getStatistics();
186 
187 	/**
188 	 * Destroy this <tt>SessionFactory</tt> and release all resources (caches,
189 	 * connection pools, etc).
190 	 * <p/>
191 	 * It is the responsibility of the application to ensure that there are no
192 	 * open {@link Session sessions} before calling this method as the impact
193 	 * on those {@link Session sessions} is indeterminate.
194 	 * <p/>
195 	 * No-ops if already {@link #isClosed closed}.
196 	 *
197 	 * @throws HibernateException Indicates an issue closing the factory.
198 	 */
199 	public void close() throws HibernateException;
200 
201 	/**
202 	 * Is this factory already closed?
203 	 *
204 	 * @return True if this factory is already closed; false otherwise.
205 	 */
206 	public boolean isClosed();
207 
208 	/**
209 	 * Obtain direct access to the underlying cache regions.
210 	 *
211 	 * @return The direct cache access API.
212 	 */
213 	public Cache getCache();
214 
215 	/**
216 	 * Evict all entries from the second-level cache. This method occurs outside
217 	 * of any transaction; it performs an immediate "hard" remove, so does not respect
218 	 * any transaction isolation semantics of the usage strategy. Use with care.
219 	 *
220 	 * @param persistentClass The entity class for which to evict data.
221 	 *
222 	 * @throws HibernateException Generally will mean that either that
223 	 * 'persisttentClass' did not name a mapped entity or a problem
224 	 * communicating with underlying cache impl.
225 	 *
226 	 * @deprecated Use {@link Cache#evictEntityRegion(Class)} accessed through
227 	 * {@link #getCache()} instead.
228 	 */
229     @Deprecated
230 	public void evict(Class persistentClass) throws HibernateException;
231 
232 	/**
233 	 * Evict an entry from the second-level  cache. This method occurs outside
234 	 * of any transaction; it performs an immediate "hard" remove, so does not respect
235 	 * any transaction isolation semantics of the usage strategy. Use with care.
236 	 *
237 	 * @param persistentClass The entity class for which to evict data.
238 	 * @param id The entity id
239 	 *
240 	 * @throws HibernateException Generally will mean that either that
241 	 * 'persisttentClass' did not name a mapped entity or a problem
242 	 * communicating with underlying cache impl.
243 	 *
244 	 * @deprecated Use {@link Cache#containsEntity(Class, Serializable)} accessed through
245 	 * {@link #getCache()} instead.
246 	 */
247     @Deprecated
248 	public void evict(Class persistentClass, Serializable id) throws HibernateException;
249 
250 	/**
251 	 * Evict all entries from the second-level cache. This method occurs outside
252 	 * of any transaction; it performs an immediate "hard" remove, so does not respect
253 	 * any transaction isolation semantics of the usage strategy. Use with care.
254 	 *
255 	 * @param entityName The entity name for which to evict data.
256 	 *
257 	 * @throws HibernateException Generally will mean that either that
258 	 * 'persisttentClass' did not name a mapped entity or a problem
259 	 * communicating with underlying cache impl.
260 	 *
261 	 * @deprecated Use {@link Cache#evictEntityRegion(String)} accessed through
262 	 * {@link #getCache()} instead.
263 	 */
264     @Deprecated
265 	public void evictEntity(String entityName) throws HibernateException;
266 
267 	/**
268 	 * Evict an entry from the second-level  cache. This method occurs outside
269 	 * of any transaction; it performs an immediate "hard" remove, so does not respect
270 	 * any transaction isolation semantics of the usage strategy. Use with care.
271 	 *
272 	 * @param entityName The entity name for which to evict data.
273 	 * @param id The entity id
274 	 *
275 	 * @throws HibernateException Generally will mean that either that
276 	 * 'persisttentClass' did not name a mapped entity or a problem
277 	 * communicating with underlying cache impl.
278 	 *
279 	 * @deprecated Use {@link Cache#evictEntity(String,Serializable)} accessed through
280 	 * {@link #getCache()} instead.
281 	 */
282     @Deprecated
283 	public void evictEntity(String entityName, Serializable id) throws HibernateException;
284 
285 	/**
286 	 * Evict all entries from the second-level cache. This method occurs outside
287 	 * of any transaction; it performs an immediate "hard" remove, so does not respect
288 	 * any transaction isolation semantics of the usage strategy. Use with care.
289 	 *
290 	 * @param roleName The name of the collection role whose regions should be evicted
291 	 *
292 	 * @throws HibernateException Generally will mean that either that
293 	 * 'roleName' did not name a mapped collection or a problem
294 	 * communicating with underlying cache impl.
295 	 *
296 	 * @deprecated Use {@link Cache#evictCollectionRegion(String)} accessed through
297 	 * {@link #getCache()} instead.
298 	 */
299     @Deprecated
300 	public void evictCollection(String roleName) throws HibernateException;
301 
302 	/**
303 	 * Evict an entry from the second-level cache. This method occurs outside
304 	 * of any transaction; it performs an immediate "hard" remove, so does not respect
305 	 * any transaction isolation semantics of the usage strategy. Use with care.
306 	 *
307 	 * @param roleName The name of the collection role
308 	 * @param id The id of the collection owner
309 	 *
310 	 * @throws HibernateException Generally will mean that either that
311 	 * 'roleName' did not name a mapped collection or a problem
312 	 * communicating with underlying cache impl.
313 	 *
314 	 * @deprecated Use {@link Cache#evictCollection(String,Serializable)} accessed through
315 	 * {@link #getCache()} instead.
316 	 */
317     @Deprecated
318 	public void evictCollection(String roleName, Serializable id) throws HibernateException;
319 
320 	/**
321 	 * Evict any query result sets cached in the named query cache region.
322 	 *
323 	 * @param cacheRegion The named query cache region from which to evict.
324 	 *
325 	 * @throws HibernateException Since a not-found 'cacheRegion' simply no-ops,
326 	 * this should indicate a problem communicating with underlying cache impl.
327 	 *
328 	 * @deprecated Use {@link Cache#evictQueryRegion(String)} accessed through
329 	 * {@link #getCache()} instead.
330 	 */
331     @Deprecated
332 	public void evictQueries(String cacheRegion) throws HibernateException;
333 
334 	/**
335 	 * Evict any query result sets cached in the default query cache region.
336 	 *
337 	 * @throws HibernateException Indicate a problem communicating with
338 	 * underlying cache impl.
339 	 *
340 	 * @deprecated Use {@link Cache#evictQueryRegions} accessed through
341 	 * {@link #getCache()} instead.
342 	 */
343     @Deprecated
344 	public void evictQueries() throws HibernateException;
345 
346 	/**
347 	 * Obtain a set of the names of all filters defined on this SessionFactory.
348 	 *
349 	 * @return The set of filter names.
350 	 */
351 	public Set getDefinedFilterNames();
352 
353 	/**
354 	 * Obtain the definition of a filter by name.
355 	 *
356 	 * @param filterName The name of the filter for which to obtain the definition.
357 	 * @return The filter definition.
358 	 * @throws HibernateException If no filter defined with the given name.
359 	 */
360 	public FilterDefinition getFilterDefinition(String filterName) throws HibernateException;
361 
362 	/**
363 	 * Determine if this session factory contains a fetch profile definition
364 	 * registered under the given name.
365 	 *
366 	 * @param name The name to check
367 	 * @return True if there is such a fetch profile; false otherwise.
368 	 */
369 	public boolean containsFetchProfileDefinition(String name);
370 
371 	/**
372 	 * Retrieve this factory's {@link TypeHelper}
373 	 *
374 	 * @return The factory's {@link TypeHelper}
375 	 */
376 	public TypeHelper getTypeHelper();
377 }

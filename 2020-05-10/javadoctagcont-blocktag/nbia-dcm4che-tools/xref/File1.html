260     private Map toUidMap(String cuids) {
261         if ( NONE.equalsIgnoreCase(cuids)) return null;
262         StringTokenizer st = new StringTokenizer(cuids, "\r\n;");
263         String uid,name;
264         Map map = new TreeMap();
265         int i = 0;
266         while ( st.hasMoreTokens() ) {
267             uid = st.nextToken().trim();
268             name = uid;
269             if ( isDigit(uid.charAt(0) ) ) {
270                 if ( ! UIDs.isValid(uid) ) 
271                     throw new IllegalArgumentException("UID "+uid+" isn't a valid UID!");
272             } else {
273                 uid = UIDs.forName( name );
274             }
275             map.put(name,uid);
276         }
277         return map;
278     }

280     private String toString(Map uids) {
281         if ( uids == null || uids.isEmpty() ) return NONE;
282         StringBuffer sb = new StringBuffer( uids.size() << 5);//StringBuffer initial size: nrOfUIDs x 32
283         Iterator iter = uids.keySet().iterator();
284         while ( iter.hasNext() ) {
285             sb.append(iter.next()).append(System.getProperty("line.separator", "\n"));
286         }
287         return sb.toString();
288     }
289 
290     private String toUidGrpString(Map<String, Map<String, String>> uids) {
291         StringBuffer sb = new StringBuffer();
292         String grpId;
293         for ( Map.Entry entry : uids.entrySet() ) {
294             grpId = (String) entry.getKey();
295             for ( String uid : ((Map<String, String>) entry.getValue()).keySet() ) {
296                 sb.append(grpId).append(':').append(uid).append(System.getProperty("line.separator", "\n"));
297             }
298         }
299         return sb.toString();
300     }

304     private Map<String, Map<String, String>> toUidGrpMap(String cuidGrps) {
305         StringTokenizer st = new StringTokenizer(cuidGrps, "\r\n;");
306         String s,grp,uid,name;
307         Map<String, Map<String, String>> mapGrps = new HashMap<String, Map<String, String>>();
308         Map<String,String> map;
309         int i = 0, pos;
310         while ( st.hasMoreTokens() ) {
311             s = st.nextToken().trim();
312             pos = s.indexOf(':');
313             if ( pos != -1 ) {
314                 grp = s.substring(0,pos);
315                 if ( grp.equals(RIDSupport.CUID_GRP_SR) 
316                         || grp.equals(RIDSupport.CUID_GRP_ECG) 
317                         || grp.equals(RIDSupport.CUID_GRP_DOC))
318                     continue;
319                 uid = s.substring(++pos);
320             } else {
321                 grp = "unknown";
322                 uid=s;
323             }
324             name = uid;
325             if ( isDigit(uid.charAt(0) ) ) {
326                 if ( ! UIDs.isValid(uid) ) 
327                     throw new IllegalArgumentException("UID "+uid+" isn't a valid UID!");
328             } else {
329                 uid = UIDs.forName( name );
330             }
331             map = mapGrps.get(grp);
332             if ( map == null ) {
333                 map = new TreeMap<String, String>();
334                 mapGrps.put(grp, map);
335             }
336             map.put(name,uid);
337         }
338         return mapGrps;
339     }

349     public String getDocTitlePatterns() {
350         Map<String,String> m = support.getDocTitlePatterns();
351         if (m == null) return NONE;
352         StringBuffer sb = new StringBuffer();
353         for (Map.Entry e : m.entrySet()) {
354             sb.append(e.getKey()).append('=').append(e.getValue()).append(System.getProperty("line.separator", "\n"));
355         }
356         return sb.toString();
357     }

359     public void setDocTitlePatterns(String docTitlePatterns) {
360         Map<String,String> m = null;
361         if (docTitlePatterns != null && !docTitlePatterns.trim().equalsIgnoreCase(NONE) ) {
362             m = new HashMap<String,String>();
363             StringTokenizer st = new StringTokenizer(docTitlePatterns, "\r\n;=");
364             while ( st.hasMoreElements() ) {
365                 m.put( st.nextToken(), st.nextToken());
366             }
367         }
368         support.setDocTitlePatterns(m);
369     }

1   package org.infinispan.remoting.rpc;
2   
3   import org.infinispan.remoting.inboundhandler.DeliverOrder;
4   
5   import java.util.concurrent.TimeUnit;
6   
7   /**
8    * Classes that wraps all the configuration parameters to configure a remote invocation.
9    *
10   * @author Pedro Ruivo
11   * @since 5.3
12   */
13  public class RpcOptions {
14  
15     private final long timeout;
16     private final TimeUnit unit;
17     private final ResponseFilter responseFilter;
18     private final ResponseMode responseMode;
19     private final DeliverOrder deliverOrder;
20     private final boolean skipReplicationQueue;
21  
22     /**
23      * @deprecated use instead {@link #RpcOptions(long, java.util.concurrent.TimeUnit, ResponseFilter, ResponseMode,
24      * boolean, org.infinispan.remoting.inboundhandler.DeliverOrder)}
25      */
26     @Deprecated
27     public RpcOptions(long timeout, TimeUnit unit, ResponseFilter responseFilter,
28                       ResponseMode responseMode, boolean skipReplicationQueue, boolean fifoOrder, boolean totalOrder) {
29        this(timeout, unit, responseFilter, responseMode, skipReplicationQueue,
30             totalOrder ? DeliverOrder.TOTAL : (fifoOrder ? DeliverOrder.PER_SENDER : DeliverOrder.NONE));
31     }
32  
33     public RpcOptions(long timeout, TimeUnit unit, ResponseFilter responseFilter, ResponseMode responseMode,
34                       boolean skipReplicationQueue, DeliverOrder deliverOrder) {
35        if (unit == null) {
36           throw new IllegalArgumentException("TimeUnit cannot be null");
37        } else if (responseMode == null) {
38           throw new IllegalArgumentException("ResponseMode cannot be null");
39        } else if (deliverOrder == null) {
40           throw new IllegalArgumentException("DeliverMode cannot be null");
41        }
42        this.timeout = timeout;
43        this.unit = unit;
44        this.responseFilter = responseFilter;
45        this.responseMode = responseMode;
46        this.skipReplicationQueue = skipReplicationQueue;
47        this.deliverOrder = deliverOrder;
48     }
49  
50     /**
51      * @return the timeout value to give up.
52      */
53     public long timeout() {
54        return timeout;
55     }
56  
57     /**
58      * @return the {@link TimeUnit} in which the timeout value is.
59      */
60     public TimeUnit timeUnit() {
61        return unit;
62     }
63  
64     /**
65      * @return {@code true} if the message is to be delivered in FIFO order.
66      * @deprecated use instead {@link #deliverOrder()}.
67      */
68     @Deprecated
69     public boolean fifoOrder() {
70        return deliverOrder == DeliverOrder.PER_SENDER;
71     }
72  
73     /**
74      * @return {@code true} if the message is to be delivered in total order.
75      * @deprecated use instead {@link #deliverOrder()}.
76      */
77     @Deprecated
78     public boolean totalOrder() {
79        return deliverOrder == DeliverOrder.TOTAL;
80     }
81  
82     /**
83      * @return the {@link org.infinispan.remoting.inboundhandler.DeliverOrder}.
84      */
85     public DeliverOrder deliverOrder() {
86        return deliverOrder;
87     }
88  
89     /**
90      * @return the {@link ResponseFilter} to be used. Default is {@code null} meaning waiting for all or none responses
91      * depending if the remote invocation is synchronous or asynchronous respectively.
92      */
93     public ResponseFilter responseFilter() {
94        return responseFilter;
95     }
96  
97     /**
98      * @return the {@link ResponseMode} to handle with the responses.
99      */
100    public ResponseMode responseMode() {
101       return responseMode;
102    }
103 
104    /**
105     * @return if {@code true}, the remote invocation will never be dispatch to the {@link
106     * org.infinispan.remoting.ReplicationQueue}. However, only asynchronous remote invocation may be dispatched to the
107     * {@link org.infinispan.remoting.ReplicationQueue}.
108     */
109    public boolean skipReplicationQueue() {
110       return skipReplicationQueue;
111    }
112 
113    @Override
114    public boolean equals(Object o) {
115       if (this == o) return true;
116       if (o == null || getClass() != o.getClass()) return false;
117 
118       RpcOptions options = (RpcOptions) o;
119 
120       if (skipReplicationQueue != options.skipReplicationQueue) return false;
121       if (timeout != options.timeout) return false;
122       if (deliverOrder != options.deliverOrder) return false;
123       if (responseFilter != null ? !responseFilter.equals(options.responseFilter) : options.responseFilter != null)
124          return false;
125       if (responseMode != options.responseMode) return false;
126       if (unit != options.unit) return false;
127 
128       return true;
129    }
130 
131    @Override
132    public int hashCode() {
133       int result = (int) (timeout ^ (timeout >>> 32));
134       result = 31 * result + unit.hashCode();
135       result = 31 * result + deliverOrder.hashCode();
136       result = 31 * result + (responseFilter != null ? responseFilter.hashCode() : 0);
137       result = 31 * result + responseMode.hashCode();
138       result = 31 * result + (skipReplicationQueue ? 1 : 0);
139       return result;
140    }
141 
142    @Override
143    public String toString() {
144       return "RpcOptions{" +
145             "timeout=" + timeout +
146             ", unit=" + unit +
147             ", deliverOrder=" + deliverOrder +
148             ", responseFilter=" + responseFilter +
149             ", responseMode=" + responseMode +
150             ", skipReplicationQueue=" + skipReplicationQueue +
151             '}';
152    }
153 }

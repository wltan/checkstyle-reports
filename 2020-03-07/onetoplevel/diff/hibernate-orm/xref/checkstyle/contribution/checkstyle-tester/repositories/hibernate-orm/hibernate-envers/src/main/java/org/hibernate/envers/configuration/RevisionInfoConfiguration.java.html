76      private String revisionPropSqlType;

78      public RevisionInfoConfiguration(GlobalConfiguration globalCfg) {
79          this.globalCfg = globalCfg;
80          if (globalCfg.isUseRevisionEntityWithNativeId()) {
81              revisionInfoEntityName = "org.hibernate.envers.DefaultRevisionEntity";
82          } else {
83              revisionInfoEntityName = "org.hibernate.envers.enhanced.SequenceIdRevisionEntity";
84          }
85          revisionInfoIdData = new PropertyData("id", "id", "field", null);
86          revisionInfoTimestampData = new PropertyData("timestamp", "timestamp", "field", null);
87          modifiedEntityNamesData = new PropertyData("modifiedEntityNames", "modifiedEntityNames", "field", null);
88          revisionInfoTimestampType = new LongType();
89  
90          revisionPropType = "integer";
91      }

93      private Document generateDefaultRevisionInfoXmlMapping() {
94          Document document = XMLHelper.getDocumentFactory().createDocument();
95  
96          Element class_mapping = MetadataTools.createEntity(document, new AuditTableData(null, null, globalCfg.getDefaultSchemaName(), globalCfg.getDefaultCatalogName()), null, null);
97  
98          class_mapping.addAttribute("name", revisionInfoEntityName);
99          class_mapping.addAttribute("table", "REVINFO");
100 
101         Element idProperty = MetadataTools.addNativelyGeneratedId(class_mapping, revisionInfoIdData.getName(),
102                 revisionPropType, globalCfg.isUseRevisionEntityWithNativeId());
103         MetadataTools.addColumn(idProperty, "REV", null, null, null, null, null, null, false);
104 
105         Element timestampProperty = MetadataTools.addProperty(class_mapping, revisionInfoTimestampData.getName(),
106                 revisionInfoTimestampType.getName(), true, false);
107         MetadataTools.addColumn(timestampProperty, "REVTSTMP", null, null, null, null, null, null, false);
108 
109         if (globalCfg.isTrackEntitiesChangedInRevisionEnabled()) {
110             generateEntityNamesTrackingTableMapping(class_mapping, "modifiedEntityNames",
111                                                     globalCfg.getDefaultSchemaName(), globalCfg.getDefaultCatalogName(),
112                                                     "REVCHANGES", "REV", "ENTITYNAME", "string");
113         }
114 
115         return document;
116     }

118     /**
119      * Generates mapping that represents a set of primitive types.<br />
120      * <code>
121      * &lt;set name="propertyName" table="joinTableName" schema="joinTableSchema" catalog="joinTableCatalog"
122      * &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;cascade="persist, delete" lazy="false" fetch="join"&gt;<br />
123      * &nbsp;&nbsp;&nbsp;&lt;key column="joinTablePrimaryKeyColumnName" /&gt;<br />
124      * &nbsp;&nbsp;&nbsp;&lt;element type="joinTableValueColumnType"&gt;<br />
125      * &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&lt;column name="joinTableValueColumnName" /&gt;<br />
126      * &nbsp;&nbsp;&nbsp;&lt;/element&gt;<br />
127      * &lt;/set&gt;
128      * </code>
129      */ 
130     private void generateEntityNamesTrackingTableMapping(Element class_mapping, String propertyName,
131                                                          String joinTableSchema, String joinTableCatalog, String joinTableName,
132                                                          String joinTablePrimaryKeyColumnName, String joinTableValueColumnName,
133                                                          String joinTableValueColumnType) {
134         Element set = class_mapping.addElement("set");
135         set.addAttribute("name", propertyName);
136         set.addAttribute("table", joinTableName);
137         set.addAttribute("schema", joinTableSchema);
138         set.addAttribute("catalog", joinTableCatalog);
139         set.addAttribute("cascade", "persist, delete");
140         set.addAttribute("fetch", "join");
141         set.addAttribute("lazy", "false");
142         Element key = set.addElement("key");
143         key.addAttribute("column", joinTablePrimaryKeyColumnName);
144         Element element = set.addElement("element");
145         element.addAttribute("type", joinTableValueColumnType);
146         Element column = element.addElement("column");
147         column.addAttribute("name", joinTableValueColumnName);
148     }

150     private Element generateRevisionInfoRelationMapping() {
151         Document document = XMLHelper.getDocumentFactory().createDocument();
152         Element rev_rel_mapping = document.addElement("key-many-to-one");
153         rev_rel_mapping.addAttribute("type", revisionPropType);
154         rev_rel_mapping.addAttribute("class", revisionInfoEntityName);
155 
156         if (revisionPropSqlType != null) {
157             // Putting a fake name to make Hibernate happy. It will be replaced later anyway.
158             MetadataTools.addColumn(rev_rel_mapping, "*" , null, null, null, revisionPropSqlType, null, null, false);
159         }
160 
161         return rev_rel_mapping;
162     }

164     private void searchForRevisionInfoCfgInProperties(XClass clazz, ReflectionManager reflectionManager,
165                                     MutableBoolean revisionNumberFound, MutableBoolean revisionTimestampFound,
166                                     MutableBoolean modifiedEntityNamesFound, String accessType) {
167         for (XProperty property : clazz.getDeclaredProperties(accessType)) {
168             RevisionNumber revisionNumber = property.getAnnotation(RevisionNumber.class);
169             RevisionTimestamp revisionTimestamp = property.getAnnotation(RevisionTimestamp.class);
170             ModifiedEntityNames modifiedEntityNames = property.getAnnotation(ModifiedEntityNames.class);
171 
172             if (revisionNumber != null) {
173                 if (revisionNumberFound.isSet()) {
174                     throw new MappingException("Only one property may be annotated with @RevisionNumber!");
175                 }
176 
177                 XClass revisionNumberClass = property.getType();
178                 if (reflectionManager.equals(revisionNumberClass, Integer.class) ||
179                         reflectionManager.equals(revisionNumberClass, Integer.TYPE)) {
180                     revisionInfoIdData = new PropertyData(property.getName(), property.getName(), accessType, null);
181                     revisionNumberFound.set();
182                 } else if (reflectionManager.equals(revisionNumberClass, Long.class) ||
183                         reflectionManager.equals(revisionNumberClass, Long.TYPE)) {
184                     revisionInfoIdData = new PropertyData(property.getName(), property.getName(), accessType, null);
185                     revisionNumberFound.set();
186 
187                     // The default is integer
188                     revisionPropType = "long";
189                 } else {
190                     throw new MappingException("The field annotated with @RevisionNumber must be of type " +
191                             "int, Integer, long or Long");
192                 }
193 
194                 // Getting the @Column definition of the revision number property, to later use that info to
195                 // generate the same mapping for the relation from an audit table's revision number to the
196                 // revision entity revision number.
197                 Column revisionPropColumn = property.getAnnotation(Column.class);
198                 if (revisionPropColumn != null) {
199                     revisionPropSqlType = revisionPropColumn.columnDefinition();
200                 }
201             }
202 
203             if (revisionTimestamp != null) {
204                 if (revisionTimestampFound.isSet()) {
205                     throw new MappingException("Only one property may be annotated with @RevisionTimestamp!");
206                 }
207 
208                 XClass revisionTimestampClass = property.getType();
209                 if (reflectionManager.equals(revisionTimestampClass, Long.class) ||
210                         reflectionManager.equals(revisionTimestampClass, Long.TYPE) ||
211                         reflectionManager.equals(revisionTimestampClass, Date.class) ||
212                         reflectionManager.equals(revisionTimestampClass, java.sql.Date.class)) {
213                     revisionInfoTimestampData = new PropertyData(property.getName(), property.getName(), accessType, null);
214                     revisionTimestampFound.set();
215                 } else {
216                     throw new MappingException("The field annotated with @RevisionTimestamp must be of type " +
217                             "long, Long, java.util.Date or java.sql.Date");
218                 }
219             }
220 
221             if (modifiedEntityNames != null) {
222                 if (modifiedEntityNamesFound.isSet()) {
223                     throw new MappingException("Only one property may be annotated with @ModifiedEntityNames!");
224                 }
225                 XClass modifiedEntityNamesClass = property.getType();
226                 if (reflectionManager.equals(modifiedEntityNamesClass, Set.class) &&
227                         reflectionManager.equals(property.getElementClass(), String.class)) {
228                     modifiedEntityNamesData = new PropertyData(property.getName(), property.getName(), accessType, null);
229                     modifiedEntityNamesFound.set();
230                 } else {
231                     throw new MappingException("The field annotated with @ModifiedEntityNames must be of Set<String> type.");
232                 }
233             }
234         }
235     }

237     private void searchForRevisionInfoCfg(XClass clazz, ReflectionManager reflectionManager,
238                                           MutableBoolean revisionNumberFound, MutableBoolean revisionTimestampFound,
239                                           MutableBoolean modifiedEntityNamesFound) {
240         XClass superclazz = clazz.getSuperclass();
241         if (!"java.lang.Object".equals(superclazz.getName())) {
242             searchForRevisionInfoCfg(superclazz, reflectionManager, revisionNumberFound, revisionTimestampFound, modifiedEntityNamesFound);
243         }
244 
245         searchForRevisionInfoCfgInProperties(clazz, reflectionManager, revisionNumberFound, revisionTimestampFound,
246                 modifiedEntityNamesFound, "field");
247         searchForRevisionInfoCfgInProperties(clazz, reflectionManager, revisionNumberFound, revisionTimestampFound,
248                 modifiedEntityNamesFound, "property");
249     }

251     public RevisionInfoConfigurationResult configure(Configuration cfg, ReflectionManager reflectionManager) {
252         Iterator<PersistentClass> classes = (Iterator<PersistentClass>) cfg.getClassMappings();
253         boolean revisionEntityFound = false;
254         RevisionInfoGenerator revisionInfoGenerator = null;
255 
256         Class<?> revisionInfoClass = null;
257 
258         while (classes.hasNext()) {
259             PersistentClass pc = classes.next();
260             XClass clazz;
261             try {
262                 clazz = reflectionManager.classForName(pc.getClassName(), this.getClass());
263             } catch (ClassNotFoundException e) {
264                 throw new MappingException(e);
265             }
266 
267             RevisionEntity revisionEntity = clazz.getAnnotation(RevisionEntity.class);
268             if (revisionEntity != null) {
269                 if (revisionEntityFound) {
270                     throw new MappingException("Only one entity may be annotated with @RevisionEntity!");
271                 }
272 
273                 // Checking if custom revision entity isn't audited
274                 if (clazz.getAnnotation(Audited.class) != null) {
275                     throw new MappingException("An entity annotated with @RevisionEntity cannot be audited!");
276                 }
277 
278                 revisionEntityFound = true;
279 
280                 MutableBoolean revisionNumberFound = new MutableBoolean();
281                 MutableBoolean revisionTimestampFound = new MutableBoolean();
282                 MutableBoolean modifiedEntityNamesFound = new MutableBoolean();
283 
284                 searchForRevisionInfoCfg(clazz, reflectionManager, revisionNumberFound, revisionTimestampFound, modifiedEntityNamesFound);
285 
286                 if (!revisionNumberFound.isSet()) {
287                     throw new MappingException("An entity annotated with @RevisionEntity must have a field annotated " +
288                             "with @RevisionNumber!");
289                 }
290 
291                 if (!revisionTimestampFound.isSet()) {
292                     throw new MappingException("An entity annotated with @RevisionEntity must have a field annotated " +
293                             "with @RevisionTimestamp!");
294                 }
295 
296                 revisionInfoEntityName = pc.getEntityName();
297                 revisionInfoClass = pc.getMappedClass();
298                 Class<? extends RevisionListener> revisionListenerClass = getRevisionListenerClass(revisionEntity.value());
299                 revisionInfoTimestampType = pc.getProperty(revisionInfoTimestampData.getName()).getType();
300                 if (globalCfg.isTrackEntitiesChangedInRevisionEnabled()
301                         || (globalCfg.isUseRevisionEntityWithNativeId() && DefaultTrackingModifiedEntitiesRevisionEntity.class.isAssignableFrom(revisionInfoClass))
302                         || (!globalCfg.isUseRevisionEntityWithNativeId() && SequenceIdTrackingModifiedEntitiesRevisionEntity.class.isAssignableFrom(revisionInfoClass))
303                         || modifiedEntityNamesFound.isSet()) {
304                     // If tracking modified entities parameter is enabled, custom revision info entity is a subtype
305                     // of DefaultTrackingModifiedEntitiesRevisionEntity class, or @ModifiedEntityNames annotation is used.
306                     revisionInfoGenerator = new DefaultTrackingModifiedEntitiesRevisionInfoGenerator(revisionInfoEntityName,
307                             revisionInfoClass, revisionListenerClass, revisionInfoTimestampData, isTimestampAsDate(),
308                             modifiedEntityNamesData);
309                     globalCfg.setTrackEntitiesChangedInRevisionEnabled(true);
310                 } else {
311                     revisionInfoGenerator = new DefaultRevisionInfoGenerator(revisionInfoEntityName, revisionInfoClass,
312                             revisionListenerClass, revisionInfoTimestampData, isTimestampAsDate());
313                 }
314             }
315         }
316 
317         // In case of a custom revision info generator, the mapping will be null.
318         Document revisionInfoXmlMapping = null;
319 
320         Class<? extends RevisionListener> revisionListenerClass = getRevisionListenerClass(RevisionListener.class);
321 
322         if (revisionInfoGenerator == null) {
323             if (globalCfg.isTrackEntitiesChangedInRevisionEnabled()) {
324                 revisionInfoClass = globalCfg.isUseRevisionEntityWithNativeId() ? DefaultTrackingModifiedEntitiesRevisionEntity.class
325                                                                                 : SequenceIdTrackingModifiedEntitiesRevisionEntity.class;
326                 revisionInfoEntityName = revisionInfoClass.getName();
327                 revisionInfoGenerator = new DefaultTrackingModifiedEntitiesRevisionInfoGenerator(revisionInfoEntityName, revisionInfoClass,
328                         revisionListenerClass, revisionInfoTimestampData, isTimestampAsDate(), modifiedEntityNamesData);
329             } else {
330                 revisionInfoClass = globalCfg.isUseRevisionEntityWithNativeId() ? DefaultRevisionEntity.class
331                                                                                 : SequenceIdRevisionEntity.class;
332                 revisionInfoGenerator = new DefaultRevisionInfoGenerator(revisionInfoEntityName, revisionInfoClass,
333                         revisionListenerClass, revisionInfoTimestampData, isTimestampAsDate());
334             }
335             revisionInfoXmlMapping = generateDefaultRevisionInfoXmlMapping();
336         }
337 
338         return new RevisionInfoConfigurationResult(
339                 revisionInfoGenerator, revisionInfoXmlMapping,
340                 new RevisionInfoQueryCreator(revisionInfoEntityName, revisionInfoIdData.getName(),
341                         revisionInfoTimestampData.getName(), isTimestampAsDate()),
342                 generateRevisionInfoRelationMapping(),
343                 new RevisionInfoNumberReader(revisionInfoClass, revisionInfoIdData),
344                 globalCfg.isTrackEntitiesChangedInRevisionEnabled() ? new ModifiedEntityNamesReader(revisionInfoClass, modifiedEntityNamesData)
345                                                                     : null,
346                 revisionInfoEntityName, revisionInfoClass, revisionInfoTimestampData);
347     }
348 
349     private boolean isTimestampAsDate() {
350     	String typename = revisionInfoTimestampType.getName();
351     	return "date".equals(typename) || "time".equals(typename) || "timestamp".equals(typename);
352     }

354     /**
355      * @param defaultListener Revision listener that shall be applied if {@code org.hibernate.envers.revision_listener}
356      *                        parameter has not been set.
357      * @return Revision listener.  
358      */
359     private Class<? extends RevisionListener> getRevisionListenerClass(Class<? extends RevisionListener> defaultListener) {
360         if (globalCfg.getRevisionListenerClass() != null) {
361             return globalCfg.getRevisionListenerClass();
362         }
363         return defaultListener;
364     }

47  /**
48   * MetaDataStateFormat is a base class to write checksummed
49   * XContent based files to one or more directories in a standardized directory structure.
50   * @param <T> the type of the XContent base data-structure
51   */
52  public abstract class MetaDataStateFormat<T> {
53      public static final String STATE_DIR_NAME = "_state";
54      public static final String STATE_FILE_EXTENSION = ".st";
55      private static final String STATE_FILE_CODEC = "state";
56      private static final int STATE_FILE_VERSION = 0;
57      private static final int BUFFER_SIZE = 4096;
58      private final XContentType format;
59      private final String prefix;
60      private final Pattern stateFilePattern;
61  
62  
63      /**
64       * Creates a new {@link MetaDataStateFormat} instance
65       * @param format the format of the x-content
66       */
67      protected MetaDataStateFormat(XContentType format, String prefix) {
68          this.format = format;
69          this.prefix = prefix;
70          this.stateFilePattern = Pattern.compile(Pattern.quote(prefix) + "(\\d+)(" + MetaDataStateFormat.STATE_FILE_EXTENSION + ")?");
71  
72      }
73  
74      /**
75       * Returns the {@link XContentType} used to serialize xcontent on write.
76       */
77      public XContentType format() {
78          return format;
79      }

81      /**
82       * Writes the given state to the given directories. The state is written to a
83       * state directory ({@value #STATE_DIR_NAME}) underneath each of the given file locations and is created if it
84       * doesn't exist. The state is serialized to a temporary file in that directory and is then atomically moved to
85       * it's target filename of the pattern <tt>{prefix}{version}.st</tt>.
86       *
87       * @param state the state object to write
88       * @param version the version of the state
89       * @param locations the locations where the state should be written to.
90       * @throws IOException if an IOException occurs
91       */
92      public final void write(final T state, final long version, final File... locations) throws IOException {
93          Preconditions.checkArgument(locations != null, "Locations must not be null");
94          Preconditions.checkArgument(locations.length > 0, "One or more locations required");
95          final long maxStateId = findMaxStateId(prefix, locations)+1;
96          assert maxStateId >= 0 : "maxStateId must be positive but was: [" + maxStateId + "]";
97          final String fileName = prefix + maxStateId + STATE_FILE_EXTENSION;
98          Path stateLocation = Paths.get(locations[0].getPath(), STATE_DIR_NAME);
99          Files.createDirectories(stateLocation);
100         final Path tmpStatePath = stateLocation.resolve(fileName + ".tmp");
101         final Path finalStatePath = stateLocation.resolve(fileName);
102         try {
103             try (OutputStreamIndexOutput out = new OutputStreamIndexOutput(Files.newOutputStream(tmpStatePath), BUFFER_SIZE)) {
104                 CodecUtil.writeHeader(out, STATE_FILE_CODEC, STATE_FILE_VERSION);
105                 out.writeInt(format.index());
106                 out.writeLong(version);
107                 try (XContentBuilder builder = newXContentBuilder(format, new org.elasticsearch.common.lucene.store.OutputStreamIndexOutput(out) {
108                     @Override
109                     public void close() throws IOException {
110                         // this is important since some of the XContentBuilders write bytes on close.
111                         // in order to write the footer we need to prevent closing the actual index input.
112                     } })) {
113 
114                     builder.startObject();
115                     {
116                         toXContent(builder, state);
117                     }
118                     builder.endObject();
119                 }
120                 CodecUtil.writeFooter(out);
121             }
122             IOUtils.fsync(tmpStatePath.toFile(), false); // fsync the state file
123             Files.move(tmpStatePath, finalStatePath, StandardCopyOption.ATOMIC_MOVE);
124             IOUtils.fsync(stateLocation.toFile(), true);
125             for (int i = 1; i < locations.length; i++) {
126                 stateLocation = Paths.get(locations[i].getPath(), STATE_DIR_NAME);
127                 Files.createDirectories(stateLocation);
128                 Path tmpPath = stateLocation.resolve(fileName + ".tmp");
129                 Path finalPath = stateLocation.resolve(fileName);
130                 try {
131                     Files.copy(finalStatePath, tmpPath);
132                     Files.move(tmpPath, finalPath, StandardCopyOption.ATOMIC_MOVE); // we are on the same FileSystem / Partition here we can do an atomic move
133                     IOUtils.fsync(stateLocation.toFile(), true); // we just fsync the dir here..
134                 } finally {
135                     Files.deleteIfExists(tmpPath);
136                 }
137             }
138         } finally {
139             Files.deleteIfExists(tmpStatePath);
140         }
141         cleanupOldFiles(prefix, fileName, locations);
142     }

144     protected XContentBuilder newXContentBuilder(XContentType type, OutputStream stream ) throws IOException {
145         return XContentFactory.contentBuilder(type, stream);
146     }

160     /**
161      * Reads the state from a given file and compares the expected version against the actual version of
162      * the state.
163      */
164     public final T read(File file) throws IOException {
165         try (Directory dir = newDirectory(file.getParentFile())) {
166             try (final IndexInput indexInput = dir.openInput(file.getName(), IOContext.DEFAULT)) {
167                 // We checksum the entire file before we even go and parse it. If it's corrupted we barf right here.
168                 CodecUtil.checksumEntireFile(indexInput);
169                 CodecUtil.checkHeader(indexInput, STATE_FILE_CODEC, STATE_FILE_VERSION, STATE_FILE_VERSION);
170                 final XContentType xContentType = XContentType.values()[indexInput.readInt()];
171                 indexInput.readLong(); // version currently unused
172                 long filePointer = indexInput.getFilePointer();
173                 long contentSize = indexInput.length() - CodecUtil.footerLength() - filePointer;
174                 try (IndexInput slice = indexInput.slice("state_xcontent", filePointer, contentSize)) {
175                     try (XContentParser parser = XContentFactory.xContent(xContentType).createParser(new InputStreamIndexInput(slice, contentSize))) {
176                         return fromXContent(parser);
177                     }
178                 }
179             } catch(CorruptIndexException ex) {
180                 // we trick this into a dedicated exception with the original stacktrace
181                 throw new CorruptStateException(ex);
182             }
183         }
184     }
185 
186     protected Directory newDirectory(File dir) throws IOException {
187         return new SimpleFSDirectory(dir);
188     }

190     private void cleanupOldFiles(String prefix, String fileName, File[] locations) throws IOException {
191         // now clean up the old files
192         for (File dataLocation : locations) {
193             final File[] files = new File(dataLocation, STATE_DIR_NAME).listFiles();
194             if (files != null) {
195                 for (File file : files) {
196                     if (!file.getName().startsWith(prefix)) {
197                         continue;
198                     }
199                     if (file.getName().equals(fileName)) {
200                         continue;
201                     }
202                     Files.delete(file.toPath());
203                 }
204             }
205         }
206     }
207 
208     long findMaxStateId(final String prefix, File... locations) throws IOException {
209         long maxId = -1;
210         for (File dataLocation : locations) {
211             final File[] files = new File(dataLocation, STATE_DIR_NAME).listFiles();
212             if (files != null) {
213                 for (File file : files) {
214                     if (!file.getName().startsWith(prefix)) {
215                         continue;
216                     }
217                     final Matcher matcher = stateFilePattern.matcher(file.getName());
218                     if (matcher.matches()) {
219                         final long id = Long.parseLong(matcher.group(1));
220                         maxId = Math.max(maxId, id);
221                     }
222                 }
223             }
224         }
225         return maxId;
226     }

228     /**
229      * Tries to load the latest state from the given data-locations. It tries to load the latest state determined by
230      * the states version from one or more data directories and if none of the latest states can be loaded an exception
231      * is thrown to prevent accidentally loading a previous state and silently omitting the latest state.
232      *
233      * @param logger an elasticsearch logger instance
234      * @param dataLocations the data-locations to try.
235      * @return the latest state or <code>null</code> if no state was found.
236      */
237     public  T loadLatestState(ESLogger logger, File... dataLocations) {
238         List<FileAndStateId> files = new ArrayList<>();
239         long maxStateId = -1;
240         boolean maxStateIdIsLegacy = true;
241         if (dataLocations != null) { // select all eligable files first
242             for (File dataLocation : dataLocations) {
243                 File stateDir = new File(dataLocation, MetaDataStateFormat.STATE_DIR_NAME);
244                 // now, iterate over the current versions, and find latest one
245                 File[] stateFiles = stateDir.listFiles();
246                 if (stateFiles == null) {
247                     continue;
248                 }
249                 for (File stateFile : stateFiles) {
250                     final Matcher matcher = stateFilePattern.matcher(stateFile.getName());
251                     if (matcher.matches()) {
252                         final long version = Long.parseLong(matcher.group(1));
253                         maxStateId = Math.max(maxStateId, version);
254                         final boolean legacy = MetaDataStateFormat.STATE_FILE_EXTENSION.equals(matcher.group(2)) == false;
255                         maxStateIdIsLegacy &= legacy; // on purpose, see NOTE below
256                         files.add(new FileAndStateId(stateFile, version, legacy));
257                     }
258                 }
259             }
260         }
261         final List<Throwable> exceptions = new ArrayList<>();
262         T state = null;
263         // NOTE: we might have multiple version of the latest state if there are multiple data dirs.. for this case
264         //       we iterate only over the ones with the max version. If we have at least one state file that uses the
265         //       new format (ie. legacy == false) then we know that the latest version state ought to use this new format.
266         //       In case the state file with the latest version does not use the new format while older state files do,
267         //       the list below will be empty and loading the state will fail
268         for (FileAndStateId fileAndVersion : Collections2.filter(files, new StateIdAndLegacyPredicate(maxStateId, maxStateIdIsLegacy))) {
269             try {
270                 final File stateFile = fileAndVersion.file;
271                 final long id = fileAndVersion.id;
272                 final XContentParser parser;
273                 if (fileAndVersion.legacy) { // read the legacy format -- plain XContent
274                     try (FileInputStream stream = new FileInputStream(stateFile)) {
275                         final byte[] data = Streams.copyToByteArray(stream);
276                         if (data.length == 0) {
277                             logger.debug("{}: no data for [{}], ignoring...", prefix, stateFile.getAbsolutePath());
278                             continue;
279                         }
280                         parser = XContentHelper.createParser(data, 0, data.length);
281                         state = fromXContent(parser);
282                         if (state == null) {
283                             logger.debug("{}: no data for [{}], ignoring...", prefix, stateFile.getAbsolutePath());
284                         }
285                     }
286                 } else {
287                     state = read(stateFile);
288                     logger.trace("state id [{}] read from [{}]", id, stateFile.getName());
289                 }
290                 return state;
291             } catch (Throwable e) {
292                 exceptions.add(e);
293                 logger.debug("{}: failed to read [{}], ignoring...", e, fileAndVersion.file.getAbsolutePath(), prefix);
294             }
295         }
296         // if we reach this something went wrong
297         ExceptionsHelper.maybeThrowRuntimeAndSuppress(exceptions);
298         if (files.size() > 0) {
299             // We have some state files but none of them gave us a usable state
300             throw new ElasticsearchIllegalStateException("Could not find a state file to recover from among " + files);
301         }
302         return state;
303     }

305     /**
306      * Filters out all {@link org.elasticsearch.gateway.local.state.meta.MetaDataStateFormat.FileAndStateId} instances with a different id than
307      * the given one.
308      */
309     private static final class StateIdAndLegacyPredicate implements Predicate<FileAndStateId> {
310         private final long id;
311         private final boolean legacy;
312 
313         StateIdAndLegacyPredicate(long id, boolean legacy) {
314             this.id = id;
315             this.legacy = legacy;
316         }
317 
318         @Override
319         public boolean apply(FileAndStateId input) {
320             return input.id == id && input.legacy == legacy;
321         }
322     }

324     /**
325      * Internal struct-like class that holds the parsed state id, the file
326      * and a flag if the file is a legacy state ie. pre 1.5
327      */
328     private static class FileAndStateId {
329         final File file;
330         final long id;
331         final boolean legacy;
332 
333         private FileAndStateId(File file, long id, boolean legacy) {
334             this.file = file;
335             this.id = id;
336             this.legacy = legacy;
337         }
338 
339         @Override
340         public String toString() {
341             return "[id:" + id + ", legacy:" + legacy + ", file:" + file.getAbsolutePath() + "]";
342         }
343     }

345     /**
346      * Deletes all meta state directories recursively for the given data locations
347      * @param dataLocations the data location to delete
348      */
349     public static void deleteMetaState(Path... dataLocations) throws IOException {
350         Path[] stateDirectories = new Path[dataLocations.length];
351         for (int i = 0; i < dataLocations.length; i++) {
352             stateDirectories[i] = dataLocations[i].resolve(STATE_DIR_NAME);
353         }
354         XIOUtils.rm(stateDirectories);
355     }

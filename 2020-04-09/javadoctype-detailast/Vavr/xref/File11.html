28      private final Comparator<? super T> comparator;

32      private PriorityQueue(Comparator<? super T> comparator, Seq<Node<T>> forest, int size) {
33          this.comparator = comparator;
34          this.forest = forest;
35          this.size = size;

38      private PriorityQueue<T> with(Seq<Node<T>> forest, int size) {
39          return new PriorityQueue<>(comparator, forest, size);
40      }

48       * Enqueues a new element.

60       * Enqueues the given elements. A queue has FIFO order, i.e. the first of the given elements is
61       * the first which will be retrieved.

64       * @return a new {@link PriorityQueue} instance, containing the new elements
65       * @throws NullPointerException if elements is null

69          return merge(ofAll(comparator, elements));

73       * Returns the first element of a non-empty {@link PriorityQueue}.
74       *
75       * @return The first element of this {@link PriorityQueue}.
76       * @throws NoSuchElementException if this is empty
77       */
78      @Override
79      public T head() {
80          if (isEmpty()) {
81              throw new NoSuchElementException("head of empty " + stringPrefix());
82          } else {
83              return findMin(comparator, forest).root;
84          }
85      }

87      /**
88       * Drops the first element of a non-empty {@link PriorityQueue}.
89       *
90       * @return A new instance of PriorityQueue containing all elements except the first.
91       * @throws UnsupportedOperationException if this is empty
92       */
93      @Override
94      public PriorityQueue<T> tail() {
95          if (isEmpty()) {
96              throw new UnsupportedOperationException("tail of empty " + stringPrefix());
97          } else {
98              return dequeue()._2;
99          }
100     }

102     @Override
103     public Tuple2<T, PriorityQueue<T>> dequeue() {
104         if (isEmpty()) {
105             throw new NoSuchElementException("dequeue of empty " + stringPrefix());
106         } else {
107             final Tuple2<T, Seq<Node<T>>> dequeue = deleteMin(comparator, this.forest);
108             return Tuple.of(dequeue._1, with(dequeue._2, this.size - 1));
109         }
110     }
111 
112     public PriorityQueue<T> merge(PriorityQueue<T> target) {
113         final Seq<Node<T>> meld = meld(comparator, this.forest, target.forest);
114         return with(meld, this.size + target.size);
115     }

161     /**
162      * Returns a {@link Collector} which may be used in conjunction with
163      * {@link java.util.stream.Stream#collect(Collector)} to obtain a {@code PriorityQueue<T>}.
164      *
165      * @param <T> Component type of the {@code PriorityQueue}.
166      * @return A {@code PriorityQueue<T>} Collector.
167      */
168     public static <T> Collector<T, ArrayList<T>, PriorityQueue<T>> collector() {
169         final Supplier<ArrayList<T>> supplier = ArrayList::new;
170         final BiConsumer<ArrayList<T>, T> accumulator = ArrayList::add;
171         final BinaryOperator<ArrayList<T>> combiner = (left, right) -> {
172             left.addAll(right);
173             return left;
174         };
175         final Function<ArrayList<T>, PriorityQueue<T>> finisher = values -> ofAll(Comparators.naturalComparator(), values);
176         return Collector.of(supplier, accumulator, combiner, finisher);
177     }

215     @SuppressWarnings("unchecked")
216     public static <T> PriorityQueue<T> ofAll(Comparator<? super T> comparator, Iterable<? extends T> elements) {
217         Objects.requireNonNull(elements, "elements is null");
218         if (elements instanceof PriorityQueue && ((PriorityQueue) elements).comparator == comparator) {
219             return (PriorityQueue<T>) elements;
220         } else {
221             int size = 0;
222             Seq<Node<T>> forest = io.vavr.collection.List.empty();
223             for (T value : elements) {
224                 forest = insert(comparator, value, forest);
225                 size++;
226             }
227             return new PriorityQueue<>(comparator, forest, size);
228         }
229     }

273     @Override
274     public io.vavr.collection.List<T> toList() {
275         io.vavr.collection.List<T> results = io.vavr.collection.List.empty();
276         for (PriorityQueue<T> queue = this; !queue.isEmpty(); ) {
277             final Tuple2<T, PriorityQueue<T>> dequeue = queue.dequeue();
278             results = results.prepend(dequeue._1);
279             queue = dequeue._2;
280         }
281         return results.reverse();
282     }

329     @Override
330     public PriorityQueue<T> dropWhile(Predicate<? super T> predicate) {
331         Objects.requireNonNull(predicate, "predicate is null");
332         PriorityQueue<T> result = this;
333         while (!result.isEmpty() && predicate.test(result.head())) {
334             result = result.tail();
335         }
336         return result;

473     @Override
474     public Tuple2<? extends PriorityQueue<T>, ? extends PriorityQueue<T>> partition(Predicate<? super T> predicate) {
475         Objects.requireNonNull(predicate, "predicate is null");
476         PriorityQueue<T> left = empty(comparator), right = left;
477         for (T t : this) {
478             if (predicate.test(t)) {
479                 left = left.enqueue(t);
480             } else {
481                 right = right.enqueue(t);
482             }
483         }
484         return Tuple.of(left, right);
485 
486     }

633     /**
634      * fun deleteMin [] = raise EMPTY
635      * * | deleteMin ts =
636      * *     val (Node (x,r,c), ts) = getMin ts
637      * *     val (ts',xs') = split ([],[],c)
638      * *     in fold insert xs' (meld (ts, ts')) end
639      **/
640     static <T> Tuple2<T, Seq<Node<T>>> deleteMin(Comparator<? super T> comparator, Seq<Node<T>> forest) {
641         /* get the minimum tree and the rest of the forest */
642         final Node<T> minTree = findMin(comparator, forest);
643         final Seq<Node<T>> forestTail = (minTree == forest.head()) ? forest.tail() : forest.remove(minTree);
644 
645         final Seq<Node<T>> newForest = rebuild(comparator, minTree.children);
646         return Tuple.of(minTree.root, meld(comparator, newForest, forestTail));
647     }
648 
649     /**
650      * fun insert (x, ts as t1 :: t2 :: rest) =
651      * *     if rank t1 = rank t2 then skewLink(Node(x,0,[]),t1,t2) :: rest
652      * *     else                      Node (x,0,[]) :: ts
653      * * | insert (x, ts) =            Node (x,0,[]) :: ts
654      **/
655     static <T> Seq<Node<T>> insert(Comparator<? super T> comparator, T element, Seq<Node<T>> forest) {
656         final Node<T> tree = Node.of(element, 0, io.vavr.collection.List.empty());
657         if (forest.size() >= 2) {
658             final Seq<Node<T>> tail = forest.tail();
659             final Node<T> t1 = forest.head(), t2 = tail.head();
660             if (t1.rank == t2.rank) {
661                 return tree.skewLink(comparator, t1, t2).appendTo(tail.tail());
662             }
663         }
664         return tree.appendTo(forest);
665     }
666 
667     /** fun meld (ts, ts') = meldUniq (uniqify ts, uniqify ts') */
668     static <T> Seq<Node<T>> meld(Comparator<? super T> comparator, Seq<Node<T>> source, Seq<Node<T>> target) {
669         return meldUnique(comparator, uniqify(comparator, source), uniqify(comparator, target));
670     }
671 
672     /**
673      * Find the minimum root in the forest
674      * <p>
675      * fun findMin [] = raise EMPTY
676      * * | findMin [t] = root t
677      * * | findMin (t :: ts) =
678      * *     let val x = findMin ts
679      * *     in if Elem.leq (root t, x) then root t else x end
680      */
681     static <T> Node<T> findMin(Comparator<? super T> comparator, Seq<Node<T>> forest) {
682         final io.vavr.collection.Iterator<Node<T>> iterator = forest.iterator();
683         Node<T> min = iterator.next();
684         for (Node<T> node : iterator) {
685             if (comparator.compare(node.root, min.root) < 0) {
686                 min = node;
687             }
688         }
689         return min;
690     }
691 
692     /**
693      * Separate the rank 0 trees from the rest, rebuild the 0 rank ones and merge them back
694      * <p>
695      * fun split (ts,xs,[]) = (ts, xs)
696      * * | split (ts,xs,t :: c) =
697      * *     if rank t = 0 then split (ts,root t :: xs,c)
698      * *     else               split (t :: ts,xs,c)
699      */
700     private static <T> Seq<Node<T>> rebuild(Comparator<? super T> comparator, Seq<Node<T>> forest) {
701         Seq<Node<T>> nonZeroRank = io.vavr.collection.List.empty(), zeroRank = io.vavr.collection.List.empty();
702         for (; !forest.isEmpty(); forest = forest.tail()) {
703             final Node<T> initialForestHead = forest.head();
704             if (initialForestHead.rank == 0) {
705                 zeroRank = insert(comparator, initialForestHead.root, zeroRank);
706             } else {
707                 nonZeroRank = initialForestHead.appendTo(nonZeroRank);
708             }
709         }
710         return meld(comparator, nonZeroRank, zeroRank);
711     }
712 
713     /**
714      * fun uniqify [] = []
715      * *  | uniqify (t :: ts) = ins (t, ts) (∗ eliminate initial duplicate ∗)
716      **/
717     private static <T> Seq<Node<T>> uniqify(Comparator<? super T> comparator, Seq<Node<T>> forest) {
718         return forest.isEmpty()
719                ? forest
720                : ins(comparator, forest.head(), forest.tail());
721     }
722 
723     /**
724      * fun ins (t, []) = [t]
725      * * | ins (t, t' :: ts) = (∗ rank t ≤ rank t' ∗)
726      * *     if rank t < rank t' then t :: t' :: ts
727      * *     else                     ins (link (t, t'), ts)
728      */
729     private static <T> Seq<Node<T>> ins(Comparator<? super T> comparator, Node<T> tree, Seq<Node<T>> forest) {
730         while (!forest.isEmpty() && tree.rank == forest.head().rank) {
731             tree = tree.link(comparator, forest.head());
732             forest = forest.tail();
733         }
734         return tree.appendTo(forest);
735     }
736 
737     /**
738      * fun meldUniq ([], ts) = ts
739      * *  | meldUniq (ts, []) = ts
740      * *  | meldUniq (t1 :: ts1, t2 :: ts2) =
741      * *      if rank t1 < rank t2 then      t1 :: meldUniq (ts1, t2 :: ts2)
742      * *      else if rank t2 < rank t1 then t2 :: meldUniq (t1 :: ts1, ts2)
743      * *      else                           ins (link (t1, t2), meldUniq (ts1, ts2))
744      **/
745     private static <T> Seq<Node<T>> meldUnique(Comparator<? super T> comparator, Seq<Node<T>> forest1, Seq<Node<T>> forest2) {
746         if (forest1.isEmpty()) {
747             return forest2;
748         } else if (forest2.isEmpty()) {
749             return forest1;
750         } else {
751             final Node<T> tree1 = forest1.head(), tree2 = forest2.head();
752 
753             if (tree1.rank == tree2.rank) {
754                 final Node<T> tree = tree1.link(comparator, tree2);
755                 final Seq<Node<T>> forest = meldUnique(comparator, forest1.tail(), forest2.tail());
756                 return ins(comparator, tree, forest);
757             } else {
758                 if (tree1.rank < tree2.rank) {
759                     final Seq<Node<T>> forest = meldUnique(comparator, forest1.tail(), forest2);
760                     return tree1.appendTo(forest);
761                 } else {
762                     final Seq<Node<T>> forest = meldUnique(comparator, forest1, forest2.tail());
763                     return tree2.appendTo(forest);
764                 }
765             }
766         }
767     }

769     /* Based on http://www.brics.dk/RS/96/37/BRICS-RS-96-37.pdf */
770     static final class Node<T> implements Serializable {
771 
772         private static final long serialVersionUID = 1L;
773 
774         final T root;
775         final int rank;
776         final Seq<Node<T>> children;
777 
778         private Node(T root, int rank, Seq<Node<T>> children) {
779             this.root = root;
780             this.rank = rank;
781             this.children = children;
782         }
783 
784         static <T> Node<T> of(T value, int rank, Seq<Node<T>> children) {
785             return new Node<>(value, rank, children);
786         }
787 
788         /*
789          * fun link (t1 as Node (x1,r1,c1), t2 as Node (x2,r2,c2)) = (∗ r1 = r2 ∗)
790          * *  if Elem.leq (x1,x2) then Node (x1,r1+1,t2 :: c1)
791          * *  else                     Node (x2,r2+1,t1 :: c2
792          */
793         Node<T> link(Comparator<? super T> comparator, Node<T> tree) {
794             return comparator.compare(this.root, tree.root) <= 0
795                    ? of(this.root, this.rank + 1, tree.appendTo(this.children))
796                    : of(tree.root, tree.rank + 1, this.appendTo(tree.children));
797         }
798 
799         /*
800          * fun skewLink (t0 as Node (x0,r0, _), t1 as Node (x1,r1,c1), t2 as Node (x2,r2,c2)) =
801          * *  if Elem.leq (x1,x0) andalso Elem.leq (x1,x2) then      Node (x1,r1+1,t0 :: t2 :: c1)
802          * *  else if Elem.leq (x2,x0) andalso Elem.leq (x2,x1) then Node (x2,r2+1,t0 :: t1 :: c2)
803          * *  else                                                   Node (x0,r1+1,[t1, t2])
804          */
805         Node<T> skewLink(Comparator<? super T> comparator, Node<T> left, Node<T> right) {
806             if (comparator.compare(left.root, root) <= 0 && comparator.compare(left.root, right.root) <= 0) {
807                 return of(left.root, left.rank + 1, appendTo(right.appendTo(left.children)));
808             } else if (comparator.compare(right.root, root) <= 0) {
809                 return of(right.root, right.rank + 1, appendTo(left.appendTo(right.children)));
810             } else {
811                 return of(root, left.rank + 1, io.vavr.collection.List.of(left, right));
812             }
813         }
814 
815         Seq<Node<T>> appendTo(Seq<Node<T>> forest) {
816             return forest.prepend(this);
817         }
818     }

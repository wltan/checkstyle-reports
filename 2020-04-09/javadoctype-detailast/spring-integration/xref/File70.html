17  package org.springframework.integration.file.transformer;
18  
19  import java.io.File;
20  import java.io.IOException;
21  
22  import org.apache.commons.logging.Log;
23  import org.apache.commons.logging.LogFactory;
24  
25  import org.springframework.beans.BeansException;
26  import org.springframework.beans.factory.BeanFactory;
27  import org.springframework.beans.factory.BeanFactoryAware;
28  import org.springframework.integration.file.FileHeaders;
29  import org.springframework.integration.support.DefaultMessageBuilderFactory;
30  import org.springframework.integration.support.MessageBuilderFactory;
31  import org.springframework.integration.support.utils.IntegrationUtils;
32  import org.springframework.integration.transformer.Transformer;
33  import org.springframework.messaging.Message;
34  import org.springframework.messaging.MessagingException;
35  import org.springframework.util.Assert;
36  
37  /**
38   * Base class for transformers that convert a File payload.
39   *
40   * @author Mark Fisher
41   * @author Artem Bilan
42   */
43  public abstract class AbstractFilePayloadTransformer<T> implements Transformer, BeanFactoryAware {
44  
45  	private final Log logger = LogFactory.getLog(this.getClass());
46  
47  	private volatile boolean deleteFiles;
48  
49  	private volatile MessageBuilderFactory messageBuilderFactory = new DefaultMessageBuilderFactory();
50  
51  	private boolean messageBuilderFactorySet;
52  
53  	private volatile BeanFactory beanFactory;
54  
55  	/**
56  	 * Specify whether to delete the File after transformation.
57  	 * Default is <em>false</em>.
58  	 *
59  	 * @param deleteFiles true to delete the file.
60  	 */
61  	public void setDeleteFiles(boolean deleteFiles) {
62  		this.deleteFiles = deleteFiles;
63  	}
64  
65  	@Override
66  	public void setBeanFactory(BeanFactory beanFactory) throws BeansException {
67  		this.beanFactory = beanFactory;
68  	}
69  
70  	protected MessageBuilderFactory getMessageBuilderFactory() {
71  		if (!this.messageBuilderFactorySet) {
72  			if (this.beanFactory != null) {
73  				this.messageBuilderFactory = IntegrationUtils.getMessageBuilderFactory(this.beanFactory);
74  			}
75  			this.messageBuilderFactorySet = true;
76  		}
77  		return this.messageBuilderFactory;
78  	}
79  
80  	@Override
81  	public final Message<?> transform(Message<?> message) {
82  		try {
83  			Assert.notNull(message, "Message must not be null");
84  			Object payload = message.getPayload();
85  			Assert.notNull(payload, "Message payload must not be null");
86  			Assert.isInstanceOf(File.class, payload, "Message payload must be of type [java.io.File]");
87  			File file = (File) payload;
88  			T result = this.transformFile(file);
89  			Message<?> transformedMessage = getMessageBuilderFactory().withPayload(result)
90  					.copyHeaders(message.getHeaders())
91  					.setHeaderIfAbsent(FileHeaders.ORIGINAL_FILE, file)
92  					.setHeaderIfAbsent(FileHeaders.FILENAME, file.getName())
93  					.build();
94  			if (this.deleteFiles) {
95  				if (!file.delete() && this.logger.isWarnEnabled()) {
96  					this.logger.warn("failed to delete File '" + file + "'");
97  				}
98  			}
99  			return transformedMessage;
100 		}
101 		catch (Exception e) {
102 			throw new MessagingException(message, "failed to transform File Message", e);
103 		}
104 	}
105 
106 	/**
107 	 * Subclasses must implement this method to transform the File contents.
108 	 *
109 	 * @param file The file.
110 	 * @return The result of the transformation.
111 	 * @throws IOException Any IOException.
112 	 */
113 	protected abstract T transformFile(File file) throws IOException;
114 
115 }

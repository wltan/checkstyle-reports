17  package org.springframework.integration.handler;
18  
19  import org.springframework.lang.Nullable;
20  import org.springframework.messaging.Message;
21  
22  /**
23   * This defines the lowest-level strategy of processing a Message and returning
24   * some Object (or null). Implementations will be focused on generic concerns,
25   * such as invoking a method, running a script, or evaluating an expression.
26   * <p>
27   * Higher level MessageHandler implementations can delegate to these processors
28   * for such functionality, but it is the responsibility of each handler type to
29   * add the semantics such as routing, splitting, transforming, etc.
30   * <p>
31   * In some cases the return value might be a Message itself, but it does not
32   * need to be. It is the responsibility of the caller to determine how to treat
33   * the return value. That may require creating a Message or even creating
34   * multiple Messages from that value.
35   * <p>
36   * This strategy and its various implementations are considered part of the
37   * internal "support" API, intended for use by Spring Integration's various
38   * message-handling components. As such, it is subject to change.
39   *
40   * @author Mark Fisher
41   * @author Artem Bilan
42   *
43   * @since 2.0
44   */
45  @FunctionalInterface
46  public interface MessageProcessor<T> {
47  
48  	/**
49  	 * Process the Message and return a value (or null).
50  	 *
51  	 * @param message The message to process.
52  	 * @return The result.
53  	 */
54  	@Nullable
55  	T processMessage(Message<?> message);
56  
57  }

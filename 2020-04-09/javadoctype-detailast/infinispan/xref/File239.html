1   package org.infinispan.notifications.cachelistener.filter;
2   
3   import org.infinispan.metadata.Metadata;
4   
5   /**
6    * A Filter that only allows post events to be accepted.
7    *
8    * @author wburns
9    * @since 7.0
10   */
11  public class PostCacheEventFilter<K, V> implements CacheEventFilter<K, V> {
12     @Override
13     public boolean accept(K key, V oldValue, Metadata oldMetadata, V newValue, Metadata newMetadata, EventType eventType) {
14        return !eventType.isPreEvent();
15     }
16  }

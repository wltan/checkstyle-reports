1   package org.infinispan.commons.util.concurrent;
2   
3   import java.util.concurrent.Future;
4   
5   /**
6    * An internal interface which adds the ability to inform the future of completion.
7    *
8    * @author Manik Surtani
9    * @since 4.0
10   */
11  public interface NotifyingNotifiableFuture<T> extends NotifyingFuture<T> {
12     /**
13      * Notify the listeners that the operation has finished. Subsequent calls for isDone()
14      * will return true and subsequent calls for get() will return the provided result.
15      * @param result
16      */
17     void notifyDone(T result);
18  
19     /**
20      * Notify the listeners that the operation has finished. Subsequent calls for isDone()
21      * will return true and subsequent calls for get() will throw the provided exception.
22      * @param exception
23      */
24     void notifyException(Throwable exception);
25  
26     /**
27      * Setup the future which is wrapped by implementation of this interface.
28      *
29      * Warning: the implementation must synchronize the call setFuture()
30      * and notifyDone() to be executed in this order. Also, all calls delegated to the
31      * future must be deferred until the network future is set up.
32      * @param future
33      */
34     void setFuture(Future<T> future);
35  }

1   package org.infinispan.marshall.core;
2   
3   import org.infinispan.commons.io.ByteBuffer;
4   import org.infinispan.metadata.InternalMetadata;
5   
6   /**
7    * Defines an externally persisted entry. External stores that keep the data in serialised form should return an
8    * MarshalledEntry that contains the data in binary form (ByteBuffer) and unmarshall it lazily when
9    * getKey/Value/Metadata are invoked. This approach avoids unnecessary object (de)serialization e.g
10   * when the entries are fetched from the external store for the sole purpose of being sent over the wire to
11   * another requestor node.
12   *
13   * @author Mircea Markus
14   * @since 6.0
15   */
16  public interface MarshalledEntry<K, V> {
17  
18     /**
19      * Returns the key in serialized format.
20      */
21     ByteBuffer getKeyBytes();
22  
23     /**
24      * Returns the value in serialize format.
25      */
26     ByteBuffer getValueBytes();
27  
28     /**
29      * @return null if there's no metadata associated with the object (e.g. expiry info, version..)
30      */
31     ByteBuffer getMetadataBytes();
32  
33     /**
34      * Returns the same key as {@link #getKeyBytes()}, but unmarshalled.
35      */
36     K getKey();
37  
38     /**
39      * Returns the same value as {@link #getKeyBytes()}, but unmarshalled.
40      */
41     V getValue();
42  
43     /**
44      * @return might be null if there's no metadata associated with the object (e.g. expiry info, version..).
45      */
46     InternalMetadata getMetadata();
47  }

1   package org.infinispan.notifications.cachelistener.cluster;
2   
3   import org.infinispan.distexec.DistributedCallable;
4   import org.infinispan.notifications.cachelistener.CacheNotifier;
5   import org.infinispan.notifications.cachelistener.event.CacheEntryEvent;
6   
7   import java.util.Collection;
8   import java.util.UUID;
9   
10  /**
11   * This interface describes methods required for a cluster listener to be able to be bootstrapped and properly notified
12   * when a new event has been raised from the cluster.
13   *
14   * @author wburns
15   * @since 7.0
16   */
17  public interface ClusterCacheNotifier<K, V> extends CacheNotifier<K, V> {
18     /**
19      * Method that is invoked on the node that has the given cluster listener that when registered generated the given
20      * listenerId.  Note this will notify only cluster listeners and regular listeners are not notified of the events.
21      * Will fire the events in the order of the iteration of the collection.
22      * @param events
23      * @param listenerId
24      */
25     void notifyClusterListeners(Collection<? extends CacheEntryEvent<K, V>> events, UUID listenerId);
26  
27     /**
28      * This method is invoked so that this node can send the details required for a new node to be bootstrapped with
29      * the existing cluster listeners that are already installed.
30      * @return A collection of callables that should be invoked on the new node to properly install cluster listener information
31      */
32     Collection<DistributedCallable> retrieveClusterListenerCallablesToInstall();
33  }

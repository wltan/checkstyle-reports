1   package org.infinispan.filter;
2   
3   import org.infinispan.metadata.Metadata;
4   
5   /**
6    * This interface is an optimization that can be used when a filter and converter are most efficiently used as the same
7    * object composing the filtering and conversion in the same method invocation.
8    * @author wburns
9    * @since 7.0
10   */
11  public interface KeyValueFilterConverter<K, V, C> extends KeyValueFilter<K, V>, Converter<K, V, C> {
12     /**
13      * Will both filter the entry and if passed subsequently convert the value to a new value.  A returned value of null
14      * will symbolize the value not passing the filter, so ensure your conversion will not return null if you want this
15      * entry to be returned.
16      * @param key The key of the entry to filter
17      * @param value The value of the entry to filter and then convert
18      * @param metadata The metadata attached to the entry
19      * @return The converted value or null if the filter didn't pass
20      */
21     public C filterAndConvert(K key, V value, Metadata metadata);
22  }

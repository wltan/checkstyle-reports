1   package org.infinispan.notifications.cachelistener.filter;
2   
3   import org.infinispan.metadata.Metadata;
4   
5   /**
6    * A filter that can be used to tell if an event should be raised or not.  This filter allows for filtering based
7    * on the previous value as well as the new updated value.  The old value and old metadata are the previous values and
8    * the new value and new metadata are the new values even for pre and post events.
9    *
10   * @author wburns
11   * @since 7.0
12   */
13  public interface CacheEventFilter<K, V> {
14     /**
15      * Whether or not this event should be raised to the listener it is attached to.
16      * @param key The key for the entry that was changed for the event
17      * @param oldValue The previous value before the event takes place
18      * @param oldMetadata The old value before the event takes place
19      * @param newValue The new value for the entry after the event takes place
20      * @param newMetadata The new metadata for the entry after the event takes place
21      * @param eventType The type of event that is being raised
22      * @return Whether or not to notify the listener
23      */
24     public boolean accept(K key, V oldValue, Metadata oldMetadata, V newValue, Metadata newMetadata, EventType eventType);
25  }

1   package org.infinispan.cdi.util.defaultbean;
2   
3   import java.lang.annotation.Annotation;
4   import java.lang.reflect.Type;
5   import java.util.Set;
6   
7   import javax.enterprise.context.spi.CreationalContext;
8   import javax.enterprise.inject.spi.Bean;
9   import javax.enterprise.inject.spi.BeanManager;
10  
11  import org.infinispan.cdi.util.Beans;
12  
13  
14  /**
15   * A helper class for implementing producer methods and fields on default beans
16   *
17   * @author Stuart Douglas
18   */
19  abstract class AbstractDefaultProducerBean<T> extends AbstractDefaultBean<T> {
20  
21      private static final Annotation[] NO_QUALIFIERS = {};
22  
23      private final Type declaringBeanType;
24      private final Annotation[] declaringBeanQualifiers;
25  
26      protected AbstractDefaultProducerBean(Bean<T> delegate, Type declaringBeanType, Set<Type> types, Set<Annotation> qualifiers, Set<Annotation> declaringBeanQualifiers, BeanManager beanManager) {
27          super(delegate, declaringBeanType, types, qualifiers, beanManager);
28          this.declaringBeanType = delegate.getBeanClass();
29          this.declaringBeanQualifiers = declaringBeanQualifiers.toArray(NO_QUALIFIERS);
30      }
31  
32      protected Annotation[] getDeclaringBeanQualifiers() {
33          return declaringBeanQualifiers.clone();
34      }
35  
36      protected Type getDeclaringBeanType() {
37          return declaringBeanType;
38      }
39  
40      protected abstract T getValue(Object receiver, CreationalContext<T> creationalContext);
41  
42      @Override
43      public T create(CreationalContext<T> creationalContext) {
44          Object receiver = getReceiver(creationalContext);
45          T instance = getValue(receiver, creationalContext);
46          Beans.checkReturnValue(instance, this, null, getBeanManager());
47          return instance;
48      }
49  
50      protected Object getReceiver(CreationalContext<T> creationalContext) {
51          Bean<?> declaringBean = getBeanManager().resolve(getBeanManager().getBeans(getDeclaringBeanType(), declaringBeanQualifiers));
52          return getBeanManager().getReference(declaringBean, declaringBean.getBeanClass(), creationalContext);
53      }
54  }

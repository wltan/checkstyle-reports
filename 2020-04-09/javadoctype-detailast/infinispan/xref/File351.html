1   package org.infinispan.spring.support;
2   
3   import org.infinispan.Cache;
4   import org.infinispan.manager.CacheContainer;
5   import org.infinispan.util.logging.Log;
6   import org.infinispan.util.logging.LogFactory;
7   import org.springframework.beans.factory.DisposableBean;
8   import org.springframework.beans.factory.FactoryBean;
9   import org.springframework.beans.factory.InitializingBean;
10  
11  /**
12   * <p>
13   * A {@link org.springframework.beans.factory.FactoryBean <code>FactoryBean</code>} for creating a
14   * native <em>default</em> Infinispan {@link org.infinispan.Cache <code>org.infinispan.Cache</code>}
15   * , delegating to a {@link #setInfinispanCacheContainer(CacheContainer) <code>configurable</code>}
16   * {@link org.infinispan.manager.CacheContainer <code>org.infinispan.manager.CacheContainer</code>}.
17   * A default <code>Cache</code> is a <code>Cache</code> that uses its <code>CacheContainer</code>'s
18   * default settings. This is contrary to a <em>named</em> <code>Cache</code> where select settings
19   * from a <code>CacheContainer</code>'s default configuration may be overridden with settings
20   * specific to that <code>Cache</code>.
21   * </p>
22   * <p>
23   * In addition to creating a <code>Cache</code> this <code>FactoryBean</code> does also control that
24   * <code>Cache</code>'s {@link org.infinispan.commons.api.Lifecycle lifecycle} by shutting it down
25   * when the enclosing Spring application context is closed. It is therefore advisable to
26   * <em>always</em> use this <code>FactoryBean</code> when creating a <code>Cache</code>.
27   * </p>
28   *
29   * @author <a href="mailto:olaf DOT bergner AT gmx DOT de">Olaf Bergner</a>
30   *
31   */
32  public class InfinispanDefaultCacheFactoryBean<K, V> implements FactoryBean<Cache<K, V>>,
33                                                                  InitializingBean, DisposableBean {
34  
35     protected final Log logger = LogFactory.getLog(getClass());
36  
37     private CacheContainer infinispanCacheContainer;
38  
39     private Cache<K, V> infinispanCache;
40  
41     /**
42      * <p>
43      * Sets the {@link org.infinispan.manager.CacheContainer
44      * <code>org.infinispan.manager.CacheContainer</code>} to be used for creating our
45      * {@link org.infinispan.Cache <code>Cache</code>} instance. Note that this is a
46      * <strong>mandatory</strong> property.
47      * </p>
48      *
49      * @param infinispanCacheContainer
50      *           The {@link org.infinispan.manager.CacheContainer
51      *           <code>org.infinispan.manager.CacheContainer</code>} to be used for creating our
52      *           {@link org.infinispan.Cache <code>Cache</code>} instance
53      */
54     public void setInfinispanCacheContainer(final CacheContainer infinispanCacheContainer) {
55        this.infinispanCacheContainer = infinispanCacheContainer;
56     }
57  
58     /**
59      * @see org.springframework.beans.factory.InitializingBean#afterPropertiesSet()
60      */
61     @Override
62     public void afterPropertiesSet() throws Exception {
63        if (this.infinispanCacheContainer == null) {
64           throw new IllegalStateException("No Infinispan CacheContainer has been set");
65        }
66        this.logger.info("Initializing named Infinispan cache ...");
67        this.infinispanCache = this.infinispanCacheContainer.getCache();
68        this.logger.info("New Infinispan cache [" + this.infinispanCache + "] initialized");
69     }
70  
71     /**
72      * @see org.springframework.beans.factory.FactoryBean#getObject()
73      */
74     @Override
75     public Cache<K, V> getObject() throws Exception {
76        return this.infinispanCache;
77     }
78  
79     /**
80      * @see org.springframework.beans.factory.FactoryBean#getObjectType()
81      */
82     @Override
83     public Class<? extends Cache> getObjectType() {
84        return this.infinispanCache != null ? this.infinispanCache.getClass() : Cache.class;
85     }
86  
87     /**
88      * Always returns <code>true</code>.
89      *
90      * @return Always <code>true</code>
91      *
92      * @see org.springframework.beans.factory.FactoryBean#isSingleton()
93      */
94     @Override
95     public boolean isSingleton() {
96        return true;
97     }
98  
99     /**
100     * Shuts down the <code>org.infinispan.Cache</code> created by this <code>FactoryBean</code>.
101     *
102     * @see org.springframework.beans.factory.DisposableBean#destroy()
103     * @see org.infinispan.Cache#stop()
104     */
105    @Override
106    public void destroy() throws Exception {
107       // Probably being paranoid here ...
108       if (this.infinispanCache != null) {
109          this.infinispanCache.stop();
110       }
111    }
112 }

1   package org.infinispan.iteration.impl;
2   
3   import org.infinispan.Cache;
4   import org.infinispan.commons.util.CloseableIterable;
5   import org.infinispan.container.entries.CacheEntry;
6   import org.infinispan.context.Flag;
7   import org.infinispan.filter.Converter;
8   import org.infinispan.filter.KeyValueFilter;
9   import org.infinispan.iteration.EntryIterable;
10  
11  import java.util.EnumSet;
12  
13  /**
14   * This is an implementation that allows for creating new EntryIterable instances by supplying a new converter.
15   *
16   * @author wburns
17   * @since 7.0
18   */
19  public class EntryIterableImpl<K, V> extends TrackingEntryIterable<K, V, V> implements EntryIterable<K, V> {
20     public EntryIterableImpl(EntryRetriever<K, V> entryRetriver, KeyValueFilter<? super K, ? super V> filter,
21                              EnumSet<Flag> flags, Cache<K, V> cache) {
22        super(entryRetriver, filter, null, flags, cache);
23     }
24  
25     @Override
26     public <C> CloseableIterable<CacheEntry<K, C>> converter(Converter<? super K, ? super V, ? extends C> converter) {
27        return new TrackingEntryIterable<>(entryRetriever, filter, converter, flags, cache);
28     }
29  }

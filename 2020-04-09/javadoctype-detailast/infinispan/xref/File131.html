1   package org.infinispan.configuration.cache;
2   
3   import static org.infinispan.configuration.cache.AsyncStoreConfiguration.*;
4   
5   import java.util.concurrent.TimeUnit;
6   
7   import org.infinispan.commons.configuration.Builder;
8   import org.infinispan.commons.configuration.attributes.AttributeSet;
9   import org.infinispan.configuration.global.GlobalConfiguration;
10  
11  /**
12   * Configuration for the async cache store. If enabled, this provides you with asynchronous writes
13   * to the cache store, giving you 'write-behind' caching.
14   *
15   * @author pmuir
16   *
17   */
18  public class AsyncStoreConfigurationBuilder<S> extends AbstractStoreConfigurationChildBuilder<S> implements Builder<AsyncStoreConfiguration> {
19     private final AttributeSet attributes;
20  
21     AsyncStoreConfigurationBuilder(AbstractStoreConfigurationBuilder<? extends AbstractStoreConfiguration, ?> builder) {
22        super(builder);
23        this.attributes = AsyncStoreConfiguration.attributeDefinitionSet();
24     }
25  
26     /**
27      * If true, all modifications to this cache store happen asynchronously, on a separate thread.
28      */
29     public AsyncStoreConfigurationBuilder<S> enable() {
30        attributes.attribute(ENABLED).set(true);
31        return this;
32     }
33  
34     public AsyncStoreConfigurationBuilder<S> disable() {
35        attributes.attribute(ENABLED).set(false);
36        return this;
37     }
38  
39     public AsyncStoreConfigurationBuilder<S> enabled(boolean enabled) {
40        attributes.attribute(ENABLED).set(enabled);
41        return this;
42     }
43  
44     /**
45      * Unused.
46      */
47     @Deprecated
48     public AsyncStoreConfigurationBuilder<S> flushLockTimeout(long l) {
49        return this;
50     }
51  
52     /**
53      * Unused.
54      */
55     @Deprecated
56     public AsyncStoreConfigurationBuilder<S> flushLockTimeout(long l, TimeUnit unit) {
57        return this;
58     }
59  
60     /**
61      * Sets the size of the modification queue for the async store. If updates are made at a rate
62      * that is faster than the underlying cache store can process this queue, then the async store
63      * behaves like a synchronous store for that period, blocking until the queue can accept more
64      * elements.
65      */
66     public AsyncStoreConfigurationBuilder<S> modificationQueueSize(int i) {
67        attributes.attribute(MODIFICATION_QUEUE_SIZE).set(i);
68        return this;
69     }
70  
71     /**
72      * Unused.
73      */
74     @Deprecated
75     public AsyncStoreConfigurationBuilder<S> shutdownTimeout(long l) {
76        return this;
77     }
78  
79     /**
80      * Unused.
81      */
82     @Deprecated
83     public AsyncStoreConfigurationBuilder<S> shutdownTimeout(long l, TimeUnit unit) {
84        return this;
85     }
86  
87     /**
88      * Size of the thread pool whose threads are responsible for applying the modifications.
89      */
90     public AsyncStoreConfigurationBuilder<S> threadPoolSize(int i) {
91        attributes.attribute(THREAD_POOL_SIZE).set(i);
92        return this;
93     }
94  
95     @Override
96     public
97     void validate() {
98     }
99  
100    @Override
101    public void validate(GlobalConfiguration globalConfig) {
102    }
103 
104    @Override
105    public AsyncStoreConfiguration create() {
106       return new AsyncStoreConfiguration(attributes.protect());
107    }
108 
109    @Override
110    public AsyncStoreConfigurationBuilder<S> read(AsyncStoreConfiguration template) {
111       this.attributes.read(template.attributes());
112       return this;
113    }
114 
115    @Override
116    public String toString() {
117       return "AsyncStoreConfigurationBuilder [attributes=" + attributes + "]";
118    }
119 }

1   package org.infinispan.cdi.util;
2   
3   import java.lang.annotation.Annotation;
4   import java.lang.reflect.Type;
5   import java.util.Set;
6   
7   import javax.enterprise.context.Dependent;
8   import javax.enterprise.context.spi.CreationalContext;
9   import javax.enterprise.inject.Default;
10  import javax.enterprise.inject.spi.Bean;
11  import javax.enterprise.inject.spi.InjectionPoint;
12  
13  /**
14   * <p>
15   * A base class for implementing {@link Bean}. The attributes are immutable, and
16   * collections are defensively copied on instantiation. It uses the defaults
17   * from the specification for properties if not specified.
18   * </p>
19   * <p/>
20   * <p>
21   * This bean delegates it's lifecycle to the callbacks on the provided
22   * {@link ContextualLifecycle}.
23   * </p>
24   *
25   * @author Stuart Douglas
26   * @author Pete Muir
27   * @see AbstractImmutableBean
28   * @see BeanBuilder
29   * @see ImmutablePassivationCapableBean
30   */
31  public class ImmutableBean<T> extends AbstractImmutableBean<T> implements Bean<T> {
32  
33      private final ContextualLifecycle<T> contextualLifecycle;
34  
35      /**
36       * Create a new, immutable bean. All arguments passed as collections are
37       * defensively copied.
38       *
39       * @param beanClass           The Bean class, may not be null
40       * @param name                The bean name
41       * @param qualifiers          The bean's qualifiers, if null, a singleton set of
42       *                            {@link Default} is used
43       * @param scope               The bean's scope, if null, the default scope of
44       *                            {@link Dependent} is used
45       * @param stereotypes         The bean's stereotypes, if null, an empty set is used
46       * @param types               The bean's types, if null, the beanClass and {@link Object}
47       *                            will be used
48       * @param alternative         True if the bean is an alternative
49       * @param nullable            True if the bean is nullable
50       * @param injectionPoints     the bean's injection points, if null an empty set
51       *                            is used
52       * @param contextualLifecycle Handler for {@link #create(CreationalContext)}
53       *                            and {@link #destroy(Object, CreationalContext)}
54       * @param toString            the string representation of the bean, if null the built
55       *                            in representation is used, which states the bean class and
56       *                            qualifiers
57       * @throws IllegalArgumentException if the beanClass is null
58       */
59      public ImmutableBean(Class<?> beanClass, String name, Set<Annotation> qualifiers, Class<? extends Annotation> scope, Set<Class<? extends Annotation>> stereotypes, Set<Type> types, boolean alternative, boolean nullable, Set<InjectionPoint> injectionPoints, ContextualLifecycle<T> contextualLifecycle, String toString) {
60          super(beanClass, name, qualifiers, scope, stereotypes, types, alternative, nullable, injectionPoints, toString);
61          this.contextualLifecycle = contextualLifecycle;
62      }
63  
64      public T create(CreationalContext<T> arg0) {
65          return contextualLifecycle.create(this, arg0);
66      }
67  
68      public void destroy(T arg0, CreationalContext<T> arg1) {
69          contextualLifecycle.destroy(this, arg0, arg1);
70      }
71  
72  }

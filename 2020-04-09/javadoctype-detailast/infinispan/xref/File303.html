1   package org.infinispan.objectfilter.impl.predicateindex;
2   
3   import org.infinispan.objectfilter.impl.util.Interval;
4   
5   /**
6    * @author anistor@redhat.com
7    * @since 7.0
8    */
9   public final class IntervalCondition<AttributeDomain extends Comparable<AttributeDomain>> extends Condition<AttributeDomain> {
10  
11     private final Interval<AttributeDomain> interval;
12  
13     public IntervalCondition(Interval<AttributeDomain> interval) {
14        this.interval = interval;
15     }
16  
17     @Override
18     public boolean match(AttributeDomain attributeValue) {
19        return attributeValue != null && interval.contains(attributeValue);
20     }
21  
22     @Override
23     public boolean equals(Object obj) {
24        if (this == obj) return true;
25        if (obj == null || getClass() != obj.getClass()) return false;
26  
27        IntervalCondition other = (IntervalCondition) obj;
28        return interval.equals(other.interval);
29     }
30  
31     @Override
32     public int hashCode() {
33        return interval.hashCode();
34     }
35  
36     @Override
37     public String toString() {
38        return "IntervalCondition(" + interval + ')';
39     }
40  }

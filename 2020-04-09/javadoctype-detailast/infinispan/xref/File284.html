1   package org.infinispan.jcache.annotation;
2   
3   import static java.util.Collections.unmodifiableSet;
4   
5   import java.lang.annotation.Annotation;
6   import java.lang.reflect.Method;
7   import java.util.List;
8   import java.util.Set;
9   
10  import javax.cache.annotation.CacheKeyGenerator;
11  
12  /**
13   * Metadata associated to a method annotated with a cache annotation.
14   *
15   * @author Kevin Pollet <kevin.pollet@serli.com> (C) 2011 SERLI
16   */
17  public class MethodMetaData<A extends Annotation> {
18  
19     private final Method method;
20     private final Set<Annotation> annotations;
21     private final A cacheAnnotation;
22     private final String cacheName;
23     private final AggregatedParameterMetaData aggregatedParameterMetaData;
24     private final CacheKeyGenerator cacheKeyGenerator;
25  
26     public MethodMetaData(Method method,
27                           AggregatedParameterMetaData aggregatedParameterMetaData,
28                           Set<Annotation> annotations,
29                           CacheKeyGenerator cacheKeyGenerator,
30                           A cacheAnnotation,
31                           String cacheName) {
32  
33        this.method = method;
34        this.aggregatedParameterMetaData = aggregatedParameterMetaData;
35        this.annotations = unmodifiableSet(annotations);
36        this.cacheKeyGenerator = cacheKeyGenerator;
37        this.cacheAnnotation = cacheAnnotation;
38        this.cacheName = cacheName;
39     }
40  
41     public Method getMethod() {
42        return method;
43     }
44  
45     public Set<Annotation> getAnnotations() {
46        return annotations;
47     }
48  
49     public A getCacheAnnotation() {
50        return cacheAnnotation;
51     }
52  
53     public String getCacheName() {
54        return cacheName;
55     }
56  
57     public CacheKeyGenerator getCacheKeyGenerator() {
58        return cacheKeyGenerator;
59     }
60  
61     public List<ParameterMetaData> getParameters() {
62        return aggregatedParameterMetaData.getParameters();
63     }
64  
65     public List<ParameterMetaData> getKeyParameters() {
66        return aggregatedParameterMetaData.getKeyParameters();
67     }
68  
69     public ParameterMetaData getValueParameter() {
70        return aggregatedParameterMetaData.getValueParameter();
71     }
72  
73     @Override
74     public String toString() {
75        return new StringBuilder()
76              .append("MethodMetaData{")
77              .append("method=").append(method)
78              .append(", annotations=").append(annotations)
79              .append(", cacheAnnotation=").append(cacheAnnotation)
80              .append(", cacheName='").append(cacheName).append('\'')
81              .append(", aggregatedParameterMetaData=").append(aggregatedParameterMetaData)
82              .append(", cacheKeyGenerator=").append(cacheKeyGenerator)
83              .append('}')
84              .toString();
85     }
86  }

1   package org.infinispan.filter;
2   
3   import net.jcip.annotations.ThreadSafe;
4   import org.infinispan.metadata.Metadata;
5   
6   /**
7    * A filter for keys with their values.  This class is complemented by the {@link org.infinispan.filter.KeyFilter}
8    * class.  This filter should be used in most cases over {@link org.infinispan.filter.KeyFilter} since it allows
9    * for more control of the filtering since it provides the ability to filter on the value and metadata in addition
10   * to the key.
11   *
12   * @author William Burns
13   * @since 7.0
14   */
15  @ThreadSafe
16  public interface KeyValueFilter<K, V> {
17     /**
18      * @param key key to test
19      * @param value value to use (could be null for the case of removal)
20      * @param metadata metadata
21      * @return true if the given key is accepted by this filter.
22      */
23     boolean accept(K key, V value, Metadata metadata);
24  }

23  /**
24   * <p> Class for constructing a new AnnotatedType. A new instance of builder
25   * should be used for each annotated type. </p> <p/> <p> {@link
26   * AnnotatedTypeBuilder} is not thread-safe. </p>
27   *
28   * @author Stuart Douglas
29   * @author Pete Muir
30   * @see AnnotatedType
31   */
32  public class AnnotatedTypeBuilder<X> {
33  
34     private static final Log log = LogFactory.getLog(AnnotatedTypeBuilder.class, Log.class);
35  
36     private Class<X> javaClass;
37     private final AnnotationBuilder typeAnnotations;
38  
39     private final Map<Constructor<?>, AnnotationBuilder> constructors;
40     private final Map<Constructor<?>, Map<Integer, AnnotationBuilder>> constructorParameters;
41     private final Map<Constructor<?>, Map<Integer, Type>> constructorParameterTypes;
42  
43     private final Map<Field, AnnotationBuilder> fields;
44     private final Map<Field, Type> fieldTypes;
45  
46     private final Map<Method, AnnotationBuilder> methods;
47     private final Map<Method, Map<Integer, AnnotationBuilder>> methodParameters;
48     private final Map<Method, Map<Integer, Type>> methodParameterTypes;

50     /**
51      * Create a new builder. A new builder has no annotations and no members.
52      *
53      * @see #readFromType(AnnotatedType)
54      * @see #readFromType(Class)
55      * @see #readFromType(AnnotatedType, boolean)
56      * @see #readFromType(Class, boolean)
57      */
58     public AnnotatedTypeBuilder() {
59        this.typeAnnotations = new AnnotationBuilder();
60        this.constructors = new HashMap<Constructor<?>, AnnotationBuilder>();
61        this.constructorParameters = new HashMap<Constructor<?>, Map<Integer, AnnotationBuilder>>();
62        this.constructorParameterTypes = new HashMap<Constructor<?>, Map<Integer, Type>>();
63        this.fields = new HashMap<Field, AnnotationBuilder>();
64        this.fieldTypes = new HashMap<Field, Type>();
65        this.methods = new HashMap<Method, AnnotationBuilder>();
66        this.methodParameters = new HashMap<Method, Map<Integer, AnnotationBuilder>>();
67        this.methodParameterTypes = new HashMap<Method, Map<Integer, Type>>();
68     }

81     /**
82      * Reads in from an existing AnnotatedType. Any elements not present are
83      * added. The javaClass will be read in. If the annotation already exists on
84      * that element in the builder the read annotation will be used.
85      *
86      * @param type the type to read from
87      * @throws IllegalArgumentException if type is null
88      */
89     public AnnotatedTypeBuilder<X> readFromType(AnnotatedType<X> type) {
90        return readFromType(type, true);
91     }

93     /**
94      * Reads in from an existing AnnotatedType. Any elements not present are
95      * added. The javaClass will be read in if overwrite is true. If the
96      * annotation already exists on that element in the builder, overwrite
97      * determines whether the original or read annotation will be used.
98      *
99      * @param type      the type to read from
100     * @param overwrite if true, the read annotation will replace any existing
101     *                  annotation
102     * @throws IllegalArgumentException if type is null
103     */
104    public AnnotatedTypeBuilder<X> readFromType(AnnotatedType<X> type, boolean overwrite) {
105       if (type == null) {
106          throw log.parameterMustNotBeNull("type");
107       }
108       if (javaClass == null || overwrite) {
109          this.javaClass = type.getJavaClass();
110       }
111       mergeAnnotationsOnElement(type, overwrite, typeAnnotations);
112       for (AnnotatedField<? super X> field : type.getFields()) {
113          if (fields.get(field.getJavaMember()) == null) {
114             fields.put(field.getJavaMember(), new AnnotationBuilder());
115          }
116          mergeAnnotationsOnElement(field, overwrite, fields.get(field.getJavaMember()));
117       }
118       for (AnnotatedMethod<? super X> method : type.getMethods()) {
119          if (methods.get(method.getJavaMember()) == null) {
120             methods.put(method.getJavaMember(), new AnnotationBuilder());
121          }
122          mergeAnnotationsOnElement(method, overwrite, methods.get(method.getJavaMember()));
123          for (AnnotatedParameter<? super X> p : method.getParameters()) {
124             if (methodParameters.get(method.getJavaMember()) == null) {
125                methodParameters.put(method.getJavaMember(), new HashMap<Integer, AnnotationBuilder>());
126             }
127             if (methodParameters.get(method.getJavaMember()).get(p.getPosition()) == null) {
128                methodParameters.get(method.getJavaMember()).put(p.getPosition(), new AnnotationBuilder());
129             }
130             mergeAnnotationsOnElement(p, overwrite, methodParameters.get(method.getJavaMember()).get(p.getPosition()));
131          }
132       }
133       for (AnnotatedConstructor<? super X> constructor : type.getConstructors()) {
134          if (constructors.get(constructor.getJavaMember()) == null) {
135             constructors.put(constructor.getJavaMember(), new AnnotationBuilder());
136          }
137          mergeAnnotationsOnElement(constructor, overwrite, constructors.get(constructor.getJavaMember()));
138          for (AnnotatedParameter<? super X> p : constructor.getParameters()) {
139             if (constructorParameters.get(constructor.getJavaMember()) == null) {
140                constructorParameters.put(constructor.getJavaMember(), new HashMap<Integer, AnnotationBuilder>());
141             }
142             if (constructorParameters.get(constructor.getJavaMember()).get(p.getPosition()) == null) {
143                constructorParameters.get(constructor.getJavaMember()).put(p.getPosition(), new AnnotationBuilder());
144             }
145             mergeAnnotationsOnElement(p, overwrite, constructorParameters.get(constructor.getJavaMember()).get(p.getPosition()));
146          }
147       }
148       return this;
149    }

162    /**
163     * Reads the annotations from an existing java type. If overwrite is true
164     * then existing annotations will be overwritten
165     *
166     * @param type      the type to read from
167     * @param overwrite if true, the read annotation will replace any existing
168     *                  annotation
169     */
170    public AnnotatedTypeBuilder<X> readFromType(Class<X> type, boolean overwrite) {
171       if (type == null) {
172          throw log.parameterMustNotBeNull("type");
173       }
174       if (javaClass == null || overwrite) {
175          this.javaClass = type;
176       }
177       for (Annotation annotation : type.getAnnotations()) {
178          if (overwrite || !typeAnnotations.isAnnotationPresent(annotation.annotationType())) {
179             typeAnnotations.add(annotation);
180          }
181       }
182 
183       for (Field field : Reflections.getAllDeclaredFields(type)) {
184          AnnotationBuilder annotationBuilder = fields.get(field);
185          if (annotationBuilder == null) {
186             annotationBuilder = new AnnotationBuilder();
187             fields.put(field, annotationBuilder);
188          }
189          field.setAccessible(true);
190          for (Annotation annotation : field.getAnnotations()) {
191             if (overwrite || !annotationBuilder.isAnnotationPresent(annotation.annotationType())) {
192                annotationBuilder.add(annotation);
193             }
194          }
195       }
196 
197       for (Method method : Reflections.getAllDeclaredMethods(type)) {
198          AnnotationBuilder annotationBuilder = methods.get(method);
199          if (annotationBuilder == null) {
200             annotationBuilder = new AnnotationBuilder();
201             methods.put(method, annotationBuilder);
202          }
203          method.setAccessible(true);
204          for (Annotation annotation : method.getAnnotations()) {
205             if (overwrite || !annotationBuilder.isAnnotationPresent(annotation.annotationType())) {
206                annotationBuilder.add(annotation);
207             }
208          }
209 
210          Map<Integer, AnnotationBuilder> parameters = methodParameters.get(method);
211          if (parameters == null) {
212             parameters = new HashMap<Integer, AnnotationBuilder>();
213             methodParameters.put(method, parameters);
214          }
215          for (int i = 0; i < method.getParameterTypes().length; ++i) {
216             AnnotationBuilder parameterAnnotationBuilder = parameters.get(i);
217             if (parameterAnnotationBuilder == null) {
218                parameterAnnotationBuilder = new AnnotationBuilder();
219                parameters.put(i, parameterAnnotationBuilder);
220             }
221             for (Annotation annotation : method.getParameterAnnotations()[i]) {
222                if (overwrite || !parameterAnnotationBuilder.isAnnotationPresent(annotation.annotationType())) {
223                   parameterAnnotationBuilder.add(annotation);
224                }
225             }
226          }
227       }
228 
229       for (Constructor<?> constructor : type.getDeclaredConstructors()) {
230          AnnotationBuilder annotationBuilder = constructors.get(constructor);
231          if (annotationBuilder == null) {
232             annotationBuilder = new AnnotationBuilder();
233             constructors.put(constructor, annotationBuilder);
234          }
235          constructor.setAccessible(true);
236          for (Annotation annotation : constructor.getAnnotations()) {
237             if (overwrite || !annotationBuilder.isAnnotationPresent(annotation.annotationType())) {
238                annotationBuilder.add(annotation);
239             }
240          }
241          Map<Integer, AnnotationBuilder> mparams = constructorParameters.get(constructor);
242          if (mparams == null) {
243             mparams = new HashMap<Integer, AnnotationBuilder>();
244             constructorParameters.put(constructor, mparams);
245          }
246          for (int i = 0; i < constructor.getParameterTypes().length; ++i) {
247             AnnotationBuilder parameterAnnotationBuilder = mparams.get(i);
248             if (parameterAnnotationBuilder == null) {
249                parameterAnnotationBuilder = new AnnotationBuilder();
250                mparams.put(i, parameterAnnotationBuilder);
251             }
252             for (Annotation annotation : constructor.getParameterAnnotations()[i]) {
253                if (overwrite || !parameterAnnotationBuilder.isAnnotationPresent(annotation.annotationType())) {
254                   annotationBuilder.add(annotation);
255                }
256             }
257          }
258       }
259       return this;
260    }
261 
262    protected void mergeAnnotationsOnElement(Annotated annotated, boolean overwriteExisting, AnnotationBuilder typeAnnotations) {
263       for (Annotation annotation : annotated.getAnnotations()) {
264          if (typeAnnotations.getAnnotation(annotation.annotationType()) != null) {
265             if (overwriteExisting) {
266                typeAnnotations.remove(annotation.annotationType());
267                typeAnnotations.add(annotation);
268             }
269          } else {
270             typeAnnotations.add(annotation);
271          }
272       }
273    }

275    /**
276     * Create an {@link AnnotatedType}. Any public members present on the
277     * underlying class and not overridden by the builder will be automatically
278     * added.
279     */
280    public AnnotatedType<X> create() {
281       Map<Constructor<?>, Map<Integer, AnnotationStore>> constructorParameterAnnnotations = new HashMap<Constructor<?>, Map<Integer, AnnotationStore>>();
282       Map<Constructor<?>, AnnotationStore> constructorAnnotations = new HashMap<Constructor<?>, AnnotationStore>();
283       Map<Method, Map<Integer, AnnotationStore>> methodParameterAnnnotations = new HashMap<Method, Map<Integer, AnnotationStore>>();
284       Map<Method, AnnotationStore> methodAnnotations = new HashMap<Method, AnnotationStore>();
285       Map<Field, AnnotationStore> fieldAnnotations = new HashMap<Field, AnnotationStore>();
286 
287       for (Entry<Field, AnnotationBuilder> field : fields.entrySet()) {
288          fieldAnnotations.put(field.getKey(), field.getValue().create());
289       }
290 
291       for (Entry<Method, AnnotationBuilder> method : methods.entrySet()) {
292          methodAnnotations.put(method.getKey(), method.getValue().create());
293       }
294       for (Entry<Method, Map<Integer, AnnotationBuilder>> parameters : methodParameters.entrySet()) {
295          Map<Integer, AnnotationStore> parameterAnnotations = new HashMap<Integer, AnnotationStore>();
296          methodParameterAnnnotations.put(parameters.getKey(), parameterAnnotations);
297          for (Entry<Integer, AnnotationBuilder> parameter : parameters.getValue().entrySet()) {
298             parameterAnnotations.put(parameter.getKey(), parameter.getValue().create());
299          }
300       }
301 
302       for (Entry<Constructor<?>, AnnotationBuilder> constructor : constructors.entrySet()) {
303          constructorAnnotations.put(constructor.getKey(), constructor.getValue().create());
304       }
305       for (Entry<Constructor<?>, Map<Integer, AnnotationBuilder>> parameters : constructorParameters.entrySet()) {
306          Map<Integer, AnnotationStore> parameterAnnotations = new HashMap<Integer, AnnotationStore>();
307          constructorParameterAnnnotations.put(parameters.getKey(), parameterAnnotations);
308          for (Entry<Integer, AnnotationBuilder> parameter : parameters.getValue().entrySet()) {
309             parameterAnnotations.put(parameter.getKey(), parameter.getValue().create());
310          }
311       }
312 
313       return new AnnotatedTypeImpl<X>(javaClass, typeAnnotations.create(), fieldAnnotations, methodAnnotations, methodParameterAnnnotations, constructorAnnotations, constructorParameterAnnnotations, fieldTypes, methodParameterTypes, constructorParameterTypes);
314    }

328     * Remove an annotation from the specified method.
329     *
330     * @param method         the method to remove the annotation from
331     * @param annotationType the annotation type to remove
332     * @throws IllegalArgumentException if the annotationType is null or if the
333     *                                  method is not currently declared on the type
334     */
335    public AnnotatedTypeBuilder<X> removeFromMethod(Method method, Class<? extends Annotation> annotationType) {
336        if (methods.get(method) == null) {
337            throw new IllegalArgumentException("Method not present " + method.toString() + " on " + javaClass);
338        } else {
339            methods.get(method).remove(annotationType);
340        }
341        return this;
342    }

384    /**
385     * Add an annotation to the specified method parameter. If the method is not
386     * already present, it will be added. If the method parameter is not already
387     * present, it will be added.
388     *
389     * @param method     the method to add the annotation to
390     * @param position   the position of the parameter to add
391     * @param annotation the annotation to add
392     * @throws IllegalArgumentException if the annotation is null
393     */
394    public AnnotatedTypeBuilder<X> addToMethodParameter(Method method, int position, Annotation annotation) {
395        if (!methods.containsKey(method)) {
396            methods.put(method, new AnnotationBuilder());
397        }
398        if (methodParameters.get(method) == null) {
399            methodParameters.put(method, new HashMap<Integer, AnnotationBuilder>());
400        }
401        if (methodParameters.get(method).get(position) == null) {
402            methodParameters.get(method).put(position, new AnnotationBuilder());
403        }
404        methodParameters.get(method).get(position).add(annotation);
405        return this;
406    }

408    /**
409     * Remove an annotation from the specified method parameter.
410     *
411     * @param method         the method to remove the annotation from
412     * @param position       the position of the parameter to remove
413     * @param annotationType the annotation type to remove
414     * @throws IllegalArgumentException if the annotationType is null, if the
415     *                                  method is not currently declared on the type or if the
416     *                                  parameter is not declared on the method
417     */
418    public AnnotatedTypeBuilder<X> removeFromMethodParameter(Method method, int position, Class<? extends Annotation> annotationType) {
419        if (methods.get(method) == null) {
420            throw new IllegalArgumentException("Method not present " + method +  " on " + javaClass);
421        } else {
422            if (methodParameters.get(method).get(position) == null) {
423                throw new IllegalArgumentException("Method parameter " + position + " not present on " + method + " on " + javaClass);
424            } else {
425                methodParameters.get(method).get(position).remove(annotationType);
426            }
427        }
428        return this;
429    }

1   package org.infinispan.iteration.impl;
2   
3   import org.infinispan.Cache;
4   import org.infinispan.commons.util.CloseableIterable;
5   import org.infinispan.commons.util.CloseableIterator;
6   import org.infinispan.container.entries.CacheEntry;
7   import org.infinispan.context.Flag;
8   import org.infinispan.filter.Converter;
9   import org.infinispan.filter.KeyValueFilter;
10  import org.infinispan.util.concurrent.ConcurrentHashSet;
11  
12  import java.util.EnumSet;
13  import java.util.Set;
14  import java.util.concurrent.atomic.AtomicBoolean;
15  
16  /**
17   * CloseableIterable that also tracks the iterators it spawns so it can properly close them when the close method is
18   * invoked.
19   *
20   * @author wburns
21   * @since 7.0
22   */
23  public class TrackingEntryIterable<K, V, C> implements CloseableIterable<CacheEntry<K, C>> {
24     protected final EntryRetriever<K, V> entryRetriever;
25     protected final KeyValueFilter<? super K, ? super V> filter;
26     protected final Converter<? super K, ? super V, ? extends C> converter;
27     protected final EnumSet<Flag> flags;
28     protected final Cache<K, V> cache;
29  
30     protected final AtomicBoolean closed = new AtomicBoolean(false);
31     protected final Set<CloseableIterator<CacheEntry<K, C>>> iterators =
32           new ConcurrentHashSet<>();
33  
34     public TrackingEntryIterable(EntryRetriever<K, V> retriever, KeyValueFilter<? super K, ? super V> filter,
35                                   Converter<? super K, ? super V, ? extends C> converter, EnumSet<Flag> flags,
36                                   Cache<K, V> cache) {
37        if (retriever == null) {
38           throw new NullPointerException("Retriever cannot be null!");
39        }
40        if (filter == null) {
41           throw new NullPointerException("Filter cannot be null!");
42        }
43        this.entryRetriever = retriever;
44        this.filter = filter;
45        this.converter = converter;
46        this.flags = flags;
47        this.cache = cache;
48     }
49  
50     @Override
51     public void close() {
52        closed.set(true);
53        for (CloseableIterator<CacheEntry<K, C>> iterator : iterators) {
54           iterator.close();
55        }
56     }
57  
58     @Override
59     public CloseableIterator<CacheEntry<K, C>> iterator() {
60        if (closed.get()) {
61           throw new IllegalStateException("Iterable has been closed - cannot be reused");
62        }
63        // TODO: when an iterator is closed we should remove it from the set - this isn't critical yet though
64        CloseableIterator<CacheEntry<K, C>> iterator = entryRetriever.retrieveEntries(filter, converter, flags, null);
65        iterators.add(iterator);
66        // Note we have to check if we were closed afterwards just in case if a concurrent close occurred.
67        if (closed.get()) {
68           // Rely on fact that multiple closes don't have adverse effects
69           iterator.close();
70           throw new IllegalStateException("Iterable has been closed - cannot be reused");
71        }
72        return new RemovableEntryIterator(iterator, cache, true);
73     }
74  }

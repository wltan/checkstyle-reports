21  /**
22   * <p>
23   * A builder class for creating immutable beans. The builder can create
24   * {@link PassivationCapable} beans, using
25   * {@link Annotateds#createTypeId(AnnotatedType)} to generate the id.
26   * </p>
27   * <p/>
28   * <p>
29   * The builder can read from an {@link AnnotatedType} and have any attribute
30   * modified. This class is not thread-safe, but the bean created by calling
31   * {@link #create()} is.
32   * </p>
33   * <p/>
34   * <p>
35   * It is advised that a new bean builder is instantiated for each bean created.
36   * </p>
37   *
38   * @author Stuart Douglas
39   * @author Pete Muir
40   * @see ImmutableBean
41   * @see ImmutablePassivationCapableBean
42   */
43  public class BeanBuilder<T> {

53      private Set<InjectionPoint> injectionPoints;

61      /**
62       * Instantiate a new bean builder.
63       *
64       * @param beanManager the bean manager to use for creating injection targets
65       *                    and determining if annotations are qualifiers, scopes or
66       *                    stereotypes.
67       */
68      public BeanBuilder(BeanManager beanManager) {
69          this.beanManager = beanManager;
70      }

72      /**
73       * <p>
74       * Read the {@link AnnotatedType}, creating a bean from the class and it's
75       * annotations.
76       * </p>
77       * <p/>
78       * <p>
79       * By default the bean lifecycle will wrap the result of calling
80       * {@link BeanManager#createInjectionTarget(AnnotatedType)}.
81       * </p>
82       * <p/>
83       * <p>
84       * {@link BeanBuilder} does <em>not</em> support reading members of the class
85       * to create producers or observer methods.
86       * </p>
87       *
88       * @param type the type to read
89       */
90      public BeanBuilder<T> readFromType(AnnotatedType<T> type) {
91          this.beanClass = type.getJavaClass();
92          InjectionTarget<T> injectionTarget;
93          if (!type.getJavaClass().isInterface()) {
94              injectionTarget = beanManager.createInjectionTarget(type);
95          } else {
96              injectionTarget = new DummyInjectionTarget<T>();
97          }
98          this.beanLifecycle = new DelegatingContextualLifecycle<T>(injectionTarget);
99          this.injectionPoints = injectionTarget.getInjectionPoints();
100         this.qualifiers = new HashSet<Annotation>();
101         this.stereotypes = new HashSet<Class<? extends Annotation>>();
102         this.types = new HashSet<Type>();
103         for (Annotation annotation : type.getAnnotations()) {
104             if (beanManager.isQualifier(annotation.annotationType())) {
105                 this.qualifiers.add(annotation);
106             } else if (beanManager.isScope(annotation.annotationType())) {
107                 this.scope = annotation.annotationType();
108             } else if (beanManager.isStereotype(annotation.annotationType())) {
109                 this.stereotypes.add(annotation.annotationType());
110             }
111             if (annotation instanceof Named) {
112                 this.name = ((Named) annotation).value();
113             }
114             if (annotation instanceof Alternative) {
115                 this.alternative = true;
116             }
117         }
118         if (this.scope == null) {
119             this.scope = Dependent.class;
120         }
121         for (Class<?> c = type.getJavaClass(); c != Object.class && c != null; c = c.getSuperclass()) {
122             this.types.add(c);
123         }
124         for (Class<?> i : type.getJavaClass().getInterfaces()) {
125             this.types.add(i);
126         }
127         if (qualifiers.isEmpty()) {
128             qualifiers.add(DefaultLiteral.INSTANCE);
129         }
130         qualifiers.add(AnyLiteral.INSTANCE);
131         this.id = ImmutableBean.class.getName() + ":" + Annotateds.createTypeId(type);
132         return this;
133     }

135     /**
136      * <p>
137      * Use the bean builder's current state to define the bean.
138      * </p>
139      *
140      * @return the bean
141      */
142     public Bean<T> create() {
143         if (!passivationCapable) {
144             return new ImmutableBean<T>(beanClass, name, qualifiers, scope, stereotypes, types, alternative, nullable, injectionPoints, beanLifecycle, toString);
145         } else {
146             return new ImmutablePassivationCapableBean<T>(id, beanClass, name, qualifiers, scope, stereotypes, types, alternative, nullable, injectionPoints, beanLifecycle, toString);
147         }
148     }

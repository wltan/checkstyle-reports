1   package org.infinispan.cdi.util.annotatedtypebuilder;
2   
3   import java.lang.reflect.Constructor;
4   import java.lang.reflect.Type;
5   import java.util.Map;
6   
7   import javax.enterprise.inject.spi.AnnotatedConstructor;
8   
9   /**
10   * @author Stuart Douglas
11   */
12  class AnnotatedConstructorImpl<X> extends AnnotatedCallableImpl<X, Constructor<X>> implements AnnotatedConstructor<X> {
13  
14     AnnotatedConstructorImpl(AnnotatedTypeImpl<X> type, Constructor<?> constructor, AnnotationStore annotations, Map<Integer, AnnotationStore> parameterAnnotations, Map<Integer, Type> typeOverrides) {
15        super(type, (Constructor<X>) constructor, constructor.getDeclaringClass(), constructor.getParameterTypes(), getGenericArray(constructor), annotations, parameterAnnotations, null, typeOverrides);
16     }
17  
18     private static Type[] getGenericArray(Constructor<?> constructor) {
19        Type[] genericTypes = constructor.getGenericParameterTypes();
20        // for inner classes genericTypes and parameterTypes can be different
21        // length, this is a hack to fix this.
22        // TODO: investigate this behavior further, on different JVM's and
23        // compilers
24        if (genericTypes.length + 1 == constructor.getParameterTypes().length) {
25           genericTypes = new Type[constructor.getGenericParameterTypes().length + 1];
26           genericTypes[0] = constructor.getParameterTypes()[0];
27           for (int i = 0; i < constructor.getGenericParameterTypes().length; ++i) {
28              genericTypes[i + 1] = constructor.getGenericParameterTypes()[i];
29           }
30        }
31        return genericTypes;
32     }

5   /**
6    * Converter that can be used to transform a given entry to a different value.  This is especially useful to reduce
7    * overall payload of given data that is sent for the given event when a notification is send to a cluster listener as
8    * this will have to be serialized and sent across the network when the cluster listener is not local to the node who
9    * owns the given key.
10   *
11   * @author William Burns
12   * @since 7.0
13   */
14  public interface Converter<K, V, C> {
15     C convert(K key, V value, Metadata metadata);
16  }

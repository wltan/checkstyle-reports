1   package org.infinispan.scripting.impl;
2   
3   import java.util.Iterator;
4   
5   import javax.script.SimpleBindings;
6   
7   import org.infinispan.commons.CacheException;
8   import org.infinispan.distexec.mapreduce.Reducer;
9   import org.infinispan.manager.EmbeddedCacheManager;
10  import org.infinispan.scripting.ScriptingManager;
11  
12  /**
13   * ReducerScript.
14   *
15   * @author Tristan Tarrant
16   * @since 7.2
17   */
18  public class ReducerScript<KOut, VOut> implements Reducer<KOut, VOut>, EnvironmentAware {
19     private final ScriptMetadata metadata;
20     private transient ScriptingManagerImpl scriptManager;
21     private transient SimpleBindings bindings;
22  
23     public ReducerScript(ScriptMetadata metadata) {
24        this.metadata  = metadata;
25     }
26  
27     @Override
28     public VOut reduce(KOut reducedKey, Iterator<VOut> iter) {
29        bindings.put("reducedKey", reducedKey);
30        bindings.put("iter", iter);
31        try {
32           return (VOut) scriptManager.execute(metadata, bindings).get();
33        } catch (Exception e) {
34           throw new CacheException("Error during reducer script", e);
35        }
36     }
37  
38     @Override
39     public void setEnvironment(EmbeddedCacheManager cacheManager) {
40        scriptManager = (ScriptingManagerImpl) SecurityActions.getGlobalComponentRegistry(cacheManager).getComponent(ScriptingManager.class);
41        bindings = new SimpleBindings();
42        bindings.put(SystemBindings.CACHE_MANAGER.toString(), cacheManager);
43     }
44  
45  }

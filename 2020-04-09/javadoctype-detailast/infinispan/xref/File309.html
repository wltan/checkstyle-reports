1   package org.infinispan.objectfilter.impl.predicateindex.be;
2   
3   import org.infinispan.objectfilter.impl.predicateindex.FilterEvalContext;
4   import org.infinispan.objectfilter.impl.predicateindex.IntervalPredicate;
5   import org.infinispan.objectfilter.impl.predicateindex.Predicate;
6   
7   import java.util.List;
8   
9   /**
10   * A PredicateNode is a leaf node in a BETree that holds a Predicate instance. A PredicateNode instance is never reused
11   * inside the same BETree or shared between multiple BETrees, but an entire BETree could be shared by multiple filters.
12   * Multiple PredicateNodes could share the same Predicate instance.
13   *
14   * @author anistor@redhat.com
15   * @since 7.0
16   */
17  public final class PredicateNode<AttributeId extends Comparable<AttributeId>> extends BENode {
18  
19     // the predicate can be shared by multiple PredicateNodes
20     private final Predicate<?> predicate;
21  
22     /**
23      * Indicates if the Predicate's condition is negated. This can be true only for condition predicates, never for
24      * interval predicates.
25      */
26     private final boolean isNegated;
27  
28     private final List<AttributeId> attributePath;
29  
30     public PredicateNode(BENode parent, Predicate<?> predicate, boolean isNegated, List<AttributeId> attributePath) {
31        super(parent);
32        if (isNegated && predicate instanceof IntervalPredicate) {
33           throw new IllegalArgumentException("Interval predicates should not be negated");
34        }
35        this.predicate = predicate;
36        this.isNegated = isNegated;
37        this.attributePath = attributePath;
38     }
39  
40     public Predicate<?> getPredicate() {
41        return predicate;
42     }
43  
44     public boolean isNegated() {
45        return isNegated;
46     }
47  
48     public List<AttributeId> getAttributePath() {
49        return attributePath;
50     }
51  
52     @Override
53     public void handleChildValue(BENode child, boolean childValue, FilterEvalContext evalContext) {
54        if (child != null) {
55           throw new IllegalArgumentException("Predicates have value but do not have children");
56        }
57  
58        final int value = childValue ? BETree.EXPR_TRUE : BETree.EXPR_FALSE;
59  
60        if (isEvaluationComplete(evalContext)) {
61           if (predicate.isRepeated() && evalContext.treeCounters[startIndex] == value) {
62              // receiving the same value multiple times is fine if this is a repeated condition
63              return;
64           }
65           throw new IllegalStateException("This should never be called again if the state of this node was previously decided.");
66        }
67  
68        if (parent == null) {
69           evalContext.treeCounters[startIndex] = value;
70           suspendSubscription(evalContext);
71        } else {
72           parent.handleChildValue(this, childValue, evalContext);
73        }
74     }
75  
76     @Override
77     public void suspendSubscription(FilterEvalContext ctx) {
78        if (predicate.isRepeated()) {
79           ctx.matcherContext.addSuspendedSubscription(predicate);
80        }
81     }
82  
83     @Override
84     public String toString() {
85        return "PredicateNode{" +
86              "attributePath=" + attributePath +
87              ", isNegated=" + isNegated +
88              ", predicate=" + predicate +
89              '}';
90     }
91  }

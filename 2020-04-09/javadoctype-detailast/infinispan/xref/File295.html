1   package org.infinispan.objectfilter.impl;
2   
3   import java.util.List;
4   
5   /**
6    * A generic representation of some of the aspects of type metadata. This is meant to ensure decoupling from the
7    * underlying metadata representation (Class, Protobuf descriptor, etc).
8    *
9    * @author anistor@redhat.com
10   * @since 7.0
11   */
12  public interface MetadataAdapter<TypeMetadata, AttributeMetadata, AttributeId extends Comparable<AttributeId>> {
13  
14     String getTypeName();
15  
16     TypeMetadata getTypeMetadata();
17  
18     /**
19      * Transforms a String property path into an internal representation of the path which might not be String based
20      * (AttributeId is an Integer in the Protobuf case).
21      */
22     List<AttributeId> translatePropertyPath(List<String> path);
23  
24     AttributeMetadata makeChildAttributeMetadata(AttributeMetadata parentAttributeMetadata, AttributeId attribute);
25  
26     boolean isComparableProperty(AttributeMetadata attributeMetadata);
27  }

12  /**
13   * This is a tree-like facade around a {@link Cache} allowing for efficient tree-style access to cached data.
14   * <p/>
15   * The primary purpose of this interface is to allow for efficient caching of tree-like structures such as directories,
16   * as well as to provide a compatibility layer with JBoss Cache 3.x and earlier.
17   * <p/>
18   * For most purposes, we expect people to use the {@link Cache} interface directly as it is simpler.
19   * <p/>
20   * The tree API assumes that a collection of {@link Node}s, organized in a tree structure underneath a root node,
21   * contains key/value attributes of data.
22   * <p/>
23   * Any locking happens on a node-level granularity, which means that all attributes on a node are atomic and in terms of
24   * locking, is coarse grained.  At the same time, replication is fine grained, and only modified attributes in a Node
25   * are replicated.
26   * <p/>
27   * Obtaining a TreeCache is done using the {@link TreeCacheFactory}.
28   * <pre>
29   *   Cache cache = new DefaultCacheFactory().getCache();
30   *   TreeCacheFactory tcf = new TreeCacheFactory();
31   *   TreeCache tree = tcf.createTreeCache(cache);
32   * </pre>
33   *
34   * @author Manik Surtani (<a href="mailto:manik AT jboss DOT org">manik AT jboss DOT org</a>)
35   * @see Node
36   * @since 4.0
37   */
38  public interface TreeCache<K, V> extends Lifecycle {
39     /**
40      * Returns the root node of this cache.
41      *
42      * @return the root node
43      */
44     Node<K, V> getRoot();
45  
46     Node<K, V> getRoot(Flag... flags);
47  
48     /**
49      * Associates the specified value with the specified key for a {@link Node} in this cache. If the {@link Node}
50      * previously contained a mapping for this key, the old value is replaced by the specified value.
51      *
52      * @param fqn   <b><i>absolute</i></b> {@link Fqn} to the {@link Node} to be accessed.
53      * @param key   key with which the specified value is to be associated.
54      * @param value value to be associated with the specified key.
55      * @return previous value associated with specified key, or <code>null</code> if there was no mapping for key. A
56      *         <code>null</code> return can also indicate that the Node previously associated <code>null</code> with the
57      *         specified key, if the implementation supports null values.
58      * @throws IllegalStateException if the cache is not in a started state.
59      */
60     V put(Fqn fqn, K key, V value);
61  
62     V put(Fqn fqn, K key, V value, Flag... flags);
63  
64     /**
65      * Convenience method that takes a string representation of an Fqn.  Otherwise identical to {@link #put(Fqn, Object,

69      * @param key   key with which the specified value is to be associated.
70      * @param value value to be associated with the specified key.
71      * @return previous value associated with specified key, or <code>null</code> if there was no mapping for key. A
72      *         <code>null</code> return can also indicate that the Node previously associated <code>null</code> with the
73      *         specified key, if the implementation supports null values.
74      * @throws IllegalStateException if the cache is not in a started state
75      */
76  
77     V put(String fqn, K key, V value);
78  
79     V put(String fqn, K key, V value, Flag... flags);
80  
81     /**
82      * Copies all of the mappings from the specified map to a {@link Node}.

84      * @param fqn  <b><i>absolute</i></b> {@link Fqn} to the {@link Node} to copy the data to
85      * @param data mappings to copy
86      * @throws IllegalStateException if the cache is not in a started state
87      */
88     void put(Fqn fqn, Map<? extends K, ? extends V> data);
89  
90     void put(Fqn fqn, Map<? extends K, ? extends V> data, Flag... flags);

92     /**
93      * Convenience method that takes a string representation of an Fqn.  Otherwise identical to {@link #put(Fqn,

201    /**
202     * Moves a part of the cache to a different subtree.
203     * <p/>
204     * E.g.:
205     * <p/>
206     * assume a cache structure such as:
207     * <p/>
208     * <pre>
209     *  /a/b/c
210     *  /a/b/d
211     *  /a/b/e
212     * <p/>
213     * <p/>
214     *  Fqn f1 = Fqn.fromString("/a/b/c");
215     *  Fqn f2 = Fqn.fromString("/a/b/d");
216     * <p/>
217     *  cache.move(f1, f2);
218     * </pre>
219     * <p/>
220     * Will result in:
221     * <pre>
222     * <p/>
223     * /a/b/d/c
224     * /a/b/e
225     * <p/>
226     * </pre>
227     * <p/>
228     * and now
229     * <p/>
230     * <pre>
231     *  Fqn f3 = Fqn.fromString("/a/b/e");
232     *  Fqn f4 = Fqn.fromString("/a");
233     *  cache.move(f3, f4);
234     * </pre>
235     * <p/>
236     * will result in:
237     * <pre>
238     * /a/b/d/c
239     * /a/e
240     * </pre>
241     * No-op if the node to be moved is the root node.
242     * <p/>
243     * <b>Note</b>: As of 3.0.0 and when using MVCC locking, more specific behaviour is defined as follows: <ul> <li>A
244     * no-op if the node is moved unto itself.  E.g., <tt>move(fqn, fqn.getParent())</tt> will not do anything.</li>
245     * <li>If a target node does not exist it will be created silently, to be more consistent with other APIs such as
246     * <tt>put()</tt> on a nonexistent node.</li> <li>If the source node does not exist this is a no-op, to be more
247     * consistent with other APIs such as <tt>get()</tt> on a nonexistent node.</li> </ul>
248     *
249     * @param nodeToMove the Fqn of the node to move.
250     * @param newParent  new location under which to attach the node being moved.
251     * @throws NodeNotExistsException may throw one of these if the target node does not exist or if a different thread
252     *                                has moved this node elsewhere already.
253     * @throws IllegalStateException  if {@link Cache#getStatus()} would not return {@link ComponentStatus#RUNNING}.
254     */
255    void move(Fqn nodeToMove, Fqn newParent) throws NodeNotExistsException;
256 
257    void move(Fqn nodeToMove, Fqn newParent, Flag... flags) throws NodeNotExistsException;
258 
259    /**
260     * Convenience method that takes in string representations of Fqns.  Otherwise identical to {@link #move(Fqn, Fqn)}
261     *
262     * @throws IllegalStateException if {@link Cache#getStatus()} would not return {@link ComponentStatus#RUNNING}.
263     */
264    void move(String nodeToMove, String newParent) throws NodeNotExistsException;
265 
266    void move(String nodeToMove, String newParent, Flag... flags) throws NodeNotExistsException;

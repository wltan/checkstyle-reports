1   package org.infinispan.client.hotrod;
2   
3   /**
4    * Besides the value, also contains a version and expiration information. Time values are server
5    * time representations as returned by {@link org.infinispan.util.TimeService#wallClockTime}
6    *
7    * @author Tristan Tarrant
8    * @since 5.2
9    */
10  public interface MetadataValue<V> extends VersionedValue<V> {
11  
12     /**
13      *
14      * @return Time when entry was created. -1 for immortal entries.
15      */
16     long getCreated();
17  
18     /**
19      *
20      * @return Lifespan of the entry in seconds. Negative values are interpreted as unlimited
21      *         lifespan.
22      */
23     int getLifespan();
24  
25     /**
26      *
27      * @return Time when entry was last used. -1 for immortal entries.
28      */
29     long getLastUsed();
30  
31     /**
32      *
33      * @return The maximum amount of time (in seconds) this key is allowed to be idle for before it
34      *         is considered as expired.
35      */
36     int getMaxIdle();
37  
38  }

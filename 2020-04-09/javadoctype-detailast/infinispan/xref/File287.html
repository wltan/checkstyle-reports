1   package org.infinispan.jcache.embedded;
2   
3   import org.infinispan.jcache.AbstractJCache;
4   import org.infinispan.jcache.AbstractJCacheListenerAdapter;
5   import org.infinispan.jcache.AbstractJCacheNotifier;
6   import org.infinispan.notifications.Listener;
7   import org.infinispan.notifications.cachelistener.annotation.CacheEntryCreated;
8   import org.infinispan.notifications.cachelistener.annotation.CacheEntryModified;
9   import org.infinispan.notifications.cachelistener.annotation.CacheEntryRemoved;
10  import org.infinispan.notifications.cachelistener.event.CacheEntryCreatedEvent;
11  import org.infinispan.notifications.cachelistener.event.CacheEntryModifiedEvent;
12  import org.infinispan.notifications.cachelistener.event.CacheEntryRemovedEvent;
13  
14  /**
15   * Adapts Infinispan notification mechanism to JSR 107 requirements.
16   *
17   * @author Vladimir Blagojevic
18   * @author Galder Zamarre√±o
19   * @since 5.3
20   */
21  @Listener
22  public class JCacheListenerAdapter<K, V> extends AbstractJCacheListenerAdapter<K, V> {
23     public JCacheListenerAdapter(AbstractJCache<K, V> jcache, AbstractJCacheNotifier<K, V> notifier) {
24        super(jcache, notifier);
25     }
26  
27     @CacheEntryCreated
28     public void handleCacheEntryCreatedEvent(CacheEntryCreatedEvent<K, V> e) {
29        // JCache listeners notified only once, so do it after the event
30        if (!e.isPre())
31           notifier.notifyEntryCreated(jcache, e.getKey(), e.getValue());
32     }
33  
34     @CacheEntryModified
35     public void handleCacheEntryModifiedEvent(CacheEntryModifiedEvent<K, V> e) {
36        // JCache listeners notified only once, so do it after the event
37        if (!e.isPre() && !e.isCreated())
38           notifier.notifyEntryUpdated(jcache, e.getKey(), e.getValue());
39     }
40  
41     @CacheEntryRemoved
42     public void handleCacheEntryRemovedEvent(CacheEntryRemovedEvent<K, V> e) {
43        // JCache listeners notified only once, so do it after the event
44        if (!e.isPre())
45           notifier.notifyEntryRemoved(jcache, e.getKey(), e.getOldValue());
46     }
47  }

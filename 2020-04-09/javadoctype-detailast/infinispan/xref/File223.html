1   package org.infinispan.notifications.cachelistener.event;
2   
3   /**
4    * This event is passed in to any method annotated with {@link org.infinispan.notifications.cachelistener.annotation.TransactionCompleted}.
5    * <p/>
6    * Note that this event is only delivered <i>after the fact</i>, i.e., you will never see an instance of this event with
7    * {@link #isPre()} being set to <tt>true</tt>.
8    *
9    * @author Manik Surtani
10   * @since 4.0
11   */
12  public interface TransactionCompletedEvent<K, V> extends TransactionalEvent<K, V> {
13     /**
14      * @return if <tt>true</tt>, the transaction completed by committing successfully.  If <tt>false</tt>, the
15      *         transaction completed with a rollback.
16      */
17     boolean isTransactionSuccessful();
18  }

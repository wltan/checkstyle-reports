1   package org.infinispan.scripting.impl;
2   
3   import javax.script.SimpleBindings;
4   
5   import org.infinispan.distexec.mapreduce.Collector;
6   import org.infinispan.distexec.mapreduce.Mapper;
7   import org.infinispan.manager.EmbeddedCacheManager;
8   import org.infinispan.scripting.ScriptingManager;
9   
10  /**
11   * MapperScript.
12   *
13   * @author Tristan Tarrant
14   * @since 7.2
15   */
16  public class MapperScript<KIn, VIn, KOut, VOut> implements Mapper<KIn, VIn, KOut, VOut>, EnvironmentAware {
17     private final ScriptMetadata metadata;
18     private transient ScriptingManagerImpl scriptManager;
19     private transient SimpleBindings bindings;
20  
21     public MapperScript(ScriptMetadata metadata) {
22        this.metadata = metadata;
23     }
24  
25     @Override
26     public void map(KIn key, VIn value, Collector<KOut, VOut> collector) {
27        bindings.put("key", key);
28        bindings.put("value", value);
29        bindings.put("collector", collector);
30        scriptManager.execute(metadata, bindings);
31     }
32  
33     @Override
34     public void setEnvironment(EmbeddedCacheManager cacheManager) {
35        scriptManager = (ScriptingManagerImpl) SecurityActions.getGlobalComponentRegistry(cacheManager).getComponent(ScriptingManager.class);
36        bindings = new SimpleBindings();
37        bindings.put("marshaller", scriptManager.getMarshaller());
38        bindings.put("cacheManager", cacheManager);
39     }
40  
41  }

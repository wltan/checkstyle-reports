1   package org.infinispan.notifications.cachelistener.event;
2   
3   import org.infinispan.Cache;
4   
5   /**
6    * An interface that defines common characteristics of events
7    *
8    * @author Manik Surtani
9    * @since 4.0
10   */
11  public interface Event<K, V> {
12     enum Type {
13        CACHE_ENTRY_ACTIVATED, CACHE_ENTRY_PASSIVATED, CACHE_ENTRY_VISITED,
14        CACHE_ENTRY_LOADED, CACHE_ENTRY_EVICTED, CACHE_ENTRY_CREATED, CACHE_ENTRY_REMOVED, CACHE_ENTRY_MODIFIED,
15        TRANSACTION_COMPLETED, TRANSACTION_REGISTERED, CACHE_ENTRY_INVALIDATED, DATA_REHASHED, TOPOLOGY_CHANGED,
16        PARTITION_STATUS_CHANGED
17     }
18  
19     /**
20      * @return the type of event represented by this instance.
21      */
22     Type getType();
23  
24     /**
25      * @return <tt>true</tt> if the notification is before the event has occurred, <tt>false</tt> if after the event has occurred.
26      */
27     boolean isPre();
28  
29     /**
30      * @return a handle to the cache instance that generated this notification.
31      */
32     Cache<K, V> getCache();
33  }

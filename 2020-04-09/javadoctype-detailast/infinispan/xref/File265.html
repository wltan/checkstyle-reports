1   package org.infinispan.util.concurrent.locks.containers;
2   
3   import java.util.concurrent.TimeUnit;
4   import java.util.concurrent.locks.Lock;
5   
6   /**
7    * A container for locks
8    *
9    * @author Manik Surtani
10   * @author Mircea.Markus@jboss.com
11   * @since 4.0
12   */
13  public interface LockContainer<L extends Lock> {
14     /**
15      * Tests if a give owner owns a lock on a specified object.
16      *
17      * @param key   object to check
18      * @param owner owner to test
19      * @return true if owner owns lock, false otherwise
20      */
21     boolean ownsLock(Object key, Object owner);
22  
23     /**
24      * @param key object
25      * @return true if an object is locked, false otherwise
26      */
27     boolean isLocked(Object key);
28  
29     /**
30      * @param key object
31      * @return the lock for a specific object.  May be null if the object is not locked, but may also be an unlocked lock.
32      */
33     L getLock(Object key);
34  
35     /**
36      * @return number of locks held
37      */
38     int getNumLocksHeld();
39  
40     /**
41      * @return the size of the shared lock pool
42      */
43     int size();
44  
45     /**
46      * Attempts to acquire a lock for the given object within certain time boundaries defined by the timeout and
47      * time unit parameters.
48      *
49      * @param key Object to acquire lock on
50      * @param timeout Time after which the lock acquisition will fail
51      * @param unit Time unit of the given timeout
52      * @return If lock was acquired it returns the corresponding Lock object. If lock was not acquired, it returns null
53      * @throws InterruptedException If the lock acquisition was interrupted
54      */
55     L acquireLock(Object lockOwner, Object key, long timeout, TimeUnit unit) throws InterruptedException;
56  
57     /**
58      * Release lock on the given key.
59      *
60      * @param key Object on which lock is to be removed
61      */
62     void releaseLock(Object lockOwner, Object key);
63  
64     /**
65      * Returns the 'id' of the lock that will be used to guard access to a given key in the cache.  Particularly useful
66      * if Lock Striping is used and locks may guard more than one key.  This mechanism can be used to check whether
67      * keys may end up sharing the same lock.
68      * <p />
69      * If lock-striping is not used, the identity hash code of the lock created for this specific key is returned, if the
70      * key is locked, or -1 if the key is not locked and a lock does not exist for the key.
71      *
72      * @param key key to test for
73      * @return the ID of the lock.
74      */
75     int getLockId(Object key);
76  }

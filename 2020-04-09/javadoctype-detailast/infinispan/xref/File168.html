1   package org.infinispan.filter;
2   
3   import org.infinispan.metadata.Metadata;
4   
5   /**
6    * This is a base class that should be used when implementing a KeyValueFilterConverter that provides default
7    * implementations for the {@link org.infinispan.filter.KeyValueFilter#accept(Object, Object, org.infinispan.metadata.Metadata)}
8    * and {@link org.infinispan.filter.Converter#convert(Object, Object, org.infinispan.metadata.Metadata)} methods so they just call the
9    * {@link org.infinispan.filter.KeyValueFilterConverter#filterAndConvert(Object, Object, org.infinispan.metadata.Metadata)}
10   * method and then do the right thing.
11   *
12   * @author wburns
13   * @since 7.0
14   */
15  public abstract class AbstractKeyValueFilterConverter<K, V, C> implements KeyValueFilterConverter<K, V, C> {
16     @Override
17     public final C convert(K key, V value, Metadata metadata) {
18        return filterAndConvert(key, value, metadata);
19     }
20  
21     @Override
22     public final boolean accept(K key, V value, Metadata metadata) {
23        return filterAndConvert(key, value, metadata) != null;
24     }
25  }

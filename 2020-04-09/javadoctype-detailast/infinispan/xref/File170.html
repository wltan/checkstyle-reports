1   package org.infinispan.filter;
2   
3   import org.infinispan.commons.io.UnsignedNumeric;
4   import org.infinispan.commons.marshall.AbstractExternalizer;
5   import org.infinispan.commons.util.Util;
6   import org.infinispan.factories.ComponentRegistry;
7   import org.infinispan.factories.annotations.Inject;
8   import org.infinispan.marshall.core.Ids;
9   
10  import java.io.IOException;
11  import java.io.ObjectInput;
12  import java.io.ObjectOutput;
13  import java.util.Set;
14  
15  /**
16   * Allows AND-composing several filters.
17   *
18   * @author Mircea Markus
19   * @since 6.0
20   */
21  public class CompositeKeyFilter<K> implements KeyFilter<K> {
22     private final KeyFilter<? super K>[] filters;
23  
24     public CompositeKeyFilter(KeyFilter<? super K>... filters) {
25        this.filters = filters;
26     }
27  
28     @Override
29     public boolean accept(K key) {
30        for (KeyFilter<? super K> k : filters)
31           if (!k.accept(key)) return false;
32        return true;
33     }
34  
35     @Inject
36     protected void injectDependencies(ComponentRegistry cr) {
37        for (KeyFilter<? super K> f : filters) {
38           cr.wireDependencies(f);
39        }
40     }
41  
42     public static class Externalizer extends AbstractExternalizer<CompositeKeyFilter> {
43  
44        @Override
45        public Set<Class<? extends CompositeKeyFilter>> getTypeClasses() {
46           return Util.<Class<? extends CompositeKeyFilter>>asSet(CompositeKeyFilter.class);
47        }
48  
49        @Override
50        public void writeObject(ObjectOutput output, CompositeKeyFilter object) throws IOException {
51           UnsignedNumeric.writeUnsignedInt(output, object.filters.length);
52           for (KeyFilter filter : object.filters) {
53              output.writeObject(filter);
54           }
55        }
56  
57        @Override
58        public CompositeKeyFilter readObject(ObjectInput input) throws IOException, ClassNotFoundException {
59           int filtersSize = UnsignedNumeric.readUnsignedInt(input);
60           KeyFilter[] filters = new KeyFilter[filtersSize];
61           for (int i = 0; i < filtersSize; ++i) {
62              filters[i] = (KeyFilter)input.readObject();
63           }
64           return new CompositeKeyFilter(filters);
65        }
66  
67        @Override
68        public Integer getId() {
69           return Ids.COMPOSITE_KEY_FILTER;
70        }
71     }
72  
73  }

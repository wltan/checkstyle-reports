1   package org.infinispan.query.dsl.impl;
2   
3   import org.infinispan.query.dsl.FilterConditionBeginContext;
4   import org.infinispan.query.dsl.FilterConditionContext;
5   import org.infinispan.query.dsl.FilterConditionEndContext;
6   import org.infinispan.query.dsl.Query;
7   import org.infinispan.query.dsl.QueryFactory;
8   
9   /**
10   * @author anistor@redhat.com
11   * @since 6.0
12   */
13  public abstract class BaseQueryFactory<T extends Query> implements QueryFactory<T> {
14  
15     @Override
16     public FilterConditionEndContext having(String attributePath) {
17        return new AttributeCondition(this, attributePath);
18     }
19  
20     @Override
21     public FilterConditionBeginContext not() {
22        return new IncompleteCondition(this).not();
23     }
24  
25     @Override
26     public FilterConditionContext not(FilterConditionContext fcc) {
27        if (fcc == null) {
28           throw new IllegalArgumentException("Argument cannot be null");
29        }
30  
31        BaseCondition baseCondition = ((BaseCondition) fcc).getRoot();
32        if (baseCondition.queryFactory != this) {
33           throw new IllegalArgumentException("The given condition was created by a different factory");
34        }
35        if (baseCondition.queryBuilder != null) {
36           throw new IllegalArgumentException("The given condition is already in use by another builder");
37        }
38        NotCondition notCondition = new NotCondition(this, baseCondition);
39        baseCondition.setParent(notCondition);
40        return notCondition;
41     }
42  }

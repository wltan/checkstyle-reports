1   package org.infinispan.distexec;
2   
3   import java.util.concurrent.Callable;
4   
5   /**
6    * DistributedTask describes all relevant attributes of a distributed task, most importantly its
7    * execution policy, fail over policy and its timeout.
8    * 
9    * @author Manik Surtani
10   * @author Vladimir Blagojevic
11   * @since 5.2
12   */
13  public interface DistributedTask<T> {
14  
15     /**
16      * Returns timeout for the execution of this task
17      * 
18      * @return task timeout
19      */
20     long timeout();
21  
22     /**
23      * Returns custom {@link DistributedTaskExecutionPolicy} for this task
24      * 
25      * @return task DistributedTaskExecutionPolicy
26      */
27     DistributedTaskExecutionPolicy getTaskExecutionPolicy();
28     
29     /**
30      * Returns custom {@link DistributedTaskFailoverPolicy}  for this task
31      * 
32      * @return
33      */
34     DistributedTaskFailoverPolicy getTaskFailoverPolicy();
35  
36     /**
37      * Returns {@link Callable} for this task
38      * 
39      * @return task callable 
40      */
41     Callable<T> getCallable();
42  
43  }

1   package org.infinispan.notifications.cachelistener.cluster;
2   
3   import org.infinispan.Cache;
4   import org.infinispan.factories.annotations.Inject;
5   import org.infinispan.metadata.Metadata;
6   import org.infinispan.notifications.cachelistener.filter.AbstractCacheEventFilterConverter;
7   import org.infinispan.notifications.cachelistener.filter.EventType;
8   
9   import java.io.Serializable;
10  
11  /**
12   * @author anistor@redhat.com
13   * @since 7.2
14   */
15  public class NoOpCacheEventFilterConverterWithDependencies<K, V> extends AbstractCacheEventFilterConverter<K, V, V> implements Serializable {
16  
17     private transient Cache cache;
18  
19     @Inject
20     protected void injectDependencies(Cache cache) {
21        this.cache = cache;
22     }
23  
24     @Override
25     public V filterAndConvert(K key, V oldValue, Metadata oldMetadata, V newValue, Metadata newMetadata, EventType eventType) {
26        if (cache == null) {
27           throw new IllegalStateException("Dependencies were not injected");
28        }
29        return newValue;
30     }
31  }

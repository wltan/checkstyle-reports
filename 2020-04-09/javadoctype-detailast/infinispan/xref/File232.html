1   package org.infinispan.notifications.cachelistener.filter;
2   
3   import org.infinispan.metadata.Metadata;
4   
5   /**
6    * This interface is an optimization that can be used when an event filter and converter are most efficiently used as
7    * the same object composing the filtering and conversion in the same method invocation.
8    *
9    * @author wburns
10   * @since 7.0
11   */
12  public interface CacheEventFilterConverter<K, V, C> extends CacheEventFilter<K, V>, CacheEventConverter<K, V, C> {
13     /**
14      * Will both filter the entry and if passed subsequently convert the value to a new value.  A returned value of null
15      * will symbolize the value not passing the filter, so ensure your conversion will not return null if you want this
16      * entry to be returned.
17      * @param key The key for the entry that was changed for the event
18      * @param oldValue The previous value before the event takes place
19      * @param oldMetadata The old value before the event takes place
20      * @param newValue The new value for the entry after the event takes place
21      * @param newMetadata The new metadata for the entry after the event takes place
22      * @param eventType The type of event that is being raised
23      * @return A non null value converted value when it also passes the filter or null for when it doesn't pass the filter
24      */
25     public C filterAndConvert(K key, V oldValue, Metadata oldMetadata, V newValue, Metadata newMetadata,
26                               EventType eventType);
27  }

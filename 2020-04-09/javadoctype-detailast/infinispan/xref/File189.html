1   package org.infinispan.iteration.impl;
2   
3   import org.infinispan.Cache;
4   import org.infinispan.commons.util.CloseableIterable;
5   import org.infinispan.container.entries.CacheEntry;
6   import org.infinispan.context.impl.TxInvocationContext;
7   import org.infinispan.filter.Converter;
8   import org.infinispan.filter.KeyValueFilter;
9   import org.infinispan.iteration.EntryIterable;
10  import org.infinispan.transaction.impl.LocalTransaction;
11  
12  /**
13   * {@inheritDoc}
14   *
15   * @author wburns
16   * @since 7.0
17   */
18  public class TransactionAwareEntryIterable<K, V> extends TransactionAwareCloseableIterable<K, V, V> implements EntryIterable<K, V> {
19     private final EntryIterable<K, V> entryIterable;
20  
21     public TransactionAwareEntryIterable(EntryIterable<K, V> entryIterable, 
22           KeyValueFilter<? super K, ? super V> filter, TxInvocationContext<LocalTransaction> ctx,
23                                          Cache<K, V> cache) {
24        super(entryIterable, filter, null, ctx, cache);
25        this.entryIterable = entryIterable;
26     }
27  
28     @Override
29     public <C> CloseableIterable<CacheEntry<K, C>> converter(Converter<? super K, ? super V, ? extends C> converter) {
30        return new TransactionAwareCloseableIterable<>(entryIterable.converter(converter),
31              filter, converter, ctx, cache);
32     }
33  }

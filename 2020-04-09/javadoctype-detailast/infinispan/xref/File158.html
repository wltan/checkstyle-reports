1   package org.infinispan.distribution.group;
2   
3   import org.infinispan.filter.KeyFilter;
4   
5   /**
6    * A key filter that accepts keys which belongs to the group.
7    *
8    * @author Pedro Ruivo
9    * @since 7.0
10   */
11  public class GroupFilter<K> implements KeyFilter<K> {
12  
13     private final String groupName;
14     private final GroupManager groupManager;
15  
16     public GroupFilter(String groupName, GroupManager groupManager) {
17        this.groupName = groupName;
18        this.groupManager = groupManager;
19     }
20  
21     @Override
22     public boolean accept(K key) {
23        String keyGroup = groupManager.getGroup(key);
24        return keyGroup != null && keyGroup.equals(groupName);
25     }
26  }

1   package org.infinispan.commons.util;
2   
3   import java.util.Iterator;
4   
5   /**
6    * This is a simple bridge class to change a plain {@link java.util.Iterator} into a
7    * {@link org.infinispan.commons.util.CloseableIterator} where the close method does nothing
8    *
9    * @author wburns
10   * @since 7.0
11   */
12  public class IteratorAsCloseableIterator<E> implements CloseableIterator<E> {
13     private final Iterator<E> iterator;
14  
15     public IteratorAsCloseableIterator(Iterator<E> iterator) {
16        this.iterator = iterator;
17     }
18  
19     @Override
20     public void close() {
21        // This does nothing
22     }
23  
24     @Override
25     public boolean hasNext() {
26        return iterator.hasNext();
27     }
28  
29     @Override
30     public E next() {
31        return iterator.next();
32     }
33  
34     @Override
35     public void remove() {
36        iterator.remove();
37     }
38  }

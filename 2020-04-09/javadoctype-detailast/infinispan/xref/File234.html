1   package org.infinispan.notifications.cachelistener.filter;
2   
3   import org.infinispan.factories.ComponentRegistry;
4   import org.infinispan.factories.annotations.Inject;
5   import org.infinispan.metadata.Metadata;
6   
7   /**
8    * Allows AND-composing several cache event filters.
9    *
10   * @author wburns
11   * @since 7.0
12   */
13  public class CompositeCacheEventFilter<K, V> implements CacheEventFilter<K, V> {
14     private final CacheEventFilter<? super K, ? super V>[] filters;
15  
16     public CompositeCacheEventFilter(CacheEventFilter<? super K, ? super V>... filters) {
17        this.filters = filters;
18     }
19  
20     @Override
21     public boolean accept(K key, V oldValue, Metadata oldMetadata, V newValue, Metadata newMetadata, EventType eventType) {
22        for (CacheEventFilter<? super K, ? super V> f : filters)
23           if (!f.accept(key, oldValue, oldMetadata, newValue, newMetadata, eventType)) return false;
24        return true;
25     }
26  
27     @Inject
28     protected void injectDependencies(ComponentRegistry cr) {
29        for (CacheEventFilter<? super K, ? super V> f : filters) {
30           cr.wireDependencies(f);
31        }
32     }
33  }

1   package org.infinispan.notifications.cachelistener.event;
2   
3   /**
4    * This event subtype is passed in to any method annotated with {@link org.infinispan.notifications.cachelistener.annotation.CacheEntryModified}
5    * <p />
6    * The {@link #getValue()} method's behavior is specific to whether the callback is triggered before or after the event
7    * in question.  For example, if <tt>event.isPre()</tt> is <tt>true</tt>, then <tt>event.getValue()</tt> would return the
8    * <i>old</i> value, prior to modification.  If <tt>event.isPre()</tt> is <tt>false</tt>, then <tt>event.getValue()</tt>
9    * would return new <i>new</i> value.  If the event is creating and inserting a new entry, the old value would be <tt>null</tt>.
10   * <p />
11   * @author Manik Surtani
12   * @since 4.0
13   */
14  public interface CacheEntryModifiedEvent<K, V> extends CacheEntryEvent<K, V> {
15  
16     /**
17      * Retrieves the value of the entry being modified.
18      * <p />
19      * @return the previous or new value of the entry, depending on whether isPre() is true or false.
20      */
21     V getValue();
22  
23     /**
24      * Indicates whether the cache entry modification event is the result of
25      * the cache entry being created. This method helps determine if the cache
26      * entry was created when <tt>event.isPre()</tt> is <tt>false</tt>.
27      *
28      * @return true if the event is the result of the entry being created,
29      * otherwise it returns false indicating that the event was the result of
30      * a cache entry being updated
31      */
32     boolean isCreated();
33  
34     /**
35      * This will be true if the write command that caused this had to be retried again due to a topology change.  This
36      * could be a sign that this event has been duplicated or another event was dropped and replaced
37      * (eg: ModifiedEvent replaced CreateEvent)
38      * @return Whether the command that caused this event was retried
39      */
40     boolean isCommandRetried();
41  
42  }

1   package org.infinispan.iteration;
2   
3   import org.infinispan.commons.util.CloseableIterable;
4   import org.infinispan.container.entries.CacheEntry;
5   import org.infinispan.filter.Converter;
6   
7   
8   /**
9    * A {@link java.lang.Iterable} instance that allows the user to iterate over the entries in the cache.  This
10   * also implements {@link java.io.Closeable} of which the {@link java.io.Closeable#close()} should be invoked
11   * when iteration of all needed Iterables is complete.
12   *
13   * @author wburns
14   * @since 7.0
15   */
16  public interface EntryIterable<K, V> extends CloseableIterable<CacheEntry<K, V>> {
17     /**
18      * This returns a {@link org.infinispan.commons.util.CloseableIterable} that will change the type of the returned
19      * value for the entry using the already provided filter in addition to the converter.
20      * @param converter The converter to apply to the iterator that is produced.  Callbacks to to this converter will
21      *                  never provide a key or value that is null.
22      * @param <C> The type of the converted value
23      * @return A CloseableIterator that will use the given converter
24      */
25     public <C> CloseableIterable<CacheEntry<K, C>> converter(Converter<? super K, ? super V, ? extends C> converter);
26  }

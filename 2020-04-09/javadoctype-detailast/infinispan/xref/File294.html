1   package org.infinispan.objectfilter.impl;
2   
3   import org.infinispan.objectfilter.FilterCallback;
4   import org.infinispan.objectfilter.FilterSubscription;
5   import org.infinispan.objectfilter.SortField;
6   import org.infinispan.objectfilter.impl.predicateindex.PredicateIndex;
7   import org.infinispan.objectfilter.impl.predicateindex.Predicates;
8   import org.infinispan.objectfilter.impl.predicateindex.be.BENode;
9   import org.infinispan.objectfilter.impl.predicateindex.be.BETree;
10  import org.infinispan.objectfilter.impl.predicateindex.be.PredicateNode;
11  import org.infinispan.objectfilter.impl.util.ComparableArrayComparator;
12  
13  import java.util.ArrayList;
14  import java.util.Comparator;
15  import java.util.List;
16  
17  /**
18   * @author anistor@redhat.com
19   * @since 7.0
20   */
21  public final class FilterSubscriptionImpl<TypeMetadata, AttributeMetadata, AttributeId extends Comparable<AttributeId>> implements FilterSubscription {
22  
23     private final String queryString;
24  
25     private final boolean useIntervals;
26  
27     public int index = -1; // todo [anistor] hide this
28  
29     private final MetadataAdapter<TypeMetadata, AttributeMetadata, AttributeId> metadataAdapter;
30  
31     private final BETree beTree;
32  
33     private final List<Predicates.Subscription<AttributeId>> predicateSubscriptions = new ArrayList<Predicates.Subscription<AttributeId>>();
34  
35     private final FilterCallback callback;
36  
37     private final String[] projection;
38  
39     private final List<List<AttributeId>> translatedProjection;
40  
41     private final SortField[] sortFields;
42  
43     private final List<List<AttributeId>> translatedSortProjection;
44  
45     private final Object[] eventTypes;
46  
47     private Comparator<Comparable[]> comparator;
48  
49     protected FilterSubscriptionImpl(String queryString, boolean useIntervals, MetadataAdapter<TypeMetadata, AttributeMetadata, AttributeId> metadataAdapter, BETree beTree, FilterCallback callback,
50                                      List<String> projection, List<List<AttributeId>> translatedProjection,
51                                      List<SortField> sortFields, List<List<AttributeId>> translatedSortProjection,
52                                      Object[] eventTypes) {
53        this.queryString = queryString;
54        this.useIntervals = useIntervals;
55        this.metadataAdapter = metadataAdapter;
56        this.beTree = beTree;
57        this.callback = callback;
58  
59        if (projection != null && !projection.isEmpty()) {
60           this.projection = projection.toArray(new String[projection.size()]);
61        } else {
62           this.projection = null;
63        }
64  
65        if (sortFields != null && !sortFields.isEmpty()) {
66           this.sortFields = sortFields.toArray(new SortField[sortFields.size()]);
67        } else {
68           this.sortFields = null;
69        }
70  
71        this.translatedProjection = translatedProjection;
72        this.translatedSortProjection = translatedSortProjection;
73        this.eventTypes = eventTypes;
74     }
75  
76     public String getQueryString() {
77        return queryString;
78     }
79  
80     public boolean isUseIntervals() {
81        return useIntervals;
82     }
83  
84     public BETree getBETree() {
85        return beTree;
86     }
87  
88     @Override
89     public String getEntityTypeName() {
90        return metadataAdapter.getTypeName();
91     }
92  
93     public MetadataAdapter<TypeMetadata, AttributeMetadata, AttributeId> getMetadataAdapter() {
94        return metadataAdapter;
95     }
96  
97     @Override
98     public FilterCallback getCallback() {
99        return callback;
100    }
101 
102    @Override
103    public String[] getProjection() {
104       return projection;
105    }
106 
107    @Override
108    public SortField[] getSortFields() {
109       return sortFields;
110    }
111 
112    @Override
113    public Comparator<Comparable[]> getComparator() {
114       if (sortFields != null && comparator == null) {
115          boolean[] direction = new boolean[sortFields.length];
116          for (int i = 0; i < sortFields.length; i++) {
117             direction[i] = sortFields[i].isAscending();
118          }
119          comparator = new ComparableArrayComparator(direction);
120       }
121       return comparator;
122    }
123 
124    @Override
125    public Object[] getEventTypes() {
126       return eventTypes;
127    }
128 
129    public void registerProjection(PredicateIndex<AttributeMetadata, AttributeId> predicateIndex) {
130       int i = 0;
131       if (translatedProjection != null) {
132          i = predicateIndex.addProjections(this, translatedProjection, i);
133       }
134       if (translatedSortProjection != null) {
135          predicateIndex.addProjections(this, translatedSortProjection, i);
136       }
137    }
138 
139    public void unregisterProjection(PredicateIndex<AttributeMetadata, AttributeId> predicateIndex) {
140       if (translatedProjection != null) {
141          predicateIndex.removeProjections(this, translatedProjection);
142       }
143       if (translatedSortProjection != null) {
144          predicateIndex.removeProjections(this, translatedSortProjection);
145       }
146    }
147 
148    public void subscribe(PredicateIndex<AttributeMetadata, AttributeId> predicateIndex) {
149       for (BENode node : beTree.getNodes()) {
150          if (node instanceof PredicateNode) {
151             PredicateNode<AttributeId> predicateNode = (PredicateNode<AttributeId>) node;
152             predicateSubscriptions.add(predicateIndex.addSubscriptionForPredicate(predicateNode, this));
153          }
154       }
155    }
156 
157    public void unsubscribe(PredicateIndex<AttributeMetadata, AttributeId> predicateIndex) {
158       for (Predicates.Subscription<AttributeId> subscription : predicateSubscriptions) {
159          predicateIndex.removeSubscriptionForPredicate(subscription);
160       }
161    }
162 }

23  /**
24   * <!-- START SNIPPET: validatorFlavours -->
25   * <p>The validators supplied by the XWork distribution (and any validators you
26   * might write yourself) come in two different flavors:</p>
27   *
28   * <ol>
29   * <li> Plain Validators / Non-Field validators </li>
30   * <li> FieldValidators </li>
31   * </ol>
32   *
33   * <p>Plain Validators (such as the ExpressionValidator) perform validation checks
34   * that are not inherently tied to a single specified field. When you declare a
35   * plain Validator in your -validation.xml file you do not associate a fieldname
36   * attribute with it. (You should avoid using plain Validators within the
37   * &lt;field-validator&gt; syntax described below.)</p>
38   *
39   * <p>FieldValidators (such as the EmailValidator) are designed to perform
40   * validation checks on a single field. They require that you specify a fieldname
41   * attribute in your -validation.xml file. There are two different (but equivalent)
42   * XML syntaxes you can use to declare FieldValidators (see "&lt;validator&gt; vs.
43   * &lt;field-Validator&gt; syntax&quot; below).</p>
44   *
45   * <p>There are two places where the differences between the two validator flavors
46   * are important to keep in mind:</p>
47   *
48   * <ol>
49   * <li> when choosing the xml syntax used for declaring a validator
50   * (either &lt;validator&gt; or &lt;field-validator&gt;)</li>
51   * <li> when using the short-circuit capability</li>
52   * </ol>
53   *
54   * <p><b>NOTE:</b>Note that you do not declare what &quot;flavor&quot; of validator you are
55   * using in your -validation.xml file, you just declare the name of the validator
56   * to use and Struts will know whether it's a &quot;plain Validator&quot; or a &quot;FieldValidator&quot;
57   * by looking at the validation class that the validator's programmer chose
58   * to implement.</p>
59   * <!-- END SNIPPET: validatorFlavours -->
60   *
61   *
62   * <!-- START SNIPPET: validationRules -->
63   * <p>To define validation rules for an Action, create a file named ActionName-validation.xml
64   * in the same package as the Action. You may also create alias-specific validation rules which
65   * add to the default validation rules defined in ActionName-validation.xml by creating
66   * another file in the same directory named ActionName-aliasName-validation.xml. In both
67   * cases, ActionName is the name of the Action class, and aliasName is the name of the
68   * Action alias defined in the xwork.xml configuration for the Action.</p>
69   *
70   * <p>The framework will also search up the inheritance tree of the Action to
71   * find validation rules for directly implemented interfaces and parent classes of the Action.
72   * This is particularly powerful when combined with ModelDriven Actions and the VisitorFieldValidator.
73   * Here's an example of how validation rules are discovered. Given the following class structure:</p>
74   *
75   * <ul>
76   * <li>interface Animal;</li>
77   * <li>interface Quadraped extends Animal;</li>
78   * <li>class AnimalImpl implements Animal;</li>
79   * <li>class QuadrapedImpl extends AnimalImpl implements Quadraped;</li>
80   * <li>class Dog extends QuadrapedImpl;</li>
81   * </ul>
82   *
83   * <p>The framework method will look for the following config files if Dog is to be validated:</p>
84   *
85   * <ul>
86   * <li>Animal</li>
87   * <li>Animal-aliasname</li>
88   * <li>AnimalImpl</li>
89   * <li>AnimalImpl-aliasname</li>
90   * <li>Quadraped</li>
91   * <li>Quadraped-aliasname</li>
92   * <li>QuadrapedImpl</li>
93   * <li>QuadrapedImpl-aliasname</li>
94   * <li>Dog</li>
95   * <li>Dog-aliasname</li>
96   * </ul>
97   *
98   * <p>While this process is similar to what the XW:Localization framework does
99   * when finding messages, there are some subtle differences. The most important
100  * difference is that validation rules are discovered from the parent downwards.
101  * </p>
102  *
103  * <p><b>NOTE:</b>Child's *-validation.xml will add on to parent's *-validation.xml
104  * according to the class hierarchy defined above. With this feature, one could have
105  * more generic validation rule at the parent and more specific validation rule at
106  * the child.</p>
107  *
108  * <!-- END SNIPPET: validationRules -->
109  *
110  *
111  * <!-- START SNIPPET: validatorVsFieldValidators1 -->
112  * <p>There are two ways you can define validators in your -validation.xml file:</p>
113  * <ol>
114  * <li> &lt;validator&gt; </li>
115  * <li> &lt;field-validator&gt; </li>
116  * </ol>
117  * <p>Keep the following in mind when using either syntax:</p>
118  *
119  * <p><b>Non-Field-Validator</b>
120  * The &lt;validator&gt; element allows you to declare both types of validators
121  * (either a plain Validator a field-specific FieldValidator).</p>
122  * <!-- END SNIPPET: validatorVsFieldValidators1 -->
123  *
124  * <pre>
125  * <!-- START SNIPPET: nonFieldValidatorUsingValidatorSyntax -->
126  *    &lt;!-- Declaring a plain Validator using the &lt;validator&gt; syntax: --&gt;
127  *
128  *    &lt;validator type=&quot;expression&gt;
129  *          &lt;param name=&quot;expression&quot;&gt;foo gt bar&lt;/param&gt;
130  *          &lt;message&gt;foo must be great than bar.&lt;/message&gt;
131  *    &lt;/validator&gt;
132  * <!-- END SNIPPET: nonFieldValidatorUsingValidatorSyntax -->
133  * </pre>
134  *
135  * <pre>
136  * <!-- START SNIPPET: fieldValidatorUsingValidatorSyntax -->
137  *    &lt;!-- Declaring a field validator using the &lt;validator&gt; syntax; --&gt;
138  *
139  *    &lt;validator type=&quot;required&quot;&gt;
140  *         &lt;param name=&quot;fieldName&quot;&gt;bar&lt;/param&gt;
141  *         &lt;message&gt;You must enter a value for bar.&lt;/message&gt;
142  *    &lt;/validator&gt;
143  * <!-- END SNIPPET: fieldValidatorUsingValidatorSyntax -->
144  * </pre>
145  *
146  * <!-- START SNIPPET: validatorVsFieldValidators2 -->
147  * <p><b>field-validator</b>
148  * The &lt;field-validator&gt; elements are basically the same as the &lt;validator&gt; elements
149  * except that they inherit the fieldName attribute from the enclosing &lt;field&gt; element.
150  * FieldValidators defined within a &lt;field-validator&gt; element will have their fieldName
151  * automatically filled with the value of the parent &lt;field&gt; element's fieldName
152  * attribute. The reason for this structure is to conveniently group the validators
153  * for a particular field under one element, otherwise the fieldName attribute
154  * would have to be repeated, over and over, for each individual &lt;validator&gt;.</p>
155  *
156  * <p><b>HINT:</b>
157  * It is always better to defined field-validator inside a &lt;field&gt; tag instead of
158  * using a &lt;validator&gt; tag and supplying fieldName as its param as the xml code itself
159  * is clearer (grouping of field is clearer)</p>
160  *
161  * <p><b>NOTE:</b>
162  * Note that you should only use FieldValidators (not plain Validators) within a
163  * &lt;field-validator&gt; block. A plain Validator inside a &lt;field&gt; will not be
164  * allowed and would generate error when parsing the xml, as it is not allowed in
165  * the defined dtd (xwork-validator-1.0.2.dtd)</p>
166  * <!-- END SNIPPET: validatorVsFieldValidators2 -->
167  *
168  * <pre>
169  * <!-- START SNIPPET: fieldValidatorUsingFieldValidatorSyntax -->
170  * Declaring a FieldValidator using the &lt;field-validator&gt; syntax:
171  *
172  * &lt;field name=&quot;email_address&quot;&gt;
173  *   &lt;field-validator type=&quot;required&quot;&gt;
174  *       &lt;message&gt;You cannot leave the email address field empty.&lt;/message&gt;
175  *   &lt;/field-validator&gt;
176  *   &lt;field-validator type=&quot;email&quot;&gt;
177  *       &lt;message&gt;The email address you entered is not valid.&lt;/message&gt;
178  *   &lt;/field-validator&gt;
179  * &lt;/field&gt;
180  * <!-- END SNIPPET: fieldValidatorUsingFieldValidatorSyntax -->
181  * </pre>
182  *
183  *
184  * <!-- START SNIPPET: validatorVsFieldValidators3 -->
185  * <p>The choice is yours. It's perfectly legal to only use &lt;validator&gt; elements
186  * without the &lt;field&gt; elements and set the fieldName attribute for each of them.
187  * The following are effectively equal:</p>
188  * <!-- END SNIPPET: validatorVsFieldValidators3 -->
189  *
190  * <pre>
191  * <!-- START SNIPPET: similarValidatorDeclaredInDiffSyntax -->
192  * &lt;field name=&quot;email_address&quot;&gt;
193  *   &lt;field-validator type=&quot;required&quot;&gt;
194  *       &lt;message&gt;You cannot leave the email address field empty.&lt;/message&gt;
195  *   &lt;/field-validator&gt;
196  *   &lt;field-validator type=&quot;email&quot;&gt;
197  *       &lt;message&gt;The email address you entered is not valid.&lt;/message&gt;
198  *   &lt;/field-validator&gt;
199  * &lt;/field&gt;
200  *
201  *
202  * &lt;validator type=&quot;required&quot;&gt;
203  *   &lt;param name=&quot;fieldName&quot;&gt;email_address&lt;/param&gt;
204  *   &lt;message&gt;You cannot leave the email address field empty.&lt;/message&gt;
205  * &lt;/validator&gt;
206  * &lt;validator type=&quot;email&quot;&gt;
207  *   &lt;param name=&quot;fieldName&quot;&gt;email_address&lt;/param&gt;
208  *   &lt;message&gt;The email address you entered is not valid.&lt;/message&gt;
209  * &lt;/validator&gt;
210  * <!-- END SNIPPET: similarValidatorDeclaredInDiffSyntax -->
211  * </pre>

214  * <!-- START SNIPPET: shortCircuitingValidators1 -->
215  * <p>It is possible to short-circuit a stack of validators.
216  * Here is another sample config file containing validation rules from the
217  * Xwork test cases: Notice that some of the &lt;field-validator&gt; and
218  * &lt;validator&gt; elements have the short-circuit attribute set to true.</p>
219  * <!-- END SNIPPET : shortCircuitingValidators1 -->
220  *
221  * <pre>
222  * &lt;!-- START SNIPPET: exShortCircuitingValidators --&gt;
223  * &lt;!DOCTYPE validators PUBLIC
224  *         &quot;-//Apache Struts//XWork Validator 1.0.3//EN&quot;
225  *  	   &quot;http://struts.apache.org/dtds/xwork-validator-1.0.3.dtd&quot;&gt;
226  * &lt;validators&gt;
227  *   &lt;!-- Field Validators for email field --&gt;
228  *   &lt;field name=&quot;email&quot;&gt;
229  *       &lt;field-validator type=&quot;required&quot; short-circuit=&quot;true&quot;&gt;
230  *           &lt;message&gt;You must enter a value for email.&lt;/message&gt;
231  *       &lt;/field-validator&gt;
232  *       &lt;field-validator type=&quot;email&quot; short-circuit=&quot;true&quot;&gt;
233  *           &lt;message&gt;Not a valid e-mail.&lt;/message&gt;
234  *       &lt;/field-validator&gt;
235  *   &lt;/field&gt;
236  *   &lt;!-- Field Validators for email2 field --&gt;
237  *   &lt;field name=&quot;email2&quot;&gt;
238  *      &lt;field-validator type=&quot;required&quot;&gt;
239  *           &lt;message&gt;You must enter a value for email2.&lt;/message&gt;
240  *       &lt;/field-validator&gt;
241  *      &lt;field-validator type=&quot;email&quot;&gt;
242  *           &lt;message&gt;Not a valid e-mail2.&lt;/message&gt;
243  *       &lt;/field-validator&gt;
244  *   &lt;/field&gt;
245  *   &lt;!-- Plain Validator 1 --&gt;
246  *   &lt;validator type=&quot;expression&quot;&gt;
247  *       &lt;param name=&quot;expression&quot;&gt;email.equals(email2)&lt;/param&gt;
248  *       &lt;message&gt;Email not the same as email2&lt;/message&gt;
249  *   &lt;/validator&gt;
250  *   &lt;!-- Plain Validator 2 --&gt;
251  *   &lt;validator type=&quot;expression&quot; short-circuit=&quot;true&quot;&gt;
252  *       &lt;param name=&quot;expression&quot;&gt;email.startsWith('mark')&lt;/param&gt;
253  *       &lt;message&gt;Email does not start with mark&lt;/message&gt;
254  *   &lt;/validator&gt;
255  * &lt;/validators&gt;
256  * &lt;!-- END SNIPPET: exShortCircuitingValidators --&gt;
257  * </pre>
258  *
259  * <!-- START SNIPPET:shortCircuitingValidators2  -->
260  * <p><b>short-circuiting and Validator flavors</b></p>
261  * <p>Plain validator takes precedence over field-validator. They get validated
262  * first in the order they are defined and then the field-validator in the order
263  * they are defined. Failure of a particular validator marked as short-circuit
264  * will prevent the evaluation of subsequent validators and an error (action
265  * error or field error depending on the type of validator) will be added to
266  * the ValidationContext of the object being validated.</p>
267  *
268  * <p>In the example above, the actual execution of validator would be as follows:</p>
269  *
270  * <ol>
271  * <li> Plain Validator 1</li>
272  * <li> Plain Validator 2</li>
273  * <li> Field Validators for email field</li>
274  * <li> Field Validators for email2 field</li>
275  * </ol>
276  *
277  * <p>Since Plain Validator 2 is short-circuited, if its validation failed,
278  * it will causes Field validators for email field and Field validators for email2
279  * field to not be validated as well.</p>
280  *
281  * <p><b>Usefull Information:</b>
282  * More complicated validation should probably be done in the validate()
283  * method on the action itself (assuming the action implements Validatable
284  * interface which ActionSupport already does).</p>
285  *
286  * <p>
287  * A plain Validator (non FieldValidator) that gets short-circuited will
288  * completely break out of the validation stack. No other validators will be
289  * evaluated and plain validators takes precedence over field validators meaning
290  * that they get evaluated in the order they are defined before field validators
291  * get a chance to be evaluated.
292  * </p>
293  * <!-- END SNIPPET: shortCircuitingValidators2 -->
294  *
295  * <!-- START SNIPPET: scAndValidatorFlavours1 -->
296  * <p><b>Short cuircuiting and validator flavours</b></p>
297  * <p>A FieldValidator that gets short-circuited will only prevent other
298  * FieldValidators for the same field from being evaluated. Note that this
299  * &quot;same field&quot; behavior applies regardless of whether the &lt;validator&gt; or
300  * &lt;field-validator&gt; syntax was used to declare the validation rule.
301  * By way of example, given this -validation.xml file:</p>
302  * <!-- END SNIPPET: scAndValidatorFlavours1 -->
303  *
304  * <pre>
305  * <!-- START SNIPPET: exScAndValidatorFlavours -->
306  * &lt;validator type=&quot;required&quot; short-circuit=&quot;true&quot;&gt;
307  *   &lt;param name=&quot;fieldName&quot;&gt;bar&lt;/param&gt;
308  *   &lt;message&gt;You must enter a value for bar.&lt;/message&gt;
309  * &lt;/validator&gt;
310  *
311  * &lt;validator type=&quot;expression&quot;&gt;
312  *   &lt;param name=&quot;expression&quot;&gt;foo gt bar&lt;/param&gt;
313  *   &lt;message&gt;foo must be great than bar.&lt;/message&gt;
314  * &lt;/validator&gt;
315  * <!-- END SNIPPET: exScAndValidatorFlavours -->
316  * </pre>
317  *
318  * <!-- START SNIPPET: scAndValidatorFlavours2 -->
319  * <p>both validators will be run, even if the &quot;required&quot; validator short-circuits.
320  * &quot;required&quot; validators are FieldValidator's and will not short-circuit the plain
321  * ExpressionValidator because FieldValidators only short-circuit other checks on
322  * that same field. Since the plain Validator is not field specific, it is
323  * not short-circuited.</p>
324  * <!-- END SNIPPET: scAndValidatorFlavours2 -->
325  *
326  *
327  * <!-- START SNIPPET: howXworkFindsValidatorForAction -->
328  * <p>As mentioned above, the framework will also search up the inheritance tree
329  * of the action to find default validations for interfaces and parent classes of
330  * the Action. If you are using the short-circuit attribute and relying on
331  * default validators higher up in the inheritance tree, make sure you don't
332  * accidentally short-circuit things higher in the tree that you really want!</p>
333  * <p>
334  * The effect of having common validators on both
335  * </p>
336  * <ul>
337  * 	<li>&lt;actionClass&gt;-validation.xml</li>
338  *     <li>&lt;actionClass&gt;-&lt;actionAlias&gt;-validation.xml</li>
339  * </ul>
340  * <p>
341  * It should be noted that the nett effect will be validation on both the validators available
342  * in both validation configuration file. For example if we have 'requiredstring' validators defined
343  * in both validation xml file for field named 'address', we will see 2 validation error indicating that
344  * the the address cannot be empty (assuming validation failed). This is due to WebWork
345  * will merge validators found in both validation configuration files.
346  * </p>
347  * <p>
348  * The logic behind this design decision is such that we could have common validators in
349  * &lt;actionClass&gt;-validation.xml and more context specific validators to be located
350  * in &lt;actionClass&gt;-&lt;actionAlias&gt;-validation.xml
351  * </p>
352  * <!-- END SNIPPET: howXworkFindsValidatorForAction -->
353  *
354  *
355  * <!-- START SNIPPET: i18n -->
356  * Validator's validation messages could be internatinalized. For example,
357  * <pre>
358  *   &lt;field-validator type=&quot;required&quot;&gt;
359  *      &lt;message key=&quot;required.field&quot; /&gt;
360  *   &lt;/field-validator&gt;
361  * </pre>
362  * or
363  * <pre>
364  *   &lt;validator type=&quot;expression&quot;&gt;
365  *      &lt;param name=&quot;expression&quot;&gt;email.startsWith('Mark')&lt;/param&gt;
366  *      &lt;message key=&quot;email.invalid&quot; /&gt;
367  *   &lt;/validator&gt;
368  * </pre>
369  * In the first case, WebWork would look for i18n with key 'required.field' as the validation error message if
370  * validation fails, and 'email.invalid' in the second case.
371  *
372  * We could also provide a default message such that if validation failed and the i18n key for the message
373  * cannot be found, WebWork would fall back and use the default message. An example would be as follows :-
374  * <pre>
375  *   &lt;field-validator type=&quot;required&quot;&gt;
376  *      &lt;message key=&quot;required.field&quot;&gt;This field is required.&lt;/message&gt;
377  *   &lt;/field-validator&gt;
378  * </pre>
379  * or
380  * <pre>
381  *   &lt;validator type=&quot;expression&quot;&gt;
382  *      &lt;param name=&quot;expression&quot;&gt;email.startsWith('Mark')&lt;/param&gt;
383  *      &lt;message key=&quot;email.invalid&quot;&gt;Email needs with starts with Mark&lt;/message&gt;
384  *   &lt;/validator&gt;
385  * </pre>
386  *
387  *
388  * <!-- END SNIPPET: i18n -->
389  * @author Jason Carreira
390  */
391 public interface Validator<T> {
392 
393     /**
394      * Sets the default message to use for validation failure
395      *
396      * @param message the default message
397      */
398     void setDefaultMessage(String message);
399 
400     /**
401      * Gets the default message used for validation failures
402      *
403      * @return the default message
404      */
405     String getDefaultMessage();
406 
407     /**
408      * Gets the validation failure message for the given object
409      *
410      * @param object object being validated (eg. a domain model object)
411      * @return the validation failure message
412      */
413     String getMessage(Object object);
414 
415     /**
416      * Sets a resource bundle key to be used for lookup of validation failure message
417      *
418      * @param key the resource bundle key
419      */
420     void setMessageKey(String key);
421 
422     /**
423      * Gets the resource bundle key used for lookup of validation failure message
424      *
425      * @return the resource bundle key
426      */
427     String getMessageKey();
428 
429     /**
430      * Sets the message parameters to be used when parsing i18n messages
431      *
432      * @param messageParameters  the message parameters
433      */
434     void setMessageParameters(String[] messageParameters);
435 
436     /**
437      * Gets the message parameters to be used when parsing i18n messages
438      *
439      * @return the message parameters
440      */
441     String[] getMessageParameters();
442 
443     /**
444      * This method will be called before validate with a non-null ValidatorContext.
445      *
446      * @param validatorContext the validation context to use.
447      */
448     void setValidatorContext(ValidatorContext validatorContext);
449 
450     /**
451      * Gets the validation context used
452      *
453      * @return the validation context
454      */
455     ValidatorContext getValidatorContext();
456 
457     /**
458      * The validation implementation must guarantee that setValidatorContext will
459      * be called with a non-null ValidatorContext before validate is called.
460      *
461      * @param object the object to be validated.
462      * @throws ValidationException is thrown if there is validation error(s).
463      */
464     void validate(Object object) throws ValidationException;
465 
466     /**
467      * Sets the validator type to use (see class javadoc).
468      *
469      * @param type the type to use.
470      */
471     void setValidatorType(String type);
472 
473     /**
474      * Gets the validator type used (see class javadoc).
475      *
476      * @return the type used
477      */
478     String getValidatorType();
479 
480     /**
481      * Sets the value stack to use to resolve values and parameters
482      *
483      * @param stack The value stack for the request
484      * @since 2.1.1
485      */
486     void setValueStack(ValueStack stack);
487 
488 }

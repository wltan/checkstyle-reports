19  package org.jsecurity.web.attr;
20  
21  import javax.servlet.ServletRequest;
22  import javax.servlet.ServletResponse;
23  
24  /**
25   * A <tt>WebAttribute</tt> is a storage mechanism for a single object accessible during a web request.
26   *
27   * <p>It is used to make objects associated with the transient request persistent beyond the request so that they can
28   * be retrieved upon a later request.
29   *
30   * @author Les Hazlewood
31   * @since 0.2
32   */
33  public interface WebAttribute<T> {
34  
35      T retrieveValue(ServletRequest request, ServletResponse response);
36  
37      void storeValue(T value, ServletRequest request, ServletResponse response);
38  
39      void removeValue(ServletRequest request, ServletResponse response);
40  }

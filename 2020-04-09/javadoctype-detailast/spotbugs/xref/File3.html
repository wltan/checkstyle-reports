20  package edu.umd.cs.findbugs.gui2;
21  
22  import java.awt.Component;
23  import java.awt.event.MouseAdapter;
24  import java.awt.event.MouseEvent;
25  
26  import javax.swing.JCheckBox;
27  import javax.swing.JList;
28  import javax.swing.ListCellRenderer;
29  import javax.swing.ListSelectionModel;
30  import javax.swing.UIManager;
31  import javax.swing.border.Border;
32  import javax.swing.border.EmptyBorder;
33  
34  /**
35   * A list of JCheckBoxes! How convenient!
36   *
37   * Adapted from: http://www.devx.com/tips/Tip/5342
38   *
39   * @author Trevor Harmon
40   */
41  @SuppressWarnings("serial")
42  public class CheckBoxList<E> extends JList<E> {
43      private static Border noFocusBorder = new EmptyBorder(1, 1, 1, 1);
44  
45      public CheckBoxList() {
46          setCellRenderer(new CellRenderer());
47  
48          addMouseListener(new MouseAdapter() {
49              @Override
50              public void mousePressed(MouseEvent e) {
51                  int index = locationToIndex(e.getPoint());
52  
53                  if (index != -1) {
54                      JCheckBox checkbox = (JCheckBox) getModel().getElementAt(index);
55                      checkbox.setSelected(!checkbox.isSelected());
56                      repaint();
57                  }
58              }
59          });
60  
61          setSelectionMode(ListSelectionModel.SINGLE_SELECTION);
62      }
63  
64      public CheckBoxList(E[] list) {
65          this();
66          setListData(list);
67      }
68  
69      @Override
70      public void setEnabled(boolean enabled) {
71          super.setEnabled(enabled);
72  
73          for (int i = 0; i < getModel().getSize(); i++) {
74              ((JCheckBox) getModel().getElementAt(i)).setEnabled(enabled);
75          }
76      }
77  
78      protected class CellRenderer implements ListCellRenderer<E> {
79          @Override
80          public Component getListCellRendererComponent(JList<? extends E> list, E value, int index, boolean isSelected,
81                  boolean cellHasFocus) {
82              JCheckBox checkbox = (JCheckBox) value;
83              checkbox.setBackground(isSelected ? getSelectionBackground() : getBackground());
84              checkbox.setForeground(isSelected ? getSelectionForeground() : getForeground());
85              checkbox.setEnabled(isEnabled());
86              checkbox.setFont(getFont());
87              checkbox.setFocusPainted(false);
88              checkbox.setBorderPainted(true);
89              checkbox.setBorder(isSelected ? UIManager.getBorder("List.focusCellHighlightBorder") : noFocusBorder);
90              return checkbox;
91          }
92  
93  
94  
95      }
96  }

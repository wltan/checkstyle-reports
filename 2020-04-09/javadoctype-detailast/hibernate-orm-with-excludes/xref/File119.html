24  package org.hibernate.ejb.metamodel;
25  import java.io.Serializable;
26  import javax.persistence.metamodel.Type;
27  
28  /**
29   * Defines commonality for the JPA {@link Type} hierarchy of interfaces.
30   *
31   * This adds a type name so we don't rely on the class name. That allows
32   * for non-class-based (metadata-driven) models. --koehn
33   *
34   * @author Steve Ebersole
35   */
36  public abstract class AbstractType<X> implements Type<X>, Serializable {
37      private final Class<X> javaType;
38      private String typeName;
39  
40      public AbstractType(Class<X> javaType) {
41          this.javaType = javaType;
42  
43          this.typeName = javaType != null ? javaType.getName() : "unknown";
44      }
45  
46      public Class<X> getJavaType() {
47          return javaType;
48      }
49  
50      public String getTypeName() {
51          return typeName;
52      }
53  
54      public void setTypeName(String name) {
55          this.typeName = name;
56      }
57  }

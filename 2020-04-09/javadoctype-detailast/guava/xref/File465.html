36  /**
37   * GWT emulated version of {@link com.google.common.collect.ImmutableSortedMap}. It's a thin wrapper
38   * around a {@link java.util.TreeMap}.
39   *
40   * @author Hayward Chan
41   */
42  public final class ImmutableSortedMap<K, V> extends ForwardingImmutableMap<K, V>
43      implements SortedMap<K, V> {
44  
45    @SuppressWarnings("unchecked")
46    static final Comparator NATURAL_ORDER = Ordering.natural();
47  
48    // This reference is only used by GWT compiler to infer the keys and values
49    // of the map that needs to be serialized.
50    private Comparator<? super K> unusedComparatorForSerialization;
51    private K unusedKeyForSerialization;
52    private V unusedValueForSerialization;

54    private final transient SortedMap<K, V> sortedDelegate;
55  
56    // The comparator used by this map.  It's the same as that of sortedDelegate,
57    // except that when sortedDelegate's comparator is null, it points to a
58    // non-null instance of Ordering.natural().
59    // (cpovirk: Is sortedDelegate's comparator really ever null?)
60    // The comparator will likely also differ because of our nullAccepting hack.
61    // See the bottom of the file for more information about it.
62    private final transient Comparator<? super K> comparator;
63  
64    ImmutableSortedMap(SortedMap<K, V> delegate, Comparator<? super K> comparator) {
65      super(delegate);
66      this.comparator = comparator;
67      this.sortedDelegate = delegate;
68    }

70    @Beta
71    public static <T, K, V> Collector<T, ?, ImmutableSortedMap<K, V>> toImmutableSortedMap(
72        Comparator<? super K> comparator,
73        Function<? super T, ? extends K> keyFunction,
74        Function<? super T, ? extends V> valueFunction) {
75      return CollectCollectors.toImmutableSortedMap(comparator, keyFunction, valueFunction);
76    }

78    @Beta
79    public static <T, K, V> Collector<T, ?, ImmutableSortedMap<K, V>> toImmutableSortedMap(
80        Comparator<? super K> comparator,
81        Function<? super T, ? extends K> keyFunction,
82        Function<? super T, ? extends V> valueFunction,
83        BinaryOperator<V> mergeFunction) {
84      checkNotNull(comparator);
85      checkNotNull(keyFunction);
86      checkNotNull(valueFunction);
87      checkNotNull(mergeFunction);
88      return Collectors.collectingAndThen(
89          Collectors.toMap(
90              keyFunction, valueFunction, mergeFunction, () -> new TreeMap<K, V>(comparator)),
91          ImmutableSortedMap::copyOfSorted);
92    }

155   public static <K, V> ImmutableSortedMap<K, V> copyOfSorted(SortedMap<K, ? extends V> map) {
156     // If map has a null comparator, the keys should have a natural ordering,
157     // even though K doesn't explicitly implement Comparable.
158     @SuppressWarnings("unchecked")
159     Comparator<? super K> comparator =
160         (map.comparator() == null) ? NATURAL_ORDER : map.comparator();
161     return copyOfInternal(map, comparator);
162   }
163 
164   private static <K, V> ImmutableSortedMap<K, V> copyOfInternal(
165       Map<? extends K, ? extends V> map, Comparator<? super K> comparator) {
166 
167     if (map instanceof ImmutableSortedMap) {
168       // TODO: Prove that this cast is safe, even though
169       // Collections.unmodifiableSortedMap requires the same key type.
170       @SuppressWarnings("unchecked")
171       ImmutableSortedMap<K, V> kvMap = (ImmutableSortedMap<K, V>) map;
172       Comparator<?> comparator2 = kvMap.comparator();
173       boolean sameComparator =
174           (comparator2 == null) ? comparator == NATURAL_ORDER : comparator.equals(comparator2);
175       if (sameComparator) {
176         return kvMap;
177       }
178     }
179 
180     SortedMap<K, V> delegate = newModifiableDelegate(comparator);
181     for (Entry<? extends K, ? extends V> entry : map.entrySet()) {
182       putEntryWithChecks(delegate, entry);
183     }
184     return newView(unmodifiableSortedMap(delegate), comparator);
185   }
186 
187   private static <K, V> void putEntryWithChecks(
188       SortedMap<K, V> map, Entry<? extends K, ? extends V> entry) {
189     K key = checkNotNull(entry.getKey());
190     V value = checkNotNull(entry.getValue());
191     if (map.containsKey(key)) {
192       // When a collision happens, the colliding entry is the first entry
193       // of the tail map.
194       Entry<K, V> previousEntry = map.tailMap(key).entrySet().iterator().next();
195       throw new IllegalArgumentException(
196           "Duplicate keys in mappings "
197               + previousEntry.getKey()
198               + "="
199               + previousEntry.getValue()
200               + " and "
201               + key
202               + "="
203               + value);
204     }
205     map.put(key, value);
206   }

220   public static final class Builder<K, V> extends ImmutableMap.Builder<K, V> {
221     private final Comparator<? super K> comparator;
222 
223     public Builder(Comparator<? super K> comparator) {
224       this.comparator = checkNotNull(comparator);
225     }
226 
227     @Override
228     public Builder<K, V> put(K key, V value) {
229       entries.add(entryOf(key, value));
230       return this;
231     }
232 
233     @Override
234     public Builder<K, V> put(Entry<? extends K, ? extends V> entry) {
235       super.put(entry);
236       return this;
237     }
238 
239     @Override
240     public Builder<K, V> putAll(Map<? extends K, ? extends V> map) {
241       return putAll(map.entrySet());
242     }
243 
244     @Override
245     public Builder<K, V> putAll(Iterable<? extends Entry<? extends K, ? extends V>> entries) {
246       for (Entry<? extends K, ? extends V> entry : entries) {
247         put(entry);
248       }
249       return this;
250     }
251 
252     Builder<K, V> combine(Builder<K, V> other) {
253       super.combine(other);
254       return this;
255     }
256 
257     @Override
258     public Builder<K, V> orderEntriesByValue(Comparator<? super V> valueComparator) {
259       throw new UnsupportedOperationException("Not available on ImmutableSortedMap.Builder");
260     }
261 
262     @Override
263     public ImmutableSortedMap<K, V> build() {
264       SortedMap<K, V> delegate = newModifiableDelegate(comparator);
265       for (Entry<? extends K, ? extends V> entry : entries) {
266         putEntryWithChecks(delegate, entry);
267       }
268       return newView(unmodifiableSortedMap(delegate), comparator);
269     }
270   }

272   private transient ImmutableSortedSet<K> keySet;
273 
274   @Override
275   public ImmutableSortedSet<K> keySet() {
276     ImmutableSortedSet<K> ks = keySet;
277     return (ks == null) ? (keySet = createKeySet()) : ks;
278   }
279 
280   @Override
281   ImmutableSortedSet<K> createKeySet() {
282     // the keySet() of the delegate is only a Set and TreeMap.navigatableKeySet
283     // is not available in GWT yet.  To keep the code simple and code size more,
284     // we make a copy here, instead of creating a view of it.
285     //
286     // TODO: revisit if it's unbearably slow or when GWT supports
287     // TreeMap.navigatbleKeySet().
288     return ImmutableSortedSet.copyOf(comparator, sortedDelegate.keySet());
289   }

303   K higher(K k) {
304     Iterator<K> iterator = keySet().tailSet(k).iterator();
305     while (iterator.hasNext()) {
306       K tmp = iterator.next();
307       if (comparator().compare(k, tmp) < 0) {
308         return tmp;
309       }
310     }
311     return null;
312   }

314   public ImmutableSortedMap<K, V> headMap(K toKey) {
315     checkNotNull(toKey);
316     return newView(sortedDelegate.headMap(toKey));
317   }
318 
319   ImmutableSortedMap<K, V> headMap(K toKey, boolean inclusive) {
320     checkNotNull(toKey);
321     if (inclusive) {
322       K tmp = higher(toKey);
323       if (tmp == null) {
324         return this;
325       }
326       toKey = tmp;
327     }
328     return headMap(toKey);
329   }

331   public ImmutableSortedMap<K, V> subMap(K fromKey, K toKey) {
332     checkNotNull(fromKey);
333     checkNotNull(toKey);
334     checkArgument(comparator.compare(fromKey, toKey) <= 0);
335     return newView(sortedDelegate.subMap(fromKey, toKey));
336   }
337 
338   ImmutableSortedMap<K, V> subMap(K fromKey, boolean fromInclusive, K toKey, boolean toInclusive) {
339     checkNotNull(fromKey);
340     checkNotNull(toKey);
341     checkArgument(comparator.compare(fromKey, toKey) <= 0);
342     return tailMap(fromKey, fromInclusive).headMap(toKey, toInclusive);
343   }
344 
345   public ImmutableSortedMap<K, V> tailMap(K fromKey) {
346     checkNotNull(fromKey);
347     return newView(sortedDelegate.tailMap(fromKey));
348   }
349 
350   public ImmutableSortedMap<K, V> tailMap(K fromKey, boolean inclusive) {
351     checkNotNull(fromKey);
352     if (!inclusive) {
353       fromKey = higher(fromKey);
354       if (fromKey == null) {
355         return new Builder<K, V>(this.comparator).build();
356       }
357     }
358     return tailMap(fromKey);
359   }

361   private ImmutableSortedMap<K, V> newView(SortedMap<K, V> delegate) {
362     return newView(delegate, comparator);

365   private static <K, V> ImmutableSortedMap<K, V> newView(
366       SortedMap<K, V> delegate, Comparator<? super K> comparator) {
367     return new ImmutableSortedMap<K, V>(delegate, comparator);
368   }

370   /*
371    * We don't permit nulls, but we wrap every comparator with nullsFirst().
372    * Why? We want for queries like containsKey(null) to return false, but the
373    * GWT SortedMap implementation that we delegate to throws
374    * NullPointerException if the comparator does. Since our construction
375    * methods ensure that null is never present in the map, it's OK for the
376    * comparator to look for it wherever it wants.
377    *
378    * Note that we do NOT touch the comparator returned by comparator(), which
379    * should be identical to the one the user passed in. We touch only the
380    * "secret" comparator used by the delegate implementation.
381    */
382 
383   private static <K, V> SortedMap<K, V> newModifiableDelegate(Comparator<? super K> comparator) {
384     return newTreeMap(nullAccepting(comparator));
385   }
386 
387   private static <E> Comparator<E> nullAccepting(Comparator<E> comparator) {
388     return Ordering.from(comparator).nullsFirst();
389   }

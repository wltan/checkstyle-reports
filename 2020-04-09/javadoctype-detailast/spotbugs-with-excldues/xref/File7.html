20  package edu.umd.cs.findbugs.ba;
21  
22  /**
23   * Convenience class for defining Dataflow classes which use a
24   * BasicAbstractDataflowAnalysis subtype. The main functionality is offering
25   * getFact{At,After}Location() methods which forward to the actual analysis
26   * object.
27   *
28   * @see edu.umd.cs.findbugs.ba.Dataflow
29   * @see edu.umd.cs.findbugs.ba.BasicAbstractDataflowAnalysis
30   * @author David Hovemeyer
31   */
32  public class AbstractDataflow<Fact, AnalysisType extends BasicAbstractDataflowAnalysis<Fact>> extends
33  Dataflow<Fact, AnalysisType> {
34  
35      /**
36       * Constructor.
37       *
38       * @param cfg
39       *            CFG of the method on which dfa is performed
40       * @param analysis
41       *            the dataflow analysis
42       */
43      public AbstractDataflow(CFG cfg, AnalysisType analysis) {
44          super(cfg, analysis);
45      }
46  
47      /**
48       * Get the fact that is true on the given control edge.
49       *
50       * @param edge
51       *            the edge
52       * @return the fact that is true on the edge
53       * @throws DataflowAnalysisException
54       */
55      @Override
56      public Fact getFactOnEdge(Edge edge) throws DataflowAnalysisException {
57          return getAnalysis().getFactOnEdge(edge);
58      }
59  
60  }

29  /**
30   * Perform a depth first search on a graph. Algorithm based on Cormen, et. al,
31   * <cite>Introduction to Algorithms</cite>, p. 478. Currently, this class
32   * <ul>
33   * <li>assigns DFS edge types (see {@link DFSEdgeTypes})
34   * <li>assigns discovery and finish times for each vertex
35   * <li>produces a topological sort of the vertices,
36   * <em>if and only if the graph is acyclic</em>
37   * </ul>
38   *
39   * <p>
40   * Concrete subclasses implement forward and reverse versions of depth first
41   * search.
42   *
43   * @author David Hovemeyer
44   * @see Graph
45   * @see DepthFirstSearch
46   * @see ReverseDepthFirstSearch
47   */
48  public abstract class AbstractDepthFirstSearch<GraphType extends Graph<EdgeType, VertexType>, EdgeType extends GraphEdge<EdgeType, VertexType>, VertexType extends GraphVertex<VertexType>>
49  implements DFSEdgeTypes {
50  
51      public final static boolean DEBUG = false;
52  
53      private final GraphType graph;
54  
55      private final int[] colorList;
56  
57      private final int[] discoveryTimeList;
58  
59      private final int[] finishTimeList;
60  
61      private final int[] dfsEdgeTypeList;
62  
63      private int timestamp;
64  
65      private final LinkedList<VertexType> topologicalSortList;
66  
67      private VertexChooser<VertexType> vertexChooser;
68  
69      private SearchTreeCallback<VertexType> searchTreeCallback;
70  
71      /**
72       * Color of a vertex which hasn't been visited yet.
73       */
74      protected static final int WHITE = 0;
75  
76      /**
77       * Color of a vertex which has been visited, but not all of whose
78       * descendents have been visited.
79       */
80      protected static final int GRAY = 1;
81  
82      /**
83       * Color of a vertex whose entire search tree has been visited.
84       */
85      protected static final int BLACK = 2;
86  
87      /**
88       * Constructor.
89       *
90       * @param graph
91       *            the graph to be searched
92       * @throws IllegalArgumentException
93       *             if the graph has not had edge ids assigned yet
94       */
95      public AbstractDepthFirstSearch(GraphType graph) {
96          this.graph = graph;
97  
98          int numBlocks = graph.getNumVertexLabels();
99          colorList = new int[numBlocks]; // initially all elements are WHITE
100         discoveryTimeList = new int[numBlocks];
101         finishTimeList = new int[numBlocks];
102 
103         int maxEdgeId = graph.getNumEdgeLabels();
104         dfsEdgeTypeList = new int[maxEdgeId];
105 
106         timestamp = 0;
107 
108         topologicalSortList = new LinkedList<>();
109     }
110 
111     // Abstract methods allow the concrete subclass to define
112     // the "polarity" of the depth first search. That way,
113     // this code can do normal DFS, or DFS of reversed GraphType.
114 
115     /**
116      * Get Iterator over "logical" outgoing edges.
117      */
118     protected abstract Iterator<EdgeType> outgoingEdgeIterator(GraphType graph, VertexType vertex);
119 
120     /**
121      * Get "logical" target of edge.
122      */
123     protected abstract VertexType getTarget(EdgeType edge);
124 
125     /**
126      * Get "logical" source of edge.
127      */
128     protected abstract VertexType getSource(EdgeType edge);
129 
130     /**
131      * Choose the next search tree root. By default, this method just scans for
132      * a WHITE vertex. Subclasses may override this method in order to choose
133      * which vertices are used as search tree roots.
134      *
135      * @return the next search tree root
136      */
137     protected VertexType getNextSearchTreeRoot() {
138         // FIXME: slow linear search, should improve
139         for (Iterator<VertexType> i = graph.vertexIterator(); i.hasNext();) {
140             VertexType vertex = i.next();
141             if (visitMe(vertex)) {
142                 return vertex;
143             }
144         }
145         return null;
146     }
147 
148     public Collection<VertexType> unvisitedVertices() {
149         LinkedList<VertexType> result = new LinkedList<>();
150 
151         for (Iterator<VertexType> i = graph.vertexIterator(); i.hasNext();) {
152             VertexType v = i.next();
153             if (getColor(v) == WHITE) {
154                 result.add(v);
155             }
156         }
157         return result;
158     }

160     /**
161      * Specify a VertexChooser object to be used to selected which vertices are
162      * visited by the search. This is useful if you only want to search a subset
163      * of the vertices in the graph.
164      *
165      * @param vertexChooser
166      *            the VertexChooser to use
167      */
168     public void setVertexChooser(VertexChooser<VertexType> vertexChooser) {
169         this.vertexChooser = vertexChooser;
170     }
171 
172     /**
173      * Set a search tree callback.
174      *
175      * @param searchTreeCallback
176      *            the search tree callback
177      */
178     public void setSearchTreeCallback(SearchTreeCallback<VertexType> searchTreeCallback) {
179         this.searchTreeCallback = searchTreeCallback;
180     }

182     /**
183      * Perform the depth first search.
184      *
185      * @return this object
186      */
187     public AbstractDepthFirstSearch<GraphType, EdgeType, VertexType> search() {
188         visitAll();
189         classifyUnknownEdges();
190 
191         return this;
192     }
193 
194     /**
195      * Return whether or not the graph contains a cycle: i.e., whether it
196      * contains any back edges. This should only be called after search() has
197      * been called (since that method actually executes the search).
198      *
199      * @return true if the graph contains a cycle, false otherwise
200      */
201     public boolean containsCycle() {
202         for (Iterator<EdgeType> i = graph.edgeIterator(); i.hasNext();) {
203             EdgeType edge = i.next();
204             if (getDFSEdgeType(edge) == BACK_EDGE) {
205                 return true;
206             }
207         }
208         return false;
209     }
210 
211     /**
212      * Get the type of edge in the depth first search.
213      *
214      * @param edge
215      *            the edge
216      * @return the DFS type of edge: TREE_EDGE, FORWARD_EDGE, CROSS_EDGE, or
217      *         BACK_EDGE
218      * @see DFSEdgeTypes
219      */
220     public int getDFSEdgeType(EdgeType edge) {
221         return dfsEdgeTypeList[edge.getLabel()];
222     }
223 
224     /**
225      * Return the timestamp indicating when the given vertex was discovered.
226      *
227      * @param vertex
228      *            the vertex
229      */
230     public int getDiscoveryTime(VertexType vertex) {
231         return discoveryTimeList[vertex.getLabel()];
232     }
233 
234     /**
235      * Return the timestamp indicating when the given vertex was finished
236      * (meaning that all of its descendents were visited recursively).
237      *
238      * @param vertex
239      *            the vertex
240      */
241     public int getFinishTime(VertexType vertex) {
242         return finishTimeList[vertex.getLabel()];
243     }
244 
245     /**
246      * Get array of finish times, indexed by vertex label.
247      *
248      * @return the array of finish times
249      */
250     @SuppressFBWarnings("EI")
251     public int[] getFinishTimeList() {
252         return finishTimeList;
253     }
254 
255     /**
256      * Get an iterator over the vertexs in topological sort order.
257      * <em>This works if and only if the graph is acyclic.</em>
258      */
259     public Iterator<VertexType> topologicalSortIterator() {
260         return topologicalSortList.iterator();
261     }

263     private class Visit {
264         private final VertexType vertex;
265 
266         private final Iterator<EdgeType> outgoingEdgeIterator;
267 
268         public Visit(VertexType vertex) {
269             if (vertex == null) {
270                 throw new IllegalStateException();
271             }
272             this.vertex = vertex;
273             this.outgoingEdgeIterator = outgoingEdgeIterator(graph, vertex);
274 
275             // Mark the vertex as visited, and set its timestamp
276             setColor(vertex, GRAY);
277             setDiscoveryTime(vertex, timestamp++);
278         }
279 
280         public VertexType getVertex() {
281             return vertex;
282         }
283 
284         public boolean hasNextEdge() {
285             return outgoingEdgeIterator.hasNext();
286         }
287 
288         public EdgeType getNextEdge() {
289             return outgoingEdgeIterator.next();
290         }
291     }
292 
293     private void visitAll() {
294         for (;;) {
295             VertexType searchTreeRoot = getNextSearchTreeRoot();
296             if (searchTreeRoot == null) {
297                 // No more work to do
298                 break;
299             }
300 
301             if (searchTreeCallback != null) {
302                 searchTreeCallback.startSearchTree(searchTreeRoot);
303             }
304 
305             ArrayList<Visit> stack = new ArrayList<>(graph.getNumVertexLabels());
306             stack.add(new Visit(searchTreeRoot));
307 
308             while (!stack.isEmpty()) {
309                 Visit visit = stack.get(stack.size() - 1);
310 
311                 if (visit.hasNextEdge()) {
312                     // Continue visiting successors
313                     EdgeType edge = visit.getNextEdge();
314                     visitSuccessor(stack, edge);
315                 } else {
316                     // Finish the vertex
317                     VertexType vertex = visit.getVertex();
318                     setColor(vertex, BLACK);
319                     topologicalSortList.addFirst(vertex);
320                     setFinishTime(vertex, timestamp++);
321 
322                     stack.remove(stack.size() - 1);
323                 }
324             }
325         }
326     }
327 
328     private void visitSuccessor(ArrayList<Visit> stack, EdgeType edge) {
329         // Get the successor
330         VertexType succ = getTarget(edge);
331         int succColor = getColor(succ);
332 
333         // Classify edge type (if possible)
334         int dfsEdgeType = 0;
335         switch (succColor) {
336         case WHITE:
337             dfsEdgeType = TREE_EDGE;
338             break;
339         case GRAY:
340             dfsEdgeType = BACK_EDGE;
341             break;
342         case BLACK:
343             dfsEdgeType = UNKNOWN_EDGE;
344             break;// We can't distinguish between CROSS and FORWARD edges at
345             // this point
346         default:
347             assert false;
348         }
349         setDFSEdgeType(edge, dfsEdgeType);
350 
351         // If successor hasn't been visited yet, visit it
352         if (visitMe(succ)) {
353             // Add to search tree (if a search tree callback exists)
354             if (searchTreeCallback != null) {
355                 searchTreeCallback.addToSearchTree(getSource(edge), getTarget(edge));
356             }
357 
358             // Add to visitation stack
359             stack.add(new Visit(succ));
360         }
361     }
362 
363     // Classify CROSS and FORWARD edges
364     private void classifyUnknownEdges() {
365         Iterator<EdgeType> edgeIter = graph.edgeIterator();
366         while (edgeIter.hasNext()) {
367             EdgeType edge = edgeIter.next();
368             int dfsEdgeType = getDFSEdgeType(edge);
369             if (dfsEdgeType == UNKNOWN_EDGE) {
370                 int srcDiscoveryTime = getDiscoveryTime(getSource(edge));
371                 int destDiscoveryTime = getDiscoveryTime(getTarget(edge));
372 
373                 if (srcDiscoveryTime < destDiscoveryTime) {
374                     // If the source was visited earlier than the
375                     // target, it's a forward edge.
376                     dfsEdgeType = FORWARD_EDGE;
377                 } else {
378                     // If the source was visited later than the
379                     // target, it's a cross edge.
380                     dfsEdgeType = CROSS_EDGE;
381                 }
382 
383                 setDFSEdgeType(edge, dfsEdgeType);
384             }
385         }
386     }
387 
388     private void setColor(VertexType vertex, int color) {
389         colorList[vertex.getLabel()] = color;
390     }
391 
392     /**
393      * Get the current color of given vertex.
394      *
395      * @param vertex
396      *            the vertex
397      * @return the color (WHITE, BLACK, or GRAY)
398      */
399     protected int getColor(VertexType vertex) {
400         return colorList[vertex.getLabel()];
401     }
402 
403     /**
404      * Predicate to determine which vertices should be visited as the search
405      * progresses. Takes both vertex color and the vertex chooser (if any) into
406      * account.
407      *
408      * @param vertex
409      *            the vertex to possibly be visited
410      * @return true if the vertex should be visited, false if not
411      */
412     protected boolean visitMe(VertexType vertex) {
413         return (getColor(vertex) == WHITE) && (vertexChooser == null || vertexChooser.isChosen(vertex));
414     }
415 
416     private void setDiscoveryTime(VertexType vertex, int ts) {
417         discoveryTimeList[vertex.getLabel()] = ts;
418     }
419 
420     private void setFinishTime(VertexType vertex, int ts) {
421         finishTimeList[vertex.getLabel()] = ts;
422     }
423 
424     private void setDFSEdgeType(EdgeType edge, int dfsEdgeType) {
425         dfsEdgeTypeList[edge.getLabel()] = dfsEdgeType;
426     }

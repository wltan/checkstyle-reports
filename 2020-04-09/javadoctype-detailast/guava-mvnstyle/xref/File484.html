15  package com.google.common.util.concurrent;
16  
17  import elemental2.promise.IThenable;
18  import elemental2.promise.Promise;
19  import java.util.concurrent.Executor;
20  import java.util.concurrent.Future;
21  import jsinterop.annotations.JsMethod;
22  import jsinterop.annotations.JsOptional;
23  
24  /**
25   * Java super source for ListenableFuture, implementing a structural thenable via a default method.
26   * For restrictions, please refer to the documentation of the then() method.
27   *
28   * <p>This class is not (explicitly) implementing IThenable<V> because "then" is overloaded there
29   * and the single parameter version would need to be marked native, which does not seem to be
30   * feasible in interfaces (see "subclassing a class with overloaded methods" in jsinterop
31   * documentation).
32   */
33  public interface ListenableFuture<V> extends Future<V>, Thenable<V> {
34    void addListener(Runnable listener, Executor executor);
35  
36    /** Note that this method is not expected to be overridden. */
37    @JsMethod
38    @Override
39    default <R> IThenable<R> then(
40        @JsOptional ThenOnFulfilledCallbackFn<? super V, ? extends R> onFulfilled,
41        @JsOptional ThenOnRejectedCallbackFn<? extends R> onRejected) {
42      return new Promise<V>(
43              (resolve, reject) -> {
44                Futures.addCallback(
45                    this,
46                    new FutureCallback<V>() {
47                      @Override
48                      public void onSuccess(V value) {
49                        resolve.onInvoke(value);
50                      }
51  
52                      @Override
53                      public void onFailure(Throwable throwable) {
54                        reject.onInvoke(throwable.getBackingJsObject());
55                      }
56                    },
57                    MoreExecutors.directExecutor());
58              })
59          .then(
60              (IThenable.ThenOnFulfilledCallbackFn) onFulfilled,
61              (IThenable.ThenOnRejectedCallbackFn) onRejected);
62    }

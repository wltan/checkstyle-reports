37  /**
38   * Corrrelates two sequences when they overlap and groups the results.
39   * 
40   * @see <a href="http://msdn.microsoft.com/en-us/library/hh244235.aspx">MSDN: Observable.GroupJoin</a>
41   * @param <T1> the left value type
42   * @param <T2> the right value type
43   * @param <D1> the value type of the left duration
44   * @param <D2> the value type of the right duration
45   * @param <R> the result value type
46   */
47  public final class OnSubscribeGroupJoin<T1, T2, D1, D2, R> implements OnSubscribe<R> {
48      protected final Observable<T1> left;
49      protected final Observable<T2> right;
50      protected final Func1<? super T1, ? extends Observable<D1>> leftDuration;
51      protected final Func1<? super T2, ? extends Observable<D2>> rightDuration;
52      protected final Func2<? super T1, ? super Observable<T2>, ? extends R> resultSelector;
53  
54      public OnSubscribeGroupJoin(
55              Observable<T1> left,
56              Observable<T2> right,
57              Func1<? super T1, ? extends Observable<D1>> leftDuration,
58              Func1<? super T2, ? extends Observable<D2>> rightDuration,
59              Func2<? super T1, ? super Observable<T2>, ? extends R> resultSelector) {
60          this.left = left;
61          this.right = right;
62          this.leftDuration = leftDuration;
63          this.rightDuration = rightDuration;
64          this.resultSelector = resultSelector;
65      }
66  
67      @Override
68      public void call(Subscriber<? super R> child) {
69          ResultManager ro = new ResultManager(new SerializedSubscriber<R>(child));
70          child.add(ro);
71          ro.init();
72      }
73  
74      /** Manages sub-observers and subscriptions. */
75      final class ResultManager implements Subscription {
76          final RefCountSubscription cancel;
77          final Subscriber<? super R> subscriber;
78          final CompositeSubscription group;
79          final Object guard = new Object();
80          /** Guarded by guard. */
81          int leftIds;
82          /** Guarded by guard. */
83          int rightIds;
84          /** Guarded by guard. */
85          final Map<Integer, Observer<T2>> leftMap = new HashMap<Integer, Observer<T2>>();
86          /** Guarded by guard. */
87          final Map<Integer, T2> rightMap = new HashMap<Integer, T2>();
88          /** Guarded by guard. */
89          boolean leftDone;
90          /** Guarded by guard. */
91          boolean rightDone;
92  
93          public ResultManager(Subscriber<? super R> subscriber) {
94              this.subscriber = subscriber;
95              this.group = new CompositeSubscription();
96              this.cancel = new RefCountSubscription(group);
97          }
98  
99          public void init() {
100 
101             Subscriber<T1> s1 = new LeftObserver();
102             Subscriber<T2> s2 = new RightObserver();
103             
104             group.add(s1);
105             group.add(s2);
106 
107             left.unsafeSubscribe(s1);
108             right.unsafeSubscribe(s2);
109         }
110 
111         @Override
112         public void unsubscribe() {
113             cancel.unsubscribe();
114         }
115         
116         @Override
117         public boolean isUnsubscribed() {
118             return cancel.isUnsubscribed();
119         }
120         /**
121          * Notify everyone and cleanup.
122          * @param e the exception
123          */
124         void errorAll(Throwable e) {
125             List<Observer<T2>> list;
126             synchronized (guard) {
127                 list = new ArrayList<Observer<T2>>(leftMap.values());
128                 leftMap.clear();
129                 rightMap.clear();
130             }
131             for (Observer<T2> o : list) {
132                 o.onError(e);
133             }
134             subscriber.onError(e);
135             cancel.unsubscribe();
136         }
137         /**
138          * Notify only the main subscriber and cleanup.
139          * @param e  the exception
140          */
141         void errorMain(Throwable e) {
142             synchronized (guard) {
143                 leftMap.clear();
144                 rightMap.clear();
145             }            
146             subscriber.onError(e);
147             cancel.unsubscribe();
148         }
149         void complete(List<Observer<T2>> list) {
150             if (list != null) {
151                 for (Observer<T2> o : list) {
152                     o.onCompleted();
153                 }
154                 subscriber.onCompleted();
155                 cancel.unsubscribe();
156             }
157         }
158         
159         /** Observe the left source. */
160         final class LeftObserver extends Subscriber<T1> {
161             @Override
162             public void onNext(T1 args) {
163                 try {
164                     int id;
165                     Subject<T2, T2> subj = PublishSubject.create();
166                     Observer<T2> subjSerial = new SerializedObserver<T2>(subj);
167                     
168                     synchronized (guard) {
169                         id = leftIds++;
170                         leftMap.put(id, subjSerial);
171                     }
172 
173                     Observable<T2> window = Observable.create(new WindowObservableFunc<T2>(subj, cancel));
174 
175                     Observable<D1> duration = leftDuration.call(args);
176 
177                     Subscriber<D1> d1 = new LeftDurationObserver(id);
178                     group.add(d1);
179                     duration.unsafeSubscribe(d1);
180 
181                     R result = resultSelector.call(args, window);
182 
183                     List<T2> rightMapValues;
184                     synchronized (guard) {
185                         rightMapValues = new ArrayList<T2>(rightMap.values());
186                     }
187                     
188                     subscriber.onNext(result);
189                     for (T2 t2 : rightMapValues) {
190                         subjSerial.onNext(t2);
191                     }
192                     
193                     
194                 } catch (Throwable t) {
195                     onError(t);
196                 }
197             }
198 
199             @Override
200             public void onCompleted() {
201                 List<Observer<T2>> list = null;
202                 synchronized (guard) {
203                     leftDone = true;
204                     if (rightDone) {
205                         list = new ArrayList<Observer<T2>>(leftMap.values());
206                         leftMap.clear();
207                         rightMap.clear();
208                     }
209                 }
210                 complete(list);
211             }
212 
213             @Override
214             public void onError(Throwable e) {
215                 errorAll(e);
216             }
217 
218         }
219 
220         /** Observe the right source. */
221         final class RightObserver extends Subscriber<T2> {
222             @Override
223             public void onNext(T2 args) {
224                 try {
225                     int id;
226                     synchronized (guard) {
227                         id = rightIds++;
228                         rightMap.put(id, args);
229                     }
230                     Observable<D2> duration = rightDuration.call(args);
231 
232                     Subscriber<D2> d2 = new RightDurationObserver(id);
233                     
234                     group.add(d2);
235                     duration.unsafeSubscribe(d2);
236 
237                     List<Observer<T2>> list;
238                     synchronized (guard) {
239                         list = new ArrayList<Observer<T2>>(leftMap.values());
240                     }
241                     for (Observer<T2> o : list) {
242                         o.onNext(args);
243                     }
244                 } catch (Throwable t) {
245                     onError(t);
246                 }
247             }
248 
249             @Override
250             public void onCompleted() {
251                 List<Observer<T2>> list = null;
252                 synchronized (guard) {
253                     rightDone = true;
254                     if (leftDone) {
255                         list = new ArrayList<Observer<T2>>(leftMap.values());
256                         leftMap.clear();
257                         rightMap.clear();
258                     }
259                 }
260                 complete(list);
261             }
262 
263             @Override
264             public void onError(Throwable e) {
265                 errorAll(e);
266             }
267         }
268 
269         /** Observe left duration and apply termination. */
270         final class LeftDurationObserver extends Subscriber<D1> {
271             final int id;
272             boolean once = true;
273 
274             public LeftDurationObserver(int id) {
275                 this.id = id;
276             }
277 
278             @Override
279             public void onCompleted() {
280                 if (once) {
281                     once = false;
282                     Observer<T2> gr;
283                     synchronized (guard) {
284                         gr = leftMap.remove(id);
285                     }
286                     if (gr != null) {
287                         gr.onCompleted();
288                     }
289                     group.remove(this);
290                 }
291             }
292 
293             @Override
294             public void onError(Throwable e) {
295                 errorMain(e);
296             }
297 
298             @Override
299             public void onNext(D1 args) {
300                 onCompleted();
301             }
302         }
303 
304         /** Observe right duration and apply termination. */
305         final class RightDurationObserver extends Subscriber<D2> {
306             final int id;
307             boolean once = true;
308             public RightDurationObserver(int id) {
309                 this.id = id;
310             }
311 
312             @Override
313             public void onCompleted() {
314                 if (once) {
315                     once = false;
316                     synchronized (guard) {
317                         rightMap.remove(id);
318                     }
319                     group.remove(this);
320                 }
321             }
322 
323             @Override
324             public void onError(Throwable e) {
325                 errorMain(e);
326             }
327 
328             @Override
329             public void onNext(D2 args) {
330                 onCompleted();
331             }
332         }

336     /**
337      * The reference-counted window observable.
338      * Subscribes to the underlying Observable by using a reference-counted
339      * subscription.
340      */
341     final static class WindowObservableFunc<T> implements OnSubscribe<T> {
342         final RefCountSubscription refCount;
343         final Observable<T> underlying;
344 
345         public WindowObservableFunc(Observable<T> underlying, RefCountSubscription refCount) {
346             this.refCount = refCount;
347             this.underlying = underlying;
348         }
349 
350         @Override
351         public void call(Subscriber<? super T> t1) {
352             Subscription ref = refCount.get();
353             WindowSubscriber wo = new WindowSubscriber(t1, ref);
354             wo.add(ref);
355             
356             underlying.unsafeSubscribe(wo);
357         }
358 
359         /** Observe activities on the window. */
360         final class WindowSubscriber extends Subscriber<T> {
361             final Subscriber<? super T> subscriber;
362             private final Subscription ref;
363 
364             public WindowSubscriber(Subscriber<? super T> subscriber, Subscription ref) {
365                 super(subscriber);
366                 this.subscriber = subscriber;
367                 this.ref = ref;
368             }
369 
370             @Override
371             public void onNext(T args) {
372                 subscriber.onNext(args);
373             }
374 
375             @Override
376             public void onError(Throwable e) {
377                 subscriber.onError(e);
378                 ref.unsubscribe();
379             }
380 
381             @Override
382             public void onCompleted() {
383                 subscriber.onCompleted();
384                 ref.unsubscribe();
385             }
386         }

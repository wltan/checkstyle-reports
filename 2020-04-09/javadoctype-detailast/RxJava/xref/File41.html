16  package rx.internal.operators;
17  
18  import rx.Notification;
19  import rx.Observable.Operator;
20  import rx.Subscriber;
21  import rx.plugins.RxJavaPlugins;
22  
23  /**
24   * Turns all of the notifications from an Observable into {@code onNext} emissions, and marks
25   * them with their original notification types within {@link Notification} objects.
26   * <p>
27   * <img width="640" src="https://github.com/ReactiveX/RxJava/wiki/images/rx-operators/materialize.png" alt="">
28   * <p>
29   * See <a href="http://msdn.microsoft.com/en-us/library/hh229453.aspx">here</a> for the Microsoft Rx equivalent.
30   */
31  public final class OperatorMaterialize<T> implements Operator<Notification<T>, T> {
32      /** Lazy initialization via inner-class holder. */
33      private static final class Holder {
34          /** A singleton instance. */
35          static final OperatorMaterialize<Object> INSTANCE = new OperatorMaterialize<Object>();
36      }
37      /**
38       * @return a singleton instance of this stateless operator.
39       */
40      @SuppressWarnings("unchecked")
41      public static <T> OperatorMaterialize<T> instance() {
42          return (OperatorMaterialize<T>)Holder.INSTANCE;
43      }
44      private OperatorMaterialize() { }
45      @Override
46      public Subscriber<? super T> call(final Subscriber<? super Notification<T>> child) {
47          return new Subscriber<T>(child) {
48  
49              @Override
50              public void onCompleted() {
51                  child.onNext(Notification.<T> createOnCompleted());
52                  child.onCompleted();
53              }
54  
55              @Override
56              public void onError(Throwable e) {
57                  RxJavaPlugins.getInstance().getErrorHandler().handleError(e);
58                  child.onNext(Notification.<T> createOnError(e));
59                  child.onCompleted();
60              }
61  
62              @Override
63              public void onNext(T t) {
64                  child.onNext(Notification.<T> createOnNext(t));
65              }
66  
67          };
68      }
69  }
